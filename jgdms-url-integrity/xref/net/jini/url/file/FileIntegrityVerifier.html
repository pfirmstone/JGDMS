19  package net.jini.url.file;
20  
21  import java.net.URL;
22  import net.jini.security.IntegrityVerifier;
23  import aQute.bnd.annotation.headers.RequireCapability;
24  import aQute.bnd.annotation.headers.ProvideCapability;
25  
26  /**
27   * Integrity verifier for FILE URLs. This class is intended to be specified
28   * in a resource to configure the operation of
29   * {@link net.jini.security.Security#verifyCodebaseIntegrity
30   * Security.verifyCodebaseIntegrity}.
31   * <p>
32   * Do not use this verifier if your operating system is configured to
33   * mount remote file systems and the underlying file system protocols do
34   * not provide adequate data integrity or authentication of the remote hosts.
35   *
36   * @author Sun Microsystems, Inc.
37   * @since 2.0
38   */
39  @RequireCapability(
40  	ns="osgi.extender",
41  	filter="(osgi.extender=osgi.serviceloader.registrar)")
42  @ProvideCapability(
43  	ns="osgi.serviceloader",
44  	name="net.jini.security.IntegrityVerifier")
45  public class FileIntegrityVerifier implements IntegrityVerifier {
46      /**
47       * Returns <code>true</code> if the specified URL uses the "file"
48       * protocol and the host is <code>null</code>, empty, "~", or "localhost";
49       * returns <code>false</code> otherwise.
50       *
51       * @throws NullPointerException {@inheritDoc}
52       */
53      public boolean providesIntegrity(URL url) {
54  	if (!"file".equals(url.getProtocol())) {
55  	    return false;
56  	}
57  	String host = url.getHost();
58  	return (host == null || host.equals("") ||
59  		host.equals("~") || host.equals("localhost"));
60      }
61  }
