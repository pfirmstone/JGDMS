54      /**
55       * Computes the message digest of data specified by a URL.
56       *
57       * @param url the URL of the data
58       * @param algorithm the message digest algorithm to use
59       * @return the message digest, as a <code>String</code> in hexadecimal
60       *	       format
61       * @throws IOException if an I/O exception occurs while reading data from
62       *	       the URL
63       * @throws NoSuchAlgorithmException if no provider is found for the message
64       *	       digest algorithm
65       * @throws NullPointerException if either argument is <code>null</code>
66       */
67      public static String computeDigest(URL url, String algorithm)
68  	throws IOException, NoSuchAlgorithmException
69      {
70  	return computeDigest(url.openStream(), algorithm);
71      }
72  
73      /** Computes the message digest for an input stream. */
74      private static String computeDigest(InputStream in, String algorithm)
75  	throws IOException, NoSuchAlgorithmException
76      {
77  	try {
78  	    if (!(in instanceof BufferedInputStream)) {
79  		in = new BufferedInputStream(in, 2048);
80  	    }
81  	    MessageDigest md = MessageDigest.getInstance(algorithm);
82  	    byte[] buf = new byte[2048];
83  	    while (true) {
84  		int n = in.read(buf);
85  		if (n < 0) {
86  		    break;
87  		}
88  		md.update(buf, 0, n);
89  	    }
90  	    return digestString(md.digest());
91  	} finally {
92  	    try {
93  		in.close();
94  	    } catch (IOException e) {
95  	    }
96  	}
97      }

99      /**
100      * Computes the message digests for a codebase with HTTPMD URLs.
101      * <p>
102      * Do not use a directory on a remote filesystem, or a directory URL, if
103      * the underlying network access protocol does not provide adequate data
104      * integrity or authentication of the remote host.
105      *
106      * @param sourceDirectory the filename or URL of the directory containing
107      *	      the source files corresponding to the URLs in
108      *	      <code>codebase</code>
109      * @param codebase a space-separated list of HTTPMD URLs. The digest
110      *	      values specified in the URLs will be ignored. The path portion of
111      *	      the URLs, without the message digest parameters, will be used to
112      *	      specify the source files, relative to
113      *	      <code>sourceDirectory</code>, to use for computing message
114      *	      digests.
115      * @return the codebase updated to include computed digests
116      * @throws IllegalArgumentException if any of the URLs in
117      *	       <code>codebase</code> fail to specify the HTTPMD protocol
118      * @throws IOException if an I/O exception occurs while reading data from
119      *	       the source files
120      * @throws MalformedURLException if any of the URLs in
121      *	       <code>codebase</code> have incorrect syntax
122      * @throws NullPointerException if either argument is <code>null</code>
123      */
124     public static String computeDigestCodebase(String sourceDirectory,
125 					       String codebase)
126 	throws IOException, MalformedURLException, NullPointerException
127     {
128 	boolean isURL;
129 	try {
130 	    new URL(sourceDirectory);
131 	    isURL = true;
132 	} catch (MalformedURLException e) {
133 	    isURL = false;
134 	}
135 	if (sourceDirectory.endsWith(isURL ? "/" : File.separator)) {
136 	    sourceDirectory =
137 		sourceDirectory.substring(0, sourceDirectory.length() - 1);
138 	}
139 	StringTokenizer specs = new StringTokenizer(codebase);
140 	StringBuffer sb = new StringBuffer();
141 	boolean first = true;
142 	while (specs.hasMoreTokens()) {
143 	    final String spec = specs.nextToken();
144 	    if (!"httpmd:".regionMatches(true, 0, spec, 0, 7)) {
145 		throw new IllegalArgumentException(
146 		    "Codebase URL does not specify HTTPMD protocol: " + spec);
147 	    }
148 	    /*
149 	     * Use doPrivileged so caller doesn't need
150 	     * java.net.NetPermission("specifyStreamHandler") to specify the
151 	     * URL stream handler.
152 	     */
153 	    URL url;
154 	    try {
155 		url = (URL) Security.doPrivileged(
156 		    new PrivilegedExceptionAction() {
157 			public Object run() throws MalformedURLException {
158 			    return new URL(null, spec, handler);
159 			}
160 		    });
161 	    } catch (PrivilegedActionException e) {
162 		throw (MalformedURLException) e.getCause();
163 	    }
164 	    String path = url.getPath();
165 	    int paramIndex = path.lastIndexOf(';');
166 	    int equalsIndex = path.indexOf('=', paramIndex);
167 	    int commentIndex = path.indexOf(',', equalsIndex);
168 	    String algorithm = path.substring(paramIndex + 1, equalsIndex);
169 	    URI relSourceURI;
170 	    try {
171 		relSourceURI = new URI(
172 		    "file:" + (path.startsWith("/") ? "" : "/") +
173 		    path.substring(0, path.indexOf(';')));
174 	    } catch (URISyntaxException e) {
175 		throw new MalformedURLException(
176 		    "Problem with codebase URL " + spec + ": " +
177 		    e.getMessage());
178 	    }
179 	    InputStream in;
180 	    if (isURL) {
181 		String relSource = relSourceURI.getRawPath();
182 		in = new URL(sourceDirectory + relSource).openStream();
183 	    } else {
184 		String relSource = relSourceURI.getPath();
185 		if ('/' != File.separatorChar) {
186 		    relSource = relSource.replace('/', File.separatorChar);
187 		}
188 		in = new FileInputStream(sourceDirectory + relSource);
189 	    }
190 	    String digest;
191 	    try {
192 		digest = computeDigest(in, algorithm);
193 	    } catch (NoSuchAlgorithmException e) {
194 		/* Shouldn't happen -- URL constructor should find this */
195 		throw new RuntimeException("Shouldn't happen: " + e);
196 	    }
197 	    URL result = new URL(
198 		url,
199 		path.substring(0, equalsIndex + 1) + digest +
200 		(commentIndex < 0 ? "" : path.substring(commentIndex)) +
201 		(url.getQuery() == null ? "" : '?' + url.getQuery()) +
202 		(url.getRef() == null ? "" : '#' + url.getRef()));
203 	    if (!first) {
204 		sb.append(' ');
205 	    } else {
206 		first = false;
207 	    }
208 	    sb.append(result);
209 	}
210 	return sb.toString();
211     }

213     /** Converts a message digest to a String in hexadecimal format. */
214     static String digestString(byte[] digest) {
215 	StringBuffer sb = new StringBuffer(digest.length * 2);
216 	for (int i = 0; i < digest.length; i++) {
217 	    byte b = digest[i];
218 	    sb.append(Character.forDigit((b >> 4) & 0xf, 16));
219 	    sb.append(Character.forDigit(b & 0xf, 16));
220 	}
221 	return sb.toString();
222     }

224     /** Converts a String in hexadecimal format to a message digest. */
225     static byte[] stringDigest(String s) throws NumberFormatException {
226 	byte[] result = new byte[(s.length() + 1) / 2];
227 	int rpos = result.length;
228 	int last = -1;
229 	for (int spos = s.length(); --spos >= 0; ) {
230 	    int digit = Character.digit(s.charAt(spos), 16);
231 	    if (digit < 0) {
232 		throw new NumberFormatException(
233 		    "Illegal hex digit: '" + s.charAt(spos) + "'");
234 	    }
235 	    if (last < 0) {
236 		last = digit;
237 	    } else {
238 		result[--rpos] = (byte) (last + (digit << 4));
239 		last = -1;
240 	    }
241 	}
242 	if (last >= 0) {
243 	    result[--rpos] = (byte) last;
244 	}
245 	return result;
246     }

248     /**
249      * Returns true if the character is permitted in an HTTPMD comment.  Legal
250      * comment characters are ASCII letters and numbers, plus '-', '_', '.',
251      * '~', '*', ''', '(', ')', ':', '@', '&amp;', '=', '+', '$', and ','.
252      */
253     static boolean commentChar(char c) {
254 	return (c >= 'a' && c <= 'z')
255 	    || (c >= 'A' && c <= 'Z')
256 	    || (c >= '0' && c <= '9')
257 	    || ("-_.~*'():@&=+$,".indexOf(c) >= 0);
258     }
