31  /**
32   * A stream handler for URLs with the HTTPMD protocol. HTTPMD URLs provide a
33   * way to insure the integrity of data retrieved from an HTTP URL. The HTTPMD
34   * URL includes a message digest for the data to be retrieved from the URL. The
35   * URL input stream insures that the data has the correct message digest when
36   * the end of file for the stream is reached. If the data has the wrong
37   * message digest, a {@link WrongMessageDigestException} is thrown. <p>
38   *
39   * HTTPMD URLs may be used to guarantee the integrity of a downloaded object's
40   * codebase if used when specifying the URLs for the JAR files containing the
41   * object's classes. Because HTTPMD URLs specify a message digest for a single
42   * item, they should not be used for directories of classes. <p>
43   *
44   * HTTPMD URLs have a syntax similar to that of HTTP URLs, but include a
45   * message digest as the last parameter stored in the last segment of the
46   * path. The parameter is introduced by the '<code>;</code>' character, and
47   * includes the name of the message digest algorithm, a '<code>=</code>', the
48   * message digest, and an optional comment introduced by the '<code>,</code>'
49   * character. In addition, a comment by itself may be specified in a relative
50   * HTTPMD URL. Comments are ignored when using <code>equals</code> to compare
51   * HTTPMD URLs. The comment specified in the context URL is ignored when
52   * parsing a relative HTTPMD URL. Adding a comment to an HTTPMD URL is useful
53   * in cases where the URL is required to have a particular suffix, for example
54   * the ".jar" file extension. A comment-only relative HTTPMD URL is useful when
55   * specifying the URL of the containing document from within the contents of
56   * the document, where the message digest cannot be specified because it is not
57   * yet known. <p>
58   *
59   * The message digest algorithm is case-insensitive, and may include ASCII
60   * letters and numbers, as well as the following characters:
61   *
62   * <pre>
63   * - _ . ~ * ' ( ) : @ &amp; + $ ,
64   * </pre> <p>
65   *
66   * The value specifies the name of the {@link MessageDigest} algorithm to
67   * use. For the URL syntax to be valid, the value must be the name of a
68   * <code>MessageDigest</code> algorithm as determined by calling
69   * {@link MessageDigest#getInstance(String)}. <p>
70   *
71   * The message digest is represented as a positive hexadecimal integer, using
72   * digits, and the letters '<code>a</code>' through '<code>f</code>', in either
73   * lowercase or uppercase. <p>
74   * 
75   * The characters following the '<code>,</code>' comment character may include
76   * ASCII letters and numbers, as well as the following characters:
77   *
78   * <pre>
79   * - _ . ~ * ' ( ) : @ &amp; = + $ ,
80   * </pre> <p>
81   *
82   * Here are some examples of HTTPMD URLs: </p> <ul>
83   *
84   * <li> An absolute URL: <br>
85   *      <code>
86   *	httpmd://www.sun.com/index.html;md5=7be207c7111e459eeea1c9b3d04f1667
87   *	</code>
88   *
89   * <li> A relative URL: <br>
90   *	<code>
91   * 	index.html;sha=99f6837808c0a79398bf69d83cfb1b82d20cf0cf,Comment
92   *	</code>
93   *
94   * <li> A comment-only relative URL: <br>
95   *	    <code>,.jar</code>
96   * </ul>
97   *
98   * @author Sun Microsystems, Inc.
99   * @see HttpmdUtil
100  * @since 2.0
101  *
102  *  <!-- Implementation Specifics -->
103  * <p>
104  * This implementation of HTTPMD URLs uses the {@link Logger} named
105  * <code>net.jini.url.httpmd</code> to log information at the following logging
106  * levels: </p>
107  *
108  * <table border="1" cellpadding="5" summary="Describes logging performed
109  *	  by the HTTPMD URL handler at different logging levels">
110  *
111  * <caption><b><code>net.jini.url.httpmd</code></b></caption>
112  *
113  * <tr> <th scope="col"> Level <th scope="col"> Description
114  *
115  * <tr> <td> {@link Levels#FAILED FAILED} <td> URL input stream detects an
116  *	incorrect message digest
117  *
118  * </table> <p>
119  *
120  * See the {@link LogManager} class for one way to use the <code>FAILED</code>
121  * logging level in standard logging configuration files.
122  */

143     /**
144      * Parses the string representation of an HTTPMD URL object.
145      *
146      * @throws IllegalArgumentException if the URL is malformed
147      */
148     protected void parseURL(URL url, String spec, int start, int limit) {
149 	/* Check for a relative URL that only specifies a comment */
150 	if (start < limit && spec.charAt(start) == ',') {
151 	    String query = url.getQuery();
152 	    int queryStart = spec.indexOf('?', start);
153 	    if (queryStart != -1) {
154 		query = spec.substring(queryStart + 1, limit);
155 		limit = queryStart;
156 	    }
157 	    String path = url.getPath() == null ? "" : url.getPath();
158 	    /* Remove the comment from the context path */
159 	    int param = path.lastIndexOf(';');
160 	    if (param != -1) {
161 		int equalsIndex = path.indexOf('=', param);
162 		if (equalsIndex < 0) {
163 		    throw new IllegalArgumentException(
164 		        "Message digest parameter is missing a '='");
165 		}
166 		int comment = path.indexOf(',', equalsIndex);
167 		if (comment != -1) {
168 		    path = path.substring(0, comment);
169 		}
170 	    }
171 	    /* Append the new comment */
172 	    path += spec.substring(start, limit);
173 	    setURL(url, url.getProtocol(), url.getHost(), url.getPort(),
174 		   url.getAuthority(), url.getUserInfo(), path, query,
175 		   url.getRef());
176 	} else {
177 	    /* Otherwise, combine spec with the context URL */
178 	    super.parseURL(url, spec, start, limit);
179 	}
180 
181 	/* Check syntax of message digest parameter */
182 	String path = url.getPath() == null ? "" : url.getPath();
183 	int semiIndex = path.lastIndexOf(';');
184 	if (semiIndex < 0) {
185 	    throw new IllegalArgumentException(
186 		"Message digest parameter is missing");
187 	}
188 	int equalsIndex = path.indexOf('=', semiIndex);
189 	if (equalsIndex < 0) {
190 	    throw new IllegalArgumentException(
191 		"Message digest parameter is missing a '='");
192 	}
193 	String algorithm = path.substring(semiIndex + 1, equalsIndex);
194 	try {
195 	    MessageDigest.getInstance(algorithm);
196 	} catch (NoSuchAlgorithmException e) {
197 	    throw new IllegalArgumentException(
198 		"Message digest parameter algorithm is not found: " +
199 		algorithm);
200 	}
201 	String digest = path.substring(equalsIndex + 1);
202 	int comment = digest.indexOf(',');
203 	if (comment != -1) {
204 	    for (int i = digest.length(); --i > comment; ) {
205 		char c = digest.charAt(i);
206 		if (!HttpmdUtil.commentChar(c)) {
207 		    throw new IllegalArgumentException(
208 			"Comment contains illegal character: '" + c + "'");
209 		}
210 	    }
211 	    digest = digest.substring(0, comment);
212 	}
213 	int length = digest.length();
214 	if (length == 0) {
215 	    throw new IllegalArgumentException(
216 		"Message digest parameter digest is empty");
217 	}
218 	for (int i = length; --i >= 0; ) {
219 	    char c = digest.charAt(i);
220 	    if (Character.digit(c, 16) < 0) {
221 		throw new IllegalArgumentException(
222 		    "Message digest parameter has invalid hex character: " +
223 		    c);
224 	    }
225 	}
226     }

228     /**
229      * The default superclass implementation performs dns lookup to determine
230      * if hosts are equal, this allows two URL's with different hashCode's
231      * to be equal, breaking the hashCode equals contract.
232      * 
233      * It also causes a test failure in the jtreg test suite.
234      * 
235      * 
236      * *** Start test: Mon Jan 23 08:11:26 EST 2012
237      * [jtreg] Test 9: TestEqual: httpmd://foo:88/bar/baz;p1=v1;md5=abcd?q#r, httpmd://alpha:88/bar/baz;p1=v1;md5=abcd?q#r
238      * [jtreg] FAIL: Should be: false
239      * [jtreg]       Result: true
240      * 
241      * URL.implies(URL url) is better suited to perform this function, why
242      * it was originally implemented in equals is unknown.
243      */
244     protected boolean hostsEqual(URL u1, URL u2) {
245 	if (u1.getHost() != null && u2.getHost() != null) 
246             return u1.getHost().equalsIgnoreCase(u2.getHost());
247 	 else
248             return u1.getHost() == null && u2.getHost() == null;
249     }

251     /**
252      * Compares two HTTPMD URLs to see if they refer to the same file. Performs
253      * case-insensitive comparison of the protocols and of the message digest
254      * parameters (ignoring the comment), calls <code>hostsEqual</code> to
255      * compare the hosts, compares the ports, with <code>-1</code> matching the
256      * default HTTP port (<code>80</code>), and performs case-sensitive
257      * comparison on the remaining portions.
258      */
259     protected boolean sameFile(URL u1, URL u2) {
260         /* Compare the protocols */
261         if (!( u1.getProtocol() != null &&
262                u1.getProtocol().equalsIgnoreCase(u2.getProtocol())))
263 	{
264             return false;
265 	}
266 	/* Compare the hosts */
267 	if (!hostsEqual(u1, u2)) {
268             return false;
269 	}
270 	/* Compare the paths */
271 	String path1 = u1.getPath();
272 	String path2 = u2.getPath();
273 	if (path1 == null || path2 == null) {
274 	    return false;
275         } else if ( path1.equals(path2)){
276             /* Paths are OK */
277 	} else {
278 	    /*
279 	     * Perform case insensitive matching on the message digest
280 	     * parameters, ignoring comments.
281 	     */
282 	    int param = path1.lastIndexOf(';');
283 	    if (param < 0 || param != path2.lastIndexOf(';')) {
284 		return false;
285 	    }
286 	    /* Case-sensitive match for non-parameter part */
287 	    if (!path1.regionMatches(0, path2, 0, param)) {
288 		return false;
289 	    }
290 	    /* Ignore comments */
291 	    int equalsIndex = path1.indexOf('=', param + 1);
292 	    if (equalsIndex < 0 || equalsIndex != path2.lastIndexOf('=')) {
293 		return false;
294 	    }
295 	    int comment1 = path1.indexOf(',', equalsIndex + 1);
296 	    int len = (comment1 != -1) ? comment1 : path1.length();
297 	    int comment2 = path2.indexOf(',', equalsIndex + 1);
298 	    int len2 = (comment2 != -1) ? comment2 : path2.length();
299 	    if (len != len2) {
300 		return false;
301 	    }
302 	    /* Case-insensitive match for algorithm and digest */
303 	    if (!path1.regionMatches(true, param + 1,
304 				     path2, param + 1, len - param - 1))
305 	    {
306 		return false;
307 	    }
308 	}
309 	/* Compare queries */
310 	if (u1.getQuery() == null
311 	    ? u2.getQuery() != null
312 	    : !u1.getQuery().equals(u2.getQuery()))
313 	{
314 	    return false;
315 	}
316 	/* Compare the ports */
317         int port1 = (u1.getPort() != -1) ? u1.getPort() : u1.getDefaultPort();
318         int port2 = (u2.getPort() != -1) ? u2.getPort() : u2.getDefaultPort();
319 	if (port1 != port2) {
320 	    return false;
321 	}
322         return true;
323     }

325     /**
326      * Computes the hash code for the specified URL. This method ignores the
327      * comment portion of the message digest parameter, and ignores the
328      * case of characters in the message digest and algorithm.
329      */
330     protected int hashCode(URL u) {
331         int h = 0;
332 
333         /* Generate the protocol part */
334         String protocol = u.getProtocol();
335         if (protocol != null) {
336 	    h += protocol.hashCode();
337 	}
338 
339         /* Generate the host part */
340 //	InetAddress addr = getHostAddress(u);
341 //	if (addr != null) {
342 //	    h += addr.hashCode();
343 //	} else {
344             String host = u.getHost();
345             if (host != null) {
346 	        h += host.toLowerCase().hashCode();
347 	    }
348 //        }
349 
350 	/*
351 	 * Generate the path part, ignoring case in the message digest and
352 	 * algorithm, and ignoring comment.
353 	 */
354 	String path = u.getPath();
355 	if (path != null) {
356 	    int param = path.lastIndexOf(';');
357 	    if (param == -1) {
358 		h += path.hashCode();
359 	    } else {
360 		h += path.substring(0, param).hashCode();
361 		int equalsIndex = path.indexOf('=', param + 1);
362 		int comment = path.indexOf(',', equalsIndex != -1 ? equalsIndex
363 								  : param + 1);
364 		if (comment != -1) {
365 		    path = path.substring(0, comment);
366 		}
367 		h += path.substring(param).toLowerCase().hashCode();
368 	    }
369 	}
370 
371         /* Generate the query part */
372         String query = u.getQuery();
373 	if (query != null) {
374 	    h += query.hashCode();
375 	}
376 
377         /* Generate the port part */
378 	if (u.getPort() == -1) {
379             h += getDefaultPort();
380 	} else {
381             h += u.getPort();
382 	}
383 
384         /* Generate the ref part */
385         String ref = u.getRef();
386 	if (ref != null)
387             h += ref.hashCode();
388 
389 	return h;
390     }
