34  import net.jini.jeri.BasicJeriExporter;

36  import net.jini.jeri.ServerEndpoint;
37  import org.apache.river.api.common.Beta;

39  /**
40   *
41   */
42  @Beta
43  public class ExportHelper
44      extends ConfigHelper
45  {
46      private static final Logger logger = Logger.getLogger(ExportHelper.class.getName());
47  
48      public ExportHelper(URL resource) throws ConfigurationException
49      {
50          super(resource);
51      }
52  
53      public ExportHelper(Configuration configuration)
54      {
55          super(configuration);
56      }
57  
58      public <T extends Remote> T export(T obj, Class<T> cls)
59          throws ExportException, ConfigurationException
60      {
61          return export(obj,cls,null);
62      }
63  
64      public <T extends Remote> T export( T obj, Class<T> cls, String component )
65          throws ExportException, ConfigurationException
66      {
67          if( component == null ) {
68              component = getComponent(cls);
69          }
70  
71          Exporter exp = getExporter(component);
72  
73          final Remote exportedObj = exp.export(obj);
74  
75          if (logger.isLoggable(Level.FINE)) {
76              logger.fine( "proxy=" + exportedObj );
77          }
78  
79          return (T) exportedObj;
80      }
81  
82      public Exporter getExporter( String component )
83          throws ConfigurationException
84      {
85          Exporter exp = (Exporter) configuration.getEntry(component, EXPORTER, Exporter.class, null);
86  
87          if( exp == null ) {
88              ServerEndpoint se = getServerEndpoint(component);
89              InvocationLayerFactory ilf = getILFactory(component);
90              boolean keepAlive = getKeepAlive(component);
91              boolean dgc = getDgc(component);
92              Uuid uuid = getUuid(component);
93  
94              exp = new BasicJeriExporter(se, ilf, dgc, keepAlive, uuid);
95          }
96  
97          return exp;
98      }
99  
100     private InvocationLayerFactory getILFactory( String component )
101         throws ConfigurationException
102     {
103         InvocationLayerFactory ilf = (InvocationLayerFactory) configuration.getEntry(component, ILF, InvocationLayerFactory.class, null);
104 
105         if( ilf == null ) {
106             //boolean verify = getVerify(component);
107 
108             InvocationConstraints ic = getInvocationConstraints(component);
109             if( ic == null ) {
110                 ic = InvocationConstraints.EMPTY ;
111             }
112             MethodConstraints serverConstraints = new BasicMethodConstraints(ic);
113             ilf = new BasicILFactory(serverConstraints,null);
114             //ilf = new ProxyTrustILFactory(serverConstraints,null);
115         }
116         
117         return ilf ;
118     }
119 
120     private ServerEndpoint getServerEndpoint( String component )
121         throws ConfigurationException
122     {
123         // entry must exist.
124         ServerEndpoint se = (ServerEndpoint) configuration.getEntry(component, SERVERENDPOINT, ServerEndpoint.class, Configuration.NO_DEFAULT);
125 
126         // check if entry is not null.
127         if( se == null ) {
128             throw new NullPointerException("serverEndpoint is null in " + component );
129         }
130 
131         return se ;
132     }
133 
134     private boolean getKeepAlive(String component) throws ConfigurationException
135     {
136         Boolean b = (Boolean) configuration.getEntry(component, KEEPALIVE, boolean.class, false);
137         return b.booleanValue();
138     }
