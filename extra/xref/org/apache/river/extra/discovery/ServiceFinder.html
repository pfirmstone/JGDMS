18  package org.apache.river.extra.discovery;
19  
20  import java.rmi.RemoteException;
21  
22  import net.jini.core.lookup.ServiceTemplate;
23  
24  /**
25   * This interface provides a convenience wrapper around service discovery simply
26   * for the purposes of demonstrating the self-healing proxy.
27   * 
28   * This implementation has at least one obvious potential failing.  The
29   * <code>ServiceFinder</code> implementation does not check that the service
30   * it replaces the proxy with is different to the service which has previous
31   * failed.  For example, <code>invoke</code> might fail against service with
32   * ID 0000-0000-0000-AAAA, if (for whatever reason) the ServiceFinder again
33   * finds the service with ID 0000-0000-0000-AAAA, then we would expect this call
34   * to fail also.
35   * 
36   * It would be wise to offer alternative implementations of <code>ServiceFinder</code>
37   * which guard against these kinds of behaviours.
38   * 
39   */
40  public interface ServiceFinder {
41  
42      Object findNewService(ServiceTemplate template) throws RemoteException;
43  
44      void terminate();
45  
46  }
