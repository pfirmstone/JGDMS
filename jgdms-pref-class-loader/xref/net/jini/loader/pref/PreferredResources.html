34  /**
35   * Internal data structure which holds preference information for a
36   * preferred class loader.  This utility is used only by the preferred
37   * class loader provider and is not intended to be a public API.
38   *
39   * A preferred resources object is created from an input stream which
40   * is formatted according to the Preferred List Syntax which is
41   * defined in the specification for
42   * <code>net.jini.loader.pref.PreferredClassProvider</code>
43   *
44   * Preferred resources instances hold preferred list expression data
45   * and the preferred state for the resources contained in a given
46   * preferred class loader.
47   * 
48   * PreferredResouces is thread safe.  
49   * Originally from package org.apache.river.loader.pref.internal.
50   *
51   * @author Sun Microsystems, Inc.
52   * @since 2.0
53   */
54  final class PreferredResources {
55      /**
56       * Constant value that indicates that there is no preference value
57       * for a given name.
58       */
59      public static final int NAME_NO_PREFERENCE = 0;
60  
61      /**
62       * Constant value that indicates that the resource for a given
63       * name is known to be not preferred.  This state is applicable
64       * when a preference expression declared that the name is not
65       * preferred or when the resource for the name (preferred or not)
66       * is known to not exist.
67       */
68      public static final int NAME_NOT_PREFERRED = 1;
69  
70      /**
71       * Constant value that indicates that a given name is preferred.
72       * This state is applicable when a preference expression declares
73       * the name to be preferred but it is uncertain whether the
74       * resource for the name the exists.
75       */
76      public static final int NAME_PREFERRED = 2;
77  
78      /**
79       * Constant value that indicates that a given resource is
80       * preferred.  This state is applicable when a preference
81       * expression declares the name to be preferred and the resource
82       * for the name is known to exist.
83       */
84      public static final int NAME_PREFERRED_RESOURCE_EXISTS = 3;
85      
86      /** string that starts preference specification headers */
87      private static final String HEADER_TITLE = "PreferredResources-Version: ";
88      private static final String HEADER_MAJOR_VERSION = "1";
89      private static final String HEADER_MINOR_VERSION = "0";
90      
91      /** preference syntax keywords */
92      private static final String PREF_PREFIX = "Preferred:";
93      private static final String NAME_PREFIX = "Name:";
94  
95      /* tables which contain relevant components of preferred names */
96      private final Map<String,Object> namespacePrefs;
97      private final Map<String,Object> packagePrefs;
98      private final Map<String,Object> completeNamePrefs;
99  
100     /** flag to signal if this preference object is empty */
101     private volatile boolean isEmpty = true;
102 
103     /** default preference setting */
104     private final Boolean defaultPreference;
105 
106     /**
107      * Create a preference object from a stream of formatted
108      * preference syntax.
109      *
110      * @see PreferredResources
111      */
112     public PreferredResources(InputStream in) throws IOException {
113         /*
114          * Read and parse preference information from the parameter input
115          * stream <code>in</code>.  When complete, the
116          * preference expression maps contain preference settings for
117          * preference names contained in the input stream.
118          */
119         BufferedReader br = new BufferedReader
120 	    (new InputStreamReader(in, "UTF8"));	
121         String line;
122 	String name = null;
123 	Boolean preference = null;
124         Boolean defaultPreference = Boolean.FALSE;
125         
126 	completeNamePrefs = new HashMap<String,Object>(53);
127 	packagePrefs = new HashMap<String,Object>(23);
128 	namespacePrefs = new HashMap<String,Object>(11);
129 	
130 	if ((line = readLineTrimComments(br)) != null) {
131 	    if (!line.startsWith(HEADER_TITLE)) {
132 		throw new IOException("unsupported preferred list header: " +
133 				      line);
134 	    } else {
135 		String version =
136 		    line.substring(HEADER_TITLE.length(),
137 				   line.length()).trim();
138 		if (!version.startsWith("1.")) {
139 		    throw new IOException("preferred list major version " +
140 					  "not supported");
141 		}
142 	    }
143 	    for (line = readLineTrimComments(br); line != null;
144 		 line = readLineTrimComments(br))
145 	    {
146 		if (line.length() == 0) {
147 		    continue;
148 		} else if (line.startsWith(NAME_PREFIX)) {
149 		    if (name != null) {
150 			throw new IOException("Preferred name without " +
151 					      "preference value");
152 		    }
153 		    name = line.substring(NAME_PREFIX.length()).trim();
154 		    continue;
155 		    
156 		} else if (line.startsWith(PREF_PREFIX)) {
157 		    String value = line.substring(PREF_PREFIX.length()).trim();
158 		    if (value.equals("")) {
159 			throw new IOException("Empty preference value not " +
160 					      "permitted");
161 		    }
162 		    preference = Boolean.valueOf(value);
163 		} else {
164 		    throw new IOException("unrecognized preference entry: " +
165 					  line);
166 		}
167 		if (name == null) {
168 		    if (preference != null) {
169 			if (!isEmpty) {
170 			    throw new IOException("default preference must " +
171 				"be the first expression and can not " +
172 				"be redefined");
173 			}
174 			defaultPreference = preference;
175 			preference = null;
176 			isEmpty = false;
177 		    }
178 		    
179 		    /* it is ok for both name and preference to be null */
180 		    
181 		} else if (preference != null) {
182 		    if (name.startsWith("/") ||
183 			name.startsWith("*") ||
184 			name.startsWith("-") ||			
185 			name.startsWith("."))
186 		    {
187 			throw new IOException("Invalid character " +
188 					      "at name beginning: " + name);
189 		    } else if (name.endsWith("/*")) {
190 			mapPut(packagePrefs,
191 			    name.substring(0, name.length() - 2), preference);
192 		    } else if (name.endsWith("/-")) {
193 			mapPut(namespacePrefs,
194 			    name.substring(0, name.length() - 2), preference);
195 		    } else if (name.endsWith("/")) {
196 			mapPut(packagePrefs,
197 			    name.substring(0, name.length() - 1), preference);
198 		    } else {
199 			/* no wildcard; must be a full resource name */
200 			int state = (preference.booleanValue() ?
201 				     NAME_PREFERRED : NAME_NOT_PREFERRED);
202 			mapPut(completeNamePrefs, name, Integer.valueOf(state));
203 		    }
204 		    preference = null;
205 		    name = null;
206 		}
207 	    }
208 	    if (name != null) {
209 		throw new IOException("Preferred name without " +
210 				      "preference value");
211 	    }
212 	}
213 	if (isEmpty) {
214 	    throw new IOException("Empty preferences list is invalid");
215 	}
216         this.defaultPreference = defaultPreference;
217     }
218 
219     /**
220      * Reads the next line from the specified BufferedReader, removing
221      * leading and trailing whitespace and comments.  Null is returned
222      * on EOF.
223      **/
224     private String readLineTrimComments(BufferedReader br) throws IOException {
225 	String line = br.readLine();
226 	if (line != null) {
227 	    line = line.trim();
228 	    if (line.indexOf('#') == 0) {
229 		line = "";
230 	    }
231 	}
232 	return line;
233     }

235     /**
236      * Insert a preference expression and value into a given map.
237      */
238     private void mapPut(Map<String,Object> map, String name, Object preference)
239 	throws IOException
240     {
241 	isEmpty = false;
242 	
243 	if ((name == null) || (name.length() == 0)) {
244 	    throw new IOException("no name specified in preference" +
245 				  " expression");
246 	}
247 	if (map.put(name, preference) != null) {
248 	    throw new IOException("duplicate map entry: " + name);
249 	}
250     }

252     /**
253      * Write the preferences to the specified OutputStream using the
254      * preference list syntax. Preference expressions are written in
255      * the following order:
256      *
257      * Complete name expressions
258      *
259      * Package expressions
260      * 
261      * Namespace expressions
262      *
263      * @param out the stream to which formatted preference information
264      * will be written
265      * @throws IOException if an error occurs while writing to the stream
266      */
267     public void write(OutputStream out) throws IOException {
268         BufferedWriter bw = new BufferedWriter
269             (new OutputStreamWriter(out, "UTF8"));
270 
271 	/* write the specification header */
272         bw.write(HEADER_TITLE + "1.0\n");
273 	bw.write(PREF_PREFIX + " " + defaultPreference + "\n\n");
274 
275 	/* write out most specific preferences first */
276         synchronized (completeNamePrefs){
277             writeMap(completeNamePrefs, bw, "");
278         }
279         synchronized (packagePrefs){
280             writeMap(packagePrefs, bw, "/*");
281         }
282         synchronized (namespacePrefs){
283             writeMap(namespacePrefs, bw, "/-");
284         }
285 
286 	bw.flush();
287     }
288 
289     /**
290      * Write the contents of the map into <code>out</code> using the
291      * preference syntax.
292      */
293     private void writeMap(Map<String,Object> prefs, Writer out, String suffix)
294 	throws IOException
295     {
296 	Iterator<String> i = (new TreeSet<String>(prefs.keySet())).iterator();
297 	while (i.hasNext()) {
298 	    String current = i.next();
299 	    out.write(NAME_PREFIX + " " + current +
300 		      suffix + "\n");
301 	    Object value = prefs.get(current);
302 	    if (value instanceof Boolean) {
303 		out.write(PREF_PREFIX + " " + value + "\n\n");
304 	    } else if (value instanceof Integer) {
305 		int state = ((Integer) value).intValue();
306 		if ((state == NAME_PREFERRED) ||
307 		    (state == NAME_PREFERRED_RESOURCE_EXISTS))
308 		{
309 		    out.write(PREF_PREFIX + " " + true + "\n\n");
310 		} else {
311 		    out.write(PREF_PREFIX + " " + false + "\n\n");   
312 		}
313 	    }
314 	}
315     }

318      * Returns the preference setting that will be applied to names
319      * which have no explicit preference setting in contained
320      * preference settings.  The default preference is set by the
321      * first preference setting with no associated name in a
322      * preferences list file.
323      *
324      * @return default boolean preference value for these preferences
325      */
326     public Boolean getDefaultPreference() {
327 	return defaultPreference;
328     }

330     /**
331      * Enable MarshalInputStream to optimize preference information:
332      * permits complete name expressions to be added for names that
333      * only match wild-card expressions.  These added expressions hold
334      * data that tells if a resource has been loaded.
335      *
336      * This method makes this object mutable.  Synchronization must be
337      * used to ensure consistent preference state when this method is
338      * called.
339      *
340      * @param name the name for which preferred state will be set
341      * @param prefState the preferred state for the given name
342      * @throws IOException if the name length is zero length
343      */
344     public void setNameState(String name, int prefState)
345 	throws IOException
346     {
347 	isEmpty = false;
348 	
349 	if (name.length() == 0) {
350 	    throw new IOException("no name specified in preference" +
351 				  " expression");
352 	}
353         synchronized (completeNamePrefs){
354             completeNamePrefs.put(name, Integer.valueOf(prefState));
355         }
356     }

358     /**
359      * Searches the preference maps to determine the preference state
360      * of the named resource.  The preference state for the given
361      * resource name is returned.  The preference state is an integer
362      * that is equal to one of the preference state values defined
363      * above.  This state integer tells the preference value of name
364      * and whether its resource is known to exist.
365      *
366      * @param name
367      * @param isClass whether the given <code>name</code> refers to a
368      *        class resource
369      *
370      * @return the state for the given name which will be set to one
371      *         of the following values: NAME_NO_PREFERENCE,
372      *         NAME_NOT_PREFERRED, NAME_PREFERRED,
373      *         NAME_PREFERRED_RESOURCE_EXISTS
374      * 
375      * @throws IOException if an error occurs getting the state for
376      *         the supplied name
377      */
378     public int getNameState(String name, boolean isClass)
379 	throws IOException
380     {
381 	Integer state;
382 
383 	if (isClass) {
384 	    state = getClassNameState(name);
385 	} else {
386             synchronized (completeNamePrefs){
387                 state = (Integer) completeNamePrefs.get(name);
388             }
389 	}
390 	if (state != null) {
391 	    return state.intValue();
392 	}
393 	return NAME_NO_PREFERENCE;
394     }

396     /**
397      * Returns the preference state for a given name (as is done in
398      * getNonclassNameState) but also interprets the notation for
399      * inner classes as a wild card so that the preference value for a
400      * container class propagates to the classes it contains.  More
401      * specific inner classes names override more general container
402      * preference.
403      */
404     private Integer getClassNameState(String name) throws IOException {
405 	if (!name.endsWith(".class")) {
406 	    throw new IOException("requested name state on a " +
407 				  "non-class resource: " + name);
408 	}
409 	
410 	Integer state = null;	
411 	String container = name;
412 	int lastDollar = -1;
413         synchronized(completeNamePrefs){
414             do {
415                 state = (Integer) completeNamePrefs.get(container);
416                 if (state == null) {
417                     lastDollar = container.lastIndexOf("$");
418                     if (lastDollar >= 0) {
419                         container =
420                             container.substring(0, lastDollar) + ".class";
421                     }
422                 }
423             } while ((lastDollar >= 0) && (state == null));
424         }
425 	return state;
426     }

428     /**
429      * Return the boolean value of the most specific wild card
430      * (package and namespace) expression which matches
431      * <code>name</code>.  Package preferences are always more
432      * specific than namespace preferences.
433      *
434      * @param name the resource name to which the returned boolean
435      * value will apply
436      *
437      * @return <code>Boolean.TRUE</code> if <code>name</code> is
438      * preferred. <code>Boolean.FALSE</code> is it is
439      * not. <code>null</code> if there is no wildcard preference for
440      * the name.
441      */
442     public Boolean getWildcardPreference(String name) {
443 	Boolean wildcardPref = null;
444 	
445 	int lastSlash = name.lastIndexOf("/");
446 	if (lastSlash >= 0) {
447 	    String mostSpecific = name.substring(0, lastSlash);
448 	    if (!mostSpecific.equals("")) {
449                 synchronized (packagePrefs){
450                     if (!packagePrefs.isEmpty()) {
451                         wildcardPref = (Boolean)
452                             packagePrefs.get(mostSpecific);
453                     }
454                 }
455 		if (wildcardPref == null) {
456 		    wildcardPref = getNamespacePreference(mostSpecific);
457 		}
458 	    }
459 	}
460 
461 	return wildcardPref;
462     }
463 
464     /**
465      * Return a Boolean for the most specific namespace expression
466      * which matches <code>name</code>. null if the name does not
467      * match a namespace preference expression.
468      */
469     private Boolean getNamespacePreference(String namespace) {
470 	Boolean namespacePref = null;
471 	synchronized (namespacePrefs){
472             if (!namespacePrefs.isEmpty()) {
473                 int lastSlash;
474                 do {
475                     namespacePref =
476                         (Boolean) namespacePrefs.get(namespace);
477 
478                     lastSlash = namespace.lastIndexOf("/");
479                     if (lastSlash >= 0) {
480                         namespace = namespace.substring(0, lastSlash);
481                     }
482 
483                 } while ((lastSlash >= 0) &&
484                          (!namespace.equals("")) &&
485                          namespacePref == null);
486             }
487         }
488 	return namespacePref;
489     }
