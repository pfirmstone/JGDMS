36  /**
37   * A <code>JrmpExporter</code> contains the information necessary to export a
38   * single remote object to the 
39   * <a href="http://java.sun.com/j2se/1.4/docs/guide/rmi/">JRMP</a> runtime.  It
40   * acts as an adapter between existing methods for (un)exporting remote objects
41   * over JRMP (e.g, {@link UnicastRemoteObject#exportObject(Remote)}, 
42   * {@link Activatable#exportObject(Remote, ActivationID, int)}) and the 
43   * {@link Exporter} interface.
44   *
45   * <p>An object exported via a <code>JrmpExporter</code> can customize the
46   * following properties that govern invocation behavior and other
47   * characteristics of the exported remote object and its stub:
48   * <ul>
49   * 
50   * <li><code>port</code>: the port number for the {@link java.net.ServerSocket}
51   * on which the JRMP runtime will listen for incoming calls to the exported
52   * object.
53   * 
54   * <li>{@link RMIClientSocketFactory}: the factory object used by client stubs
55   * to create {@link java.net.Socket} objects over which to issue calls to the
56   * exported object.  If <code>null</code>, then the global JRMP socket factory
57   * will be used (i.e., the value returned by 
58   * {@link java.rmi.server.RMISocketFactory#getSocketFactory}), or if there
59   * isn't one set, then the default global JRMP socket factory will be used
60   * (i.e., the value returned by 
61   * {@link java.rmi.server.RMISocketFactory#getDefaultSocketFactory}).
62   * 
63   * <li>{@link RMIServerSocketFactory}: the factory object used by the JRMP
64   * runtime to create <code>ServerSocket</code> objects over which to receive
65   * calls to the exported object.  If <code>null</code>, then the global JRMP
66   * socket factory will be used, or if there isn't one set, then the default
67   * global JRMP socket factory will be used.  The default global JRMP socket
68   * factory returns a <code>ServerSocket</code> for an anonymous port if passed
69   * a port number of zero.
70   * 
71   * <li>{@link ActivationID}: the <code>ActivationID</code> identifying this
72   * remote object to the activation system, or <code>null</code> if the remote
73   * object is not to be exported as activatable.
74   * </ul>
75   * 
76   * <p>This exporter is a front-end adapter on top of
77   * <code>UnicastRemoteObject</code> and <code>Activatable</code>; exporting
78   * remote objects through this exporter is equivalent to doing so directly via
79   * the various <code>exportObject</code> methods defined by the aforementioned
80   * classes.
81   * 
82   * @author	Sun Microsystems, Inc.
83   * @since	2.0
84   *
85   * 
86   *
87   * <p>This implementation uses the {@link Logger} named
88   * <code>net.jini.jrmp.JrmpExporter</code> to log
89   * information at the following levels:
90   *
91   * <table summary="Describes what is logged by JrmpExporter at various
92   *        logging levels" border=1 cellpadding=5>
93   *
94   * <tr> <th scope="col"> Level <th scope="col"> Description
95   *
96   * <tr> <td> {@link Level#FINE FINE} <td> successful export of object
97   * 
98   * <tr> <td> {@link Level#FINE FINE} <td> attempted unexport of object
99   *
100  * </table>
101  */
102 public final class JrmpExporter implements Exporter {
103     
104     private static final Logger logger =
105 	Logger.getLogger("net.jini.jrmp.JrmpExporter");
106 
107     private final int port;
108     private final RMIClientSocketFactory csf;
109     private final RMIServerSocketFactory ssf;
110     private final ActivationID id;
111     private WeakReference ref;
112 
113     /**
114      * Creates an exporter for a non-activatable JRMP "unicast" remote object
115      * that exports on an anonymous port and does not use custom socket
116      * factories.
117      */
118     public JrmpExporter() {
119 	this(0);
120     }
121     
122     /**
123      * Creates an exporter for a non-activatable JRMP "unicast" remote object
124      * that exports on the specified TCP port and does not use custom socket
125      * factories.
126      * 
127      * @param	port number of the port on which an exported object will
128      * 		receive calls (if zero, an anonymous port will be chosen)
129      */
130     public JrmpExporter(int port) {
131 	this(port, null, null);
132     }
133     
134     /**
135      * Creates an exporter for a non-activatable JRMP "unicast" remote object
136      * that exports on the specified TCP port and uses sockets created by the
137      * given custom socket factories.
138      * 
139      * @param	port number of the port on which an exported object will
140      * 		receive calls (if zero, an anonymous port will be chosen)
141      * @param	csf client-side socket factory (if null, the global JRMP socket
142      * 		factory or, if necessary, the default global JRMP socket
143      * 		factory will be used to create client-side sockets)
144      * @param	ssf server-side socket factory (if null, the global JRMP socket
145      * 		factory or, if necessary, the default global JRMP socket
146      * 		factory will be used to create server-side sockets)
147      */
148     public JrmpExporter(int port,
149 			RMIClientSocketFactory csf,
150 			RMIServerSocketFactory ssf)
151     {
152 	this.port = port;
153 	this.csf = csf;
154 	this.ssf = ssf;
155 	id = null;
156     }
157     
158     /**
159      * Creates an exporter for an activatable JRMP remote object with the given
160      * activation ID that exports on the specified TCP port and does not use
161      * custom socket factories.
162      * 
163      * @param	id activation ID associated with the object to export
164      * @param	port number of the port on which an exported object will
165      * 		receive calls (if zero, an anonymous port will be chosen)
166      * @throws	NullPointerException if <code>id</code> is <code>null</code>
167      */
168     public JrmpExporter(ActivationID id, int port) {
169 	this(id, port, null, null);
170     }
171     
172     /**
173      * Creates an exporter for an activatable JRMP remote object with the given
174      * activation ID that exports on the specified TCP port and uses sockets
175      * created by the given custom socket factories.
176      * 
177      * @param	id activation ID associated with the object to export
178      * @param	port number of the port on which an exported object will
179      * 		receive calls (if zero, an anonymous port will be chosen)
180      * @param	csf client-side socket factory (if null, the global JRMP socket
181      * 		factory or, if necessary, the default global JRMP socket
182      * 		factory will be used to create client-side sockets)
183      * @param	ssf server-side socket factory (if null, the global JRMP socket
184      * 		factory or, if necessary, the default global JRMP socket
185      * 		factory will be used to create server-side sockets)
186      * @throws	NullPointerException if <code>id</code> is <code>null</code>
187      */
188     public JrmpExporter(ActivationID id, int port,
189 			RMIClientSocketFactory csf,
190 			RMIServerSocketFactory ssf)
191     {
192 	if (id == null) {
193 	    throw new NullPointerException();
194 	}
195 	this.id = id;
196 	this.port = port;
197 	this.csf = csf;
198 	this.ssf = ssf;
199     }
200     
201     /**
202      * Returns the port used by this exporter, or zero if an anonymous port is
203      * used.
204      * 
205      * @return	port number, or zero if anonymous
206      */
207     public int getPort() {
208 	return port;
209     }
210     
211     /**
212      * Returns the client socket factory for this exporter, or
213      * <code>null</code> if none (in which case {@link java.net.Socket} objects
214      * are created directly).
215      * 
216      * @return	client socket factory, or <code>null</code> if none
217      */
218     public RMIClientSocketFactory getClientSocketFactory() {
219 	return csf;
220     }
221     
222     /**
223      * Returns the server socket factory for this exporter, or
224      * <code>null</code> if none (in which case
225      * <code>java.net.ServerSocket</code> objects are created directly).
226      * 
227      * @return	server socket factory, or <code>null</code> if none
228      */
229     public RMIServerSocketFactory getServerSocketFactory() {
230 	return ssf;
231     }
232     
233     /**
234      * Returns the activation ID associated with the object exported by this
235      * exporter, or <code>null</code> if activation is not being used with this
236      * exporter.
237      * 
238      * @return	activation ID, or <code>null</code> if none
239      */
240     public ActivationID getActivationID() {
241 	return id;
242     }
243     
244     /**
245      * Exports a remote object, <code>impl</code>, to the JRMP runtime and
246      * returns a proxy (stub) for the remote object.  This method cannot be
247      * called more than once to export a remote object or an
248      * <code>IllegalStateException</code> will be thrown.
249      * 
250      * <p>If this exporter was created with a constructor that accepted a
251      * <code>java.rmi.activation.ActivationID</code>, then calling this method
252      * is equivalent to invoking
253      * <code>java.rmi.activation.Activatable.exportObject</code> with the
254      * appropriate impl, <code>ActivationID</code>, port,
255      * <code>RMIClientSocketFactory</code>, and
256      * <code>RMIServerSocketFactory</code> values.  Otherwise, calling this
257      * method is equivalent to invoking
258      * <code>java.rmi.server.UnicastRemoteObject.exportObject</code> with the
259      * appropriate impl, port, <code>RMIClientSocketFactory</code>, and
260      * <code>RMIServerSocketFactory</code> values.
261      *
262      * @throws 	NullPointerException {@inheritDoc}
263      * @throws	IllegalStateException {@inheritDoc}
264      */
265     public synchronized Remote export(Remote impl)
266 	throws ExportException
267     {
268 	if (impl == null) {
269 	    throw new NullPointerException();
270 	} else if (ref != null) {
271 	    throw new IllegalStateException(
272 		"object already exported via this exporter");
273 	}
274 	ref = new WeakReference(impl);
275 
276 	try {
277 	    Remote proxy = (id != null) ? 
278 		Activatable.exportObject(impl, id, port, csf, ssf) :
279 		UnicastRemoteObject.exportObject(impl, port, csf, ssf);
280 	    if (logger.isLoggable(Level.FINE)) {
281 		logger.log(Level.FINE, 
282 		    "export of {0} via {1} returns proxy {2}",
283 		    new Object[]{ impl, this, proxy });
284 	    }
285 	    return proxy;
286 	} catch (ExportException ex) {
287 	    throw ex;
288 	} catch (RemoteException ex) {
289 	    throw new ExportException("export failed", ex);
290 	}
291     }
292     
293     /**
294      * Unexports the remote object exported via this exporter's
295      * <code>export</code> method such that the object can no longer
296      * accept incoming remote calls that were possible as a result of
297      * exporting via this exporter.
298      *
299      * <p>If <code>force</code> is <code>true</code>, the object is forcibly
300      * unexported even if there are pending or in progress calls to the remote
301      * object via this exporter.  If <code>force</code> is <code>false</code>,
302      * the object is only unexported if there are no pending or in progress
303      * calls to the remote object via this exporter.  This method is equivalent
304      * to calling <code>java.rmi.activation.Activatable.unexportObject</code>
305      * or <code>java.rmi.server.UnicastRemoteObject.unexportObject</code> and
306      * passing the "impl" that was previously passed to this exporter's
307      * <code>export</code> method, depending on whether or not this exporter
308      * was created with a constructor that accepted a
309      * <code>java.rmi.activation.ActivationID</code> value.
310      * 
311      * <p>The return value is <code>true</code> if the object is (or was
312      * previously) unexported, and <code>false</code> if the object is still
313      * exported.
314      *
315      * @throws	IllegalStateException {@inheritDoc}
316      */
317     public synchronized boolean unexport(boolean force) {
318 	if (ref == null) {
319 	    throw new IllegalStateException(
320 		"an object has not been exported via this exporter");
321 	}
322 	Remote impl = (Remote) ref.get();
323 	if (impl == null) {
324 	    return true;
325 	}
326 	try {
327 	    boolean result = (id != null) ?
328 		Activatable.unexportObject(impl, force) :
329 		UnicastRemoteObject.unexportObject(impl, force);
330 	    if (logger.isLoggable(Level.FINE)) {
331 		logger.log(Level.FINE, "unexport on {0} returns {1}",
332 		    new Object[]{ this, Boolean.valueOf(result) });
333 	    }
334 	    return result;
335 	} catch (NoSuchObjectException ex) {
336 	    return true;
337 	}
338     }
339     
340     /**
341      * Returns the string representation for this exporter.
342      * 
343      * @return the string representation for this exporter
344      */
345     public String toString() {
346 	return (id != null) ?
347 	    "JrmpExporter[" + id + "," + port + "," + csf + "," + ssf + "]" :
348 	    "JrmpExporter[" + port + "," + csf + "," + ssf + "]";
349     }
