20  import java.io.InvalidObjectException;
21  import java.lang.reflect.Method;
22  import net.jini.core.constraint.MethodConstraints;
23  import net.jini.core.constraint.RemoteMethodControl;
24  
25  /**
26   * Provided for backward compatibility, migrate to new name space.
27   */
28  @Deprecated
29  public class ConstrainableProxyUtil {
30      
31      /** This class cannot be instantiated. */
32      private ConstrainableProxyUtil() {
33  	throw new AssertionError();
34      }
35  
36      /**
37       * Creates a {@link MethodConstraints} using the constraints in
38       * <code>methodConstraints</code>, but with the methods remapped according
39       * to <code>mappings</code>, where the first element of each pair of
40       * elements is mapped to the second.  For example, if
41       * <code>methodConstraints</code> returns constraints <code>C1</code> for
42       * method <code>M1</code>, and the elements of mappings are methods
43       * <code>M1</code> and <code>M2</code>, then the resulting method
44       * constraints return <code>C1</code> for method <code>M2</code>.
45       *
46       * @param methodConstraints the method constraints whose methods should be
47       *	      translated, or <code>null</code> for empty constraints
48       * @param mappings the method mappings
49       * @return the translated method constraints
50       * @throws NullPointerException if <code>mappings</code> is
51       *	       <code>null</code> or contains <code>null</code> elements
52       * @throws IllegalArgumentException if <code>mappings</code> contains an
53       *	       odd number of elements
54       */
55      public static MethodConstraints translateConstraints(
56  	MethodConstraints methodConstraints,
57  	Method[] mappings)
58      {
59  	return org.apache.river.proxy.ConstrainableProxyUtil.translateConstraints(
60                  methodConstraints, mappings);
61      }
62  
63      /**
64       * Test to see if two {@link MethodConstraints} instances are
65       * equivalent given a method-to-method mapping. Only the
66       * constraints for methods that appear in the mapping are
67       * compared. The mapping is represented by an array, <code>
68       * mappings</code>, of 2n {@link Method} objects.  For all values
69       * p less than n the constraints associated with
70       * <code>mappings[2p]</code> in <code>methodConstraints1</code>
71       * are compared to the constraints associated with
72       * <code>mappings[2p+1]</code> in
73       * <code>methodConstraints2</code>. If null is passed in for both
74       * instances they are considered equivalent.
75       *
76       * @param methodConstraints1 the first <code>MethodConstraints</code>
77       *        object to compare.
78       * @param methodConstraints2 the second <code>MethodConstraints</code>
79       *        object to compare.
80       * @param mappings the method-to-method mapping.
81       * @return <code>true</code> if the <code>MethodConstraints</code> 
82       *         instances represent equivalent constraints, returns
83       *         <code>false</code> otherwise.
84       * @throws NullPointerException if <code>mapping</code> is
85       *         <code>null</code> contains <code>null</code> elements.
86       * @throws IllegalArgumentException if <code>mapping</code> contains an
87       *         odd number of elements 
88       */
89      public static boolean equivalentConstraints(
90          MethodConstraints methodConstraints1, 
91  	MethodConstraints methodConstraints2,
92  	Method[] mappings) 
93      {
94  	return org.apache.river.proxy.ConstrainableProxyUtil.equivalentConstraints(
95                  methodConstraints1, methodConstraints2, mappings);
96      }
97  
98      /**
99       * Verify that an object, <code>proxy</code>, is an instance of
100      * {@link RemoteMethodControl} its {@link MethodConstraints} are
101      * equivalent to another <code>MethodConstraints</code> instance,
102      * <code>methodConstraints</code> once a mapping has been applied.
103      * If <code>proxy</code> does not implement
104      * <code>RemoteMethodControl</code> or the associated constraints are
105      * not equivalent throw an {@link InvalidObjectException}. The
106      * mapping is represented by an array, <code>mappings</code>, of
107      * 2n {@link Method} objects. For all values p less than n the
108      * constraints associated with <code>mappings[2p]</code> in
109      * <code>methodConstraints</code> are compared to the constraints
110      * associated with <code>mappings[2p+1]</code> in the
111      * <code>MethodConstraints</code> returned by
112      * <code>proxy.getConstraints</code>. Will also return normally if
113      * both <code>methodConstraints</code> and the value returned by
114      * <code>proxy.getConstraints</code> are <code>null</code>.
115      *
116      * @param methodConstraints the method constraints 
117      *        <code>proxy</code> should have.
118      * @param proxy the proxy to test, must implement 
119      *        <code>RemoteMethodControl</code>.
120      * @param mappings the method to method mapping     
121      * @throws NullPointerException if <code>mappings</code> or
122      *         <code>proxy</code> is <code>null</code> or if
123      *         <code>mapping</code> contains <code>null</code> elements
124      * @throws IllegalArgumentException if <code>mappings</code> contains an
125      *	       odd number of elements
126      * @throws InvalidObjectException if <code>proxy</code> does
127      *         not implement <code>RemoteMethodControl</code>, or if
128      *         the constraints on <code>proxy</code> are not
129      *         equivalent to <code>methodConstraints</code>.  
130      */
131     public static void verifyConsistentConstraints(
132 	MethodConstraints methodConstraints, Object proxy, Method[] mappings)
133 	throws InvalidObjectException
134     {
135 	org.apache.river.proxy.ConstrainableProxyUtil.verifyConsistentConstraints(
136                 methodConstraints, proxy, mappings);
137     }
