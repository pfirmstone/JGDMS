18  package com.sun.jini.proxy;
19  
20  import java.io.IOException;
21  import java.io.InvalidObjectException;
22  import java.io.ObjectInput;
23  import java.io.ObjectInputStream;
24  import java.io.ObjectStreamException;
25  import java.io.Serializable;
26  import java.util.Collection;
27  import java.util.Iterator;
28  import net.jini.io.MarshalledInstance;
29  import net.jini.io.ObjectStreamContext;
30  import net.jini.io.context.IntegrityEnforcement;
31  import org.apache.river.api.io.AtomicSerial;
32  import org.apache.river.api.io.AtomicSerial.GetArg;
33  import org.apache.river.api.io.AtomicSerial.ReadInput;
34  import org.apache.river.api.io.AtomicSerial.ReadObject;
35  
36  /**
37   * Wrapper around {@link MarshalledInstance} that samples the integrity setting
38   * (if any) of the stream it is unmarshalled from, and uses that setting to
39   * determine whether or not to verify codebase integrity when calling the
40   * {@link MarshalledInstance#get(boolean) get} method of the contained
41   * <code>MarshalledInstance</code>.
42   *
43   * @author Sun Microsystems, Inc.
44   *
45   * @since 2.0
46   */
47  @Deprecated
48  @AtomicSerial
49  public class MarshalledWrapper implements Serializable {
50      
51      private static class RO implements ReadObject{
52  	boolean integrity;
53  	@Override
54  	public void read(ObjectInput input) throws IOException, ClassNotFoundException {
55  	    integrity = integrityEnforced(input);
56  	}
57  	
58      }
59      
60      @ReadInput
61      static ReadObject read(){
62  	return new RO();
63      }
64  
65      private static final long serialVersionUID = 2L;
66  
67      /** 
68       * The wrapped MarshalledInstance.
69       *
70       * @serial
71       */
72      private final MarshalledInstance instance;
73      /** 
74       * Flag set to true if this instance was unmarshalled from an
75       * integrity-protected stream, or false otherwise
76       */
77      private transient boolean integrity = false;
78  
79      /**
80       * Returns the integrity protection setting of the given stream, determined
81       * as follows: if the stream implements {@link ObjectStreamContext} and has
82       * in its context collection an object of type
83       * {@link IntegrityEnforcement}, the
84       * {@link IntegrityEnforcement#integrityEnforced integrityEnforced} method
85       * of that object is called and the resulting value returned; otherwise,
86       * <code>false</code> is returned.
87       *
88       * @param stream the given stream
89       * @return integrity protection setting of the given stream
90       */
91      public static boolean integrityEnforced(ObjectInput stream) {
92  	if (stream instanceof ObjectStreamContext) {
93  	    return  integrityEnforced(((ObjectStreamContext) stream));
94  	}
95  	return false;
96      }
97      
98      /**
99       * Returns the integrity protection setting in the given ObjectStreamContext.
100      * @param context
101      * @return 
102      */
103     public static boolean integrityEnforced(ObjectStreamContext context) {
104 	Collection ctx = context.getObjectStreamContext();
105 	    for (Iterator i = ctx.iterator(); i.hasNext(); ) {
106 		Object obj = i.next();
107 		if (obj instanceof IntegrityEnforcement) {
108 		    return ((IntegrityEnforcement) obj).integrityEnforced();
109 		}
110 	    }
111 	return false;
112     }
113 
114     /**
115      * Creates a new <code>MarshalledWrapper</code> wrapping a
116      * <code>MarshalledInstance</code> containing the given object.
117      *
118      * @param obj object to create <code>MarshalledInstance</code> with
119      * @throws IOException if <code>MarshalledInstance</code> creation fails
120      */
121     public MarshalledWrapper(Object obj) throws IOException {
122 	instance = new MarshalledInstance(obj);
123     }
124 
125     /**
126      * Creates a new <code>MarshalledWrapper</code> wrapping the given
127      * <code>MarshalledInstance</code>.
128      *
129      * @param instance <code>MarshalledInstance</code> to wrap
130      * @throws NullPointerException if <code>instance</code> is
131      * <code>null</code>
132      */
133     public MarshalledWrapper(MarshalledInstance instance) {
134 	this(check(instance), false);
135     }
136     
137     public MarshalledWrapper(GetArg arg) throws IOException {
138 	this(validate(arg.get("instance", null, MarshalledInstance.class)),
139 		((RO) arg.getReader()).integrity);
140     }
141     
142     private MarshalledWrapper(MarshalledInstance instance, boolean integrity){
143 	this.instance = instance;
144 	this.integrity = integrity;
145     }
146     
147     private static MarshalledInstance check( MarshalledInstance instance){
148 	if (instance == null) {
149 	    throw new NullPointerException();
150 	}
151 	return instance;
152     }
153     
154     private static MarshalledInstance validate(MarshalledInstance instance) throws InvalidObjectException{
155 	if (instance == null) {
156 	    throw new InvalidObjectException("null instance");
157 	}
158 	return instance;
159     }
160 
161     /**
162      * Returns the result of calling the {@link MarshalledInstance#get(boolean)
163      * get} method of the wrapped <code>MarshalledInstance</code>, passing the
164      * integrity value sampled during deserialization as the
165      * <code>verifyCodebaseIntegrity</code> argument.  If this
166      * <code>MarshalledWrapper</code> instance was not produced by
167      * deserialization or was deserialized from a stream with no integrity
168      * protection setting, then a <code>verifyCodebaseIntegrity</code> value of
169      * <code>false</code> is used.
170      *
171      * @return the object unmarshalled by the wrapped
172      * <code>MarshalledInstance</code>
173      * @throws IOException if an <code>IOException</code> occurs during
174      * unmarshalling
175      * @throws ClassNotFoundException if any classes necessary for
176      * reconstructing the object being unmarshalled cannot be found
177      */
178     public Object get() throws IOException, ClassNotFoundException {
179 	return instance.get(integrity);
180     }
181 
182     /**
183      * Returns the <code>MarshalledInstance</code> wrapped by this
184      * <code>MarshalledWrapper</code>.
185      *
186      * @return wrapped <code>MarshalledInstance</code>
187      */
188     public MarshalledInstance getMarshalledInstance() {
189 	return instance;
190     }
191 
192     /**
193      * Returns the integrity value sampled during deserialization of this
194      * <code>MarshalledWrapper</code> instance, or <code>false</code> if this
195      * instance was not produced by deserialization or was deserialized from a
196      * stream with no integrity protection setting.
197      *
198      * @return sampled integrity value
199      */
200     public boolean getIntegrity() {
201 	return integrity;
202     }
203 
204     /**
205      * Returns the hash code value for this <code>MarshalledWrapper</code>.
206      *
207      * @return the hash code value for this <code>MarshalledWrapper</code>.
208      */
209     public int hashCode() {
210 	return MarshalledWrapper.class.hashCode() ^ instance.hashCode();
211     }
212 
213     /**
214      * Compares the specified object with this <code>MarshalledWrapper</code>
215      * for equality.  This method returns <code>true</code> if and only if the
216      * specified object is of the same class as this object and its wrapped
217      * <code>MarshalledInstance</code> is equal to the one in this object.
218      *
219      * @param obj object to compare with
220      * @return <code>true</code> if <code>obj</code> is equivalent to
221      * this object; <code>false</code> otherwise
222      */
223     public boolean equals(Object obj) {
224 	return obj == this ||
225 	       (obj != null &&
226 		obj.getClass() == getClass() &&
227 		instance.equals(((MarshalledWrapper) obj).instance));
228     }
229 
230     /**
231      * Samples integrity protection setting (if any) of the stream from which
232      * this instance is being deserialized.
233      */
234     private void readObject(ObjectInputStream in)
235 	throws IOException, ClassNotFoundException
236     {
237 	in.defaultReadObject();
238 	if (instance == null) {
239 	    throw new InvalidObjectException("null instance");
240 	}
241 	integrity = integrityEnforced(in);
242     }
243 
244     /**
245      * Throws InvalidObjectException, since data for this class is required.
246      */
247     private void readObjectNoData() throws ObjectStreamException {
248 	throw new InvalidObjectException("no data");
249     }
250 }
