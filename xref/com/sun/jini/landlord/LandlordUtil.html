18  package com.sun.jini.landlord;

20  import java.util.Map;
21  import net.jini.id.Uuid;
22  
23  /**
24   * Provided for backward compatibility, migrate to new name space.
25   */
26  @Deprecated
27  public class LandlordUtil {
28      /**
29       * Call <code>landlord.renew()</code> for each object in 
30       * <code>cookie[]</code>, passing <code>cookie[i]</code> and
31       * <code>durations[i]</code>.  Gather the results and 
32       * exceptions into a <code>RenewResults</code> object.
33       *
34       * @param landlord   A grantor of leases.
35       * @param cookies an array of <code>Uuid</code>s, each uniquely and
36       *                universally identifying a lease
37       * @param durations an array of longs, each representing a
38       *                  duration in milliseconds that the client
39       *                  wants the lease associated with the object
40       *                  from the corresponding element of
41       *                  <code>cookies</code> renewed for.
42       * @return A <code>RenewResults</code> object that contains the new
43       *         duration of each lease that was successfully renewed or
44       *         the exception encountered for each lease that could not
45       *         be renewed.
46       */ 
47      public static Landlord.RenewResults renewAll(LocalLandlord landlord,
48  					Uuid[] cookies, long[] durations) 
49      {
50          org.apache.river.landlord.Landlord.RenewResults results =
51  	org.apache.river.landlord.LandlordUtil.renewAll(landlord, cookies, durations);
52          int len = cookies.length;
53          long [] granted = new long[len];
54          Exception [] denied = new Exception[len];
55          for (int i = 0; i<len ; i++){
56              granted[i] = results.getGranted(i);
57              denied[i] = results.getDenied(i);
58          }
59          
60          return new Landlord.RenewResults(granted, denied);
61      }
62  
63      /**
64       * Call <code>landlord.cancel()</code> for each object in 
65       * <code>cookies[]</code>, passing <code>cookies[i]</code>.
66       *
67       * @param landlord a grantor of leases
68       * @param cookies an array of <code>Uuid</code>s, each uniquely and
69       *                universally identifying a lease
70       * @return If no exceptions are encountered, return
71       *         <code>null</code>, otherwise, return a Map that 
72       *         for each failed <code>cancel()</code> call maps
73       *         the passed cookie object to the exception that 
74       *         was generated
75       */
76      public static Map cancelAll(LocalLandlord landlord, Uuid[] cookies) {
77  	return org.apache.river.landlord.LandlordUtil.cancelAll(landlord, cookies);
78      }
79  }
