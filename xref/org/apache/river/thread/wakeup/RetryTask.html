20  import java.util.logging.Level;
21  import java.util.logging.Logger;
22  import org.apache.river.constants.TimeConstants;

24  /**
25   * An abstract class for building tasks that retry on failure after a
26   * timeout period.  This builds upon <code>TaskManager</code> for task
27   * execution and <code>WakeupManager</code> for retry scheduling.  You
28   * extend <code>RetryTask</code>, implementing <code>tryOnce</code> to
29   * represent a single attempt at the task.  If <code>tryOnce</code>
30   * returns <code>true</code>, the attempt was successful and the task
31   * is complete.  If <code>tryOnce</code> returns <code>false</code>,
32   * the task is scheduled for a future retry using <code>WakeupManager</code>.
33   * <p>
34   * The default retry times are defined by this class's implementation
35   * of <code>retryTime</code></code>.  You can override this method to
36   * change the retry times.
37   * <p>
38   * It is legal to reuse the same task again and again, but only after
39   * the task is complete and <code>reset</code> has been called.
40   * Inserting the same task multiple times before it has either been
41   * cancelled or completed successfully will generate unpredictable
42   * behavior.<p>
43   *
44   * This class uses the {@link Logger} named
45   * <code>org.apache.river.thread.RetryTask</code> to log information at
46   * the following logging levels: <p>
47   *
48   * <table border=1 cellpadding=5
49   *       summary="Describes logging performed by RetryTask at different
50   *	          logging levels">
51   *
52   * <tr> <th> Level <th> Description
53   *
54   * <tr> <td> FINEST <td> after a failed attempt, when should 
55   *                       the task be scheduled for a re-try
56   *
57   * </table>
58   *
59   * @author Sun Microsystems, Inc.
60   *
61   * @see TaskManager
62   * @see WakeupManager
63   */

79  /**
80   *
81   * @param <V>
82   */
83  public abstract class RetryTask<V> implements RunnableFuture<V>, ObservableFuture<V>, TimeConstants {
84      private final TaskManager	  manager;	// the TaskManager for this task
85      private final ExecutorService executor;
86      private RetryTime	  retry;	// the retry object for this task
87      private volatile boolean	  cancelled;	// have we been cancelled?
88      private boolean	  complete;	// have we completed successfully?
89      private volatile Ticket	  ticket;	// the WakeupManager ticket
90      private final long	  startTime;	// the time when we were created 
91      private final AtomicInteger attempt;	// the current attempt number
92      private final WakeupManager wakeup;       // WakeupManager for retry scheduling
93      private final List<FutureObserver<V>> listeners;
94  
95      /**
96       * Default delay backoff times.  These are converted from
97       * intervals to "time since start" by the static block below.
98       *
99       * @see #retryTime 
100      */
101     private static final long[] delays = {
102 	 0, // First value is never read
103 	 1 * SECONDS,
104 	 5 * SECONDS,
105 	10 * SECONDS,
106 	 1 * MINUTES,
107 	 1 * MINUTES,
108 	 5 * MINUTES,
109     };
110 
111     /** Logger for this class */
112     protected static final Logger logger = 
113 	Logger.getLogger("org.apache.river.thread.RetryTask");
114 
115     /**
116      * Create a new <code>RetryTask</code> that will be scheduled with
117      * the given task manager, and which will perform retry scheduling 
118      * using the given wakeup manager.
119      * @deprecated 
120      */
121     @Deprecated
122     public RetryTask(TaskManager manager, WakeupManager wakeupManager) {
123 	this.manager = manager;
124         this.executor = null;
125         this.wakeup = wakeupManager;
126         attempt = new AtomicInteger();
127         listeners = new ArrayList<FutureObserver<V>>();
128         startTime = System.currentTimeMillis();
129     }
130     
131     /**
132      * Create a new <code>RetryTask</code> that will be scheduled with
133      * the given executor service, and which will perform retry scheduling 
134      * using the given wakeup manager.
135      * 
136      * TaskManager's Task.runAfter() is not called when using
137      * this constructor.
138      * 
139      * @param executor
140      * @param wakeupManager 
141      */
142     public RetryTask(ExecutorService executor, WakeupManager wakeupManager){
143         this.manager = null;
144         this.executor = executor;
145         this.wakeup = wakeupManager;
146         attempt = new AtomicInteger();
147         listeners = new ArrayList<FutureObserver<V>>();
148         startTime = System.currentTimeMillis();
149     }
150     
151     public boolean addObserver(FutureObserver<V> listener){
152         synchronized (this){
153             if (cancelled) return false;
154             if (complete) {
155                 listener.futureCompleted(this);
156                 return false;
157             }
158             return listeners.add(listener);
159         }
160     }
161 
162     /**
163      * Make a single attempt.  Return <code>true</code> if the attempt
164      * was successful.  If the attempt is not successful, the task
165      * will be scheduled for a future retry.
166      */
167     public abstract boolean tryOnce();
168 
169     /**
170      * The <code>run</code> method used as a
171      * <code>TaskManager&#046;Task</code>.  This invokes
172      * <code>tryOnce</code>.  If it is not successful, it schedules
173      * the task for a future retry at the time it gets by invoking
174      * <code>retryTime</code>.
175      *
176      * @see #tryOnce
177      * @see #startTime 
178      */
179     public void run() {		// avoid retry if cancelled
180         if (cancelled) return;			// do nothing
181 	boolean success = false;
182         try {
183             success = tryOnce();
184         } catch (Throwable t){
185             t.printStackTrace(System.err);
186             if (t instanceof Error) throw (Error) t;
187             if (t instanceof RuntimeException) throw (RuntimeException) t;
188         }
189         if (!success) {		// if at first we don't succeed ...
190             attempt.incrementAndGet();
191             long at = retryTime();	// ... try, try again
192 
193             if (logger.isLoggable(Level.FINEST)) {
194                 logger.log(Level.FINEST, "retry of {0} in {1} ms", 
195                     new Object[]{this, 
196                         Long.valueOf(at - System.currentTimeMillis())});
197             }
198             RetryTime time = null;
199             synchronized (this){
200                 // only create it if we need to
201                 if (retry == null) retry = new RetryTime();	            
202                 time = retry;
203             }
204             ticket = wakeup.schedule(at, time);
205         } else {
206             // Notify was here, however I noticed that during some tests,
207             // the wakeup manager task was scheduled after cancelling.
208             synchronized (this){
209                 complete = true;
210                 notifyAll();	
211                 Iterator<FutureObserver<V>> it = listeners.iterator();
212                 while (it.hasNext()){
213                 it.next().futureCompleted(this);
214                 }
215             } // see waitFor()
216         }
217     }
218 
219     /**
220      * Return the next time at which we should make another attempt.
221      * This is <em>not</em> an interval, but the actual time.
222      * <p>
223      * The implementation is free to do as it pleases with the policy
224      * here.  The default implementation is to delay using intervals of
225      * 1 second, 5 seconds, 10 seconds, 1 minute, and 1 minute between
226      * attempts, and then retrying every five minutes forever.
227      * <p>
228      * The default implementation assumes it is being called from
229      * the default <code>run</code> method and that the current thread
230      * holds the lock on this object. If the caller does
231      * not own the lock the result is undefined and could result in an
232      * exception.
233      */
234     public long retryTime() {
235         int attempt = this.attempt.get();
236 	int index = (attempt < delays.length ? attempt : delays.length - 1); 
237 	return delays[index] + System.currentTimeMillis();
238     }
239 
240     /**
241      * Return the time this task was created.
242      * @return the time this task was created.
243      */
244     public long startTime() {
245 	return startTime;
246     }
247 
248     /**
249      * Return the attempt number, starting with zero.
250      */
251     public int attempt() {
252 	return attempt.get();
253     }
254 
255     /**
256      * Cancel the retrying of the task.  This ensures that there will be
257      * no further attempts to invoke <code>tryOnce</code>.  It will not
258      * interfere with any ongoing invocation of <code>tryOnce</code>
259      * unless a subclass overrides this to do so.  Any override of this
260      * method should invoke <code>super.cancel()</code>.
261      */
262     @Override
263     public boolean cancel(boolean mayInterruptIfRunning) {
264         
265         Ticket ticket = this.ticket;
266         if (ticket != null) wakeup.cancel(ticket);
267         synchronized (this) {
268             cancelled = true;
269             notifyAll();		// see waitFor()
270             Iterator<FutureObserver<V>> it = listeners.iterator();
271             while (it.hasNext()){
272                 it.next().futureCompleted(this);
273             }
274         }
275         return true;
276     }
277 
278     /**
279      * Return <code>true</code> if <code>cancel</code> has been invoked.
280      */
281     @Override
282     public boolean isCancelled() {
283 	return cancelled;
284     }
285 
286     /**
287      * Return <code>true</code> if <code>tryOnce</code> has returned
288      * successfully.
289      */
290     @Override
291     public boolean isDone() {
292         synchronized (this){
293             return complete;
294         }
295     }
296 
297     public boolean waitFor(long duration) throws InterruptedException {
298         if (cancelled) throw new CancellationException("RetryTask was cancelled");
299         synchronized (this){
300             // Moved inside sync block to make check atomic
301             while (!cancelled && !complete) {
302                 if (duration == 0 )wait();
303                 else wait(duration);
304             }
305             return complete;
306         }
307             
308         
309     }
310     
311     @Override
312     public V get() throws InterruptedException, ExecutionException {
313         waitFor(0L);
314         return null;
315     }
316     
317     @Override
318     public V get(long time, TimeUnit unit) throws InterruptedException, 
319             ExecutionException, TimeoutException
320     {
321         waitFor(unit.toMillis(time));
322         return null;
323     }

331     /**
332      * This is the runnable class for the <code>WakeupManager</code>,
333      * since we need different implementations of
334      * <code>WakeupManager&#046;run</code> and
335      * <code>TaskManager&#046;run</code>.  
336      */
337     private class RetryTime implements Runnable {
338 	/**
339 	 * Time to retry the task.
340 	 */
341 	public void run() {
342             ticket = null;
343 	    if (manager != null ) manager.add(RetryTask.this);
344             if (executor != null) executor.submit(RetryTask.this);
345 	}
346     };
