30  /**
31   * A Queue of timed tasks.  Each task implements {@link Runnable}.
32   * Events can either be executed in the queue's thread or in their own thread.
33   * <p>
34   * A task is an object that implements <code>Runnable</code>.  It is
35   * scheduled by invoking {@link #schedule(long, Runnable, WakeupManager.ThreadDesc)
36   * schedule} with a time at which it should be run.  When that time
37   * arrives (approximately) the task will be pulled off the queue and
38   * have its {@link Runnable#run run} method invoked.  <p>
39   *
40   * A <code>schedule</code> request can specify a
41   * {@link WakeupManager.ThreadDesc}, which will define the parameters
42   * of a thread to be created to run the <code>Runnable</code>.  You can
43   * specify the group, whether the thread is a daemon thread, and the priority.
44   * Additionally you can use a subclass of <code>WakeupManager.ThreadDesc</code>
45   * and override the {@link WakeupManager.ThreadDesc#thread thread} method
46   * to further customize thread creation.
47   * <p>
48   * 
49   * When a task is scheduled, a {@link WakeupManager.Ticket} is returned
50   * that can be used to cancel the event if desired.
51   * <p>
52   * 
53   * The queue requires its own thread, whose parameters can be defined
54   * via a <code>ThreadDesc</code> if desired. The queue's thread
55   * will be started when the first task is scheduled. If the queue
56   * becomes empty the thread will be terminated after a 
57   * <a href=#queueThreadTimeout>configurable delay</a>. The thread
58   * will be re-started if a new task is scheduled.
59   * <p>
60   *
61   * While it is theoretically possible to obtain the queue's thread and
62   * interrupt it, the results of doing so are undefined. If a client
63   * wishes to stop the queue's thread the client should either remove
64   * all the tasks or call {@link #stop}. Note, calling
65   * <code>stop</code> will cause future <code>schedule</code> calls to
66   * fail with an <code>IllegalStateException</code>. <p>
67   *
68   * <a name="ConfigEntries"></a>
69   * <code>WakeupManager</code> supports the <code>queueThreadTimeout</code>
70   * configuration entry, with the component
71   * <code>org.apache.river.thread.wakeup.WakeupManager</code>.
72   *
73   * <a name="queueThreadTimeout"></a>
74   * <table summary="Describes the queueThreadTimeout configuration entry" 
75   *                border="0" cellpadding="2">
76   *   <tr valign="top">
77   *     <th scope="col">&#X2022;
78   *     <th scope="col" align="left" colspan="2">
79   *     <code>queueThreadTimeout</code>
80   * 
81   *   <tr valign="top"> <td> &nbsp; <th scope="row" align="right">
82   *     Type: <td> <code>long</code>
83   * 
84   *   <tr valign="top"> <td> &nbsp; <th scope="row" align="right">
85   *     Default: <td> 30,000 milliseconds
86   * 
87   *   <tr valign="top"> <td> &nbsp; <th scope="row" align="right">
88   *     Description:
89   *       <td> How long, in milliseconds, the queue's thread will be
90   *       left running if there are no scheduled tasks. Must be
91   *       a non-negative long value. This configuration entry is 
92   *       consulted when the <code>WakeupManager</code> is initially created.
93   *           
94   * </table>
95   *
96   *
97   * This class uses the {@link Logger} named
98   * <code>org.apache.river.thread.wakeup.WakeupManager</code> to log information at
99   * the following logging levels: 
100  * 
101  * <table border=1 cellpadding=5
102  *       summary="Describes logging performed by WakeupManager at different
103  *	          logging levels">
104  *
105  * <tr> <th> Level <th> Description
106  *
107  * <tr> <td> SEVERE <td> exceptions thrown when we attempt to
108  *                       create the queue's thread
109  * 
110  * <tr> <td> WARNING <td> exceptions thrown by the run methods of tasks, 
111  *                     by the <code>ThreadDesc</code>'s of tasks, or
112  *                     if the queue's thread is interrupted
113  *
114  * <tr> <td> FINEST <td> how many milliseconds until the next event
115  *                       and when the queue's thread is stopped or started
116  *
117  * </table>
118  *
119  * @author Sun Microsystems, Inc.
120  *
121  * @see java.lang.Runnable */
122 public class WakeupManager {

173     /**
174      * Description of a future thread.
175      *
176      * @see WakeupManager#schedule
177      * @see WakeupManager#WakeupManager(WakeupManager.ThreadDesc)
178      */
179     public static class ThreadDesc {
180 	private final ThreadGroup group;	// group to create in
181 	private final boolean daemon;		// create as daemon?
182 	private final int priority;		// priority
183 
184 	/**
185 	 * Equivalent to
186 	 * <pre>
187 	 *     ThreadDesc(null, false)
188 	 * </pre>
189 	 */
190 	public ThreadDesc() {
191 	    this(null, false);
192 	}
193 
194 	/**
195 	 * Equivalent to
196 	 * <pre>
197 	 *     ThreadDesc(group, deamon, Thread.NORM_PRIORITY)
198 	 * </pre>
199 	 */
200 	public ThreadDesc(ThreadGroup group, boolean daemon) {
201 	    this(group, daemon, Thread.NORM_PRIORITY);
202 	}
203 
204 	/**
205 	 * Describe a future thread that will be created in the given group,
206 	 * deamon status, and priority.
207 	 *
208 	 * @param group The group to be created in.  If <code>null</code>,
209 	 *		the thread will be created in the default group.
210 	 * @param daemon The thread will be a daemon thread if this is
211 	 *		<code>true</code>.
212 	 * @param priority The thread's priority.
213 	 * @throws IllegalArgumentException if priority is not 
214 	 *    in between {@link Thread#MIN_PRIORITY} and 
215 	 *    {@link Thread#MAX_PRIORITY}
216 	 */
217 	public ThreadDesc(ThreadGroup group, boolean daemon, int priority) {
218 	    if (priority < Thread.MIN_PRIORITY ||
219 		priority > Thread.MAX_PRIORITY)
220 	    {
221 		throw new IllegalArgumentException("bad value for priority:" +
222 						   priority);
223 	    }
224 
225 	    this.group = group;
226 	    this.daemon = daemon;
227 	    this.priority = priority;
228 	}
229 
230 	/** 
231 	 * The {@link ThreadGroup} the thread will be created in.
232 	 * @return the {@link ThreadGroup} the thread will be created in.
233 	 */
234 	public ThreadGroup getGroup() { return group; }
235 
236 	/** 
237 	 * Returns <code>true</code> if the the thread will be daemon
238 	 * thread, returns <code>false</code> otherwise.
239 	 * @return <code>true</code> if the the thread will be daemon
240 	 * thread, returns <code>false</code> otherwise.
241 	 */
242 	public boolean isDaemon() { return daemon; }
243 
244 	/**
245 	 * The priority the thread should be created with.
246 	 * @return the priority the thread should be created with.
247 	 */
248 	public int getPriority() { return priority; }
249 
250 	/**
251 	 * Create a thread for the given runnable based on the values in this
252 	 * object. May be overridden to give full control over creation
253 	 * of thread.
254 	 * @return a thread to run <code>r</code>, unstarted
255 	 */
256 	public Thread thread(Runnable r) {
257 	    Thread thr;
258 	    if (getGroup() == null)
259 		thr = new Thread(r);
260 	    else
261 		thr = new Thread(getGroup(), r);
262 	    thr.setDaemon(isDaemon());
263 	    thr.setPriority(getPriority());
264 	    return thr;
265 	}
266 
267 	public String toString() {
268 	    return "[" + getGroup() + ", " + isDaemon() + ", " 
269 		+ getPriority() + "]";
270 	}
271     }

273     /**
274      * A ticket that can be used for cancelling a future task.  It
275      * describes the task itself as well. The {@link
276      * WakeupManager#newTicket WakeupManager.newTicket} method
277      * can be used by subclasses of <code>WakeupManager</code> to
278      * create new <code>Ticket</code> instances. 
279      */
280     public static class Ticket implements Comparable<Ticket> {
281 	/** When the task should occur. */
282 	public final long when;
283 	/** The task object to be executed */
284 	public final Runnable task;
285 	/** The <code>ThreadDesc</code>, or <code>null</code> if none. */
286 	public final ThreadDesc desc;
287 
288 	/** Tie beaker used when two tickets have the same value for when */
289 	private final long breaker;
290 
291         private Ticket(long when, Runnable task, ThreadDesc threadDesc,
292 		       long breaker)
293 	{
294 	    if (task == null)
295 		throw new NullPointerException("task not specified");
296 	    this.when = when;
297 	    this.task = task;
298 	    this.desc = threadDesc;
299 	    this.breaker = breaker;
300 	}
301 
302 	public String toString() {
303 	    return dateFmt.format(Long.valueOf(when)) + "(" + when + ")" + ", "
304 		+ task.getClass().getName() + ", " + desc;
305 	}
306 
307 	public boolean equals(Object o) {
308 	    if (!(o instanceof Ticket))
309 		return false;
310 
311 	    final Ticket that = (Ticket)o;
312 
313 	    return that.when == when && that.breaker == breaker;
314 	}
315 
316 	public int hashCode() {
317 	    return (int)breaker;
318 	}
319 
320 	public int compareTo(Ticket o) {
321 	    final Ticket that = o;	    
322 	    
323 	    final long whenDiff = when - that.when;
324 	    if (whenDiff > 0)
325 		return 1;
326 	    else if (whenDiff < 0)
327 		return -1;
328 	    else {
329 		final long breakerDiff = breaker - that.breaker;	
330 
331 		if (breakerDiff > 0)
332 		    return 1;
333 		else if (breakerDiff < 0)
334 		    return -1;
335 		else
336 		    return 0;
337 	    }
338 	}
339     }

366     /**
367      * Create a new <code>WakeupManager</code>.  The thread used for
368      * timing will be created according to the provided <code>ThreadDesc</code>.
369      * Optionally pass a configuration to control various implementation
370      * specific behaviors.
371      * @throws ConfigurationException if if an exception
372      *         occurs while retrieving an item from the given
373      *         <code>Configuration</code> object
374      * @throws NullPointerException if either argument is null
375      */
376     public WakeupManager(ThreadDesc desc, Configuration config)
377         throws ConfigurationException
378     {
379 	if (desc == null)
380 	    throw new NullPointerException("desc must be non-null");
381 
382 	kickerDesc = desc;
383 	queueThreadTimeout = Config.getLongEntry(config, COMPONENT_NAME, 
384 		"queueThreadTimeout", DEFAULT_QUEUE_THREAD_TIMEOUT, 
385 		0, Long.MAX_VALUE);

388     /**
389      * Create a new ticket with the specified values for when the task
390      * should be run, what task should be run, and what sort of
391      * thread the task should be run in. 
392      *
393      * @param when when the task should run, an absolute time
394      * @param task what task should be run
395      * @param threadDesc if non-<code>null</code> the object to use to 
396      *        create the thread the task should be run in, if
397      *        <code>null</code> the task should be run in the
398      *        manager's thread.
399      * @throws NullPointerException if task is <code>null</code>
400      */
401     protected Ticket newTicket(long when, Runnable task, ThreadDesc threadDesc) {
402 	synchronized (contents) {
403 	    return new Ticket(when, task, threadDesc, nextBreaker++);
404 	}
405     }

407     /**
408      * Schedule the given task for the given time.  The task's <code>run</code>
409      * method will be executed synchronously in the queue's own thread, so it
410      * should be brief or it will affect whether future events will be executed
411      * at an appropriate time.
412      * @throws NullPointerException if <code>task</code> is <code>null</code>
413      * @throws IllegalStateException if the manager has been stopped
414      */
415     public Ticket schedule(long when, Runnable task) {
416 	return schedule(when, task, null);
417     }
418 
419     /**
420      * Schedule the given task for the given time, to be run in a thread.
421      * When the time comes, a new thread will be created according to the
422      * <code>ThreadDesc</code> object provided.  If <code>threadDesc</code> is
423      * <code>null</code>, this is equivalent to the other form of
424      * <code>schedule</code>.
425      * @throws NullPointerException if <code>task</code> is <code>null</code>
426      * @throws IllegalStateException if the manager has been stopped
427      */
428     public Ticket schedule(long when, Runnable task, ThreadDesc threadDesc) {
429 	synchronized (contents) {
430 	    if (dead)
431 		throw new IllegalStateException(
432 		    "trying to add task to stopped WakeupManager");
433 
434 	    Ticket t = newTicket(when, task, threadDesc);
435 	    contents.add(t);
436 	    
437 	    if (kickerThread == null) {
438 		logger.log(Level.FINEST, "starting queue's thread");
439 
440 		try {
441 		    final Thread thread = kickerDesc.thread(kicker);
442 		    thread.start();
443 
444 		    // Only set once we know start worked
445 		    kickerThread = thread;
446 		} catch (Throwable tt) {
447 		    try {
448 			logger.log(Level.SEVERE, 
449 				   "queue thread creation exception",tt);
450 		    } catch (Throwable ttt) {
451 			// don't let a problem in logging kill the thread
452 		    }
453 		}
454 	    }
455 
456 	    // need to call checkHead (even if we just (re)created the
457 	    // kickerThread), because that is how head gets set (note,
458 	    // this is ok to call even if thread creation failed)
459 	    checkHead();
460 
461 	    return t;
462 	}
463     }
464 
465     /**
466      * Cancel the given ticket.
467      */
468     public void cancel(Ticket t) {
469 	synchronized (contents) {
470 	    if (dead) return;
471 
472 	    contents.remove(t);
473 	    checkHead();
474 	}
475     }
476 
477     /**
478      * Cancel all tickets.
479      */
480     public void cancelAll() {
481 	synchronized (contents) {
482 	    if (dead) return;
483 
484 	    contents.clear();
485 	    checkHead();
486 	}
487     }

490     /**
491      * Called whenever we change contents to update head
492      * and see if we need to wake up the queue thread.
493      * Assumes the caller holds the lock on contents.
494      */
495     private void checkHead() {
496         synchronized (contents){
497 	final Ticket oldHead = head;
498 
499 	if (contents.isEmpty())
500 	    head = null;
501 	else
502 	    head = (Ticket)contents.first();
503 
504 	if (head == oldHead) return;
505 
506 	// New first event (including possibly no events), run
507 	// needs to wake up and change its sleep time.
508 	contents.notifyAll();
509     }
510     }
511 
512     /**
513      * Return whether the queue is currently empty.
514      */
515     public boolean isEmpty() {
516 	synchronized (contents) {
517 	    return (contents.isEmpty());
518 	}
519     }
520 
521     /**
522      * Stop executing.
523      */
524     public void stop() {
525 	synchronized (contents) {
526 	    contents.clear();
527 	    kickerThread = null;
528 	    head = null;
529 	    dead = true;
530 	    contents.notifyAll();
531 	}
532     }

534     /**
535      * The kicker work.  This is what sleeps until the time of
536      * the next event.
537      */
538     private class Kicker implements Runnable {
539 	public void run() {
540 	    /* Set when contents is empty to control when the kicker will
541 	     * exit. Long.MIN_VALUE used as flag value to indicate 
542 	     * kickerExitTime is invalid
543 	     */
544 	    long kickerExitTime = Long.MIN_VALUE;
545 	    
546 	    while (true) {
547 		final Ticket ticketToRun;
548 	    
549 		synchronized (contents) {
550 		    while (true) {
551 			if (dead)
552 			    return;
553 
554 			final long now = System.currentTimeMillis();
555 			final long timeToNextEvent;
556 		
557 			if (contents.isEmpty()) {			
558 			    if (kickerExitTime == Long.MIN_VALUE) {
559 				kickerExitTime = now + queueThreadTimeout;
560 
561 				if (kickerExitTime < 0) {
562 				    // overflow
563 				    kickerExitTime = Long.MAX_VALUE;
564 				} 
565 			    }
566 
567 			    // Since contents is empty the next event is exit
568 			    timeToNextEvent = kickerExitTime - now;
569 
570 			    if (timeToNextEvent <= 0) {
571 				// been idle long enough, depart
572 
573 				/* $$$ Do this in a finally block for the run?
574 				 * so no mater how this thread ends kickerThread
575 				 * get set to null?
576 				 */
577 				kickerThread = null;	    
578 
579 				logger.log(Level.FINEST,
580 					   "stopping queue's thread");
581 				return;
582 			    }
583 			} else { // contents is non-empty
584 			    kickerExitTime = Long.MIN_VALUE;
585 			    timeToNextEvent = head.when - now;
586 
587 			    if (timeToNextEvent <= 0) { 
588 				// The head's time has come, consume and
589 				// break out of inner loop to run it.
590 
591 				ticketToRun = head;
592 				contents.remove(head);
593 				checkHead();
594 				break;
595 			    }
596 			}
597 
598 			if (logger.isLoggable(Level.FINEST)) {
599 			    logger.log(Level.FINEST, "timeToNextEvent:{0}",
600 				   (timeToNextEvent == Long.MAX_VALUE ?
601 				    "Long.MAX_VALUE" : 
602 				    Long.toString(timeToNextEvent)));
603 			}
604 			
605 			assert timeToNextEvent > 0;
606 
607 			try {
608 			    contents.wait(timeToNextEvent);
609 			} catch (InterruptedException e) {
610 			    /* This should never happen, our thread is
611 			     * private to WakeupManager and tasks
612 			     * calling Thread.currentThread().interrupt() is
613 			     * decidedly anti-social. Log, but keep on
614 			     * going.
615 			     */
616 
617 			    try {				
618 				logger.log(Level.WARNING, 
619 					   "Attempt to interrupt Queue's thread");
620 			    } catch (Throwable t) {
621 				// ignore
622 			    }
623 
624 			    /* This loop already deals with wait returning
625 			     * early for no reason, so going to the top
626 			     * of the loop is ok here - if there are no
627 			     * new tasks and we are not dead we will 
628 			     * just calc a new value for timeToNextEvent
629 			     */
630 			}
631 
632 			/* Something has changed or the time has arived
633 			 * for action, don't know which, go back to the 
634 			 * the top of the inner loop to figure out what to
635 			 * do next
636 			 */
637 		    }
638 		}
639 
640 		// Run the task outside of the lock
641 		if (ticketToRun.desc == null) {
642 		    // ... in this thread
643 		    try {
644 			ticketToRun.task.run();
645 		    } catch (Throwable e) {
646                         if (e instanceof Error) throw (Error) e;
647 			try {
648 			    logger.log(Level.WARNING, "Runnable.run exception", e);
649 			} catch (Throwable t) {
650 			    // don't let a problem in logging kill the thread
651 			}
652 		    }
653 		} else {
654 		    // ... in its own thread
655 		    try {
656 			ticketToRun.desc.thread(ticketToRun.task).start();
657 		    } catch (Throwable t) {
658                         t.printStackTrace(System.out);
659                         if (t instanceof Error) throw (Error) t;
660 //                        if (t instanceof RuntimeException) throw (RuntimeException) t;
661 			try {
662 			    logger.log(Level.WARNING, 
663 				       "task thread creation exception", t);
664 			} catch (Throwable tt) {
665 			    // don't let a problem in logging kill the thread
666 			}			
667 		    }
668 		}
669 	    }
670 	}
671     }
