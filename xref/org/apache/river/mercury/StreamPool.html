23  import java.io.EOFException;
24  import java.io.File;
25  import java.io.IOException;
26  import java.io.PrintWriter;
27  import java.io.RandomAccessFile;
28  import java.io.SyncFailedException;
29  import java.util.HashMap;
30  import java.util.LinkedList;
31  import java.util.logging.Level;
32  import java.util.logging.Logger;

34  /** 
35   * This class provides a pool of <tt>LogStream</tt> objects.  Each
36   * <tt>LogStream</tt> has an associated <tt>FileDescriptor</tt>, which
37   * is the system resource we are trying to manage. This pool limits the
38   * (user configurable) number of concurrent, open <tt>FileDescriptor</tt>s.
39   *
40   * @author Sun Microsystems, Inc.
41   *
42   * @since 1.1
43   */
44  class StreamPool {
45      // Class fields
46      /** Logger for lease related messages */
47      private static final Logger persistenceLogger = 
48          MailboxImpl.PERSISTENCE_LOGGER;
49      
50      /** 
51       * Maximum limit for the number of concurrent <tt>LogStream</tt>s
52       * in the stream pool.
53       */
54      private final int maxPoolSize;
55  
56      /** Holds stream references by associated key */
57      private final HashMap pool;
58  
59      /**
60       * Holds stream references in least recently used (released) order.
61       * It's used in order determine which stream to discard upon
62       * reaching the maximum pool limit.
63       */
64      private final LinkedList freeList;
65  
66      /**
67       * Simple constructor that creates a pool of given <code>size</code>.
68       *
69       * @exception IllegalArgumentException Thrown if the value of 
70       *                <tt>maxPoolSize</tt> is less than 1.
71       */
72      StreamPool(int size) {
73          if (size < 1) 
74              throw new IllegalArgumentException(
75  		"Pool size must be greater than 0."); 
76          maxPoolSize = size;
77          pool = new HashMap(maxPoolSize);
78          freeList = new LinkedList();
79  
80          if (persistenceLogger.isLoggable(Level.FINEST)) {
81              persistenceLogger.log(Level.FINEST,
82  	        "Created StreamPool of size {0}", 
83  		Integer.valueOf(maxPoolSize));
84  	}
85      }
86  
87      /**
88       * Returns a <tt>ControlLog</tt> object for the specified <tt>file</tt>
89       * from the pool if it already exists.
90       * Otherwise, it creates a new instance and adds it to the pool.
91       *
92       * @exception IOException if an I/O error occurs
93       */
94      synchronized ControlLog getControlLog(File file) 
95  	throws IOException 
96      {
97  	StreamKey key = new StreamKey(file, StreamType.CONTROL);
98  	ControlLog log = (ControlLog)pool.get(key);
99  	if (log != null) { // found it!
100 	    if (freeList.remove(key) == false)
101 	        throw new InternalMailboxException("Did not find re-used control log "
102 	            + "stream in freeList.");
103 	} else { // Log was not found, so attempt to add it
104 
105 	    ensurePoolSpace();
106 
107             //
108 	    // Create new ControlLog and add it the pool.
109             //
110             log = new ControlLog(file, key);
111             pool.put(key, log);
112 
113             if(freeList.remove(key))
114                 throw new InternalMailboxException("Found newly created ControlLog "
115                     + "in freeList");
116 	}
117 
118         return log;
119     }
120 
121     /**
122      * Returns a <tt>LogInputStream</tt> object from the pool if it 
123      * already exists. Otherwise, it creates a new instance and adds 
124      * it to the pool.
125      *
126      * @exception IOException if an I/O error occurs
127      */
128     synchronized LogInputStream getLogInputStream(File file, long offset) 
129 	throws IOException
130     {
131 	StreamKey key = new StreamKey(file, StreamType.INPUT);
132 	LogInputStream in = (LogInputStream)pool.get(key);
133 	if (in != null) { //found it!
134 	    if (freeList.remove(key) == false)
135 	        throw new InternalMailboxException("Did not find re-used input log "
136 	            + "stream in freelist.");
137 	} 
138 
139 	if (in == null ||            // if log not found OR 
140 	    in.getOffset() > offset) // current read offset is past desired 
141 	{                            // then create a new log and add it
142 	    ensurePoolSpace();
143 
144 	    in = new LogInputStream(file, key);
145 	    pool.put(key, in);
146 
147             if(freeList.remove(key))
148                 throw new InternalMailboxException("Found newly created ControlLog "
149                     + "on freeList");
150 	}
151 
152 	// Sanity check for offset value
153 	if (offset > file.length()) 
154 	    throw new EOFException("Attempting to read past end of file.");
155 
156 	// Check if log offset needs adjusting. 
157 	// By this point in.offset <= offset.
158 	while (in.getOffset() < offset) {
159 	    in.skip(offset - in.getOffset());
160 	}
161 
162 	return in;
163     }
164 
165     /**
166      * Returns a <tt>LogOutputStream</tt> object for the specified <tt>file</tt>
167      * from the pool if it already exists. 
168      * Otherwise, it creates a new instance and adds it to the pool.
169      *
170      * @exception IOException if an I/O error occurs
171      */
172     synchronized LogOutputStream getLogOutputStream(File file, long offset) 
173 	throws IOException 
174     {
175 	StreamKey key = new StreamKey(file, StreamType.OUTPUT);
176 	LogOutputStream out = (LogOutputStream)pool.get(key);
177 
178 	if (out != null) { // found it!
179 	    if (freeList.remove(key) == false)
180 	        throw new InternalMailboxException("Did not find re-used output log "
181 	            + "stream in freelist");
182 
183 	     // Sanity check to see if we are still in sync. If not,
184 	     // then we need to close this stream and create another
185 	     // one (done in the next code block).
186 	    if (out.getOffset() != offset) {
187 	        removeLogStream(out);
188 	        out = null;
189 	    }
190 	}
191 
192         // Check to see if we need to create another log.
193 	if (out == null) { 
194 
195 	    ensurePoolSpace();
196 
197 	    if (offset == 0L) { // Create new log, without appending
198 	        out = new LogOutputStream(file, key, false);
199 	    } else { // Create new log, appending to existing file
200 	        long len = file.length();
201 
202 	        if (offset > len) 
203 	            throw new EOFException("Attempting to write past end "
204 	                + "of file");
205 
206 		if (offset < len) {
207 		    RandomAccessFile raf = new RandomAccessFile(file, "rw");
208 		    raf.setLength(offset);
209 		    raf.close();
210 		}
211 
212 		out = new LogOutputStream(file, key, true);
213 	    }
214 
215             pool.put(key, out);
216 
217             if (freeList.remove(key))
218                 throw new InternalMailboxException("Found newly created output log "
219                     + "in freeList");
220 
221 	}
222 
223         return out;
224     }
225 
226     /**
227      * Ensures that room is available in the pool. If the pool is currently
228      * full, then the least recently used <tt>LogStream</tt> will be removed 
229      * and closed to make room. This method will block if the pool is full and
230      * no <tt>LogStream</tt> objects can be closed. 
231      *
232      * @exception IOException if an I/O error occurs
233      */
234     private synchronized void ensurePoolSpace() throws IOException {
235         if (pool.size() >= maxPoolSize) {
236             while(freeList.size() < 1) {
237                 try {
238 		    wait();
239 		} catch (InterruptedException ie) { ; }
240 	    }
241 	    StreamKey key = (StreamKey)freeList.removeFirst();
242 	    LogStream els = (LogStream)pool.remove(key);
243 	    els.close();
244 	}
245     }
246 
247     /**
248      * Marks a stream as available for closing.
249      * A log will only be closed if a new log is requested and the
250      * pool has reached its maximum size.
251      */
252     synchronized void releaseLogStream(LogStream stream) {
253         StreamKey key = (StreamKey)stream.getKey();
254         if (pool.get(key) == null)
255             throw new InternalMailboxException("Not managing stream: " 
256                 + stream + ":" + key + " -- release failed");
257 	freeList.add(key);
258 	notifyAll();
259     }
260 
261     /**
262      * Removes the given <tt>LogStream</tt> from the pool and closes it,
263      * if possible. The intent is for this method to be called for 
264      * unusable logs so that they will no longer be returned by a 
265      * subsequent call to one of the "get" methods.
266      */
267     synchronized void removeLogStream(LogStream stream) {
268         StreamKey key = (StreamKey)stream.getKey();
269         if (pool.remove(key) == null)
270             throw new InternalMailboxException("Not managing stream: " 
271                 + stream + ":" + key + " -- remove failed");
272 
273         // Remove it from freeList, if present
274         freeList.remove(key);
275         try {
276             stream.close();
277 	} catch (IOException ioe) {
278 	    // Note the exception, but otherwise ignore
279 	    if (persistenceLogger.isLoggable(Levels.HANDLED)) {
280                 persistenceLogger.log(Levels.HANDLED,
281 		    "Exception closing Log", ioe);
282 	    }
283 	}
284     }
285 
286     //
287     // Debug use only!
288     //
289     synchronized int getPoolSize() { return pool.size(); }
290     synchronized int getFreeSize() { return freeList.size(); }
291     synchronized void dump() { 
292 	System.out.println("Pool:\n" + pool); 
293 	System.out.println("Free:\n" + freeList); 
294     }
295 }
