18  package org.apache.river.mercury;
19  
20  /**
21   * Class that serves the purpose of an enumeration type. It's used 
22   * in conjunction with <tt>StreamKey</tt> in order to distinguish among 
23   * different <tt>LogStream</tt> objects that can potentially act upon the
24   * same underlying log file.
25   *
26   * @author Sun Microsystems, Inc.
27   *
28   * @since 1.1
29   */
30  class StreamType {
31  
32      /**
33       * Simple constructor. Private protection is used in order to prevent
34       * external instantiation of this class type.
35       */
36      private StreamType() {;}
37  
38      /** Event data output stream enumeration */
39      static final StreamType OUTPUT = new StreamType();
40  
41      /** Event data input stream enumeration */
42      static final StreamType INPUT = new StreamType();
43  
44      /** 
45       * Control data stream enumeration. Control data will not use
46       * separate input/output data streams and therefore no distinction
47       * is made.
48       */
49      static final StreamType CONTROL = new StreamType();
50  }
