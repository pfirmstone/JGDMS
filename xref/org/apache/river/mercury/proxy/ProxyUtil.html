18  package org.apache.river.mercury.proxy;
19  
20  import java.lang.reflect.Method;
21  
22  /**
23   * Convenience class that contains a package protected static utility method
24   * used by the proxy classes of this package in proxy trust verification
25   * process.
26   *
27   * Note that this class cannot be instantiated.
28   *
29   * @author Sun Microsystems, Inc.
30   */
31  class ProxyUtil {
32  
33      /** This class cannot be instantiated. */
34      private ProxyUtil() {
35  	throw new AssertionError("class cannot be instantiated");
36      }//end constructor
37  
38      /**
39       * Returns the public method for the specified <code>Class</code> type,
40       * method name, and array of parameter types.
41       * <p>
42       * This method is typically used in place of {@link Class#getMethod
43       * Class.getMethod} to get a method that should definitely be defined;
44       * thus, this method throws an error instead of an exception if the
45       * given method is missing.
46       * <p>
47       * This method is convenient for the initialization of a static
48       * variable for use as the <code>mappings</code> argument to 
49       * {@link org.apache.river.proxy.ConstrainableProxyUtil#translateConstraints 
50       * ConstrainableProxyUtil.translateConstraints}.
51       *
52       * @param type           the <code>Class</code> type that defines the
53       *                       method of interest
54       * @param name           <code>String</code> containing the name of the
55       *                       method of interest
56       * @param parameterTypes the <code>Class</code> types of the parameters
57       *                       to the method of interest
58       *
59       * @return a <code>Method</code> object that provides information about,
60       *         and access to, the method of interest
61       *
62       * @throws <code>NoSuchMethodError</code> if the method of interest cannot
63       *         be found
64       * @throws <code>NullPointerException</code> if <code>type</code> or
65       *         <code>name</code> is <code>null</code> 
66       */
67      static Method getMethod(Class type,
68  				   String name,
69  				   Class[] parameterTypes)
70      {
71          try {
72              return type.getMethod(name, parameterTypes);
73          } catch (NoSuchMethodException e) {
74              throw (Error)(new NoSuchMethodError(e.getMessage()).initCause(e));
75          }
76      }//end getMethod
77  }//end class ProxyUtil
