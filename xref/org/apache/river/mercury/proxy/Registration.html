47  /**
48   * The <tt>Registration</tt> class is the client-side proxy
49   * returned to event mailbox clients as the result of the 
50   * registration process. It implements the <tt>MailboxRegistration</tt>
51   * interface and delegates functionality to the mailbox service
52   * where necessary.

54   * @author Sun Microsystems, Inc.

56   * @since 1.1

58  @AtomicSerial
59  public class Registration implements MailboxPullRegistration, 
60      Serializable, ReferentUuid, ProxyAccessor

63      private static final long serialVersionUID = 2L;

65      /** Unique identifier for this registration */
66      final Uuid registrationID;
67  
68      /** Reference to service implementation */
69      final MailboxBackEnd mailbox;
70  
71      /** Reference to service provided RemoteEventListener implementation */
72      final ListenerProxy listener;
73  
74      /** The service's registration lease */ 
75      final Lease lease;

77      /**
78       * Creates a mailbox registration proxy, returning an instance 
79       * that implements RemoteMethodControl if the server does too.
80       *
81       * @param server the server proxy
82       * @param id the ID of the lease set
83       * @param lease the lease set's lease
84       */
85      public static Registration create(Uuid id, MailboxBackEnd server, Lease lease) {
86  	if (id == null || server == null || lease == null)
87              throw new IllegalArgumentException("Cannot accept null arguments");
88  	if (server instanceof RemoteMethodControl) {
89  	    return new ConstrainableRegistration(id, server, lease, null);
90  	} else {
91  	    return new Registration(id, server, ListenerProxy.create(id, server), lease);
92  	}
93      }
94  
95      /** Convenience constructor */
96      private Registration(Uuid id, MailboxBackEnd srv, ListenerProxy proxy, Lease l) {
97          registrationID = id;
98          mailbox = srv;
99          listener = proxy;
100         lease = l;
101     }

103     Registration(GetArg arg) throws IOException {

106 		(ListenerProxy) arg.get("listener", null),
107 		(Lease) arg.get("lease", null)
108 		);
109     }

111     private static Uuid check(GetArg arg) throws IOException {
112 	Uuid registrationID = (Uuid) arg.get("registrationID", null);
113 	MailboxBackEnd mailbox = (MailboxBackEnd) arg.get("mailbox", null);
114 	ListenerProxy listener = (ListenerProxy) arg.get("listener", null);
115 	Lease lease = (Lease) arg.get("lease", null);
116 	/* Verify server */
117         if(mailbox == null) {
118             throw new InvalidObjectException("Registration.readObject "
119                                              +"failure - mailbox "
120                                              +"field is null");
121         }//endif
122         /* Verify registrationID */
123         if(registrationID == null) {
124             throw new InvalidObjectException
125                                   ("Registration.readObject "
126                                    +"failure - registrationID field is null");
127         }//endif
128         /* Verify regLease */
129         if(lease == null) {
130             throw new InvalidObjectException
131                                         ("Registration.readObject "
132                                          +"failure - lease field is null");
133         }//endif
134         /* Verify listener */
135         if(listener == null) {
136             throw new InvalidObjectException
137                                         ("Registration.readObject "
138                                          +"failure - listener field is null");
139         }//endif
140 	return registrationID;
141     }

153     // inherit javadoc from supertype
154     public void enableDelivery(RemoteEventListener target) 
155 	throws RemoteException
156     {
157         // Prevent resubmission of this registration's listener
158 	if ((target instanceof ListenerProxy) &&
159 	    (listener.equals((ListenerProxy)target))) {
160 	    throw new IllegalArgumentException("Cannot resubmit " +
161 		"a target that was provided by the EventMailbox service");
162 	} else { // OK to make the call, now
163 	    try {
164 	        mailbox.enableDelivery(registrationID, target);
165 	    } catch (ThrowThis tt) { 
166 	        tt.throwRemoteException();
167 	    }
168 	}
169     }

233    /** When an instance of this class is deserialized, this method is

240     private void readObject(ObjectInputStream s)

242     {
243         s.defaultReadObject();

246             throw new InvalidObjectException("Registration.readObject "
247                                              +"failure - mailbox "
248                                              +"field is null");
249         }//endif
250         /* Verify registrationID */
251         if(registrationID == null) {
252             throw new InvalidObjectException
253                                   ("Registration.readObject "
254                                    +"failure - registrationID field is null");
255         }//endif
256         /* Verify regLease */
257         if(lease == null) {
258             throw new InvalidObjectException
259                                         ("Registration.readObject "
260                                          +"failure - lease field is null");
261         }//endif
262         /* Verify listener */
263         if(listener == null) {
264             throw new InvalidObjectException
265                                         ("Registration.readObject "
266                                          +"failure - listener field is null");
267         }//endif
268     }//end readObject

289     /** A subclass of Registration that implements RemoteMethodControl. */
290     @AtomicSerial
291     final static class ConstrainableRegistration extends Registration 
292         implements RemoteMethodControl 
293     {
294 	private static final long serialVersionUID = 1L;
295 
296 	// Mappings from client to server methods, 
297 	private static final Method[] methodMap1 = {
298 	    ProxyUtil.getMethod(MailboxPullRegistration.class,
299 	        "getRemoteEvents", new Class[] {}),
300             ProxyUtil.getMethod(MailboxBackEnd.class, 
301 	        "getRemoteEvents", new Class[] {Uuid.class}), 
302             // Use the same constraints for getNextBatch as getRemoteEvents
303 	    ProxyUtil.getMethod(MailboxPullRegistration.class,
304 	        "getRemoteEvents", new Class[] {}),
305             ProxyUtil.getMethod(MailboxBackEnd.class, 
306 	        "getNextBatch", new Class[] {
307 		    Uuid.class, Uuid.class, long.class, Object.class}), 
308 	    ProxyUtil.getMethod(MailboxPullRegistration.class,
309 	        "addUnknownEvents", new Class[] {Collection.class}),
310             ProxyUtil.getMethod(MailboxBackEnd.class, 
311 	        "addUnknownEvents", new Class[] {Uuid.class, Collection.class}), 
312 	    ProxyUtil.getMethod(MailboxRegistration.class,
313 	        "enableDelivery", new Class[] {RemoteEventListener.class}),
314 	    ProxyUtil.getMethod(MailboxBackEnd.class, 
315 	        "enableDelivery", new Class[] {Uuid.class, 
316 		RemoteEventListener.class}), 
317 	    ProxyUtil.getMethod(MailboxRegistration.class,
318 	        "disableDelivery", new Class[] {}),
319 	    ProxyUtil.getMethod(MailboxBackEnd.class, 
320 	        "disableDelivery", new Class[] {Uuid.class})
321 	};
322 	/**
323 	 * The client constraints placed on this proxy or <code>null</code>.
324 	 *
325 	 * @serial
326 	 */
327 	private MethodConstraints methodConstraints;
328 
329 	/** Creates an instance of this class. */
330 	private ConstrainableRegistration(Uuid id, MailboxBackEnd server,
331             Lease lease, MethodConstraints methodConstraints)
332 	{
333 	    super(id, 
334 		  constrainServer(server, methodConstraints),
335 		  ListenerProxy.create(id, server),
336 	          lease);
337 	    this.methodConstraints = methodConstraints;
338 	}
339 	
340 	ConstrainableRegistration(GetArg arg) throws IOException {
341 	    super(check(arg));
342 	    methodConstraints = (MethodConstraints) 
343 		    arg.get("methodConstraints", null);
344 	}
345 	
346 	private static GetArg check(GetArg arg) throws IOException {
347 	    Registration r = new Registration(arg);
348 	    MethodConstraints methodConstraints = (MethodConstraints) 
349 		    arg.get("methodConstraints", null);
350 	    /* Verify the server and its constraints */
351             ConstrainableProxyUtil.verifyConsistentConstraints(methodConstraints,
352                                                         r.mailbox,
353                                                         methodMap1);
354             if( !(r.lease instanceof ConstrainableLandlordLease) ) {
355                 throw new InvalidObjectException
356                                 ("Registration.readObject failure - "
357                                  +"lease is not an instance of "
358                                  +"ConstrainableLandlordLease");
359             }//endif
360 
361             if( !(r.listener instanceof ListenerProxy.ConstrainableListenerProxy) ) {
362                 throw new InvalidObjectException
363                                 ("Registration.readObject failure - "
364                                  +"listener is not an instance of "
365                                  +"ListenerProxy.ConstrainableListenerProxy");
366             }//endif
367 
368             /* Verify listener's ID */
369             if(!r.registrationID.equals(
370 	       ((ListenerProxy.ConstrainableListenerProxy)r.listener).registrationID)) 
371             {
372                 throw new InvalidObjectException
373                                         ("Registration.readObject "
374                                          +"failure - listener ID "
375                                          +"is not equal to "
376                                          +"proxy ID");
377             }   
378 	    return arg;
379 	}

391 	/**
392 	 * Returns a copy of the server proxy with the specified client
393 	 * constraints and methods mapping.
394 	 */
395 	private static MailboxBackEnd constrainServer(
396 	    MailboxBackEnd server,
397 	    MethodConstraints methodConstraints)
398 	{
399 	    return (MailboxBackEnd) 
400 	        ((RemoteMethodControl)server).setConstraints(
401 		    ConstrainableProxyUtil.translateConstraints(
402 		        methodConstraints, methodMap1));
403 	}
404 	/** {@inheritDoc} */
405 	public RemoteMethodControl setConstraints(
406 	    MethodConstraints constraints)
407 	{
408 	    return new ConstrainableRegistration(registrationID, mailbox, lease, 
409 	        constraints);
410 	}

412 	/** {@inheritDoc} */

430 	/**
431 	 * Verifies that the registrationID, lease and mailbox fields are 
432 	 * not null, that mailbox implements RemoteMethodControl, and that the 
433 	 * mailbox proxy has the appropriate method constraints.
434 	 *
435 	 * @throws InvalidObjectException if registrationID, lease or mailbox
436 	 *         is null, if mailbox does not implement RemoteMethodControl, 
437 	 *         or if server has the wrong constraints
438 	 */
439 	private void readObject(ObjectInputStream s)
440 	    throws IOException, ClassNotFoundException
441 	{
442             /* Note that basic validation of the fields of this class was
443              * already performed in the readObject() method of this class'
444              * super class.
445              */
446             s.defaultReadObject();
447             /* Verify the server and its constraints */
448             ConstrainableProxyUtil.verifyConsistentConstraints(methodConstraints,
449                                                         mailbox,
450                                                         methodMap1);
451             if( !(lease instanceof ConstrainableLandlordLease) ) {
452                 throw new InvalidObjectException
453                                 ("Registration.readObject failure - "
454                                  +"lease is not an instance of "
455                                  +"ConstrainableLandlordLease");
456             }//endif
457 
458             if( !(listener instanceof ListenerProxy.ConstrainableListenerProxy) ) {
459                 throw new InvalidObjectException
460                                 ("Registration.readObject failure - "
461                                  +"listener is not an instance of "
462                                  +"ListenerProxy.ConstrainableListenerProxy");
463             }//endif
464 
465             /* Verify listener's ID */
466             if(!registrationID.equals(
467 	       ((ListenerProxy.ConstrainableListenerProxy)listener).registrationID)) 
468             {
469                 throw new InvalidObjectException
470                                         ("Registration.readObject "
471                                          +"failure - listener ID "
472                                          +"is not equal to "
473                                          +"proxy ID");
474             }            
475 	}
