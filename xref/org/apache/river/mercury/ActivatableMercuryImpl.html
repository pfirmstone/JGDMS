18  package org.apache.river.mercury;
19  
20  import java.rmi.activation.ActivationID;
21  import java.rmi.MarshalledObject;
22  import net.jini.export.DynamicProxyCodebaseAccessor;
23  
24  /**
25   * Convenience class intended for use with the 
26   * {@link org.apache.river.start.ServiceStarter} framework to start
27   * an implementation of Mercury that is activatable, and which will
28   * log its state information to persistent storage.
29   *
30   * @author Sun Microsystems, Inc.
31   * @since 2.0
32   */
33  class ActivatableMercuryImpl extends MailboxImpl  
34  			     implements DynamicProxyCodebaseAccessor {
35  
36      /**
37       * Constructs a new instance of <code>MailboxImpl</code> that is 
38       * activatable, and which will persist its state.
39       * <p>
40       * A constructor having this signature is required for the class to
41       * be activatable. This constructor is automatically called by the 
42       * activation group when the service is activated.
43       *
44       * @param activationID the activation ID generated by the activation
45       *                     system and assigned to the instance of the server
46       *                     being activated
47       * @param data         state data (represented as a 
48       *                     <code>MarshalledObject</code>) which is needed to
49       *                     re-activate this server
50       *
51       * @throws Exception If there was a problem initializing the service.
52       */
53      ActivatableMercuryImpl(ActivationID activationID, MarshalledObject data)
54          throws Exception
55      {
56          super(activationID, data);
57      }//end constructor
58  
59  }//end class ActivatableMercuryImpl
