18  package org.apache.river.landlord;
19  
20  import net.jini.core.lease.Lease;
21  import net.jini.core.lease.LeaseDeniedException;
22  
23  /**
24   * Simple implementation of <code>LeasePeriodPolicy</code> that grants
25   * lease times based on a fixed default and maximum lease. Will grant
26   * renewals longer than the maximum if the current lease and the
27   * request are both longer than the maximum.
28   *
29   * @author Sun Microsystems, Inc.
30   * @since 2.0
31   */
32  public class FixedLeasePeriodPolicy implements LeasePeriodPolicy {
33      /** Under normal circumstances, the maximum lease or renewal to grant */
34      final private long maximum;
35  
36      /** The default lease length */
37      final private long defaultLength;
38  
39      /**
40       * Create a new <code>FixedLeasePeriodPolicy</code> with 
41       * the specified values for the maxium and default lease
42       * lengths.
43       * @param maximum the length in milliseconds of the longest lease
44       *                this object should normally grant
45       * @param defaultLength the length in milliseconds of the
46       *                default lease grants
47       * @throws IllegalArgumentException if either argument
48       *         is not positive.
49       */
50      public FixedLeasePeriodPolicy(long maximum, long defaultLength) {
51  	if (maximum <= 0)
52  	    throw new IllegalArgumentException("FixedLeasePeriodPolicy:" +
53  	        "maximum lease time must be larger than 0, " +
54  		 "passed:" + maximum);
55  
56  	if (defaultLength <= 0)
57  	    throw new IllegalArgumentException("FixedLeasePeriodPolicy:" +
58  	        "default lease time must be larger than 0, " + 
59  		 "passed:" + defaultLength);
60  
61  	this.maximum = maximum;
62  	this.defaultLength = defaultLength;
63      }
64  
65  
66      /**
67       * Returns the duration this policy is willing to grant for the
68       * passed resource at this time.  The duration actually granted
69       * will be shorter if the duration extends pass the end of the
70       * epoch.  Must return a positive number.
71       * <p>
72       * Note the duration returned by this method will be shorter than
73       * the final duration granted if the requested duration extends
74       * past the current expiration, and duration return by this 
75       * method is before the current expiration.
76       *     
77       * @param resource          the resource having a lease granted
78       *                          or renewed
79       * @param requestedDuration the duration the client wants 
80       * @return  duration in milliseconds.
81       * @exception IllegalArgumentException thrown if requestedDuration
82       *            is less than 0 and not equal to
83       *            <code>Lease.ANYLENGTH</code> or
84       *            <code>Lease.FOREVER</code>.  
85       */
86      protected long calculateDuration(LeasedResource resource, 
87          long requestedDuration) 
88      {
89  	if (requestedDuration == Lease.FOREVER)
90  	    requestedDuration = Long.MAX_VALUE;
91  	else if (requestedDuration == Lease.ANY)
92  	    requestedDuration = defaultLength;
93  	else if (requestedDuration < 0)
94  	    throw new 
95  		IllegalArgumentException("Negative lease duration " +
96  		    "requested");
97  
98  	return Math.min(requestedDuration, maximum);
99      }
100 
101     /**
102      * Method that provides some notion of the current time in milliseconds
103      * since the beginning of the epoch. Default implementation
104      * calls System.currentTimeMillis()
105      * @return current time in milliseconds.
106      */
107     protected long currentTime() {
108 	return System.currentTimeMillis();
109     }
110 
111     /**
112      * Calculates an expiration based on the passed time and a requested
113      * duration. Will clip the expiration to the end of the epoch
114      * @param preferredDuration the duration the policy wants to grant
115      * @param now               the current time in milliseconds since
116      *                          the beginning of the epoch
117      * @return the new expiration time for the lease in milliseconds since 
118      *         the beginning of the epoch
119      */
120     private long calcExpiration(long preferredDuration, long now) {
121 	long expiration = now + preferredDuration;
122 	// Any addition of two positive longs is guaranteed to be
123 	// negative if it overflowed
124 	if (expiration < 0) {
125 	    // Not enough time for the duration we want, set to end of
126 	    // epoch 
127 	    return Long.MAX_VALUE;
128 	}
129 
130 	return expiration;
131     }
132 
133 
134     public Result grant(LeasedResource resource, long requestedDuration) 
135         throws LeaseDeniedException
136     {
137 	final long now = currentTime();
138 	final long expiration =
139 	    calcExpiration(calculateDuration(resource, requestedDuration), 
140 			   now);
141 	final long duration = expiration - now;
142 	return new Result(expiration, duration);
143     }
144 
145     public Result renew(LeasedResource resource, long requestedDuration)
146         throws LeaseDeniedException
147     {
148 	final long now = currentTime();
149 
150 	// The new expiration we would grant independent of the
151 	// current expiration time 
152 	long newExpiration =
153 	    calcExpiration(calculateDuration(resource, requestedDuration), 
154 			   now);	
155 
156 	// Do not change the expiration time to be sooner that it is
157 	// now unless the client explicitly asked for it.  If they did
158 	// ask for a sooner expiration give them exactly what they
159 	// asked for.
160 
161 	// The current expiration of the lease
162 	final long oldExpiration = resource.getExpiration();
163 
164 	if (oldExpiration > newExpiration) {
165 	    // need to make sure they asked for it
166 
167 	    if (requestedDuration == Lease.ANY) {
168 		// they did not explicitly ask for a sooner
169 		// expiration, give them at least what they had
170 		newExpiration = oldExpiration; 
171 	    } else {
172 		// The new expiration time they are requesting
173 		// (possibly clipped to the end of the epoch) this is
174 		// not affected by what we are willing to grant.
175 		// [Note we know at this point that requestedDuration
176 		// is >= 0]
177 		final long requestedExpiration = calcExpiration(
178 		    requestedDuration, now);
179 		
180 		if (requestedExpiration > oldExpiration) {
181 		    // they made a request for something longer, give
182 		    // them at least what they had
183 		    newExpiration = oldExpiration; 
184 		} else {
185 		    // They made a request for something shorter,
186 		    // give them what they at least what they want
187 		    newExpiration = requestedExpiration;
188 		}
189 	    }
190 	}
191 
192 	// Now that we finally have decided on a new expiration,
193 	// get the real duration and return the result.
194 	return new Result(newExpiration, newExpiration - now);
195     }
196 }
