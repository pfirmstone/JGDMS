31  /**
32   * This class parses a command line that uses multi-character options,
33   * such as <code>-verbose</code> or <code>-help</code>.
34   * <p>
35   * To use <code>MultiCommandLine</code>, create a <code>MultiCommandLine</code>
36   * object with the array of strings you wish to parse (typically the
37   * array passed to the utility's <code>main</code> method), and then
38   * consume options from it, providing default values in case the option
39   * is not specified by the user.  When you have consumed all the
40   * options, you invoke the <code>MultiCommandLine</code> object's
41   * <code>getOperands</code> method to return the remaining operands on
42   * the command line.  If ``<code>--</code>'' is specified it is neither an
43   * option nor an operand, just a separator between the two lists.  The
44   * <code>CommandLine.BadInvocationException</code> is used to signal
45   * errors in the construction of the strings, that is, a user error,
46   * such as specifying a option that takes an argument but forgetting to
47   * provide that argument.  See the documentation for
48   * <code>MultiCommandLine.main</code> for an example.
49   * <p>
50   * You must call <code>getOperands</code> for proper behavior, even if
51   * you do not use any operands in your command.  <code>getOperands</code> 
52   * checks for several user errors, including unknown options.  If you
53   * do not expect to use operands, you should check the return value of
54   * <code>getOperands</code> and complain if any are specified.
55   * <p>
56   * No options
57   * can be consumed after <code>getOperands</code> is invoked.  Each
58   * option may be used only once.  Failure to follow these
59   * rule is a programmer error that will result in a
60   * <code>CommandLine.ProgrammingException</code>.
61   * <p>
62   * <code>MultiCommandLine</code> provides you several methods to get input
63   * streams from the command line.  If these do not suffice for your
64   * particular needs, you can get the argument as a <code>String</code>
65   * and do your own processing.
66   *
67   * @author Sun Microsystems, Inc.
68   *
69   * @see java.util.StringTokenizer
70   */
71  public class MultiCommandLine extends CommandLine {

73      private String[] args;

134     /**
135      * Return <code>true</code> if the given option is specified on the
136      * command line.
137      */
138     public synchronized boolean getBoolean(String opt) {
139 	addOpt(opt, null);
140 	boolean retval = false;
141 	while (findOpt(opt))
142 	    retval = true;
143 	return retval;
144     }

146     /**
147      * Return the argument for the given option.  This is a workhorse
148      * routine shared by all the methods that get options with arguments.
149      */
150     private String getArgument(String opt)
151 	throws BadInvocationException
152     {
153 	if (findOpt(opt))
154 	    return optArg();
155 	return null;
156     }

371     /**
372      * Find the given option somewhere in the command line.  If the
373      * option is not found, return <code>false</code>.  Otherwise set
374      * <code>str</code>, <code>pos</code>, and <code>opt</code> fields,
375      * mark the option character as <code>USED</code>, and then return
376        <code>true</code>.
377      */
378     private boolean findOpt(String opt) {
379 	if (allUsed)
380 	    return false;
381 	boolean seenUnused = false;
382 	for (int i = 0; i < args.length; i++) {
383 	    if (used.get(i))			// already consumed
384 		continue;
385 	    if (!args[i].startsWith("-"))	// not an option
386 		continue;
387 	    if (args[i].equals("--"))		// "--" ends the list
388 		break;
389 	    seenUnused = true;
390 	    if (args[i].length() - 1 == opt.length() &&
391 	        args[i].regionMatches(1, opt, 0, opt.length()))
392 	    {
393 		str = i;
394 		this.opt = opt;
395 		used.set(i);
396 		return true;
397 	    }
398 	}
399 	if (!seenUnused)
400 	    allUsed = true;
401 	return false;
402     }

404     /**
405      * Return the current option's argument, marking its characters
406      * as <code>USED</code>.
407      *
408      * @exception BadInvocationException No argument is given.
409      */
410     private String optArg() throws BadInvocationException {
411 	if (str >= args.length)
412 	    throw new BadInvocationException(opt);
413 	used.set(str + 1);
414 	return args[str + 1];
415     }

417     /**
418      * Return the command line operands that come after the options.
419      * This checks to make sure that all specified options have been
420      * consumed -- any options remaining at this point are assumed to
421      * be unknown options.  If no operands remain, an empty array is
422      * returned.
423      * <p>
424      * This is also where <code>-?</code> is handled.  If the user
425      * specifies <code>-?</code> then the method <code>usage</code> is
426      * invoked and <code>HelpOnlyException</code> is thrown.  The
427      * program is expected to catch this exception and simply exit
428      * successfully.
429      *
430      * @see #usage
431      */
432     public String[] getOperands()
433 	throws BadInvocationException, HelpOnlyException
434     {
435 	if (getBoolean("?") || getBoolean("help")) {
436 	    usage();
437 	    throw new HelpOnlyException();
438 	}
439 
440 	StringBuffer unknown = new StringBuffer();
441 	int a;
442 	for (a = 0; a < args.length; a++) {
443 	    if (used.get(a))			// skip used parameters
444 		continue;
445 	    if (!args[a].startsWith("-"))	// first non-option argument
446 		break;
447 	    if (args[a].equals("--")) {		// "--" ends things
448 		a++;				// skip the "--"
449 		break;
450 	    }
451 	    unknown.append(' ').append(args[a]);
452 	}
453 	if (unknown.length() != 0) {
454 	    String ustr = unknown.toString();
455 	    throw new BadInvocationException("unknown option" +
456 		(ustr.indexOf(' ') > 0 ? "s" : "") + ":" + ustr);
457 	}
458 
459 	String[] remains = new String[args.length - a];
460 	System.arraycopy(args, a, remains, 0, remains.length);
461 	return remains;
462     }

464     /**
465      * Add the given option of the given type to the list of known options.
466      * "-?" and "-help" are handled separately in <code>getOperands</code>.
467      *
468      * @see #getOperands
469      * @see #usage
470      */
471     private void addOpt(String opt, String optType) {
472 	// ensure this is a new, not a redundant, option.
473 	Iterator it = options.iterator();
474 	while (it.hasNext()) {
475 	    Opt o = (Opt) it.next();
476 	    if (o.opt.equals(opt)) {
477 		o.multi = true;
478 		return;		// already known
479 	    }
480 	}
481 
482 	if (!isHelp(opt))
483 	    options.add(new Opt(opt, optType));
484     }
485 
486     private static boolean isHelp(String opt) {
487 	return (opt.equals("?") || opt.equals("help"));
488     }

490     /**
491      * Print out a summary of the commands usage, inferred from the
492      * requested options.  You can override this to provide a more
493      * specific summary.  This implementation is only valid after all
494      * known options have been requested and <code>getOperands</code>
495      * has been (or is being) called.  Adds <code>...</code> for
496      * operands.
497      *
498      * @see #getOperands
499      */
500     public void usage() {
501 	if (prog != null) {
502 	    System.out.print(prog);
503 	    System.out.print(' ');
504 	}
505 	System.out.print("[-?]");
506 
507 	// print out options take arguments
508 	Iterator it = options.iterator();
509 	while (it.hasNext()) {
510 	    Opt o = (Opt) it.next();
511 	    System.out.print(" [-");
512 	    System.out.print(o.opt);
513 	    if (o.argType != null) {
514 		System.out.print(' ');
515 		System.out.print(o.argType);
516 	    }
517 	    System.out.print("]");
518 	    if (o.multi)
519 		System.out.print("...");
520 	}
521 
522 	// assume other arguments
523 	System.out.println(" ...");
524     }
