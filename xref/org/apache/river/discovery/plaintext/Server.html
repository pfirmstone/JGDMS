41  /**
42   * Implements the server side of the <code>net.jini.discovery.plaintext</code>
43   * discovery format.
44   *
45   * @author Sun Microsystems, Inc.
46   * @since 2.0
47   */
48  @RequireCapability(
49  	ns="osgi.extender",
50  	filter="(osgi.extender=osgi.serviceloader.registrar)")
51  @ProvideCapability(
52  	ns="osgi.serviceloader",
53  	name="org.apache.river.discovery.DiscoveryFormatProvider")
54  public class Server
55      implements MulticastRequestDecoder,
56  	       MulticastAnnouncementEncoder,
57  	       UnicastDiscoveryServer
58  {
59      /**
60       * Constructs a new instance.
61       */
62      public Server() {
63      }
64  
65      // documentation inherited from DiscoveryFormatProvider
66      public String getFormatName() {
67  	return "net.jini.discovery.plaintext";
68      }
69      
70      // documentation inherited from MulticastRequestDecoder
71      public MulticastRequest decodeMulticastRequest(
72  					    ByteBuffer buf,
73  					    InvocationConstraints constraints,
74  					    ClientSubjectChecker checker)
75  	throws IOException
76      {
77  	Plaintext.checkConstraints(constraints);
78  	if (checker != null) {
79  	    checker.checkClientSubject(null);
80  	}
81  	return Plaintext.decodeMulticastRequest(buf);
82      }
83  
84      // documentation inherited from MulticastAnnouncementEncoder
85      public void encodeMulticastAnnouncement(MulticastAnnouncement announcement,
86  					    DatagramBufferFactory bufs,
87  					    InvocationConstraints constraints)
88  	throws IOException
89      {
90  	Plaintext.checkConstraints(constraints);
91  	Plaintext.encodeMulticastAnnouncement(announcement, bufs);
92      }
93  
94      // documentation inherited from UnicastDiscoveryServer
95      public void checkUnicastDiscoveryConstraints(
96  					    InvocationConstraints constraints)
97  	throws UnsupportedConstraintException
98      {
99  	Plaintext.checkConstraints(constraints);
100     }
101 
102     // documentation inherited from UnicastDiscoveryServer
103     public void handleUnicastDiscovery(UnicastResponse response,
104 				       Socket socket,
105 				       InvocationConstraints constraints,
106 				       ClientSubjectChecker checker,
107 				       Collection context,
108 				       ByteBuffer received,
109 				       ByteBuffer sent)
110 	throws IOException
111     {
112 	Plaintext.checkConstraints(constraints);
113 	if (checker != null) {
114 	    checker.checkClientSubject(null);
115 	}
116 	OutputStream out = new BufferedOutputStream(socket.getOutputStream());
117 	Plaintext.writeUnicastResponse(out, response, context);
118 	out.flush();
119     }
120 }
