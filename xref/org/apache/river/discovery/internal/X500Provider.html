102      * Creates an instance with the given attributes.

104     X500Provider(String formatName,
105 		 String signatureAlgorithm,
106 		 int maxSignatureLength,
107 		 String keyAlgorithm,
108 		 String keyAlgorithmOID)
109     {
110 	super(formatName);
111 	if (maxSignatureLength < 0) {
112 	    throw new IllegalArgumentException();
113 	}
114 	if (keyAlgorithm == null || keyAlgorithmOID == null) {
115 	    throw new NullPointerException();
116 	}
117 	this.signatureAlgorithm = signatureAlgorithm;
118 	this.maxSignatureLength = maxSignatureLength;
119 	this.keyAlgorithm = keyAlgorithm;
120 	this.keyAlgorithmOID = keyAlgorithmOID;
121     }

123     /**
124      * Returns certificate corresponding to the given principal, or null if no
125      * matching certificate can be found.  Subclasses can override this method
126      * to customize the certificate search mechanism.
127      * <p>
128      * The default implementation of this method does the following: the first
129      * time this method is called on this instance, a keystore containing trust
130      * anchors for the certificate to return is loaded.  The location of the
131      * file to load the keystore from can be specified (in order of precedence)
132      * by the org.apache.river.discovery.x500.trustStore and
133      * javax.net.ssl.trustStore system properties; if no location is specified,
134      * then the cacerts file in the lib/security subdirectory of the JDK
135      * installation directory is used.  If specified, the location is treated as
136      * a URL. If no protocol is specified in the URL or it is an unknown
137      * protocol, then, the location is treated as a file name.
138      * Depending on which system property is used to specify the keystore
139      * location, the org.apache.river.discovery.x500.trustStoreType and
140      * org.apache.river.discovery.x500.trustStorePassword or
141      * javax.net.ssl.trustStoreType and javax.net.ssl.trustStorePassword system
142      * properties can be used to specify the type of the keystore and the
143      * password to use when loading it.  If no keystore type is specified, then
144      * the type returned by KeyStore.getDefaultType() is used; if no password
145      * is specified, then no password is used when loading the keystore.
146      * Additionally, if the org.apache.river.discovery.x500.ldapCertStores system
147      * property is set, its value is interpreted as a comma-separated list of
148      * "host[:port]" elements which are used to obtain references to LDAP-based
149      * CertStore instances.
150      * <p>
151      * For each call, the default implementation of this method creates a PKIX
152      * CertPathBuilder and calls its build method, passing as the argument a
153      * PKIXBuilderParameters instance initialized with the aforementioned
154      * keystore, CertStores (if any), and a CertSelector based on the provided
155      * X.500 principal and the key algorithm OID for this instance.  If the
156      * build operation succeeds, the resulting certificate is returned.
157      */
158     protected Certificate getCertificate(final X500Principal principal)
159 	throws IOException, GeneralSecurityException
160     {
161 	try {
162 	    return (Certificate) AccessController.doPrivileged(
163 		new PrivilegedExceptionAction() {
164 		    public Object run()
165 			throws IOException, GeneralSecurityException
166 		    {
167 			return getCertificate0(principal);
168 		    }
169 		});
170 	} catch (PrivilegedActionException e) {
171 	    Throwable t = e.getCause();
172 	    if (t instanceof IOException) {
173 		throw (IOException) t;
174 	    } else {
175 		throw (GeneralSecurityException) t;
176 	    }
177 	}
178     }

180     /**
181      * Returns non-null array containing the usable X.500 private credentials
182      * of the current subject (if any).  This method does not check that the
183      * caller has AuthenticationPermission to use the credentials.
184      */
185     X500PrivateCredential[] getPrivateCredentials() {
186 	final AccessControlContext acc = AccessController.getContext();
187 	Collection[] subjInfo = (Collection[]) AccessController.doPrivileged(
188 	    new PrivilegedAction() {
189 		public Object run() {
190 		    Subject s = Subject.getSubject(acc);
191 		    // filter principals & credentials manually, due to 4892841
192 		    return (s != null) ?
193 			new Collection[]{
194 			    syncGetInstances(
195 				s.getPrincipals(), X500Principal.class),
196 			    syncGetInstances(
197 				s.getPrivateCredentials(),
198 				X500PrivateCredential.class)
199 			} :
200 			new Collection[]{ 
201 			    Collections.EMPTY_SET, Collections.EMPTY_SET
202 			};
203 		}
204 	    });
205 	Collection ppals = subjInfo[0];
206 	Collection creds = subjInfo[1];
207 	List l = new ArrayList();
208 	for (Iterator i = creds.iterator(); i.hasNext(); ) {
209 	    X500PrivateCredential cred = (X500PrivateCredential) i.next();
210 	    X509Certificate cert = cred.getCertificate();
211 	    try {
212 		checkCertificate(cert);
213 	    } catch (CertificateException e) {
214 		logger.log(Levels.HANDLED, "invalid certificate", e);
215 		continue;
216 	    }
217 	    if (keyAlgorithm.equals(cred.getPrivateKey().getAlgorithm()) &&
218 		ppals.contains(cert.getSubjectX500Principal()))
219 	    {
220 		l.add(cred);
221 	    }
222 	}
223 	if (logger.isLoggable(Level.FINEST)) {
224 	    logger.log(Level.FINEST, "obtained private credentials {0}",
225 		       new Object[]{ l });
226 	}
227 	return (X500PrivateCredential[]) l.toArray(
228 	    new X500PrivateCredential[l.size()]);
229     }

231     /**
232      * Test whether the caller has AuthPermission("getSubject").
233      * 
234      * @return true if the caller has AuthPermission("getSubject"),
235      *         false otherwise
236      */
237     private static boolean canGetSubject() {
238 	try {
239 	    SecurityManager sm = System.getSecurityManager();
240 	    if (sm != null) {
241 		sm.checkPermission(authPermission);
242 	    }
243 	    return true;
244 	} catch (SecurityException e) {
245 	    return false;
246 	}
247     }

249     /**
250      * Only throw non-generic exception if caller has getSubject
251      * permission.
252      *
253      * @param detailedException the real
254      *        <code>SecurityException</code> to be thrown if caller
255      *        has the "getSubject" <code>AuthPermission</code>
256      * @param genericException the generic
257      *        <code>UnsupportedConstraintException</code> to be thrown
258      *        if caller does not have the "getSubject"
259      *        <code>AuthPermission</code>
260      */
261     static void secureThrow(SecurityException detailedException,
262 			    UnsupportedConstraintException genericException)
263 	throws UnsupportedConstraintException
264     {
265 	if (canGetSubject()) { // has "getSubject" permission
266 	    throw detailedException;
267 	} else {
268 	    throw genericException;
269 	}
270     }

286     /**
287      * Returns true if the sig buffer contains the signature of the contents of
288      * the data buffer; returns false otherwise. The passed in buffers will be
289      * modified in case they do not have a backing array.
290      */
291     boolean verify(ByteBuffer data, ByteBuffer sig, PublicKey key)
292 	throws SignatureException, InvalidKeyException, NoSuchAlgorithmException
293     {
294 	data = ensureArrayBacking(data);
295 	sig = ensureArrayBacking(sig);
296 	Signature s = getSignature();
297 	s.initVerify(key);
298 	s.update(
299 	    data.array(),
300 	    data.arrayOffset() + data.position(),
301 	    data.remaining());
302 	return s.verify(
303 	    sig.array(),
304 	    sig.arrayOffset() + sig.position(),
305 	    sig.remaining());
306     }

308     /**
309      * Main body of getCertificate(), called from within a doPrivileged block.
310      */
311     private Certificate getCertificate0(X500Principal principal)
312 	throws IOException, GeneralSecurityException
313     {
314 	synchronized (storeLock) {
315 	    if (trustStore == null) {
316 		initStores();
317 	    }
318 	}
319 
320 	X509CertSelector selector = new X509CertSelector();
321 	selector.setSubject(principal.getName());
322 	selector.setSubjectPublicKeyAlgID(keyAlgorithmOID);
323 	selector.setCertificateValid(new Date());
324 	// element 0 of keyUsage array is for digital signatures
325 	selector.setKeyUsage(new boolean[]{ true });
326 	PKIXBuilderParameters params =
327 	    new PKIXBuilderParameters(trustStore, selector);
328 	for (int j = 0; j < certStores.length; j++) {
329 	    params.addCertStore(certStores[j]);
330 	}
331 	PKIXCertPathBuilderResult result;
332 	try {
333 	    result = (PKIXCertPathBuilderResult)
334 		CertPathBuilder.getInstance("PKIX").build(params);
335 	} catch (CertPathBuilderException e) {
336 	    logger.log(Levels.HANDLED,
337 		       "exception building certificate path", e);
338 	    return null;
339 	}
340 
341 	List certs = result.getCertPath().getCertificates();
342 	return certs.isEmpty() ?
343 	    result.getTrustAnchor().getTrustedCert() :
344 	    (Certificate) certs.get(0);
345     }

347     /**
348      * Initializes trust store and cert stores based on system property values.
349      */
350     private void initStores() throws IOException, GeneralSecurityException {
351 	String path, type, passwd;
352 	if ((path = System.getProperty(NAME + ".trustStore")) != null) {
353 	    type = System.getProperty(
354 		NAME + ".trustStoreType", KeyStore.getDefaultType());
355 	    passwd = System.getProperty(NAME + ".trustStorePassword");
356 	} else if ((path = System.getProperty(JSSE + ".trustStore")) != null) {
357 	    type = System.getProperty(
358 		JSSE + ".trustStoreType", KeyStore.getDefaultType());
359 	    passwd = System.getProperty(JSSE + ".trustStorePassword");
360 	} else {
361 	    path = System.getProperty("java.home") + "/lib/security/cacerts";
362 	    type = KeyStore.getDefaultType();
363 	    passwd = null;
364 	}
365 	KeyStore kstore = KeyStore.getInstance(type);
366 	InputStream in;
367 	URL url = null;
368 	try {
369 	    url = new URL(path);
370 	} catch (MalformedURLException e) {
371 	}
372 	if (url != null) {
373 	    in = url.openStream();
374 	} else {
375 	    in = new FileInputStream(path);
376 	}
377 	try {
378 	    kstore.load(in, (passwd != null) ? passwd.toCharArray() : null);
379 	} finally {
380 	    in.close();
381 	}
382 	if (logger.isLoggable(Level.FINEST)) {
383 	    logger.log(Level.FINEST, "loaded trust store from {0} ({1})",
384 		       new Object[]{ path, type });
385 	}
386 
387 	String cstores = System.getProperty(NAME + ".ldapCertStores");
388 	List l = new ArrayList();
389 	if (cstores != null) {
390 	    StringTokenizer tok = new StringTokenizer(cstores, ",");
391 	    while (tok.hasMoreTokens()) {
392 		String s = tok.nextToken().trim();
393 		Matcher m = hostPortPattern.matcher(s);
394 		try {
395 		    CertStoreParameters params = m.matches() ?
396 			new LDAPCertStoreParameters(
397 			    m.group(1), Integer.parseInt(m.group(2))) :
398 			new LDAPCertStoreParameters(s);
399 		    l.add(CertStore.getInstance("LDAP", params));
400 		} catch (Exception e) {
401 		    logger.log(Level.WARNING,
402 			       "exception initializing cert store", e);
403 		}
404 	    }
405 	}
406 	if (logger.isLoggable(Level.FINEST)) {
407 	    logger.log(Level.FINEST, "using cert stores {0}",
408 		       new Object[]{ l });
409 	}
410 	certStores = (CertStore[]) l.toArray(new CertStore[l.size()]);
411 	trustStore = kstore;
412     }

422     /**
423      * Returns a new collection containing all instances of the specified class
424      * contained in the given collection.  All operations on the given
425      * collection are performed while synchronized on the collection.
426      */
427     private static Collection syncGetInstances(Collection coll, Class cl) {
428 	synchronized (coll) {
429 	    Collection c = new ArrayList(coll.size());
430 	    for (Iterator i = coll.iterator(); i.hasNext(); ) {
431 		Object obj = i.next();
432 		if (cl.isInstance(obj)) {
433 		    c.add(obj);
434 		}
435 	    }
436 	    return c;
437 	}
438     }

440     /**
441      * Throws a CertificateException if the given certificate is not currently
442      * valid, or specifies a KeyUsage extension which prohibits use in digital
443      * signatures.
444      */
445     private static void checkCertificate(X509Certificate cert)
446 	throws CertificateException
447     {
448 	cert.checkValidity();
449 	boolean[] keyUsage = cert.getKeyUsage();
450 	// element 0 of keyUsage array is for digital signatures
451 	if (keyUsage != null && keyUsage.length > 0 && !keyUsage[0]) {
452 	    throw new CertificateException(
453 		"certificate not permitted for digital signatures: " + cert);
454 	}
455     }

457     /**
458      * Returns given buffer if it is backed by an array; otherwise, returns a
459      * newly created array-backed buffer into which the remaining contents of
460      * the given buffer have been transferred.
461      */
462     private static ByteBuffer ensureArrayBacking(ByteBuffer buf) {
463 	return buf.hasArray() ?
464 	    buf : (ByteBuffer)
465 		    ByteBuffer.allocate(buf.remaining()).put(buf).flip();
466     }

468     /**
469      * Buffer factory which signs data written into the buffers it dispenses.
470      */
471     class SigningBufferFactory implements DatagramBufferFactory {
472 
473 	private final List buffers = new ArrayList();
474 	private final DatagramBufferFactory factory;
475 	private final byte[] principalName;
476 	private final Signature signature;
477 
478 	SigningBufferFactory(DatagramBufferFactory factory,
479 			     X500PrivateCredential cred)
480 	    throws InvalidKeyException,
481 		   UTFDataFormatException,
482 		   NoSuchAlgorithmException
483 	{
484 	    this.factory = factory;
485 	    principalName = Plaintext.toUtf(
486 		cred.getCertificate().getSubjectX500Principal().getName());
487 	    signature = getSignature();
488 	    signature.initSign(cred.getPrivateKey());
489 	}
490 
491 	public ByteBuffer newBuffer() {
492 	    BufferInfo bi = new BufferInfo(factory.newBuffer());
493 	    buffers.add(bi);
494 	    return bi.getDataBuffer();
495 	}
496 
497 	public void sign() throws SignatureException {
498 	    for (Iterator i = buffers.iterator(); i.hasNext(); ) {
499 		((BufferInfo) i.next()).sign();
500 	    }
501 	}
502 
503 	private class BufferInfo {
504 
505 	    private final ByteBuffer buf;
506 	    private final ByteBuffer data;
507 	    private final boolean overflow;
508 
509 	    BufferInfo(ByteBuffer buf) {
510 		this.buf = buf;
511 		data = buf.duplicate();
512 		int authBlockLen = principalName.length + maxSignatureLength;
513 		if (data.remaining() >= INT_LEN + authBlockLen) {
514 		    data.position(data.position() + INT_LEN);
515 		    data.limit(data.limit() - authBlockLen);
516 		    overflow = false;
517 		} else {
518 		    data.limit(data.position());
519 		    overflow = true;
520 		}
521 	    }
522 
523 	    ByteBuffer getDataBuffer() {
524 		return data;
525 	    }
526 
527 	    void sign() throws SignatureException {
528 		if (overflow) {
529 		    throw new BufferOverflowException();
530 		}
531 		buf.putInt(data.position() - (buf.position() + INT_LEN));
532 		buf.position(data.position());
533 		buf.put(principalName);
534 
535 		ByteBuffer b =
536 		    ensureArrayBacking((ByteBuffer) data.duplicate().flip());
537 		signature.update(
538 		    b.array(), b.arrayOffset() + b.position(), b.remaining());
539 		buf.put(signature.sign());
540 	    }
541 	}
