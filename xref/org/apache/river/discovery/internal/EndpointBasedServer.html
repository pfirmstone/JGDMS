64  /**
65   * Provides an abstract server endpoint-based UnicastDiscoveryServer
66   * implementation, which serves as a superclass for server-side providers for
67   * the net.jini.discovery.ssl and net.jini.discovery.kerberos unicast discovery
68   * formats.
69   */
70  public abstract class EndpointBasedServer
71      extends EndpointBasedProvider implements UnicastDiscoveryServer
72  {
73      /**
74       * Constructs instance with the given format name and object providing
75       * access to non-public endpoint operations.
76       */
77      protected EndpointBasedServer(String formatName,
78  				  EndpointInternals endpointInternals)
79      {
80  	super(formatName, endpointInternals);
81      }

83      // documentation inherited from UnicastDiscoveryServer
84      public void checkUnicastDiscoveryConstraints(
85  					    InvocationConstraints constraints)
86  	throws UnsupportedConstraintException
87      {
88  	if (constraints == null) {
89  	    constraints = InvocationConstraints.EMPTY;
90  	}
91  	ServerEndpoint ep = getServerEndpoint(null);
92  	InvocationConstraints unfulfilled = ep.checkConstraints(constraints);
93  	checkIntegrity(unfulfilled);
94  	checkAtomicity(unfulfilled);
95      }

97      // documentation inherited from UnicastDiscoveryServer
98      public void handleUnicastDiscovery(UnicastResponse response,
99  				       Socket socket,
100 				       InvocationConstraints constraints,
101 				       ClientSubjectChecker checker,
102 				       Collection context,
103 				       ByteBuffer received,
104 				       ByteBuffer sent)
105 	throws IOException
106     {
107 	if (response == null || socket == null ||
108 	    received == null || sent == null)
109 	{
110 	    throw new NullPointerException();
111 	}
112 	if (constraints == null) {
113 	    constraints = InvocationConstraints.EMPTY;
114 	}
115 	ServerEndpoint ep = 
116 	    getServerEndpoint(new PrearrangedServerSocketFactory(socket));
117 	ServerConnManagerImpl mgr = new ServerConnManagerImpl();
118 	endpointInternals.setServerConnManager(ep, mgr);
119 
120 	ListenContextImpl lc = new ListenContextImpl();
121 	ep.enumerateListenEndpoints(lc);
122 	ServerConnection conn = mgr.getServerConnection();
123 	try {
124 	    InputStream in = new BufferedInputStream(conn.getInputStream());
125 	    OutputStream out = new BufferedOutputStream(conn.getOutputStream());
126 	    InboundRequestHandle handle = conn.processRequestData(in, out);
127 	    conn.checkPermissions(handle);
128 	    InvocationConstraints higherLayerConstraints = 
129 		    conn.checkConstraints(handle, constraints);
130 	    checkIntegrity(higherLayerConstraints);
131 	    boolean atomicity = checkAtomicity(higherLayerConstraints);
132 	    if (atomicity){
133 		context = context == null ? new ArrayList() : new ArrayList(context);
134 		context.add( new AtomicValidationEnforcement(){
135 
136 		    @Override
137 		    public boolean enforced() {
138 			return true;
139 		    }
140 		});
141 		context = Collections.unmodifiableCollection(context);
142 	    }
143 	    if (checker != null) {
144 		checker.checkClientSubject(getClientSubject(conn, handle));
145 	    }
146 
147 	    byte[] hash = calcHandshakeHash(received, sent);
148 	    byte[] clientHash = new byte[hash.length];
149 	    new DataInputStream(in).readFully(clientHash);
150 	    if (!Arrays.equals(clientHash, hash)) {
151 		throw new DiscoveryProtocolException(
152 		    "handshake hash mismatch");
153 	    }
154 
155 	    writeUnicastResponse(out, response, context);
156 	} finally {
157 	    conn.close();
158 	    lc.getListenHandle().close();
159 	}
160     }

163      * Allows server providers to use a different unicast response.
164      * 
165      * 
166      * @param out
167      * @param response
168      * @param context
169      * @throws IOException 
170      */
171     protected void writeUnicastResponse(OutputStream out,
172 					UnicastResponse response,
173 					Collection context)
174 	throws IOException
175     {
176 	Plaintext.writeUnicastResponse(out, response, context);
177 	out.flush();
178     }

180     protected void writeClassAnnotationCerts(OutputStream out,
181 					UnicastResponse response) throws IOException
182     {
183 	String classAnnotation = "";
184 	    String certFactoryType = classAnnotation;
185 	    String certPathEncoding = certFactoryType;
186 	    byte [] encodedCerts = new byte[0];
187 	    ServiceRegistrar reg = response.getRegistrar();
188 	    if (reg instanceof ProxyAccessor){
189 		Object proxy = ((ProxyAccessor) reg).getProxy();
190 		if (proxy instanceof CodebaseAccessor){
191 		    CodebaseAccessor sca = (CodebaseAccessor) proxy;
192 		    classAnnotation = sca.getClassAnnotation();
193 		    certFactoryType = sca.getCertFactoryType();
194 		    certPathEncoding = sca.getCertPathEncoding();
195 		    encodedCerts = sca.getEncodedCerts();
196 		}
197 	    }
198 	    DataOutputStream dout = new DataOutputStream(out);
199 	    dout.writeUTF(classAnnotation);
200 	    dout.writeUTF(certFactoryType);
201 	    dout.writeUTF(certPathEncoding);
202 	    dout.writeShort(encodedCerts.length);
203 	    dout.write(encodedCerts);
204     }

216     /**
217      * Returns the subject that the client of the given connection has
218      * authenticated as, or null if the client is not authenticated.
219      */
220     private static Subject getClientSubject(ServerConnection connection,
221 					    InboundRequestHandle handle)
222     {
223 	Collection ctx = new ArrayList();
224 	connection.populateContext(handle, ctx);
225 	for (Iterator i = ctx.iterator(); i.hasNext(); ) {
226 	    Object obj = i.next();
227 	    if (obj instanceof ClientSubject) {
228 		return ((ClientSubject) obj).getClientSubject();
229 	    }
230 	}
231 	return null;
232     }

234     /**
235      * Server connection manager that stores the connection it is given to
236      * handle.
237      */
238     private static class ServerConnManagerImpl implements ServerConnManager {
239 
240 	private ServerConnection conn = null;
241 
242 	ServerConnManagerImpl() {
243 	}
244 
245 	public synchronized void handleConnection(ServerConnection conn,
246 						  RequestDispatcher disp)
247 	{
248 	    if (conn == null || disp == null) {
249 		throw new NullPointerException();
250 	    }
251 	    this.conn = conn;
252 	    notifyAll();
253 	}
254 
255 	synchronized ServerConnection getServerConnection() {
256 	    while (conn == null) {
257 		try { wait(); } catch (InterruptedException e) {}
258 	    }
259 	    return conn;
260 	}
261     }

264      * Listen context that listens on the endpoint it is given, and stores the

266      */
267     private static class ListenContextImpl implements ListenContext {
268 
269 	private ListenHandle handle = null;

272 	    throws IOException
273 	{
274 	    handle = endpoint.listen(new RequestDispatcher() {
275 		public void dispatch(InboundRequest req) {
276 		    throw new AssertionError("dispatch should not occur");
277 		}
278 	    });
279 	    return handle.getCookie();
280 	}
281 
282 	ListenHandle getListenHandle() {
283 	    return handle;
284 	}

287     /**
288      * Server socket that returns a prearranged socket once from its accept
289      * method, and then blocks on subsequent calls to accept until closed.
290      */
291     private static class PrearrangedServerSocket extends ServerSocket {
292 
293 	private Socket socket;
294 	private boolean closed = false;
295 
296 	PrearrangedServerSocket(Socket socket) throws IOException {
297 	    this.socket = socket;
298 	}
299 
300 	public synchronized Socket accept() throws IOException {
301 	    if (!closed && socket != null) {
302 		Socket s = socket;
303 		socket = null;
304 		return s;
305 	    }
306 	    while (!closed) {
307 		try { wait(); } catch (InterruptedException e) {}
308 	    }
309 	    throw new SocketException("socket closed");
310 	}
311 
312 	public int getLocalPort() {
313 	    /*
314 	     * Although this port number is bogus, it has to be plausible,
315 	     * otherwise calling enumerateListenEndpoints on the server
316 	     * endpoint will fail since an endpoint cannot be created.
317 	     */
318 	    return 1;
319 	}
320 
321 	public synchronized void close() throws IOException {
322 	    if (!closed) {
323 		closed = true;
324 		notifyAll();
325 	    }
326 	}
327     }
