36  /**
37   * When clients request a registration with the Fiddler implementation of
38   * the lookup discovery service, leases of type FiddlerLease are granted
39   * on those registrations. Under certain circumstances it may be desirable
40   * to collect multiple granted leases in a set which implements the
41   * <code>net.jini.core.lease.LeaseMap</code> interface. 
42   * <p>
43   * This class is the implementation class of the <code>LeaseMap</code>
44   * interface that is employed by the Fiddler implementation of the lookup
45   * discovery service. When a client wishes to "batch" leases granted by
46   * that service, the are placed in an instance of this class.
47   * <p>
48   * Clients only see instances of this class via the  <code>LeaseMap</code>
49   * interface.
50   *
51   * @author Sun Microsystems, Inc.
52   *
53   */
54  class FiddlerLeaseMap extends AbstractIDLeaseMap<FiddlerLease> {
55  
56      /**
57       * The reference to the back-end server of the lookup discovery service
58       *
59       * @serial
60       */
61      final Fiddler server;
62      /**
63       * The unique ID associated with the server referenced in this class
64       * (used to compare server references).
65       *
66       * @serial
67       */
68      final Uuid serverID;
69  
70      /**
71       * Static factory method that creates and returns an instance of 
72       * <code>FiddlerLeaseMap</code>. If the lease input to this method
73       * is and instance of <code>ConstrainableFiddlerLease</code>, then
74       * the object returned by this method will be an instance of
75       * <code>ConstrainableFiddlerLeaseMap</code>.
76       *
77       * Note that because of the way the <code>Lease</code> class for the
78       * associated service is implemented, together with the way the
79       * <code>canBatch</code> method of that class is implemented, if the
80       * <code>lease</code> object input to this method implements
81       * <code>ConstrainableFiddlerLease</code>, then the <code>server</code>
82       * object referenced in this class will implement 
83       * <code>RemoteMethodControl</code> as well.
84       * 
85       * @param lease    reference to a lease to add to the map as a key value
86       * @param duration the duration of the corresponding lease. This value
87       *                 is the "mapped" value corresponding to the lease key.
88       * 
89       * @return an instance of <code>FiddlerLeaseMap</code>, or an instance
90       *         of <code>ConstrainableFiddlerLeaseMap</code> if the given 
91       *         <code>lease</code> is an instance of
92       *         <code>ConstrainableFiddlerLease</code>.
93       */
94      static FiddlerLeaseMap createLeaseMap(FiddlerLease lease, long duration) {
95          if(lease instanceof FiddlerLease.ConstrainableFiddlerLease) {
96              MethodConstraints leaseConstraints =
97          ((FiddlerLease.ConstrainableFiddlerLease)lease).getConstraints();
98  
99              return new ConstrainableFiddlerLeaseMap(lease.getServer(),
100                                                     lease,
101                                                     duration,
102                                                     leaseConstraints);
103         } else {
104             return new FiddlerLeaseMap(lease.getServer(), lease, duration);
105         }//endif
106     }//end createLeaseMap
107 
108     /**
109      * Constructs a new instance of FiddlerLeaseMap.
110      *
111      * @param server   reference to the server object through which 
112      *                 communication occurs between the client-side and
113      *                 server-side of the associated service.
114      * @param lease    reference to a lease to add to the map as a key value
115      * @param duration the duration of the corresponding lease. This value
116      *                 is the "mapped" value corresponding to the lease key.
117      */
118     private FiddlerLeaseMap(Fiddler server, FiddlerLease lease, long duration){
119         super();
120         this.server = server;
121         this.serverID = lease.getServerID();
122         put(lease, duration);
123     }//end constructor
124 
125     /**
126      * Examines the input parameter to determine if that parameter will be
127      * accepted or rejected by this map as a "legal" key value.
128      * <p>
129      * This method will return true if the <code>key</code> parameter is 
130      * the type of lease which can be "batch-wise" renewed and cancelled 
131      * along with all of the other leases in the map.
132      * <p>
133      * For the Fiddler implementation of the lookup discovery service, two
134      * leases can be batched (placed in the same </code>FiddlerLeaseMap</code>)
135      * if they are both instances of <code>FiddlerLease</code> and they
136      * were both granted by the same Fiddler implementation of the lookup
137      * discovery service. That is, they are the same type, and they were
138      * granted by the same server.
139      *                      
140      * @param key reference to the object that this method examines to
141      *        determine if this map will accept or reject it as a key
142      *                      
143      * @return true if this map will accept the <code>key</code> parameter,
144      *         false otherwise
145      *
146      * @see net.jini.core.lease.Lease#canBatch
147      */
148     public boolean canContainKey(Object key) {
149         return (    (key instanceof FiddlerLease)
150                  && (serverID.equals(((FiddlerLease)key).getServerID()))  );
151     }
152 
153     /**
154      * Renews all leases in this map. For each lease (key) in the map, the
155      * duration used to renew the lease is the lease's corresponding map
156      * value. If all renewal attempts are successful, this method returns
157      * normally; otherwise, this method removes from this map all leases
158      * that could not be renewed, and throws a <code>LeaseMapException</code>.
159      *
160      * @throws net.jini.core.lease.LeaseMapException this exception is thrown
161      *         when one or more leases in the map could not be renewed.
162      * 
163      * @throws java.rmi.RemoteException typically, this exception occurs when
164      *         there is a communication failure between the client and the
165      *         server. When this exception does occur, the leases in the map
166      *         may or may not have been renewed successfully.
167      */
168     public void renewAll() throws LeaseMapException, RemoteException {
169         if (isEmpty()) return;
170         List<FiddlerLease> keys = new LinkedList<FiddlerLease>();
171         List<Uuid> registrationIDs = new LinkedList<Uuid>();
172         List<Uuid> leaseIDs = new LinkedList<Uuid>();
173         List<Long> durations = new LinkedList<Long>();
174         Iterator<Entry<FiddlerLease,Long>> it = entrySet().iterator();
175         while (it.hasNext()){
176             Entry<FiddlerLease,Long> e = it.next();
177             FiddlerLease ls = e.getKey();
178             keys.add(ls);
179             registrationIDs.add(ls.getRegistrationID());
180             leaseIDs.add(ls.getLeaseID());
181             durations.add(e.getValue());
182         }
183         long [] dur = new long[durations.size()];
184         int i = 0;
185         for (Iterator<Long> durIt = durations.iterator(); durIt.hasNext(); i++){
186             dur[i] = durIt.next().intValue();
187         }
188         FiddlerRenewResults results = server.renewLeases(
189                 registrationIDs.toArray(new Uuid[registrationIDs.size()]),
190                 leaseIDs.toArray(new Uuid[leaseIDs.size()]),
191                 dur
192                 );
193         long now = System.currentTimeMillis();
194         HashMap<Lease,Throwable> emap = (results.exceptions != null) ?
195                          new HashMap<Lease,Throwable>(2 * results.exceptions.length + 1) : null;
196         i = 0;
197         int j = 0;
198         for (Iterator<FiddlerLease> iter = keys.iterator(); iter.hasNext(); i++) {
199             FiddlerLease e = iter.next();
200             long duration = results.durations[i];
201             if (duration >= 0) {
202                 // Mutation of volatile field
203                 e.setExpiration(duration + now);
204             } else {
205                 emap.put(e, results.exceptions[j++]);
206                 remove(e);
207             }
208         }
209         if (emap != null) {
210             throw new LeaseMapException("lease renewal failures", emap);
211         }
212     }
213 
214     /**
215      * Cancels all leases in this map. If all cancellation attempts are
216      * successful, this method returns normally; otherwise, this method
217      * removes from this map all leases that could not be cancelled, and
218      * throws a <code>LeaseMapException</code>.
219      * <p>
220      * Note that a lease is removed from this map only when an attempt to
221      * cancel it fails; that is, every lease that is successfully cancelled
222      * is left in the map.
223      *
224      * @throws net.jini.core.lease.LeaseMapException this exception is thrown
225      *         when one or more leases in the map could not be cancelled.
226      * 
227      * @throws java.rmi.RemoteException typically, this exception occurs when
228      *         there is a communication failure between the client and the
229      *         server. When this exception does occur, the leases in the map
230      *         may or may not have been cancelled successfully.
231      */
232     public void cancelAll() throws LeaseMapException, RemoteException {
233         if( isEmpty()) return;
234         List<FiddlerLease> leases = new LinkedList<FiddlerLease>();
235         List<Uuid> registrationIDs = new LinkedList<Uuid>();
236         List<Uuid> leaseIDs = new LinkedList<Uuid>();
237         int i = 0;
238         for (Iterator<FiddlerLease> iter = keySet().iterator(); iter.hasNext(); i++) {
239             FiddlerLease ls = iter.next();
240             leases.add(ls);
241             registrationIDs.add(ls.getRegistrationID());
242             leaseIDs.add(ls.getLeaseID());
243         }
244         Exception[] exceptions = server.cancelLeases(
245                 registrationIDs.toArray(new Uuid[registrationIDs.size()]),
246                 leaseIDs.toArray(new Uuid[leaseIDs.size()])
247                 );
248         if (exceptions == null) {
249             return;
250         }
251         i = 0;
252         Map<Lease,Throwable> emap = new HashMap<Lease,Throwable>(13);
253         for (Iterator<FiddlerLease> iter = leases.iterator(); iter.hasNext(); i++) {
254             FiddlerLease ls = iter.next();
255             Exception ex = exceptions[i];
256             if (ex != null) {
257                 emap.put(ls, ex);
258                 remove(ls);
259             }
260         }
261         throw new LeaseMapException("lease cancellation failures", emap);
262     }

264     /** The constrainable version of the class <code>FiddlerLeaseMap</code>. 
265      *
266      * @since 2.0
267      */
268     static final class ConstrainableFiddlerLeaseMap extends FiddlerLeaseMap {
269 
270         /* Convenience fields containing, respectively, the renew and cancel
271          * methods defined in the Lease interface. These fields are used in
272          * the method mapping array, and when retrieving method constraints
273          * for comparison in canContainKey().
274          */
275         private static final Method renewMethod =
276                              ProxyUtil.getMethod(Lease.class,
277                                                  "renew",
278                                                  new Class[] {long.class} );
279         private static final Method cancelMethod =
280                              ProxyUtil.getMethod(Lease.class,
281                                                  "cancel",
282                                                  new Class[] {} );
283 
284         /* When a <code>LeaseMap</code> is created, an implicit set of 
285          * constraints is generated and associated with that lease map.
286          * The constraints that are generated are based on the constraints
287          * on the first lease placed in that lease map. As such, the array
288          * defined here contains element pairs in which each pair of elements
289          * represents a correspondence 'mapping' between two methods having
290          * the following characteristics:
291          *  - the first element in the pair is one of the public, remote
292          *    method(s) that may be invoked by the client through a lease
293          *    proxy, and which has method constraints that are intended 
294          *    to apply to the corresponding method that is invoked on
295          *    server's backend
296          *  - the second element in the pair is the method, implemented
297          *    in the backend server class, that is intended to be 
298          *    executed with the same method constraints as its corresponding
299          *    method, specified in the first element of the pair
300          */
301         private static final Method[] methodMapArray = 
302         {
303             renewMethod, ProxyUtil.getMethod(Fiddler.class,
304                                              "renewLeases", 
305                                              new Class[] {Uuid[].class,
306                                                           Uuid[].class,
307                                                           long[].class}),
308 
309             cancelMethod, ProxyUtil.getMethod(Fiddler.class,
310                                               "cancelLeases", 
311                                               new Class[] {Uuid[].class,
312                                                            Uuid[].class})
313         };//end methodMapArray
314 
315         /** In order to determine if a given lease will be accepted by this
316          *  map as a "legal" key value, the method <code>canContainKey</code>
317          *  must verify that the corresponding methods of the initial
318          *  lease used to create this map and the lease input to
319          *  <code>canContainKey</code> have equivalent constraints. The
320          *  array defined here contains the set of methods whose constraints
321          *  will be compared in <code>canContainKey</code>.
322          */
323         private static final Method[] canContainKeyMethodMapArray =
324                                             { renewMethod,  renewMethod,
325                                               cancelMethod, cancelMethod
326                                             };//end canContainKeyMethodMapArray
327 
328         /** Client constraints placed on this proxy (may be <code>null</code>).
329          *
330          * @serial
331          */
332         private MethodConstraints methodConstraints;
333 
334         /** Constructs a new <code>ConstrainableFiddlerLeaseMap</code>
335          *  instance.
336          *  <p>
337          *  For a description of all but the <code>methodConstraints</code>
338          *  argument (provided below), refer to the description for the
339          *  constructor of this class' super class.
340          *
341          *  @param methodConstraints the client method constraints to place on
342          *                           this proxy (may be <code>null</code>).
343          */
344         private ConstrainableFiddlerLeaseMap
345                                         (Fiddler server, 
346                                          FiddlerLease lease,
347                                          long duration,
348                                          MethodConstraints methodConstraints)
349         {
350             super(constrainServer(server, methodConstraints), lease, duration);
351 	    this.methodConstraints = methodConstraints;
352         }//end constructor
353 
354         /**
355          * Examines the input parameter to determine if that parameter will
356          * be accepted or rejected by this map as a "legal" key value.
357          * <p>
358          * This method will return true if the <code>key</code> parameter is 
359          * the type of lease which can be "batch-wise" renewed and cancelled 
360          * along with all of the other leases in the map.
361          * <p>
362          * For this implementation of the service, two leases can be
363          * batched (placed in the same service-specific instance of
364          * <code>LeaseMap</code>) if those leases satisfy the following
365          * conditions:
366          * <p><ul>
367          *    <li> the leases are the same type; that is, the leases are 
368          *         both instances of the same, constrainable, service-specific
369          *         <code>Lease</code> implementation
370          *    <li> the leases were granted by the same backend server
371          *    <li> the leases have the same constraints
372          *                      
373          * @param key reference to the object that this method examines to
374          *        determine if this map will accept or reject it as a key
375          *                      
376          * @return <code>true</code> if this map will accept the
377          *         <code>key</code> parameter, <code>false</code> otherwise
378          *
379          * @see net.jini.core.lease.Lease#canBatch
380          * @see net.jini.core.lease.LeaseMap#canContainKey
381          */
382         public boolean canContainKey(Object key) {
383             if( !(super.canContainKey(key)) )  return false;
384             /* Non-constrainable criteria satisfied, now handle constrainable
385              * case.
386              */
387             if( !(key instanceof FiddlerLease.ConstrainableFiddlerLease) ) {
388                 return false;
389             }//endif
390             /* Compare constraints */
391             MethodConstraints keyConstraints =
392           ((FiddlerLease.ConstrainableFiddlerLease)key).getConstraints();
393             return ConstrainableProxyUtil.equivalentConstraints
394                                               ( methodConstraints,
395                                                 keyConstraints,
396                                                 canContainKeyMethodMapArray );
397         }//end ConstrainableFiddlerLeaseMap.canContainKey
398 
399         /** Returns a copy of the given server proxy having the client method
400          *  constraints that result after the specified method mapping is
401          *  applied to the given client method constraints.
402          */
403         private static Fiddler constrainServer( Fiddler server,
404                                                 MethodConstraints constraints )
405         {
406             MethodConstraints newConstraints 
407                = ConstrainableProxyUtil.translateConstraints(constraints,
408                                                              methodMapArray);
409             RemoteMethodControl constrainedServer = 
410                 ((RemoteMethodControl)server).setConstraints(newConstraints);
411 
412             return ((Fiddler)constrainedServer);
413         }//end constrainServer
414 
415     }//end class ConstrainableFiddlerLeaseMap
