18  package org.apache.river.norm.proxy;
19  
20  import java.rmi.MarshalledObject;
21  import java.rmi.Remote;
22  import java.rmi.RemoteException;
23  
24  import net.jini.admin.Administrable;
25  import net.jini.admin.JoinAdmin;
26  import net.jini.core.event.EventRegistration;
27  import net.jini.core.event.RemoteEventListener;
28  import net.jini.core.lease.Lease;
29  import net.jini.id.Uuid;
30  import net.jini.io.MarshalledInstance;
31  import net.jini.lease.LeaseRenewalService;
32  
33  import org.apache.river.admin.DestroyAdmin;
34  import org.apache.river.landlord.Landlord;
35  import org.apache.river.proxy.ThrowThis;
36  import net.jini.lookup.ServiceProxyAccessor;
37  
38  /**
39   * This interface is the private wire protocol to that the various
40   * proxy objects created by a Norm server (the lease renewal service
41   * itself, sets, leases, admins, etc.) use to communicate back to the
42   * server.
43   *
44   * @author Sun Microsystems, Inc.
45   */
46  public interface NormServer extends Landlord, LeaseRenewalService, 
47      Administrable, JoinAdmin, DestroyAdmin, ServiceProxyAccessor, Remote 
48  {
49      /**
50       * If calling <code>setExpirationWarningListener</code> with a
51       * <code>null</code> listener, this is the value that should be passed.
52       */
53      final static long NO_LISTENER = -1;
54  
55      /**
56       * Add a lease to a set.
57       *
58       * @param id what set the lease should be added to
59       * @param leaseToRenew the lease to be added to the set
60       * @param membershipDuration how long the lease should be in the set
61       * @param renewDuration how long the lease should be renewed for
62       *	      each time it is renewed
63       * @throws ThrowThis when another exception has to be thrown by the proxy
64       * @throws RemoteException if a communication-related exception occurs
65       */
66      public void renewFor(Uuid id, Lease leaseToRenew,
67  			 long membershipDuration, long renewDuration)
68  	throws RemoteException, ThrowThis;
69  
70      /**
71       * Remove a lease from a set.
72       *
73       * @param id of set being operated on
74       * @param leaseToRemove the lease to be removed from the set
75       * @throws ThrowThis when another exception has to be thrown by the proxy
76       * @throws RemoteException if a communication-related exception occurs
77       */
78      public Lease remove(Uuid id, Lease leaseToRemove) 
79  	throws RemoteException, ThrowThis;
80  
81      /**
82       * Return all the leases in the set.  Returns <code>null</code>
83       * or a zero-length array if there are no leases in the set.
84       *
85       * @param id of set being operated on
86       * @return an object containing an array of {@link MarshalledInstance}s,
87       *	       one for each lease
88       * @throws ThrowThis when another exception has to be thrown by the proxy
89       * @throws RemoteException if a communication-related exception occurs
90       */
91      public GetLeasesResult getLeases(Uuid id) 
92  	throws RemoteException, ThrowThis;
93  
94      /**
95       * Set the expiration warning listener for a set.  Also used to 
96       * cancel a registration.
97       *
98       * @param id of set being operated on
99       * @param listener listener to be notified when this event occurs.
100      *        Pass <code>null</code> to clear the registration.
101      * @param minWarning how long be for the lease on the set expires
102      *        should the event be sent. Ignored if <code>listener</code>
103      *        is <code>null</code>.
104      * @param handback an object to be handed back to the listener when
105      *        the warning event occurs. Ignored if <code>listener</code>
106      *        is <code>null</code>.
107      * @return an <code>EventRegistration</code> object for the new
108      *	       registration if <code>listener</code> is
109      *	       non-<code>null</code> and <code>null</code> otherwise
110      * @throws ThrowThis when another exception has to be thrown by the proxy
111      * @throws RemoteException if a communication-related exception occurs
112      */
113     public EventRegistration setExpirationWarningListener(
114 			         Uuid                id,
115 	                         RemoteEventListener listener, 
116 				 long                minWarning, 
117 				 MarshalledObject    handback)
118 	throws RemoteException, ThrowThis;
119 
120     /**
121      * Set the renewal failure listener for a set.  Also used to 
122      * cancel a registration.
123      * @param id of set being operated on
124      * @param listener listener to be notified when this event occurs.
125      *        Pass <code>null</code> to clear the registration.
126      * @param handback an object to be handed back to the listener when
127      *        the failure event occurs.  Ignored if
128      *        <code>listener</code> is <code>null</code>.
129      * @return an <code>EventRegistration</code> object for the new
130      *	       registration if <code>listener</code> is
131      *	       non-<code>null</code> and <code>null</code> otherwise.
132      * @throws ThrowThis when another exception has to be thrown by the proxy
133      * @throws RemoteException if a communication-related exception occurs
134      */
135     public EventRegistration setRenewalFailureListener(
136 			         Uuid                id,
137 	                         RemoteEventListener listener, 
138 				 MarshalledObject    handback)
139 	throws RemoteException, ThrowThis;
140 }
