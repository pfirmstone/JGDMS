44  /**
45   * NormServerInitializer creates the internal state of NormServerBaseImpl allowing
46   * many fields to be final after construction.  This simplifies auditing of
47   * synchronization by reducing scope to mutable fields and references 
48   * to mutable objects.
49   */
50  class NormServerInitializer {
51      boolean persistent;
52      LifeCycle lifeCycle;
53      LoginContext loginContext;
54      String persistenceDirectory;
55      List renewedList = new LinkedList();
56      float snapshotWt;
57      int logToSnapshotThresh;
58      ProxyPreparer leasePreparer;
59      ProxyPreparer listenerPreparer;
60      ProxyPreparer locatorPreparer;
61      ProxyPreparer recoveredLeasePreparer;
62      ProxyPreparer recoveredListenerPreparer;
63      ProxyPreparer recoveredLocatorPreparer;
64      LeasePeriodPolicy setLeasePolicy;
65      boolean isolateSets;
66      LeaseRenewalManager lrm;
67      Exporter exporter;
68      LeaseExpirationMgr expMgr;
69      EventTypeGenerator generator;
70      LRMEventListener lrmEventListener;
71      NormServerBaseImpl.RenewLogThread renewLogger;
72      AccessControlContext context;
73      Configuration config;
74      String codebase;
75      String certFactoryType;
76      String certPathEncoding;
77      byte[] encodedCerts;
78  
79      /**
80       * Initializer object for NormServer implementations.  Can be overridden by
81       * subclasses of NormServerBaseImpl
82       * 
83       * @param persistent true if implementation is persistent
84       * @param lifeCycle object to notify when NormServer is destroyed.
85       */
86      NormServerInitializer(boolean persistent, LifeCycle lifeCycle) {
87          this.persistent = persistent;
88          this.lifeCycle = lifeCycle;
89      }
90  
91      /**
92       * Common construction for activatable and non-activatable cases, run
93       * under the proper Subject.
94       */
95      void initAsSubject(Configuration config) throws Exception {
96          this.config = config;
97          /* Get configuration entries first */
98          if (persistent) {
99              persistenceDirectory = (String) Config.getNonNullEntry(config, NormServerBaseImpl.NORM, "persistenceDirectory", String.class);
100             snapshotWt = Config.getFloatEntry(config, NormServerBaseImpl.NORM, "persistenceSnapshotWeight", 10, 0, Float.MAX_VALUE);
101             logToSnapshotThresh = Config.getIntEntry(config, NormServerBaseImpl.NORM, "persistenceSnapshotThreshold", 200, 0, Integer.MAX_VALUE);
102         }
103         leasePreparer = (ProxyPreparer) Config.getNonNullEntry(config, NormServerBaseImpl.NORM, "leasePreparer", ProxyPreparer.class, new BasicProxyPreparer());
104         listenerPreparer = (ProxyPreparer) Config.getNonNullEntry(config, NormServerBaseImpl.NORM, "listenerPreparer", ProxyPreparer.class, new BasicProxyPreparer());
105         locatorPreparer = (ProxyPreparer) Config.getNonNullEntry(config, NormServerBaseImpl.NORM, "locatorPreparer", ProxyPreparer.class, new BasicProxyPreparer());
106         if (persistent) {
107             recoveredLeasePreparer = (ProxyPreparer) Config.getNonNullEntry(config, NormServerBaseImpl.NORM, "recoveredLeasePreparer", ProxyPreparer.class, new BasicProxyPreparer());
108             recoveredListenerPreparer = (ProxyPreparer) Config.getNonNullEntry(config, NormServerBaseImpl.NORM, "recoveredListenerPreparer", ProxyPreparer.class, new BasicProxyPreparer());
109             recoveredLocatorPreparer = (ProxyPreparer) Config.getNonNullEntry(config, NormServerBaseImpl.NORM, "recoveredLocatorPreparer", ProxyPreparer.class, new BasicProxyPreparer());
110         }
111         setLeasePolicy = (LeasePeriodPolicy) Config.getNonNullEntry(config, NormServerBaseImpl.NORM, "leasePolicy", LeasePeriodPolicy.class, new FixedLeasePeriodPolicy(2 * 60 * 60 * 1000, 60 * 60 * 1000));
112         isolateSets = ((Boolean) config.getEntry(NormServerBaseImpl.NORM, "isolateSets", boolean.class, Boolean.FALSE)).booleanValue();
113         try {
114             lrm = (LeaseRenewalManager) Config.getNonNullEntry(config, NormServerBaseImpl.NORM, "leaseManager", LeaseRenewalManager.class);
115         } catch (NoSuchEntryException e) {
116             lrm = new LeaseRenewalManager(config);
117         }
118         exporter = getExporter(config);
119 	codebase = Config.getNonNullEntry(config, NormServerBaseImpl.NORM,
120 		"Codebase_Annotation", String.class, "");
121 	certFactoryType = Config.getNonNullEntry(config, NormServerBaseImpl.NORM,
122 		"Codebase_CertFactoryType", String.class, "X.509");
123 	certPathEncoding = Config.getNonNullEntry(config, NormServerBaseImpl.NORM,
124 		"Codebase_CertPathEncoding", String.class, "PkiPath");
125 	encodedCerts = Config.getNonNullEntry(config, NormServerBaseImpl.NORM,
126 		"Codebase_Certs", byte[].class, new byte[0]);
127 	
128         // We use some of these during the recovery process
129         expMgr = new LeaseExpirationMgr();
130         generator = new EventTypeGenerator();
131         lrmEventListener = new LRMEventListener();
132         renewLogger = new NormServerBaseImpl.RenewLogThread(renewedList);
133         context = AccessController.getContext();
134     }
135 
136     /**
137      * Returns the exporter to use to export this server.
138      *
139      * @param config the configuration to use for supplying the exporter
140      * @return the exporter to use to export this server
141      * @throws ConfigurationException if a problem occurs retrieving entries
142      *	       from the configuration
143      */
144     Exporter getExporter(Configuration config) throws ConfigurationException {
145         return (Exporter) Config.getNonNullEntry(config, NormServerBaseImpl.NORM,
146 		"serverExporter", Exporter.class,
147 		new BasicJeriExporter(
148 			TcpServerEndpoint.getInstance(0),
149 			new AtomicILFactory(null, null, NormServer.class.getClassLoader())
150 		)
151 	);
152     }
