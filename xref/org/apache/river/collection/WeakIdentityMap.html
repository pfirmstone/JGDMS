19  package org.apache.river.collection;
20  
21  import java.lang.ref.Reference;
22  import java.lang.ref.ReferenceQueue;
23  import java.lang.ref.WeakReference;
24  import java.util.Collection;
25  import java.util.HashMap;
26  import java.util.Map;
27  
28  /**
29   * Identity-based weak hash map.
30   *
31   * @author Sun Microsystems, Inc.
32   *
33   * @since 2.0
34   */
35  public class WeakIdentityMap<K,V> {
36  
37      // REMIND: optimize implementation (clone new java.util.WeakHashMap?)
38  
39      private final Map<Key<K>,V> map = new HashMap<Key<K>,V>();
40      private final ReferenceQueue<K> queue = new ReferenceQueue<K>();
41  
42      /**
43       * Associates value with given key, returning value previously associated
44       * with key, or null if none.
45       * @param key
46       * @param value
47       * @return 
48       */
49      public V put(K key, V value) {
50  	processQueue();
51  	return map.put(Key.create(key, queue), value);
52      }
53  
54      /**
55       * Returns value associated with given key, or null if none.
56       * @param key
57       * @return 
58       */
59      public V get(Object key) {
60  	processQueue();
61  	return map.get(Key.create(key, null));
62      }
63  
64      /**
65       * Removes association for given key, returning value previously associated
66       * with key, or null if none.
67       * @param key
68       * @return 
69       */
70      public V remove(Object key) {
71  	processQueue();
72  	return map.remove(Key.create(key, null));
73      }
74  
75      /**
76       * Returns collection containing all values currently held in this map.
77       * @return 
78       */
79      public Collection<V> values() {
80  	processQueue();
81  	return map.values();
82      }
83  
84      /**
85       * Removes all associations from this map.
86       */
87      public void clear() {
88  	processQueue();
89  	map.clear();
90      }
91  
92      private void processQueue() {
93  	Reference<? extends K> k;
94  	    while ((k = queue.poll()) != null) {
95  	    map.remove(k);
96  	}
97      }
98  
99      private static class Key<T> extends WeakReference<T> {
100 
101 	private final int hash;
102 
103 	static <T> Key<T> create(T k, ReferenceQueue<T> q) {
104 	    if (k == null) {
105 		return null;
106 	    } else if (q == null) {
107 		return new Key<T>(k);
108 	    } else {
109 		return new Key<T>(k, q);
110 	    }
111 	}
112 
113 	private Key(T k) {
114 	    super(k);
115 	    hash = System.identityHashCode(k);
116 	}
117 
118 	private Key(T k, ReferenceQueue<T> q) {
119 	    super(k, q);
120 	    hash = System.identityHashCode(k);
121 	}
122 
123 	public boolean equals(Object o) {
124 	    if (this == o) {
125 		return true;
126 	    } else if (!(o instanceof Key)) {
127 		return false;
128 	    }
129 	    Object k1 = get(), k2 = ((Key) o).get();
130 	    return (k1 != null && k2 != null && k1 == k2);
131 	}
132 
133 	public int hashCode() {
134 	    return hash;
135 	}
136     }
137 }
