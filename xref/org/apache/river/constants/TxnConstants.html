18  package org.apache.river.constants;
19  
20  import net.jini.core.transaction.server.TransactionConstants;
21  
22  /**
23   * Names for constants common to transaction managers and participants.
24   *
25   * @author Sun Microsystems, Inc.
26   *
27   */
28  
29  public class TxnConstants implements TransactionConstants {
30      /** Names of each of the states */
31      private static final String[] stateNames = {
32  	null, "active", "voting", "prepared",
33  	"notchanged", "committed", "aborted"
34      };
35  
36      /**
37       * Returns a <code>String</code> which describes
38       * the state of the <code>Transaction</code>.
39       *
40       * @see net.jini.core.transaction.server.TransactionConstants
41       */
42      public static String getName(int state) {
43  	switch (state) {
44  	  case ACTIVE: return "ACTIVE";
45  	  case VOTING: return "VOTING";
46  	  case PREPARED: return "PREPARED";
47  	  case NOTCHANGED: return "NOTCHANGED";
48  	  case COMMITTED: return "COMMITTED";
49  	  case ABORTED: return "ABORTED";
50            default: return "UNKNOWN";
51  	}
52      }
53  }
