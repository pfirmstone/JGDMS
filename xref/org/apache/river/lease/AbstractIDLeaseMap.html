19  package org.apache.river.lease;

21  import java.util.AbstractMap;
22  import java.util.Set;
23  import java.util.concurrent.ConcurrentHashMap;
24  import java.util.concurrent.ConcurrentMap;
25  import java.util.concurrent.ConcurrentSkipListSet;
26  import net.jini.core.lease.Lease;
27  import net.jini.core.lease.LeaseMap;
28  import net.jini.id.Uuid;
29  import org.apache.river.api.io.AtomicSerial;
30  import org.apache.river.concurrent.RC;
31  import org.apache.river.concurrent.Ref;
32  import org.apache.river.concurrent.Referrer;
33  import org.apache.river.lease.ID;
34  
35  /**
36   * AbstractIDLeaseMap is intended to work around some minor design warts in the 
37   * {@link Lease} interface:
38   * 
39   * In the real world, when a Lease is renewed, a new Lease contract document
40   * is issued, however when an electronic Lease is renewed the Lease expiry
41   * date is changed and the record of the previous Lease is lost.  Ideally the
42   * renew method would return a new Lease.
43   * 
44   * Current Lease implementations rely on a {@link Uuid} to represents the lease,
45   * the expiry date is not included the equals or hashCode calculations.  For this
46   * reason, two Lease objects, one expired and one valid, may be equal, this
47   * is undesirable.
48   * 
49   * The Lease interface doesn't specify a contract for equals or hashCode,
50   * all Lease implementations are also mutable, previous implementations
51   * of {@link LeaseMap} used Leases as keys.
52   * 
53   * AbstractIDLeaseMap uses only the {@link ID}, usually a {@link Uuid}
54   * provided by a Lease for internal map keys, if {@link ID} is not implemented
55   * then the Lease itself is used as the key.
56   * 
57   * Both Lease keys and Long values are actually stored internally as values
58   * referred to by ID keys, allowing Lease implementations to either not override
59   * hashCode and equals object methods or allow implementations that more
60   * accurately model reality.
61   * 
62   * This implementation is thread safe, concurrent and doesn't require external 
63   * synchronization.
64   * 
65   * @param <K> 
66   * @author peter
67   */
68  public abstract class AbstractIDLeaseMap<K extends Lease> extends AbstractMap<K,Long> 
69                                implements LeaseMap<K,Long> {
70      
71      private final ConcurrentMap<Object,K> leaseMap;
72      private final ConcurrentMap<Object,Long> durationMap;
73      private final Set<Entry<K,Long>> set;
74      
75      /**
76       * Constructor for subclasses.
77       */
78      protected AbstractIDLeaseMap(){
79          leaseMap = RC.concurrentMap(
80                  new ConcurrentHashMap<Referrer<Object>,Referrer<K>>(),
81                  Ref.WEAK, Ref.STRONG, 10000, 10000);
82          durationMap = RC.concurrentMap(
83                  new ConcurrentHashMap<Referrer<Object>,Referrer<Long>>(),
84                  Ref.WEAK, Ref.STRONG, 10000, 10000);
85          set = new ConcurrentSkipListSet<Entry<K,Long>>();
86      }
87  
88      @Override
89      public Set<Entry<K,Long>> entrySet() {
90          return set;
91      }
92      
93      void checkKey(Object key) throws IllegalArgumentException {
94          if (!canContainKey(key)) throw new IllegalArgumentException("Key not valid for this LeaseMap");
95      }
96      
97      void checkValue(Object value) throws IllegalArgumentException {
98          if (!(value instanceof Long)) throw new IllegalArgumentException("Value not valid for this LeaseMap, must be Long");
99      }
100     
101     public boolean containsValue(Object value){
102         checkValue(value);
103         return durationMap.containsValue(value);
104     }
105     
106     /**
107      * {@inheritDoc}
108      * 
109      * Determines whether the ID of the key matches the ID of another key
110      * in the map.
111      * 
112      * @param key
113      * @return 
114      */
115     public boolean containsKey(Object key){
116         checkKey(key);
117         return set.contains(new LeaseEntry<K>(getIdentity(key), null, null, null));
118     }
119     
120     public Long get(Object key){
121         checkKey(key);
122         return durationMap.get(getIdentity(key));
123     }
124     
125     /**
126      * {@inheritDoc}
127      * 
128      * This implementation will place a new key value pair association in the map,
129      * or it will replace both the key and the value if an equivalent association
130      * currently exists in the map.
131      * 
132      * @param key
133      * @param value
134      * @return 
135      */
136     public Long put(K key, Long value) {
137         checkKey(key);
138         Object identity = getIdentity(key);
139         LeaseEntry<K> entry = new LeaseEntry<K>(identity, leaseMap, durationMap, set);
140         if (entry.isNew(key, value)){
141             return null;
142         } else { // existing identity, replace Lease and duration.
143             leaseMap.replace(identity, key);
144             return durationMap.replace(identity, value);
145         }
146     }
147     
148     public Long remove(Object key){
149         checkKey(key);
150         LeaseEntry<K> entry = new LeaseEntry<K>(getIdentity(key), null, durationMap, null);
151         if (set.remove(entry)) return entry.getValue();
152         return null;
153     }
154     
155     private Object getIdentity(Object key){
156         if (key instanceof ID) {
157                 return((ID) key).identity();
158             } else {
159                 return key;
160             } // Allows for support of legacy lease implementations where equals is based on Uuid.
161     }
162 
163     /**
164      * The logic behind this Entry is that the identity which maintains strong
165      * references to the key and value will not be added to the set or leaseMap
166      * if it's already present.
167      */
168     private static class LeaseEntry<K extends Lease> implements Entry<K,Long>, Comparable<LeaseEntry<K>> {
169         
170         private final ConcurrentMap<Object,K> leaseMap;
171         private final ConcurrentMap<Object,Long> durationMap;
172         private final Set<Entry<K,Long>> set;
173         private final Object identity;
174         private volatile K key;
175         private volatile Long value;
176         
177                 
178         
179         LeaseEntry(Object identity, ConcurrentMap<Object,K> leaseMap, ConcurrentMap<Object,Long> durationMap, Set<Entry<K,Long>> set){
180             if (identity == null) throw new NullPointerException("Identity cannot be null");
181             this.set = set;
182             this.leaseMap = leaseMap;
183             this.durationMap = durationMap;
184             this.identity = identity;
185         }
186         
187         boolean isNew(K key, Long value){
188             if (set.add(this)){
189                 Object exists = leaseMap.putIfAbsent(identity, key);
190                 Object valExists = durationMap.putIfAbsent(identity, value);
191                 // If exists, there's a problem if identity is not the same object
192                 // we are forced to remove it, because key and value may not be 
193                 // strongly referenced by our identity and risk being garbage collected.
194                 if (exists != null || valExists != null){
195                     this.key = key;
196                     this.value = value;
197                     try {
198                         leaseMap.remove(identity);
199                         durationMap.remove(identity);
200                         leaseMap.put(identity, key);
201                         durationMap.put(identity, value);
202                     } finally {
203                         this.key = null;
204                         this.value = null;
205                     }
206                 }
207                 return true;
208             }
209             return false;
210         }
211 
212         @Override
213         public K getKey() {
214             K key = this.key;
215             if (key != null) return key;
216             return leaseMap.get(identity);
217         }
218 
219         @Override
220         public Long getValue() {
221             Long value = this.value;
222             if (value != null) return value;
223             return durationMap.get(identity);
224         }
225 
226         @Override
227         public Long setValue(Long value) {
228             return durationMap.replace(identity, value);
229         }
230 
231         @Override
232         public int hashCode() {
233             return identity.hashCode();
234         }
235         
236         public boolean equals(Object o){
237             if (!(o instanceof LeaseEntry)) return false;
238             LeaseEntry that = (LeaseEntry) o;
239             return this.identity.equals(that.identity);
240         }
241 
242         public int compareTo(LeaseEntry<K> o) {
243             if (identity instanceof Uuid && o.identity instanceof Uuid){
244                 long mine = ((Uuid) identity).getLeastSignificantBits();
245                 long his = ((Uuid) o.identity).getLeastSignificantBits();
246                 if ( mine < his) return -1;
247                 if ( mine > his) return 1;
248                 if ( mine == his){
249                     mine = ((Uuid) identity).getMostSignificantBits();
250                     his = ((Uuid) o.identity).getMostSignificantBits();
251                     if ( mine < his) return -1;
252                     if ( mine > his) return 1;
253                     return 0;
254                 }
255             }
256             int myHash = hashCode();
257             int hisHash = o.hashCode();
258             if (myHash < hisHash) return -1;
259             if (myHash > hisHash) return 1;
260             return 0;
261         }
262     
263     }
