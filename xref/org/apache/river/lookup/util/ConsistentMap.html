19  package org.apache.river.lookup.util;
20  
21  import java.io.IOException;
22  import java.io.Serializable;
23  import java.util.AbstractMap;
24  import java.util.Collections;
25  import java.util.HashSet;
26  import java.util.Iterator;
27  import java.util.Map;
28  import java.util.Set;
29  import org.apache.river.api.io.AtomicSerial;
30  import org.apache.river.api.io.AtomicSerial.GetArg;
31  
32  /**
33   * An implementation of the <code>java.util.Map</code> interface that has
34   * a serialized form consistent in all virtual machines. <code>ConsistentMap</code>
35   * instances are unmodifiable. All mutator methods, such as <code>add</code> and
36   * <code>remove</code>, throw <code>UnsupportedOperationException</code>.
37   * This class permits <code>null</code> values and the <code>null</code> key.
38   *
39   * <p>
40   * Although instances of this class are unmodifiable, they are not necessarily
41   * immutable. If a client retrieves a mutable object (either a key or value) contained in a
42   * <code>ConsistentMap</code> and mutates that object, the client in effect
43   * mutates the state of the <code>ConsistentMap</code>. In this case, the
44   * serialized form of the <code>ConsistentMap</code> will most likely also
45   * have been mutated. A <code>ConsistentMap</code> that contains only immutable
46   * objects will maintain a consistent serialized form indefinitely. But a 
47   * <code>ConsistentMap</code> that contains mutable objects will maintain a
48   * consistent serialized form only so long as the mutable objects are not
49   * mutated.
50   *
51   * @author Bill Venners
52   */
53  @AtomicSerial
54  public class ConsistentMap<K,V> extends AbstractMap<K,V> implements Serializable {
55  
56      private static final long serialVersionUID = -5223157327307155247L;
57  
58      /**
59       * @serial A <code>Set</code> of <code>java.util.Map.Entry</code> objects,
60       *     the key-value pairs contained in this <code>ConsistentMap</code>.
61       */
62      private final Set<Map.Entry<K,V>> entrySet;
63  
64      /**
65       * Constructs a new, empty <code>ConsistentMap</code>. All instances
66       * of <code>ConsistentMap</code> are unmodifiable.
67       */
68      public ConsistentMap() {
69          entrySet = new ConsistentSet<Map.Entry<K,V>>(new HashSet<Map.Entry<K,V>>());
70      }
71  
72      /**
73       * Constructs a new <code>ConsistentMap</code> containing the elements
74       * in the passed collection. All instances of <code>ConsistentMap</code>
75       * are unmodifiable.
76       *
77       * @param init the map whose elements are to be placed into this map.
78       * @throws  NullPointerException if the passed <code>init</code> reference
79       *     is <code>null</code>
80       */
81      public ConsistentMap(Map<K,V> init) {
82  	this(entrySet(init));
83      }
84  
85      public ConsistentMap(GetArg arg) throws IOException {
86  	this(entrySet((Map<K, V>) arg.get("entrySet", null)));
87      }
88      
89      private static <K,V> Set<Map.Entry<K,V>> entrySet(Map<K,V> init){
90          if (init == null) {
91              throw new NullPointerException();
92          }
93  
94          // Must put the key-value pairs into ConsistentMapEntry objects,
95          // so they'll behave correctly when setValue() is invoked on them.
96          Set<Map.Entry<K,V>> unmodEntries = new HashSet<Map.Entry<K,V>>(init.size());
97          Set<Map.Entry<K,V>> entries = init.entrySet();
98          Iterator<Map.Entry<K,V>> it = entries.iterator();
99          while (it.hasNext()) {
100 
101             Map.Entry<K,V> entry = it.next();
102             Map.Entry<K,V> unmodEntry = new ConsistentMapEntry<K,V>(entry.getKey(), entry.getValue());
103             unmodEntries.add(unmodEntry);
104         }
105         return new ConsistentSet<Map.Entry<K,V>>(unmodEntries);
106     }
107 
108     private ConsistentMap(Set<Map.Entry<K,V>> set){
109 	entrySet = set;
110     }
111 
112     /**
113      * Returns a set view of the mappings contained in this
114      * <code>ConsistentMap</code>. Each element in the returned
115      * set is a <code>Map.Entry</code>
116      *
117      * @return a set view of the mappings contained in this
118      *     <code>ConsistentMap</code>.
119      */
120     @Override
121     public Set<Map.Entry<K,V>> entrySet() {
122         return Collections.unmodifiableSet(entrySet);
123     }
124 }
