20  import java.rmi.MarshalledObject;
21  import java.rmi.Remote;
22  import java.rmi.RemoteException;
23  import net.jini.admin.Administrable;
24  import net.jini.admin.JoinAdmin;
25  import net.jini.core.discovery.LookupLocator;
26  import net.jini.core.event.EventRegistration;
27  import net.jini.core.event.RemoteEventListener;
28  import net.jini.core.lease.UnknownLeaseException;
29  import net.jini.core.lookup.ServiceID;
30  import net.jini.core.lookup.ServiceRegistration;
31  import net.jini.lookup.ServiceProxyAccessor;
32  import net.jini.id.Uuid;
33  import net.jini.io.MarshalledInstance;
34  import net.jini.lookup.DiscoveryAdmin;
35  import org.apache.river.admin.DestroyAdmin;
36  import org.apache.river.proxy.MarshalledWrapper;

38  /**
39   * Registrar defines the private protocol between the various client-side
40   * proxies and the registrar server.
41   * <p>
42   * The declared methods are pretty straightforward transformations of the
43   * ServiceRegistrar and ServiceRegistration interfaces, with external classes
44   * (ServiceItem, ServiceTemplate, ServiceMatches, Entry) converted to internal
45   * classes (Item, Template, Matches, EntryRep).  In addition, there are
46   * methods for transformed Lease and LeaseMap interfaces, for service and
47   * event leases.
48   *
49   * @author Sun Microsystems, Inc.
50   *
51   */
52  public interface Registrar 
53      extends Remote, ServiceProxyAccessor, Administrable,
54  	    DiscoveryAdmin, JoinAdmin, DestroyAdmin
55  {
56      /**
57       * Register a new service or re-register an existing service.
58       * @see net.jini.core.lookup.ServiceRegistrar#register
59       */
60      ServiceRegistration register(Item item, long leaseDuration)
61  	throws RemoteException;
62  
63      /**
64       * Returns the service object (i.e., just ServiceItem.service) from an
65       * item matching the template, or null if there is no match.
66       * @see net.jini.core.lookup.ServiceRegistrar#lookup
67       * @param tmpl Template used for matching service items.
68       * @return MarshalledWrapper containing service object
69       * @throws RemoteException if connection related exception occurs.
70       */
71      MarshalledWrapper lookup(Template tmpl) throws RemoteException;
72  
73      /**
74       * Returns at most maxMatches items matching the template, plus the total
75       * number of items that match the template.
76       * @see net.jini.core.lookup.ServiceRegistrar#lookup
77       */
78      Matches lookup(Template tmpl, int maxMatches) throws RemoteException;
79      
80      Object [] lookUp(Template tmpl, int maxProxys) throws RemoteException;
81  
82      /**
83       * Registers for event notification.
84       * @see net.jini.core.lookup.ServiceRegistrar#notify
85       */
86      EventRegistration notify(Template tmpl,
87  			     int transitions,
88  			     RemoteEventListener listener,
89  			     MarshalledObject handback,
90  			     long leaseDuration)
91  	throws RemoteException;
92      
93      /**
94       * Registers for event notification.
95       * @see net.jini.core.lookup.ServiceRegistrar#notify
96       */
97      EventRegistration notiFy(Template tmpl,
98  			     int transitions,
99  			     RemoteEventListener listener,
100 			     MarshalledInstance handback,
101 			     long leaseDuration)
102 	throws RemoteException;
103 
104     /**
105      * Looks at all service items that match the specified template, finds
106      * every entry (among those service items) that either doesn't match any
107      * entry templates or is a subclass of at least one matching entry
108      * template, and returns the set of the (most specific) classes of those
109      * entries.
110      * @see net.jini.core.lookup.ServiceRegistrar#getEntryClasses
111      */
112     EntryClassBase[] getEntryClasses(Template tmpl) throws RemoteException;
113 
114     /**
115      * Looks at all service items that match the specified template, finds
116      * every entry (among those service items) that matches
117      * tmpl.attributeSetTemplates[setIndex], and returns the set of values
118      * of the specified field of those entries.
119      * The field name has been converted to an index (fields numbered
120      * from super to subclass).
121      *
122      * @see net.jini.core.lookup.ServiceRegistrar#getFieldValues
123      * @param tmpl template to match service items
124      * @param setIndex index in the template of attributes
125      * @param field field index.
126      * @return Object array
127      * @throws RemoteException if a connection related exception occurs.
128      */
129     Object[] getFieldValues(Template tmpl, int setIndex, int field)
130 	throws RemoteException;
131 
132     /**
133      * Looks at all service items that match the specified template, and for
134      * every service item finds the most specific type (class or interface)
135      * or types the service item is an instance of that are neither equal to,
136      * nor a superclass of, any of the service types in the template and that
137      * have names that start with the specified prefix, and returns the set
138      * of all such types.
139      * @see net.jini.core.lookup.ServiceRegistrar#getServiceTypes
140      * @param tmpl template to match service items.
141      * @param prefix to match start of names
142      * @return ServiceTypeBase array
143      * @throws RemoteException if connection related exception occurs.
144      */
145     ServiceTypeBase[] getServiceTypes(Template tmpl, String prefix)
146 	throws RemoteException;
147 
148     /**
149      * Returns a LookupLocator that can be used if necessary for unicast
150      * discovery of the lookup service.
151      * @see net.jini.core.lookup.ServiceRegistrar#getLocator
152      * @return LookupLocator for lookup service
153      * @throws RemoteException if connection related exception occurs.
154      */
155     LookupLocator getLocator() throws RemoteException;
156 
157     /**
158      * Adds the specified attribute sets (those that aren't duplicates of
159      * existing attribute sets) to the registered service item.
160      * @see net.jini.core.lookup.ServiceRegistration#addAttributes
161      * @param serviceID the serviceID used to identify the service
162      * @param leaseID the Uuid of the lease.
163      * @param attrSets the array of EntryRep's.
164      * @throws UnknownLeaseException if Lease expired.
165      * @throws RemoteException if a connection related problem occurs.
166      */
167     void addAttributes(ServiceID serviceID, Uuid leaseID, EntryRep[] attrSets)
168 	throws UnknownLeaseException, RemoteException;
169 
170     /**
171      * Modifies existing attribute sets of a registered service item.
172      * @see net.jini.core.lookup.ServiceRegistration#modifyAttributes
173      */
174     void modifyAttributes(ServiceID serviceID,
175 			  Uuid leaseID,
176 			  EntryRep[] attrSetTmpls,
177 			  EntryRep[] attrSets)
178 	throws UnknownLeaseException, RemoteException;
179 
180     /**
181      * Deletes all of the service item's existing attributes, and replaces
182      * them with the specified attribute sets.
183      * @see net.jini.core.lookup.ServiceRegistration#setAttributes
184      */
185     void setAttributes(ServiceID serviceID, Uuid leaseID, EntryRep[] attrSets)
186 	throws UnknownLeaseException, RemoteException;
187 
188     /**
189      * Cancels a service lease.
190      * @see net.jini.core.lease.Lease#cancel
191      * @param serviceID ServiceID
192      * @param leaseID Uuid of lease.
193      * @throws UnknownLeaseException if leaseID is unknown.
194      * @throws RemoteException of a connection related exception occurs.
195      */
196     void cancelServiceLease(ServiceID serviceID, Uuid leaseID)
197 	throws UnknownLeaseException, RemoteException;
198 
199     /**
200      * Renews a service lease.
201      * @see net.jini.core.lease.Lease#renew
202      */
203     long renewServiceLease(ServiceID serviceID, Uuid leaseID, long duration)
204 	throws UnknownLeaseException, RemoteException;
205 
206     /**
207      * Cancels an event lease.
208      * @see net.jini.core.lease.Lease#cancel
209      * @param eventID the event id.
210      * @param leaseID Uuid of the lease.
211      * @throws UnknownLeaseException if leaseID unknown.
212      * @throws RemoteException if connection related exception occurs.
213      */
214     void cancelEventLease(long eventID, Uuid leaseID)
215 	throws UnknownLeaseException, RemoteException;
216 
217     /**
218      * Renews an event lease.
219      * @see net.jini.core.lease.Lease#renew
220      */
221     long renewEventLease(long eventID, Uuid leaseID, long duration)
222 	throws UnknownLeaseException, RemoteException;
223 
224     /**
225      * Renews service and event leases from a LeaseMap.
226      * @see net.jini.core.lease.LeaseMap#renewAll
227      */
228     RenewResults renewLeases(Object[] regIDs,
229 			     Uuid[] leaseIDs,
230 			     long[] durations)
231 	throws RemoteException;
232 
233     /**
234      * Cancels service and event leases from a LeaseMap.
235      * @see net.jini.core.lease.LeaseMap#cancelAll
236      * @param regIDs array of service registration id's.
237      * @param leaseIDs array of lease Uuid's
238      * @return array containing Exceptions
239      * @throws RemoteException if a connection related exception occurs.
240      */
241     Exception[] cancelLeases(Object[] regIDs, Uuid[] leaseIDs)
242 	throws RemoteException;
243 }
