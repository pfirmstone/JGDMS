62  /**
63   * A RegistrarProxy is a proxy for a registrar.  Clients only see instances
64   * via the ServiceRegistrar, Administrable and ReferentUuid interfaces.
65   *
66   * @author Sun Microsystems, Inc.
67   *
68   */
69  @AtomicSerial
70  public class RegistrarProxy 
71      implements ServiceRegistrar, SafeServiceRegistrar, ProxyAccessor, Administrable, ReferentUuid, Serializable
72  {
73      private static final long serialVersionUID = 2L;
74  
75      private static final Logger logger = 
76  	Logger.getLogger("org.apache.river.reggie");

78      /**
79       * The registrar.
80       *
81       * @serial
82       */
83      final Registrar server;

85       * The registrar's service ID.
86       */
87      transient ServiceID registrarID;

89      /**
90       * Returns RegistrarProxy or ConstrainableRegistrarProxy instance,
91       * depending on whether given server implements RemoteMethodControl.
92       */
93      public static RegistrarProxy getInstance(Registrar server,
94  				      ServiceID registrarID)
95      {
96  	return (server instanceof RemoteMethodControl) ?
97  	    new ConstrainableRegistrarProxy(server, registrarID, null) :
98  	    new RegistrarProxy(server, registrarID);
99      }

106     private static boolean check(GetArg arg) throws IOException{
107 	Registrar server = (Registrar) arg.get("server", null);
108 	if (server == null) throw new InvalidObjectException("null server");
109 	RO r = (RO) arg.getReader();
110 	if (r.registrarID == null) throw new InvalidObjectException("null ServiceID");
111 	return true;
112     }
113     
114     RegistrarProxy(GetArg arg) throws IOException{
115 	this(arg, check(arg));
116     }
117     
118     RegistrarProxy(GetArg arg, boolean check) throws IOException{
119 	server = (Registrar) arg.get("server", null);
120 	RO r = (RO) arg.getReader();
121 	registrarID = r.registrarID;
122     }

151     @Override
152     public Object lookup(ServiceTemplate tmpl) throws RemoteException {
153 	MarshalledWrapper wrapper = server.lookup(new Template(tmpl));
154 	if (wrapper == null)
155 	    return null;
156 	try {
157 	    return wrapper.get();
158 	} catch (IOException e) {
159 	    throw new UnmarshalException("error unmarshalling return", e);
160 	} catch (ClassNotFoundException e) {
161 	    throw new UnmarshalException("error unmarshalling return", e);
162 	}
163     }

173     @Override
174     public Object [] lookUp(
175 	    ServiceTemplate tmpl, int maxProxies) throws RemoteException
176     {
177 	Object [] proxys = server.lookUp(new Template(tmpl), maxProxies);
178 	List result = new ArrayList(proxys.length);
179 	for (int i = 0, l = proxys.length; i < l; i++){
180 	    if(!(proxys[i] instanceof RemoteMethodControl)) continue;
181 	    if(!(proxys[i] instanceof TrustEquivalence)) continue;
182 	    if(!(proxys[i] instanceof ServiceProxyAccessor)) continue;
183 	    if(!(proxys[i] instanceof ServiceAttributesAccessor)) continue;
184 	    if(!Proxy.isProxyClass(proxys[i].getClass())) continue;
185 	    result.add(proxys[i]);
186 	}
187 	return result.toArray();
188     }

225     // Inherit javadoc
226     @Override
227     public Object[] getFieldValues(ServiceTemplate tmpl,
228 				   int setIndex, String field)
229 	throws NoSuchFieldException, RemoteException
230     {
231 	/* check that setIndex and field are valid, convert field to index */
232 	ClassMapper.EntryField[] efields =
233 	    ClassMapper.getFields(
234 			     tmpl.attributeSetTemplates[setIndex].getClass());
235 	int fidx;
236 	for (fidx = efields.length; --fidx >= 0; ) {
237 	    if (field.equals(efields[fidx].field.getName()))
238 		break;
239 	}
240 	if (fidx < 0)
241 	    throw new NoSuchFieldException(field);
242 	Object[] values = server.getFieldValues(new Template(tmpl),
243 						setIndex, fidx);
244 	/* unmarshal each value, replacing with null on exception */
245 	if (values != null && efields[fidx].marshal) {
246 	    for (int i = values.length; --i >= 0; ) {
247 		try {
248 		    values[i] = ((MarshalledWrapper) values[i]).get();
249 		    continue;
250 		} catch (Throwable e) {
251 		    handleException(e);
252 		}
253 		values[i] = null;
254 	    }
255 	}
256 	return values;
257     }

259     /**
260      * Rethrow the exception if it is an Error, unless it is a LinkageError,
261      * OutOfMemoryError, or StackOverflowError.  Otherwise print the
262      * exception stack trace if debugging is enabled.
263      */
264     static void handleException(final Throwable e) {
265 	if (e instanceof Error &&
266 	    !(e instanceof LinkageError ||
267 	      e instanceof OutOfMemoryError ||
268 	      e instanceof StackOverflowError))
269 	{
270 	    throw (Error)e;
271 	}
272 	logger.log(Level.INFO, "unmarshalling failure", e);
273     }

333     /**
334      * Writes the default serializable field value for this instance, followed
335      * by the registrar's service ID encoded as specified by the
336      * ServiceID.writeBytes method.
337      */
338     private void writeObject(ObjectOutputStream out) throws IOException {
339 	out.defaultWriteObject();
340 	registrarID.writeBytes(out);
341     }
342 
343     /**
344      * Reads the default serializable field value for this instance, followed
345      * by the registrar's service ID encoded as specified by the
346      * ServiceID.writeBytes method.  Verifies that the deserialized registrar
347      * reference is non-null.
348      */
349     private void readObject(ObjectInputStream in)
350 	throws IOException, ClassNotFoundException
351     {
352 	in.defaultReadObject();
353 	registrarID = new ServiceID(in);
354 	if (server == null) {
355 	    throw new InvalidObjectException("null server");
356 	}
357     }

364     }
