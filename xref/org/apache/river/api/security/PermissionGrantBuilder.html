32  /**
33   * The PermissionGrantBuilder creates Dynamic PermissionGrant's based on
34   * information provided by the user.  The user must have access to the
35   * system policy and have permission to grant permissions.
36   * 
37   * A PermissionGrantBuilder implementation should also be used as the serialized form
38   * for PermissionGrant's, the implementation of PermissionGrant's should
39   * remain package private.
40   * 
41   * This prevents the serialized form becoming part of the public api.
42   * 
43   * Single Thread use only.
44   * @author Peter Firmstone.
45   * @since 3.0.0
46   * @see PermissionGrant
47   */
48  public abstract class PermissionGrantBuilder {
49     
50      /**
51       * The PermissionGrant generated will apply to all classes loaded by
52       * the ClassLoader
53       */ 
54      public static final int CLASSLOADER = 0;
55    
56      /**
57       * The PermissionGrant generated will apply to all classes belonging to
58       * the ProtectionDomain.  This is actually a simplification for the 
59       * programmer the PermissionGrant will apply to the CodeSource and the
60       * ClassLoader combination, the reason for this is the DomainCombiner may
61       * create new instances of ProtectionDomain's from those that exist on
62       * the stack.
63       * <p>
64       * DNS is not consulted, the RFC3986 normalized URI and all Certificates contained
65       * by the CodeSources must be equal.
66       * 
67       * @see java.security.AccessControlContext
68       * @see java.security.DomainCombiner
69       * @see javax.security.auth.SubjectDomainCombiner
70       */
71      public static final int PROTECTIONDOMAIN = 2;
72      /**
73       * The PermissionGrant generated will apply to the Subject that has 
74       * all the principals provided.
75       * 
76       * @see Subject
77       */
78      public static final int PRINCIPAL = 3;
79      /**
80       * The PermissionGrant generated will apply to all classes loaded from
81       * CodeSource's that have at a minimum the defined array Certificate[]
82       * provided the logged in Subject also has all Principals when defined.
83       */
84      public static final int CODESOURCE_CERTS = 4;
85      
86      /**
87       * <p>
88       * The PermissionGrant generated will imply the ProtectionDomain run as
89       * a Subject with all Principals (when applicable) and
90       * CodeSource that has the Certificates and URI RFC3986 location as specified.
91       * </p>
92       * <p>
93       * The outcome of URI comparison is similar to 
94       * {@link CodeSource#implies(CodeSource)}.</p>
95       * <p>
96       * DNS lookup is avoided for security and performance reasons,
97       * DNS is not authenticated and therefore cannot be trusted.  Doing so,
98       * could allow an attacker to use DNS Cache poisoning to escalate
99       * Permission, by imitating a URL with privilege, such as AllPermission.</p>
100      * <p>
101      * CodeSource URL are converted to URI and normalized according
102      * to <a href="http://www.ietf.org/rfc/rfc3986.txt">RFC3986</a> before being 
103      * compared as Strings.</p>
104      * <p>
105      * A URI based PermissionGrant "implies" a specified {@link ProtectionDomain} if:
106      * </P>
107      * <ol>
108      * <li> The {@link ProtectionDomain}'s <i>{@link CodeSource}</i> is not null.
109      * <li> All {@link Principal}'s, if defined in the {@link PermissionGrant} are present in the 
110      *      {@link ProtectionDomain}, when run as a {@link Subject}, in any order.
111      * <li> A URI based PermissionGrant "implies" a specified {@link ProtectionDomain} 
112      *      or non null {@link CodeSource} if:
113      * <ol>
114      * <li> All {@link Certificate}'s included in a URI based 
115      * PermissionGrant are present in that {@link ProtectionDomain}'s 
116      * <i>{@link CodeSource}</i>'s certificates, in any order, or no Certificates
117      *    are defined by the PermissionGrant.
118      * <li> For any {@link URI} in a PermissionGrant, checks are made in the 
119      * following order:
120      *   <ol>
121      *     <li>  Any null {@link URI} implies any 
122      *           {@link ProtectionDomain} that contains a non null
123      *           {@link CodeSource}, including a null {@link URL} returned by
124      *           {@link CodeSource#getLocation()}.  
125      *
126      *     <li>  If any RFC3986 normalized URI equals the {@link ProtectionDomain}'s 
127      *           <i>{@link CodeSource#getLocation()}</i>'s {@link URL}
128      *           after it is converted to a RFC3986 normalized {@link URI}, 
129      *           the {@link PermissionGrant#implies(java.security.ProtectionDomain)}
130      *           method will return true.
131      * 
132      *     <li>  The {@link CodeSource#getLocation()}'s {@link URL} is checked
133      *           against each {@link URI} contained in a PermissionGrant and
134      *           returns true if all the following conditions are met for at least
135      *           one {@link URI}:
136      *     <ol>
137      *     <li>  The {@link URI#getScheme()} scheme must be
138      *           equal to a {@link CodeSource}'s {@link URL#getProtocol()}
139      *           protocol, after normalization to RFC3986 rules.
140      *
141      *     <li>  If {@link URI#getHost()} is non null,  
142      *           and {@link URL#getHost()} is equal after RFC3986 compliant normalization
143      *           performed.
144      *
145      *     <li>  If the {@link URI#getPort()} port is not 
146      *           equal to -1 (that is, if a port is specified), it must equal the 
147      *           CodeSource URL's port.
148      *
149      *     <li>  URI and URL path's are normalized to RFC3986, in addition file: scheme
150      *           paths are normalized to upper case, on platforms with a backslash
151      *           path separator.
152      *     <li>  After normalization, if this {@link URI#getPath()} path doesn't equal
153      *           <i>codesource</i>'s {@link URL#getPath()} path, then the following checks are made:
154      *           If this URI's path ends with "/-",
155      *           then <i>codesource</i>'s URL path must start with this URI's
156      *           path (exclusive the trailing "-").
157      *           If this URI's path ends with a "/*",
158      *           then <i>codesource</i>'s URL path must start with this URI's
159      *           path and must not have any further "/" separators.
160      *           If this URI's path doesn't end with a "/", 
161      *           then <i>codesource</i>'s URL path must match this URI's 
162      *           path with a '/' appended.
163      *
164      *     <li>  If this {@link URI#getFragment()} fragment is 
165      *           not null, it must equal <i>codesource</i>'s 
166      *           {@link URL#getRef()} reference.
167      * 
168      *     <li>  Unlike {@link CodeSource#implies(java.security.CodeSource) }
169      *           {@link URI#getQuery()} query is not appended to the path because
170      *           normalization to specific platforms is undefined.  It appears
171      *           that {@link URL} was developed prior to RFC2396 and as such
172      *           the {@link URL#getFile()} included the query component, later
173      *           in Java 1.3 the {@link URL#getPath() } method was added, however
174      *           earlier developed classes like {@link URLClassLoader} continued
175      *           to use {@link URL#getFile() } and append this with path
176      *           separators and wild cards after the query, if it existed.  In 
177      *           any case, Certificate and Principal are more prudent identifiers for 
178      *           privileges.
179      *      </ol>
180      *   </ol>
181      * </ol>
182      * </ol>
183      */
184     public static final int URI = 5;
185     
186     public static PermissionGrantBuilder newBuilder(){
187         return new PermissionGrantBuilderImp();
188     }
189     
190     /**
191      * resets the state for reuse, identical to a newly created 
192      * PermissionGrantBuilder, this step must be performed to avoid unintentional
193      * grants to previously added URI.
194      * @return PermissionGrantBuilder
195      */
196     public abstract PermissionGrantBuilder reset();
197    
198     /**
199      * Sets the context of the PermissionGrant to on of the static final 
200      * fields in this class.
201      * 
202      * @param context value of a static field defined in this builder.
203      * @return PermissionGrantBuilder
204      * @throws IllegalStateException if context out of range of static fields 
205      * defined in this builder.
206      */
207     public abstract PermissionGrantBuilder context(int context) throws IllegalStateException;
208     
209     /**
210      * The URI will be added to the PermissionGrant, multiple may be specified by
211      * calling multiple times.
212      * 
213      * @param path - RFC3986 compliant URI or null.
214      * @return PermissinoGrantBuilder
215      */
216     public abstract PermissionGrantBuilder uri(String path);
217     /**
218      * Extracts ProtectionDomain
219      * from the Class for use in the PermissionGrantBuilder.  The ClassLoader
220      * and ProtectionDomain are weakly referenced, when collected any 
221      * created PermissionGrant affected will be voided.
222      * @param cl Class used to determine the ProtectionDomain to be
223      * used for PermissionGrant.
224      * @return PermissionGrantBuilder.
225      */
226     public abstract PermissionGrantBuilder clazz(Class cl);
227     /**
228      * Sets the Certificate[] a CodeSource must have to receive the PermissionGrant.
229      * @param certs Certificates
230      * @return a PermissionGrantBuilder
231      */
232     public abstract PermissionGrantBuilder certificates(Certificate[] certs);
233     
234     /**
235      * Sets the Certificate[] a CodeSource must have to receive the PermissionGrant.
236      * 
237      * @param certs Certificates
238      * @param aliases of certificates.
239      * @return a PermissionGrantBuilder
240      */
241     public abstract PermissionGrantBuilder certificates(Certificate[] certs, String[] aliases);
242     /**
243      * Sets the Principal[] that a Subject must have to be entitled to receive
244      * the PermissionGrant.
245      * 
246      * @param pals Principals.
247      * @return a PermissionGrantBuilder
248      */
249     public abstract PermissionGrantBuilder principals(Principal[] pals);
250     /**
251      * Specifies Permission's to be granted.
252      * @param perm Permissions.
253      * @return a PermissionGrantBuilder
254      */
255     public abstract PermissionGrantBuilder permissions(Permission[] perm);
256     /**
257      * An Exclusion specifically excludes some code from receiving a 
258      * PermissionGrant.  This may be to avoid a known security vulnerability,
259      * where code that we don't have control over allows a reference to
260      * escape without performing adequate security checks.
261      * 
262      * EG: I trust code signed by XXX, but they have a security vulnerability
263      * in xxx.jar
264      * 
265      * A better implementation would be to use a deny policy, where exclusions
266      * are checked before grants are checked.
267      * 
268      * In the default implementation, this doesn't apply to Principal only 
269      * grants, only Certificate and ClassLoader based grants.
270      * @param e
271      * @return 
272      */
273     //public abstract PermissionGrantBuilder exclude(Exclusion e);
274     /**
275      * Build the PermissionGrant using information supplied.
276      * @return an appropriate PermissionGrant.
277      */
278     public abstract PermissionGrant build();
279 
280     /**
281      * 
282      * 
283      * @param domain WeakReference containing ProtectionDomain
284      * @return a PermissionGrantBuilder
285      */
286     public abstract PermissionGrantBuilder setDomain(WeakReference<ProtectionDomain> domain);
