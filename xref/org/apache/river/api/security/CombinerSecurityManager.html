57  /**
58   * CombinerSecurityManager, is intended to be a highly scalable
59   * SecurityManager implementation that caches the results of security checks
60   * for each context, which may be an instance of SecurityContext or
61   * AccessControlContext.  Stale records are pruned from the cache.
62   * 
63   * This SecurityManager should be tuned for garbage collection for a large
64   * young generation heap, since many young objects are created and discarded.
65   * 
66   * It is recommended that this SecurityManager be installed from the command
67   * line in order to load as early as possible.
68   * 
69   * Apart from Permission objects and java.security.Policy.getPolicy() class
70   * lock (Bug ID: 7093090 fixed in jdk8(b15), this SecurityManager is non
71   * blocking, including the cache it keeps to prevent repeat security checks.
72   * 
73   * Note: this security manager is not compatible with the jvm option
74   * -Djava.security.manager=
75   * 
76   * @see Security
77   * @see SecurityContext
78   * @see AccessControlContext
79   * 
80   * @author Peter Firmstone
81   * @since 3.0.0
82   */
83  public class CombinerSecurityManager 
84  extends SecurityManager implements CachingSecurityManager {

89       * Logger is lazily loaded, the SecurityManager can be loaded prior to
90       * the system ClassLoader, attempting to load a Logger will cause a 
91       * NullPointerException, when calling ClassLoader.getSystemClassLoader(),
92       * in order to load the logger class.
93       * 
94       * Note, that since we call Security.getContext(), it too must lazily load
95       * its loggers.
96       * 
97       * @return the logger
98       */
99      private static Logger getLogger() {
100         synchronized (loggerLock){
101             if (logger != null) return logger;
102             logger = Logger.getLogger(CombinerSecurityManager.class.getName());
103             return logger;
104         }

136     private CombinerSecurityManager(boolean check){
137         super(); //checked the permission to create a SecurityManager.
138 	/* Get the policy & refresh, in case it hasn't been initialized. 
139          * While there is no SecurityManager,
140          * no policy checks are performed, however the policy must be in a
141          * constructed working state, before the SecurityManager is set,
142          * otherwise the Policy, if it's a non jvm policy, won't have permission
143          * to read the properties and files it needs in order to be constructed.
144 	 * 
145 	 * Note: we don't want the sun policy provider by default because it
146 	 *	 cannot parse httpmd policy grants that don't have a message
147 	 *	 digest, only use it if specifically requested.
148          */
149 	String policy_provider = System.getProperty("policy.provider");
150 	if (policy_provider == null || "sun.security.provider.PolicyFile".equals(policy_provider) )
151 	{
152 	    try {
153 		Policy.setPolicy(new ConcurrentPolicyFile());
154 	    } catch (PolicyInitializationException ex) {
155 		throw new ExceptionInInitializerError(ex);
156 	    }
157 	}
158 //        Policy policy = java.security.Policy.getPolicy(); 
159 	// Get context before this becomes a SecurityManager.
160         SMConstructorContext = AccessController.getContext();
161 //        SMSecurityContext = Security.getContext(); // Force class loading of Security.
162         ProtectionDomain [] context = new ProtectionDomain[1];
163         privilegedDomain = this.getClass().getProtectionDomain();
164         context[0] = privilegedDomain;
165         SMPrivilegedContext = new AccessControlContext(context);
166         dc = new DelegateDomainCombiner();
167         ConcurrentMap<Referrer<AccessControlContext>, 
168                 Referrer<AccessControlContext>> internal = 
169                 new ConcurrentHashMap<Referrer<AccessControlContext>, 
170                 Referrer<AccessControlContext>>();
171         contextCache = RC.concurrentMap(internal, Ref.TIME, Ref.STRONG, 60000L, 60000L);
172         ConcurrentMap<Referrer<Object>, Referrer<NavigableSet<Permission>>> refmap 
173                 = new ConcurrentHashMap<Referrer<Object>, 
174                 Referrer<NavigableSet<Permission>>>();
175         checked = RC.concurrentMap(refmap, Ref.TIME, Ref.STRONG, 20000L, 20000L);
176         g = new SecurityPermission("getPolicy");
177         action = new Action();
178         // Make this a tunable property.
179         double blocking_coefficient = 0.6; // 0 CPU intensive to 0.9 IO intensive
180         int numberOfCores = Runtime.getRuntime().availableProcessors();
181         int poolSizeLimit = (int) (numberOfCores / ( 1 - blocking_coefficient));
182         // The intent here is to parallelise security checks as well as weed
183         // out blocking SocketPermission's to execute them in parallel to 
184         // reduce the wait on network IO.
185         // Once the pool size has reached it's maximum, the tasks are handed
186         // back to the calling thread to execute, this strategy also eliminates
187         // the possiblity of deadlock caused by circular dependencies between
188         // permission checks.
189         executor = 
190                 new ThreadPoolExecutor(numberOfCores, poolSizeLimit, 20L, 
191                 TimeUnit.SECONDS, new SynchronousQueue<Runnable>(), 
192                 new NamedThreadFactory("CombinerSecurityManager", true),
193                 new ThreadPoolExecutor.CallerRunsPolicy());
194         permCompare = RC.comparator(new PermissionComparator());
195         threadContext = new ThreadLocal<SecurityContext>();
196         inTrustedCodeRecursiveCall = new ThreadLocal<Boolean>();
197         // This is to avoid unnecessarily refreshing the policy.
198 //        Permission createAccPerm = new SecurityPermission("createAccessControlContext");
199 //	if (!policy.implies(context[0], createAccPerm)) policy.refresh();
200         /* Bug ID: 7093090 Reduce synchronization in java.security.Policy.getPolicyNoCheck
201          * This bug may cause contention between ProtectionDomain implies
202          * calls, also it could be a point of attack for Denial of service,
203          * since the lock used is a static class lock.  This bug has been fixed
204          * in jdk8(b15).
205          */
206 	/* The following ensures the classes we need are loaded early to avoid
207 	 * class loading deadlock during permission checks */
208 	try {
209 	    checkPermission(new RuntimePermission("setIO"), SMPrivilegedContext);
210 	} catch (ExceptionInInitializerError er){
211 	    Error e = new ExceptionInInitializerError(
212 		"All domains on stack when starting a security manager must have AllPermission");
213 	    e.initCause(er);
214 	    throw e;
215 	}
216 	constructed = Security.class != null;
217     }

221 	Object context = null;
222         inTrustedCodeRecursiveCall.set(Boolean.TRUE);
223         try {
224             context = Security.getContext();
225         }finally {
226             inTrustedCodeRecursiveCall.set(Boolean.FALSE); // Must always happen, no matter what.
227         }
228         return context;

231     /**
232      * Throws a <code>SecurityException</code> if the requested
233      * access, specified by the given permission, is not permitted based
234      * on the security policy currently in effect.
235      * 
236      * This method obtains the current SecurityContext and checks
237      * the give permission in that context.
238      * 
239      * @see SecurityContext
240      * @see Security
241      * @param perm
242      * @throws SecurityException 
243      */
244     @Override
245     public void checkPermission(Permission perm) throws SecurityException {
246         Boolean call = inTrustedCodeRecursiveCall.get();
247         if (call == Boolean.TRUE) return; // In Security and Policy static methods we trust.
248         Object context = getSecurityContext();
249         checkPermission(perm, context);
250     }

252     /**
253      * Throws a <code>SecurityException</code> if the requested
254      * access, specified by the given permission and context, is not permitted based
255      * on the security policy currently in effect.
256      * 
257      * It is absolutely essential that the SecurityContext override equals 
258      * and hashCode.
259      * 
260      * @param perm permission to be checked
261      * @param context - AccessControlContext or SecurityContext
262      * @throws SecurityException if context doesn't have permission.
263      */
264     @Override
265     public final void checkPermission(Permission perm, Object context) throws SecurityException {
266         if (perm == null ) throw new NullPointerException("Permission Collection null");
267 	perm.getActions(); // Ensure any lazy state has been instantiated before publication.
268         AccessControlContext executionContext = null;
269         SecurityContext securityContext = null;
270 	if (context instanceof AccessControlContext){
271             executionContext = (AccessControlContext) context;
272         } else if (context instanceof SecurityContext){
273             securityContext = (SecurityContext) context;
274             executionContext = securityContext.getAccessControlContext();
275         } else {
276             throw new SecurityException();
277         }
278         threadContext.set(securityContext); // may be null.
279         /* The next line speeds up permission checks related to this SecurityManager. */
280         if ( constructed
281 		&& (SMPrivilegedContext.equals(executionContext)
282 		|| SMConstructorContext.equals(executionContext) 
283 //		|| SMSecurityContext.equals(securityContext) 
284 		)
285             ) return; // prevents endless loop in debug.
286         // Checks if Permission has already been checked for this context.
287         NavigableSet<Permission> checkedPerms = checked.get(context);
288         if (checkedPerms == null){
289             /* A ConcurrentSkipListSet is used to avoid blocking during
290              * removal operations that occur while the garbage collector
291              * recovers softly reachable memory.  Since this happens while
292              * the jvm's under stress, it's important that permission checks
293              * continue to perform well.
294              * 
295              * Although I considered a multi read, single write Set, I wanted
296              * to avoid blocking under stress, that would be caused as a result
297              * of garbage collection.
298              * 
299              * IMPORTANT:
300              * The Set "checkedPerms" must be obtained prior to executing a permission
301              * check and the result written to the same Set, interleaved
302              * clear operations may remove the Set from the ConcurrentMap "checked",
303              * this prevents revoked permissions from entering the "checked"
304              * cache after clear is called and allows tasks to run to completion 
305              * without needing to be concerned about revocation.
306              */
307             NavigableSet<Referrer<Permission>> internal = 
308                     new ConcurrentSkipListSet<Referrer<Permission>>(permCompare);
309             checkedPerms = RC.navigableSet(internal, Ref.TIME, 10000L);
310             inTrustedCodeRecursiveCall.set(Boolean.TRUE);
311             try {
312                 NavigableSet<Permission> existed = checked.putIfAbsent(context, checkedPerms);
313                 if (existed != null) checkedPerms = existed;
314             }finally {
315                 inTrustedCodeRecursiveCall.set(Boolean.FALSE); // Must always happen, no matter what.
316             }
317         }
318         if (checkedPerms.contains(perm)) return; // don't need to check again.
319         // Cache the created AccessControlContext.
320         AccessControlContext delegateContext = contextCache.get(executionContext);
321         if (delegateContext == null ) {
322             final AccessControlContext finalExecutionContext = executionContext;
323             // Create a new AccessControlContext with the DelegateDomainCombiner
324             inTrustedCodeRecursiveCall.set(Boolean.TRUE);
325             try {
326                 delegateContext = AccessController.doPrivileged( 
327                     new PrivilegedAction<AccessControlContext>(){
328                         public AccessControlContext run() {
329                             return new AccessControlContext(finalExecutionContext, dc);
330                         }
331                     }
332                 );
333             }finally {
334                 inTrustedCodeRecursiveCall.set(Boolean.FALSE); // Must always happen, no matter what.
335             }
336             // Optimise the delegateContext, this runs the DelegateDomainCombiner
337             // and returns the AccessControlContext.
338             // This is a mutator method, the delegateContext returned
339             // is actually the same object passed in, after it is
340             // mutated, but just in case that changes in future we
341             // return it.
342             delegateContext = AccessController.doPrivileged(action, delegateContext);
343             inTrustedCodeRecursiveCall.set(Boolean.TRUE);
344             try {
345                 contextCache.putIfAbsent(executionContext, delegateContext);
346                 // Above putIfAbsent: It doesn't matter if it already existed,
347                 // the context we have is valid to perform a permissionCheck.
348             }finally {
349                 inTrustedCodeRecursiveCall.set(Boolean.FALSE); // Must always happen, no matter what.
350             }
351             
352         }
353         // Normal execution, same as SecurityManager.
354         delegateContext.checkPermission(perm); // Throws SecurityException.
355         /* It's ok to cache SocketPermission if we use a comparator */
356         // If we get to here, no exceptions were thrown, caller has permission.
357         checkedPerms.add(perm);
358     }

360     /**
361      * This method is intended to be called only by a Policy.
362      * 
363      * To clear the cache of checked Permissions requires the following Permission:
364      * java.security.SecurityPermission("getPolicy");
365      * 
366      * @throws SecurityException if caller isn't permitted to clear cache.
367      */
368     @Override
369     public void clearCache() throws SecurityException {
370         /* Clear the cache, out of date permission check tasks are still
371          * writing to old Set's, while new checks will write to new Sets.
372          */
373         g.checkGuard(this);
374         inTrustedCodeRecursiveCall.set(Boolean.TRUE);
375         try {
376             checked.clear();
377         }finally {
378             inTrustedCodeRecursiveCall.set(Boolean.FALSE); // Must always happen, no matter what.
379         }
380     }

392     private class DelegateDomainCombiner implements DomainCombiner {
393         
394         private DelegateDomainCombiner (){
395         }
396 
397         public ProtectionDomain[] combine(final ProtectionDomain[] currentDomains, final ProtectionDomain[] assignedDomains) {
398             /* We're only interested in the assignedDomains, since these
399              * are from the Context that the SecurityManager has been asked
400              * to check.
401              * 
402              * assignedDomains are inherited domains.
403              * 
404              * This code wraps assignedDomains in a DelegateProtectionDomain
405              * to ensure we check for the DelegatePermission or it's candidate
406              * Permission.
407              * 
408              * The AccessControlContext instance will be the new instance
409              * we just created moments earlier, but with the context returned
410              * by this DomainCombiner.
411              *
412              * The SecurityManager's ProtectionDomain must be removed
413              * from the Context, for the following case:
414              * 
415              * If using sun.security.provider.PolicyFile, the policy will
416              * cache it's own domain prior to it being instantiated and it
417              * may perform a PrivilegedAction when it's 
418              * getPermissions(ProtectionDomain pd) is later called for
419              * ProtectionDomain's not in policy cache.
420              * However, CombinerSecurityManager and
421              * net.jini.security.Security cannot cache their shared 
422              * ProtectionDomain, relying on the underlying policy instead.
423              * 
424              * When a standard java permission check
425              * is made, the AccessController picks up the domain of 
426              * CombinerSecurityManager and net.jini.security.Security,
427              * as well as that of the policy provider.  Since the policy
428              * provider will cache it's own ProtectionDomain, but not that
429              * of the SecurityManager and Security, a infinite circular call 
430              * loop will preceed until a StackOverflowError occurs.
431              * 
432              * This will be caused by PolicyFile, attempting to determine
433              * which permissions apply to the ProtectionDomain of
434              * CombinerSecurityManager and Security, then asking
435              * the SecurityManager if it has a FilePermission.
436              * 
437              * The policy provider org.apache.river.security.ConcurrentPolicyFile
438              * has no such issue, unless using CodeSource based PermissionGrant's,
439              * which have been deprecated.
440              */
441             int l = assignedDomains.length;
442             List<ProtectionDomain> list = new ArrayList<ProtectionDomain>(l);
443             for (int i = 0; i < l ; i++){
444                 if (assignedDomains[i] != privilegedDomain){
445                     list.add(assignedDomains[i]);
446                 }
447             }
448             ProtectionDomain [] context = list.toArray(new ProtectionDomain[list.size()]);
449             DelegateProtectionDomain[] delegated = new DelegateProtectionDomain[1];
450             delegated[0] = new DelegateProtectionDomain(context);
451             return delegated;
452         }
453     }

455     /*
456      * DelegateProtectionDomain executes checks on ProtectionDomain's in parallel.
457      */
458     private class DelegateProtectionDomain extends ProtectionDomain {
459         // Context from AccessControlContext.
460         private final ProtectionDomain[] context;
461         
462         DelegateProtectionDomain(ProtectionDomain[] context){
463             // Use static domain so we don't strongly reference the ClassLoader
464             // which has a strong reference to ProtectionDomain.
465             super(null, null);
466             this.context = context; // Not mutated so don't need to clone.
467         }
468         
469         /* An earlier implementation used interruption to cancel running tasks,
470          * this interruption only added complexity, in most cases permission
471          * checks are expected to pass and failure occurs far less often, 
472          * for that reason, it is acceptable for all tasks to run to completion.  
473          * The overall performance cost of using task interruption was likely 
474          * greater, due to increased access of shared memory for only a small
475          * performance benefit for failling permission checks.
476          * 
477          * If the current thread is interrupted, the interrupt status is
478          * preserved, this is done in cases where permission is required to perform
479          * safe shutdown
480          */
481         @Override
482         public boolean implies(Permission perm) {
483             Thread currentThread = Thread.currentThread();
484             boolean interrupt = Thread.interrupted(); // Clears the interrupt and stores it.
485             int l = context.length;
486             /* This is both a performance optimisation and a safety precaution.
487              * When there are only a few domains on the stack, they are 
488              * normally privileged and will return very quickly.
489              * 
490              * Also, permission checks performed inside PrivilegedAction
491              * calls by the Policy may come from setting or getting context security
492              * sensitive variables when wrappingPrivilegedAction
493              * permission checks.
494              * 
495              * The policy may accept Objects from other ProtectionDomain's
496              * as part of a PermissionGrant, this domain must be included
497              * in the context so it can be checked, but since that would
498              * create a recursive call, we avoid recursion
499              * by not splitting that permission check among multiple threads.
500              */
501             if ( l < 4 ){ 
502                 for ( int i = 0; i < l; i++ ){
503                     if (! checkPermission(context[i], perm)) {
504                         if (interrupt) currentThread.interrupt();
505                         return false;
506                     }
507                 }
508                 if (interrupt) currentThread.interrupt();
509                 return true;
510             }
511             CountDownLatch latch = new CountDownLatch(l);
512             List<RunnableFuture<Boolean>> resultList = new ArrayList<RunnableFuture<Boolean>>(l);
513             for ( int i = 0; i < l; i++ ){
514                 resultList.add(new FutureTask<Boolean>(
515                     new PermissionCheck(context[i], perm, latch, threadContext.get())
516                 ));
517             }
518             Iterator<RunnableFuture<Boolean>> it = resultList.iterator();
519             while (it.hasNext()){
520                 executor.execute(it.next());
521             }
522             try {
523                 // We can change either call to add a timeout.
524                 if (!latch.await(180L, TimeUnit.SECONDS)) return false; // Throws InterruptedException
525                 it = resultList.iterator();
526                 try {
527                     while (it.hasNext()){
528                             Boolean result = it.next().get(); // Throws InterruptedException
529                             if (result.equals(Boolean.FALSE)) {
530                                 if (interrupt) currentThread.interrupt();
531                                 return false;
532                             }
533                     }
534                     if (interrupt) currentThread.interrupt();
535                     return true;
536                 } catch (ExecutionException ex) {
537                     // This should never happen, unless a runtime exception occurs.
538                     if (getLogger().isLoggable(Level.SEVERE)) getLogger().log(Level.SEVERE, null, ex);
539                     throw new RuntimeException("Unrecoverable: ", ex.getCause()); // Bail out.
540                 }
541             } catch (InterruptedException ex) {
542                 // REMIND: Java Memory Model and thread interruption.           
543                 // We've been externally interrupted, during execution.
544                 // Do this the slow way to avoid reinterruption during shutdown cleanup!
545                 if (getLogger().isLoggable(Level.FINEST)) getLogger().log(Level.FINEST, "External Interruption", ex);
546                 for ( int i = 0; i < l; i++ ){
547                     if (!checkPermission(context[i], perm)) {
548                         currentThread.interrupt(); // restore external interrupt.
549                         return false;
550                     }
551                 }
552                 currentThread.interrupt(); // restore external interrupt.
553                 return true;
554             }
555         }
556         
557         @Override
558         public String toString(){
559             /* Unfortunately we don't know exactly which domain has failed
560              * in fact, multiple domains may fail the permission check since
561              * they are executed concurrently, for that reason, we'll print
562              * all domains on the stack.
563              */
564             StringBuilder sb = new StringBuilder(800);
565             sb.append("DomainCombinerSecurityManager full stack: \n");
566             int l = context.length;
567             for (int i = 0; i < l; i++ ){
568                 sb.append(context[i].toString());
569             }
570             return sb.toString();
571         }

575     /**
576      * Immutable callable task, discarded immediately after use.
577      */
578     private class PermissionCheck implements Callable<Boolean> {
579         private final ProtectionDomain pd;
580         private final Permission p;
581         private final CountDownLatch latch;
582         private final SecurityContext securityContext; // Preserves context accross calls.
583         
584         PermissionCheck(ProtectionDomain pd, Permission p, CountDownLatch c, SecurityContext sc){
585             if (pd == null || p == null) throw new NullPointerException();
586             this.pd = pd;
587             this.p = p;
588             latch = c;
589             securityContext = sc;
590             
591         }
592 
593         public Boolean call() throws Exception {
594             try {
595                 // Required for AggregatePolicyProvider.
596                 Boolean result = AccessController.doPrivileged( 
597                     securityContext != null ?
598                         securityContext.wrap(
599                             new PrivilegedAction<Boolean>(){
600                                 public Boolean run() {
601                                     boolean result = checkPermission(pd, p);
602                                     return Boolean.valueOf(result);
603                                 }
604                             }
605                         ) 
606                     :new PrivilegedAction<Boolean>(){
607                         public Boolean run() {
608                             boolean result = checkPermission(pd, p);
609                             return Boolean.valueOf(result);
610                         }
611                     }  
612                 );
613                 return result;
614             } finally {
615                 // In case we exit with a runtime exception, ensure threads
616                 // aren't left waiting.
617                 latch.countDown();
618             }
619         }
620         
621     }
