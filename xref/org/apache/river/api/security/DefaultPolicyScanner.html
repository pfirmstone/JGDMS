37  /**
38   * This is a basic high-level tokenizer of policy files. It takes in a stream,
39   * analyzes data read from it and returns a set of structured tokens. <br>
40   * This implementation recognizes text files, consisting of clauses with the
41   * following syntax:
42   * 
43   * <pre>
44   * 
45   *     keystore &quot;some_keystore_url&quot;, &quot;keystore_type&quot;;
46   *  
47   * </pre>
48   * <pre>
49   * 
50   *     grant [SignedBy &quot;signer_names&quot;] [, CodeBase &quot;URL&quot;]
51   *      [, Principal [principal_class_name] &quot;principal_name&quot;]
52   *      [, Principal [principal_class_name] &quot;principal_name&quot;] ... {
53   *      permission permission_class_name [ &quot;target_name&quot; ] [, &quot;action&quot;] 
54   *      [, SignedBy &quot;signer_names&quot;];
55   *      permission ...
56   *      };
57   *  
58   * </pre>
59   * 
60   * <br>
61   * Keywords are case-insensitive in contrast to quoted string literals.
62   * Comma-separation rule is quite forgiving, most commas may be just omitted.
63   * Whitespaces, line- and block comments are ignored. Symbol-level tokenization
64   * is delegated to java.io.StreamTokenizer. <br>
65   * <br>
66   * This implementation is effectively thread-safe, as it has no field references
67   * to data being processed (that is, passes all the data as method parameters).
68   * 
69   * @see org.apache.river.api.security.DefaultPolicyParser
70   */
71  class DefaultPolicyScanner {

96      /**
97       * Configures passed tokenizer accordingly to supported syntax.
98       */
99      StreamTokenizer configure(StreamTokenizer st) {
100         st.slashSlashComments(true);
101         st.slashStarComments(true);
102 	st.wordChars('_', '_');
103 	st.wordChars('$', '$');
104 	return st;
105     }

107     /**
108      * Performs the main parsing loop. Starts with creating and configuring a
109      * StreamTokenizer instance; then tries to recognize <i>keystore </i> or
110      * <i>grant </i> keyword. When found, invokes read method corresponding to
111      * the clause and collects result to the passed collection.
112      * 
113      * @param r
114      *            policy stream reader
115      * @param grantEntries
116      *            a collection to accumulate parsed GrantEntries
117      * @param keystoreEntries
118      *            a collection to accumulate parsed KeystoreEntries
119      * @throws IOException
120      *             if stream reading failed
121      * @throws InvalidFormatException
122      *             if unexpected or unknown token encountered
123      */
124     void scanStream(Reader r, Collection<GrantEntry> grantEntries,
125             List<KeystoreEntry> keystoreEntries) throws IOException,
126             InvalidFormatException {
127         StreamTokenizer st = configure(new StreamTokenizer(r));
128         //main parsing loop
129         parsing: while (true) {
130             switch (st.nextToken()) {
131             case StreamTokenizer.TT_EOF: //we've done the job
132                 break parsing;
133 
134             case StreamTokenizer.TT_WORD:
135                 if (Util.equalsIgnoreCase("keystore", st.sval)) { //$NON-NLS-1$
136                     keystoreEntries.add(readKeystoreEntry(st));
137                 } else if (Util.equalsIgnoreCase("grant", st.sval)) { //$NON-NLS-1$
138                     grantEntries.add(readGrantEntry(st));
139                 } else {
140                     handleUnexpectedToken(st, Messages.getString("security.89")); //$NON-NLS-1$
141                 }
142                 break;
143 
144             case ';': //just delimiter of entries
145                 break;
146 
147             default:
148                 handleUnexpectedToken(st);
149                 break;
150             }
151         }
152     }

154     /**
155      * Tries to read <i>keystore </i> clause fields. The expected syntax is
156      * 
157      * <pre>
158      * 
159      *     &quot;some_keystore_url&quot;[, &quot;keystore_type&quot;];
160      *  
161      * </pre>
162      * 
163      * @return successfully parsed KeystoreEntry
164      * @throws IOException
165      *             if stream reading failed
166      * @throws InvalidFormatException
167      *             if unexpected or unknown token encountered
168      */
169     KeystoreEntry readKeystoreEntry(StreamTokenizer st)
170             throws IOException, InvalidFormatException {
171         String url = null, type = null;
172         if (st.nextToken() == '"') {
173             url = st.sval;
174             if ((st.nextToken() == '"')
175                     || ((st.ttype == ',') && (st.nextToken() == '"'))) {
176                 type = st.sval;
177             } else { // handle token in the main loop
178                 st.pushBack();
179             }
180         } else {
181             handleUnexpectedToken(st, Messages.getString("security.8A")); //$NON-NLS-1$
182         }
183         return new KeystoreEntry(url, type);
184     }
185 
186     /**
187      * Tries to read <i>grant </i> clause. <br>
188      * First, it reads <i>codebase </i>, <i>signedby </i>, <i>principal </i>
189      * entries till the '{' (opening curly brace) symbol. Then it calls
190      * readPermissionEntries() method to read the permissions of this clause.
191      * <br>
192      * Principal entries (if any) are read by invoking readPrincipalEntry()
193      * method, obtained PrincipalEntries are accumulated. <br>
194      * The expected syntax is
195      * 
196      * <pre>
197      * 
198      *     [ [codebase &quot;url&quot;] | [signedby &quot;name1,...,nameN&quot;] | 
199      *          principal ...] ]* { ... }
200      *  
201      * </pre>
202      * 
203      * @return successfully parsed GrantEntry
204      * @throws IOException
205      *             if stream reading failed
206      * @throws InvalidFormatException
207      *             if unexpected or unknown token encountered
208      */
209     GrantEntry readGrantEntry(StreamTokenizer st) throws IOException,
210             InvalidFormatException {
211         String signer = null, codebase = null;
212         Collection<PrincipalEntry> principals = new ArrayList<PrincipalEntry>();
213         Collection<PermissionEntry> permissions = null;
214         
215         parsing: while (true) {
216             switch (st.nextToken()) {
217 
218             case StreamTokenizer.TT_WORD:
219                 if (Util.equalsIgnoreCase("signedby", st.sval)) { //$NON-NLS-1$
220                     if (st.nextToken() == '"') {
221                         signer = st.sval;
222                     } else {
223                         handleUnexpectedToken(st, Messages.getString("security.8B")); //$NON-NLS-1$
224                     }
225                 } else if (Util.equalsIgnoreCase("codebase", st.sval)) { //$NON-NLS-1$
226                     if (st.nextToken() == '"') {
227                         codebase = st.sval;
228                     } else {
229                         handleUnexpectedToken(st, Messages.getString("security.8C")); //$NON-NLS-1$
230                     }
231                 } else if (Util.equalsIgnoreCase("principal", st.sval)) { //$NON-NLS-1$
232                     principals.add(readPrincipalEntry(st));
233                 } else {
234                     handleUnexpectedToken(st);
235                 }
236                 break;
237 
238             case ',': //just delimiter of entries
239                 break;
240 
241             case '{':
242                 permissions = readPermissionEntries(st);
243                 break parsing;
244 
245             default: // handle token in the main loop
246                 st.pushBack();
247                 break parsing;
248             }
249         }
250 
251         return new GrantEntry(signer, codebase, principals, permissions);
252     }

254     /**
255      * Tries to read <i>Principal </i> entry fields. The expected syntax is
256      * 
257      * <pre>
258      * 
259      *     [ principal_class_name ] &quot;principal_name&quot;
260      *  
261      * </pre>
262      * 
263      * Both class and name may be wildcards, wildcard names should not
264      * surrounded by quotes.
265      * 
266      * @return successfully parsed PrincipalEntry
267      * @throws IOException
268      *             if stream reading failed
269      * @throws InvalidFormatException
270      *             if unexpected or unknown token encountered
271      */
272     PrincipalEntry readPrincipalEntry(StreamTokenizer st)
273             throws IOException, InvalidFormatException {
274         String classname = null, name = null;
275         if (st.nextToken() == StreamTokenizer.TT_WORD) {
276             classname = st.sval;
277             st.nextToken();
278         } else if (st.ttype == '*') {
279             classname = PrincipalEntry.WILDCARD;
280             st.nextToken();
281         }
282         if (st.ttype == '"') {
283             name = st.sval;
284         } else if (st.ttype == '*') {
285             name = PrincipalEntry.WILDCARD;
286         } else {
287             handleUnexpectedToken(st, Messages.getString("security.8D")); //$NON-NLS-1$
288         }
289         return new PrincipalEntry(classname, name);
290     }

292     /**
293      * Tries to read a list of <i>permission </i> entries. The expected syntax
294      * is
295      * 
296      * <pre>
297      * 
298      *     permission permission_class_name
299      *          [ &quot;target_name&quot; ] [, &quot;action_list&quot;]
300      *          [, signedby &quot;name1,name2,...&quot;];
301      *  
302      * </pre>
303      * 
304      * List is terminated by '}' (closing curly brace) symbol.
305      * 
306      * @return collection of successfully parsed PermissionEntries
307      * @throws IOException
308      *             if stream reading failed
309      * @throws InvalidFormatException
310      *             if unexpected or unknown token encountered
311      */
312     Collection<PermissionEntry> readPermissionEntries(
313             StreamTokenizer st) throws IOException, InvalidFormatException {
314         Collection<PermissionEntry> permissions = new HashSet<PermissionEntry>();
315         parsing: while (true) {
316             switch (st.nextToken()) {
317 
318             case StreamTokenizer.TT_WORD:
319                 if (Util.equalsIgnoreCase("permission", st.sval)) { //$NON-NLS-1$
320                     String klass = null, name = null, actions = null, signers = null;
321                     
322                     if (st.nextToken() == StreamTokenizer.TT_WORD) {
323                         klass = st.sval;
324                         if (st.nextToken() == '"') {
325                             name = st.sval;
326                             st.nextToken();
327                         }
328                         if (st.ttype == ',') {
329                             st.nextToken();
330                         }
331                         if (st.ttype == '"') {
332                             actions = st.sval;
333                             if (st.nextToken() == ',') {
334                                 st.nextToken();
335                             }
336                         }
337                         if (st.ttype == StreamTokenizer.TT_WORD
338                                 && Util.equalsIgnoreCase("signedby", st.sval)) { //$NON-NLS-1$
339                             if (st.nextToken() == '"') {
340                                 signers = st.sval;
341                             } else {
342                                 handleUnexpectedToken(st);
343                             }
344                         } else { // handle token in the next iteration
345                             st.pushBack();
346                         }
347                         PermissionEntry pe = new PermissionEntry(klass, name, actions, signers);
348                         permissions.add(pe);
349                         continue parsing;
350                     }
351                 }
352                 handleUnexpectedToken(st, Messages.getString("security.8E")); //$NON-NLS-1$
353                 break;
354 
355             case ';': //just delimiter of entries
356                 break;
357 
358             case '}': //end of list
359                 break parsing;
360 
361             default: // invalid token
362                 handleUnexpectedToken(st);
363                 break;
364             }
365         }
366 
367         return permissions;
368     }

370     /**
371      * Formats a detailed description of tokenizer status: current token,
372      * current line number, etc.
373      */
374     String composeStatus(StreamTokenizer st) {
375         return st.toString();
376     }

378     /**
379      * Throws InvalidFormatException with detailed diagnostics.
380      * 
381      * @param st
382      *            a tokenizer holding the erroneous token
383      * @param message
384      *            a user-friendly comment, probably explaining expected syntax.
385      *            Should not be <code>null</code>- use the overloaded
386      *            single-parameter method instead.
387      */
388     final void handleUnexpectedToken(StreamTokenizer st,
389             String message) throws InvalidFormatException {
390         throw new InvalidFormatException(Messages.getString("security.8F", //$NON-NLS-1$
391                 composeStatus(st), message));
392     }
393 
394     /**
395      * Throws InvalidFormatException with error status: which token is
396      * unexpected on which line.
397      * 
398      * @param st
399      *            a tokenizer holding the erroneous token
400      */
401     final void handleUnexpectedToken(StreamTokenizer st)
402             throws InvalidFormatException {
403         throw new InvalidFormatException(Messages.getString("security.90", //$NON-NLS-1$
404                 composeStatus(st)));
405     }

414     /**
415      * Compound token representing <i>keystore </i> clause. See policy format
416      * {@link org.apache.river.api.security.ConcurrentPolicyFile description}for details.
417      * 
418      * @see org.apache.river.api.security.DefaultPolicyParser
419      * @see org.apache.river.api.security.DefaultPolicyScanner
420      */
421     static class KeystoreEntry {
422 
423         /**
424          * The URL part of keystore clause.
425          */
426         private final String url;
427 
428         /**
429          * The typename part of keystore clause.
430          */
431         private final String type;
432         
433         KeystoreEntry(String url, String type){
434             this.url= url;
435             this.type= type;
436         }
437         
438         public String toString(){
439             String newline = "\n";
440             int l = getUrl() == null? 0 : getUrl().length();
441             l = l + (getType() == null? 0 : getType().length());
442             l = l + 4;
443             StringBuffer sb = new StringBuffer(l);
444             if ( getUrl() != null ) sb.append(getUrl()).append(newline);
445             if ( getType() != null ) sb.append(getType()).append(newline);
446             return sb.toString();
447         }
448 
449         /**
450          * @return the url
451          */
452         String getUrl() {
453             return url;
454         }
455 
456         /**
457          * @return the type
458          */
459         String getType() {
460             return type;
461         }
462     }

464     /**
465      * Compound token representing <i>grant </i> clause. See policy format
466      * {@link org.apache.river.api.security.ConcurrentPolicyFile description}for details.
467      * 
468      * @see org.apache.river.api.security.DefaultPolicyParser
469      * @see org.apache.river.api.security.DefaultPolicyScanner
470      */
471     static class GrantEntry {
472 
473         /**
474          * The signers part of grant clause. This is a comma-separated list of
475          * certificate aliases.
476          */
477         private final String signers;
478 
479         /**
480          * The codebase part of grant clause. This is an URL from which code
481          * originates.  Comma separate list allowed?
482          */
483         private final String codebase;
484 
485         /**
486          * Collection of PrincipalEntries of grant clause.
487          */
488         private final Collection<PrincipalEntry> principals;
489 
490         /**
491          * Collection of PermissionEntries of grant clause.
492          */
493         private final Collection<PermissionEntry> permissions;
494         
495         GrantEntry(String signers, String codebase, 
496                     Collection<PrincipalEntry> pe,
497                     Collection<PermissionEntry> perms){
498             this.signers = signers;
499             this.codebase = codebase;
500             this.principals = pe;
501             this.permissions = perms;
502         }
503         
504         public String toString(){
505             String newline = "\n";
506             StringBuilder sb = new StringBuilder(400);
507             if (signers != null ) sb.append(signers).append(newline);
508             if (codebase != null ) sb.append(codebase).append(newline);
509             if (principals != null ) sb.append(principals).append(newline);
510             if (permissions != null ) sb.append(permissions).append(newline);
511             return sb.toString();
512         }
513 
514         /**
515          * @return the signers
516          */
517         String getSigners() {
518             return signers;
519         }
520 
521         /**
522          * @return the codebase
523          */
524         String getCodebase(Properties system) {
525             if (system == null) return codebase;
526             try {
527                 return PolicyUtils.expand(codebase, system);
528             } catch (ExpansionFailedException ex) {
529 //                Logger.getLogger(DefaultPolicyScanner.class.getName()).log(Level.SEVERE, null, ex);
530                 ex.printStackTrace(System.err);
531                 return codebase;
532             }
533         }
534 
535         /**
536          * @return the principals
537          */
538         Collection<PrincipalEntry> getPrincipals(Properties system) {
539             return principals;
540         }
541 
542         /**
543          * @return the permissions
544          */
545         Collection<PermissionEntry> getPermissions() {
546             return permissions;
547         }

551     /**
552      * Compound token representing <i>principal </i> entry of a <i>grant </i>
553      * clause. See policy format
554      * {@link org.apache.river.api.security.ConcurrentPolicyFile description}for details.
555      * 
556      * @see org.apache.river.api.security.DefaultPolicyParser
557      * @see org.apache.river.api.security.DefaultPolicyScanner
558      */
559     static class PrincipalEntry {
560 
561         /**
562          * Wildcard value denotes any class and/or any name.
563          * Must be asterisk, for proper general expansion and 
564          * PrivateCredentialsPermission wildcarding
565          */
566         public static final String WILDCARD = "*"; //$NON-NLS-1$
567         
568         /**
569          * The classname part of principal clause.
570          */
571         private final String klass;
572 
573         /**
574          * The name part of principal clause.
575          */
576         private final String name;
577         
578         PrincipalEntry(String classname, String name){
579             klass = classname;
580             this.name = name;
581         }
582         
583         public String toString(){
584             String newline = "\n";
585             StringBuilder sb = new StringBuilder(100);
586             if ( getKlass() != null ) sb.append(getKlass()).append(newline);
587             if ( getName() != null ) sb.append(getName()).append(newline);
588             return sb.toString();
589         }
590 
591         /**
592          * @return the klass
593          */
594         String getKlass() {
595             return klass;
596         }
597 
598         /**
599          * @return the name
600          */
601         String getName() {
602             return name;
603         }
604     }

606     /**
607      * Compound token representing <i>permission </i> entry of a <i>grant </i>
608      * clause. See policy format
609      * {@link org.apache.river.api.security.ConcurrentPolicyFile description}for details.
610      * 
611      * @see org.apache.river.api.security.DefaultPolicyParser
612      * @see org.apache.river.api.security.DefaultPolicyScanner
613      */
614     static class PermissionEntry {
615 
616         /**
617          * The classname part of permission clause.
618          */
619         private final String klass;
620 
621         /**
622          * The name part of permission clause.
623          */
624         private final String name;
625 
626         /**
627          * The actions part of permission clause.
628          */
629         private final String actions;
630 
631         /**
632          * The signers part of permission clause. This is a comma-separated list
633          * of certificate aliases.
634          */
635         private final String signers;
636         
637         PermissionEntry(String klass, String name, String actions, String signers){
638             if (klass == null) throw new NullPointerException();
639             this.klass= klass;
640             this.name= name == null ? "" : name;
641             this.actions= actions == null ? "" : actions;
642             this.signers= signers;
643         }
644         
645         public String toString(){
646             String endline = "\n";
647             int l = getKlass() == null ? 0 : getKlass().length();
648             l = l + (getName() == null? 0 : getName().length());
649             l = l + (getActions() == null? 0 : getActions().length());
650             l = l + (getSigners() == null? 0 : getSigners().length());
651             l = l + 8;
652             StringBuffer sb = new StringBuffer(l);
653             if ( getKlass() != null ) sb.append(getKlass()).append(endline);
654             if ( getName() != null ) sb.append(getName()).append(endline);
655             if ( getActions() != null ) sb.append(getActions()).append(endline);
656             if ( getSigners() != null ) sb.append(getSigners()).append(endline);
657             return sb.toString();
658         }
659 
660         /**
661          * @return the klass
662          */
663         String getKlass() {
664             return klass;
665         }
666 
667         /**
668          * @return the name
669          */
670         String getName() {
671             return name;
672         }
673 
674         /**
675          * @return the actions
676          */
677         String getActions() {
678             return actions;
679         }
680 
681         /**
682          * @return the signers
683          */
684         String getSigners() {
685             return signers;
686         }
