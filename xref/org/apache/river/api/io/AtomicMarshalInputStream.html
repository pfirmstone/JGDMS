80  /**
81   * <h1>ObjectInputStream hardened against DOS attack.</h1>
82   * 
83   *    <h2>De-serialization Not supported:</h2>
84   * <p>
85   * <ul>
86   *      <li><code>private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException</code></li>
87   *      <li>Object graphs with circular references.</li>
88   *      <li>Object construction using the first non serializable superclass zero
89   *      arg constructor.</li>
90   * </ul>
91   * <p>
92   *    <h2>De-serialization Supported:</h2>
93   * <p>
94   * <ul>
95   *      <li>Classes annotated with @AtomicSerial, that provide a single GetArg parameter 
96   *      constructor and throw IOException.</li>
97   *      <li>Classes annotated with @AtomicExternal, that provide a single ObjectInput parameter
98   *      constructor and throw IOException.</li>
99   *      <li>Classes that implement {@link Externalizable}
100  *      <li>String's, arrays, enums, primitive values.</li>
101  *      <li>{@link Serializable} object's with a public zero arg constructor,
102  *      with serial forms that contain only primitive fields, any object fields must
103  *      be marked transient.</li>
104  *      <li>{@link net.jini.core.entry.Entry}, stream data will be checked against
105  *      each field type.</li>
106  * </ul>
107  * 
108  * 
109  * <p>
110  *      Any of the above classes that have the appropriate {@link DeSerializationPermission},
111  *      {@link Serializable} object's that have only primitive serial form, don't
112  *      require {@link DeSerializationPermission}.
113  *      The Serialization stream protocol.
114  * <p>
115  *    <h2>Informational:</h2>
116  * <p>
117  *      Collection, List Set, SortedSet, Map and SortedMap, are replaced in
118  *      AtomicObjectOutputStream with immutable implementations that guard
119  *      against denial of service attacks.  These collections are not intended
120  *      to be used in de-serialized form, other than for passing as an argument
121  *      to create a new collection.  Collections should be type checked during
122  *      validation before a superclass constructor is called.
123  * </p><p>
124  *      AtomicMarshalInputStream is restricted to caching 2^16 objects, and a total combined
125  *      array length of Integer.MAX_VALUE - 8, for all arrays, the
126  *      stream must be reset prior to exceeding these limits or a 
127  *      StreamCorruptedException will be thrown and control will return to the caller.
128  * </p><p>
129  *	JVM arguments should be adjusted to ensure that an OOME will not be thrown
130  *	if these limits are reached.
131  *</p>
132  * @author peter
133  */

461     /**
462      * Checks to if it is ok to read primitive types from this stream at
463      * this point. One is not supposed to read primitive types when about to
464      * read an object, for example, so an exception has to be thrown.
465      * 
466      * @throws IOException
467      *             If any IO problem occurred when trying to read primitive type
468      *             or if it is illegal to read primitive types
469      */
470     private void checkReadPrimitiveTypes() throws IOException {
471         // If we still have primitive data, it is ok to read primitive data
472         if (primitiveData == input || primitiveData.available() > 0) {
473             return;
474         }
475 
476         // If we got here either we had no Stream previously created or
477         // we no longer have data in that one, so get more bytes
478         do {
479             byte tc = nextTCNoEOFEx();
480             switch (tc) {
481                 case TC_BLOCKDATA:
482 //		    System.out.println("TC_BLOCKDATA");
483                     primitiveData = new ByteArrayInputStream(readBlockData());
484                     return;
485                 case TC_BLOCKDATALONG:
486 //		    System.out.println("TC_BLOCKDATALONG");
487                     primitiveData = new ByteArrayInputStream(
488                             readBlockDataLong());
489                     return;
490                 case TC_RESET:
491 //		    System.out.println("TC_RESET");
492                     resetState();
493                     break;
494                 default:
495                     if (tc != -1) {
496                         pushbackTC(tc);
497                     }
498                     return;
499             }
500             // Only TC_RESET falls through
501         } while (true);
502     }
503     
504     private void readyPrimitiveData() throws IOException{
505 	byte tc = nextTC();
506 	switch (tc) {
507 	    case TC_BLOCKDATA:
508 //				System.out.println("TC_BLOCKDATA");
509 		primitiveData = new ByteArrayInputStream(readBlockData());
510 		break;
511 	    case TC_BLOCKDATALONG:
512 //				System.out.println("TC_BLOCKDATALONG");
513 		primitiveData = new ByteArrayInputStream(
514 			readBlockDataLong());
515 		break;
516 	    case TC_RESET:
517 //				System.out.println("TC_RESET");
518 		resetState();
519 		break;
520 	    default:
521 		if (tc >= 0 && (tc < TC_BASE || tc > TC_MAX)) {
522 		    throw new StreamCorruptedException("invalid type code: " + tc);
523 		}
524 	}
525     }

589             if (currentManager != null) {

591             }
592         }
593         boolean originalValue = enableResolve;
594         enableResolve = enable;
595         return originalValue;
596     }

598     /**
599      * Checks if two classes belong to the same package.
600      * 
601      * @param c1
602      *            one of the classes to test.
603      * @param c2
604      *            the other class to test.
605      * @return {@code true} if the two classes belong to the same package,
606      *         {@code false} otherwise.
607      */
608     private boolean inSamePackage(Class<?> c1, Class<?> c2) {
609         String nameC1 = c1.getName();
610         String nameC2 = c2.getName();
611         int indexDotC1 = nameC1.lastIndexOf('.');
612         int indexDotC2 = nameC2.lastIndexOf('.');
613         if (indexDotC1 != indexDotC2) {
614             return false; // cannot be in the same package if indices are not
615         }
616         // the same
617         if (indexDotC1 < 0) {
618             return true; // both of them are in default package
619         }
620         return nameC1.substring(0, indexDotC1).equals(
621                 nameC2.substring(0, indexDotC2));
622     }
