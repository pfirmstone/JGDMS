63  /**
64   * A wrapper for activatable objects, providing separation of the import
65   * codebase (where the server classes are loaded from by the activation
66   * group) from the export codebase (where clients should load classes from
67   * for stubs, etc.) as well as providing an independent security policy file 
68   * for each activatable object. This functionality allows multiple 
69   * activatable objects to be placed in the same activation group, with each 
70   * object maintaining a distinct codebase and policy.
71   * <p>
72   * This wrapper class is assumed to be available directly in the activation
73   * group VM; that is, it is assumed to be in the application classloader,
74   * the extension classloader, or the boot classloader, rather than being
75   * downloaded. Since this class also needs considerable permissions, the
76   * easiest thing to do is to make it an installed extension.
77   * <p>
78   * This wrapper class performs a security check to control what 
79   * policy files can be used with a given codebase. 
80   * It does this by querying the VM's (global) policy for 
81   * {@link org.apache.river.start.SharedActivationPolicyPermission} 
82   * grants. The service's associated 
83   * {@link org.apache.river.start.ActivateWrapper.ActivateDesc#importLocation
84   * ActivateDesc.importLocation} is used as 
85   * the {@link java.security.CodeSource}
86   * for selecting the appropriate permission set to 
87   * check against. If multiple codebases are used, then all the codebases must
88   * have the necessary <code>SharedActivationPolicyPermission</code> grants.
89   * <p>
90   * An example of how to use this wrapper:
91   * <pre>
92   * URL[] importURLs = new URL[] {new URL("http://myhost:8080/service.jar")};
93   * URL[] exportURLs = new URL[] {new URL("http://myhost:8080/service-dl.jar")};
94   * ActivationID aid 
95   *     = ActivateWrapper.register(
96   *		gid,
97   *		new ActivateWrapper.ActivateDesc(
98   *			"foo.bar.ServiceImpl",
99   *			importURLs,
100  *			exportURLs,
101  *			"http://myhost:8080/service.policy",
102  *			new MarshalledObject(
103  *                          new String[] { "/tmp/service.config" })
104  *              ),
105  *		true,
106  *              activationSystem);
107  * </pre>
108  * <A NAME="serviceConstructor"></A>
109  * Clients of this wrapper service need to implement the following "activation
110  * constructor":
111  * <blockquote><pre>
112  * &lt;impl&gt;(ActivationID activationID, MarshalledObject data)
113  * </pre></blockquote>
114  * where,
115  * <UL>
116  * <LI>activationID - is the object's activation identifier
117  * <LI>data         - is the object's activation data
118  * </UL>
119  *
120  * Clients of this wrapper service can also implement 
121  * {@link net.jini.export.ProxyAccessor}, which allows the service 
122  * implementation to provide a remote reference of its choosing.
123  * <P>
124  * <A NAME="configEntries"></A>
125  * This implementation of <code>ActivateWrapper</code>
126  * supports the
127  * following {@link java.security.Security} property:
128  *
129  *   <table summary="Describes the org.apache.river.start.servicePolicyProvider
130  *          security property"
131  *          border="0" cellpadding="2">
132  *     <tr valign="top">
133  *       <th scope="col"> <font size="+1">&#X2022;</font>
134  *       <th scope="col" align="left" colspan="2"> <font size="+1"><code>
135  *       org.apache.river.start.servicePolicyProvider</code></font>
136  *     <tr valign="top"> <td> &nbsp; <th scope="row" align="right">
137  *       Default: <td> <code>
138  *         "net.jini.security.policy.DynamicPolicyProvider"
139  *         </code>
140  *     <tr valign="top"> <td> &nbsp; <th scope="row" align="right">
141  *       Description: <td> The fully qualified class name of a
142  *       dynamic policy provider (see {@link net.jini.security.policy.DynamicPolicy})
143  *       which will be used to "wrap" all service policy files. 
144  *       The implementation class needs to:
145  *       <UL>
146  *       <LI> implement the following constructor:
147  *           <blockquote><pre>
148  *   public &lt;impl&gt;(Policy servicePolicy)
149  *           </pre></blockquote>
150  *           where,
151  *           <LI>servicePolicy - is the service policy object to be wrapped
152  *           
153  *       <LI> implement {@link net.jini.security.policy.DynamicPolicy}
154  *       <LI> be a public, non-interface, non-abstract class
155  *       </UL>
156  *       
157  *       <P>
158  *       A custom service policy provider can be very useful when trying to
159  *       debug security related issues.
160  *       <code>org.apache.river.tool.DebugDynamicPolicyProvider</code> is an example
161  *       policy provider that provides this functionality and can be located
162  *       via the following URL:
163  *       <A HREF="http://starterkit-examples.jini.org/">
164  *           http://starterkit-examples.jini.org/
165  *       </A><BR>
166  *       <I>Note:</I>The custom policy implementation is assumed to be
167  *       available from the system classloader of the virtual machine
168  *       hosting the service. Its codebase should also be granted
169  *       {@link java.security.AllPermission}.
170  *   </table>
171  *
172  * @see org.apache.river.start.SharedActivationPolicyPermission
173  * @see java.rmi.activation.ActivationID
174  * @see java.rmi.MarshalledObject
175  * @see java.rmi.Remote
176  * @see java.security.CodeSource
177  * @see net.jini.export.ProxyAccessor
178  *
179  * @author Sun Microsystems, Inc.
180  *
181  */

379     /**
380      * Activatable constructor. This constructor: 
381      * <UL>
382      * <LI>Retrieves an <code>ActivateDesc</code> from the 
383      *     provided <code>data</code> parameter.
384      * <LI>creates an <code>ExportClassLoader</code> using the 
385      *     import and export codebases obtained from the provided 
386      *     <code>ActivateDesc</code>, 
387      * <LI>checks the import codebase(s) for the required 
388      *     <code>SharedActivationPolicyPermission</code>
389      * <LI>associates the newly created <code>ExportClassLoader</code>
390      *     and the corresponding policy file obtained from the 
391      *     <code>ActivateDesc</code> with the 
392      *     <code>AggregatePolicyProvider</code>
393      * <LI>loads the "wrapped" activatable object's class and
394      *     calls its activation constructor with the context classloader
395      *     set to the newly created <code>ExportClassLoader</code>.
396      * <LI> resets the context class loader to the original 
397      *      context classloader
398      * </UL>
399      * The first instance of this class will also replace the VM's 
400      * existing <code>Policy</code> object, if any,  
401      * with a <code>AggregatePolicyProvider</code>. 
402      *
403      * @param id The <code>ActivationID</code> of this object
404      * @param data The activation data for this object
405      * @throws java.lang.Exception
406      *
407      * @see org.apache.river.start.ActivateWrapper.ExportClassLoader
408      * @see org.apache.river.start.ActivateWrapper.ActivateDesc
409      * @see org.apache.river.start.AggregatePolicyProvider
410      * @see org.apache.river.start.SharedActivationPolicyPermission
411      * @see java.security.Policy
412      *
413      */
414     public ActivateWrapper(ActivationID id, MarshalledObject data)
415 	throws Exception
416     {
417          try {
418             logger.entering(ActivateWrapper.class.getName(), 
419 	        "ActivateWrapper", new Object[] { id, data });
420 
421 	    ActivateDesc desc = (ActivateDesc)
422                     new MarshalledInstance(data).get(false);
423 	    logger.log(Level.FINEST, "ActivateDesc: {0}", desc);
424 
425 	    Thread t = Thread.currentThread();
426 	    ClassLoader ccl = t.getContextClassLoader();
427  	    logger.log(Level.FINEST, "Saved current context class loader: {0}",
428 	       ccl);
429 
430 	    ExportClassLoader cl = null;
431             try {
432 	        cl = new ExportClassLoader(desc.importLocation, 
433 	                                   desc.exportLocation,
434 					   ccl);
435 	        logger.log(Level.FINEST, "Created ExportClassLoader: {0}", cl);
436             } catch (Exception e) {
437 	        logger.throwing(ActivateWrapper.class.getName(), 
438 	            "ActivateWrapper", e);
439 	        throw e;
440 	    }
441 	
442 	    checkPolicyPermission(desc.policy, desc.importLocation);
443 	
444 	    synchronized (ActivateWrapper.class) {
445 	        // supplant global policy 1st time through
446 	        if (globalPolicy == null) { 
447 		    initialGlobalPolicy = Policy.getPolicy();
448                     if (!(initialGlobalPolicy instanceof DynamicPolicy)) {
449                         initialGlobalPolicy = 
450                             new DynamicPolicyProvider(initialGlobalPolicy);
451                     }
452 		    globalPolicy = 
453 		        new AggregatePolicyProvider(initialGlobalPolicy);
454 		    Policy.setPolicy(globalPolicy);
455 	            logger.log(Level.FINEST, 
456 		        "Global policy set: {0}", globalPolicy);
457 	        }
458 		Policy service_policy = 
459 		    getServicePolicyProvider(
460 		        new PolicyFileProvider(desc.policy));
461 		Policy backstop_policy = 
462 		    getServicePolicyProvider(initialGlobalPolicy);
463                 LoaderSplitPolicyProvider split_service_policy = 
464                     new LoaderSplitPolicyProvider(
465                         cl, service_policy, backstop_policy);
466 		/* Grant "this" code enough permission to do its work
467 		* under the service policy, which takes effect (below)
468 		* after the context loader is (re)set.
469 		* Note: Throws UnsupportedOperationException if dynamic grants
470 		* aren't supported (because underlying policies don't support it).
471 		*/
472 		split_service_policy.grant(
473 	            this.getClass(), 
474 	            null, /* Principal[] */
475 	            new Permission[] { new AllPermission() } );	
476 	        globalPolicy.setPolicy(cl, split_service_policy);
477 	        logger.log(Level.FINEST, 
478 		    "Added policy to set: {0}", desc.policy);
479 	    }
480 	
481 	    boolean initialize = false;
482 	    Class ac = LoadClass.forName(desc.className, initialize, cl);
483  	    logger.log(Level.FINEST, "Obtained implementation class: {0}", ac);
484 
485 	    t.setContextClassLoader(cl);
486 
487 	    try {
488  	        logger.log(Level.FINEST, 
489 		    "Set new context class loader: {0}", cl);
490 		Constructor constructor =
491 		    ac.getDeclaredConstructor(actTypes);
492  	        logger.log(Level.FINEST, 
493 		    "Obtained implementation constructor: {0}", 
494 		    constructor);
495 		constructor.setAccessible(true);
496 		impl =
497 		    constructor.newInstance(new Object[]{id, desc.data});
498                 if (impl instanceof Startable) {
499                     ((Startable) impl).start();
500                 } else {
501                     logger.log( Level.FINE,
502                         "Service {0} doesn''t implement {1} {2} {3} {4} {5} {6}", 
503                         new Object []
504                             {
505                                 impl.getClass().getCanonicalName(),
506                                 Startable.class.getCanonicalName(),
507                                 "this service is likely to suffer from race",
508                                 "conditions caused by export performed during", 
509                                 "construction, or threads started while ''this''",
510                                 "has been allowed to escape during construction",
511                                 "https://www.securecoding.cert.org/confluence/display/java/TSM01-J.+Do+not+let+the+this+reference+escape+during+object+construction"
512                             } 
513                     );
514                 }
515  	        logger.log(Level.FINEST, 
516 		    "Obtained implementation instance: {0}", impl);
517 	    } finally {
518 	        t.setContextClassLoader(ccl);
519  	        logger.log(Level.FINEST, "Context class loader reset to: {0}", 
520 	            ccl);
521 	    }
522         } catch (Exception e) {
523 	    logger.throwing(ActivateWrapper.class.getName(), 
524 	        "ActivateWrapper", e);
525             throw e;
526         }
527         logger.exiting(ActivateWrapper.class.getName(), 
528 	    "ActivateWrapper");
529     }

572 	MarshalledObject data;
573 	try {
574 	    data = new MarshalledInstance(desc).convertToMarshalledObject();
575 	} catch (Exception e) {
576             MarshalException me = 
577 	        new MarshalException("marshalling ActivateDesc", e);
578 	    logger.throwing(ActivateWrapper.class.getName(), 
579 	        "register", me);
580 	    throw me;
581 	}
582 	
583 	ActivationDesc adesc =
584 	    new ActivationDesc(gid,
585 		ActivateWrapper.class.getName(),
586 		null,
587 		data,
588 		restart
589 	);      
590  	logger.log(Level.FINEST, 
591 	    "Registering descriptor with activation: {0}", adesc);
592 
593 	ActivationID aid = sys.registerObject(adesc);
594 
595         logger.exiting(ActivateWrapper.class.getName(), 
596 	    "register", aid);
597 	return aid;
598     }

600     /**
601      * Checks that all the provided <code>URL</code>s have permission to
602      * use the given policy.
603      */
604     private static void checkPolicyPermission(String policy, URL[] urls) {
605         logger.entering(ActivateWrapper.class.getName(), 
606 	    "checkPolicyPermission", new Object[] { policy, urlsToPath(urls) });
607         // Create desired permission object
608 	Permission perm = new SharedActivationPolicyPermission(policy);
609         Certificate[] certs = null;
610         CodeSource cs;
611 	 ProtectionDomain pd;
612 	// Loop over all codebases
613         int l = urls.length;
614 	for (int i=0; i < l; i++) {
615             // Create ProtectionDomain for given codesource
616 	    cs = new CodeSource(urls[i], certs);
617 	    pd = new ProtectionDomain(cs, null, null, null);
618 	    AccessControlContext acc = 
619 		    new AccessControlContext(new ProtectionDomain[]{pd});
620 	    SecurityManager sm = System.getSecurityManager();
621  	    logger.log(Level.FINEST, 
622 	        "Checking protection domain: {0}", pd);
623 	    
624 	    // Check if current domain allows desired permission
625 	    try {
626 		sm.checkPermission(perm, acc);
627 	    } catch (SecurityException e){
628 		SecurityException se =  new SecurityException(
629 		    "ProtectionDomain " + pd
630 		    + " does not have required permission: " + perm);
631                 logger.throwing(ActivateWrapper.class.getName(), 
632 	            "checkPolicyPermission", se);
633 		throw se;
634 	    }
635         }
636         logger.exiting(ActivateWrapper.class.getName(), 
637 	    "checkPolicyPermission");
638     }

648     private static String urlsToPath(URL[] urls) {
649 //TODO - check if spaces in file paths are properly escaped (i.e.% chars)	
650         if (urls == null) {
651             return null;
652         } else if (urls.length == 0) {
653             return "";
654         } else if (urls.length == 1) {
655             return urls[0].toExternalForm();
656         } else {
657             StringBuilder path = new StringBuilder(urls[0].toExternalForm());
658             for (int i = 1; i < urls.length; i++) {
659                 path.append(' ');
660                 path.append(urls[i].toExternalForm());
661             }
662             return path.toString();
663         }
664     }

666     static Policy getServicePolicyProvider(Policy service_policy) throws Exception {
667         Policy servicePolicyWrapper;
668         if (servicePolicyProvider != null) {
669  	    Class sp = Class.forName(servicePolicyProvider);
670 	    logger.log(Level.FINEST, 
671 	        "Obtained custom service policy implementation class: {0}", sp);
672 	    Constructor constructor =
673 	        sp.getConstructor(policyTypes);
674 	    logger.log(Level.FINEST, 
675 	        "Obtained custom service policy implementation constructor: {0}", 
676 	        constructor);
677 	    servicePolicyWrapper = (Policy)
678 		constructor.newInstance(new Object[]{service_policy});
679 	    logger.log(Level.FINEST, 
680 		"Obtained custom service policy implementation instance: {0}", 
681 		servicePolicyWrapper);
682 	} else {
683 	   servicePolicyWrapper = new DynamicPolicyProvider(service_policy);
684 	   logger.log(Level.FINEST, 
685 		"Using default service policy implementation instance: {0}", 
686 		servicePolicyWrapper);
687 	}
688 	return servicePolicyWrapper;
689     }
