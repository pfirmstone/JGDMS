18  package org.apache.river.outrigger;
19  
20  /**
21   * Mix-in interface for <code>QueryWatcher</code>s associated
22   * with ifExists queries.
23   */
24  interface IfExistsWatcher {
25      /**
26       * If the query has been resolved by the locked entry set
27       * becoming empty returns <code>true</code>, otherwise
28       * return <code>false</code>.
29       * @return <code>true</code> if locked entry set
30       * associated with this query is empty.
31       * @throws IllegalStateException if the query has not 
32       * be resolved.
33       */
34      public boolean isLockedEntrySetEmpty();
35  
36      /**
37       * Called after the last transition that was recored 
38       * during the initial search has been processed. This
39       * means that any entry that replaced an entry that 
40       * was removed before the initial search had gotten to
41       * it has been seen by the watcher. The query should now be
42       * resolved with a <code>null</code> if the locked
43       * entry set goes empty.
44       */
45      public void caughtUp();
46  }
