18  package org.apache.river.outrigger;
19  
20  /**
21   * This interface defines the methods that any OutriggerServerImpl store must
22   * implement.
23   *
24   * @author Sun Microsystems, Inc.
25   *
26   * @see OutriggerServerImpl
27   * @see LogOps
28   * @see Recover
29   */
30  public interface Store {
31  
32      /**
33       * Set up the store. Recovery (if any) will be completed
34       * before this method returns. That is, no calls will be made by
35       * <code>setupStore</code> on the <code>Recover</code> object
36       * after it returns.
37       *
38       * @param space object used for recovery of previous state (if any)
39       *
40       * @return object used to persist state
41       */
42      public LogOps setupStore(Recover space);
43  
44      /**
45       * Destroy any persistent state and release
46       * any resources associated with this store.
47       * @throws java.io.IOException if an IO related exception occurs.
48       */
49      public void destroy() throws java.io.IOException;
50  
51      /**
52       * Close the store, release VM resources (stop independent
53       * threads, close files, etc.) but do not destroy any persistent
54       * state. This method is used when there is a failure
55       * in <code>OutriggerServerImpl</code> constructor.
56       * @throws java.io.IOException if an IO related exception occurs.
57       */
58      public void close() throws java.io.IOException;
59  }
