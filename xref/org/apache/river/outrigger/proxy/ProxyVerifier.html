18  package org.apache.river.outrigger.proxy;
19  
20  import org.apache.river.landlord.Landlord;
21  import org.apache.river.landlord.LandlordProxyVerifier;
22  import java.io.IOException;
23  import java.io.InvalidObjectException;
24  import java.io.ObjectInputStream;
25  import java.io.Serializable;
26  import java.rmi.RemoteException;
27  import net.jini.core.constraint.MethodConstraints;
28  import net.jini.core.constraint.RemoteMethodControl;
29  import net.jini.id.Uuid;
30  import net.jini.security.TrustVerifier;
31  import net.jini.security.proxytrust.TrustEquivalence;
32  import org.apache.river.api.io.AtomicSerial.GetArg;
33  
34  /** 
35   * This class defines a trust verifier for the proxies related to the 
36   * Outrigger implementation of JavaSpaces technology. Uses 
37   * {@link LandlordProxyVerifier} to verify Leases.
38   *
39   * @see net.jini.security.TrustVerifier
40   * @author Sun Microsystems, Inc.
41   * @since 2.0
42   */
43  public final class ProxyVerifier implements Serializable, TrustVerifier {
44      private static final long serialVersionUID = 1L;
45  
46      /** 
47       * The canonical instance of the server reference. This
48       * instance will be used by the <code>isTrusted</code> method 
49       * as the known trusted object used to determine whether or not a
50       * given proxy is equivalent in trust, content, and function.
51       *
52       * @serial
53       */
54      private final RemoteMethodControl server;
55  
56      /**
57       * The top level <code>Uuid</code> that has been assigned to
58       * the Outrigger server this verifier is for.
59       */
60      private final Uuid uuid;
61  
62      /**
63       * Returns a verifier for the smart proxies of an Outrigger server with
64       * the specified server reference.
65       *
66       * @param server the reference to the <code>OutriggerServer</code>.
67       * @param uuid the <code>Uuid</code> assigned to the Outrigger
68       *             server <code>server</code> is a reference to.
69       * @throws UnsupportedOperationException if <code>server</code> does
70       *	       not implement both {@link RemoteMethodControl} and {@link
71       *	       TrustEquivalence}
72       * @throws NullPointerException if either argument is 
73       *         <code>null</code>.
74       */
75      public ProxyVerifier(OutriggerServer server, Uuid uuid) {
76  	this(check(server, uuid), server, uuid);
77      }
78      
79      ProxyVerifier(GetArg arg) throws IOException {
80  	this(check(arg),
81  		(OutriggerServer) arg.get("server", null),
82  		(Uuid) arg.get("uuid", null));
83      }
84      
85      private ProxyVerifier(boolean check, OutriggerServer server, Uuid uuid){
86  	this.uuid = uuid;
87          this.server = (RemoteMethodControl)server;
88      }
89      
90      private static boolean check(GetArg arg) throws IOException {
91  	try {
92  	    return check((OutriggerServer) arg.get("server", null),
93  		    (Uuid) arg.get("uuid", null));
94  	} catch (NullPointerException ex){
95  	    InvalidObjectException e = new InvalidObjectException("Invariants unsatisfied");
96  	    e.initCause(ex);
97  	    throw e;
98  	}
99      }
100     
101     private static boolean check(OutriggerServer server, Uuid uuid){
102 	if (server == null)
103 	    throw new NullPointerException("server can not be null");
104 
105 	if (uuid == null)
106 	    throw new NullPointerException("uuid can not be null");
107 
108         if(!(server instanceof RemoteMethodControl))
109             throw new UnsupportedOperationException
110 		("cannot construct verifier - server reference does not " +
111 		 "implement RemoteMethodControl");
112 
113 	if (!(server instanceof TrustEquivalence)) 
114             throw new UnsupportedOperationException
115 		("cannot construct verifier - server reference does not " +
116 		 "implement TrustEquivalence");
117 	return true;
118     }
119 
120     /** 
121      * Returns <code>true</code> if the specified proxy object (that is
122      * not yet known to be trusted) is equivalent in trust, content, and
123      * function to the canonical server reference this object was 
124      * constructed with; otherwise returns <code>false</code>.
125      *
126      * @param obj proxy object that will be compared to this class' stored
127      *            canonical proxy to determine whether or not the given
128      *            proxy object is equivalent in trust, content, and function.
129      *            
130      * @return <code>true</code> if the specified object (that is not yet
131      *                           known to be trusted) is equivalent in trust,
132      *                           content, and function to the canonical inner
133      *                           proxy object referenced in this class;
134      *                           otherwise returns <code>false</code>.
135      *
136      * @throws NullPointerException if any argument is <code>null</code>
137      */
138     public boolean isTrustedObject(Object obj, TrustVerifier.Context ctx)
139 	throws RemoteException
140     {
141         /* Validate the arguments */
142         if (obj == null || ctx == null) {
143             throw new NullPointerException("arguments must not be null");
144 	}
145 
146         /* Prepare the input proxy object for trust verification. The types
147          * of proxies, specific to the service, that this method will
148          * handle are:
149          *  - ConstrainableSpaceProxy2
150 	 *  - ConstrainableAdminProxy
151 	 *  - ConstrainableParticipantProxy
152 	 *  - ConstrainableLandlordLease (via LandlordProxyVerifier)
153          */
154 
155 	// Server reference from input proxy
156         final RemoteMethodControl inputProxyServer;
157 	final Uuid inputProxyUuid; 
158         if (obj instanceof ConstrainableSpaceProxy2) {
159 	    final ConstrainableSpaceProxy2 csp = (ConstrainableSpaceProxy2)obj;
160 
161 	    inputProxyUuid = csp.getReferentUuid();
162             inputProxyServer = (RemoteMethodControl)csp.space;
163         } else if (obj instanceof ConstrainableAdminProxy) {
164 	    final ConstrainableAdminProxy cap = (ConstrainableAdminProxy)obj;
165 
166 	    inputProxyUuid = cap.getReferentUuid();
167             inputProxyServer = (RemoteMethodControl)cap.admin;
168 	} else if (obj instanceof ConstrainableParticipantProxy) {
169 	    final ConstrainableParticipantProxy cpp = 
170 		(ConstrainableParticipantProxy)obj;
171 
172 	    inputProxyUuid = cpp.getReferentUuid();
173 	    inputProxyServer = (RemoteMethodControl)cpp.space;
174 	} else if (obj instanceof OutriggerServer &&
175 		   obj instanceof RemoteMethodControl) 
176 	{
177 	    // obj may be our inner proxy, which does not hold
178 	    // a ref to uuid, to simplify the logic just make
179 	    // inputProxyUuid = uuid.
180 	    inputProxyUuid = uuid;
181 	    inputProxyServer = (RemoteMethodControl)obj;
182         } else {
183 	    final LandlordProxyVerifier lpv = 
184 		new LandlordProxyVerifier((Landlord)server, uuid);
185 	    return lpv.isTrustedObject(obj, ctx);
186         }
187 
188 	// Check Uuid first
189 	if (!uuid.equals(inputProxyUuid))
190 	    return false;
191 
192         /* Get the client constraints currently set on the server reference
193 	 * of contained in the input proxy 
194 	 */
195         final MethodConstraints mConstraints = 
196 	    inputProxyServer.getConstraints();
197 
198         /* Create a copy of the canonical server reference with its method
199 	 * constraints replaced with the method constraints on
200 	 * server reference of the input proxy.
201          */
202         final TrustEquivalence constrainedServer =
203              (TrustEquivalence)server.setConstraints(mConstraints);
204  
205        /* With respect to trust, content, and function, test whether
206 	* the server reference from the input proxy is equivalent to
207 	* the canonical server reference we have (once method
208 	* constraints have been normalized.)
209 	*/
210         return constrainedServer.checkTrustEquivalence(inputProxyServer);
211     }
212 
213     /**
214      * Verifies that the server reference implements
215      * <code>TrustEquivalence</code>.
216      * @param in stream used to de-serialize.
217      * @throws IOException
218      * @throws ClassNotFoundException
219      */
220     private void readObject(ObjectInputStream in)
221 	throws IOException, ClassNotFoundException
222     {
223 	in.defaultReadObject();
224 
225 	if (server == null)
226 	    throw new InvalidObjectException("null server reference");
227 
228 	if (uuid == null)
229 	    throw new InvalidObjectException("null uuid reference");
230 
231 	if (!(server instanceof TrustEquivalence))
232 	    throw new InvalidObjectException(
233 		"server does not implement TrustEquivalence");
234 
235 	if (!(server instanceof Landlord))
236 	    throw new InvalidObjectException(
237 		 "server does not implement landlord");
238     }
