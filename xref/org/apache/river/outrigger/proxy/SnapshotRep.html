18  package org.apache.river.outrigger.proxy;
19  
20  import java.rmi.MarshalException;
21  import net.jini.core.entry.Entry;
22  
23  /**
24   * This class is used by the <code>snapshot</code> implementation of the
25   * proxy to create a <code>snapshot</code>.  Here's what the JavaSpaces
26   * specification says about <code>snapshot</code>.
27   *
28   * The process of serializing an entry for transmission to a JavaSpaces
29   * service will be identical if the same entry is used twice. This is most
30   * likely to be an issue with templates that are used repeatedly to search
31   * for entries with <code>read</code> or <code>take</code>. The client-side 
32   * implementations of <code>read</code> and <code>take</code> cannot
33   * reasonably avoid this duplicated effort, since they have no efficient
34   * way of checking whether the same template is being used without
35   * intervening modification.
36   *
37   * The <code>snapshot</code> method gives the JavaSpaces service implementor
38   * a way to reduce the impact of repeated use of the same entry. Invoking
39   * <code>snapshot</code> with an <code>Entry</code> will return another
40   * <code>Entry</code> object that contains a <i>snapshot</i> of the
41   * original entry. Using the returned snapshot entry is equivalent to
42   * using the unmodified original entry in all operations on the same
43   * JavaSpaces service. Modifications to the original entry will not
44   * affect the snapshot. You can <code>snapshot</code> a <code>null</code>
45   * template; <code>snapshot</code> may or may not return null given a
46   * <code>null</code> template.
47   *
48   * The entry returned from <code>snapshot</code> will be guaranteed
49   * equivalent to the original unmodified object only when used with
50   * the space. Using the snapshot with any other JavaSpaces service
51   * will generate an <code>IllegalArgumentException</code> unless the
52   * other space can use it because of knowledge about the JavaSpaces
53   * service that generated the snapshot. The snapshot will be a different
54   * object from the original, may or may not have the same hash code,
55   * and <code>equals</code> may or may not return <code>true</code>
56   * when invoked with the original object, even if the original
57   * object is unmodified.
58   *
59   * A snapshot is guaranteed to work only within the virtual machine
60   * in which it was generated. If a snapshot is passed to another 
61   * virtual machine (for example, in a parameter of an RMI call),
62   * using it--even with the same JavaSpaces service--may generate
63   * an <code>IllegalArgumentException</code>.
64   *
65   * For more information, please review the appropriate specifications.
66   *
67   * @author Sun Microsystems, Inc.
68   *
69   */
70  // @see SpaceProxy#snapshot
71  class SnapshotRep implements Entry {
72      static final long serialVersionUID = 5126328162389368097L;
73  
74      private final EntryRep rep;	// the rep from the snapshot
75  
76      /**
77       * Create a new <code>SnapshotRep</code> that is a snapshot of
78       * <code>e</code>.
79       */
80      SnapshotRep(Entry e) throws MarshalException {
81  	rep = new EntryRep(e);
82      }
83  
84      /**
85       * Construct an SnapshotRep from an existing EntryRep
86       */
87      SnapshotRep(EntryRep e) {
88  	rep = e;
89      }
90  
91      /**
92       * Return the pre-computed <code>EntryRep</code>.
93       */
94      EntryRep rep() {
95  	return rep;
96      }
97  }
