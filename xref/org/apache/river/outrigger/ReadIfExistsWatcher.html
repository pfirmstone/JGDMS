18  package org.apache.river.outrigger;
19  
20  import java.util.Set;
21  import net.jini.id.Uuid;
22  import org.apache.river.outrigger.proxy.EntryRep;
23  
24  /**
25   * Subclass of <code>QueryWatcher</code> for non-transactional if
26   * exists reads. Resolves with the first matching
27   * transition where the transaction is <code>null</code> and the entry
28   * is visible (the entry's current state is ignored) or if 
29   * the locked entry set goes empty.
30   */
31  class ReadIfExistsWatcher extends SingletonQueryWatcher
32      implements IfExistsWatcher 
33  {
34      /**
35       * The set of entries that would match but are currently
36       * unavailable (e.g. they are locked). We only keep
37       * the ids, not the entries themselves.
38       */
39      private final Set<Uuid> lockedEntries;
40  
41      /**
42       * Set <code>true</code> once the query thread is 
43       * done processing the backlog. Once this is 
44       * <code>true</code> it is ok to resolve if
45       * <code>lockedEntries</code> is empty.
46       */
47      private boolean backlogFinished = false;
48  
49      /**
50       * Create a new <code>ReadIfExistsWatcher</code>.
51       * @param expiration the initial expiration time
52       *        for this <code>TransitionWatcher</code> in 
53       *        milliseconds since the beginning of the epoch.
54       * @param timestamp the value that is used
55       *        to sort <code>TransitionWatcher</code>s.
56       * @param startOrdinal the highest ordinal associated
57       *        with operations that are considered to have occurred 
58       *        before the operation associated with this watcher.
59       * @param lockedEntries Set of entries (by their IDs)
60       *        that match but are unavailable. Must be non-empty.
61       *        Keeps a reference to this object.
62       * @throws NullPointerException if <code>lockedEntries</code> is
63       *         <code>null</code>.
64       */
65      ReadIfExistsWatcher(long expiration, long timestamp, long startOrdinal, 
66  			Set<Uuid> lockedEntries)
67      {
68  	super(expiration, timestamp, startOrdinal);
69  
70  	if (lockedEntries == null) 
71  	    throw new NullPointerException("lockedEntries must be non-null");
72  	
73  	this.lockedEntries = lockedEntries;
74      }
75  
76      boolean isInterested(EntryTransition transition, long ordinal) {
77  	/* If we are unresolved pretty much all transitions are
78  	 * interesting because we may need to update
79  	 * lockedEntries. The only exception is read locks being
80  	 * resolved. It is important that transitions triggered by the
81  	 * release of read locks get filtered out - otherwise process
82  	 * could end up adding and removing elements to lockedEntries
83  	 * when it shouldn't.
84  	 * 
85  	 * Note, !isResolved() without the lock will result only in
86  	 * false positives, not false negatives - it will only
87  	 * cause isInterested() to return false if we are resolved,
88  	 * we may still return true if we are resolved though.  
89  	 */
90  
91  	if (!transition.isVisible() && transition.isAvailable()) {
92  	    /* must be a transition triggered by a read lock release,
93  	     * wont change anything so ignore it.
94  	     */
95  	    return false;
96  	}
97  
98  	return (ordinal>startOrdinal) && !isResolved();
99      }
100 
101     synchronized void process(EntryTransition transition, long now) {
102 	if (isResolved())
103 	    return; // Already done.
104 
105 	final EntryRep rep = transition.getHandle().rep();
106 	final boolean isVisible = transition.isVisible();
107 
108 	/* If the entry was visible at one time to the null 
109 	 * transaction we can just resolve.
110 	 */
111 	if (isVisible && (transition.getTxn() == null)) {
112 	    resolve(transition.getHandle(), null);
113 	} else if (isVisible) { // && getTxn() != null
114 	    /* If we are here transition.getTxn() must be != null 
115 	     * and the entry was visible, implying that it was an entry
116 	     * written under a transaction and is interesting to 
117 	     * us, but not yet visible. We need to add it lockedEntries
118 	     * even if has been removed since it could have gotten
119 	     * replaced before it was removed. If we did not
120 	     * add it we would be acting on the future.
121 	     */
122 	    lockedEntries.add(rep.id());
123 	} else {
124 	    /* Must not be visible (and becaues of the test in
125 	     * isInteresting can't be available either - that is
126 	     * transition can't just be the release of a read lock),
127 	     * transition must mark the resolution of the transaction
128 	     * in such away that the entry has been removed, remove it
129 	     * from the set and see if that makes the set empty.  
130 	     */
131 	    lockedEntries.remove(rep.id());
132 	    if (backlogFinished && lockedEntries.isEmpty())
133 		resolve(null, null);
134 	}
135     }
136 
137     synchronized boolean catchUp(EntryTransition transition, long now) {
138 	if (isResolved())
139 	    return true; // Already done.
140 	
141 	final EntryHandle handle = transition.getHandle();
142 	final EntryRep rep = handle.rep();
143 	final boolean isVisible = transition.isVisible();
144 
145 	/* Was this the resolution of a read lock? if so ignore */
146 	if (!isVisible && transition.isAvailable())
147 	    return false;
148 
149 	/* If the entry was visible at one time to the null 
150 	 * transaction we can just resolve.
151 	 */
152 	if (isVisible && (transition.getTxn() == null)) {
153 	    resolve(handle, null);
154 	    return true;
155 	} 
156 
157 	if (isVisible) { // && getTxn() != null
158 	    /* If we are here transition.getTxn() must be != null and
159 	     * the entry is/was visible to someone, implying that it
160 	     * was an entry written under a transaction and is
161 	     * interesting to us, but not yet visible to us. We only
162 	     * add if it has not already been removed.  It might have
163 	     * gotten replaced before removal, but since we won't let
164 	     * this query get resolved with a definitive null before
165 	     * we get past the point in the journal where it was
166 	     * removed it is ok to never put it in (and if we did put
167 	     * it in it might never get removed since process() may
168 	     * have already processed the removal). We don't need to
169 	     * check to see it the entry has been provisionally
170 	     * removed since if has been provisional removal does not
171 	     * put in entries in the journal and if it is
172 	     * provisionally removed it has not yet been removed so
173 	     * the remove recored has not yet been created (must less
174 	     * processed).  
175 	     */
176 	    synchronized (handle) {
177 		if (!handle.removed()) {
178 		    lockedEntries.add(rep.id());
179 		}		
180 	    }
181 
182 	    // Either way, still not resolved.
183 	    return false;
184 	}
185 	 
186 	/* Must not be visible (and because of the first test can't be
187 	 * available either - that is transition can't just be the
188 	 * release of a read lock), transition must mark the
189 	 * resolution of the transaction in such away that the entry
190 	 * has been removed, remove it from the set (don't need to
191 	 * check for empty because we haven't gotten to the point
192 	 * where we can resolve with a definitive null.)  
193 	 */
194 	lockedEntries.remove(rep.id());
195 	return false;
196     }
197 
198     /**
199      * Once the backlog is complete we can resolve if 
200      * lockedEntries is/becomes empty.
201      */
202     public synchronized void caughtUp() {
203 	backlogFinished = true;
204 	
205 	if (isResolved())
206 	    return; // Don't much mater.
207 
208 	if (lockedEntries.isEmpty())
209 	    resolve(null, null);	
210     }
211 
212     public synchronized boolean isLockedEntrySetEmpty() {
213 	if (!isResolved())
214 	    throw new IllegalStateException("Query not yet resolved");	
215 	return lockedEntries.isEmpty();
216     }
