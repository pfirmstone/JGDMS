45  /**
46   * The notifier thread.  This thread is responsible for notifying
47   * objects for which interest has been registered.  It operates in
48   * transient space as much as possible.  Pending notifications will be
49   * lost when the server goes down, but registrations of interest
50   * survive across server crashes for persistent servers.
51   *
52   * @author Sun Microsystems, Inc.
53   *
54   * @see JavaSpace#notify
55   * @see OutriggerServerImpl#notify
56   */
57  // @see NotifyChit
58  class Notifier implements org.apache.river.constants.TimeConstants {
59      /** 
60       * The object to use for the <code>source</code> when creating
61       * events.
62       */
63      private final JavaSpace source;
64  
65      /** Proxy preparer to use on recovered listeners */
66      private final ProxyPreparer recoveredListenerPreparer;
67      
68      /** wakeup manager for <code>NotifyTask</code> */
69      private final WakeupManager wakeupMgr = 
70  	new WakeupManager(new WakeupManager.ThreadDesc(null, true));
71      
72      /** pending notifications tasks */
73      private final ExecutorService pending;
74  
75      private final static int	MAX_ATTEMPTS = 10;	// max times to retry
76  
77      /** Logger for logging event related information */
78      private static final Logger logger = 
79  	Logger.getLogger(OutriggerServerImpl.eventLoggerName);
80  
81      /**
82       * Create a notifier connected to the given <code>space</code>.
83       * @param source the value to use for the <code>source</code> in
84       *               remote event objects.
85       * @param recoveredListenerPreparer <code>ProxyPreparer</code> to
86       *               apply to recovered listeners.
87       * @param config a source of configuration data.a
88       * @throws ConfigurationException if there is a problem
89       *         with the passed configuration.
90       * @throws NullPointerException if <code>source</code> or 
91       *         <code>config</code> arguments are <code>null</code>.
92       */
93      Notifier(JavaSpace source, ProxyPreparer recoveredListenerPreparer, 
94  	     Configuration config) 
95  	throws ConfigurationException 
96      {
97  	if (source == null)
98  	    throw new NullPointerException("source must be non-null");
99  	this.source = source;
100 
101 	this.recoveredListenerPreparer = recoveredListenerPreparer;
102 
103 	pending = Config.getNonNullEntry(config,
104 	    OutriggerServerImpl.COMPONENT_NAME, "notificationsExecutorService", 
105 	    ExecutorService.class, 
106             new ThreadPoolExecutor(
107                 10,
108                 10, /* Ignored */
109                 15,
110                 TimeUnit.SECONDS, 
111                 new LinkedBlockingQueue<Runnable>(), /* Unbounded queue */
112                 new NamedThreadFactory("OutriggerServerImpl Notifier", false)
113             )
114         );
115     }
116 
117     /**
118      * Terminate the notifier, shutting down any threads
119      * it has running. This method can assume that
120      * the constructor completed.
121      */
122     void terminate() {
123 	pending.shutdown();
124 	wakeupMgr.stop();	
125 	wakeupMgr.cancelAll();	
126     }
127 
128     /**
129      * Queue up an event for delivery.
130      * @param sender An object that on request will
131      *               attempt to deliver its event
132      *               to the associated listener.
133      * @throws NullPointerException if <code>sender</code> is
134      * <code>null</code>
135      */
136     void enqueueDelivery(EventSender sender, AccessControlContext context) {
137 	pending.execute(new NotifyTask(sender, context));
138     }
139 
140     /*
141      * Static stuff for Pending (can't put it in the class, unfortunately).
142      */
143 				// 1 day =hrs  mins secs milliseconds
144     private static final long	MAX_TIME = 1 * DAYS;
145     private static final long	delays[] = {
146 				    1 * SECONDS, 5 * SECONDS,
147 				    10 * SECONDS, 60 * SECONDS, 60 * SECONDS
148 				};
149 
150     static {
151 	/*
152 	 * Make the delays the amount of time since the start -- it
153 	 * is easier to declare the intervals, but the elapsed time is
154 	 * more <i>useful</i>.
155 	 */
156 	for (int i = 1; i < delays.length; i++)
157 	    delays[i] += delays[i - 1];
158     }
159 
160     /**
161      * A task that represent a notification of matching a particular
162      * template under a given transaction.
163      */
164     private class NotifyTask extends RetryTask {
165 	/** Who and what to send a event to. */
166 	private final EventSender sender;	
167 	private final AccessControlContext context;
168 
169 	/**
170 	 * Create an object to represent this list of chits needing
171 	 * notification.
172 	 * @param sender An object that on request will
173 	 *               attempt to deliver its event
174 	 *               to the associated listener.
175 	 * @throws NullPointerException if <code>sender</code> is
176 	 * <code>null</code>
177 	 */
178 	NotifyTask(EventSender sender, AccessControlContext context) {
179 	    super(Notifier.this.pending, Notifier.this.wakeupMgr);
180 	    if (sender == null)
181 		throw new NullPointerException("sender must be non-null");
182 	    this.sender = sender;
183 	    this.context = context;
184 	}
185 
186 	/**
187 	 * Try to notify the target.  Return <code>true</code> if the
188 	 * notification was successful.
189 	 * <p>
190 	 * We know that we are the only one dealing with the given chit
191 	 * because <code>runAfter</code> makes sure of it.
192 	 */
193 	@Override
194 	public boolean tryOnce() {
195 	    long curTime = System.currentTimeMillis();
196 	    if (curTime - startTime() > MAX_TIME) {
197 		if (logger.isLoggable(Levels.FAILED)) {
198 		    logger.log(Levels.FAILED, 
199 			"giving up on delivering event, keeping registration");
200 		}
201 
202 		return true;	// just stop here, we are declaring "success"
203 	    }
204 
205 	    boolean successful = true;	// notification successful?
206 	    try {
207 		try {
208 		    AccessController.doPrivileged(
209 			    new PrivilegedExceptionAction(){
210 				
211 				@Override
212 				public Object run() throws Exception {
213 				    sender.sendEvent(source, curTime, recoveredListenerPreparer);
214 				    return null;
215 				}
216 			    }, context
217 		    );
218 		} catch (PrivilegedActionException ex) {
219 		    Exception e = ex.getException();
220 		    if (e instanceof IOException) throw (IOException) e;
221 		    if (e instanceof UnknownEventException) throw (UnknownEventException)e;
222 		    if (e instanceof ClassNotFoundException) throw (ClassNotFoundException)e;
223 		    if (e instanceof RuntimeException) throw (RuntimeException)e;
224 		    throw new IOException("Unexpected Exception", ex);
225 		}
226 		
227 	    } catch (UnknownEventException e) {
228 		// they didn't want to know about this, so stop them getting
229 		// future notifications, too.
230 		logFailure("UnknownEventException", Level.FINER, true, e);
231 		sender.cancelRegistration();
232 		// this is still "successful" -- we know to stop sending this
233 	    } catch (RemoteException e) {
234 		final int cat = ThrowableConstants.retryable(e);
235 
236 		if (cat == ThrowableConstants.BAD_INVOCATION ||
237 		    cat == ThrowableConstants.BAD_OBJECT)
238 		{
239 		    // Listener probably bad, retry likely to fail.
240 		    logFailure("definite exception", Level.INFO, true, e);
241 		    sender.cancelRegistration();
242 		} else if (cat == ThrowableConstants.INDEFINITE) {
243 		    // try, try, again
244 		    logFailure("indefinite exception", Levels.FAILED,
245 			       false, e);
246 		    successful = false;	
247 		} else if (cat == ThrowableConstants.UNCATEGORIZED) {
248 		    // Same as above but log differently.
249 		    logFailure("uncategorized exception", Level.INFO, false, 
250 			       e);
251 		    successful = false;	
252 		} else {
253 		    logger.log(Level.WARNING, "ThrowableConstants.retryable " +
254 			"returned out of range value, " + cat,
255 			new AssertionError(e));
256 		    successful = false;			    
257 		}
258 	    } catch (IOException e) {
259 		// corrupted listener? unlikely to get better, cancel
260 		logFailure("IOException", Level.INFO, true, e);
261 		sender.cancelRegistration();
262 	    } catch (ClassNotFoundException e) {
263 		// probably a codebase problem, retry
264 		logFailure("ClassNotFoundException", Levels.FAILED, false, e);
265 		successful = false;			
266 	    } catch (RuntimeException e) {
267 		/* bad listener, or preparer, either way unlikely to
268 		 * get better
269 		 */
270 		logFailure("RuntimeException", Level.INFO, true, e);
271 		sender.cancelRegistration();
272 	    }
273 
274 	    if (!successful && attempt() > MAX_ATTEMPTS) {
275 		if (logger.isLoggable(Levels.FAILED)) {
276 		    logger.log(Levels.FAILED, 
277 			"giving up on delivering event, keeping registration");
278 		}
279 		return true;		// as successful as we're going to be
280 	    }
281 
282 	    return successful;
283 	}
284 
285 	/** Log a failed delivery attempt */
286 	private void logFailure(String exceptionDescription, Level level,
287 				boolean terminal, Throwable t) 
288 	{
289 	    if (logger.isLoggable(level)) {
290 		logger.log(level, "Encountered " + exceptionDescription +
291 		     "while preparing to send/sending event, " +
292 		     (terminal?"dropping":"keeping") +  " registration", t);
293 	    }
294 	}
295     }
