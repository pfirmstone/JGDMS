86      /**
87       * Return an <code>EntryHandle</code> object that matches the given
88       * template, or <code>null</code> if none does. Optionally
89       * removes (perhaps provisionally) the found entry.
90       *
91       * @param tmpl	The template to match against
92       * @param txn       If non-null the transaction (represented as
93       *                  a <code>TransactableMgr</code> to perform
94       *                  the operation under. May be <code>null</code>
95       *                  if the operation is not to be done under
96       *                  a transaction.
97       * @param takeIt	<code>true</code> if <code>hasMatch</code> should
98       *			remove the matching entry.
99       * @param conflictSet If non-null the <code>TransactableMgr</code>
100      *                  objects of any transactions that prevent 
101      *                  a non-null value from being retured will
102      *                  be added to <code>conflictSet</code>. May
103      *                  be <code>null</code> in which case
104      *                  conflicting transaction will not be recorded.
105      *                  This method assumes that any concurrent access
106      *                  is being arbitrated by the set or by the caller.
107      * @param lockedEntrySet If non-null the ID of any entries that
108      *                  can't be retured because of conflicting
109      *                  transaction will be added to
110      *                  <code>lockedEntrySet</code>. May be
111      *                  <code>null</code> in which case unavailable
112      *                  entries will not be recorded.  This method 
113      *                  assumes that any concurrent access is being
114      *                  arbitrated by the set or by the caller.
115      * @param provisionallyRemovedEntrySet If the entry can not be
116      *              read/taken because it has been provisionally
117      *              removed then its handle will be placed in the
118      *              passed <code>WeakHashMap</code> as a key (with
119      *              null as the value).  May be <code>null</code> in
120      *              which case provisionally removed entries will not
121      *              be recorded. This method assumes that any
122      *              concurrent access is being arbitrated by the set
123      *              or by the caller.
124      * @throws CannotJoinException if a match is found and
125      *         the operation is to be performed under a transaction,
126      *         but the transaction is no longer active.
127      * @see #attemptCapture 
128      */
129     EntryHandle hasMatch(EntryRep tmpl, TransactableMgr txn, boolean takeIt,
130             Set conflictSet, Set lockedEntrySet,
131             Set provisionallyRemovedEntrySet)
132             throws CannotJoinException {
133         matchingLogger.entering("EntryHolder", "hasMatch");
134         EntryHandleTmplDesc desc = null;
135         long startTime = 0;
136 
137         for (EntryHandle handle : content) {
138 
139             if (startTime == 0) {
140                 // First time through
141                 desc = EntryHandle.descFor(tmpl, handle.rep().numFields());
142                 startTime = System.currentTimeMillis();
143             }
144 
145             if (handle.removed())
146                 continue;
147 
148             // Quick reject -- see the if handle mask is incompatible
149             if ((handle.hash() & desc.mask) != desc.hash)
150                 continue;
151 
152             final EntryRep rep = handle.rep();
153 
154             if (!tmpl.matches(rep))
155                 continue;
156 
157             final boolean available = confirmAvailabilityWithTxn(rep, handle,
158                     txn, takeIt, startTime, conflictSet, lockedEntrySet,
159                     provisionallyRemovedEntrySet);
160 
161             if (available)
162                 return handle;
163         }
164 
165         return null;
166     }

186     /**
187      * Atomically check to see if the passed entry can be read/taken by
188      * the specified operation using the specified transaction and if
189      * it can read/take it and return <code>true</code>, otherwise
190      * return <code>false</code>. If the entry is removed. Note,
191      * if the entry is removed, removal is logged.
192      * @param handle The <code>EntryHandle</code> of the entry
193      *              the caller wants to read/take.
194      * @param txn   If non-null the transaction to perform
195      *              this operation under. Note, if non-null and 
196      *              <code>txn</code> is not active <code>false</code>
197      *              will be returned.
198      * @param takeIt <code>true</code> if the caller is trying
199      *              take the passed entry, <code>false</code>
200      *              otherwise.
201      * @param conflictSet If non-null and the entry can not be
202      *              read/taken because of transaction conflicts the
203      *              conflicting transaction(s) will be added to this set.
204      *              This method assumes that any concurrent access is
205      *              being arbitrated by the set or by the caller.
206      * @param lockedEntrySet If the entry can not be read/taken
207      *              because of a transaction conflict, the ID of the
208      *              entry will be added to this set. This method
209      *              assumes that any concurrent access is being arbitrated
210      *              by the set or by the caller.
211      * @param provisionallyRemovedEntrySet If the entry can not be
212      *              read/taken because it has been provisionally
213      *              removed then its handle will be placed in the
214      *              passed <code>WeakHashMap</code> as a key (with
215      *              null as the value).  May be <code>null</code> in
216      *              which case provisionally removed entries will not
217      *              be recorded. This method assumes that any
218      *              concurrent access is being arbitrated by the set
219      *              or by the caller.
220      * @param now an estimate of the current time in milliseconds
221      *            since the beginning of the epoch.
222      * @return <code>true</code> if the entry could be read/taken and
223      *         <code>false</code> otherwise.
224      * @throws NullPointerException if entry is <code>null</code>.  
225      */
226     boolean attemptCapture(EntryHandle handle, TransactableMgr txn,
227 	boolean takeIt, Set conflictSet, Set lockedEntrySet, 
228         Set provisionallyRemovedEntrySet, long now)
229     {
230 	try {
231 	    return confirmAvailabilityWithTxn(handle.rep(), handle,
232 		txn, takeIt, now, conflictSet, lockedEntrySet, 
233 		provisionallyRemovedEntrySet);
234 	} catch (CannotJoinException e) {
235 	    return false;
236 	}
237     }

239     /* Now that we know we have a match, make sure that the the
240      * item in question is still in the space and hasn't been
241      * subject to other transactional-interference...
242      */
243     private boolean confirmAvailabilityWithTxn(EntryRep rep, 
244 	      EntryHandle handle, TransactableMgr txnMgr, boolean takeIt, 
245 	      long time, Set conflictSet, Set<Uuid> lockedEntrySet,
246 	      Set<EntryHandle> provisionallyRemovedEntrySet)
247 	throws CannotJoinException
248     {
249 	// Now that we know we have a match, make sure that the the
250 	// item in question is still in the space and hasn't been
251 	// subject to other transactional-interference...
252 
253 	// Lock the txn first (since that what everybody else does that)
254 	// to ensure that the locking always happens in A cannonical
255 	// order. We lock the transaction (txn) object to ensure that the
256 	// transaction hasn't been altered behind our backs. Other, cheaper
257 	// actions with the space (write, etc.) are handled nicely in 
258 	// OutriggerServerImpl. But, this one (reading/taking) was expensive
259 	// enough that we wanted to do it here, where the window is
260 	// smallest.
261 	// Bug 4394263...
262 	
263 	final Txn txn = (Txn)txnMgr;
264 	try {
265 	    if (txn != null) 
266 		txn.ensureActive();
267 
268 	    return confirmAvailability(rep, handle, txn,
269 		takeIt, time, conflictSet, lockedEntrySet, 
270 		provisionallyRemovedEntrySet);
271 	} finally {
272 	    if (txn != null) 
273 		txn.allowStateChange();
274 	}
275     }

277     /**
278      * With the EntryRep <code>rep</code> passed in, verify that the
279      * entry hasn't been taken by someone else, hasn't expired, etc.
280      * Also, verify that the entry is really (legally) visible to this
281      * transaction at this time. If this is a <code>take</code>, it it
282      * is removed or provisionally removed. If this operation is under
283      * a transaction, the entry is locked appropriately.
284      */
285     // @see grab -- a helper routine
286     private boolean
287 	confirmAvailability(EntryRep rep, EntryHandle handle,
288 	      TransactableMgr txn, boolean takeIt, long time,
289 	      Set conflictSet, Set<Uuid> lockedEntrySet,
290 	      Set<EntryHandle> provisionallyRemovedEntrySet)
291     {
292 	synchronized (handle) {
293             if (handle.removed()) return false;
294 	    // get rid of stale entries
295 	    if (isExpired(time, handle))
296 		return false;
297 		
298 	    if (handle.isProvisionallyRemoved()) {
299 		if (provisionallyRemovedEntrySet != null)
300 		    provisionallyRemovedEntrySet.add(handle);
301 		return false;
302 	    }
303 
304 	    int op = (takeIt ? TransactableMgr.TAKE : TransactableMgr.READ);
305 	    if (!handle.canPerform(txn, op)) {
306 		// Before deciding this is a conflict, we have to
307 		// make sure we are not conflicting with ourselves.
308 		if (handle.onlyMgr(txn))
309 		    // We are just conflicting with ourselves, this entry
310 		    // must have been taken by this txn already ... oh well
311 		    return false;
312 	       
313 		// Must be a real conflict, log if we were told too
314 		if (matchingLogger.isLoggable(Level.FINER)) {
315 		    matchingLogger.log(Level.FINER, "match, but can''t " +
316 			"perform {0}; handle.knownMgr(txn) == {1}", 
317 			new Object[] {Integer.valueOf(op), 
318 				      Boolean.valueOf(handle.knownMgr(txn))});
319 		}
320 
321 		if (conflictSet != null) {
322 		    handle.addTxns(conflictSet);
323 		} 
324 
325 		if (lockedEntrySet != null) {
326 		    lockedEntrySet.add(rep.id());
327 		}
328 
329 		return false;
330 	    }
331 
332 	    if (grab(handle, txn, op, takeIt, false))
333 		return true;
334 	    else 
335 		throw 
336 		    new AssertionError("entry became non-available while locked");
337 	}
338     }

340     /**
341      * Given an entry that we want to return as the result of a query
342      * and we have confirmed we can return it, make the results of the
343      * query visible to the rest of the service. We must either [a]
344      * remove it from the holder (if the operation is a take under the
345      * same transaction the entry was written under), [b] mark it as
346      * provisionally removed pending commiting the operation to disk
347      * (if the operation is a take without a transaction), [c]
348      * lock it under the transaction passed into this method (if the
349      * operation is a read or take under a transaction), or [d] do
350      * nothing (read not under a transaction) as appropriate. 
351      * <p>
352      * Also used during log recovery to recover takes.
353      *
354      * @param handle The handle attached to the particular EntryRep
355      * @param txn The Txn object
356      * @param op TAKE or READ (as a TransactableMgr constant)
357      * @param takeIt Is this a TAKE (or is it [false] a READ)
358      * @param recovery <code>true</code> if being called as
359      *        part of store recovery.
360      * @return <code>true</code> if the entry could be grabbed.
361      * @throws NullPointerException if <code>handle</code> is
362      *         <code>null</code>.  
363      */
364     private boolean grab(EntryHandle handle, TransactableMgr txn, int op,
365 			 boolean takeIt, boolean recovery)
366     {
367 	assert op==(takeIt?TransactableMgr.TAKE:TransactableMgr.READ);
368 	assert !recovery || takeIt;
369 
370 	/*
371 	 * If this manager (txn) is known to the handle, then txn
372 	 * is a transaction which can operate directly on the
373 	 * existing handle.  If so, then we can operate on it
374 	 * directly.  Note that if txn is null and the handle is
375 	 * not managed, then txn is "known" in this sense.
376 	 *
377 	 * Else, this operation is new relative to the
378 	 * known transactions (possibly null) and txn.  
379 	 */
380 
381 	if (handle.knownMgr(txn)) {    // added or read in this txn if taking
382 	    // take it if we have to (otherwise it must be a read
383 	    // there is no house keeping to be done since must already
384 	    // be read locked)
385 
386 	    if (takeIt) {
387 		// is this an entry that was written under this
388 		// transaction or another?
389 
390 		if (handle.managed()) {
391 		    // It is locked by a transaction and it must
392 		    // be the one this take is under.
393 
394 		    if (!handle.promoteToTakeIfNeeded()) {
395 			// this transaction wrote the entry (note txn
396 			// must be non-null, otherwise either
397 			// handle.managed or handle.knownMgr(txn)
398 			// would have returned null), just need to try
399 			// and remove it
400 			assert txn != null;
401 			
402 			// It is ok to call remove before committing this
403 			// op to disk since a crash will cause the original
404 			// write to be undone and the entry will still 
405 			// end up being removed.
406 			if (!remove(handle, recovery))
407 			    // Someone got to it first
408 			    return false;
409 		    }
410 
411 		    // Otherwise it was read locked (now take
412 		    // locked and needs to still be semi-visible
413 		    // to ifExists queries outside of the
414 		    // transaction, leave it in place with its new
415 		    // lock type (aka "state")
416 
417 		} else {
418 		    // Nether the entry nor this take are under a
419 		    // transaction (txn must be null because the
420 		    // that is the only way knownMgr return true, and
421 		    // managed() false)
422 		    assert txn == null;
423                     synchronized (handle){
424                         if (handle.removed() || handle.isProvisionallyRemoved())
425                             // Someone got to it first
426                             return false;
427                         handle.provisionallyRemove();
428                     }
429 		}
430 	    }
431 	} else {
432 	    if (txn != null) {
433 		// We need to add this txn to this handle and
434 		// add this handle to this txn.
435 		handle.add(txn, op, this); // add manager to handle's list
436 		txn.add(handle);	    // add handle to mgr's list
437 	    }
438 	}
439 	return true;
440     }

458     /**
459      * Return <code>true</code> if the entry held by the given
460      * <code>handle</code> has expired by the time in <code>now</code>.
461      */
462     private boolean isExpired(long now, EntryHandle handle) {
463 	/* Some callers already own the lock on handle when
464 	 * they call us, but the general feeling is that
465 	 * re-synchronizing is not costly and it is important
466 	 * to only call remove if we own the lock.
467 	 */
468 	synchronized (handle) {
469 	    final EntryRep rep = handle.rep();
470 	    synchronized (rep) {
471 		if (rep.getExpiration() > now) {
472 		    // Not expired
473 		    return false;
474 		}
475 
476 		/* Expired, set expiration to before the beginning of
477 		 * time so renew() (which does not lock on the handle,
478 		 * but the rep...) can't renew the lease once we leave
479 		 * this block, even it already has a ref to the
480 		 * rep. Everyone else locks on the handle so
481 		 * and makes sure the handle has not been removed
482 		 * before doing anything important.
483 		 */
484 		rep.setExpiration(Long.MIN_VALUE);
485 	    }
486 
487 	    // If we are here the lease must have expired
488 	    if (matchingLogger.isLoggable(Level.FINER)) {
489 		matchingLogger.log(Level.FINER, "expired {0} at {1} (now {2})",
490 		    new Object[] {rep.id(), 
491 				  Long.valueOf(rep.getExpiration()),
492 				  Long.valueOf(now)});
493 	    }
494 
495 	    if (!handle.isProvisionallyRemoved() && remove(handle, false)) {
496 		/* If we got to do the remove, schedule
497 		 * logging of the removal, otherwise
498 		 * someone else already did it, or will do it
499 		 */
500 		space.scheduleCancelOp(rep.id());
501 	    }
502 
503 	    return true;
504 	}
505     }

516     /**
517      * Add new new entry to the holder. Assumes the lock
518      * on the handle is held if there is a possibility
519      * of concurrent access.
520      * @param handle The <code>EntryHandle</code> for the
521      *               entry being added.
522      * @param txn If the add is being done under a 
523      *            transaction the <code>TransactableMgr</code> for
524      *            that transaction.
525      * @throws NullPointerException if <code>handle</code> is 
526      *         <code>null</code>.
527      */
528     void add(EntryHandle handle, TransactableMgr txn) {
529 	final EntryRep rep = handle.rep();
530 
531 	/* Make sure info duplicated across all the handles in this
532 	 * holder is shared.
533 	 *
534 	 * We make the head call before calling txn.add because
535 	 * it seems like better hygiene to fix reps state before
536 	 * exposing it to others (even though shareWith will
537 	 * not change handle's state materially).  
538 	 */
539 	final EntryHandle head = getContentsHead();
540 	if (head != null && head != handle) rep.shareWith(head.rep());
541         synchronized (handle){ //typically synchronized externally anyway.
542             if (txn != null) txn.add(handle);
543             content.add(handle);
544             EntryHandle existed = idMap.putIfAbsent(rep.getCookie(), handle);
545             if (existed != null) throw new IllegalStateException("An EntryHandle with that Cookie already exists in idMap");
546         }
547     }

569     /**
570      * The class that implements <code>RepEnum</code> for this class.
571      */
572     private class SimpleRepEnum implements RepEnum {
573         private final Iterator<EntryHandle> contentsIterator;
574 	private final TransactableMgr mgr;
575 	private final long startTime;
576 
577         SimpleRepEnum(TransactableMgr mgr) {
578             this.mgr = mgr;
579             startTime = System.currentTimeMillis();
580             contentsIterator = content.iterator();
581         }
582 
583         // inherit doc comment from superclass
584         public EntryRep nextRep() {
585             iteratorLogger.entering("SimpleRepEnum", "nextRep");
586             while (contentsIterator.hasNext()) {
587                 EntryHandle handle = contentsIterator.next();
588                 iteratorLogger.log(Level.FINEST,
589                         "advanced current handle to {0}", handle);
590                 /*
591                  * Skip over handles which are either removed or unable to
592                  * perform a READ operation.
593                  */
594                 synchronized (handle){
595                     if (handle.canPerform(mgr, TransactableMgr.READ)
596                             && !isExpired(startTime, handle) && !handle.removed()) {
597                         return handle.rep();
598                     }
599                 }
600 
601             }
602 
603             return null;
604         }
605     }

630     /**
631      * Object that can be used to perform a query that can
632      * return multiple matches and be restarted in another thread.
633      * Assumes that is being invoked from only one thread
634      * at a time, but does handle synchronization between concurrent
635      * queries on the parent <code>EntryHolder</code>.
636      */
637     class ContinuingQuery {
638 	/** Templates being used for the query */
639 	final private EntryRep[] tmpls;
640 
641 	/** Transaction (if any) being used for the query */
642 	final private TransactableMgr txn;
643 
644 	/**
645 	 * <code>true</code> if entries yielded by the query should 
646 	 * removed.
647 	 */
648 	final private boolean takeThem;
649 
650 	/** <code>EntryHandleTmplDesc</code> for the templates */
651         final private ThreadLocal<EntryHandleTmplDesc[]> descLocal;
652 	    
653 
654 	/** Time used to weed out expired entries, ok if old */
655 	volatile long now;
656 
657 	/** 
658 	 * Current position in parent <code>EntryHolder</code>'s
659 	 * <code>contents</code> 
660 	 */
661 	private final Iterator<EntryHandle> contentsIterator;

663 	/**
664 	 * Create a new <code>ContinuingQuery</code> object.
665 	 *
666 	 * @param tmpls    An array of templates. Query will yield any 
667 	 *                 entry that matches one or more of the templates.
668 	 * @param txn      Transaction that should be used with the query.
669 	 *                 May be <code>null</code>. If
670 	 *                 non-<code>null</code> any entries yielded by the
671 	 *                 query will be locked under the transaction.
672 	 * @param takeThem If <code>true</code> any entries yielded by
673 	 *                 the query should be removed.
674 	 * @param now      Estimate of current time used to weed out
675 	 *                 expired entries, ok if old
676 	 */
677         // 	 * @return a new ContinuingQuery object. (?)
678 	private ContinuingQuery(EntryRep[] tmpls, TransactableMgr txn,
679 				boolean takeThem, long now)
680 	{
681 	    this.tmpls = tmpls;
682 	    this.txn = txn;
683 	    this.takeThem = takeThem;
684 	    this.now = now;
685 	    contentsIterator = content.iterator();
686             descLocal = new ThreadLocal<EntryHandleTmplDesc[]>();
687 	}

689 	/**
690 	 * <code>EntryHolder</code> queries have thread local state
691 	 * that get clobbered if another query (on a different or same
692 	 * <code>EntryHolder</code>, including <code>hasMatch</code>
693 	 * calls) is started in the same thread. The state also needs
694 	 * to be restored if it the query is continued in another
695 	 * thread. The <code>restart</code> method must be called to
696 	 * restore the thread local state.
697 	 * @param now      Estimate of current time used to weed out
698 	 *                 expired entries, ok if old
699 	 */
700 	void restart(long now) {
701 	    if (!contentsIterator.hasNext()) return;
702 	    this.now = now;
703 	}
704 
705 	/**
706 	 * Return the next matching entry. Returns <code>null</code>
707 	 * if there are no matches remaining. Call {@link #restart
708 	 * restart} first if query is being used in a different thread
709 	 * from the last <code>next</code> call and/or the current
710 	 * thread has done any form of <code>EntryHolder</code> query
711 	 * on any holder since the last next call.
712 	 *
713 	 * @param conflictSet If non-null the <code>TransactableMgr</code>
714 	 *              objects of any transactions that prevent 
715 	 *              a entry from being retured will
716 	 *              be added to <code>conflictSet</code>. May
717 	 *              be <code>null</code> in which case
718 	 *              conflicting transaction will not be recorded.
719 	 *              This method assumes that any concurrent access
720 	 *              is being arbitrated by the set or by the caller.
721 	 * @param lockedEntrySet If non-null the ID of any entries that
722 	 *              can't be retured because of conflicting
723 	 *              transaction will be added to
724 	 *              <code>lockedEntrySet</code>. May be
725 	 *              <code>null</code> in which case unavailable
726 	 *              entries will not be recorded.  This method 
727 	 *              assumes that any concurrent access is being
728 	 *              arbitrated by the set or by the caller.
729 	 * @param provisionallyRemovedEntrySet If the entry can not be
730 	 *              read/taken because it has been provisionally
731 	 *              removed then its handle will be placed in the
732 	 *              passed <code>WeakHashMap</code> as a key (with
733 	 *              null as the value).  May be <code>null</code> in
734 	 *              which case provisionally removed entries will not
735 	 *              be recorded. This method assumes that any
736 	 *              concurrent access is being arbitrated by the set
737 	 *              or by the caller.
738 	 * @return a matching entry or <code>null</code>.
739 	 * @throws CannotJoinException if a match is found and
740 	 *         the operation is to be performed under a transaction,
741 	 *         but the transaction is no longer active.
742 	 */
743 	EntryHandle next(Set conflictSet, Set<Uuid> lockedEntrySet, Set<EntryHandle> provisionallyRemovedEntrySet) 
744 	    throws CannotJoinException
745 	{
746 	    matchingLogger.entering("ContinuingQuery", "next");
747             EntryHandleTmplDesc[] descs = descLocal.get();
748 	    while (contentsIterator.hasNext()) {
749 	        EntryHandle handle = contentsIterator.next();
750 	        if(descs == null){
751 	            // first time
752 	            descs = new EntryHandleTmplDesc[tmpls.length];
753 	            for (int i=0; i<tmpls.length; i++) {
754 	                descs[i] = EntryHandle.descFor(tmpls[i], 
755 	                                               handle.rep().numFields());
756 	            }
757                     descLocal.set(descs);
758 	        }
759 		if (handleMatch(handle, descs)) {
760 		    final boolean available =
761 			confirmAvailabilityWithTxn(handle.rep(), handle, txn, 
762 			    takeThem, now, conflictSet, lockedEntrySet, 
763 			    provisionallyRemovedEntrySet);
764 
765 		    if (available) return handle;
766 		}
767 	    }
768 
769 	    return null;
770 	}
771 
772 	/**
773 	 * Returns <code>true</code> if handle has not been removed
774 	 * and matches one or more of the templates 
775 	 */
776 	private boolean handleMatch(EntryHandle handle, EntryHandleTmplDesc[] descs) {
777 	    if (handle.removed()) return false;
778             int length = tmpls.length;
779 	    for (int i=0; i<length; i++) {
780 		final EntryRep tmpl = tmpls[i];
781 		final EntryHandleTmplDesc desc = descs[i];
782 
783 		// Quick reject
784 		if ((handle.hash() & desc.mask) != desc.hash)
785 		    continue;
786 		if (!tmpl.matches(handle.rep()))
787 		    continue;
788 		return true;
789 	    }
790 
791 	    return false;
792 	}
793     }

795     /**
796      * Remove the given handle from this holder and the <code>idMap</code>.
797      * If the handle isn't in this holder, this does nothing.
798      * @param h the <code>EntryHandle</code> to remove.
799      * @param recovery <code>true</code> if being called as part
800      *        of log recovery.
801      * @return <code>true</code> if this call removed <code>h</code> and
802      *         <code>false</code> otherwise.  
803      */
804     boolean remove(EntryHandle h, boolean recovery) {
805 	boolean ok = false;
806         synchronized (h){
807             ok = h.remove();
808             if (!ok) throw new AssertionError("EntryHandle not removed");
809             h.removalComplete();
810             // Ensure removal of EntryHandle is atomic.
811             boolean removed = idMap.remove(h.rep().getCookie(), h);
812             if (!removed) throw new IllegalStateException ("EntryHandle was missing from idMap at time of removal");
813             /* This may cause an ifExists query to be resolved,
814              * even if this entry was not locked under a transaction 
815              */
816             if (!recovery) {
817                 space.recordTransition(
818                     new EntryTransition(h, null, false, false, false));
819             }
820         }
821 	return ok;
822     }

824     /**
825      * Reap the expired elements.
826      */
827     void reap() {
828 
829 	// Examine each of the elements to determine if
830 	// any of them have expired. If they have, ensure that they are
831 	// removed ("reaped") from the collection. 
832 
833 	long now = System.currentTimeMillis();
834 	for(EntryHandle handle : content){
835 	    // Don't try to remove things twice
836 	    if (handle.removed()) continue;
837 	    // Calling isExpired() will both make the check and remove it
838 	    // if necessary.
839 	    isExpired(now, handle);
840 	}
841     }
