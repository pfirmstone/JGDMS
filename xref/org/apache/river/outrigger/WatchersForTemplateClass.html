18  package org.apache.river.outrigger;
19  
20  import java.util.Queue;
21  import java.util.Set;
22  import java.util.concurrent.ConcurrentLinkedQueue;
23  import org.apache.river.outrigger.proxy.EntryRep;
24  
25  /**
26   * Holds a collection of <code>TemplateHandle</code>s who's templates
27   * are all of exactly the same class. Unless otherwise noted all
28   * methods are thread safe. This method provides the linkage between
29   * <code>TemplateHandle</code>s and <code>TransitionWatchers</code>
30   * and for the most part is not visible to the clients of either.
31   */
32  class WatchersForTemplateClass {
33      /** All the templates we know about */
34      private final Queue<TemplateHandle> content = new ConcurrentLinkedQueue<TemplateHandle>();
35  
36      /** The OutriggerServerImpl we belong to */
37      private final OutriggerServerImpl owner;
38  
39      /**
40       * Create a new <code>WatchersForTemplateClass</code> object
41       * associated with the specified <code>TransitionWatchers</code> object.
42       * @param owner The <code>OutriggerServerImpl</code> that
43       *              this object will be a part of.
44       * @throws NullPointerException if <code>owner</code> is
45       *         <code>null</code>.
46       */
47      WatchersForTemplateClass(OutriggerServerImpl owner) {
48  	if (owner == null)
49  	    throw new NullPointerException("owner must be non-null");
50  	this.owner = owner;
51      }
52  
53      /**
54       * Add a <code>TransitionWatcher</code> to the list
55       * of watchers looking for visibility transitions in
56       * entries that match the specified template. Associates
57       * a <code>TemplateHandle</code> using 
58       * <code>TransitionWatcher.setTemplateHandle</code> method.
59       *
60       * @param watcher The <code>TransitionWatcher</code> being added.
61       * @param template The <code>EntryRep</code> that represents
62       *                 the template of interest.
63       * @throws NullPointerException if either argument is
64       *         <code>null</code>.
65       */
66      void add(TransitionWatcher watcher, EntryRep template) {
67  	/* We try to find an existing handle, but it is ok
68  	 * if we have more than one with the same template.
69           * It isn't possible to add a watcher to a removed
70           * handle even if present during iteration.
71  	 */	
72          for(TemplateHandle handle : content) {
73  	    if (template.equals(handle.rep()) &&
74                  handle.addTransitionWatcher(watcher)) return;
75  	}
76          
77  	TemplateHandle handle = new TemplateHandle(template, owner, content);
78          if (handle.addTransitionWatcher(watcher)) content.add(handle);
79          // else the new handle is discarded.
80      }
81      
82      /**
83       * Iterate over the watchers associated with 
84       * this object calling <code>isInterested</code> on each
85       * and if it returns <code>true</code> adding the watcher to the
86       * passed set.
87       *
88       * @param set The set to accumulate interested watchers
89       *            into.
90       * @param transition The transition being processed.
91       * @param ordinal The ordinal associated with <code>transition</code>.
92       * @throws NullPointerException if either argument is <code>null</code>.
93       */
94      void collectInterested(Set<TransitionWatcher> set, EntryTransition transition, 
95  			   long ordinal) 
96      {
97  	final EntryHandle entryHandle = transition.getHandle();
98  	final EntryRep rep = entryHandle.rep();
99  	final long entryHash = entryHandle.hash();
100 	final int repNumFields = rep.numFields();
101 
102 	/* Look at each of handles, check to see if they match
103 	 * the changed entry and if they do ask them to 
104 	 * put the appropriate watchers in the set.
105 	 */
106 	for (TemplateHandle handle : content)
107 	{
108 	    // See the if handle mask is incompatible
109 	    EntryHandleTmplDesc desc = handle.descFor(repNumFields); // final
110 
111 	    if ((entryHash & desc.mask) != desc.hash) continue;
112 
113 	    if (handle.matches(rep)) handle.collectInterested(set, transition, ordinal);
114 	}
115     }
116 
117     /**
118      * Visit each <code>TransitionWatcher</code> and check to see if
119      * it has expired, removing it if it has. Also reaps the
120      * <code>FastList</code> associated with this object.
121      * @param now an estimate of the current time expressed as
122      *            milliseconds since the beginning of the epoch.
123      */
124     void reap(long now) {
125 	// First remove empty handles
126 	for (TemplateHandle handle : content)
127 	{
128 	    // Dump any expired watchers.
129 	    handle.reap(now);
130             handle.removeIfEmpty();
131 	}
132     }
133 }
