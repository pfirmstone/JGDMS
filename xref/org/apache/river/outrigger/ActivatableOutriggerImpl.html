16  package org.apache.river.outrigger;
17  
18  import java.io.IOException;
19  import java.rmi.MarshalledObject;
20  import java.rmi.activation.ActivationException;
21  import java.rmi.activation.ActivationID;
22  import javax.security.auth.login.LoginException;
23  import net.jini.config.ConfigurationException;
24  import net.jini.export.DynamicProxyCodebaseAccessor;
25  
26  /**
27   *
28   * @author peter
29   */
30  class ActivatableOutriggerImpl extends PersistentOutriggerImpl  
31  			      implements DynamicProxyCodebaseAccessor {
32  
33      /**
34       * Create a new incarnation of an activatable
35       * <code>OutriggerServerImpl</code> server.
36       * @param activationID of the server, may be <code>null</code>.
37       * @param data an array of <code>String</code>s (packaged in
38       *        a marshalled object) that will be used 
39       *        to obtain a <code>Configuration</code>.
40       * @throws IOException if there is problem recovering data
41       *         from disk, exporting the server, or unpacking
42       *         <code>data</code>.
43       * @throws ClassCastException if the value of <code>data.get()</code>
44       *         is not an array of <code>String</code>s.
45       * @throws ConfigurationException if the <code>Configuration</code> is 
46       *         malformed.  
47       * @throws ActivationException if activatable and there
48       *         is a problem getting a reference to the activation system.
49       * @throws LoginException if the <code>loginContext</code> specified
50       *         in the configuration is non-null and throws 
51       *         an exception when login is attempted.
52       * @throws ClassNotFoundException if the classes of the objects
53       *         encapsulated inside <code>data</code> can not be found.
54       */
55      ActivatableOutriggerImpl(ActivationID activationID, 
56  				    MarshalledObject data) 
57  	    throws IOException, ConfigurationException, LoginException,
58  	    ActivationException, ClassNotFoundException 
59      {
60  	super(activationID, data);
61      }
62      
63  }
