18  package org.apache.river.outrigger.snaplogstore;
19  
20  import org.apache.river.outrigger.proxy.StorableObject;
21  import org.apache.river.outrigger.StoredObject;
22  
23  import java.io.ByteArrayInputStream;
24  import java.io.ByteArrayOutputStream;
25  import java.io.IOException;
26  import java.io.ObjectInputStream;
27  import java.io.ObjectOutputStream;
28  import java.io.Serializable;
29  
30  import net.jini.space.InternalSpaceException;
31  
32  /**
33   * Top level wrapper class for persisting outrigger objects.
34   * The target object is serialized and stored here as a byte
35   * array.
36   */
37  class BaseObject<T extends StorableObject<T>> implements StoredObject<T>, Serializable {
38      static final long serialVersionUID = -400804064969360164L;
39  
40      /**
41       * @serialField containing a binary blob.
42       */
43      private final byte[]	blob;
44  
45      BaseObject(T object) {
46  	try {
47  	    ByteArrayOutputStream baos = new ByteArrayOutputStream();
48  	    ObjectOutputStream oos = new ObjectOutputStream(baos);
49  	    object.store(oos);
50  	    oos.flush();
51  	    blob = baos.toByteArray();
52  	    oos.close();
53  	} catch (IOException e) {
54  	    throw new InternalSpaceException("Exception serializing resource", e);
55  	}
56      }
57  
58      public T restore(T object)
59        throws IOException, ClassNotFoundException {
60  	ObjectInputStream ois = new ObjectInputStream(new ByteArrayInputStream(blob));
61  	T result = object.restore(ois);
62  	ois.close();
63          return result;
64      }
65      
66      /**
67       * Added to enable the serial form to be modified
68       * in a backward compatible manner (if necessary) with 3.0.0 and later.  
69       * Modified serial form would be a breaking change for versions
70       * prior to 3.0.0 
71       * 
72       * @serialData 
73       * @param ois
74       * @throws IOException
75       * @throws ClassNotFoundException 
76       * @since 3.0.0
77       */
78      private void readObject(ObjectInputStream ois) throws IOException, ClassNotFoundException {
79          ois.defaultReadObject();
80      }
81  }
