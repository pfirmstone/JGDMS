18  package org.apache.river.outrigger;
19  
20  import java.util.Set;
21  import net.jini.core.transaction.TransactionException;
22  import net.jini.space.InternalSpaceException;
23  
24  /**
25   * Subclass of QueryWatcher for takes and transactional reads.
26   * Resolves with the first matching transition where the entry is
27   * visible to the associated transaction and the entry is still
28   * available.
29   */
30  class ConsumingWatcher extends SingletonQueryWatcher implements Transactable {
31      /**
32       * If non-null the transaction this query is
33       * being performed under. If <code>null</code> 
34       * this query is not associated with a transaction.
35       */
36      private final Txn txn;
37  
38      /**
39       * <code>true</code> if this query is a take and
40       * <code>false</code> otherwise.
41       */
42      private final boolean takeIt;
43  
44      /**
45       * Set of entries (represented by <code>EntryHolder</code>s) that
46       * we would have liked to return, but have been provisionally
47       * removed.
48       */
49      private final Set<EntryHandle> provisionallyRemovedEntrySet;
50  
51      /**
52       * Create a new <code>ConsumingWatcher</code>.
53       * @param expiration the initial expiration time
54       *        for this <code>TransitionWatcher</code> in 
55       *        milliseconds since the beginning of the epoch.
56       * @param timestamp the value that is used
57       *        to sort <code>TransitionWatcher</code>s.
58       * @param startOrdinal the highest ordinal associated
59       *        with operations that are considered to have occurred 
60       *        before the operation associated with this watcher.
61       * @param provisionallyRemovedEntrySet If the watcher encounters
62       *        an entry that can not be read/taken because it has been
63       *        provisionally removed then its handle will be placed in
64       *        this <code>WeakHashMap</code> as a key (with null as the
65       *        value).  May be <code>null</code> in which case
66       *        provisionally removed entries will not be
67       *        recorded. Ensures that object is only accessed by one
68       *        thread at a time
69       * @param txn If the query is being performed under
70       *        a transaction the <code>Txn</code> object
71       *        associated with that transaction.
72       * @param takeIt <code>true</code> if this query is a take and
73       *        <code>false</code> otherwise.  
74       */
75      ConsumingWatcher(long expiration, long timestamp, long startOrdinal, 
76  		     Set<EntryHandle> provisionallyRemovedEntrySet, Txn txn,
77  		     boolean takeIt)
78      {
79  	super(expiration, timestamp, startOrdinal);
80  	this.txn = txn;
81  	this.takeIt = takeIt;
82  	this.provisionallyRemovedEntrySet = provisionallyRemovedEntrySet; 
83      }
84  
85      boolean isInterested(EntryTransition transition, long ordinal) {
86  	/* Note, !isResolved() without the lock will result only in
87  	 * false positives, not false negatives - it will only
88  	 * cause isInterested() to return false if we are resolved,
89  	 * we may still return true if we are resolved though.
90  	 *
91  	 * Note, we don't bother with isVisible because isAvailable is
92  	 * the right test for a take, and using isAvailable for read too
93  	 * simplifies things a bit. The worst that will happen is we
94  	 * will return true for a transition that should not change
95  	 * anything - but we are unlikely to get these transitions
96  	 * (since the fact the transition happened at all suggests
97  	 * that there is some entry that would have resolved this
98  	 * watcher), if we do generate a false positive we do a
99  	 * positive check against the entry in question anyway - so it
100 	 * can't cause us to produce the wrong answer, and it may well be
101 	 * that such a "false positive" allows us to get legitimately
102 	 * resolved anyway - so it is not so "false" a positive after all.
103 	 */
104 	final TransactableMgr transitionTxn = transition.getTxn();
105 	return (ordinal>startOrdinal) && !isResolved() &&
106 	    transition.isAvailable() &&
107 	    ((null == transitionTxn) || (txn == transitionTxn));
108     }
109 
110     synchronized void process(EntryTransition transition, long now) {
111 	if (isResolved())
112 	    return; // Already done.
113 
114 	final EntryHandle handle = transition.getHandle();
115 
116 	// Is it still available?
117 	if (getServer().attemptCapture(handle, txn, takeIt, null, 
118 				       provisionallyRemovedEntrySet, now, this)) 
119 	{
120 	    // Got it
121 	    resolve(handle, null);
122 	} 
123     }
124 
125     synchronized boolean catchUp(EntryTransition transition, long now) {
126 	if (isResolved())
127 	    return true;
128 
129 	final TransactableMgr transitionTxn = transition.getTxn();
130 	final EntryHandle handle = transition.getHandle();
131 
132 	/* See note in isInterested about not calling isVisible */
133 	if (transition.isAvailable() &&
134 	    ((null == transitionTxn) || (txn == transitionTxn)) &&
135 	    (getServer().attemptCapture(handle, txn, takeIt, null,
136 		provisionallyRemovedEntrySet, now, this)))
137 	{
138 	    // Got it
139 	    resolve(handle, null);
140 	    return true;
141 	} 
142 
143 	// Not interesting or could not get it, either way not resolved
144 	return false;
145     }
146

148     /**
149      * If a transaction ends in the middle of a query we want
150      * to throw an exception to the client making the query
151      * not the <code>Txn</code> calling us here.)
152      */
153     public synchronized int prepare(TransactableMgr mgr,
154 				    OutriggerServerImpl space) 
155     {
156 	assert txn != null:"Transactable method called on a " +
157 	    "non-transactional ConsumingWatcher";
158 
159 	// only throw an exception if we are not resolved.
160 	if (!isResolved()) {
161 	    // Query still in progress, kill it
162 	    resolve(null, new TransactionException("completed while " +
163 						   "operation in progress"));
164 	}
165 
166 	// If this object has made changes they have been recorded elsewhere
167 	return NOTCHANGED;
168     }
169 
170     /**
171      * This should never happen since we always return
172      * <code>NOTCHANGED</code> from <code>prepare</code>.
173      */
174     public void commit(TransactableMgr mgr, OutriggerServerImpl space) {
175 	throw new InternalSpaceException("committing a blocking query");
176 			   
177     }
178 
179     /**
180      * If a transaction ends in the middle of a query we want
181      * to throw an exception to the client making the query 
182      * (not the <code>Txn</code> calling us here.)
183      */
184     public void abort(TransactableMgr mgr, OutriggerServerImpl space) {
185 	// prepare does the right thing, and should forever
186 	prepare(mgr, space);
187     }
188 }
