18  package org.apache.river.outrigger;

20  import org.apache.river.logging.Levels;
21  import java.io.IOException;
22  import java.util.Iterator;
23  import java.util.Set;
24  import java.util.concurrent.ConcurrentHashMap;
25  import java.util.concurrent.ConcurrentMap;
26  import java.util.concurrent.ConcurrentSkipListSet;
27  import java.util.logging.Logger;
28  import net.jini.core.transaction.server.ServerTransaction;
29  import net.jini.core.transaction.server.TransactionManager;
30  import net.jini.security.ProxyPreparer;
31  
32  /**
33   * Keeps a mapping from {@link TransactionManager}/id pairs, to {@link Txn}
34   * objects. Some <code>Txn</code>s may be <i>broken</i>, that is the
35   * <code>TransactionManager</code> they are associated with can not be
36   * unmarshalled and prepared, and thus can't currently be used (but may be
37   * usable in the future).
38   * 
39   * @author Sun Microsystems, Inc.
40   * @since 2.0
41   */
42  class TxnTable {
43  	/**
44  	 * Key class for the primary map (from manager/id pairs to <code>Txn</code>
45  	 * s. We use a new class instead of <code>ServerTransaction</code> objects
46  	 * so we can make sure we don't call <code>equals</code> on unprepared
47  	 * managers.
48  	 */
49  	private static class Key {
50  		/** The manager for the transaction */
51  		private final TransactionManager manager;
52  
53  		/** The id for the transaction */
54  		private final long id;
55  
56  		/**
57  		 * True if it has been asserted that <code>manager</code> has been
58  		 * prepared. Note, we only put Keys in the table that have this flag
59  		 * set.
60  		 */
61  		private final boolean prepared;
62  
63  		/**
64  		 * Create a new key from the specified manager and id.
65  		 * 
66  		 * @param manager
67  		 *            the manager for the transaction
68  		 * @param id
69  		 *            the id for the transaction
70  		 * @param prepared
71  		 *            should be <code>true</code> if the manager has been
72  		 *            prepared and false otherwise
73  		 * @throws NullPointerException
74  		 *             if manager is <code>null</code>.
75  		 */
76  		private Key(TransactionManager manager, long id, boolean prepared) {
77  			if (manager == null)
78  				throw new NullPointerException("manager must be non-null");
79  			this.manager = manager;
80  			this.id = id;
81  			this.prepared = prepared;
82  		}
83  
84  		// Inherit doc from super
85  		public int hashCode() {
86  			return (int) id ^ manager.hashCode();
87  		}
88  
89  		// Inherit doc from super
90  		public boolean equals(Object other) {
91  			if (!(other instanceof Key))
92  				return false;
93  
94  			final Key o = (Key) other;
95  			if (id != o.id)
96  				return false;
97  
98  			/*
99  			 * Either this or o should have prepared == true, call
100 			 * manager.equals on the one who has been prepared.
101 			 */
102 			if (o.prepared)
103 				// Common case, usually the passed in object is in the map
104 				return o.manager.equals(manager);
105 			else if (prepared)
106 				return manager.equals(o.manager);
107 			else
108 				throw new AssertionError("TxnTable.Key equals call with two "
109 						+ "unprepared managers");
110 		}
111 	}
112 
113 	/**
114 	 * Map of manager,id pairs (represented as <code>Keys</code>s) to non-broken
115 	 * <code>Txn</code>s. Only <code>Key</code>s that have their prepared flag
116 	 * set should go in this Map.
117 	 */
118 	final private ConcurrentMap<Key,Txn> txns = new ConcurrentHashMap<Key,Txn>();
119 
120 	/**
121 	 * Map of transaction ids to the <code>List</code> of broken
122 	 * <code>Txn</code> objects that have the id. <code>null</code> if there are
123 	 * no broken <code>Txn</code>s.
124 	 */
125 	private final ConcurrentMap<Long,Set<Txn>> brokenTxns = new ConcurrentHashMap<Long,Set<Txn>>();
126 
127 	/**
128 	 * <code>ProxyPreparer</code> to use when unpacking transactions, may be
129 	 * <code>null</code>.
130 	 */
131 	private final ProxyPreparer proxyPreparer;
132         
133 	/** The logger to use */
134 	static private final Logger logger = Logger
135 			.getLogger(OutriggerServerImpl.txnLoggerName);
136 
137 	/**
138 	 * Create a new <code>TxnTable</code>.
139 	 * 
140 	 * @param proxyPreparer
141 	 *            the proxy preparer to use on recovered
142 	 *            <code>TransactionManager</code>s.
143 	 */
144 	TxnTable(ProxyPreparer proxyPreparer) {
145 		this.proxyPreparer = proxyPreparer;
146 	}
147 
148 	/**
149 	 * Given a <code>TransactionManager</code>, <code>manager</code>, and a
150 	 * transaction id, return the associated <code>Txn</code>, or
151 	 * <code>null</code> if there is no <code>Txn</code> for this manager/id
152 	 * pair. The only method this method will call on <code>manager</code> is
153 	 * <code>hashCode</code>. The returned <code>Txn</code> object will not be
154 	 * broken.
155 	 * 
156 	 * @param manager
157 	 *            a <code>TransactionManager</code>.
158 	 * @param id
159 	 *            a transaction id.
160 	 * @return the <code>Txn</code> for the specified pair or null if none can
161 	 *         be found.
162 	 * @throws IOException
163 	 *             if there was one or more possible matches, but they are
164 	 *             broken and the attempt to unpack them yielded an
165 	 *             <code>IOException</code>.
166 	 * @throws ClassNotFoundException
167 	 *             if there was one or more possible matches, but they are
168 	 *             broken and the attempt to unpack them yielded a
169 	 *             <code>ClassNotFoundException</code>.
170 	 * @throws SecurityException
171 	 *             if there was one or more possible matches, but they are
172 	 *             broken and the attempt to unpack them yielded a
173 	 *             <code>SecurityException</code>.
174 	 */
175 	Txn get(TransactionManager manager, long id) throws IOException,
176 			ClassNotFoundException {
177 		final Long idAsLong;
178                 Set<Txn> txnsForId;
179 		// Try the table of non-broken txns first
180                 {
181                     final Txn r = txns.get(new Key(manager, id, false));
182                     if (r != null) return r;
183                            
184                     idAsLong = Long.valueOf(id);
185                     txnsForId = brokenTxns.get(idAsLong);
186                     if (txnsForId == null)
187                             /*
188                              * Broken Txns, but none with the right ID so txns is definitive
189                              * for this manager/id pair.
190                              */
191                             return null;
192                 }
193 
194 		/*
195 		 * try and fix each element of brokenTxnsForId until we find the right
196 		 * Txn, or until we run out of elements
197 		 */
198 		Txn match = null;
199 		Throwable t = null; // The first throwable we get, if any
200                 Iterator<Txn> txnsit = txnsForId.iterator();
201                 while (txnsit.hasNext()){
202 			try {
203                                 Txn txn = txnsit.next();
204 				final ServerTransaction st = txn
205 						.getTransaction(proxyPreparer);
206 
207 				/*
208 				 * We fixed a Txn, so we can move it to txns.
209 				 */
210                                 put(txn);
211                                 txnsit.remove();
212 				/*
213 				 * Did this match? Pass unprepared manager to prepared one!
214 				 */
215 				if (st.mgr.equals(manager)) {
216 					// bingo!
217 					match = txn;
218 					break;
219 				}
220 			} catch (Throwable tt) {
221 				try {
222 					if (logger.isLoggable(Levels.FAILED))
223 						logger.log(Levels.FAILED, "Encountered " + tt
224 								+ "while recovering/re-preparing "
225 								+ "transaction, will retry latter", tt);
226 				} catch (Throwable ttt) {
227 					// don't let a problem in logging kill the thread
228 				}
229 
230 				if (t == null)
231 					t = tt;
232                                 // We could potentially swallow an Error if we find a match
233                                 // Is this advisable?  A denial of service
234                                 // attack could throw an Error.  But then it's not
235                                 // necessarily a good thing to swallow an Error either.
236 			}
237 		}
238                 
239                 // Now remove empty collection from brokenTxns so it doesn't leak memory
240                 while (txnsForId.isEmpty()){
241                     boolean removed = brokenTxns.remove(idAsLong, txnsForId);
242                     if (!removed) break;
243                     try {
244                         Thread.sleep(10L); // Allow time for other threads to complete.
245                     } catch (InterruptedException ex) {
246                         Thread.currentThread().interrupt(); // Restore interrupt.
247                     }
248                     // Check that set is still empty.           
249                     if (removed && !txnsForId.isEmpty()){
250                         // Somebody had a reference and added to the set.
251                         Set<Txn> existed = brokenTxns.putIfAbsent(idAsLong, txnsForId);
252                         if (existed != null){
253                             // Someone else replaced our set!
254                             // Add all to this new collection
255                             existed.addAll(txnsForId);
256                             txnsForId = existed;
257                         }
258                     } else {
259                         break;
260                     }
261                 }
262 
263 		// Did we run out of candidates, or did we find a match?
264 		if (match != null) return match;
265 
266 		/*
267 		 * Was there some Txn that may have been a match but is still broken so
268 		 * we don't know? [this could happen if : o the managers codebase
269 		 * changed so old proxies can't be unmarshalled (but new ones can), o
270 		 * the log is corrupted, o a bad preparer has been supplied, o etc.,
271 		 * etc. ]
272 		 */
273 		if (t != null) {
274 			/*
275 			 * We may have some recored of this manager/id pair, but we can't
276 			 * tell since one or more of the broken Txns that could be a match
277 			 * could not be fixed. Throw an exception to indicate an ambiguous
278 			 * result.
279 			 */
280 			if (t instanceof IOException)
281 				throw (IOException) t;
282 			else if (t instanceof ClassNotFoundException)
283 				throw (ClassNotFoundException) t;
284 			else if (t instanceof Error)
285 				throw (Error) t;
286 			else if (t instanceof RuntimeException)
287 				throw (RuntimeException) t;
288 			else
289 				throw new AssertionError(t);
290 		}
291 
292 		/*
293 		 * If we are here there were one or more broken Txns with the same ID,
294 		 * but all them could be fixed and none of them had a manager that
295 		 * matched. Definitively return null, we have no record of this
296 		 * manager/id pair.
297 		 */
298 		return null;
299 	}
300 
301 	/**
302 	 * Atomically test if there is a <code>Txn</code> for the specified
303 	 * <code>ServerTransaction</code> in the table, creating a new
304 	 * <code>Txn</code>, and place in table if there is not. If there is already
305 	 * a <code>Txn</code> for the specified <code>ServerTransaction</code>
306 	 * return the existing one, otherwise return the new one. Does not check for
307 	 * matches against broken txns.
308 	 * 
309 	 * @param tr
310 	 *            <code>ServerTransaction</code> to add to the table. The
311 	 *            contained manager proxy should already been prepared
312 	 * @return the <code>Txn</code> for <code>tr</code>.
313 	 */
314 	Txn put(ServerTransaction tr) {
315 		final Key k = new Key(tr.mgr, tr.id, true);
316 		final long internalID = OutriggerServerImpl.nextID();
317 
318 		// Atomic test and set
319                 Txn r = txns.get(k);
320                 if (r == null) {
321                     // not in table, put a new one in and return it.
322                     r = new Txn(tr, internalID);
323                     Txn existed = txns.putIfAbsent(k, r);
324                     if (existed != null) r = existed;
325                 }
326                 return r;
327 	}
328 
329 	/**
330 	 * Used to put a formerly broken <code>Txn</code> in the main table. Only
331 	 * puts it in the table if it is not already there.
332 	 * 
333 	 * @param txn
334 	 *            the <code>Txn</code> being moved, should have been prepared
335 	 */
336 	private void put(Txn txn) {
337 		final Key k = new Key(txn.getManager(), txn.getTransactionId(), true);
338                 final Txn r = txns.get(k);
339                 if (r == null) {
340                         // Doesn't add the key if already in there.
341                         txns.putIfAbsent(k, txn);
342                 }
343 	}
344 
345 	/**
346 	 * Restore a <code>Txn</code> in the table as part of log recovery. An
347 	 * attempt will be made unpack and prepare the transaction manager. The
348 	 * <code>Txn</code> may be broken. This method is not synchronized.
349 	 * 
350 	 * @param txn
351 	 *            the <code>Txn</code> being recovered.
352 	 */
353 	void recover(Txn txn) {
354 		// Try to get the transaction
355 		ServerTransaction st = null;
356 		try {
357 			st = txn.getTransaction(proxyPreparer);
358 		} catch (Throwable t) {
359 			// log, but otherwise ignore
360 			try {
361 				if (logger.isLoggable(Levels.FAILED))
362 					logger.log(Levels.FAILED, "Encountered " + t + " while "
363 							+ "recovering/re-preparing transaction, "
364 							+ "will retry latter", t);
365 			} catch (Throwable tt) {
366 				// don't let a problem in logging kill the thread
367 			}
368 		}
369 
370 		if (st == null) {
371 			// txn is broken, put in brokenTxns
372 			final Long id = Long.valueOf(txn.getTransactionId());
373 			Set<Txn> txnsForId = brokenTxns.get(id);
374 			if (txnsForId == null) {
375                             txnsForId = new ConcurrentSkipListSet<Txn>();
376                             Set<Txn> existed = brokenTxns.putIfAbsent(id, txnsForId);
377                             if (existed != null) {
378                                 txnsForId = existed;
379                             }        
380 			} 
381                         txnsForId.add(txn);
382 		} else {
383 			// Txn is ok, put it in txns
384 			final Key k = new Key(st.mgr, st.id, true);
385 			txns.put(k, txn);
386 		}
387 	}
388 
389 	/**
390 	 * Remove the mapping for the given <code>TransactionManager</code>, id
391 	 * pair. Will not remove broken <code>Txn</code>s.
392 	 * 
393 	 * @param manager
394 	 *            the <code>TransactionManager</code> for transaction being
395 	 *            removed.
396 	 * @param id the manager assigned to the transaction being removed.
397 	 */
398 	void remove(TransactionManager manager, long id) {
399 		final Key k = new Key(manager, id, false);
400                 txns.remove(k);
401 	}
