18  package org.apache.river.outrigger;
19  
20  import net.jini.id.Uuid;
21  import org.apache.river.outrigger.proxy.StorableObject;
22  import org.apache.river.outrigger.proxy.StorableResource;
23  
24  /**
25   * Methods that log an operation.  These are used
26   * when writing to the store. <p>
27   *
28   * The <code>writeOp</code> and <code>takeOp</code> methods can
29   * be called under a transaction. For these methods an in-progress
30   * transaction is indicated by a non-null <code>txnId</code> parameter.
31   * The <code>txnId</code> is an identifier for a transaction.
32   * Each unique transaction must have a unique identifier and all
33   * write or take operations under the same transaction should
34   * use the same identifier. The store does not interpret the
35   * identifier in any way. <p>
36   *
37   * <i>Note: Because the transaction identifier
38   * must be unique, it can not be <code>ServerTransaction.id</code>.
39   * Instead this identifier is the <code>Txn</code> ID. </i> <p>
40   *
41   * When a transaction is closed, the <code>prepareOp</code>,
42   * <code>commitOp</code>, and <code>abortOp</code> methods are
43   * passed the identifier (<code>txnId</code>) for that transaction. If
44   * <code>prepareOp</code> is called and there is a restart, the
45   * <code>txnId</code> passed to the write and take operations will
46   * be passed back to the server via the <code>Recover.recoverWrite</code>
47   * and <code>Recover.recoverTake</code> methods. Likewise the same
48   * identifier will be passed to <code>Recover.recoverTransaction</code>.
49   *
50   * @see Store
51   * @see Recover
52   *
53   * @author Sun Microsystems, Inc.
54   *
55   */
56  public interface LogOps {
57  
58      /**
59       * Log a server boot (first time start or any reactivation).
60       *
61       * @see Recover#recoverSessionId
62       *
63       * @param time stamp for this boot
64       *
65       * @param sessionId of this boot
66       */
67      void bootOp(long time, long sessionId);
68  
69      /**
70       * Log an update to the join state
71       *
72       * @see Recover#recoverJoinState
73       *
74       * @param state to be logged
75       *
76       */
77      void joinStateOp(StorableObject state);
78  
79      /**
80       * Log a <code>write</code> operation. If the operation was
81       * performed under a transaction the <code>txnId</code> is
82       * the identifier for that transaction.
83       *
84       * @see Recover#recoverWrite
85       *
86       * @param entry to be logged
87       *
88       * @param txnId transaction identifier or <code>null</code> if
89       *        no transaction is active for this write
90       */
91      void writeOp(StorableResource entry, Long txnId);
92  
93      /**
94       * Log a batch <code>write</code> operation. If the operation was
95       * performed under a transaction the <code>txnId</code> is
96       * the identifier for that transaction.
97       *
98       * @see Recover#recoverWrite
99       * @param entries to be logged
100      * @param txnId transaction identifier or <code>null</code> if
101      *        no transaction is active for this write
102      */
103     void writeOp(StorableResource entries[], Long txnId);
104 
105     /**
106      * Log a <code>take</code> operation. If the operation was
107      * performed under a transaction the <code>txnId</code> is
108      * the identifier for that transaction.
109      *
110      * @see Recover#recoverTake
111      *
112      * @param cookie ID identifying the entry target to be taken
113      *
114      * @param txnId transaction identifier or <code>null</code> if
115      *        no transaction is active for this take
116      */
117     void takeOp(Uuid cookie, Long txnId);
118 
119     /**
120      * Log a batch <code>take</code> operation. If the operation was
121      * performed under a transaction the <code>txnId</code> is
122      * the identifier for that transaction.
123      *
124      * @see Recover#recoverTake
125      *
126      * @param cookies IDs identifying the entries to be taken
127      *
128      * @param txnId transaction identifier or <code>null</code> if
129      *        no transaction is active for this take
130      */
131     void takeOp(Uuid[] cookies, Long txnId);
132 
133     /**
134      * Log a <code>notify</code> operation.  Notifications under
135      * transactions are lost at the end of the transaction, so the
136      * only ones that are logged are those that are under no
137      * transaction.
138      *
139      * @see Recover#recoverRegister
140      *
141      * @param registration to be logged
142      *
143      * @param type of registration, passed back via <code>type</code>
144      *             parameter of corresponding <code>recoverRegister</code>
145      *             call
146      *
147      * @param templates associated with this registration 
148      */
149     void registerOp(StorableResource registration, String type,
150 		    StorableObject[] templates);
151 
152     /**
153      * Log a <code>renew</code> operation.  We use the expiration, not
154      * the extension, because we don't want to calculate the
155      * expiration relative to when we read the log -- we want to use
156      * the exact expiration granted.
157      *
158      * @see StoredResource
159      *
160      * @param cookie ID of the entry or registration associated with this
161      *               renew
162      *
163      * @param expiration time 
164      */
165     void renewOp(Uuid cookie, long expiration);
166 
167     /**
168      * Log a <code>cancel</code> and entry or registration. The entry or
169      * registration associated with <code>cookie</code> will no longer
170      * be recoverable and may be removed from the log records.
171      *
172      * @param cookie ID of the entry or registration to cancel
173      * @param expired is true if the cancel was due to a lease expiration
174      */
175     void cancelOp(Uuid cookie, boolean expired);
176 
177     /**
178      * Log a transaction <code>prepare</code>. If there is a restart
179      * before either <code>commitOp</code> or <code>abortOp</code> is
180      * called for the transaction identified by <code>txnId</code>,
181      * all write and take operations associated with <code>txnId</code>
182      * will be recovered and <code>Recover.recoverTransaction</code>
183      * called with the the same <code>txnId</code>.
184      *
185      * @see Recover#recoverTransaction
186      *
187      * @param txnId identifier of the transaction to be prepared
188      *
189      * @param transaction object associated with this transaction
190      */
191     void prepareOp(Long txnId, StorableObject transaction);
192 
193     /**
194      * Log a transaction <code>commit</code> or
195      * <code>prepareAndCommit</code>.  The store will commit the write
196      * and take operations associated with <code>txnId</code>. A call
197      * to <code>prepareOP</code> is not required for
198      * <code>commitOp</code> to be called.
199      *
200      * @param txnId identifier of the transaction to be committed 
201      */
202     void commitOp(Long txnId);
203 
204     /**
205      * Log a transaction <code>abort</code>. Any write and take operations
206      * associated with <code>txnId</code> will no longer be recoverable
207      * and may be removed from the log records.
208      *
209      * @param txnId identifier of the transaction to be aborted
210      */
211     void abortOp(Long txnId);
212 
213     /**
214      * Log the <code>Uuid</code> that identifies the space as a whole.
215      * @see Recover#recoverUuid
216      * @param uuid The <code>Uuid</code> to be stored.
217      */
218     void uuidOp(Uuid uuid);
219 }
