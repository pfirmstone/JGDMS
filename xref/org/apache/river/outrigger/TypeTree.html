20  import java.util.Collection;
21  import java.util.Iterator;
22  import java.util.LinkedList;
23  import java.util.NoSuchElementException;
24  import java.util.Random;
25  import java.util.Set;
26  import java.util.concurrent.ConcurrentHashMap;
27  import java.util.concurrent.ConcurrentMap;
28  import java.util.concurrent.ConcurrentSkipListSet;

31  /**
32   * A type tree for entries.  It maintains, for each class, a list of
33   * known subclasses so that we can walk down the relevant subpart of
34   * the subtype tree for a template, looking for matching entries.  This
35   * list of subtypes is not current garbage collected -- if a subtype
36   * was once written, it's subtype entry will never be removed from this
37   * tree.  All operations are done via class name.
38   *
39   * @author Sun Microsystems, Inc.
40   *
41   * @see OutriggerServerImpl
42   */
43  class TypeTree {
44      /** For each type, a set of known subtypes.
45       * 
46       * Note: This was originally Hashtable<String,Vector<String>>, although
47       * a synchronized Map<String<Set<String>> could have been used in place
48       * of the concurrent collections, on this occasion, the concurrent
49       * implementations were easier to implement and understand atomically.
50       * 
51       * The decision to use concurrent collections was not made for performance
52       * reasons, but rather simplicity of implementation, see addKnown method.
53       */
54      private final ConcurrentMap<String,Set<String>> subclasses = new ConcurrentHashMap<String,Set<String>>();
55  
56      /**
57       * A generator used to randomize the order of iterator returns
58       */
59      // @see #RandomizedIterator
60      static final private Random numgen = new Random();
61  
62      /**
63       * Name of the root bucket of the type tree
64       */
65      static final private String ROOT = 
66  	net.jini.core.entry.Entry.class.getName();
67  
68      /**
69       * Return the set of subclasses for the given class.
70       */
71      private Set<String> classSet(String whichClass) {
72          return subclasses.get(whichClass);
73      }
74  
75      /**
76       * An iterator that will walk through a list of known types.
77       */
78      // @see #RandomizedIterator
79      private static abstract class TypeTreeIterator<T> implements Iterator<T> {
80  	protected int cursor;		// the current position in the list
81  	protected final T [] typearray;	// the list of types as an array
82          
83          protected TypeTreeIterator (T [] types){
84              cursor = 0;
85              typearray = types;
86          }
87  
88  	// inherit doc comment
89          public boolean hasNext() {
90              if (cursor < typearray.length) return true;
91              return false;
92          }
93   
94  	// inherit doc comment
95          public T next() throws NoSuchElementException {
96              T val = null;
97   
98              if (cursor >= typearray.length)
99                  throw new NoSuchElementException("TypeTreeIterator: next");
100  
101             try {
102                 val = typearray[cursor];
103                 cursor++;
104             } catch (ArrayIndexOutOfBoundsException e) {
105                 throw new NoSuchElementException("TypeTreeIterator: next" +
106                                                         e.getMessage());
107             }
108 
109             return val;
110         }
111  
112 	/**
113 	 * Unimplemented operations
114 	 * @throws UnsupportedOperationException Always
115 	 */
116         public void remove() throws UnsupportedOperationException {
117             throw new UnsupportedOperationException(
118 		"TypeTreeIterator: remove not supported");
119         }
120     }
121 
122     /**
123      * This class implements a randomized iterator over the
124      * <code>TypeTree</code>.  Given a <code>className</code>, it
125      * maintains a randomized list of subtypes for the given
126      * <code>className</code>, including the class itself.
127      */
128     private static class RandomizedIterator extends TypeTreeIterator<String> {
129 	/**
130 	 * Create a new <code>RandomizedIterator</code> for the given
131 	 * class.
132 	 */
133 	RandomizedIterator(String className, TypeTree tree) {
134 	    super(init(className, tree));
135 	}
136 
137 
138 	/*
139 	 * Traverse the given type tree and add to the list all the
140 	 * subtypes encountered within.
141 	 */
142 	private static void walkTree(Collection<String> children, Collection<String> list, TypeTree tree) {
143 	    if (children != null) {
144 		list.addAll(children);
145 	        String[] kids = children.toArray(new String[children.size()]);
146 		for (int i = 0; i< kids.length; i++) {
147 		    walkTree(tree.classSet(kids[i]), list, tree);
148 		}
149 	    }
150 	}
151 
152 	/**
153 	 * Set up this iterator to walk over the subtypes of this class,
154 	 * including the class itself.  It then randomizes the list.
155 	 */
156 	private static String [] init(String className, TypeTree tree) {
157             // Use a Linked to avoid resizing.
158             Collection<String> types = new LinkedList<String>();
159             String [] typearray;
160 	    if (className.equals(EntryRep.matchAnyClassName())) {
161 		// handle "match any" specially" -- search from ROOT
162 		// Simplification suggested by 
163 		// Lutz Birkhahn <lutz.birkhahn@GMX.DE>
164 		className = ROOT;
165 	    } else {
166 		// add this class
167 		types.add(className);
168 	    }
169 
170 	    // add all subclasses
171 	    walkTree(tree.classSet(className), types, tree);
172 
173 	    // Convert it to an array and then randomize
174 	    typearray = types.toArray(new String[types.size()]);
175 	    int randnum = 0;
176 	    String tmpobj = null;
177             int length = typearray.length;
178 	    for (int i = 0; i < length; i++) {
179 		randnum = numgen.nextInt(length - i);
180 		tmpobj = typearray[i];
181 		typearray[i] = typearray[randnum];
182 		typearray[randnum] = tmpobj;
183 	    }
184             return typearray;
185 	}
186     }
187 
188     /**
189      * Return an iterator over the subtypes of the given class
190      * (including the type itself).  This implementation always returns
191      * an iterator that randomizes the order of the classes returned.
192      * In other words, it returns the names of all classes that are
193      * instances of the class that named, in a random ordering.
194      */
195     Iterator<String> subTypes(String className) {
196 	return new RandomizedIterator(className, this);
197     }
198 
199     /**
200      * Update the type tree with the given bits.  This will traverse the
201      * given EntryRep's list of superclasses, retrieve the subclass list
202      * at each list item and update it with the given EntryRep's type.
203      *
204      *  SupClass List
205      *   |
206      *   V
207      *   SupC1-->Sub1OfSupC1--Sub2OfSupC1...SubNOfSupC1--EntryRep
208      *   |
209      *   |
210      *   SupC2-->Sub1OfSupC2--Sub2OfSupC2...SubNOfSupC2--EntryRep
211      *   .
212      *   .
213      *   .
214      *   SupCN-->Sub1OfSupCN--Sub2OfSupCN...SubNOfSupCN--EntryRep
215      */
216     void addTypes(EntryRep bits) {
217 	String classFor = bits.classFor();
218 	String[] superclasses = bits.superclasses();
219 
220 	//The given EntryRep will add its className to the
221 	//subtype list of all its supertypes.
222 
223 	String prevClass = classFor;
224 	for (int i = 0; i < superclasses.length; i++) {
225 	    if (!addKnown(superclasses[i], prevClass)) {
226 		return;
227 	    }
228 	    prevClass = superclasses[i];
229 	}
230 
231 	// If we are here prevClass must have java.Object as its
232 	// direct superclass (we don't store "java.Object" in
233 	// EntryRep.superclasses since that would be redundant) and
234 	// prevClass is not already in the the tree.  Place it in the
235 	// "net.jini.core.entry.Entry" bucket so it does not get lost
236 	// if it does not have any sub-classes.
237 	//
238 	// Fix suggested by Lutz Birkhahn <lutz.birkhahn@GMX.DE>
239 	addKnown(ROOT, prevClass);
240     }
241 
242     /**
243      * Add the subclass to the list of known subclasses of this superclass.  
244      */
245     private boolean addKnown(String superclass, String subclass) {
246 	Set<String> v;
247         v = classSet(superclass);
248         if (v == null) {
249             v = new ConcurrentSkipListSet<String>();
250             Set<String> existed = subclasses.putIfAbsent(superclass, v);
251             if (existed != null) v = existed; // discard new set.
252         }
253 	return v.add(subclass);
254     }
