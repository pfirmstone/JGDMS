18  package org.apache.river.outrigger;
19  
20  import java.util.Iterator;
21  import java.util.SortedSet;
22  import java.util.concurrent.atomic.AtomicLong;
23  import java.util.logging.Level;
24  import java.util.logging.Logger;
25  
26  /**   
27   * Maintain a journal of operations on entries (writes, takes, and
28   * resolution of locks) and a thread that will process the
29   * journal. This journal is kept for three reasons. One is so events
30   * can be generated asynchronously from the writes and transaction
31   * commits that cause them. Second so blocked queries can be alerted
32   * to changes that will allow them to be resolved.
33   * <p>
34   * Third to make sure that transitions occur between the time a query
35   * starts its initial search and the time it adds its watcher to the
36   * <code>TransitionWatchers</code> object do not get missed.  This
37   * last goal is accomplished by queries calling
38   * <code>newTransitionIterator</code> before initial search, doing the
39   * initial search, adding their watcher to the
40   * <code>TransitionWatchers</code> object and then calling
41   * <code>watcherRegistered</code> on the
42   * <code>TransitionIterator</code> and then feeding their watcher the
43   * chain of transitions yielded by the
44   * <code>TransitionIterator</code>.  (skipping any transitions with
45   * entries that don't match).
46   * <p>
47   * Each operation that is recored in the journal is assigned an
48   * <em>ordinal</em>. Operations with higher ordinals must be considered
49   * to have taken place after operations with lower ordinals.
50   */
51  class OperationJournal extends Thread {
52      /** The object to ask about who is interested in a transition */
53      private final TransitionWatchers watchers;
54  
55      /** The current tail of the transitions list. */
56      private volatile JournalNode tail;
57  
58      /**
59       * The <code>JournalNode</code> we are currently processing or if
60       * none are in process the last one we processed.
61       */
62      private volatile JournalNode lastProcessed;
63  
64      /** If <code>true</code> stop thread */
65      private volatile boolean dead = false;
66  
67      /** The last ordinal value used */
68      private final AtomicLong lastOrdinalUsed = new AtomicLong(1L);
69  
70      /** Logger for logging exceptions */
71      private static final Logger logger = 
72  	Logger.getLogger(OutriggerServerImpl.opsLoggerName);
73  
74      /**
75       * The nodes of our Journal
76       */
77      private class JournalNode {
78  	/** The next node in the journal */
79  	private JournalNode next;
80  
81  	/** The sequence number of this journal entry */
82  	private final long ordinal;
83  
84  	/** The contents of this journal entry */
85  	private final Object payload;
86  
87  	/**
88  	 * Create a new <code>JournalNode</code> with the specified
89  	 * value for the <code>payload</code>, <code>null</code> for
90  	 * <code>next</code>, and the appropriate value for
91  	 * <code>ordinal</code>.
92  	 * <p>
93  	 * Assumes the lock on the <code>OperationJournal</code> is held,
94  	 * @param payload The value for the payload field.
95  	 */
96  	private JournalNode(Object payload) {
97  	    ordinal = lastOrdinalUsed.incrementAndGet();
98  	    this.payload = payload;
99  	}
100 
101 	/**
102 	 * Set the next element in the list.
103 	 * @param n the new value for the next field.
104 	 * @throws IllegalStateException if the next field has
105 	 * already been set.
106 	 * @throws NullPointerException if <code>n</code> is 
107 	 * <code>null</code>.
108 	 */
109 	private synchronized void setNext(JournalNode n) {
110 	    if (next != null)
111 		throw new IllegalStateException("Already has next");
112 
113 	    if (n == null)
114 		throw new NullPointerException("n must be non-null");
115 
116 	    next = n;
117 	}
118 
119 	/**
120 	 * Get the next element in the journal.
121 	 * @return the next element in the list, or <code>null</code>
122 	 * if there is currently no next element.
123 	 */
124 	private synchronized JournalNode getNext() {
125 	    return next;
126 	}
127     }
128 
129     /**
130      * <code>JournalNode</code> payload value used for 
131      * caught up markers.
132      */
133     private static class CaughtUpMarker {
134 	/** The watcher to notify */
135 	private final IfExistsWatcher watcher;
136 
137 	/** 
138 	 * Create a new <code>CaughtUpMarker</code> that
139 	 * will notify the given <code>watcher</code>.
140 	 * @throws NullPointerException if <code>watcher</code> is 
141 	 *         <code>null</code>.     
142 	 */
143 	private CaughtUpMarker(IfExistsWatcher watcher) {
144 	    if (watcher == null)
145 		throw new NullPointerException("watcher must not be null");
146 	    this.watcher = watcher;
147 	}
148     }
149 
150     /**
151      * An iterator that will yield (in the order they were posted) all the
152      * <code>EntryTransition</code>s added after the iterator was
153      * created and processed before <code>watcherRegistered</code> was
154      * called. This call assumes it is only used by a single thread.
155      */
156     class TransitionIterator {
157 	/** 
158 	 * The place to end, <code>null</code> if 
159 	 * <code>watcherRegistered</code> has not yet 
160 	 * been called.
161 	 */
162 	private volatile JournalNode end;
163 
164 	/** Our current position in the journal */
165 	private volatile JournalNode current;
166 
167 	/**
168 	 * Create a new <code>TransitionIterator</code> that will
169 	 * start with the first <code>EntryTransition</code> that
170 	 * appears in the journal after the passed
171 	 * <code>JournalNode</code>.
172 	 * @param node Start the iteration with the first
173 	 *        <code>JournalNode</code> after node
174 	 *        that is for a <code>EntryTransition</code>.
175 	 * 
176 	 * @throws NullPointerException if node is <code>null</code>.
177 	 */
178 	private TransitionIterator(JournalNode node) {
179 	    if (node == null)
180 		throw new NullPointerException("node must be non-null");
181 
182 	    current = node;
183 	}
184 
185 	/**
186 	 * Return the next <code>EntryTransition</code> in the
187 	 * sequence, or <code>null</code> if the end
188 	 * of the sequence has been reached.
189 	 * @return The next <code>EntryTransition</code> in the
190 	 *         sequence, or <code>null</code> if the
191 	 *         end of the sequence has been reached.
192 	 * @throws IllegalStateException if 
193 	 *         <code>watcherRegistered</code> has not yet 
194 	 *         been called.
195 	 */
196 	EntryTransition next() { 
197 	    if (end == null)
198 		throw new IllegalStateException(
199 		    "watcherRegistered() not yet called");
200 
201 	    /* We don't normally go off the end, but
202 	     * we set current equal to null when we are done
203 	     */
204 	    if (current == null)
205 		return null;
206 
207 	    /* Note, the logic in here is a bit tricky since
208 	     * current is pre-incremented.
209 	     */
210 
211 	    // Skip if payload is not an EntryTransition
212             synchronized (this){
213                 Object payload = current.payload;
214                 while (true) {
215                     if (current == end) {
216                         /* This is the last one...still need to return 
217                          * it's payload if applicable.
218                          */
219                         current = null;
220                         if (payload instanceof EntryTransition)	
221                             // Might be null, but that's ok
222                             return (EntryTransition)payload;
223 
224                         return null;
225                     }
226 
227                     current = current.getNext();	    
228                     assert current != null : "Iteration when off end";
229 
230                     if ((payload != null) && 
231                         (payload instanceof EntryTransition)) 
232                     {
233                         return (EntryTransition)payload;
234                     }		    
235 
236                     payload = current.payload;
237                 }
238             }
239 	}	
240 
241 	/**
242 	 * Set the end of the iteration to ensure that
243 	 * any <code>EntryTransition</code> added after 
244 	 * this iterator was created will either be returned by
245 	 * this iterator, or passed to the process() method
246 	 * of any watcher that was added to the watcher associated
247 	 * with the <code>OperationJournal</code> before
248 	 * this method was called.
249 	 * @throws IllegalStateException if 
250 	 *         <code>watcherRegistered</code> has been called.
251 	 */
252 	void watcherRegistered() {
253             synchronized (this){
254                 if (end != null)
255                     throw new IllegalStateException(
256                         "watcherRegistered() called more than once");
257             
258                 end = lastProcessed(current);
259 
260                 if (current == end) {
261                     /* Noting has been processed since we were created.
262                      * There are no elements in the iteration.
263                      */
264                     current = null;
265                     return;
266                 }
267 
268                 /* Skip the tail when we were created, we don't
269                  * need to return it since it was in the journal before
270                  * we were created.
271                  */
272                 current = current.getNext();
273             }
274 	}
275 
276 	/**
277 	 * Return the ordinal of the last operation posted
278 	 * when this iterator was created. All the 
279 	 * <code>EntryTransition</code>s yielded by this
280 	 * iterator will have higher ordinals.
281 	 * @return the current ordinal when this 
282 	 *         iterator was created.
283 	 * @throws IllegalStateException if 
284 	 *         <code>watcherRegistered</code> has been called.
285 	 */
286 	synchronized long currentOrdinalAtCreation() {
287 	    if (end != null)
288 		throw new IllegalStateException(
289 		    "watcherRegistered() has been called");
290 
291 	    return current.ordinal;
292 	}
293     }
294 
295     /**
296      * Create a new <code>OperationJournal</code>.
297      * @param watchers Set of watchers that need
298      *                 to be asked if they are interested in the transitions.
299      * @throws NullPointerException if watchers is <code>null</code>.
300      */
301     OperationJournal(TransitionWatchers watchers) {
302 	super("OperationJournal");
303 	if (watchers == null)
304 	    throw new NullPointerException("watchers must be non-null");
305 	this.watchers = watchers;
306         tail = new JournalNode(null);
307 	lastProcessed = tail;
308     }
309 
310     /**
311      * Records an operation on an entry. This method should be called
312      * <em>after</em> the transition has been made visible in 
313      * <code>contents</code> (including any subsidiary 
314      * objects such as the appropriate <code>EntryHandle</code>).
315      * @param transition an object describing the visibility
316      *        transition of an entry.
317      * @throws NullPointerException if <code>transition</code> is
318      *         <code>null</code>.
319      */
320     void recordTransition(EntryTransition transition) {
321 	if (transition == null)
322 	    throw new NullPointerException("transition must be non-null");
323 
324 	post(new JournalNode(transition));
325     }
326 
327     /**
328      * Schedules a job that will call the <code>caughtUp</code> method
329      * on the passed <code>IfExistsWatcher</code> after the last
330      * posted <code>EntryTransition</code> is processed.
331      * @param watcher The watcher to notify when it is caught up.
332      * @throws NullPointerException if <code>watcher</code> is 
333      *         <code>null</code>.     
334      */
335     void markCaughtUp(IfExistsWatcher watcher) {
336 	post(new JournalNode(new CaughtUpMarker(watcher)));
337     }
338 
339     /**
340      * Return the ordinal of the last operation posted.
341      * Any operations with a higher ordinal should be considered
342      * to have taken place after this point in time.
343      * @return the ordinal of the last operation posted.
344      */
345     synchronized long currentOrdinal() {
346 	return lastOrdinalUsed.get();
347     }
348 
349     /**
350      * Post a <code>JournalNode</code> 
351      * @param node The node to post.
352      */
353     private synchronized void post(JournalNode node) {
354         tail.setNext(node);
355         tail = node;
356         notifyAll();
357     }
358 
359     /**
360      * Return an iterator that can latter be used to 
361      * get all the <code>EntryTransition</code>s added after 
362      * this point that have been processed.
363      * <p>
364      * We synchronize to make sure the initial state
365      * of the returned transition has been fetched from main memory.
366      * @return An iterator that can latter be used to 
367      * get all the <code>EntryTransition</code>s added after 
368      * this point that have been processed.
369      */
370     synchronized TransitionIterator newTransitionIterator() {
371 	return new TransitionIterator(tail);
372     }
373 
374     /**
375      * Return the node currently being processed, or
376      * if no entry is currently being processed the 
377      * last one that was processed.  A watcher
378      * that was added to the <code>TransitionWatchers</code> object
379      * associated with this object before this call was made is
380      * guaranteed to be asked by the journal about any transition posted
381      * after the node returned by this method was posted. 
382      * @param noEarlierThan The returned <code>JournalNode</code>
383      *        is guaranteed to have not been posted before
384      *        <code>noEarlierThan</code>. If the last
385      *        node processes was posted before
386      *        <code>noEarlierThan</code>, then 
387      *        <code>noEarlierThan</code> will be returned.
388      * @return The last node that has been at least
389      *         partially processed.
390      */
391     private synchronized JournalNode lastProcessed(
392 	JournalNode noEarlierThan) 
393     {
394 	if (lastProcessed.ordinal < noEarlierThan.ordinal) {
395 	    return noEarlierThan;
396 	}
397 
398 	return lastProcessed;
399     }
400 
401     /**
402      * Terminate queue processing.
403      */
404     void terminate() {
405         /* Only set dead to true while synchronized, this means that all
406          * pending transitions will be processed then wait() is called
407          * releasing the lock so queue processing can terminate.
408          */
409         synchronized (this){
410             dead = true;
411             notifyAll();
412         }
413     }
414 
415     /**
416      * Loop pulling transitions off the queue and
417      * process them by getting the set of interested watchers
418      * from the <code>TransitionWatchers</code> object associated
419      * with this object and then calling process on each
420      * of the watchers.
421      */
422     public void run() {
423 	while (!dead) {
424 	    try {
425 		// Wait until there is something to process
426                 final Object payload;
427                 long ordinal;
428 		synchronized (this) {
429 		    JournalNode n = lastProcessed.getNext();
430 		    while (n == null && !dead) {
431 			wait();
432 			n = lastProcessed.getNext();
433 		    }
434 
435 		    if (dead)
436 			return;
437 
438 		    lastProcessed = n;
439                     // Process based on payload
440                     payload = lastProcessed.payload;
441                     ordinal = lastProcessed.ordinal;
442 		}
443 
444 		
445 
446 		if (payload == null) {
447 		    throw new 
448 			AssertionError("JournalNode with null payload");
449 		} else if (payload instanceof EntryTransition) {
450 		    final EntryTransition t = (EntryTransition)payload;
451 		    final SortedSet<TransitionWatcher> set = 
452 			watchers.allMatches(t, ordinal);
453 		    final long now = System.currentTimeMillis();
454 
455 		    for (Iterator<TransitionWatcher> i = set.iterator(); i.hasNext() && !dead; ) {
456 			final TransitionWatcher watcher = i.next();
457 			watcher.process(t, now);
458 		    }
459 		} else if (payload instanceof CaughtUpMarker) {
460 		    ((CaughtUpMarker)payload).watcher.caughtUp();
461 		} else {
462 		    throw new 
463 			AssertionError("JournalNode with unknown payload:" +
464 				       payload.getClass());
465 		}
466 	    } catch (InterruptedException e) {
467 		// fin
468 		return;
469 	    } catch (Throwable t) {
470 		try {
471 		    logger.log(Level.INFO,
472 			       "OperationJournal.run encountered " +
473 			           t.getClass().getName() + ", continuing",
474 			       t);
475 		} catch (Throwable tt) {
476 		    // don't let a problem in logging kill the thread
477 		}
478                 if (t instanceof Error) throw (Error) t;
479                 if (t instanceof RuntimeException) throw (RuntimeException) t;
480 	    }
481 	}
482     }
