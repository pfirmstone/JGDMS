140     /**
141      * Marshals <code>value</code> to an <code>ObjectOutput</code> stream,
142      * <code>out</code>, using RMI's serialization format for arguments or
143      * return values.  For primitive types, the primitive type's class should
144      * be specified (i.e., for the type <code>int</code>, specify
145      * <code>int.class</code>), and the primitive value should be wrapped in
146      * instances of the appropriate wrapper class, such as
147      * <code>java.lang.Integer</code> or <code>java.lang.Boolean</code>.
148      *
149      * @param	type <code>Class</code> object for the value to be marshalled
150      * @param	value value to marshal
151      * @param	out stream to which the value is marshalled
152      * @throws	IOException if an I/O error occurs marshalling
153      *		the value to the output stream
154      */
155     public static void marshalValue(Class type, Object value, ObjectOutput out)
156 	throws IOException
157     {
158 	if (type.isPrimitive()) {
159 	    if (type == int.class) {
160 		out.writeInt(((Integer) value).intValue());
161 	    } else if (type == boolean.class) {
162 		out.writeBoolean(((Boolean) value).booleanValue());
163 	    } else if (type == byte.class) {
164 		out.writeByte(((Byte) value).byteValue());
165 	    } else if (type == char.class) {
166 		out.writeChar(((Character) value).charValue());
167 	    } else if (type == short.class) {
168 		out.writeShort(((Short) value).shortValue());
169 	    } else if (type == long.class) {
170 		out.writeLong(((Long) value).longValue());
171 	    } else if (type == float.class) {
172 		out.writeFloat(((Float) value).floatValue());
173 	    } else if (type == double.class) {
174 		out.writeDouble(((Double) value).doubleValue());
175 	    } else {
176 		throw new AssertionError(
177 		    "Unrecognized primitive type: " + type);
178 	    }
179 	} else {
180 	    out.writeObject(value);
181 	}
182     }

229     /**
230      * Computes the "method hash" of a remote method, <code>m</code>.  The
231      * method hash is a <code>long</code> containing the first 64 bits of the
232      * SHA digest from the UTF encoded string of the method name followed by
233      * its "method descriptor".  See section 4.3.3 of The Java(TM) Virtual
234      * Machine Specification for the definition of a "method descriptor".
235      *
236      * @param	m remote method
237      * @return	the method hash
238      */
239     private static long computeMethodHash(Method m) {
240 	long hash = 0;
241 	ByteArrayOutputStream sink = new ByteArrayOutputStream(127);
242 	try {
243 	    MessageDigest md = MessageDigest.getInstance("SHA");
244 	    DataOutputStream out = new DataOutputStream(
245 		new DigestOutputStream(sink, md));
246 
247 	    String s = getMethodNameAndDescriptor(m);
248 	    out.writeUTF(s);
249 
250 	    // use only the first 64 bits of the digest for the hash
251 	    out.flush();
252 	    byte hasharray[] = md.digest();
253 	    for (int i = 0; i < Math.min(8, hasharray.length); i++) {
254 		hash += ((long) (hasharray[i] & 0xFF)) << (i * 8);
255 	    }
256 	} catch (IOException ignore) {
257 	    /* can't happen, but be deterministic anyway. */
258 	    hash = -1;
259 	} catch (NoSuchAlgorithmException complain) {
260 	    throw new SecurityException(complain.getMessage());
261 	}
262 	return hash;
263     }

265     /**
266      * Returns the method hash for the method <code>m</code>.  Subsequent
267      * calls to <code>getMethodHash</code> passing the same method argument
268      * should be faster since this method caches internally the result of the
269      * method to method hash mapping.  The method hash is calculated using the
270      * <code>computeMethodHash</code> method.
271      *
272      * @param	m the remote method
273      * @return	the method hash for the method <code>m</code>
274      */
275     public static long getMethodHash(Method m) {
276 	Map table = methodToHash_TableCache.getTable(m.getDeclaringClass());
277 	Long hash = (Long) table.get(m);
278 	return hash.longValue();
279     }

281     private static class TableCache extends WeakHashMap {
282 	/**
283 	 * if true, the tables map methods to method hashes; if false,
284 	 * the tables map method hashes to methods
285 	 */
286 	private boolean mapsMethodToHash;
287 
288 	public TableCache(boolean mapsMethodToHash) {
289 	    super();
290 	    this.mapsMethodToHash = mapsMethodToHash;
291 	}
292 	
293 	public Map getTable(Class remoteClass) {
294 	    SoftReference[] tableRef;
295 
296 	    /*
297 	     * Method tables for remote classes are cached in a hash table
298 	     * using weak references to hold the Class object keys, so that
299 	     * the cache does not prevent the class from being unloaded, and
300 	     * using soft references to hold the values, so that the computed
301 	     * method tables will generally persist while no objects of the
302 	     * remote class are exported, but their storage may be reclaimed
303 	     * if necessary, and accidental reachability of the remote class
304 	     * through its interfaces is avoided.
305 	     */
306 	    synchronized (this) {
307 		/*
308 		 * Look up class in cache; add entry if not found.
309 		 */
310 		tableRef = (SoftReference[]) get(remoteClass);
311 		if (tableRef == null) {
312 		    tableRef = new SoftReference[] { null };
313 		    put(remoteClass, tableRef);
314 		}
315 	    }
316 
317 	    /*
318 	     * Check cached reference to method table for this class;
319 	     * if it is null, go and create the table.
320 	     */
321 	    synchronized (tableRef) {
322 		Map table = null;
323 		if (tableRef[0] != null) {
324 		    table = (Map) tableRef[0].get();
325 		}
326 		if (table == null) {
327 		    if (mapsMethodToHash) {
328 			/*
329 			 * REMIND: if we hand out this table directly, via a
330 			 * public API, we need to make this map "unmodifiable".
331 			 */
332 			table = new LazyMethodToHash_Map();
333 		    } else {
334 			throw new UnsupportedOperationException();
335 		    }
336 		    tableRef[0] = new SoftReference(table);
337 		}
338 		return table;
339 	    }
340 	}
341     }

345      * Verifies that the supplied method has at least one declared exception
346      * type that is RemoteException or one of its superclasses.  If not,
347      * then this method throws IllegalArgumentException.
348      *
349      * @throws IllegalArgumentException if m is an illegal remote method

351     private static void checkMethod(Method m) {
352 	Class[] ex = m.getExceptionTypes();
353 	for (int i = 0; i < ex.length; i++) {
354 	    if (ex[i].isAssignableFrom(RemoteException.class))
355 		return;
356 	}
357 	throw new IllegalArgumentException(
358 	    "illegal remote method encountered: " + m);

362 
363 	public LazyMethodToHash_Map() {
364 	    super();

367 	public synchronized Object get(Object key) {
368 	    Object hash = super.get(key);
369 	    if (hash == null) {
370 		Method method = (Method) key;
371 		hash = Long.valueOf(computeMethodHash(method));
372 		put(method, hash);
373 	    }
374 	    return (Long) hash;
375 	}

384     /**
385      * Returns a string consisting of the given method's name followed by
386      * its "method descriptor", as appropriate for use in the computation
387      * of the "method hash".
388      *
389      * See section 4.3.3 of The Java(TM) Virtual Machine Specification for
390      * the definition of a "method descriptor".
391      */
392     public static String getMethodNameAndDescriptor(Method m) {
393 	StringBuffer desc = new StringBuffer(m.getName());
394 	desc.append('(');
395 	Class[] paramTypes = m.getParameterTypes();
396 	for (int i = 0; i < paramTypes.length; i++) {
397 	    desc.append(getTypeDescriptor(paramTypes[i]));
398 	}
399 	desc.append(')');
400 	Class returnType = m.getReturnType();
401 	if (returnType == void.class) {	// optimization: handle void here
402 	    desc.append('V');
403 	} else {
404 	    desc.append(getTypeDescriptor(returnType));
405 	}
406 	return desc.toString();
407     }

409     /**
410      * Returns the descriptor of a particular type, as appropriate for either
411      * a parameter type or return type in a method descriptor.
412      */
413     private static String getTypeDescriptor(Class type) {
414 	if (type.isPrimitive()) {
415 	    if (type == int.class) {
416 		return "I";
417 	    } else if (type == boolean.class) {
418 		return "Z";
419 	    } else if (type == byte.class) {
420 		return "B";
421 	    } else if (type == char.class) {
422 		return "C";
423 	    } else if (type == short.class) {
424 		return "S";
425 	    } else if (type == long.class) {
426 		return "J";
427 	    } else if (type == float.class) {
428 		return "F";
429 	    } else if (type == double.class) {
430 		return "D";
431 	    } else if (type == void.class) {
432 		return "V";
433 	    } else {
434 		throw new Error("unrecognized primitive type: " + type);
435 	    }
436 	} else if (type.isArray()) {
437 	    /*
438 	     * According to JLS 20.3.2, the getName() method on Class does
439 	     * return the virtual machine type descriptor format for array
440 	     * classes (only); using that should be quicker than the otherwise
441 	     * obvious code:
442 	     *
443 	     *     return "[" + getTypeDescriptor(type.getComponentType());
444 	     */
445 	    return type.getName().replace('.', '/');
446 	} else {
447 	    return "L" + type.getName().replace('.', '/') + ";";
448 	}
449     }

465     /**
466      * Fills the given array list with the remote interfaces implemented
467      * by the given class.
468      *
469      * @throws	IllegalArgumentException if the specified class implements
470      * 		any illegal remote interfaces
471      * @throws	NullPointerException if the specified class or list is null
472      */
473     private static void getRemoteInterfaces(ArrayList list, Class cl) {
474 	Class superclass = cl.getSuperclass();
475 	if (superclass != null) {
476 	    getRemoteInterfaces(list, superclass);
477 	}
478 	
479 	Class[] interfaces = cl.getInterfaces();
480 	for (int i = 0; i < interfaces.length; i++) {
481 	    Class intf = interfaces[i];
482 	    /*
483 	     * If it is a remote interface (if it extends from
484 	     * java.rmi.Remote) and is not already in the list,
485 	     * then add the interface to the list.
486 	     */
487 	    if (Remote.class.isAssignableFrom(intf)) {
488 		if (!(list.contains(intf))) {
489 		    Method[] methods = intf.getMethods();
490 		    for (int j = 0; j < methods.length; j++) {
491 			checkMethod(methods[j]);
492 		    }
493 		    list.add(intf);
494 		}
495 	    }
496 	}
497     }

499     /**
500      * Throws IllegalArgumentException if any superinterface of c declares a
501      * method with the same name and parameter types as m that does not
502      * declare RemoteException or a superclass in its throws clause, or if
503      * any superinterface of c has its name in prohibitedProxyInterfaces.
504      */
505     public static void checkProxyRemoteMethod(Class c, Method m) {
506 	WeakIdentityMap<Method,Boolean> map;
507 	synchronized (proxyRemoteMethodCache) {
508 	    SoftReference ref = (SoftReference) proxyRemoteMethodCache.get(c);
509 	    map = (ref == null) ? null : (WeakIdentityMap) ref.get();
510 	    if (map == null && ref != null) {
511 		map = new WeakIdentityMap<Method,Boolean>();
512 		proxyRemoteMethodCache.put(c, new SoftReference(map));
513 	    }
514 	}
515 	if (map == null) {
516 	    checkProhibitedProxyInterfaces(c);
517 	    synchronized (proxyRemoteMethodCache) {
518 		SoftReference ref =
519 		    (SoftReference) proxyRemoteMethodCache.get(c);
520 		map = (ref == null) ? null : (WeakIdentityMap) ref.get();
521 		if (map == null) {
522 		    map = new WeakIdentityMap<Method,Boolean>();
523 		    proxyRemoteMethodCache.put(c, new SoftReference(map));
524 		}
525 	    }
526 	}
527 	synchronized (map) {
528 	    if (map.get(m) != null) {
529 		return;
530 	    }
531 	}
532 	checkExceptions(c, m.getName(), m.getParameterTypes());
533 	synchronized (map) {
534 	    map.put(m, Boolean.TRUE);
535 	}
536     }

538     /**
539      * Throws IllegalArgumentException if any superinterface of c declares a
540      * method with the given name and parameter types that does not declare
541      * RemoteException or a superclass in its throws clause.
542      */
543     private static void checkExceptions(Class c, String name, Class[] types) {
544 	Class[] ifaces = c.getInterfaces();
545 	for (int i = ifaces.length; --i >= 0; ) {
546 	    try {
547 		checkMethod(ifaces[i].getMethod(name, types));
548 		checkExceptions(ifaces[i], name, types);
549 	    } catch (NoSuchMethodException e) {
550 	    }
551 	}
552     }

554     /**
555      * Returns collection of prohibited proxy interfaces read from resources.
556      */
557     private static Collection getProhibitedProxyInterfaces() {
558 	Collection names = new HashSet();
559 	names.add("javax.management.MBeanServerConnection");
560 	Enumeration resources;
561 	try {
562 	    resources = ClassLoader.getSystemResources(
563 					  prohibitedProxyInterfacesResource);
564 	} catch (IOException e) {
565 	    throw new ExceptionInInitializerError(
566 			new IOException(
567 			     "problem getting resources: " +
568 			     prohibitedProxyInterfacesResource, e));
569 	}
570 	while (resources.hasMoreElements()) {
571 	    URL url = (URL) resources.nextElement();
572 	    try {
573 		InputStream in = url.openStream();
574 		BufferedReader r = null;
575 		try {
576 		    r = new BufferedReader(new InputStreamReader(in, "utf-8"));
577 		    while (true) {
578 			String s = r.readLine();
579 			if (s == null) {
580 			    break;
581 			}
582 			int i = s.indexOf('#');
583 			if (i >= 0) {
584 			    s = s.substring(0, i);
585 			}
586 			s = s.trim();
587 			int n = s.length();
588 			if (n == 0) {
589 			    continue;
590 			}
591 			char prev = '.';
592 			for (i = 0; i < n; i++) {
593 			    char c = s.charAt(i);
594 			    if (prev == '.' ?
595 				!Character.isJavaIdentifierStart(c) :
596 				!(Character.isJavaIdentifierPart(c) ||
597 				  (c == '.' && i < n - 1)))
598 			    {
599 				throw new ExceptionInInitializerError(
600 					      "illegal interface name in " +
601 					      url + ": " + s);
602 			    }
603 			    prev = c;
604 			}
605 			names.add(s);
606 		    }
607 		} finally {
608 		    try {
609 			if (r != null){
610 			    r.close();
611 			} else {
612 			    in.close();
613 			}
614 		    } catch (IOException e) {
615 		    }
616 		}
617 	    } catch (IOException e) {
618 		throw new ExceptionInInitializerError(
619 		      new IOException("problem reading " + url, e));
620 	    }
621 	}
622 	return names;
623     }

625     /**
626      * Throws IllegalArgumentException if any superinterface of c has its
627      * name in prohibitedProxyInterfaces.
628      */
629     private static void checkProhibitedProxyInterfaces(Class c) {
630 	Class[] ifaces = c.getInterfaces();
631 	for (int i = ifaces.length; --i >= 0; ) {
632 	    String name = ifaces[i].getName();
633 	    if (prohibitedProxyInterfaces.contains(name)) {
634 		throw new IllegalArgumentException(
635 		       "prohibited proxy interface encountered: " + name);
636 	    }
637 	    checkProhibitedProxyInterfaces(ifaces[i]);
638 	}

654     /**
655      * Returns true either if both arguments are null or if an
656      * invocation of Object.equals on "subject" with "object" as the
657      * argument returns true; returns false otherwise;
658      **/
659     public static boolean equals(Object subject, Object object) {
660 	return subject == null ? object == null : subject.equals(object);
661     }

663     /**
664      * Returns true either if both arguments are null or if both
665      * arguments refer to objects of the same class and an invocation
666      * of Object.equals on "subject" with "object" as the argument
667      * returns true; returns false otherwise.
668      *
669      * This method is used to compare to possibly-null references for
670      * object equality when neither object's class is trusted, with
671      * the restriction that only objects of the same class can be
672      * considered equal.
673      **/
674     public static boolean sameClassAndEquals(Object subject, Object object) {
675 	return subject == null ? object == null :
676 	    object != null &&
677 	    subject.getClass() == object.getClass() &&
678 	    subject.equals(object);
679     }

682      * Returns true either if both arguments are null of if "subject"
683      * is an instance of TrustEquivalence and an invocation of
684      * TrustEquivalence.checkTrustEquivalence on "subject" with
685      * "object" as the argument returns true; returns false otherwise.
686      **/
687     public static boolean checkTrustEquivalence(Object subject,
688 						Object object)
689     {

691 	    subject instanceof TrustEquivalence &&
692 	    ((TrustEquivalence) subject).checkTrustEquivalence(object);
693     }

696      * Returns true if proxy2 is a generated Proxy (proxy1 is assumed to
697      * be one) and the classes of both proxies implement the same ordered
698      * list of interfaces, and returns false otherwise.
699      */
700     public static boolean sameProxyClass(Object proxy1, Object proxy2) {
701 	return (proxy1.getClass() == proxy2.getClass() ||
702 		(Proxy.isProxyClass(proxy2.getClass()) &&
703 		 equalInterfaces(proxy1, proxy2)));
704     }

706     /**
707      * Returns true if the interfaces implemented by obj1's class
708      * are the same (and in the same order) as obj2's class.
709      */
710     public static boolean equalInterfaces(Object obj1, Object obj2) {
711 	Class[] intf1 = obj1.getClass().getInterfaces();
712 	Class[] intf2 = obj2.getClass().getInterfaces();
713 	if (intf1.length != intf2.length) {
714 	    return false;
715 	} else {
716 	    for (int i = 0; i < intf1.length; i++) {
717 		if (intf1[i] != intf2[i]) {
718 		    return false;
719 		}
720 	    }
721 	    return true;
722 	}
723     }
