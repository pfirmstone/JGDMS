18  package org.apache.river.jeri.internal.runtime;
19  
20  import java.io.IOException;
21  import java.io.OutputStream;
22  
23  /**
24   * This class implements a BASE64 Character encoder as specified in RFC1521.
25   * This RFC is part of the MIME specification as published by the Internet 
26   * Engineering Task Force (IETF). Unlike some other encoding schemes there 
27   * is nothing in this encoding that indicates
28   * where a buffer starts or ends.
29   *
30   * This means that the encoded text will simply start with the first line
31   * of encoded text and end with the last line of encoded text.
32   *
33   * @author Sun Microsystems, Inc.
34   * 
35   * @see CharacterEncoder
36   */
37  
38  public class BASE64Encoder extends CharacterEncoder {
39  	
40      /** this class encodes three bytes per atom. */
41      protected int bytesPerAtom() {
42  	return (3);
43      }
44  
45      /** 
46       * this class encodes 57 bytes per line. This results in a maximum
47       * of 57/3 * 4 or 76 characters per output line. Not counting the
48       * line termination.
49       */
50      protected int bytesPerLine() {
51  	return (57);
52      }
53  
54      /** This array maps the characters to their 6 bit values */
55      private final static char pem_array[] = {
56  	//       0   1   2   3   4   5   6   7
57  		'A','B','C','D','E','F','G','H', // 0
58  		'I','J','K','L','M','N','O','P', // 1
59  		'Q','R','S','T','U','V','W','X', // 2
60  		'Y','Z','a','b','c','d','e','f', // 3
61  		'g','h','i','j','k','l','m','n', // 4
62  		'o','p','q','r','s','t','u','v', // 5
63  		'w','x','y','z','0','1','2','3', // 6
64  		'4','5','6','7','8','9','+','/'  // 7
65  	};
66  
67      /** 
68       * encodeAtom - Take three bytes of input and encode it as 4
69       * printable characters. Note that if the length in len is less
70       * than three is encodes either one or two '=' signs to indicate
71       * padding characters.
72       */
73      protected void encodeAtom(OutputStream outStream, byte data[], int offset, int len) 
74  	throws IOException {
75  	byte a, b, c;
76  
77  	if (len == 1) {
78  	    a = data[offset];
79  	    b = 0;
80  	    c = 0;
81  	    outStream.write(pem_array[(a >>> 2) & 0x3F]);
82  	    outStream.write(pem_array[((a << 4) & 0x30) + ((b >>> 4) & 0xf)]);
83  	    outStream.write('=');
84  	    outStream.write('=');
85  	} else if (len == 2) {
86  	    a = data[offset];
87  	    b = data[offset+1];
88  	    c = 0;
89  	    outStream.write(pem_array[(a >>> 2) & 0x3F]);
90  	    outStream.write(pem_array[((a << 4) & 0x30) + ((b >>> 4) & 0xf)]);
91  	    outStream.write(pem_array[((b << 2) & 0x3c) + ((c >>> 6) & 0x3)]);
92  	    outStream.write('=');
93  	} else {
94  	    a = data[offset];
95  	    b = data[offset+1];
96  	    c = data[offset+2];
97  	    outStream.write(pem_array[(a >>> 2) & 0x3F]);
98  	    outStream.write(pem_array[((a << 4) & 0x30) + ((b >>> 4) & 0xf)]);
99  	    outStream.write(pem_array[((b << 2) & 0x3c) + ((c >>> 6) & 0x3)]);
100 	    outStream.write(pem_array[c & 0x3F]);
101 	}
102     }
103 }
