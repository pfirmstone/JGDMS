19  package org.apache.river.jeri.internal.connection;
20  
21  import net.jini.jeri.OutboundRequestIterator;
22  import net.jini.jeri.connection.ConnectionEndpoint;
23  import net.jini.jeri.connection.ConnectionManager;
24  import net.jini.jeri.connection.OutboundRequestHandle;
25  
26  /**
27   * Creates {@link ConnManager} instances which delegate directly to
28   * {@link ConnectionManager}.
29   *
30   * @author Sun Microsystems, Inc.
31   * @since 2.0
32   */
33  public class BasicConnManagerFactory implements ConnManagerFactory {
34  
35      /**
36       * Returns <code>ConnManager</code> containing a
37       * <code>ConnectionManager</code> constructed with the given {@link
38       * ConnectionEndpoint}; the {@link ConnManager#newRequest newRequest}
39       * method of the returned <code>ConnManager</code> delegates directly to
40       * the corresponding method of the contained
41       * <code>ConnectionManager</code>.
42       *
43       * @throws NullPointerException if <code>endpoint</code> is
44       * <code>null</code>
45       */
46      @Override
47      public ConnManager create(final ConnectionEndpoint endpoint) {
48  	return new ConnManager() {
49  
50  	    private final ConnectionManager manager =
51                                              new ConnectionManager(endpoint);
52  
53              @Override
54  	    public OutboundRequestIterator newRequest(
55  						OutboundRequestHandle handle)
56  	    {
57  		return manager.newRequest(handle);
58  	    }
59  	};
60      }
61  }
