40  /**
41   * Class representing a server-side HTTP connection used to receive and
42   * dispatch incoming HTTP requests.
43   *
44   * @author Sun Microsystems, Inc.
45   * 
46   */
47  public abstract class HttpServerConnection implements TimedConnection {
48      
49      private static final int HTTP_MAJOR = 1;
50      private static final int HTTP_MINOR = 1;
51      
52      private static final int UNSTARTED = 0;
53      private static final int IDLE      = 1;
54      private static final int BUSY      = 2;
55      private static final int CLOSED    = 3;
56  
57      private static final String serverString = (String)
58  	AccessController.doPrivileged(new PrivilegedAction() {
59  	    public Object run() {
60  		return "Java/" + System.getProperty("java.version", "???") +
61  		       " " + HttpServerConnection.class.getName();
62  	    }
63  	});
64  
65      private static final Executor userThreadPool = (Executor)
66  	java.security.AccessController.doPrivileged(
67  	    new GetThreadPoolAction(true));
68  
69      private final Socket sock;
70      private final InputStream in;
71      private final OutputStream out;
72      private final RequestDispatcher dispatcher;
73      private final HttpServerManager manager;
74      private final Object stateLock = new Object();
75      private int state = UNSTARTED;
76  
77      /**
78       * Creates new HttpServerConnection on top of given socket.
79       */
80      public HttpServerConnection(Socket sock, 
81  				RequestDispatcher dispatcher,
82  				HttpServerManager manager)
83  	throws IOException
84      {
85  	if (dispatcher == null) {
86  	    throw new NullPointerException();
87  	}
88  	this.sock = sock;
89  	this.dispatcher = dispatcher;
90  	this.manager = manager;
91  	in = new BufferedInputStream(sock.getInputStream());
92  	out = new BufferedOutputStream(sock.getOutputStream());
93      }
94  
95      /**
96       * Starts request dispatch thread.  Throws IllegalStateException if
97       * connection has already been started, or is closed.
98       */
99      protected void start() {
100 	synchronized (stateLock) {
101 	    if (state != UNSTARTED) {
102 		throw new IllegalStateException();
103 	    }
104 	    state = IDLE;
105 	    userThreadPool.execute(new Dispatcher(), "HTTP dispatcher");
106 	}
107     }

109     /**
110      * Verifies that calling context has sufficient security permissions to
111      * receive a request on this connection.
112      */
113     protected void checkPermissions() {
114     }
115     
116     /**
117      * Checks that the specified requirements are either fully or
118      * partially satisfied by the constraints actually in force for
119      * this connection, and returns any constraints that must be fully
120      * or partially implemented by higher layers in order to fully
121      * satisfy all of the specified requirements.
122      **/
123     protected InvocationConstraints
124 	checkConstraints(InvocationConstraints constraints)
125 	throws UnsupportedConstraintException
126     {
127 	if (constraints.requirements().isEmpty()) {
128 	    return InvocationConstraints.EMPTY;
129 	}
130 	throw new UnsupportedConstraintException(
131 	    "cannot satisfy constraints: " + constraints);
132     }

134     /**
135      * Populates the given context collection with context information
136      * representing this connection.
137      */
138     protected abstract void populateContext(Collection context);
139     
140     /**
141      * Upcall indicating that connection has become idle.  Subclasses may
142      * override this method to perform an appropriate action, such as
143      * scheduling an idle timeout.
144      */
145     protected void idle() {
146     }
147     
148     /**
149      * Upcall indicating that connection is about to become busy.  Subclasses
150      * may override this method to perform an appropriate action, such as
151      * cancelling an idle timeout.
152      */
153     protected void busy() {
154     }
155     
156     /**
157      * Attempts to shut down connection, returning true if connection is
158      * closed.  If force is true, connection is always shut down; if force is
159      * false, connection is only shut down if idle.
160      */
161     public boolean shutdown(boolean force) {
162 	synchronized (stateLock) {
163 	    if (state == CLOSED) {
164 		return true;
165 	    }
166 	    if (!force && state == BUSY) {
167 		return false;
168 	    }
169 	    state = CLOSED;
170 	}
171 	try { sock.close(); } catch (IOException ex) {}
172 	return true;
173     }

175     /**
176      * Incoming request dispatcher.
177      */
178     private class Dispatcher implements Runnable {
179 
180 	/**
181 	 * Dispatch loop.
182 	 */
183 	public void run() {
184 	    try {
185 		for (;;) {
186 		    idle();
187 
188 		    MessageReader reader = new MessageReader(in, false);
189 		    StartLine sline = reader.readStartLine();
190 		    
191 		    busy();
192 		    synchronized (stateLock) {
193 			if (state == CLOSED) {
194 			    return;
195 			}
196 			state = BUSY;
197 		    }
198 
199 		    Header header = reader.readHeader();
200 		    String reqType = header.getField("RMI-Request-Type");
201 		    if (!"POST".equals(sline.method)) {
202 			handleBadRequest(sline, header, reader);
203 		    } else if ("standard".equalsIgnoreCase(reqType)) {
204 			handleRequest(sline, header, reader);
205 		    } else if ("ping".equalsIgnoreCase(reqType)) {
206 			handlePing(sline, header, reader);
207 		    } else {
208 			handleBadRequest(sline, header, reader);
209 		    }
210 		    
211 		    synchronized (stateLock) {
212 			if (state == CLOSED) {
213 			    return;
214 			}
215 			state = IDLE;
216 		    }
217 		}
218 	    } catch (IOException ex) {
219 	    } finally {
220 		shutdown(true);
221 	    }
222 	}
223 
224 	/**
225 	 * Handles unacceptable HTTP request.
226 	 */
227 	private void handleBadRequest(StartLine inLine,
228 				      Header inHeader,
229 				      MessageReader reader)
230 	    throws IOException
231 	{
232 	    inHeader.merge(reader.readTrailer());
233 	    registerAcks(inHeader.getField("RMI-Response-Ack"));
234 	    boolean persist = supportsPersist(inLine, inHeader);
235 	    
236 	    MessageWriter writer = new MessageWriter(out, false);
237 	    writer.writeStartLine(new StartLine(HTTP_MAJOR, HTTP_MINOR,
238 				      HttpURLConnection.HTTP_BAD_REQUEST, 
239 				      "Bad Request"));
240 	    writer.writeHeader(createResponseHeader(persist));
241 	    writer.writeTrailer(null);
242 	    
243 	    if (!persist) {
244 		shutdown(true);
245 	    }
246 	}
247 
248 	/**
249 	 * Handles ping request.
250 	 */
251 	private void handlePing(StartLine inLine, Header inHeader,
252 				MessageReader reader)
253 	    throws IOException
254 	{
255 	    inHeader.merge(reader.readTrailer());
256 	    registerAcks(inHeader.getField("RMI-Response-Ack"));
257 	    boolean persist = supportsPersist(inLine, inHeader);
258 
259 	    MessageWriter writer = new MessageWriter(out, false);
260 	    writer.writeStartLine(new StartLine(HTTP_MAJOR, HTTP_MINOR,
261 				      HttpURLConnection.HTTP_OK, "OK"));
262 	    writer.writeHeader(createResponseHeader(persist));
263 	    writer.writeTrailer(null);
264 	    
265 	    if (!persist) {
266 		shutdown(true);
267 	    }
268 	}
269 	
270 	/**
271 	 * Handles "standard" (i.e., dispatchable) request.
272 	 */
273 	private void handleRequest(StartLine inLine, Header inHeader,
274 				   MessageReader reader)
275 	    throws IOException
276 	{
277 	    registerAcks(inHeader.getField("RMI-Response-Ack"));
278 	    boolean persist = supportsPersist(inLine, inHeader);
279 	    boolean chunk = supportsChunking(inLine, inHeader);
280 
281 	    MessageWriter writer = new MessageWriter(out, chunk);
282 	    writer.writeStartLine(new StartLine(HTTP_MAJOR, HTTP_MINOR,
283 				      HttpURLConnection.HTTP_OK, "OK"));
284 	    writer.writeHeader(createResponseHeader(persist));
285 	    
286 	    InboundRequestImpl req = new InboundRequestImpl(reader, writer);
287 	    try { dispatcher.dispatch(req); } catch (Throwable th) {}
288 	    req.finish();
289 	    
290 	    if (!persist || req.streamCorrupt()) {
291 		shutdown(true);
292 	    }
293 	}
294     }

296     /**
297      * HTTP-based implementation of InboundRequest abstraction.
298      */
299     private class InboundRequestImpl 
300 	extends Request implements InboundRequest 
301     {
302 	private final MessageReader reader;
303 	private final MessageWriter writer;
304 	private String cookie;
305 	private boolean corrupt = false;
306 
307 	/**
308 	 * Creates new InboundRequestImpl which uses given MessageReader and
309 	 * MessageWriter instances to read/write request content.
310 	 */
311 	InboundRequestImpl(MessageReader reader, MessageWriter writer) {
312 	    this.reader = reader;
313 	    this.writer = writer;
314 	}
315 	
316 	public void checkPermissions() {
317 	    HttpServerConnection.this.checkPermissions();
318 	}
319 	
320 	public InvocationConstraints
321 	    checkConstraints(InvocationConstraints constraints)
322 	    throws UnsupportedConstraintException
323 	{
324 	    return HttpServerConnection.this.checkConstraints(constraints);
325 	}
326 
327 	public void populateContext(Collection context) {
328 	    context.add(new AcknowledgmentSource() {
329 		public boolean addAcknowledgmentListener(
330 		    AcknowledgmentSource.Listener listener)
331 		{
332 		    if (listener == null) {
333 			throw new NullPointerException();
334 		    }
335 		    return InboundRequestImpl.this.addAcknowledgmentListener(
336 			listener);
337 		}
338 	    });
339 	    HttpServerConnection.this.populateContext(context);
340 	}
341 	
342 	public InputStream getRequestInputStream() {
343 	    return getInputStream();
344 	}
345 	
346 	public OutputStream getResponseOutputStream() {
347 	    return getOutputStream();
348 	}
349 	
350 	/**
351 	 * Returns true if stream corrupted, false if stream ok.
352 	 */
353 	boolean streamCorrupt() {
354 	    return corrupt;
355 	}
356 
357 	void startOutput() throws IOException {
358 	    // start line, header already written
359 	}
360 
361 	void write(byte[] b, int off, int len) throws IOException {
362 	    writer.writeContent(b, off, len);
363 	}
364 
365 	void endOutput() throws IOException {
366 	    if (cookie != null) {
367 		Header trailer = new Header();
368 		trailer.setField("RMI-Response-Cookie", cookie);
369 		writer.writeTrailer(trailer);
370 	    } else {
371 		writer.writeTrailer(null);
372 	    }
373 	}
374 
375 	boolean startInput() throws IOException {
376 	    return true;	// header already read
377 	}
378 
379 	int read(byte[] b, int off, int len) throws IOException {
380 	    return reader.readContent(b, off, len);
381 	}
382 
383 	int available() throws IOException {
384 	    return reader.availableContent();
385 	}
386 
387 	void endInput() throws IOException {
388 	    Header trailer = reader.readTrailer();
389 	    if (trailer != null) {
390 		registerAcks(trailer.getField("RMI-Response-Ack"));
391 	    }
392 	}
393 
394 	void addAckListener(AcknowledgmentSource.Listener listener) {
395 	    if (cookie == null) {
396 		cookie = manager.newCookie();
397 	    }
398 	    manager.addAckListener(cookie, listener);
399 	}
400 
401 	void done(boolean corrupt) {
402 	    this.corrupt = corrupt;
403 	}
404     }
405     
406     /**
407      * Notifies listeners for response ack cookies parsed from (possibly null)
408      * comma-separated cookie list string.
409      */
410     private void registerAcks(String ackList) {
411 	if (ackList != null) {
412 	    StringTokenizer tok = new StringTokenizer(ackList, ",");
413 	    while (tok.hasMoreTokens()) {
414 		manager.notifyAckListeners(tok.nextToken().trim());
415 	    }
416 	}
417     }

419     /**
420      * Returns true if the received message start line and header indicate that
421      * the connection can be persisted.
422      */
423     private static boolean supportsPersist(StartLine sline, Header header) {
424 	if (header.containsValue("Connection", "close", true)) {
425 	    return false;
426 	} else if (header.containsValue("Connection", "Keep-Alive", true)) {
427 	    return true;
428 	} else {
429 	    int c = StartLine.compareVersions(sline.major, sline.minor, 1, 1);
430 	    return c >= 0;
431 	}
432     }
433 
434     /**
435      * Returns true if the received message start line indicates that the
436      * sender understands chunked transfer coding.
437      */
438     private static boolean supportsChunking(StartLine sline, Header header) {
439 	int c = StartLine.compareVersions(sline.major, sline.minor, 1, 1);
440 	// REMIND: is requiring "TE: trailers" too strict?
441 	return c >= 0 && header.containsValue("TE", "trailers", true);
442     }
443     
444     /**
445      * Creates base header to be used in response message.  If persist is true,
446      * adds fields indicating a persistent connection.
447      */
448     private static Header createResponseHeader(boolean persist) {
449 	Header header = new Header();
450 	long now = System.currentTimeMillis();
451 	header.setField("Date", Header.getDateString(now));
452 	header.setField("Server", serverString);
453 	header.setField("Connection", persist ? "Keep-Alive" : "close");
454 	return header;
455     }
