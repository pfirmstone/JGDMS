19  package org.apache.river.jeri.internal.http;
20  
21  import java.io.IOException;
22  import java.io.InputStream;
23  import java.io.OutputStream;
24  import java.util.StringTokenizer;
25  
26  /**
27   * Class representing first line of an HTTP message.
28   *
29   * @author Sun Microsystems, Inc.
30   * 
31   */
32  class StartLine {
33      
34      /** major version number */
35      final int major;
36      /** minor version number */
37      final int minor;
38      /** request or response line? */
39      final boolean isRequest;
40      /** request method */
41      final String method;
42      /** request URI */
43      final String uri;
44      /** response status code */
45      final int status;
46      /** response status reason */
47      final String reason;
48      
49      /**
50       * Creates start line for HTTP request message.
51       */
52      StartLine(int major, int minor, String method, String uri) {
53  	this.major = major;
54  	this.minor = minor;
55  	this.method = method;
56  	this.uri = uri;
57  	status = -1;
58  	reason = null;
59  	isRequest = true;
60      }
61      
62      /**
63       * Creates start line for HTTP response message.
64       */
65      StartLine(int major, int minor, int status, String reason) {
66  	this.major = major;
67  	this.minor = minor;
68  	this.status = status;
69  	this.reason = reason;
70  	method = null;
71  	uri = null;
72  	isRequest = false;
73      }
74      
75      /**
76       * Reads start line from given input stream.
77       */
78      StartLine(InputStream in) throws IOException {
79  	String line = MessageReader.readLine(in);
80  	if (line == null) {
81  	    throw new HttpParseException("unexpected EOF in start line");
82  	}
83  
84  	try {
85  	    StringTokenizer tok = new StringTokenizer(line, "", true);
86  	    if (line.startsWith("HTTP")) {
87  		if (!tok.nextToken("/").equals("HTTP")) {
88  		    throw new HttpParseException();
89  		}
90  		tok.nextToken();
91  		major = Integer.parseInt(tok.nextToken("."));
92  		tok.nextToken();
93  		minor = Integer.parseInt(tok.nextToken(" "));
94  		tok.nextToken();
95  		status = Integer.parseInt(tok.nextToken());
96  		tok.nextToken();
97  		reason = tok.nextToken("\n");
98  
99  		method = null;
100 		uri = null;
101 		isRequest = false;
102 	    } else {
103 		method = tok.nextToken(" ");
104 		tok.nextToken();
105 		uri = tok.nextToken();
106 		tok.nextToken();
107 		if (!tok.nextToken("/").equals("HTTP")) {
108 		    throw new HttpParseException();
109 		}
110 		tok.nextToken();
111 		major = Integer.parseInt(tok.nextToken("."));
112 		tok.nextToken();
113 		minor = Integer.parseInt(tok.nextToken("\n"));
114 		
115 		status = -1;
116 		reason = null;
117 		isRequest = true;
118 	    }
119 	} catch (Exception ex) {
120 	    throw (HttpParseException) 
121 		new HttpParseException("invalid start line").initCause(ex);
122 	}
123     }
124     
125     /**
126      * Writes start line to given output stream.
127      */
128     void write(OutputStream out) throws IOException {
129 	String version = "HTTP/" + major + "." + minor;
130 	MessageWriter.writeLine(out, isRequest ?
131 				     method + " " + uri + " " + version :
132 				     version + " " + status + " " + reason);
133     }
134     
135     /**
136      * Compares two sets of major/minor version numbers.  Returns -1 if
137      * major1/minor1 is less than major2/minor2, 1 if major1/minor1 is more
138      * than major2/minor2, and 0 if the two pairs are equal.
139      */
140     static int compareVersions(int major1, int minor1, int major2, int minor2)
141     {
142 	if (major1 != major2) {
143 	    return (major1 > major2) ? 1 : -1;
144 	} else if (minor1 != minor2) {
145 	    return (minor1 > minor2) ? 1 : -1;
146 	} else {
147 	    return 0;
148 	}
149     }
150 }
