19  package org.apache.river.jeri.internal.http;
20  
21  import java.io.IOException;
22  import java.net.Socket;
23  
24  /**
25   * Abstraction for objects which provide/configure sockets used by
26   * HttpClientConnection instances.
27   *
28   * @author Sun Microsystems, Inc.
29   * 
30   */
31  public interface HttpClientSocketFactory {
32      
33      /**
34       * Creates client socket connected to the given host and port.
35       */
36      Socket createSocket(String host, int port) throws IOException;
37      
38      /**
39       * Creates layered socket on top of given base socket, for use when
40       * tunneling HTTP messages through a proxy.
41       */
42      Socket createTunnelSocket(Socket s) throws IOException;
43  }
