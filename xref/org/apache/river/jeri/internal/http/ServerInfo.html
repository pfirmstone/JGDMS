46      /** blank timestamp value */
47      static final long NO_TIMESTAMP = -1L;
48      
49      /** hexadecimal char conversion table */
50      private static final char[] hexChars = {
51  	'0', '1', '2', '3', '4', '5', '6', '7',
52  	'8', '9', 'a', 'b', 'c', 'd', 'e', 'f'
53      };

55      /** server host name */
56      final String host;
57      /** server port */
58      final int port;
59      /** HTTP major version */
60      int major = 1;
61      /** HTTP minor version */
62      int minor = 0;
63      /** authentication scheme, if any */
64      String authScheme;
65      /** authentication realm */
66      String authRealm;
67      /** authentication algorithm */
68      String authAlgorithm;
69      /** authentication cookie */
70      String authOpaque;
71      /** authentication challenge */
72      String authNonce;
73      /** authentication username */
74      String authUser;
75      /** authentication password */
76      String authPassword;
77      /** time of last update */
78      long timestamp = NO_TIMESTAMP;

88      /**
89       * Sets authentication information based on contents of given challenge
90       * string (which should be the value of either the "WWW-Authenticate" or
91       * "Proxy-Authenticate" header fields).  If given string is null or empty,
92       * clears any previous authentication information.
93       */
94      void setAuthInfo(String str) throws HttpParseException {
95  	if (str == null || str.length() == 0) {
96  	    authScheme = null;
97  	    authRealm = null;
98  	    authAlgorithm = null;
99  	    authOpaque = null;
100 	    authNonce = null;
101 	    authUser = null;
102 	    authPassword = null;
103 	    return;
104 	}
105 	
106 	LineParser lp = new LineParser(str);
107 	Map entries = lp.getEntries("Digest");
108 	if (entries != null) {
109 	    String realm = (String) entries.get("realm");
110 	    String nonce = (String) entries.get("nonce");
111 	    if (realm != null && nonce != null) {
112 		authScheme = "Digest";
113 		authRealm = realm;
114 		authNonce = nonce;
115 		authAlgorithm = (String) entries.get("algorithm");
116 		authOpaque = (String) entries.get("opaque");
117 		if (!"true".equalsIgnoreCase((String) entries.get("stale"))) {
118 		    authUser = null;
119 		    authPassword = null;
120 		}
121 		return;
122 	    }
123 	}
124 
125 	if ((entries = lp.getEntries("Basic")) != null) {
126 	    String realm = (String) entries.get("realm");
127 	    if (realm != null) {
128 		authScheme = "Basic";
129 		authRealm = realm;
130 		authAlgorithm = null;
131 		authOpaque = null;
132 		authNonce = null;
133 		authUser = null;
134 		authPassword = null;
135 		return;
136 	    }
137 	}
138 
139 	// REMIND: no supported schemes found; clear auth info?
140     }

142     /**
143      * Updates authentication information based on contents of given string
144      * (which should be the value of either the "Authorization-Info" or
145      * "Proxy-Authorization-Info" header fields).  If given string is null or
146      * empty, current authentication settings are left unchanged.
147      */
148     void updateAuthInfo(String str) throws HttpParseException {
149 	if (str == null || str.length() == 0) {
150 	    return;
151 	}
152 	if ("Digest".equals(authScheme)) {
153 	    LineParser lp = new LineParser(str);
154 	    Map entries = lp.getAllEntries();
155 	    String nextNonce = (String) entries.get("nextnonce");
156 	    if (nextNonce != null) {
157 		authNonce = nextNonce;
158 	    }
159 	}
160     }

162     /**
163      * Returns (possibly null) authorization string based on current
164      * authentication information in conjunction with the given request
165      * arguments.
166      */
167     String getAuthString(String protocol, String method, String uri) {
168 	if (authScheme == null) {
169 	    return null;
170 	}
171 
172 	if (authUser == null) {
173 	    PasswordAuthentication pa = getPassword(protocol);
174 	    if (pa == null) {
175 		return null;
176 	    }
177 	    String user = pa.getUserName();
178 	    char[] password = pa.getPassword();
179 	    if (user == null || password == null) {
180 		return null;
181 	    }
182 	    authUser = user;
183 	    authPassword = new String(password);
184 	}
185 	
186 	if (authScheme.equals("Basic")) {
187 	    BASE64Encoder enc = new BASE64Encoder();
188 	    try {
189 		return "Basic " +
190 			enc.encode((authUser + ":" + authPassword).getBytes("UTF-8"));
191 	    } catch (UnsupportedEncodingException ex) {
192 		return null;
193 	    }
194 	} else if (authScheme.equals("Digest")) {
195 	    String digest;
196 	    try {
197 		digest = computeDigest(method, uri);
198 	    } catch (NoSuchAlgorithmException ex) {
199 		return null;
200 	    } catch (UnsupportedEncodingException ex) {
201 		return null;
202 	    }
203 
204 	    String response = "Digest " +
205 		"username=\"" + authUser + "\", " +
206 		"realm=\"" + authRealm + "\", " +
207 		"nonce=\"" + authNonce + "\", " +
208 		"uri=\"" + uri + "\", " +
209 		"response=\"" + digest + "\"";
210 	    if (authOpaque != null) {
211 		response += ", opaque=\"" + authOpaque + "\"";
212 	    }
213 	    if (authAlgorithm != null) {
214 		response += ", algorithm=" + authAlgorithm;
215 	    }
216 	    return response;
217 	} else {
218 	    throw new InternalError();
219 	}
220     }

222     /**
223      * Computes digest authentication response for request using the given
224      * method and uri.  Throws NoSuchAlgorithmException if server-specified
225      * digest algorithm not supported.
226      */
227     private String computeDigest(String method, String uri) 
228 	throws NoSuchAlgorithmException, UnsupportedEncodingException
229     {
230 	// REMIND: cache MessageDigest?
231 	MessageDigest md = MessageDigest.getInstance(
232 	    (authAlgorithm != null) ? authAlgorithm : "MD5");
233 	String hashA1 = 
234 	    encode(md, authUser + ":" + authRealm + ":" + authPassword);
235 	String hashA2 = encode(md, method + ":" + uri);
236 	return encode(md, hashA1 + ":" + authNonce + ":" + hashA2);
237     }
238     
239     /**
240      * Returns digest of the given string, represented as string of hexadecimal
241      * digits.
242      */
243     private String encode(MessageDigest md, String str) throws UnsupportedEncodingException {
244 	md.reset();
245 	byte[] digest = md.digest(str.getBytes("UTF-8"));
246 	StringBuffer sbuf = new StringBuffer(digest.length * 2);
247 	for (int i = 0; i < digest.length; i++) {
248 	    sbuf.append(hexChars[(digest[i] >>> 4) & 0xF]);
249 	    sbuf.append(hexChars[digest[i] & 0xF]);
250 	}
251 	return sbuf.toString();
252     }

262     /**
263      * Class for parsing multi-part HTTP header lines that may appear as the
264      * values of the WWW-Authenticate, Proxy-Authenticate, Authorization-Info
265      * or Proxy-Authorization-Info header lines.
266      */
267     private static class LineParser {
268 	
269 	/* token types */
270 	private static final int EOL    = -1;
271 	private static final int WORD   = 0;
272 	private static final int QUOTE  = 1;
273 	private static final int COMMA  = 2;
274 	private static final int EQUALS = 3;
275 
276 	private final List entries = new ArrayList();
277 	private final char[] ca;
278 	private int pos = 0;
279 	private String tokenString = null;
280 	
281 	/**
282 	 * Attempts to parse the given line into a series of key/optional value
283 	 * definitions.  Throws an HttpParseException if the line contains
284 	 * syntax errors.
285 	 */
286 	LineParser(String line) throws HttpParseException {
287 	    ca = line.toCharArray();
288 	    int tok = nextToken();
289 
290 	    while (tok != EOL) {
291 		String key;
292 
293 		if (tok == COMMA) {
294 		    tok = nextToken();
295 		    continue;
296 		} else if (tok == WORD) {
297 		    key = tokenString;
298 		} else {
299 		    throw new HttpParseException("illegal key");
300 		}
301 		
302 		tok = nextToken();
303 		if (tok == COMMA) {
304 		    entries.add(new String[] { key, null });
305 		    tok = nextToken();
306 		    continue;
307 		} else if (tok != EQUALS) {
308 		    entries.add(new String[] { key, null });
309 		    continue;
310 		}
311 		
312 		tok = nextToken();
313 		if (tok != WORD && tok != QUOTE) {
314 		    throw new HttpParseException("illegal value");
315 		}
316 		entries.add(new String[] { key, tokenString });
317 		
318 		tok = nextToken();
319 		if (tok == COMMA) {
320 		    tok = nextToken();
321 		    continue;
322 		} else if (tok != EOL) {
323 		    throw new HttpParseException("illegal separator");
324 		}
325 	    }
326 	}
327 	
328 	/**
329 	 * Returns code indicating next token in line.  If token type is WORD,
330 	 * tokenString is set to the word text; if returned type is QUOTE,
331 	 * tokenString is set to the quoted string's contents.
332 	 */
333 	private int nextToken() throws HttpParseException {
334 	    int mark;
335 
336 	    while (pos < ca.length && Character.isWhitespace(ca[pos])) {
337 		pos++;
338 	    }
339 	    if (pos >= ca.length) {
340 		return EOL;
341 	    }
342 	    switch (ca[pos]) {
343 		case ',':
344 		    pos++;
345 		    return COMMA;
346 		    
347 		case '=':
348 		    pos++;
349 		    return EQUALS;
350 		    
351 		case '\"':
352 		    mark = ++pos;
353 		    while (pos < ca.length && ca[pos] != '\"') {
354 			pos++;
355 		    }
356 		    if (pos >= ca.length) {
357 			throw new HttpParseException(
358 			    "unterminated quote string");
359 		    }
360 		    tokenString = new String(ca, mark, pos++ - mark);
361 		    return QUOTE;
362 		    
363 		default:
364 		    mark = pos;
365 		    while (pos < ca.length) {
366 			char c = ca[pos];
367 			if (c == ',' || c == '=' || c == '\"' ||
368 			    Character.isWhitespace(c))
369 			{
370 			    break;
371 			}
372 			pos++;
373 		    }
374 		    tokenString = new String(ca, mark, pos - mark);
375 		    return WORD;
376 	    }
377 	}
378 
379 	/**
380 	 * Returns all key/value entries associated with the given
381 	 * authorization scheme in the parsed line, or null if the given scheme
382 	 * was not described in the parsed line.  Authorization scheme
383 	 * identifiers are those which appear without a subsequent '=' or ','
384 	 * character before the next word; key/value entries are associated
385 	 * with the nearest preceding scheme identifier.  All key strings are
386 	 * converted to lower case.
387 	 */
388 	Map getEntries(String scheme) {
389 	    Map map = null;
390 	    String[][] ea = 
391 		(String[][]) entries.toArray(new String[entries.size()][]);
392 
393 	    int i;
394 	    for (i = 0; i < ea.length; i++) {
395 		if (ea[i][1] == null && scheme.equalsIgnoreCase(ea[i][0])) {
396 		    map = new HashMap();
397 		    break;
398 		}
399 	    }
400 	    if (map != null) {
401 		for (i++; i < ea.length && ea[i][1] != null; i++) {
402 		    map.put(ea[i][0].toLowerCase(), ea[i][1]);
403 		}
404 	    }
405 	    return map;
406 	}
407 	
408 	/**
409 	 * Returns the key/value entries encountered in the parsed line.  This
410 	 * method should be used to obtain the parse results for
411 	 * Authorization-Info and Proxy-Authorization-Info values, which do not
412 	 * contain authorization scheme identifiers.  All key strings are
413 	 * converted to lower case.
414 	 */
415 	Map getAllEntries() {
416 	    Map map = new HashMap();
417 	    String[][] ea = 
418 		(String[][]) entries.toArray(new String[entries.size()][]);
419 	    for (int i = 0; i < ea.length; i++) {
420 		map.put(ea[i][0].toLowerCase(), ea[i][1]);
421 	    }
422 	    return map;
423 	}
424     }

426     /**
427      * Obtains PasswordAuthentication from the currently installed
428      * Authenticator.
429      */
430     private PasswordAuthentication getPassword(final String protocol) {
431 	return (PasswordAuthentication) AccessController.doPrivileged(
432 	    new PrivilegedAction() {
433 		public Object run() {
434 		    InetAddress addr = null;
435 		    try {
436 			addr = InetAddress.getByName(host);
437 		    } catch (UnknownHostException ex) {
438 		    }
439 
440 		    return Authenticator.requestPasswordAuthentication(
441 			addr, port, protocol, authRealm, authScheme);
442 		}
443 	    }
444 	);
445     }
