86     /*
87      * Table of valid state transitions which a
88      * TransactionManager may make.
89      *
90      * This represents the following diagram from the
91      * Jini Transaction Spec:
92      *
93      *  ACTIVE ----> VOTING ------->COMMITTED
94      *     \           |
95      *      \          |
96      *       ---------------------->ABORTED
97      *
98      *
99      *		    ACTIVE  VOTING  PREPARED  NOTCHANGED  COMMITTED  ABORTED
100     * ----------------------------------------------------------------------
101     * ACTIVE	    true     true   false     false       false       true
102     * VOTING        false    true   false     false       true        true
103     * PREPARED      false    false  false     false       false       false
104     * NOTCHANGED    false    false  false     false       false       false
105     * COMMITTED     false    false  false     false       true        false
106     * ABORTED       false    false  false     false       false       true
107     *
108     * The table is indexed using the ordered pair
109     * <current_state, next_state>.  A value of true means
110     * that the transition is possible, while a false
111     * means that the transition is not possible.
112     *
113     * Note:  Some rows are "{false, false, false, false}" as
114     * 	     unused filler to account for the fact that the
115     *	     TransactionManager's valid states are a subset
116     *	     of the TransactionConstants.
117     *
118     *    <zero>
119     *    ACTIVE = 1
120     *    VOTING = 2
121     *    PREPARED = 3
122     *    NOTCHANGED = 4
123     *    COMMITTED = 5
124     *    ABORTED = 6
125     */
126     private static final boolean states[][] = {
127 	/* <zero>    */ {false, false, false, false, false, false, false},
128 	/* ACTIVE    */ {false, true,  true,  false, false, false, true},
129 	/* VOTING    */ {false, false, true,  false, false, true,  true},
130 	/* PREPARED  */ {false, false, false, false, false, false, false},
131 	/* NOTCHANGED*/ {false, false, false, false, false, false, false},
132 	/* COMMITTED */ {false, false, false, false, false, true,  false},
133 	/* ABORTED   */ {false, false, false, false, false, false, true}};

161    /**
162     * "Parallelizing" the interaction between the manager
163     * and participants means using threads to interact with
164     * participants on behalf of the manager. In the thread
165     * pool model, a TaskManager provides a finite set of
166     * threads used to accomplish a variety of tasks.
167     * A Job encapsulates a body of work which needs to be
168     * performed during the two-phase commit: preparing,
169     * committing and aborting.  Each work item is broken
170     * into smaller pieces of work- interactions with a
171     * single participant- each assigned to a task.
172     *
173     * When a transaction is committing, a PrepareJob is
174     * created and its tasks are scheduled.  After completion,
175     * the PrepareJob's outcome is computed. Depending on
176     * the outcome, either an AbortJob or CommitJob is
177     * scheduled.  When a transaction is aborted, an AbortJob
178     * is scheduled.
179     *
180     * A caller may specify a timeout value for commit/abort.
181     * The timeout represents the length of time a caller is
182     * willing to wait for participants to be instructed to
183     * roll-forward/back.  Should this timeout expire, a
184     * TimeoutExpiredException is thrown.  This causes the
185     * caller's thread to return back to the caller.  Someone
186     * needs to finish contacting all the participants.  This
187     * is accomplished by the SettlerTask.  SettlerTasks must
188     * use a different thread pool from what is used by the
189     * various Job types, otherwise deadlock will occur.
190     *
191     * @serial
192     */
193     private final ExecutorService threadpool;

410      * method makes only valid state changes and informs the

435

439         if (operationsLogger.isLoggable(Level.FINER)) {
440             operationsLogger.exiting(TxnManagerTransaction.class.getName(), 
441 	        "modifyTxnState", Boolean.valueOf(result));
442 	}

447     /**
448      * Implementation of the join method. 
449      *
450      * @param preparedPart The joining <code>TransactionParticpant</code>
451      *
452      * @param crashCount The crashcount associated with the joining
453      *			 <code>TransactionParticipant</code>
454      *
455      * @see net.jini.core.transaction.server.TransactionParticipant
456      */
457     public void
458 	join(TransactionParticipant preparedPart, long crashCount)
459 	    throws CannotJoinException, CrashCountException, RemoteException
460     {
461         if (operationsLogger.isLoggable(Level.FINER)) {
462             operationsLogger.entering(TxnManagerTransaction.class.getName(), 
463 	        "join", new Object[] {preparedPart, Long.valueOf(crashCount)});
464 	}
465 	//if the lease has expired, or the state is not
466 	//amenable there is no need to continue
467         synchronized (jobLock){ // Following checked atomically.
468             if (getState() != ACTIVE)
469                 throw new CannotJoinException("not active");
470             if ((getState() == ACTIVE) && (ensureCurrent() == false)) {
471                 doAbort(0);
472                 throw new CannotJoinException("Lease expired");
473             }
474         }
475 
476 	//Create a ParticipantHandle for the new participant
477 	//and mark the transactional state as ACTIVE
478 	try {
479 	    ParticipantHandle ph = 
480 	        new ParticipantHandle(preparedPart, crashCount);
481 	    ParticipantHandle phtmp = null;
482 	    // FindBugs noticed that phtmp was null, and null check on known null
483 	    // referenc original programmer had assigned it to ph instaed of phtmp.
484 	    // This might have resulted in ParticipantHandles being 
485 	    // added to parts more than once, and CrashCountException to have never been
486 	    // thrown.
487 	    synchronized (parts){
488 		int index = parts.indexOf(ph);
489 		if (index > -1) phtmp = parts.get(index);
490 	    }
491 
492             if (transactionsLogger.isLoggable(Level.FINEST)) {
493                 transactionsLogger.log(Level.FINEST,
494                 "Retrieved ParticipantHandle: {0}", phtmp);
495             }
496 	    if (phtmp != null) {
497 		long oldcount = phtmp.getCrashCount();
498 		if (oldcount == crashCount) {
499 		    return;
500 		} else {
501 		    throw new CrashCountException("TxnManagerTransaction: " +
502 					    "join: old = " + oldcount +
503 					    " new = " + crashCount);
504 		}
505 	    }
506 
507 	    add(ph);
508 
509 	} catch (InternalManagerException ime) {
510             if (transactionsLogger.isLoggable(Level.SEVERE)) {
511                 transactionsLogger.log(Level.SEVERE,
512                 "TransactionParticipant unable to join", ime);
513             }
514 	    throw ime;
515 	} catch (RemoteException re) {
516             if (transactionsLogger.isLoggable(Level.FINEST)) {
517                 transactionsLogger.log(Level.FINEST,
518                 "TransactionParticipant unable to be stored", re);
519             }
520 	    throw re;
521 	}
522         if (operationsLogger.isLoggable(Level.FINER)) {
523             operationsLogger.exiting(TxnManagerTransaction.class.getName(), 
524 	        "join");
525 	}
526     }

554             int state = getState();

562      * Atomic check that state is ACTIVE.
563      * @return ACTIVE if lease hasn't expired.
564      * @throws TransactionException otherwise.
565      */

567         synchronized (stateLock){

576     /**
577      * Commits the transaction.
578      * This initiates the two-phase commit protocol.  First,
579      * each <code>net.jini.core.transaction.server.TransactionParticipant</code>
580      * in the set of participants joined in the
581      * <code>net.jini.core.transaction.server.Transaction</code>
582      * is instructed to vote and the votes are tallied. This is the
583      * first phase (prepare phase).
584      *
585      * Depending on the outcome of the votes, the transaction
586      * is considered committed or aborted.  Once commit/abort
587      * status is known, the participants are notified with
588      * a message to either roll-forward (commit case) or 
589      * roll-back (abort case).  This is the roll-phase.
590      * 
591      * Since there may be a one-to-many relationship between
592      * a transaction and its participants,
593      * <code>org.apache.river.thread.TaskManager</code>s are used
594      * as a generic mechanism to provide the threads needed
595      * to interact with the participants.
596      * 
597      */
598     void commit(long waitFor)
599         throws CannotCommitException, TimeoutExpiredException, RemoteException
600     {
601         if (operationsLogger.isLoggable(Level.FINER)) {
602             operationsLogger.entering(TxnManagerTransaction.class.getName(), 
603 	        "commit", Long.valueOf(waitFor));
604 	}
605 	long starttime = System.currentTimeMillis();
606 
607 	//If the transaction has already expired or the state
608 	//is not amenable, don't even try to continue
609         boolean abort = false;
610         try {
611             synchronized (jobLock){
612                 if ((getState() == ACTIVE) && (ensureCurrent() == false)) {
613                     abort = true;
614                     throw new CannotCommitException("Lease expired");
615                 }
616                 if (getState() == ABORTED)
617                     throw new CannotCommitException("attempt to commit " +
618                                                     "ABORTED transaction");
619             }
620         } finally {
621             if (abort) doAbort(0);
622         }
623 
624 	//Check to see if anyone joined the transaction.  Even
625 	//if no one has joined, at this point, attempt to
626 	//get to the COMMITTED state through valid state changes
627 
628         List<ParticipantHandle> joinvec = parthandles();
629  
630         if (joinvec == null) {
631 	    if (!modifyTxnState(VOTING))
632 		throw new CannotCommitException("attempt to commit " +
633 						"ABORTED transaction");
634 
635 	    if (modifyTxnState(COMMITTED))
636                 return;
637 	    else
638 		throw new CannotCommitException("attempt to commit " +
639 						"ABORTED transaction");
640         }
641 
642 	try {
643 	    int numparts = joinvec.size();
644 	    ParticipantHandle[] phs = joinvec.toArray(new ParticipantHandle[numparts]);
645 
646             long now = starttime;
647             long transpired = 0;
648             long remainder = 0;
649 
650 	    ClientLog log = logmgr.logFor(str.id);
651 
652             if (transactionsLogger.isLoggable(Level.FINEST)) {
653                 transactionsLogger.log(Level.FINEST,
654                 "{0} TransactionParticipants have joined", 
655 		Integer.valueOf(numparts));
656             }
657 
658 	    //If commit is called after recovery, do not
659 	    //log a CommitRecord since it already exists
660 	    //exists in the Log file for this transaction.
661 	    //Remember that a log is not invalidated until
662 	    //after the transaction is committed.
663 	    //
664 	    //Only new occurrences of activities requiring
665 	    //logging which happen after recovery should
666 	    //be added to the log file.  So, we add records
667 	    //for voting and roll forward/back activity for
668 	    //ACTIVE participants.
669 	    //
670 	    //If the state cannot validly transition to VOTING,
671 	    //it is either because someone already aborted or
672 	    //committed.  Only throw an exception if  someone
673 	    //has previously aborted.  In the case of a prior
674 	    //commit, fall through and wait for the CommitJob
675 	    //to complete.
676 
677 	    int oldstate = getState();
678 	    Integer result = Integer.valueOf(ABORTED);
679             Exception alternateException = null;
680 
681 
682 	    //On an ACTIVE to VOTING transition, create
683 	    //and schedule a Prepare or PrepareAndCommitJob.
684 	    //If the state transition is VOTING to VOTING,
685 	    //then the PrepareJob has already been created
686 	    //in the past, so just fall through and wait
687 	    //for it to complete.
688 	    //
689 	    //Only log the commit on ACTIVE to VOTING
690 	    //transitions.
691 
692 	    if (modifyTxnState(VOTING)) {
693 
694 		if (oldstate == ACTIVE)
695 		    log.write(new CommitRecord(phs));
696 
697 
698 		//preparing a participant can never override
699 		//the other activities (abort or commit),
700 		//so only set when the job is null.
701 
702 	        synchronized (jobLock) {
703 		    if (job == null) {
704 	                if (phs.length == 1)
705 		            job = new
706 			      PrepareAndCommitJob(
707 				  str, threadpool, wm, log, phs[0], context);
708 	                else
709 	                    job = new PrepareJob(str, threadpool, wm, log, phs, context);
710 
711 	                job.scheduleTasks();
712 		    }
713 		}
714 
715 		//Wait for the PrepareJob to complete.
716 		//PrepareJobs are given maximum time for
717 		//completion.  This is required in order to
718 		//know the transaction's completion status.
719 		//Remember that the timeout ONLY controls how
720 		//long the caller is willing to wait to inform
721 		//participants.  This means that a completion
722 		//status for the transaction MUST be computed
723 		//before consulting the timeout.
724 		//Timeout is ignored until completion status
725 		//is known.  If extra time is left, wait for
726 		//the remainder to inform participants.
727 
728 		//We must explicitly check for Job type
729 		//because someone else could have aborted
730 		//the transaction at this point.
731 
732 
733                 synchronized (jobLock) {
734 		    if ((job instanceof PrepareJob) ||
735 			    (job instanceof PrepareAndCommitJob)) {
736                         try {
737                             if (job.isCompleted(Long.MAX_VALUE)) {
738                                 result = (Integer) job.computeResult();
739                                 if (result.intValue() == ABORTED &&
740                                     job instanceof PrepareAndCommitJob) {
741                                         PrepareAndCommitJob pj = 
742                                             (PrepareAndCommitJob)job;
743                                         alternateException = 
744                                                pj.getAlternateException();
745                                 }
746                             }
747                         } catch (JobNotStartedException jnse) {
748                             //no participants voted, so do nothing
749                             result = Integer.valueOf(NOTCHANGED);
750                         } catch (ResultNotReadyException rnre) {
751                             //consider aborted
752                         } catch (JobException je) {
753                             //consider aborted
754                         }
755 		    }
756 		}
757 	    } else {
758 		//Cannot be VOTING, so we either have
759 		//an abort or commit in progress.
760 
761 		if (getState() == ABORTED)
762 		    throw new CannotCommitException("transaction ABORTED");
763 
764  
765                 //If a CommitJob is already in progress 
766 		//(the state is COMMITTED) cause a fall
767 		//through to the code which waits for
768                 //the CommitJob to complete.
769 
770                 if (getState() == COMMITTED)
771                     result = Integer.valueOf(COMMITTED);
772 	    }
773 
774             if (transactionsLogger.isLoggable(Level.FINEST)) {
775                 transactionsLogger.log(Level.FINEST,
776                 "Voting result: {0}", 
777 		TxnConstants.getName(result.intValue()));
778             }
779 
780 	    switch (result.intValue()) {
781 	      case NOTCHANGED:
782 		break;
783 
784 	      case ABORTED:
785                 now = System.currentTimeMillis();
786                 transpired = now - starttime;
787                 remainder = waitFor - transpired;
788 
789 		if (remainder >=0)
790 		    doAbort(remainder);
791 		else
792 		    doAbort(0);
793 
794                 if (alternateException == null) {
795                     throw new CannotCommitException(
796                         "Unable to commit transaction: "
797                         + getParticipantInfo());
798                 } else {
799                     throw new RemoteException(
800                         "Problem communicating with participant", 
801                         alternateException);
802                 }
803 
804 	      case PREPARED:
805 		//This entrypoint is entered if a PrepareJob
806 		//tallied the votes with an outcome of
807 		//PREPARED.  In order to inform participants,
808 		//a CommitJob must be scheduled.
809 
810 		if(modifyTxnState(COMMITTED)) {
811 //TODO - log committed state record?		
812                     synchronized (jobLock) {
813                         job = new CommitJob(str, threadpool, wm, log, phs, context);
814                         job.scheduleTasks();
815                     }
816 		} else {
817 		    throw new CannotCommitException("attempt to commit " +
818 					        "ABORTED transaction");
819 		}
820 
821 		//Fall through to code with waits
822 		//for CommitJob to complete.
823 
824 
825 	      case COMMITTED:
826 		//This entrypoint is the starting place for the code
827 		//which waits for a CommitJob to complete and
828 		//computes its resulting outcome. In addition,
829 		//the wait time is enforced.  Should the wait time
830 		//expire, a SettlerTask is scheduled on a thread 
831 		//pool.  The SettlerTask is needed to complete
832 		//the commit (instruct participants to roll-forward)
833 		//on behalf of the thread which exits when
834 		//the TimeoutExpiredException is thrown.
835 		//
836                 //It is reached when...
837 		//
838 		// a) A commit was called on the same transaction twice.
839 		//    When the thread comes in on the commit call, a
840 		//    CommitJob already exists and the state is COMMITTED.
841 		//
842 		// b) The normal case where a PrepareJob was found to
843 		//    have prepared the transaction and the tally of
844 		//    votes resulted in a PREPARED outcome.  This causes
845 		//    the state to be changed to COMMITTED and a
846 		//    CommitJob to be created.
847 		//
848 		// c) A PrepareAndCommitJob has successfully prepared
849 		//    a participant which rolled its changes forward.
850 		//
851                 //Note:  By checking to see if the CommitJob is already
852                 //       present, this check allows us to use the same
853 		//	 wait-for-CommitJob code for the regular
854 		//       PREPARE/COMMIT, the COMMIT/COMMIT and
855 		//       the PREPAREANDCOMMIT cases.
856  
857 		  synchronized (jobLock) {
858 		      //A prepareAndCommitJob is done at this
859 		      //point since the TransactionParticipant
860 		      //would have instructed itself to roll
861 		      //forward.
862 
863 		      if (job instanceof PrepareAndCommitJob) {
864 		          if(!modifyTxnState(COMMITTED))
865 			      throw new CannotCommitException("transaction " +
866 							    	"ABORTED");	
867 		          break;
868 		      }
869  
870 
871 		      //If the abort already arrived, then stop
872 
873 		      if (job instanceof AbortJob) 
874 			   throw new CannotCommitException("transaction " +
875 								"ABORTED");
876 		  }
877 
878                   if (getState() != COMMITTED){
879                       synchronized (jobLock){
880                         throw new
881                             InternalManagerException("TxnManagerTransaction: " +
882                                     "commit: " + job + " got bad state: " +
883                                     TxnConstants.getName(result.intValue()));
884                       }
885                   }
886 
887 		now = System.currentTimeMillis();
888 		transpired = now - starttime;
889 
890 		boolean committed = false;
891 
892 		//If the commit is asynchronous then...
893 		//
894 		// a) check to see if the wait time has transpired
895 		//
896 		// b) If it hasn't, sleep for what's left from the wait time
897 
898 		try {
899 		    remainder = waitFor - transpired;
900 		    synchronized (jobLock) {
901 		        if (remainder <= 0 || !job.isCompleted(remainder)) {
902 /*
903  * Note - SettlerTask will kick off another Commit/Abort task for the same txn
904  * which will try go through the VOTING->Commit states again.
905  */
906 //TODO - Kill off existing task? Postpone SettlerTask? 			
907 		            settler.noteUnsettledTxn(str.id);
908 			    throw new TimeoutExpiredException(
909 					    "timeout expired", true);
910 			} else {
911 			    result = (Integer) job.computeResult();
912 			    committed = true;
913 			}
914 		    }
915 		} catch (ResultNotReadyException rnre) {
916 		    //this should not happen, so flag
917 		    //as an error.
918 		} catch (JobNotStartedException jnse) {
919 		    //an error
920 		} catch (JobException je) {
921 		    //an error
922 		}
923 
924 		if (committed)
925 		    break;
926 
927 	      default:
928                   synchronized (jobLock){
929                     throw new InternalManagerException("TxnManagerTransaction: " +
930                                         "commit: " + job + " got bad state: " +
931                                         TxnConstants.getName(result.intValue()));
932                   }
933 	    }
934 
935 	    //We don't care about the result from
936 	    //the CommitJob
937 	    log.invalidate();
938 	} catch (RuntimeException rte) {
939             if (transactionsLogger.isLoggable(Level.FINEST)) {
940                 transactionsLogger.log(Level.FINEST,
941                 "Problem committing transaction", 
942 	         rte);
943             }
944             throw rte;
945 	} catch (LogException le) {
946             if (transactionsLogger.isLoggable(Level.FINEST)) {
947                 transactionsLogger.log(Level.FINEST,
948                 "Problem persisting transaction", 
949 	         le);
950             }
951 	    throw new CannotCommitException("Unable to log");
952 	}
953         if (operationsLogger.isLoggable(Level.FINER)) {
954             operationsLogger.exiting(TxnManagerTransaction.class.getName(), 
955 	        "commit");
956 	}
957     }

960     /**
961      * Aborts the transaction.
962      * This method attempts to set the state of the transaction
963      * to the ABORTED state.  If successful, it then creates
964      * an AbortJob which schedules tasks executed on a
965      * thread pool.  These tasks interact with the participants
966      * joined in this transaction to inform them to roll back.
967      * 
968      * @param waitFor Timeout value which controls how long,
969      *		      the caller is willing to wait for the
970      *		      participants joined in the transaction 
971      *		      to be instructed to roll-back.
972      *
973      * @see org.apache.river.mahalo.AbortJob
974      * @see org.apache.river.mahalo.ParticipantTask
975      * @see org.apache.river.thread.TaskManager
976      * @see net.jini.core.transaction.server.TransactionParticipant
977      */
978     void abort(long waitFor)
979 	throws CannotAbortException, TimeoutExpiredException
980     {
981         if (operationsLogger.isLoggable(Level.FINER)) {
982             operationsLogger.entering(TxnManagerTransaction.class.getName(), 
983 	        "abort", Long.valueOf(waitFor));
984 	}
985 	long starttime = System.currentTimeMillis();
986 
987 	/*
988 	 * Since lease cancellation process sets expiration to 0 
989 	 * and then calls abort, can't reliably check expiration 
990 	 * at this point.
991 	 */
992 //TODO - Change internal, lease logic to call overload w/o expiration check
993 //TODO - Add expiration check to abort for external clients     
994 	try {
995             List<ParticipantHandle> joinvec = parthandles();
996 
997 	    if (joinvec == null) {
998 		if (modifyTxnState(ABORTED))
999 		    return;
1000 		else 
1001 		    throw new
1002 			CannotAbortException("Transaction already COMMITTED");
1003 	    }
1004 
1005             int numparts = joinvec.size();
1006             ParticipantHandle[] phs = joinvec.toArray(new ParticipantHandle[numparts]);
1007 
1008             ClientLog log = logmgr.logFor(str.id);
1009 
1010 
1011 	    //When attempting to abort, if someone has already
1012 	    //committed the transaction, then throw an Exception.
1013 	    //
1014 	    //If an abort is possible, and you find that an AbortJob
1015 	    //is already in progress, let the existing AbortJob
1016 	    //proceed.
1017 	    //
1018 	    //If an abort is possible, but a PrepareJob is in progress,
1019 	    //go ahead an halt the PrepareJob and replace it with
1020 	    //an AbortJob.
1021 
1022 	    if (modifyTxnState(ABORTED)) {
1023                 log.write(new AbortRecord(phs));
1024 
1025 	        synchronized (jobLock) {
1026 	            if (!(job instanceof AbortJob)) {
1027 		        if (job != null)
1028 		            job.stop();
1029 	                job = new AbortJob(str, threadpool, wm, log, phs, context);
1030 	                job.scheduleTasks();
1031 	            }
1032 	        }
1033 	    } else {
1034 		throw new CannotAbortException("Transaction already COMMITTED");
1035 	    }
1036 
1037 
1038 	    //This code waits for an AbortJob to complete and
1039 	    //computes its resulting outcome. In addition,
1040 	    //the wait time is enforced.  Should the wait time
1041 	    //expire, a SettlerTask is scheduled on a thread
1042 	    //pool.  The SettlerTask is needed to complete
1043 	    //the abort (instruct participants to roll-back)
1044 	    //on behalf of the thread which exits when
1045 	    //the TimeoutExpiredException is thrown.
1046 
1047 	    long now = System.currentTimeMillis();
1048 	    long transpired = now - starttime;
1049 
1050 	    Integer result = Integer.valueOf(ACTIVE);
1051 	    boolean aborted = false;
1052 
1053 	    long remainder = waitFor - transpired;
1054 
1055 	    try {
1056 		synchronized (jobLock) {
1057 	            if (remainder<= 0 || !job.isCompleted(remainder)) {
1058 		        settler.noteUnsettledTxn(str.id);
1059 		        throw new TimeoutExpiredException(
1060 				        "timeout expired",false);
1061 		    } else {
1062 	    	       	result = (Integer) job.computeResult();
1063 	    	       	aborted = true;
1064 		    }
1065 		}
1066 	    }  catch (ResultNotReadyException rnre) {
1067 		//should not happen, so flag as error
1068 	    } catch (JobNotStartedException jnse) {
1069 		//error
1070 	    } catch (JobException je) {
1071                 synchronized (jobLock){
1072                     settler.noteUnsettledTxn(str.id);
1073                 }
1074 		throw new TimeoutExpiredException("timeout expired", false);
1075 	    }
1076 
1077 
1078 	    if (!aborted)
1079                 throw new InternalManagerException("TxnManagerTransaction: " +
1080                                 "abort: AbortJob got bad state: " +
1081                 		TxnConstants.getName(result.intValue()));
1082 
1083 	    log.invalidate();
1084         } catch (RuntimeException rte) {
1085             if (transactionsLogger.isLoggable(Level.SEVERE)) {
1086                 transactionsLogger.log(Level.SEVERE,
1087                 "Problem aborting transaction", 
1088 	         rte);
1089             }
1090 	    throw new InternalManagerException("TxnManagerTransaction: " +
1091 						    "abort: fatal error");
1092         } catch (LogException le) {
1093             if (transactionsLogger.isLoggable(Level.FINEST)) {
1094                 transactionsLogger.log(Level.FINEST,
1095                 "Problem persisting transaction", 
1096 	         le);
1097             }
1098             throw new CannotAbortException("Unable to log");
1099         }
1100         if (operationsLogger.isLoggable(Level.FINER)) {
1101             operationsLogger.exiting(TxnManagerTransaction.class.getName(), 
1102 	        "abort");
1103 	}
1104     }

1176                 "Trouble aborting  transaction", te);
1177 	    }

1179             //If abort has problems, swallow
1180             //it because the abort must have
1181             //happened
