19  package org.apache.river.mahalo.proxy;
20  
21  import org.apache.river.landlord.ConstrainableLandlordLease;
22  import org.apache.river.landlord.Landlord;
23  import org.apache.river.landlord.LandlordProxyVerifier;
24  import java.io.IOException;
25  import java.io.Serializable;
26  import java.rmi.RemoteException;
27  import java.util.logging.Level;
28  import java.util.logging.Logger;
29  import net.jini.core.constraint.MethodConstraints;
30  import net.jini.core.constraint.RemoteMethodControl;
31  import net.jini.id.ReferentUuid;
32  import net.jini.id.Uuid;
33  import net.jini.security.TrustVerifier;
34  import net.jini.security.proxytrust.TrustEquivalence;
35  import org.apache.river.api.io.AtomicSerial;
36  import org.apache.river.api.io.AtomicSerial.GetArg;
37  
38  /** Defines a trust verifier for the smart proxies of a Mahalo server. */
39  @AtomicSerial
40  public final class ProxyVerifier implements TrustVerifier, Serializable {
41  
42      private static final long serialVersionUID = 1L;
43  
44      /** Logger for logging information about this instance */
45      private static final Logger logger = 
46  	Logger.getLogger("net.jini.security.trust");
47  
48      /** The Mahalo server proxy. */
49      private final RemoteMethodControl serverProxy;
50  
51      /**
52       * The unique identifier associated with the backend server referenced
53       * by the <code>serverProxy</code>, used for comparison with the IDs
54       * extracted from the smart proxies being verified.
55       *
56       * @serial
57       */
58      private final Uuid proxyID;
59  
60      /**
61       * Returns a verifier for the smart proxies of the specified Mahalo server
62       * proxy.
63       *
64       * @param serverProxy the Mahalo server proxy
65       * @throws UnsupportedOperationException if <code>serverProxy</code> does
66       *	       not implement both {@link RemoteMethodControl} and {@link
67       *	       TrustEquivalence}
68       */
69      public ProxyVerifier(TxnManager serverProxy, Uuid proxyID) {
70  	this(check(serverProxy, proxyID), serverProxy, proxyID);
71      }
72      
73      ProxyVerifier(GetArg arg) throws IOException{
74  	this((TxnManager) arg.get("serverProxy", null),
75  		(Uuid) arg.get("proxyID", null));
76      }
77      
78      private ProxyVerifier(boolean check,TxnManager serverProxy, Uuid proxyID ){
79  	this.serverProxy = (RemoteMethodControl) serverProxy;
80  	this.proxyID = proxyID;
81      }
82      
83      private static boolean check(TxnManager serverProxy, Uuid proxyID){
84  	if (!(serverProxy instanceof RemoteMethodControl)) {
85  	    throw new UnsupportedOperationException(
86  		"No verifier available for non-constrainable service");
87  	} else if (!(serverProxy instanceof TrustEquivalence)) {
88  	    throw new UnsupportedOperationException(
89  		"Verifier requires service proxy to implement " +
90  		"TrustEquivalence");
91  	} else if (proxyID == null) {
92  	    throw new IllegalArgumentException(
93  	        "Proxy id cannot be null");
94  	}
95  	return true;
96      }
97  
98      /**
99       * @throws NullPointerException {@inheritDoc}
100      */
101     public boolean isTrustedObject(Object obj, TrustVerifier.Context ctx)
102 	throws RemoteException
103     {
104 	if (logger.isLoggable(Level.FINER)) {
105 	    logger.entering(ProxyVerifier.class.getName(), "isTrustedObject",
106 	        new Object[] { obj, ctx });
107 	}
108 	if (obj == null || ctx == null) {
109 	    throw new NullPointerException("Arguments must not be null");
110 	}
111 	RemoteMethodControl otherServerProxy;
112 	Uuid inputProxyID = null;
113 	if (obj instanceof TxnMgrProxy.ConstrainableTxnMgrProxy) {
114 	    otherServerProxy = (RemoteMethodControl) 
115 		((TxnMgrProxy)obj).backend;
116 	    inputProxyID = ((ReferentUuid)obj).getReferentUuid();
117 	} else if (obj instanceof ConstrainableLandlordLease) {
118 	    final LandlordProxyVerifier lpv =
119 		new LandlordProxyVerifier((Landlord)serverProxy, proxyID);
120 	    return lpv.isTrustedObject(obj, ctx);
121 	} else if (
122 	    obj instanceof TxnMgrAdminProxy.ConstrainableTxnMgrAdminProxy) {
123 	    otherServerProxy = (RemoteMethodControl) 
124 		((TxnMgrAdminProxy)obj).server;
125 	    inputProxyID = ((ReferentUuid)obj).getReferentUuid();
126 	} else if (obj instanceof TxnManager &&
127 	           obj instanceof RemoteMethodControl) {
128 	    otherServerProxy = (RemoteMethodControl)obj;
129 	    inputProxyID = proxyID;
130 	} else {
131 	    logger.log(Level.FINEST, "Object {0} is not a supported type",
132 		obj);
133 	    return false;
134 	}
135 
136 	// For top-level proxies, quickly verify proxy Uuid
137 	if ((inputProxyID != null) &&
138 	    !(proxyID.equals(inputProxyID))) {
139 	    return false;
140 	}
141 
142 	MethodConstraints mc = otherServerProxy.getConstraints();
143 	TrustEquivalence trusted =
144 	    (TrustEquivalence) serverProxy.setConstraints(mc);
145 	boolean result = trusted.checkTrustEquivalence(otherServerProxy);
146 	if (logger.isLoggable(Level.FINER)) {
147 	    logger.exiting(ProxyVerifier.class.getName(), "isTrustedObject", 
148 	        Boolean.valueOf(result));
149 	}
150 	return result;
151     }
152 }
