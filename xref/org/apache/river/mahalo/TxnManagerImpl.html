414     public void start() throws Exception {
415         synchronized (this){
416             if (started) return;
417             started = true;
418         }
419         try {
420             if (thrown != null) throw thrown;
421             AccessController.doPrivileged(new PrivilegedExceptionAction<Object>(){
422 
423                 @Override
424                 public Object run() throws Exception {
425                     if(initLogger.isLoggable(Level.FINEST)) {
426                         initLogger.log(Level.FINEST, "Exporting server");
427                     }
428                     serverStub = (TxnManager)exporter.export(TxnManagerImpl.this);
429                     if(initLogger.isLoggable(Level.FINEST)) {
430                         initLogger.log(Level.FINEST, "Server stub: {0}", serverStub);
431                     }
432                     // Create the proxy that will be registered in the lookup service
433                     txnMgrProxy = 
434                         TxnMgrProxy.create(serverStub, topUuid);
435                     if(initLogger.isLoggable(Level.FINEST)) {
436                         initLogger.log(Level.FINEST, "Service proxy is: {0}", 
437                             txnMgrProxy);		
438                     }
439                     // Create the admin proxy for this service
440                     txnMgrAdminProxy = 
441                         TxnMgrAdminProxy.create(serverStub, topUuid);
442                     if(initLogger.isLoggable(Level.FINEST)) {
443                         initLogger.log(Level.FINEST, "Service admin proxy is: {0}", 
444                             txnMgrAdminProxy);		
445                     }
446                     // Create leaseFactory
447                     leaseFactory = new LeaseFactory(serverStub, topUuid);
448                     expMgr = new LeaseExpirationMgr(TxnManagerImpl.this);
449                     if(initLogger.isLoggable(Level.FINEST)) {
450                         initLogger.log(Level.FINEST, "Setting up log manager");
451                     }
452                     if (persistent) {
453                         logmgr = new MultiLogManager(TxnManagerImpl.this, persistenceDirectory);
454                     } else {
455                         logmgr = new MultiLogManager();
456                     }
457 
458                     try {
459                         logmgr.recover();
460                         if(initLogger.isLoggable(Level.FINEST)) {
461                             initLogger.log(Level.FINEST, "Settling incomplete transactions");
462                         }
463 
464                         settleThread.start();
465                     } catch (LogException le) {
466                         RemoteException re =  
467                             new RemoteException("Problem recovering state");
468                         initLogger.throwing(TxnManagerImpl.class.getName(), "doInit", re);
469                         throw re;
470                     }
471 
472                     /*
473                      * With SecureRandom, the first ID requires generation of a
474                      * secure seed, which can take several seconds.  We do it here
475                      * so it doesn't affect the first call's time.  (I tried doing
476                      * this in a separate thread so some of the startup would occur
477                      * during the roundtrip back the client, but it didn't help
478                      * much and this is simpler.)
479                      */
480                     nextID();
481 
482 
483                     /*
484                      * Create the object that manages and persists our join state
485                      */
486                     if(initLogger.isLoggable(Level.FINEST)) {
487                         initLogger.log(Level.FINEST, "Starting JoinStateManager");
488                     }
489                     // Starting causes snapshot to occur
490                     joinStateManager.startManager(config, txnMgrProxy, 
491                         new ServiceID(topUuid.getMostSignificantBits(),
492                                 topUuid.getLeastSignificantBits()),
493                         attributesFor());
494 
495                     if (startupLogger.isLoggable(Level.INFO)) {
496                         startupLogger.log
497                                (Level.INFO, "Mahalo started: {0}", this);
498                     }
499                     readyState.ready();
500 
501                     if (operationsLogger.isLoggable(Level.FINER)) {
502                         operationsLogger.exiting(
503                             TxnManagerImpl.class.getName(), "doInit");
504                     }
505                     return null;
506                 }
507 
508             }, context);
509         } catch (Throwable e) {
510             if (e instanceof PrivilegedActionException){
511                 e = e.getCause();
512             }
513             cleanup();
514 	    initFailed(e); 
515         } finally {
516             // Clear references.
517             config = null;
518             thrown = null;
519         }
520         
521     }

525     public TransactionManager.Created create(long lease)
526         throws LeaseDeniedException
527     {
528         if (operationsLogger.isLoggable(Level.FINER)) {
529             operationsLogger.entering(
530 		TxnManagerImpl.class.getName(), "create", 
531 	        Long.valueOf(lease));
532 	}
533         readyState.check();
534     
535 	TxnManagerTransaction txntr;
536         Lease txnmgrlease = null;
537         ServerTransaction str = null;
538         // following section must be performed atomically, or duplicate ID might be assigned.
539         boolean todoPut = true;
540         while (todoPut){
541             long tid = nextID();
542             Uuid uuid = createLeaseUuid(tid);
543 
544             if (transactionsLogger.isLoggable(Level.FINEST)) {
545                 transactionsLogger.log(Level.FINEST, 
546                     "Transaction ID is: {0}", Long.valueOf(tid));
547             }
548 
549             txntr = new TxnManagerTransaction(
550                 txnMgrProxy, logmgr, tid, taskpool, 
551                 taskWakeupMgr, this, uuid, context);
552             try {
553                 Result r = txnLeasePeriodPolicy.grant(txntr, lease);
554                 txntr.setExpiration(r.expiration);
555                 txnmgrlease = 
556                     leaseFactory.newLease(
557                         uuid,
558                         r.expiration);
559                 		
560             } catch (LeaseDeniedException lde) {
561                 // Should never happen in our implementation.
562                 throw new AssertionError("Transaction lease was denied" + lde);
563             }
564 
565             if (transactionsLogger.isLoggable(Level.FINEST)) {
566                 transactionsLogger.log(Level.FINEST, 
567                     "Created new TxnManagerTransaction ID is: {0}", Long.valueOf(tid));
568             }
569 
570             Transaction tr = txntr.getTransaction();
571 
572             try {
573                 str = serverTransaction(tr);
574                 TxnManagerTransaction existed = txns.putIfAbsent(Long.valueOf(str.id), txntr);
575                 /* This should never happen, but in the improbable event we get a collision */
576                 if (existed == null){ 
577                     todoPut = false;
578                     expMgr.register(txntr);
579                 } else {
580                     
581                 }
582                 if (transactionsLogger.isLoggable(Level.FINEST)) {
583                     transactionsLogger.log(Level.FINEST,
584                         "recorded new TxnManagerTransaction", txntr);
585                 }
586 
587 
588             } catch(Exception e) {
589                 if (transactionsLogger.isLoggable(Level.FINEST)) {
590                     transactionsLogger.log(Level.FINEST,
591                     "Problem creating transaction", e);
592                 }
593                 todoPut = false;
594                 RuntimeException wrap =
595                     new RuntimeException("Unable to create transaction", e);
596                 transactionsLogger.throwing(
597                     TxnManagerImpl.class.getName(), "create", wrap);
598                 throw wrap;
599             }
600         }
601 
602         TransactionManager.Created tmp = 	
603 	    new TransactionManager.Created(str.id, txnmgrlease);		       
604         if (operationsLogger.isLoggable(Level.FINER)) {
605             operationsLogger.exiting(
606 		TxnManagerImpl.class.getName(), "create", tmp);
607 	}
608 
609         return tmp;
610     }

612     public void
613         join(long id, TransactionParticipant part, long crashCount)
614         throws UnknownTransactionException, CannotJoinException,
615 	       CrashCountException, RemoteException
616     {
617         if (operationsLogger.isLoggable(Level.FINER)) {
618             operationsLogger.entering(
619 		TxnManagerImpl.class.getName(), "join", 
620 	        new Object[] {Long.valueOf(id), part, Long.valueOf(crashCount) });
621 	}
622         readyState.check();
623 
624 								
625         TransactionParticipant preparedTarget = null;
626         preparedTarget =
627             (TransactionParticipant)
628                 participantPreparer.prepareProxy(part);
629         
630 	if (participantLogger.isLoggable(Level.FINEST)) {
631             participantLogger.log(Level.FINEST, 
632 	        "prepared participant: {0}", preparedTarget);
633 	}
634 
635         TxnManagerTransaction txntr = txns.get(Long.valueOf(id));
636 
637 	if (txntr == null)
638 	    throw new UnknownTransactionException("unknown transaction");
639 
640 	// txntr.join does expiration check
641 	txntr.join(preparedTarget, crashCount);
642         if (operationsLogger.isLoggable(Level.FINER)) {
643             operationsLogger.exiting(
644 		TxnManagerImpl.class.getName(), "join");
645 	}
