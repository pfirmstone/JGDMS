42  /**
43   *
44   * @author peter
45   */
46  public final class Bootstrap {
47      
48      /**
49       * The bootstrap proxy must be limited to the following interfaces, in case
50       * additional interfaces implemented by the proxy aren't available remotely.
51       */
52      private static final Class[] bootStrapProxyInterfaces = 
53  	{
54  	    ServiceProxyAccessor.class, 
55  	    ServiceAttributesAccessor.class,
56  	    ServiceIDAccessor.class,
57  	    RemoteMethodControl.class,
58  	    TrustEquivalence.class
59  	};
60      
61      private Bootstrap(){}
62      
63      /**
64       * 
65       * @param svc Service instance
66       * @return 
67       */
68      public static Proxy create(Object svc){
69  	Object proxy;
70  	if (svc instanceof ProxyAccessor){ // Obtain exported proxy
71  	    proxy = ((ProxyAccessor)svc).getProxy();
72  	} else {
73  	    proxy = svc;
74  	    // Proxy may not have been exported yet, it is legal for
75  	    // a Serializable object to use writeReplace to export and serialize a proxy.
76  	    // This occurs locally, so we don't require a codebase download.
77  	    // REMIND: Should we set the context ClassLoader?
78  	    if (!Proxy.isProxyClass(proxy.getClass()) && proxy instanceof Serializable){
79  		try {
80  		    ByteArrayOutputStream baos = new ByteArrayOutputStream();
81  		    ObjectOutput out = new ObjectOutputStream(baos);
82  		    out.writeObject(proxy);
83  		    ByteArrayInputStream bain = new ByteArrayInputStream(baos.toByteArray());
84  		    ObjectInput in = new ObjectInputStream(bain);
85  		    proxy = in.readObject();
86  		} catch (IOException ex) {
87  		    Logger.getLogger(Bootstrap.class.getName()).log(Level.FINE, null, ex);
88  		} catch (ClassNotFoundException ex) {
89  		    Logger.getLogger(Bootstrap.class.getName()).log(Level.FINE, null, ex);
90  		}
91  	    }
92  	}
93  	Class proxyClass = proxy.getClass();
94  	if (proxy instanceof RemoteMethodControl //JERI
95  	    && proxy instanceof TrustEquivalence //JERI
96  	    // REMIND: The next three interfaces may not be available locally.
97  	    // so must be found in jsk-dl.jar
98  	    && proxy instanceof ServiceIDAccessor
99  	    && proxy instanceof ServiceProxyAccessor
100 	    && proxy instanceof ServiceAttributesAccessor
101 	    && Proxy.isProxyClass(proxyClass)
102 	    ) 
103 	{
104 	    // REMIND: InvocationHandler must be available locally, for now
105 	    // it must be an instance of BasicInvocationHandler.
106 	    InvocationHandler h = Proxy.getInvocationHandler(proxy);
107 	    return (Proxy) 
108 		Proxy.newProxyInstance(
109 			getProxyLoader(proxyClass),
110 			bootStrapProxyInterfaces, 
111 			h
112 		);
113 	}
114 	return null;
115     }
116     
117     /**
118      * Returns the class loader for the specified proxy class.
119      */
120     private static ClassLoader getProxyLoader(final Class proxyClass) {
121 	return (ClassLoader)
122 	    AccessController.doPrivileged(new PrivilegedAction() {
123 		public Object run() {
124 		    return proxyClass.getClassLoader();
125 		}
126 	    });
127     }
128     
129 }
