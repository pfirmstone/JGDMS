19  package org.apache.river.proxy;
20  
21  
22  import java.io.IOException;
23  import java.io.InvalidObjectException;
24  import java.io.ObjectInputStream;
25  import java.io.Serializable;
26  import net.jini.core.constraint.MethodConstraints;
27  import net.jini.core.constraint.RemoteMethodControl;
28  import net.jini.security.TrustVerifier;
29  import net.jini.security.proxytrust.ServerProxyTrust;
30  import net.jini.security.proxytrust.TrustEquivalence;
31  import org.apache.river.api.io.AtomicSerial;
32  import org.apache.river.api.io.AtomicSerial.GetArg;
33  
34  /**
35   * A basic trust verifier for proxies.  This trust verifier is used to
36   * verify that object passed to its {@link #isTrustedObject
37   * isTrustedObject} method is equivalent in trust, content, and function to
38   * the known trusted object that the trust verifier is constructed with.
39   * This trust verifier is typically returned by an implementation of the
40   * {@link ServerProxyTrust#getProxyVerifier
41   * ServerProxyTrust.getProxyVerifier} method.
42   *
43   * @author Sun Microsystems, Inc.
44   * @since 2.0
45   **/
46  @AtomicSerial
47  public final class BasicProxyTrustVerifier
48      implements TrustVerifier, Serializable
49  {
50      private static final long serialVersionUID = 2L;
51      
52      /**
53       * The trusted proxy.
54       *
55       * @serial
56       */
57      private final RemoteMethodControl proxy;
58  
59      /**
60       * Creates a trust verifier containing the specified trusted proxy. 
61       *
62       * @param proxy the trusted proxy
63       * @throws IllegalArgumentException if the specified proxy is
64       * not an instance of {@link RemoteMethodControl} or
65       * {@link TrustEquivalence}
66       */
67      public BasicProxyTrustVerifier(Object proxy) {
68  	this(check(proxy));
69      }
70      
71      BasicProxyTrustVerifier(GetArg arg) throws IOException{
72  	this(check(arg.get("proxy", null, RemoteMethodControl.class)));
73      }
74          
75      private static RemoteMethodControl check(Object proxy){
76  	if (!(proxy instanceof RemoteMethodControl)) {
77  	    throw new IllegalArgumentException(
78  		"proxy not a RemoteMethodControl instance");
79  	} else if  (!(proxy instanceof TrustEquivalence)) {
80  	    throw new IllegalArgumentException(
81  		"proxy not a TrustEquivalence instance");
82  	}
83  	return (RemoteMethodControl) proxy;
84      }
85      
86      private BasicProxyTrustVerifier(RemoteMethodControl proxy){
87  	this.proxy = proxy;
88      }
89  	
90      /**
91       * Verifies trust in a proxy. Returns <code>true</code> if and only if
92       * the specified object is an instance of {@link RemoteMethodControl}
93       * and invoking the {@link TrustEquivalence#checkTrustEquivalence
94       * checkTrustEquivalence} method on a proxy that is this verifier's
95       * proxy with the same method constraints of the specified object,
96       * passing the specified object returns <code>true</code>.
97       **/
98      public boolean isTrustedObject(Object obj, TrustVerifier.Context ctx) {
99  	if (obj == null || ctx == null) {
100 	    throw new NullPointerException();
101 	} else if (!(obj instanceof RemoteMethodControl)) {
102 	    return false;
103 	}
104 	RemoteMethodControl unverifiedProxy = (RemoteMethodControl) obj;
105 	MethodConstraints constraints = unverifiedProxy.getConstraints();
106 	TrustEquivalence trustedProxy =
107 	    (TrustEquivalence) proxy.setConstraints(constraints);
108 	return trustedProxy.checkTrustEquivalence(unverifiedProxy);
109     }
110 
111     /**
112      * @throws InvalidObjectException if the proxy is not an instance of
113      * {@link RemoteMethodControl} and {@link TrustEquivalence}
114      **/
115     private void readObject(ObjectInputStream in)
116 	throws IOException, ClassNotFoundException
117     {
118 	in.defaultReadObject();
119 	check(proxy);
120     }
121     
122     private static RemoteMethodControl check(RemoteMethodControl proxy) 
123 	    throws InvalidObjectException
124     {
125 	if (proxy == null) {
126 	    throw new InvalidObjectException(
127 		"proxy not a RemoteMethodControl instance");
128 	} else if  (!(proxy instanceof TrustEquivalence)) {
129 	    throw new InvalidObjectException(
130 		"proxy not a TrustEquivalence instance");
131 	}
132 	return (RemoteMethodControl) proxy;
133     }
134 }
