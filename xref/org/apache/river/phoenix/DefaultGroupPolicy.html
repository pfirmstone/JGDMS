19  package org.apache.river.phoenix;
20  
21  import java.rmi.activation.ActivationGroupDesc;
22  import java.security.AccessControlException;
23  import java.security.Permission;
24  import java.util.Enumeration;
25  import java.util.Properties;
26  import net.jini.jeri.BasicInvocationDispatcher;
27  import org.apache.river.phoenix.dl.ExecOptionPermission;
28  import org.apache.river.phoenix.dl.ExecPermission;
29  
30  /**
31   * Group policy that requires the appropriate {@link ExecPermission} and
32   * set of {@link ExecOptionPermission} have been granted to the client
33   * subject or the empty protection domain (a domain with all
34   * <code>null</code> elements) if there is no client subject.
35   *
36   * @author Sun Microsystems, Inc.
37   * 
38   * @since 2.0
39   **/
40  public class DefaultGroupPolicy implements GroupPolicy {
41  
42      /**
43       * Checks permissions for the specified group descriptor. If the group
44       * class name in the descriptor is not <code>null</code> and is not
45       * <code> ActivationGroupImpl</code>, or if the class location or the
46       * initialization data in the descriptor is not <code>null</code>, an
47       * <code>AccessControlException</code> is thrown. For each property in
48       * the descriptor's property overrides, {@link #checkPermission
49       * checkPermission} is called with an {@link ExecOptionPermission}
50       * constructed with a target of the form "-D<i>name</i>=<i>value</i>",
51       * where <i>name</i> is the name of the property and <i>value</i> is the
52       * value of the property. If the command environment in the descriptor is
53       * not <code>null</code>, then additional checks are made as follows. If
54       * the command path is not <code>null</code>, <code>checkPermission</code>
55       * is called with an {@link ExecPermission} constructed with the path as
56       * a target. For each command option, <code>checkPermission</code> is
57       * called with an <code>ExecOptionPermission</code> constructed with the
58       * option as a target.
59       *
60       * @param desc the activation group descriptor
61       * @throws AccessControlException if permission is not granted to create
62       * the specified group
63       */
64      public void checkGroup(ActivationGroupDesc desc) {
65  	String groupClassName = desc.getClassName();
66  	if ((groupClassName != null &&
67  	     !groupClassName.equals(
68  			       "org.apache.river.phoenix.ActivationGroupImpl")) ||
69  	    desc.getLocation() != null ||
70  	    desc.getData() != null)
71  	{
72  	    throw new AccessControlException(
73  		"access denied (custom group implementation not allowed)");
74  	}
75  
76  	Properties props = desc.getPropertyOverrides();
77  	if (props != null) {
78  	    Enumeration p = props.propertyNames();
79  	    while (p.hasMoreElements()) {
80  		String name = (String) p.nextElement();
81  		String option = ("\"-D" + name + "=" +
82  				 props.getProperty(name) + "\"");
83  		checkPermission(new ExecOptionPermission(option));
84  	    }
85  	}
86  
87  	ActivationGroupDesc.CommandEnvironment cmdenv;
88  	cmdenv = desc.getCommandEnvironment();
89  	if (cmdenv != null) {
90  	    String path = cmdenv.getCommandPath();
91  	    if (path != null) {
92  		checkPermission(new ExecPermission(path));
93  	    }
94  
95  	    String[] options = cmdenv.getCommandOptions();
96  	    if (options != null) {
97  		for (int i = 0; i < options.length; i++) {
98  		    checkPermission(new ExecOptionPermission(
99  						   "\"" + options[i] + "\""));
100 		}
101 	    }
102 	}
103     }
104 
105     /**
106      * Calls {@link BasicInvocationDispatcher#checkClientPermission
107      * BasicInvocationDispatcher.checkClientPermission} with the
108      * specified permission.
109      *
110      * @param p the permission being checked
111      * @throws AccessControlException if permission is not granted
112      */
113     protected void checkPermission(Permission p) {
114 	BasicInvocationDispatcher.checkClientPermission(p);
115     }
116 }
