183     /**
184      * Returns the offset of a string in an array of strings.  Returns -1 if
185      * the string is not found.
186      */
187     static int position(String string, String[] list) {
188 	for (int i = list.length; --i >= 0; ) {
189 	    if (string != null && string.equals(list[i])) {
190 		return i;
191 	    }
192 	}
193 	return -1;
194     }

196     /**
197      * Returns the key exchange algorithm for the specified cipher suite. <p>
198      *
199      * The key exchange algorithm is found following the first underscore and
200      * up to the first occurrence of "_WITH_".
201      */
202     static String getKeyExchangeAlgorithm(String cipherSuite) {
203 	int start = cipherSuite.indexOf('_') + 1;
204 	if (start >= 1) {
205 	    int end = cipherSuite.indexOf("_WITH_", start);
206 	    if (end >= start) {
207 		return cipherSuite.substring(start, end);
208 	    }
209 	}
210 	return "NULL";
211     }
212     
213     /**
214      * Returns the key algorithm for the specified cipher suite, one of "RSA",
215      * "DSA", "ECDSA" or "NULL".  Throws an IllegalArgumentException if the algorithm
216      * is not recognized. <p>
217      *
218      * The key algorithm is specified by the key exchange algorithm.
219      */
220     static String getKeyAlgorithm(String cipherSuite) {
221 	String alg = getKeyExchangeAlgorithm(cipherSuite);
222 	if (position(alg, RSA_KEY_EXCHANGE_ALGORITHMS) != -1) {
223 	    return "RSA";
224 	} else if (position(alg, DSA_KEY_EXCHANGE_ALGORITHMS) != -1) {
225 	    return "DSA";
226 	} else if (position(alg, ECDSA_KEY_EXCHANGE_ALGORITHMS) != -1){
227 	    return "ECDSA";
228 	} else if (position(alg, ANONYMOUS_KEY_EXCHANGE_ALGORITHMS) != -1) {
229 	    return "NULL";
230 	} else {
231 	    throw new IllegalArgumentException(
232 		"Unrecognized key exchange algorithm: " + alg);
233 	}
234     }

242     /**
243      * Checks if the X.509 certificates in the CertPath are currently valid.
244      * If issuers is non-null, insures the path contains one of the issuers.
245      */
246     static void checkValidity(CertPath x509CertPath, X500Principal[] issuers) throws CertificateException {
247 	boolean issuerOK = issuers == null;
248 	List certs = x509CertPath.getCertificates();
249 	for (int i = certs.size(); --i >= 0;) {
250 	    X509Certificate cert = (X509Certificate) certs.get(i);
251 	    cert.checkValidity();
252 	    if (!issuerOK) {
253 		X500Principal certIssuer = cert.getIssuerX500Principal();
254 		for (int j = issuers.length; --j >= 0;) {
255 		    if (certIssuer.equals(issuers[j])) {
256 			issuerOK = true;
257 			break;
258 		    }
259 		}
260 	    }
261 	}
262 	if (!issuerOK) {
263 	    throw new CertificateException("No match for permitted issuers: " + toString(issuers) + "\nCertificate chain: " + x509CertPath);
264 	}
265     }

268     /** Converts the contents of an Object array to a String. */
269     static String toString(Object[] array) {
270 	if (array == null) {
271 	    return "null";
272 	}
273 	StringBuilder buf = new StringBuilder("[");
274 	for (int i = 0; i < array.length; i++) {
275 	    if (i != 0) {
276 		buf.append(", ");
277 	    }
278 	    buf.append(array[i]);
279 	}
280 	buf.append("]");
281 	return buf.toString();
282     }

302     /**
303      * Returns the SSLContext to use for creating client
304      * socket factories.  A new SSLContext is created each time.
305      *
306      * @param callContext the client call context
307      * @return an SSLContextInfo containing an SSLContext and ClientAuthManager
308      *	       to use for a connection described by the argument
309      * @throws RuntimeException if an error occurs during initialization of
310      *	       JSSE
311      */
312     static SSLContext getClientSSLContextInfo(Subject clientSubject) {
313 	SSLContext sslContext = null;
314 	ClientSubjectKeyManager authManager = CLIENT_TLS_MANAGER_MAP.get(clientSubject);
315 	while (true){
316 	    if (authManager != null){
317 		try {
318 		    sslContext = CLIENT_TLS_CONTEXT_MAP.get(clientSubject);
319 		    authManager.checkAuthentication();
320 		} catch (GeneralSecurityException ex) {
321 		    CLIENT_TLS_MANAGER_MAP.remove(clientSubject, authManager);
322 		    CLIENT_TLS_CONTEXT_MAP.remove(clientSubject, sslContext);
323 		    sslContext = null;
324 		}
325 	    }
326 	    if (sslContext != null) {
327 		if (CLIENT_LOGGER.isLoggable(Level.FINEST)) {
328 		    CLIENT_LOGGER.log(Level.FINEST,
329 				     "get client SSL context for {0}\nreturns existing {1}",
330 				     new Object[] { clientSubject, sslContext });
331 		}
332 		return sslContext;
333 	    }
334 
335 	    /* Create a new SSL context */
336 	    try {
337 		if (JSSE_PROVIDER != null){
338 		    sslContext = SSLContext.getInstance(SSL_PROTOCOL,JSSE_PROVIDER);
339 		} else {
340 		    sslContext = SSLContext.getInstance(SSL_PROTOCOL);
341 		}
342 	    } catch (NoSuchAlgorithmException e) {
343 		throw initializationError(e, "finding SSL context");
344 	    } catch (NoSuchProviderException e) {
345 		throw initializationError(e, "finding SSL context");
346 	    }
347 
348 	    try {
349 		authManager = new ClientSubjectKeyManager(
350 		    clientSubject
351 		);
352 		ClientSubjectKeyManager existed = CLIENT_TLS_MANAGER_MAP.putIfAbsent(clientSubject, authManager);
353 		if (existed == null) break;
354 		authManager = existed;
355 	    } 
356 	    catch (NoSuchAlgorithmException e) {
357 		throw initializationError(e, "creating key or trust manager");
358 	    } 
359 	    catch (NoSuchProviderException e) {
360 		throw initializationError(e, "creating key manager");
361 	    } 
362 	}
363 	
364 	/* Initialize SSL context */
365 	try {
366 	    SecureRandom random = null;
367 	    if (RAN_ALG != null){
368 		if ( JCE_PROVIDER != null){
369 		    random = SecureRandom.getInstance(RAN_ALG, JCE_PROVIDER);
370 		} else {
371 		    random = SecureRandom.getInstance(RAN_ALG);
372 		}
373 	    }
374 	    sslContext.init(
375 		    new KeyManager[]  { authManager },
376 		    new TrustManager[] { authManager },
377 		    random
378 	    );
379 	} 
380 	catch (KeyManagementException e) {
381 	    throw initializationError(e, "initializing SSL context");
382 	} 
383 	catch (NoSuchAlgorithmException e) {
384 	    throw initializationError(e, "initializing SSL context");
385 	} 
386 	catch (NoSuchProviderException e) {
387 	    throw initializationError(e, "initializing SSL context");
388 	}
389 
390 	if (CLIENT_LOGGER.isLoggable(Level.FINEST)) {
391 	    CLIENT_LOGGER.log(Level.FINEST,
392 			     "get client SSL context for {0}\nreturns new {1}",
393 			     new Object[] { clientSubject, sslContext });
394 	}
395 	SSLContext existed = CLIENT_TLS_CONTEXT_MAP.putIfAbsent(clientSubject, sslContext);
396 	if (existed != null) {
397 	    if (CLIENT_LOGGER.isLoggable(Level.FINEST)) {
398 		CLIENT_LOGGER.log(Level.FINEST,
399 				 "get client SSL context for {0}\nreturns new {1},\n but it already existed {2}",
400 				 new Object[] { clientSubject, sslContext, existed });
401 	    }
402 	}
403 	return sslContext;
404     }

501 		    Level.FINEST,
502 		    "get server SSL context for {0}\n" +
503 		    "returns new {1}",
504 		    new Object[] { subjectString(serverSubject),
505 				   sslContext });
506 	    }
507 	    return sslContext;
508 	}

513 /** Returns a String that includes relevant information about a Subject */
514     static String subjectString(Subject subject) {
515 	if (subject == null) {
516 	    return "null subject";
517 	} else {
518 	    return "Subject@" +
519 		Integer.toHexString(System.identityHashCode(subject)) +
520 		"{\n" + credentialsString(subject) + "}";
521 	}

524     /**
525      * Returns a String that describes the credentials in the subject.
526      *
527      * @param subject the Subject containing the credentials
528      * @return a String describing the credentials
529      * @throws NullPointerException if the subject is null
530      */
531     static String credentialsString(Subject subject) {
532 	List certPaths = getCertificateChains(subject);
533 	if (certPaths == null) {
534 	    return "";
535 	}
536 	StringBuffer buf = new StringBuffer();
537 	for (int i = certPaths.size(); --i >= 0; ) {
538 	    CertPath chain = (CertPath) certPaths.get(i);
539 	    X509Certificate cert = firstX509Cert(chain);
540 	    X500Principal principal = getPrincipal(subject, cert);
541 	    if (principal != null) {
542 		buf.append("  Principal: ").append(principal).append('\n');
543 		buf.append("    Public key: ");
544 		appendKeyString(cert.getPublicKey(), buf);
545 		buf.append('\n');
546 		buf.append("    Private key: ");
547 		try {
548 		    X500PrivateCredential cred =
549 			(X500PrivateCredential) Security.doPrivileged(
550 			    new GetPrivateCredentialAction(subject, cert));
551 		    PrivateKey privateKey =
552 			cred != null ? cred.getPrivateKey() : null;
553 		    if (privateKey == null) {
554 			buf.append("Not found");
555 		    } else {
556 			appendKeyString(privateKey, buf);
557 		    }
558 		} catch (SecurityException e) {
559 		    buf.append("No permission");
560 		}
561 	    }
562 	}
563 	return buf.toString();
564     }

566     /** Appends information about a key to a StringBuffer. */
567     private static void appendKeyString(Key key, StringBuffer buf) {
568 	String className = key.getClass().getName();
569 	buf.append(className.substring(className.lastIndexOf('.') + 1));
570 	buf.append('@');
571 	buf.append(Integer.toHexString(System.identityHashCode(key)));
572     }

583     /**
584      * Returns the latest time for which all of the X.509 certificates in the
585      * certificate chain are valid.
586      */
587     static long certificatesValidUntil(CertPath chain) {
588 	long result = Long.MAX_VALUE;
589 	List<? extends Certificate> certs = chain.getCertificates();
590 	for (int i = certs.size(); --i >= 0;) {
591 	    X509Certificate cert = (X509Certificate) certs.get(i);
592 	    long until = cert.getNotAfter().getTime();
593 	    if (until < result) {
594 		result = until;
595 	    }
596 	}
597 	return result;
598     }

616     /**
617      * Retrieves the X.509 CertPath for a credential name.  Returns null if the
618      * chain associated with the credential name is not found.  Does not check
619      * if either principal or private key associated with the chain are
620      * present.
621      *
622      * @param subject the Subject containing the credentials or null
623      * @param name the name of the credentials
624      * @return the certificate chain or null
625      */
626     static CertPath getCertificateChain(Subject subject, String name) {
627 	if (subject == null) {
628 	    return null;
629 	}
630 	SubjectKeyManager.CertificateMatcher matcher = SubjectKeyManager.CertificateMatcher.create(name);
631 	if (matcher != null) {
632 	    Set publicCreds = subject.getPublicCredentials();
633 	    synchronized (publicCreds) {
634 		for (Iterator it = publicCreds.iterator(); it.hasNext();) {
635 		    Object cred = it.next();
636 		    if (isX509CertificateChain(cred)) {
637 			CertPath chain = (CertPath) cred;
638 			if (matcher.matches(firstX509Cert(chain))) {
639 			    return chain;
640 			}
641 		    }
642 		}
643 	    }
644 	}
645 	return null;
646     }

648     /**
649      * Checks if the subject's public credentials contain a certificate chain
650      * that starts with a certificate with the same subject and public key, and
651      * returns the certificate chain if it does.  Does not check the validity
652      * of the certificate chain, or for associated private credentials or
653      * principal.
654      *
655      * @param cert the certificate
656      * @return the certificate chain starting with an equivalent certificate,
657      *	       if present, otherwise null
658      */
659     static CertPath getCertificateChain(Subject subject, X509Certificate cert) {
660 	if (subject != null) {
661 	    Principal subjectDN = null;
662 	    PublicKey key = null;
663 	    Set publicCreds = subject.getPublicCredentials();
664 	    synchronized (publicCreds) {
665 		for (Iterator it = publicCreds.iterator(); it.hasNext();) {
666 		    Object cred = it.next();
667 		    if (!isX509CertificateChain(cred)) {
668 			continue;
669 		    }
670 		    CertPath chain = (CertPath) cred;
671 		    X509Certificate start = firstX509Cert(chain);
672 		    if (cert.equals(start)) {
673 			return chain;
674 		    }
675 		    if (subjectDN == null) {
676 			subjectDN = cert.getSubjectDN();
677 			key = cert.getPublicKey();
678 		    }
679 		    if (subjectDN.equals(start.getSubjectDN()) && key.equals(start.getPublicKey())) {
680 			return chain;
681 		    }
682 		}
683 	    }
684 	}
685 	return null;
686     }

688     /**
689      * Returns the X.509 CertPaths stored in the public credentials of the
690      * subject.  Does not check if the associated principals or private keys
691      * are present.  Returns null if none are found.
692      *
693      * @param subject the subject containing the X.509 CertPaths or null
694      * @return List of the X.509 CertPaths in the subject
695      */
696     static List getCertificateChains(Subject subject) {
697 	List result = null;
698 	if (subject != null) {
699 	    Set publicCreds = subject.getPublicCredentials();
700 	    synchronized (publicCreds) {
701 		for (Iterator it = publicCreds.iterator(); it.hasNext();) {
702 		    Object cred = it.next();
703 		    if (isX509CertificateChain(cred)) {
704 			if (result == null) {
705 			    result = new ArrayList(publicCreds.size());
706 			}
707 			result.add(cred);
708 		    }
709 		}
710 	    }
711 	}
712 	return result;
713     }

725     /**
726      * Returns the subject principal matching the X.509 certificate.  Returns
727      * null if the principal is not found.  Does not check if the associated
728      * private key is present.  Assumes that the subject is non-null.
729      *
730      * @param subject the Subject containing the credentials
731      * @param cert the X.509 certificate
732      * @return the X.500 principal or null
733      */
734     static X500Principal getPrincipal(Subject subject, X509Certificate cert) {
735 	X500Principal x500 = cert.getSubjectX500Principal();
736 	String name = x500.getName(X500Principal.CANONICAL);
737 	Set principals = subject.getPrincipals();
738 	synchronized (principals) {
739 	    for (Iterator i = principals.iterator(); i.hasNext();) {
740 		Object next = i.next();
741 		if (!(next instanceof X500Principal)) {
742 		    continue;
743 		}
744 		X500Principal principal = (X500Principal) next;
745 		if (principal.getName(X500Principal.CANONICAL).equals(name)) {
746 		    return principal;
747 		}
748 	    }
749 	}
750 	return null;
751     }

753     /**
754      * Returns the X500PrivateCredential for an X.509 certificate.  Returns
755      * null if the associated private credential is missing from the subject.
756      * Does not check if the public credential or principal are present.
757      * Assumes that the subject is non-null.  The caller should check for
758      * AuthenticationPermission and then call this method from within
759      * AccessController.doPrivileged to give it private credential permissions.
760      *
761      * @param subject the Subject containing the credentials
762      * @param cert the X.509 certificate
763      * @return the X500PrivateCredential or null
764      */
765     static X500PrivateCredential getPrivateCredential(Subject subject, X509Certificate cert) {
766 	X500PrivateCredential result = null;
767 	Set privateCreds = subject.getPrivateCredentials();
768 	synchronized (privateCreds) {
769 	    /*
770 	     * XXX: Include this synchronization to work around BugID 4892913,
771 	     * Subject.getPrivateCredentials not thread-safe against changes to
772 	     * principals.  -tjb[18.Jul.2003]
773 	     *
774 	     * synchronized (subject.getPrincipals()) {
775 	     */
776 	    for (Iterator it = privateCreds.iterator(); it.hasNext();) {
777 		Object cred = it.next();
778 		if (cred instanceof X500PrivateCredential) {
779 		    X500PrivateCredential xpc = (X500PrivateCredential) cred;
780 		    if (cert.equals(xpc.getCertificate())) {
781 			result = xpc;
782 			break;
783 		    }
784 		}
785 	    }
786 	}
787 	return result;
788     }

790     /**
791      * Determines if the argument is an X.509 certificate CertPath.  Returns
792      * true if the argument is a non-null CertPath, has at least one
793      * certificate, and has type X.509.
794      */
795     protected static boolean isX509CertificateChain(Object credential) {
796 	if (!(credential instanceof CertPath)) {
797 	    return false;
798 	}
799 	CertPath certPath = (CertPath) credential;
800 	if (certPath.getCertificates().isEmpty()) {
801 	    return false;
802 	} else if (!certPath.getType().equals("X.509")) {
803 	    return false;
804 	}
805 	return true;
806     }

837     /**
838      * Provides utilities for converting between X.509 certificates and unique
839      * certificate names.
840      */
841     static class CertificateMatcher {
842 	private final BigInteger serialNumber;
843 	private final String issuerName;
844 
845 	/**
846 	 * Creates an object that can be compared with an X.509 certificate.
847 	 * Returns null if the argument is not a valid certificate name.
848 	 */
849 	static CertificateMatcher create(String certificateName) {
850 	    if (certificateName == null) {
851 		return null;
852 	    }
853 	    int atSignPosition = certificateName.indexOf('@');
854 	    if (atSignPosition < 0) {
855 		return null;
856 	    }
857 	    BigInteger serialNumber;
858 	    try {
859 		serialNumber = new BigInteger(
860 		    certificateName.substring(0, atSignPosition),
861 		    16);
862 	    } catch (NumberFormatException e) {
863 		return null;
864 	    }
865 	    String issuerName = certificateName.substring(atSignPosition + 1);
866 	    return new CertificateMatcher(serialNumber, issuerName);
867 	}
868 
869 	private CertificateMatcher(BigInteger serialNumber, String issuerName)
870 	{
871 	    this.serialNumber = serialNumber;
872 	    this.issuerName = issuerName;
873 	}
874 
875 	/** Returns the unique certificate name for an X.509 certificate */
876 	static String getName(X509Certificate certificate) {
877 	    /*
878 	     * Use the certificate serial number, which is unique for all
879 	     * certificates from a given issuer, plus the issuer name, to get a
880 	     * unique name for the certificate.
881 	     */
882 	    return certificate.getSerialNumber().toString(16) + "@" +
883 		getIssuerName(certificate);
884 	}
885 
886 	/**
887 	 * Returns true if an X.509 certificate matches the certificate name
888 	 * specified in the constructor.
889 	 */
890 	boolean matches(X509Certificate certificate) {
891 	    return certificate.getSerialNumber().equals(serialNumber)
892 		&& getIssuerName(certificate).equals(issuerName);
893 	}
894 
895 	/** Returns the canonical issuer name for an X.509 certificate. */
896 	private static String getIssuerName(X509Certificate certificate) {
897 	    return certificate.getIssuerX500Principal().getName(
898 		X500Principal.CANONICAL);
899 	}
900     }

906     static class GetPrivateCredentialAction implements PrivilegedAction {
