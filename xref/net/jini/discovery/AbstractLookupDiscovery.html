308 	public void run() {
309             logger.finest("LookupDiscovery - Notifier thread started");
310 	    while (!interrupted()) {
311 		final NotifyTask task;
312                 try {
313                     task = pendingNotifies.takeFirst();
314                 } catch (InterruptedException ex) {
315                     Thread.currentThread().interrupt(); // restore
316                     break;
317                 }
318 		    
319                 /* The call to notify() on the registered listeners is
320                  * performed inside a doPrivileged block that restores the
321                  * access control context that was in place when this utility
322                  * was created.
323                  * 
324                  * This is done because the notify() method called below 
325                  * executes in the client. But the listener object that
326                  * defines that notify() method may have been obtained by
327                  * the client from some (possibly untrusted) 3rd party. With
328                  * respect to 3rd party code executing in the client, it is
329                  * not desirable to allow such code to execute with more
330                  * priviledges than the client that created this utility.
331                  * Therefore, before executing notify() on any of the 
332                  * registered listeners, the client's Subject should be
333                  * restored, and the listner code should be restricted to
334                  * doing nothing more than the client itself is allowed to do.
335                  */
336 		AccessController.doPrivileged
337                 (securityContext.wrap(new PrivilegedAction() {
338 		    public Object run() {
339                         boolean firstListener = true;
340 			for (Iterator iter = task.listeners.iterator();
341 			     iter.hasNext(); )
342 			{
343 			    DiscoveryListener l =
344                                              (DiscoveryListener)iter.next();
345                             /* Always send discovered and discarded events */
346                             if(     (task.eventType == CHANGED)
347                                 && !(l instanceof DiscoveryChangeListener) )
348                             {
349                                 continue;
350                             }//endif
351 			    DiscoveryEvent e =
352 				new DiscoveryEvent
353                                         ( AbstractLookupDiscovery.this,
354                                           deepCopy(task.groupsMap) );
355                             /* Log the event info about the lookup(s) */
356                             if(     firstListener
357                                 && (logger.isLoggable(Level.FINEST)) )
358                             {
359                                 String eType =
360                                        new String[]{"discovered",
361                                                     "discarded",
362                                                     "changed"}[task.eventType];
363                                 ServiceRegistrar[] regs = e.getRegistrars();
364                                 logger.log(Level.FINEST, "{0} event  -- {1} lookup(s)", new Object[]{eType, regs.length});
365                                 Map groupsMap = e.getGroups();
366                                 for(int i=0;i<regs.length;i++) {
367                                     LookupLocator loc = null;
368                                     try {
369                                         loc = regs[i].getLocator();
370                                     } catch (Throwable ex) { /* ignore */ }
371                                     String[] groups = 
372                                              (String[])groupsMap.get(regs[i]);
373                                     logger.log(Level.FINEST, "    {0} locator  = {1}", new Object[]{eType, loc});
374                                     if(groups.length == 0) {
375                                         logger.log(Level.FINEST,"    {0}"+" group    "
376                                                             +"= NO_GROUPS", eType);
377                                     } else {
378                                         for(int j=0;j<groups.length;j++) {
379                                             logger.log(Level.FINEST, "    {0} group[{1}] = {2}", new Object[]{eType, j, groups[j]});
380                                         }//end loop
381                                     }//endif(groups.length)
382                                 }//end loop
383                             }//endif(firstListener && isLoggable(Level.FINEST)
384                             try {
385                         	switch(task.eventType) {
386                                     case DISCOVERED:
387                                 	l.discovered(e);
388                                 	break;
389                                     case DISCARDED:
390                                 	l.discarded(e);
391                                 	break;
392                                     case CHANGED:
393                                 	((DiscoveryChangeListener)l).changed(e);
394                                 	break;
395                         	}//end switch(eventType)
396                             } catch (Throwable t) {
397                                 logger.log(Levels.HANDLED, "a discovery listener failed to process a " +
398                                 	(task.eventType == DISCARDED ? "discard" : task.eventType == DISCOVERED ? "discover" : "changed") + " event", t);
399                             }
400                             firstListener = false;
401 			}//end loop
402 			return null;
403 		    }//end run
404 		}),//end PrivilegedAction and wrap
405                 securityContext.getAccessControlContext());//end doPrivileged
406 	    }//end loop
407 	}//end run

410     /** Thread that listens for multicast announcements from lookup services.
411      *  <p>
412      *  If the announcements are from a lookup service that has not already
413      *  been discovered, and if it is determined that the lookup service 
414      *  belongs to at least one group of interest, a "pendingDiscovery" is
415      *  queued for the UnicastDiscoveryTask to process asynchronously,
416      *  completing the discovery process by performing unicast discovery.
417      *  <p>
418      *  If the announcements are from a lookup service that has already 
419      *  been discovered, the lookup service's member groups - as indicated
420      *  in the announcements - are analyzed for changes that may result
421      *  in either the lookup service being discarded, or in a changed event
422      *  being sent.
423      *  <p>
424      *  Only 1 instance of this thread is run.

430          * which encountered failure when setting the interface or joining
431          * the desired multicast group, and which will be retried periodically.

435 	public AnnouncementListener() throws IOException {
436 	    super("multicast discovery announcement listener");
437 	    setDaemon(false);
438 	    sock = new MulticastSocket(Constants.discoveryPort);
439             switch(nicsToUse) {
440                 case NICS_USE_ALL:
441                     /* Using all interfaces. Skip (but report) any interfaces
442                      * that are "bad" or not configured for multicast.
443                      */
444                     for(int i=0;i<nics.length;i++) {
445                         try {
446                             sock.setNetworkInterface(nics[i]);
447                             sock.joinGroup(Constants.getAnnouncementAddress());
448                         } catch(IOException e) {
449                             if(retryNics == null) {
450                                 retryNics = new ArrayList<NetworkInterface>(nics.length);
451                             }//endif
452                             retryNics.add(nics[i]);
453                             if( logger.isLoggable(Levels.HANDLED) ) {
454                                 LogRecord logRec = 
455                                   new LogRecord(Levels.HANDLED,
456 						"network interface "
457                                                 +"is bad or not configured "
458                                                 +"for multicast: {0}");
459                                 logRec.setParameters(new Object[]{nics[i]});
460                                 logRec.setThrown(e);
461                                 logger.log(logRec);
462                             }//endif
463                         }
464                     }//end loop
465                     break;
466                 case NICS_USE_LIST:
467                     /* Using a configured list of specific interfaces. Skip
468                      * (but report) any interfaces that are "bad" or not
469                      * configured for multicast.
470                      */
471                     for(int i=0;i<nics.length;i++) {
472                         try {
473                             sock.setNetworkInterface(nics[i]);
474                             sock.joinGroup(Constants.getAnnouncementAddress());
475                         } catch(IOException e) {
476                             if(retryNics == null) {
477                                 retryNics = new ArrayList<NetworkInterface>(nics.length);
478                             }//endif
479                             retryNics.add(nics[i]);
480                             if( logger.isLoggable(Level.SEVERE) ) {
481                                 LogRecord logRec = 
482                                   new LogRecord(Level.SEVERE,
483                                                 "network interface is bad or "
484                                                 +"not configured for "
485                                                 +"multicast: {0}");
486                                 logRec.setParameters(new Object[]{nics[i]});
487                                 logRec.setThrown(e);
488                                 logger.log(logRec);
489                             }//endif
490                         }
491                     }//end loop
492                     break;
493                 case NICS_USE_SYS:
494                     /* Using the system-dependent default interface. Don't
495                      * need to specifically set the interface. If that
496                      * interface is "bad" or not configured for multicast,
497                      * log it and try again later.
498                      */
499                     try {
500                         sock.joinGroup(Constants.getAnnouncementAddress());
501                     } catch(IOException e) {
502                         retryNics = new ArrayList<NetworkInterface>(0);
503                         if( logger.isLoggable(Level.SEVERE) ) {
504                             logger.log(Level.SEVERE, "system default network "
505                                        +"interface is bad or not configured "
506                                        +"for multicast", e);
507                         }//endif
508                     }
509                     break;
510                 case NICS_USE_NONE:
511                     break;//multicast disabled, do nothing
512                 default:
513                     throw new AssertionError("nicsToUse flag out of range "
514                                              +"(0-3): "+nicsToUse);
515             }//end switch(nicsToUse)
516 	}//end constructor

534         /** Convenience method that retries any previously failed interfaces.*/
535 	private void retryBadNics() {
536             if(retryNics == null) return;//no failed NICs to retry
537             if( !retryNics.isEmpty() ) {
538                 String recoveredStr = "network interface has recovered "
539                                       +"from previous failure: {0}";
540                 @SuppressWarnings("unchecked")
541                 ArrayList<NetworkInterface> tmpList = (ArrayList<NetworkInterface>) retryNics.clone();
542                 retryNics.clear();
543                 for(int i=0; i<tmpList.size(); i++) {
544                     NetworkInterface nic =tmpList.get(i);
545                     try {
546                         sock.setNetworkInterface(nic);
547                         sock.joinGroup(Constants.getAnnouncementAddress());
548                         if(nicsToUse == NICS_USE_LIST) {
549                             logger.log(Level.INFO, recoveredStr, nic);
550                         } else {
551                             logger.log(Level.FINE, recoveredStr, nic);
552                         }//endif
553                     } catch(IOException e1) {
554                         retryNics.add(nic);//put back for another retry later
555                     }
556                 }//end loop
557                 if(retryNics.isEmpty()) retryNics = null;//future retries off
558             } else {//(retryNics.size() == 0) ==> sys default interface
559                 try {
560                     sock.joinGroup(Constants.getAnnouncementAddress());
561                     retryNics = null;
562                     logger.log(Level.INFO, "system default network "
563                                            +"interface has recovered from "
564                                            +"previous failure");
565                 } catch(IOException e1) { }
566             }//endif(!retryNics.isEmpty())
567 	}//end retryBadNics

569 	public void run() {
570             logger.finest("LookupDiscovery - AnnouncementListener thread "
571                           +"started");
572 	    byte[] buf = new byte[
573 		multicastAnnouncementConstraints.getMulticastMaxPacketSize(
574 		    DEFAULT_MAX_PACKET_SIZE)];
575 	    DatagramPacket pkt = new DatagramPacket(buf, buf.length);
576             long endTime = System.currentTimeMillis() + nicRetryInterval;
577 	    while (!isInterrupted()) {
578 		try {
579                     int delta_t = 0;
580                     if(retryNics != null) {//bad NICs, retry when time's up
581                         delta_t = (int)(endTime - System.currentTimeMillis());
582                         if( delta_t <= 0) {
583                             retryBadNics();
584                             if(retryNics != null) {//still bad, reset timer
585                                 delta_t = nicRetryInterval;
586                                 endTime = System.currentTimeMillis() + delta_t;
587                             } else {//all NICs recovered, turn off timer
588                                 delta_t = 0;
589                             }//endif
590                         }//endif
591                     }//endif
592                     sock.setSoTimeout(delta_t);
593 		    pkt.setLength(buf.length);
594 		    try {

901      * quickly loop and receive new announcement packets; the act of decoding
902      * packets may involve relatively slow cryptographic operations such as
903      * signature verification, and would impede the packet receiving loop if it
904      * were performed inline.
905      */
906     private class DecodeAnnouncementTask implements Runnable {
907 
908 	private final DatagramPacket datagram;
909 
910 	/**
911 	 * Creates a task for decoding the given multicast announcement packet.
912 	 */
913 	public DecodeAnnouncementTask(DatagramPacket datagram) {
914 	    this.datagram = datagram;
915 	}

917 	/**
918 	 * Restore the privileged context and run
919 	 */
920 	public void run() {
921 	    Security.doPrivileged(new PrivilegedAction() {
922 		public Object run() {
923 		    doRun();
924 		    return null;
925 		}//end run
926 	    });//end doPrivileged
927 	}
928 	/**
929 	 * Decodes this task's multicast announcement packet.  If the
930 	 * constraints for decoding multicast announcements are satisfied and
931 	 * the announcement merits further processing, an appropriate object is
932 	 * added to the pendingDiscoveries set, and control is transferred to a
933 	 * UnicastDiscoveryTask.
934 	 */
935 	private void doRun() {
936 	    MulticastAnnouncement ann;
937 	    try {
938 		ann = decodeMulticastAnnouncement(datagram);
939 	    } catch (Exception e) {
940 		if (!(e instanceof InterruptedIOException)) {
941 		    logger.log(Levels.HANDLED,
942 			       "exception decoding multicast announcement", e);
943 		}
944 		return;
945 	    }
946 
947 	    /* If the registrars map contains the service ID of the registrar
948 	     * that sent the current announcement then that registrar has
949 	     * already been discovered.
950 	     * 
951 	     * Determine if the member groups of the already-discovered
952 	     * registrar have been replaced by a set containing none of the
953 	     * desired groups. If yes, then discard the registrar.
954 	     * 
955 	     * If the registrar that sent the current announcement has not
956 	     * already been discovered, then check to see if any of the
957 	     * group(s) in which the registrar is a member are in the set of
958 	     * desired groups to discover. If yes, then queue the registrar for
959 	     * unicast discovery.
960 	     */
961 	    Object pending = null;
962 	    ServiceID srvcID = ann.getServiceID();
963             UnicastResponse resp = registrars.get(srvcID);
964             if (resp != null) {
965                 // already in discovered set, timestamp announcement
966                 AnnouncementInfo aInfo = regInfo.get(srvcID);
967                 if (aInfo == null){
968                     // Possible interleaved putIfAbsent
969                     aInfo = new AnnouncementInfo( System.currentTimeMillis(), -1);
970                     AnnouncementInfo existed = regInfo.putIfAbsent(srvcID, aInfo);
971                     if (existed != null) {
972                         // The time will be almost identical so reuse existing.
973                         aInfo = existed;
974                     }
975                 } else {
976                     AnnouncementInfo newAInfo = new AnnouncementInfo( System.currentTimeMillis(), aInfo.getSeqNum());
977                     if (regInfo.replace(srvcID, aInfo, newAInfo)){ 
978                         aInfo = newAInfo;
979                     } else {
980                         // May have been removed or replaced already
981                         AnnouncementInfo existed = regInfo.putIfAbsent(srvcID, newAInfo);
982                         if (existed != null){
983                             aInfo = existed;
984                         } else {
985                             // Was absent
986                             aInfo = newAInfo;
987                         }
988                     }
989                 }
990                 long currNum = ann.getSequenceNumber();
991                 if ((newSeqNum(currNum, aInfo.getSeqNum())) &&
992                     (!groupSetsEqual(resp.getGroups(), ann.getGroups()))) {
993                     /* Check if the groups have changed. In the case of
994                      * split announcement messages, eventually, group difference
995                      * will be seen for the given sequence number. This
996                      * check ignores other differences, such as port numbers,
997                      * but for the purposes of LookupDiscovery, this is not
998                      * important.
999                      */			
1000                     pending = new CheckGroupsMarker(ann);
1001                 }
1002             } else if (groupsOverlap(ann.getGroups())) {
1003                 // newly discovered
1004                 pending = new LookupLocator(ann.getHost(), ann.getPort());
1005             }
1006 	    if (pending != null) {
1007 		try {
1008 		    checkAnnouncementConstraints(ann);
1009 		} catch (Exception e) {
1010 		    if (!(e instanceof InterruptedIOException)) {
1011 			logger.log(Levels.HANDLED,
1012 			       "exception decoding multicast announcement", e);
1013 		    }
1014 		    return;
1015 		}
1016 		if (pending instanceof CheckGroupsMarker) {
1017                     // Since this is a valid announcement, update the
1018                     // sequence number.
1019                     AnnouncementInfo aInfo = regInfo.get(srvcID);
1020                     if (aInfo != null && !regInfo.replace // Avoid null pointer exception
1021                             (   srvcID, 
1022                                 aInfo, 
1023                                 new AnnouncementInfo( 
1024                                     aInfo.gettStamp(), 
1025                                     ann.getSequenceNumber()
1026                                 ) 
1027                             )
1028                         )
1029                     {
1030                         logger.fine("aInfo changed and was not replaced");
1031                     }
1032 		}
1033 		// enqueue and handle pending action, if not already enqueued
1034 		if (pendingDiscoveries.add(pending)) {
1035 		    if (unicastDelayRange <= 0) {
1036 			new UnicastDiscoveryTask(pending).run();
1037 		    } else {
1038 			final UnicastDiscoveryTask ud =
1039 			    new UnicastDiscoveryTask(pending, true);
1040 			final Ticket t = restoreContextScheduleRunnable(ud);
1041 			synchronized (ud) {
1042 			    ud.ticket = t;
1043 			    ud.delayRun = false;
1044 			    synchronized (tickets) {
1045 				tickets.add(t);
1046 			    }
1047 			    ud.notifyAll();
1048 			}
1049 		    }
1050 		}
1051 	    }
1052 	}

1056 	 * sequence number (for e.g. DiscoveryV1) or the service had been

1074     /** Task which retrieves elements from the set of pendingDiscoveries and
1075      *  performs the appropriate processing based on the object type of
1076      *  the element.
1077      *  <p>
1078      *  Each element of the set of pendingDiscoveries is one of the following
1079      *  object types: Socket, LookupLocator, CheckGroupsMarker,
1080      *  or CheckReachabilityMarker.
1081      *  <p>
1082      *  When the element to process is a Socket, the element was a result
1083      *  of a multicast request/response exchange (see the Requestor and
1084      *  ResponseListener Thread classes). In this case, this task completes
1085      *  the discovery of the associated lookup service by performing the
1086      *  final stage of unicast discovery, ultimately resulting in a discovered
1087      *  event being sent to all registered listeners.
1088      *  <p>
1089      *  When the element to process is a LookupLocator, the element was a
1090      *  result of a multicast announcement received from a lookup service -
1091      *  belonging to at least one group of interest - which has not already
1092      *  been discovered. In this case, this task also completes the discovery
1093      *  of the lookup service referenced in the announcement by performing the
1094      *  final stage of unicast discovery, ultimately resulting in a discovered
1095      *  event being sent to all registered listeners.
1096      *  <p>
1097      *  When the element to process is a CheckGroupsMarker, the
1098      *  element was a result of a multicast announcement received from an
1099      *  already-discovered lookup service whose member groups have changed
1100      *  in some way. In this case, this task determines how those member
1101      *  groups have changed and, based on how they have changed, whether 
1102      *  (or not) to send a discarded event or a changed event to the
1103      *  appropriate registered listeners.
1104      *  <p>
1105      *  When the element to process is a CheckReachabilityMarker, the
1106      *  element was a result of a determination that the multicast 
1107      *  announcements from an already-discovered lookup service have
1108      *  stopped being received (see the AnnouncementTimerThread class).
1109      *  In this case, this task determines if the affected lookup service
1110      *  is still available ("reachable"). If this task cannot communicate
1111      *  with the lookup service, a discarded event is queued to be sent
1112      *  to all registered listeners.
1113      *  <p>
1114      *  Rather than performing unicast discovery synchronously, after multicast
1115      *  discovery has occurred in either the AnnouncementListener thread 
1116      *  (the multicast announcement protocol) or the ResponseListener thread
1117      *  (the multicast request protocol), the unicast discovery processing 
1118      *  that is required to complete the discovery process is queued for
1119      *  asynchronous execution in this task. Unicast discovery is performed
1120      *  asynchronously because unicast discovery can take quite a while to
1121      *  fail if a lookup service "disappears" (because the network or the
1122      *  lookup service itself has crashed) between the time a multicast
1123      *  announcement or response indicates the existence of a lookup service
1124      *  eligible for unicast discovery, and the time unicast discovery 
1125      *  actually starts. If unicast discovery is performed synchronously
1126      *  in the threads that implement the multicast announcement and multicast
1127      *  request protocols, other multicast announcements (as well as other
1128      *  unicast discoveries) will be missed whenever a lookup service
1129      *  disappears prior to the commencement of the unicast discovery stage.
1130      */

1134 	private boolean delayRun = false;

1139 	    this.req = req;
1140 	    this.delayRun = delayRun;
1141 	}

1143 	 * Restore the privileged context and run
1144 	 */

1146 	    Security.doPrivileged(new PrivilegedAction() {
1147 		public Object run() {
1148 		    doRun();
1149 		    return null;
1150 		}//end run
1151 	    });//end doPrivileged
1152 	}
1153 	private void doRun() {
1154             logger.finest("LookupDiscovery - UnicastDiscoveryTask started");
1155 	    try {
1156 		synchronized (this) {
1157 		    while (delayRun) {
1158 			this.wait();
1159 		    }
1160 		    synchronized (tickets) {
1161 			// If this was run by a WakeupManager, remove its
1162 			// ticket from the list of outstanding tickets.
1163 			if (ticket != null) {
1164 			    tickets.remove(ticket);
1165 			}
1166 		    }
1167 		}
1168 		Socket sock = null;
1169 		MulticastAnnouncement announcement = null;
1170 		UnicastResponse response = null;
1171 		if (req instanceof Socket) {
1172 		    // Perform unicast discovery on the connected socket.
1173 		    DiscoveryConstraints unicastDiscoveryConstraints =
1174 			DiscoveryConstraints.process(
1175 			    rawUnicastDiscoveryConstraints);
1176 		    sock = (Socket)req;
1177 		    UnicastResponse resp;
1178 		    try {
1179 			prepareSocket(sock, unicastDiscoveryConstraints);
1180 			resp = doUnicastDiscovery(sock,
1181 						  unicastDiscoveryConstraints);
1182 		    } finally {
1183 			try {
1184 			    sock.close();
1185 			} catch (IOException e) { /* ignore */ }
1186 		    }
1187 		    maybeAddNewRegistrar(resp);
1188 		} else if(req instanceof LookupLocator) {
1189 		    // Perform unicast discovery using the LookupLocator
1190 		    // host and port.
1191 		    LookupLocator loc = (LookupLocator)req;
1192 		    UnicastResponse resp = new MultiIPDiscovery() {
1193 			protected UnicastResponse performDiscovery(
1194 							Discovery disco,
1195 							DiscoveryConstraints dc,
1196 							Socket s)
1197 			    throws IOException, ClassNotFoundException
1198 			{
1199 			    return doUnicastDiscovery(s, dc, disco);
1200 			}
1201 			protected void singleResponseException(Exception e,
1202 							       InetAddress addr,
1203 							       int port)
1204 			{
1205                                 logger.log(
1206 				Levels.HANDLED,
1207 				"Exception occured during unicast discovery " +
1208 				addr + ":" + port, e);
1209                             } 
1210 		    }.getResponse(loc.scheme(), loc.getHost(),
1211 				      loc.getPort(),

2466     /** After a possible change in the member groups of the 
2467      *  <code>ServiceRegistrar</code> corresponding to the given 
2468      *  <code>UnicastResponse</code> parameter, this method
2469      *  determines whether or not the registrar's member groups have
2470      *  changed in such a way that either a changed event or a discarded
2471      *  event is warranted.
2472      *  <p>
2473      *  Note that even if the contents of the new set of groups initially 
2474      *  indicate that the corresponding registrar is a candidate for
2475      *  a discarded or a changed event, further analysis must be performed.
2476      *  This is because there is no guarantee that the new set of member
2477      *  groups have not been "split" across the multicast announcements
2478      *  sent by the lookup service; and so there is no guarantee that the
2479      *  contents of the new group set actually reflect a change that warrants
2480      *  an event. To guarantee that the new group set accurately reflects
2481      *  the registrar's member groups, this method makes a remote call to
2482      *  the registrar to retrieve its actual member groups.
2483      *  <p>
2484      *  There is one situation where it is not necessary to query the
2485      *  registrar for its current member groups. That situation is 
2486      *  when the set of groups input to the <code>newGroups</code> parameter
2487      *  is equivalent to NO_GROUPS. If that new group set is equivalent
2488      *  to NO_GROUPS, it is guaranteed that the registrar's member groups
2489      *  have not been split across the multicast announcements.
2490      *
2491      * @param response  instance of <code>UnicastResponse</code> 
2492      *                  corresponding to the registrar whose current and
2493      *                  previous member groups are to be compared
2494      * @param newGroups <code>String</code> array containing the new
2495      *                  member groups of the registrar corresponding to the 
2496      *                  <code>response</code> parameter (just after a
2497      *                  possible change)
2498      */
2499     private void maybeSendEvent(UnicastResponse response, String[] newGroups) {
2500         ServiceRegistrar reg = response.getRegistrar();
2501         boolean getActual    = true;
2502         if(newGroups == null) { // newGroups null means get actual groups now
2503             newGroups = getActualGroups(reg);
2504             if(newGroups == null) return; // if null, then it was discarded
2505             getActual = false;
2506         }//endif
2507 
2508         if(groupSetsEqual(response.getGroups(),newGroups)) return;
2509 
2510         String[] actualGroups = newGroups;
2511         if( getActual && (newGroups.length > 0) ) {
2512             actualGroups = getActualGroups(reg);
2513             if(actualGroups == null) return; // null ==> was already discarded
2514         }//endif
2515 	
2516         // Other events may have occured to registrars while we were
2517         // making our remote call.
2518         UnicastResponse resp =
2519             registrars.get(reg.getServiceID());
2520         if (resp == null) {
2521             // The registrar was discarded in the meantime. Oh well.
2522             return;
2523         }
2524         notifyOnGroupChange(reg, resp.getGroups(), actualGroups);
2525     }//end maybeSendEvent

2527     /** After a possible change in the member groups of the given
2528      *  <code>ServiceRegistrar</code> parameter, this method compares
2529      *  the registrar's original set of member groups to its new set
2530      *  of member groups.
2531      *  <p>
2532      *  If the criteria shown below is satisfied, either a discarded event
2533      *  or a changed event will be sent to any registered listeners. The
2534      *  criteria is based on whether the old and new groups are equal,
2535      *  and whether one or more elements of the new group set also belong
2536      *  to the set of groups to discover (the new groups are "still of
2537      *  interest"). The criteria is as follows:
2538      *  <p>
2539      *  if (old groups and new groups)
2540      *  <p><ul>
2541      *       <li> (not equal but stillInterested) --&gt; send a changed event
2542      *       <li> (!stillInterested)              --&gt; send a discarded event
2543      *    </ul>
2544      *  <p>
2545      *
2546      * @param reg       instance of <code>ServiceRegistrar</code> 
2547      *                  corresponding to the registrar whose current and
2548      *                  previous member groups are to be compared; and 
2549      *                  whose corresponding service ID is used as the key
2550      *                  into the various data structures that contain
2551      *                  pertinent information about that registrar
2552      * @param oldGroups <code>String</code> array containing the member
2553      *                  groups of the <code>reg</code> parameter prior to
2554      *                  being changed
2555      * @param newGroups <code>String</code> array containing the current
2556      *                  member groups of the <code>reg</code> parameter
2557      *                  (just after a possible change)
2558      */
2559     private void notifyOnGroupChange(ServiceRegistrar reg,
2560                                      String[]         oldGroups,
2561                                      String[]         newGroups)
2562     {
2563         boolean equal           = groupSetsEqual(oldGroups,newGroups);
2564         boolean stillInterested = groupsOverlap(newGroups);
2565         if(!equal && stillInterested) {
2566             sendChanged(reg,newGroups);
2567         } else if(!stillInterested) {
2568             sendDiscarded(reg,newGroups);
2569         }//endif
2570     }//end notifyOnGroupChange

2648     /** This method retrieves from the given <code>ServiceRegistrar</code>,
2649      *  the current groups in which that registrar is a member. If the
2650      *  registrar is un-reachable, then this method will discard the
2651      *  registrar.

2684              * occurred. This means that the reg is unreachable; discard it.
