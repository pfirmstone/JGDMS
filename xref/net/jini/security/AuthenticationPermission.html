37  /**
38   * Represents permission to use the private credentials of subjects for the
39   * purpose of authenticating as any subset of the local principals specified
40   * in the target name, during secure remote calls with any peer that
41   * authenticates as at least the set of peer principals specified in the
42   * target name. In general, security providers check for this permission
43   * instead of checking for
44   * {@link javax.security.auth.PrivateCredentialPermission}. This
45   * permission does not need to be granted for anonymous communication;
46   * it only needs to be granted if an entity needs to authenticate itself.
47   * <p>
48   * An instance of this class contains a name (also referred to as a "target
49   * name") and a set of actions. The target name specifies both the maximum
50   * set of principals that an entity can authenticate as, and the minimum
51   * set of principals that the peer must authenticate as. The actions specify
52   * whether the permission is granted for making outbound remote calls with or
53   * without delegation, listening for incoming remote calls, receiving
54   * incoming remote calls, or some combination.
55   * <p>
56   * The syntax of the target name is either:
57   * <pre><i>LocalPrincipals</i></pre>
58   * or:
59   * <pre><i>LocalPrincipals</i> <code>peer</code> <i>PeerPrincipals</i></pre>
60   * where <i>LocalPrincipals</i> specifies the maximum set of principals that
61   * an entity can authenticate as (that is, the entity can authenticate as any
62   * subset of these principals), and <i>PeerPrincipals</i> specifies the
63   * minimum set of principals that the peer must authenticate as (that is,
64   * the peer must authenticate as at least all of these principals). If the
65   * first syntactic form is used, the peer can authenticate as anyone (and can
66   * be anonymous). The syntax of both <i>LocalPrincipals</i> and
67   * <i>PeerPrincipals</i> is:
68   * <pre><i>PrincipalClass</i> "<i>PrincipalName</i>" ...</pre>
69   * That is, alternating principal classes and principal names, separated by
70   * spaces, with each principal name surrounded by quotes. The order in which
71   * principals are specified does not matter, but both class names and
72   * principal names are case sensitive. For <i>LocalPrincipals</i>, in any
73   * given principal specification, a wildcard value of "*" can be used for
74   * both <i>PrincipalClass</i> and <i>PrincipalName</i> or for just
75   * <i>PrincipalName</i>, but it is illegal to use a wildcard value for just
76   * <i>PrincipalClass</i>. Explicit wildcard values cannot be used in
77   * <i>PeerPrincipals</i>; only complete wildcarding of the peer is supported,
78   * and is expressed by using the first syntactic form instead.
79   * <p>
80   * The syntax of the actions is a comma-separated list of any of the following
81   * (case-insensitive) action names: <code>listen</code>, <code>accept</code>,
82   * <code>connect</code>, <code>delegate</code>. The <code>listen</code> action
83   * grants permission to authenticate as the server when listening for
84   * incoming remote calls; in this case, the peer principals are ignored
85   * (because it is assumed that in general servers authenticate themselves
86   * before clients do). The <code>accept</code> action grants permission to
87   * receive authenticated incoming remote calls; in this case, the entity has
88   * authenticated as the server, and the peer has authenticated as the client.
89   * If the <code>accept</code> action is specified, the <code>listen</code>
90   * action is implied and need not be specified explicitly. The
91   * <code>connect</code> action grants permission to authenticate when
92   * making outgoing remote calls; in this case, the entity authenticates as
93   * the client, and the peer authenticates as the server. The
94   * <code>delegate</code> action grants permission to authenticate with
95   * (or without) delegation when making outgoing remote calls. If the
96   * <code>delegate</code> action is specified, the <code>connect</code>
97   * action is implied and need not be specified explicitly.
98   * <p>
99   * A principal <code>p</code> matches <i>LocalPrincipals</i> if
100  * <i>LocalPrincipals</i> has any of the following principal specifications:
101  * <ul>
102  * <li>"*" for both <i>PrincipalClass</i> and <i>PrincipalName</i>
103  * <li>a <i>PrincipalClass</i> equal to the value of
104  * <code>p.getClass().getName()</code> and a <i>PrincipalName</i> equal to "*"
105  * <li>a <i>PrincipalClass</i> equal to the value of
106  * <code>p.getClass().getName()</code> and a <i>PrincipalName</i> equal to
107  * the value of <code>p.getName()</code>
108  * </ul>
109  * A principal <code>p</code> matches <i>PeerPrincipals</i> if
110  * <i>PeerPrincipals</i> has a <i>PrincipalClass</i> equal to the value of
111  * <code>p.getClass().getName()</code> and a <i>PrincipalName</i> equal to
112  * the value of <code>p.getName()</code>.
113  * <p>
114  * Some example policy file permissions:
115  * <pre>
116  * // client authenticate as jack, with or without delegation, to any server
117  * permission net.jini.security.AuthenticationPermission
118  *     "javax.security.auth.x500.X500Principal \"CN=jack\"", "delegate";
119  *
120  * // client authenticate as joe and/or sue, without delegation, to any server
121  * permission net.jini.security.AuthenticationPermission
122  *     "javax.security.auth.x500.X500Principal \"CN=joe\" javax.security.auth.x500.X500Principal \"CN=sue\"", "connect";
123  *
124  * // client authenticate as any X500 principals, without delegation, to jack
125  * permission net.jini.security.AuthenticationPermission
126  *     "javax.security.auth.x500.X500Principal \"*\" peer javax.security.auth.x500.X500Principal \"CN=jack\"", "connect";
127  *
128  * // authenticate as jack to jack, bi-directional, with or without delegation
129  * permission net.jini.security.AuthenticationPermission
130  *     "javax.security.auth.x500.X500Principal \"CN=jack\" peer javax.security.auth.x500.X500Principal \"CN=jack\"", "accept,delegate";
131  *
132  * // authenticate as anyone to jack, bi-directional, without delegation
133  * permission net.jini.security.AuthenticationPermission
134  *     "* \"*\" peer javax.security.auth.x500.X500Principal \"CN=jack\"", "accept,connect";
135  * </pre>
136  *
137  * @author Sun Microsystems, Inc.
138  * 
139  * @since 2.0
140  */

260     /**
261      * Parses the actions field and initializes the transient mask field.
262      */
263     private static int parseActions(String actions) {
264 	StringTokenizer st = new StringTokenizer(actions, " ,", true);
265 	int mask = 0;
266 	boolean comma = false;
267 	while (st.hasMoreTokens()) {
268 	    String act = st.nextToken();
269 	    if (act.equals(" ")) {
270 		continue;
271 	    } else if (comma) {
272 		if (!act.equals(",")) {
273 		    comma = false;
274 		    break;
275 		}
276 	    } else if (act.equalsIgnoreCase("connect")) {
277 		mask |= CONNECT;
278 	    } else if (act.equalsIgnoreCase("accept")) {
279 		mask |= ACCEPT;
280 	    } else if (act.equalsIgnoreCase("delegate")) {
281 		mask |= DELEGATE;
282 	    } else if (act.equalsIgnoreCase("listen")) {
283 		mask |= LISTEN;
284 	    } else {
285 		break;
286 	    }
287 	    comma = !comma;
288 	}
289 	if (!comma) {
290 	    throw new IllegalArgumentException("invalid actions");
291 	}
292 	return mask;
293     }

295     private static Data parseName(Data data){
296 	StringTokenizer st = new StringTokenizer(data.name, " ", true);
297 	return parseName(st, false, data);

300     /**
301      * Parses what's left of the target name in the specified tokenizer,
302      * and initializes the transient fields. Peer is false when parsing
303      * the local principals, true when parsing the peer principals.
304      */
305     private static Data parseName(StringTokenizer st, boolean peer, Data data) {
306 	List vals = new ArrayList(2);
307     outer:
308 	while (true) {
309 	    String cls;
310 	    do {
311 		if (!st.hasMoreTokens()) {
312 		    break outer;
313 		}
314 		cls = st.nextToken();
315 	    } while (cls.equals(" "));
316 	    if (!peer && cls.equalsIgnoreCase("peer")) {
317 		parseName(st, true, data);
318 		break;
319 	    }
320 	    if (cls.equals("*")) {
321 		if (peer) {
322 		    throw new IllegalArgumentException(
323 						   "peer class cannot be *");
324 		}
325 		cls = null;
326 		vals = null;
327 	    }
328 	    String nm;
329 	    do {
330 		if (!st.hasMoreTokens()) {
331 		    throw new IllegalArgumentException(
332 						 "missing name after class");
333 		}
334 		nm = st.nextToken();
335 	    } while (nm.equals(" "));
336 	    if (!nm.startsWith("\"")) {
337 		throw new IllegalArgumentException("name must be in quotes");
338 	    }
339             StringBuilder sb = new StringBuilder(120);
340             sb.append(nm);
341 	    while (!sb.substring(sb.length()-1).equals("\"")) {
342 		if (!st.hasMoreTokens()) {
343 		    throw new IllegalArgumentException(
344 						   "name must be in quotes");
345 		}
346 		sb.append(st.nextToken());
347 	    }
348             nm = sb.toString();
349 	    if (nm.equals("\"*\"")) {
350 		if (peer) {
351 		    throw new IllegalArgumentException(
352 						"peer name cannot be \"*\"");
353 		}
354 		if (cls == null) {
355 		    continue;
356 		}
357 		nm = null;
358 	    } else if (cls == null) {
359 		throw new IllegalArgumentException(
360 				   "class cannot be * unless name is \"*\"");
361 	    } else {
362 		nm = nm.substring(1, nm.length() - 1);
363 	    }
364 	    if (vals != null) {
365 		for (int i = vals.size(); i > 0; ) {
366 		    String onm = (String) vals.get(--i);
367 		    String ocls = (String) vals.get(--i);
368 		    if (cls.equals(ocls)) {
369 			if (onm == null || onm.equals(nm)) {
370 			    continue outer;
371 			} else if (nm == null) {
372 			    vals.remove(i);
373 			    vals.remove(i);
374 			}
375 		    }
376 		}
377 		vals.add(cls);
378 		vals.add(nm);
379 	    }
380 	}
381 	String[] res = null;
382 	if (vals != null) {
383 	    if (vals.isEmpty()) {
384 		throw new IllegalArgumentException(
385 					  "target name is missing elements");
386 	    }
387 	    res = (String[]) vals.toArray(new String[vals.size()]);
388 	}
389 	if (peer) {
390 	    data.peer = res;
391 	} else {
392 	    data.me = res;
393 	}
394 	return data;
395     }

397     /**
398      * Returns an array of alternating class and principal names for the
399      * specified set of principals, and appends all of those strings to
400      * the specified buffer, separated by spaces, with the principal
401      * names in quotes.
402      */
403     private static String[] cons(Set s, StringBuffer b) {
404 	String[] vals = new String[s.size() * 2];
405 	int i = 0;
406 	for (Iterator iter = s.iterator(); iter.hasNext(); ) {
407 	    Principal p;
408 	    try {
409 		p = (Principal) iter.next();
410 	    } catch (ClassCastException e) {
411 		throw new IllegalArgumentException(
412 					    "sets must contain Principals");
413 	    }
414 	    String v = p.getClass().getName();
415 	    if (i > 0) {
416 		b.append(' ');
417 	    }
418 	    b.append(v);
419 	    vals[i++] = v;
420 	    v = p.getName();
421 	    b.append(" \"");
422 	    b.append(v);
423 	    b.append('"');
424 	    vals[i++] = v;
425 	}
426 	return vals;
427     }

429     /**
430      * Constructs the target name and transient field data for the
431      * specified principal sets.
432      */
433     private static Data parseName(Set me, Set peer) {
434 	if (me == null) {
435 	    throw new NullPointerException(
436 					"local principals must be non-empty");
437 	} else if (me.isEmpty()) {
438 	    throw new IllegalArgumentException(
439 					"local principals must be non-empty");
440 	}
441 	Data data = new Data();
442 	StringBuffer b = new StringBuffer();
443 	data.me = cons(me, b);
444 	if (peer != null && !peer.isEmpty()) {
445 	    b.append(" peer ");
446 	    data.peer = cons(peer, b);
447 	}
448 	data.name = b.toString();
449 	return data;
450     }

452     /**
453      * Returns <code>true</code> if the specified permission is an instance
454      * of <code>AuthenticationPermission</code>, and every action included in
455      * the specified permission is included as an action of this permission,
456      * and every principal that matches the local principals of the specified
457      * permission also matches the local principals of this permission, and
458      * (if the specified permission has any action besides
459      * <code>listen</code>) every principal that matches the peer principals
460      * of this permission also matches the peer principals of the specified
461      * permission; returns <code>false</code> otherwise.
462      *
463      * @param perm the permission to check
464      * @return <code>true</code> if the specified permission is an instance
465      * of <code>AuthenticationPermission</code>, and every action included in
466      * the specified permission is included as an action of this permission,
467      * and every principal that matches the local principals of the specified
468      * permission also matches the local principals of this permission, and
469      * (if the specified permission has any action besides
470      * <code>listen</code>) every principal that matches the peer principals
471      * of this permission also matches the peer principals of the specified
472      * permission; <code>false</code> otherwise
473      */
474     public boolean implies(Permission perm) {
475 	if (!(perm instanceof AuthenticationPermission)) {
476 	    return false;
477 	}
478 	AuthenticationPermission ap = (AuthenticationPermission) perm;
479 	return (mask & ap.mask) == ap.mask && implies0(ap);
480     }
481 
482     private boolean implies0(AuthenticationPermission ap) {
483 	return ((me == null || (ap.me != null && covers(me, ap.me))) &&
484 		(ap.mask == LISTEN || peer == null ||
485 		 (ap.peer != null && covers(ap.peer, peer))));
486     }
487 
488     /**
489      * Returns true if every principal that matches sub also matches sup.
490      */
491     private static boolean covers(String[] sup, String[] sub) {
492     outer:
493 	for (int i = sub.length; i > 0; ) {
494 	    String onm = sub[--i];
495 	    String ocls = sub[--i];
496 	    for (int j = sup.length; j > 0; ) {
497 		String nm = sup[--j];
498 		String cls = sup[--j];
499 		if (cls.equals(ocls) &&
500 		    (nm == null || (onm != null && nm.equals(onm))))
501 		{
502 		    continue outer;
503 		}
504 	    }
505 	    return false;
506 	}
507 	return true;
508     }

528     /**
529      * Two instances of this class are equal if each implies the other;
530      * that is, both instances have the same actions, every principal that
531      * matches the local principals of one instance matches the local
532      * principals of the other instance, and (if the instances have any
533      * action besides <code>listen</code>) every principal that matches the
534      * peer principals of one instance matches the peer principals of the
535      * other instance.
536      */
537     public boolean equals(Object obj) {
538 	if (!(obj instanceof AuthenticationPermission)) {
539 	    return false;
540 	}
541 	AuthenticationPermission ap = (AuthenticationPermission) obj;
542 	return (mask == ap.mask && same(me, ap.me) &&
543 		(mask == LISTEN || same(peer, ap.peer)));
544     }

546     /**
547      * Returns true if both arrays are null, or both arrays are the same
548      * length and contain the same pairs (ignoring order).
549      */
550     private static boolean same(String[] s1, String[] s2) {
551 	if (s1 == null) {
552 	    return s2 == null;
553 	} else if (s2 == null || s1.length != s2.length) {
554 	    return false;
555 	}
556     outer:
557 	for (int i = s2.length; i > 0; ) {
558 	    String onm = s2[--i];
559 	    String ocls = s2[--i];
560 	    for (int j = s1.length; j > 0; ) {
561 		String nm = s1[--j];
562 		String cls = s1[--j];
563 		if (cls.equals(ocls) &&
564 		    (nm == null ? onm == null : nm.equals(onm)))
565 		{
566 		    continue outer;
567 		}
568 	    }
569 	    return false;
570 	}
571 	return true;
572     }

574     /**
575      * Returns a hash code value for this object.
576      */
577     public int hashCode() {
578 	int h = mask;
579 	if (me != null) {
580 	    for (int i = me.length; --i >= 0; ) {
581 		if (me[i] != null) {
582 		    h += me[i].hashCode();
583 		}
584 	    }
585 	}
586 	if (mask != LISTEN && peer != null) {
587 	    for (int i = peer.length; --i >= 0; ) {
588 		h += peer[i].hashCode();
589 	    }
590 	}
591 	return h;
592     }

594     /**
595      * Verifies the syntax of the target name and recreates any transient
596      * state.
597      *
598      * @throws java.io.InvalidObjectException if the target name or actions
599      * string is <code>null</code>, or if the target name or actions string
600      * does not match the syntax specified in the comments at the beginning
601      * of this class
602      */
603     private void readObject(ObjectInputStream s)
604 	throws IOException, ClassNotFoundException
605     {
606 	s.defaultReadObject();
607 	try {
608 	    /**
609 	     * Parses the target name and actions, and initializes the transient
610 	     * fields.
611 	     */
612 	    mask = parseActions(actions);
613 	    Data data = parseName(new Data(getName()));
614 	    me = data.me;
615 	    peer = data.peer;
616 	} catch (RuntimeException e) {
617 	    if (e instanceof NullPointerException ||
618 		e instanceof IllegalArgumentException)
619 	    {
620 		InvalidObjectException ee =
621 		    new InvalidObjectException(e.getMessage());
622 		ee.initCause(e);
623 		throw ee;
624 	    }
625 	    throw e;
626 	}
627     }

630      * @serial include
631      */
632     static class AuthenticationPermissionCollection
633 				extends PermissionCollection
634     {
635 	private static final long serialVersionUID = -2967578431368213049L;
636 
637 	/**
638 	 * @serialField permissions List The permissions.
639 	 */
640 	private static final ObjectStreamField[] serialPersistentFields = {
641 	    new ObjectStreamField("permissions", List.class, true)
642 	};
643 
644 	private List permissions = new ArrayList();
645 
646 	AuthenticationPermissionCollection() {
647 	}
648 
649 	public synchronized void add(Permission perm) {
650 	    if (!(perm instanceof AuthenticationPermission)) {
651 		throw new IllegalArgumentException(
652 			       "element must be an AuthenticationPermission");
653 	    } else if (isReadOnly()) {
654 		throw new SecurityException("collection is read-only");
655 	    }
656 	    permissions.add(perm);
657 	}
658 
659 	public synchronized boolean implies(Permission perm) {
660 	    if (!(perm instanceof AuthenticationPermission)) {
661 		return false;
662 	    }
663 	    AuthenticationPermission ap = (AuthenticationPermission) perm;
664 	    int needed = ap.mask;
665 	    for (int i = permissions.size(); --i >= 0; ) {
666 		AuthenticationPermission cp =
667 		    (AuthenticationPermission) permissions.get(i);
668 		if ((needed & cp.mask) != 0 && cp.implies0(ap)) {
669 		    needed &= ~cp.mask;
670 		    if (needed == 0) {
671 			return true;
672 		    }
673 		}
674 	    }
675 	    return false;
676 	}

705 		permissions = new ArrayList(permissions);

711 			       "element must be an AuthenticationPermission");

713 	    }
