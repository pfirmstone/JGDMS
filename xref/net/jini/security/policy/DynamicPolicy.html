19  package net.jini.security.policy;
20  
21  import java.security.Permission;
22  import java.security.Principal;
23  
24  /**
25   * Interface implemented by security policy providers that may support dynamic
26   * granting of permissions at run-time.  The <code>grant</code> methods of the
27   * <code>net.jini.security.Security</code> class delegate to the
28   * methods declared by this interface when this interface is implemented by the
29   * installed security policy provider.  Permissions are granted on the
30   * granularity of class loader; granting a permission requires (of the calling
31   * context) {@link net.jini.security.GrantPermission} for that
32   * permission.
33   *
34   * @author Sun Microsystems, Inc.
35   * 
36   * @since 2.0
37   */
38  public interface DynamicPolicy {
39      /**
40       * Returns <code>true</code> if this policy provider supports dynamic
41       * permission grants; returns <code>false</code> otherwise.  Note that this
42       * method may return different values for a given
43       * <code>DynamicPolicy</code> instance, depending on context.  For example,
44       * a policy provider that delegates to different underlying policy
45       * implementations depending on thread state would return <code>true</code>
46       * from this method when the current delegate supports dynamic permission
47       * grants, but return <code>false</code> when another delegate lacking such
48       * support is in effect.
49       *
50       * @return  <code>true</code> if policy supports dynamic permission grants
51       * 		under current context, <code>false</code> otherwise
52       */
53      boolean grantSupported();
54  
55      /**
56       * If this security policy provider supports dynamic permission grants,
57       * grants the specified permissions to all protection domains (including
58       * ones not yet created) that are associated with the class loader of the
59       * given class and possess at least the given set of principals.  If the
60       * given class is <code>null</code>, then the grant applies across all
61       * protection domains that possess at least the specified principals.  If
62       * the list of principals is <code>null</code> or empty, then principals
63       * are effectively ignored in determining the protection domains to which
64       * the grant applies.  If this policy provider does not support dynamic
65       * permission grants, then no permissions are granted and an
66       * <code>UnsupportedOperationException</code> is thrown.
67       * <p>
68       * The given class, if non-<code>null</code>, must belong to either the
69       * system domain or a protection domain whose associated class loader is
70       * non-<code>null</code>.  If the class does not belong to such a
71       * protection domain, then no permissions are granted and an
72       * <code>UnsupportedOperationException</code> is thrown.
73       * <p>
74       * If a security manager is installed, its <code>checkPermission</code>
75       * method is called with a <code>GrantPermission</code> containing the
76       * permissions to grant; if the permission check fails, then no permissions
77       * are granted and the resulting <code>SecurityException</code> is thrown.
78       * The principals and permissions arrays passed in are neither modified nor
79       * retained; subsequent changes to the arrays have no effect on the grant
80       * operation.
81       *
82       * @param   cl class to grant permissions to the class loader of, or
83       * 		<code>null</code> if granting across all class loaders
84       * @param	principals if non-<code>null</code>, minimum set of principals
85       * 		to which grants apply
86       * @param	permissions if non-<code>null</code>, permissions to grant
87       * @throws	UnsupportedOperationException if policy does not support
88       *          dynamic grants, or if <code>cl</code> is non-<code>null</code>
89       *          and belongs to a protection domain with a <code>null</code>
90       *          class loader other than the system domain
91       * @throws	SecurityException if a security manager is installed and the
92       *          calling context does not have sufficient permissions to grant
93       *          the given permissions
94       * @throws	NullPointerException if any element of the principals or
95       *          permissions arrays is <code>null</code>
96       */
97      void grant(Class cl, Principal[] principals, Permission[] permissions);
98  
99      /**
100      * If this security policy provider supports dynamic permission grants,
101      * returns a new array containing the cumulative set of permissions
102      * dynamically granted to protection domains (including ones not yet
103      * created) that are associated with the class loader of the given class
104      * and possess at least the given set of principals.  If the given class is
105      * <code>null</code>, then this method returns the cumulative set of
106      * permissions dynamically granted across all protection domains that
107      * possess at least the specified principals (i.e., through calls to the
108      * grant method where the specified class was <code>null</code>).  If the
109      * list of principals is <code>null</code> or empty, then the permissions
110      * returned reflect only grants not qualified by principals (i.e., those
111      * performed through calls to the grant method where the specified
112      * principals array was <code>null</code> or empty).  If this policy
113      * provider does not support dynamic permission grants, then an
114      * <code>UnsupportedOperationException</code> is thrown.
115      * <p>
116      * The given class, if non-<code>null</code>, must belong to either the
117      * system domain or a protection domain whose associated class loader is
118      * non-<code>null</code>.  If the class does not belong to such a
119      * protection domain, then an <code>UnsupportedOperationException</code> is
120      * thrown.
121      *
122      * @param   cl class to query the permissions dynamically granted to the
123      *          class loader of, or <code>null</code> if querying permissions
124      *          granted across all class loaders
125      * @param   principals if non-<code>null</code>, principals to query
126      * 		dynamic grants for
127      * @return  new array containing the permissions dynamically granted to the
128      *          indicated class loader (if any) and principals
129      * @throws	UnsupportedOperationException if policy does not support
130      *          dynamic grants, or if <code>cl</code> is non-<code>null</code>
131      *          and belongs to a protection domain with a <code>null</code>
132      *          class loader other than the system domain
133      * @throws	NullPointerException if any element of the principals array is
134      *          <code>null</code>
135      */
136     Permission[] getGrants(Class cl, Principal[] principals);
137 }
