39  /**
40   * Security policy provider that wraps the J2SE(TM) default
41   * <a href="http://java.sun.com/j2se/1.4/docs/guide/security/PolicyFiles.html">
42   * "PolicyFile" security policy provider</a> distributed as part of the
43   * Java(TM) 2 Platform, Standard Edition.  This provider augments the J2SE
44   * default policy provider in two ways: it provides an additional constructor
45   * for creating a policy based on an explicitly named policy file, and supports
46   * the use of {@link UmbrellaGrantPermission}s as shorthand notation for
47   * {@link GrantPermission}s covering all permissions authorized to given
48   * protection domains.
49   *
50   * @author Sun Microsystems, Inc.
51   * @since 2.0
52   *
53   *  <!-- Implementation Specifics -->
54   *
55   * This implementation's no-argument constructor uses a default class name of
56   * <code>"org.apache.river.impl.security.policy.se.ConcurrentPolicyFile"</code> 
57   * to instantiate base policy objects, if the
58   * <code>net.jini.security.policy.PolicyFileProvider.basePolicyClass</code>
59   * security property is not set.
60   */
61  public class PolicyFileProvider extends AbstractPolicy {
62  
63      private static final String basePolicyClassProperty =
64  	"net.jini.security.policy.PolicyFileProvider.basePolicyClass";
65      private static final String defaultBasePolicyClass =
66          // Having our own implementation removes a platform dependency
67         "org.apache.river.api.security.ConcurrentPolicyFile";
68  //	"sun.security.provider.PolicyFile";
69      private static final String policyProperty = "java.security.policy";
70      private static final Object propertyLock = new Object();
71  
72      private final String policyFile;
73      private final Policy basePolicy;

75      /**
76       * Creates a <code>PolicyFileProvider</code> whose starting set of
77       * permission mappings is the same as those that would result from
78       * constructing a new instance of the J2SE default security policy provider
79       * with the current <code>java.security.policy</code> system property
80       * setting (if any), except that <code>UmbrellaGrantPermission</code>s are
81       * expanded into <code>GrantPermission</code>s as described in the
82       * documentation for {@link UmbrellaGrantPermission}.
83       * <p>
84       * The constructed <code>PolicyFileProvider</code> contains an instance of
85       * the J2SE default security policy provider, which is created as follows:
86       * if the
87       * <code>net.jini.security.policy.PolicyFileProvider.basePolicyClass</code>
88       * security property is set, then its value is interpreted as the class
89       * name of the base (underlying) J2SE default policy provider; otherwise,
90       * an implementation-specific default class name is used.  The base policy
91       * is then instantiated using the no-arg public constructor of the named
92       * class.  If the base policy class is not found or is not instantiable via
93       * a public no-arg constructor, or if invocation of its constructor fails,
94       * then a <code>PolicyInitializationException</code> is thrown.
95       * <p>
96       * Note that this constructor requires the appropriate
97       * <code>"getProperty"</code> {@link java.security.SecurityPermission} to
98       * read the
99       * <code>net.jini.security.policy.PolicyFileProvider.basePolicyClass</code>
100      * security property, and may require <code>"accessClassInPackage.*"</code>
101      * {@link RuntimePermission}s, depending on the package of the base policy
102      * class.
103      *
104      * @throws  PolicyInitializationException if unable to construct the base
105      *          policy
106      * @throws  SecurityException if there is a security manager and the
107      *          calling context does not have adequate permissions to read the
108      *          <code>net.jini.security.policy.PolicyFileProvider.basePolicyClass</code>
109      *          security property, or if the calling context does not have
110      *          adequate permissions to access the base policy class
111      * @deprecated DynamicPolicyProvider now supports Umbrella grants directly.
112      */
113     @Deprecated
114     public PolicyFileProvider() throws PolicyInitializationException {
115 	policyFile = null;
116 
117 	String cname = Security.getProperty(basePolicyClassProperty);
118 	if (cname == null) {
119 	    cname = defaultBasePolicyClass;
120 	}
121 	try {
122 	    basePolicy = (Policy) Class.forName(cname).newInstance();
123 	} catch (SecurityException e) {
124 	    throw e;
125 	} catch (Exception e) {
126 	    throw new PolicyInitializationException(
127 		"unable to construct base policy", e);
128 	}
129 	ensureDependenciesResolved();
130     }

132     /**
133      * Creates a <code>PolicyFileProvider</code> whose starting set of
134      * permission mappings is the same as those that would result from
135      * constructing a new instance of the J2SE default security policy provider
136      * with the <code>java.security.policy</code> system property set to the
137      * value of <code>policyFile</code>, except that
138      * <code>UmbrellaGrantPermission</code>s are expanded into
139      * <code>GrantPermission</code>s as described in the documentation for
140      * {@link UmbrellaGrantPermission}.
141      * <p>
142      * The constructed <code>PolicyFileProvider</code> contains an instance of
143      * the J2SE default security policy provider, which is created as described
144      * in the documentation for {@link #PolicyFileProvider()}.  Before
145      * instantiating the base (underlying) J2SE default policy provider, this
146      * constructor sets the <code>java.security.policy</code> system property
147      * to the value of <code>policyFile</code>; after instantiation of the base
148      * policy instance has completed (normally or otherwise), the
149      * <code>java.security.policy</code> system property is reset to its prior
150      * value.  Internal synchronization ensures that concurrent calls to this
151      * constructor and/or the {@link #refresh} method of this class (which may
152      * also modify <code>java.security.policy</code>) will not interfere with
153      * the <code>java.security.policy</code> values set and restored by each.
154      * No synchronization is done with any other accesses or modifications to
155      * <code>java.security.policy</code>.
156      * <p>
157      * Note that this constructor requires {@link java.util.PropertyPermission}
158      * to read and write the <code>java.security.policy</code> system property,
159      * {@link java.security.SecurityPermission} to read the
160      * <code>net.jini.security.policy.PolicyFileProvider.basePolicyClass</code>
161      * security property, and may require <code>"accessClassInPackage.*"</code>
162      * {@link RuntimePermission}s, depending on the package of the base policy
163      * class.
164      *
165      * @param	policyFile URL string specifying location of the policy file to
166      * 		use
167      * @throws	NullPointerException if <code>policyFile</code> is
168      * 		<code>null</code>
169      * @throws  PolicyInitializationException if unable to construct the base
170      *          policy
171      * @throws  SecurityException if there is a security manager and the
172      *          calling context does not have adequate permissions to read and
173      *          write the <code>java.security.policy</code> system property, to
174      *          read the
175      *          <code>net.jini.security.policy.PolicyFileProvider.basePolicyClass</code>
176      *          security property, or to access the base policy class
177      */
178     public PolicyFileProvider(String policyFile) 
179 	throws PolicyInitializationException
180     {
181 	if (policyFile == null) {
182 	    throw new NullPointerException();
183 	}
184 	this.policyFile = policyFile;
185 
186 	String cname = Security.getProperty(basePolicyClassProperty);
187 	if (cname == null) {
188 	    cname = defaultBasePolicyClass;
189 	}
190 	try {
191 	    Class cl = Class.forName(cname);
192 	    synchronized (propertyLock) {
193 		String oldp = System.getProperty(policyProperty);
194                 System.setProperty(policyProperty, policyFile);
195 		try {
196 		    basePolicy = (Policy) cl.newInstance();
197 		} finally {
198 		    resetPolicyProperty(oldp);
199 		}
200 	    }
201 	} catch (SecurityException e) {
202 	    throw e;
203 	} catch (Exception e) {
204 	    throw new PolicyInitializationException(
205 		"unable to construct base policy", e);
206 	}
207 	ensureDependenciesResolved();
208     }

212      */
213     public PermissionCollection getPermissions(CodeSource source) {
214 	PermissionCollection pc = basePolicy.getPermissions(source);
215 	expandUmbrella(pc);
216 	return pc;
217     }

229     /**
230      * Behaves as specified by {@link Policy#implies}.
231      */
232     public boolean implies(ProtectionDomain domain, Permission permission) {
233 	// REMIND: cache expanded permission collections?
234 	return (basePolicy.implies(domain, permission) ||
235 		(permission instanceof GrantPermission &&
236 		 getPermissions(domain).implies(permission)));
237     }

239     /**
240      * Refreshes the policy configuration by calling <code>refresh</code> on
241      * the base policy.  If this <code>PolicyFileProvider</code> instance was
242      * constructed with an explicit policy file value, then the
243      * <code>java.security.policy</code> system property is set to that value
244      * prior to invoking <code>refresh</code> on the base policy; once the base
245      * policy <code>refresh</code> call has completed, the
246      * <code>java.security.policy</code> system property is reset to its prior
247      * value.  Internal synchronization ensures that concurrent invocations of
248      * this method and/or the {@link #PolicyFileProvider(String)} constructor
249      * (which also modifies <code>java.security.policy</code>) will not
250      * interfere with the <code>java.security.policy</code> values set and
251      * restored by each. No synchronization is done with any other accesses
252      * or modifications to <code>java.security.policy</code>.
253      */
254     public void refresh() {
255 	if (policyFile != null) {
256 	    synchronized (propertyLock) {
257 		String oldp = System.getProperty(policyProperty);
258                 System.setProperty(policyProperty, policyFile);
259 		try {
260 		    basePolicy.refresh();
261 		} finally {
262 		    resetPolicyProperty(oldp);
263 		}
264 	    }
265 	} else {
266 	    basePolicy.refresh();
267 	}
268     }

270     /**
271      * Ensures that any classes depended on by this policy provider are
272      * resolved.  This is to preclude lazy resolution of such classes during
273      * operation of the provider, which can result in deadlock as described by
274      * bug 4911907.
275      */
276     private void ensureDependenciesResolved() {
277 	// force resolution of GrantPermission and UmbrellaGrantPermission
278 	new GrantPermission(umbrella);
279     }
280     
281     /** Resets policyProperty system property, removing it if the value to set
282      * is null. We do this in a privileged block to make sure that the operation
283      * does not fail, even if the calling context may not have the requisite
284      * permissions anymore (could happen in the refresh case).
285      */
286     private static void resetPolicyProperty(final String value) {
287 	AccessController.doPrivileged(new PrivilegedAction() {
288 	    public Object run() {
289 		if (value == null) {
290 		    // TODO: Use System.clearProperty when we move to 1.5
291                     System.clearProperty(policyProperty);
292 		    //System.getProperties().remove(policyProperty);
293 		} else {
294 		    System.setProperty(policyProperty, value);
295 		}
296 		return null;
297 	   } 
298 	});
299     }
