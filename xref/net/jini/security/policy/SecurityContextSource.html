19  package net.jini.security.policy;
20  
21  import net.jini.security.SecurityContext;
22  
23  /**
24   * Interface that a security manager or policy provider can optionally
25   * implement in order to support the saving and restoring of custom security
26   * context state.  If the installed security manager or policy provider
27   * implements this interface, then its <code>getContext</code> method is
28   * delegated to by the corresponding method of the
29   * {@link net.jini.security.Security} class, with precedence given to the
30   * security manager.
31   * <p>
32   * This interface is intended to be implemented by security managers and policy
33   * providers whose security contexts include state in addition to that provided
34   * by {@link java.security.AccessControlContext}.  For example, a security
35   * policy provider that considers thread local values when evaluating security
36   * checks may want to include those values in snapshots of the current security
37   * context, so that they can be properly restored when the context is
38   * reapplied--it can achieve this by implementing the <code>getContext</code>
39   * method to return a {@link SecurityContext} instance containing the
40   * snapshotted thread local values as well as the current access control
41   * context, with <code>wrap</code> methods implemented to return privileged
42   * action wrappers that properly restore the thread local state.
43   *
44   * @author Sun Microsystems, Inc.
45   * 
46   * @since 2.0
47   */
48  public interface SecurityContextSource {
49  
50      /**
51       * Returns a snapshot of the current security context, which can be used to
52       * restore the context at a later time.
53       *
54       * @return snapshot of the current security context
55       */
56      SecurityContext getContext();
57  }
