31  /**
32   * A <code>ProxyPreparer</code> for verifying that proxies are trusted,
33   * dynamically granting permissions to trusted proxies, and optionally
34   * setting the client constraints on trusted proxies.
35   *
36   * Determining permissions granted to verified and trusted proxies may be 
37   * either passed explicitly via a constructor or if null may be advised by the
38   * proxy using {@link AdvisoryDynamicPermissions}
39   *
40   * @author Sun Microsystems, Inc.
41   * @since 2.1
42   */
43  public final class VerifyingProxyPreparer implements ProxyPreparer {
44  
45      /** Set constraints on proxy from context. */
46      private static final int SET_CONSTRAINTS = 1;
47      /** No change to proxy or context. */
48      private static final int AS_IS = 2;
49      /** Add proxy's constraints to context. */
50      private static final int ADD_CONSTRAINTS = 3;
51  
52      /** SET_CONSTRAINTS, AS_IS, or ADD_CONSTRAINTS. */
53      private final int type;
54      /** Class loader to pass to verifyObjectTrust. */
55      private final ClassLoader loader;
56      /** Trust verifier context elements. */
57      private final Object[] contextElements;
58      /** Principals to scope the permission grant, if any. */
59      private final Principal[] principals;
60      /** Permissions to dynamically grant. */
61      private final Permission[] permissions;
62  
63      /**
64       * Creates a proxy preparer that verifies proxies using the context
65       * class loader and specified trust verifier context elements, dynamically
66       * grants the specified permissions to trusted proxies for the principals
67       * of the preparing thread's subject, and returns trusted proxies with
68       * their client constraints set to the constraints specified as a trust
69       * verifier context element. The arrays passed to this constructor are
70       * neither modified nor retained; subsequent changes to the arrays have no
71       * effect on the instance created.
72       *
73       * @param contextElements the trust verifier context elements
74       * @param permissions the permissions to dynamically grant, or
75       * <code>null</code> if no permissions should be granted
76       * @throws NullPointerException if <code>contextElements</code> is
77       * <code>null</code> or any element of <code>permissions</code> is
78       * <code>null</code>
79       * @throws IllegalArgumentException if no element of
80       * <code>contextElements</code> is an instance of {@link MethodConstraints}
81       */
82      public VerifyingProxyPreparer(Object[] contextElements,
83  				  Permission[] permissions)
84      {
85  	this(null, contextElements, null, permissions);
86      }

88      /**
89       * Creates a proxy preparer that verifies proxies using the specified
90       * class loader and trust verifier context elements, dynamically grants
91       * the specified permissions to trusted proxies for the specified
92       * principals, and returns trusted proxies with their client constraints
93       * set to the constraints specified as a trust verifier context element.
94       * The arrays passed to this constructor are neither modified nor
95       * retained; subsequent changes to the arrays have no effect on the
96       * instance created.
97       *
98       * @param loader the class loader for finding trust verifiers, or
99       * <code>null</code> to use the context class loader
100      * @param contextElements the trust verifier context elements
101      * @param principals minimum set of principals to which grants apply, or
102      * <code>null</code> to use the principals of the preparing thread's
103      * subject
104      * @param permissions the permissions to dynamically grant, or
105      * <code>null</code> if no permissions should be granted
106      * @throws NullPointerException if <code>contextElements</code> is
107      * <code>null</code> or any element of <code>principals</code> or
108      * <code>permissions</code> is <code>null</code>
109      * @throws IllegalArgumentException if no element of
110      * <code>contextElements</code> is an instance of {@link MethodConstraints}
111      */
112     public VerifyingProxyPreparer(ClassLoader loader,
113 				  Object[] contextElements,
114 				  Principal[] principals,
115 				  Permission[] permissions)
116     {
117 	type = SET_CONSTRAINTS;
118 	this.loader = loader;
119 	this.contextElements = (Object[]) contextElements.clone();
120 	this.principals = checkPrincipals(principals);
121 	this.permissions = checkPermissions(permissions);
122 	for (int i = this.contextElements.length; --i >= 0; ) {
123 	    if (this.contextElements[i] instanceof MethodConstraints) {
124 		return;
125 	    }
126 	}
127 	throw new IllegalArgumentException("no MethodConstraints in context");
128     }

130     /**
131      * Creates a proxy preparer that verifies proxies using the specified
132      * class loader and trust verifier context elements (optionally with
133      * the proxy's client constraints as an additional context element),
134      * dynamically grants the specified permissions to trusted proxies for the
135      * specified principals, and returns trusted proxies with their original
136      * client constraints intact. The arrays passed to this constructor are
137      * neither modified nor retained; subsequent changes to the arrays have no
138      * effect on the instance created.
139      *
140      * @param addProxyConstraints <code>true</code> if the proxy's client
141      * constraints should be included as a trust verifier context element,
142      * <code>false</code> otherwise
143      * @param loader the class loader for finding trust verifiers, or
144      * <code>null</code> to use the context class loader
145      * @param contextElements the trust verifier context elements, or
146      * <code>null</code> if no elements need to be supplied
147      * @param principals minimum set of principals to which grants apply, or
148      * <code>null</code> to use the principals of the preparing thread's
149      * subject
150      * @param permissions the permissions to dynamically grant, or
151      * <code>null</code> if no permissions should be granted
152      * @throws NullPointerException if any element of <code>principals</code>
153      * or <code>permissions</code> is <code>null</code>
154      */
155     public VerifyingProxyPreparer(boolean addProxyConstraints,
156 				  ClassLoader loader,
157 				  Object[] contextElements,
158 				  Principal[] principals,
159 				  Permission[] permissions)
160     {
161 	type = addProxyConstraints ? ADD_CONSTRAINTS : AS_IS;
162 	this.loader = loader;
163 	this.contextElements = contextElements == null ?
164 	    new Object[0] : (Object[]) contextElements.clone();
165 	this.principals = checkPrincipals(principals);
166 	this.permissions = checkPermissions(permissions);
167     }
168 
169     /** Clones the argument, checks for null elements, returns non-null. */
170     private static Permission[] checkPermissions(Permission[] permissions) {
171 	if (permissions == null) return AdvisoryDynamicPermissions.DEFAULT_PERMISSIONS;
172 	permissions = (Permission[]) permissions.clone();
173 	for (int i = permissions.length; --i >= 0; ) {
174 	    if (permissions[i] == null) {
175 		throw new NullPointerException("permission cannot be null");
176 	    }
177 	}
178 	return permissions;
179     }
180 
181     /** Clones the argument, checks for null elements. */
182     private static Principal[] checkPrincipals(Principal[] principals) {
183 	if (principals == null) {
184 	    return null;
185 	}
186 	principals = (Principal[]) principals.clone();
187 	for (int i = principals.length; --i >= 0; ) {
188 	    if (principals[i] == null) {
189 		throw new NullPointerException("principal cannot be null");
190 	    }
191 	}
192 	return principals;
193     }

195     /**
196      * Performs operations on a proxy to prepare it for use, returning the
197      * prepared proxy, which may or may not be the argument itself.
198      * <p>
199      * If this preparer was created using the two-argument or four-argument
200      * constructor, or using the five-argument constructor
201      * with <code>addProxyConstraints</code> set to <code>true</code>, and if
202      * the specified proxy is not an instance of {@link RemoteMethodControl},
203      * then a <code>SecurityException</code> is thrown. Otherwise,
204      * {@link Security#verifyObjectTrust Security.verifyObjectTrust} is
205      * invoked with the specified proxy, the class loader that was passed to
206      * the constructor of this preparer (or <code>null</code> if the
207      * two-argument constructor was used), and a trust verifier context
208      * collection containing all of the context elements that were passed to
209      * the constructor of this preparer. If this preparer was created using
210      * the five-arguent constructor with <code>addProxyConstraints</code>
211      * set to <code>true</code>, then the proxy's client constraints (obtained
212      * by calling {@link RemoteMethodControl#getConstraints getConstraints} on
213      * the proxy) are included as an additional context element. Any exception
214      * thrown by <code>verifyObjectTrust</code> is thrown by this method. If
215      * this preparer was created with a non-<code>null</code> array of
216      * principals and one or more permissions, then
217      * {@link Security#grant(Class,Principal[],Permission[]) Security.grant}
218      * is invoked with the proxy's class and those principals and permissions.
219      * If this preparer was created with no array of principals (either
220      * <code>null</code> was specified or the two-argument constructor was
221      * used) but one or more permissions, then
222      * {@link Security#grant(Class,Permission[]) Security.grant} is invoked
223      * with the proxy's class and those permissions. In either case, if
224      * <code>grant</code>
225      * throws an {@link UnsupportedOperationException}, this method throws
226      * a <code>SecurityException</code>. Finally, if this preparer was
227      * created using the five-argument constructor, then the original proxy
228      * is returned, otherwise what is returned is the result of calling
229      * {@link RemoteMethodControl#setConstraints
230      * RemoteMethodControl.setConstraints} on the proxy, passing the first
231      * trust verifier context element that is an instance of
232      * {@link MethodConstraints}.
233      *
234      * @param proxy the proxy to prepare
235      * @return the prepared proxy
236      * @throws NullPointerException if <code>proxy</code> is <code>null</code>
237      * @throws RemoteException if a communication-related exception occurs
238      * @throws SecurityException if a security exception occurs
239      */
240     public Object prepareProxy(Object proxy) throws RemoteException {
241 	if (proxy == null) {
242 	    throw new NullPointerException("proxy cannot be null");
243 	} else if (type != AS_IS && !(proxy instanceof RemoteMethodControl)) {
244 	    throw new SecurityException(
245 				  "proxy must implement RemoteMethodControl");
246 	}
247 	Object[] elts = contextElements;
248 	if (type == ADD_CONSTRAINTS) {
249 	    elts = new Object[contextElements.length + 1];
250 	    elts[0] = ((RemoteMethodControl) proxy).getConstraints();
251 	    System.arraycopy(contextElements, 0, elts, 1,
252 			     contextElements.length);
253 	}
254 	Security.verifyObjectTrust(proxy, loader,
255 				   Collections.unmodifiableCollection(
256 							Arrays.asList(elts)));
257 	if (permissions.length > 0) {
258 	    try {
259 		if (principals == null) {
260 		    Security.grant(proxy.getClass(), permissions);
261 		} else {
262 		    Security.grant(proxy.getClass(), principals, permissions);
263 		}
264 	    } catch (UnsupportedOperationException e) {
265 		SecurityException se = new SecurityException(
266 			       "dynamic permission grants are not supported");
267 		se.initCause(e);
268 		throw se;
269 	    }
270 	} else {
271 	    Permission [] perms;
272 	    Class klass = proxy.getClass();
273 	    ClassLoader ldr = klass.getClassLoader();
274 	    if (ldr instanceof AdvisoryDynamicPermissions) {
275 		perms = ((AdvisoryDynamicPermissions) ldr).getPermissions();
276 		if (perms.length > 0){
277 		    try {
278 			if (principals == null) {
279 			    Security.grant(klass, perms);
280 			} else {
281 			    Security.grant(klass, principals, perms);
282 	}
283 		    } catch (UnsupportedOperationException e) {
284 			Logger.getLogger("net.jini.security")
285 			    .config("Local configuration doesn't allow advisory dynamic permission grants, consider using a DynamicPolicy provider");
286 		    }
287 		}
288 	    }
289 	}
290 	if (type == SET_CONSTRAINTS) {
291 	    for (int i = 0; i < contextElements.length; i++) {
292 		Object elt = contextElements[i];
293 		if (elt instanceof MethodConstraints) {
294 		    return ((RemoteMethodControl) proxy).setConstraints(
295 						     (MethodConstraints) elt);
296 		}
297 	    }
298 	}
299 	return proxy;
300     }

302     /** Returns a string representation of this object. */
303     public String toString() {
304 	StringBuffer sb = new StringBuffer("VerifyingProxyPreparer[");
305 	if (type != SET_CONSTRAINTS) {
306 	    sb.append(type == AS_IS ? "false, " : "true, ");
307 	}
308 	sb.append(loader);
309 	sb.append(", {");
310 	for (int i = 0; i < contextElements.length; i++) {
311 	    if (i > 0) {
312 		sb.append(", ");
313 	    }
314 	    sb.append(contextElements[i]);
315 	}
316 	if (principals == null) {
317 	    sb.append("}, null, {");
318 	} else {
319 	    sb.append("}, {");
320 	    for (int i = 0; i < principals.length; i++) {
321 		if (i > 0) {
322 		    sb.append(", ");
323 		}
324 		sb.append(principals[i]);
325 	    }
326 	    sb.append("}, {");
327 	}
328 	for (int i = 0; i < permissions.length; i++) {
329 	    if (i > 0) {
330 		sb.append(", ");
331 	    }
332 	    sb.append(permissions[i]);
333 	}
334 	sb.append("}]");
335 	return sb.toString();
336     }

338     /**
339      * Returns <code>true</code> if the specified object and this object
340      * are both instances of this class that were constructed with equivalent
341      * arguments. The order of trust verifier context elements, principals,
342      * and permissions in the arrays that were passed to the constructor is
343      * not significant.
344      */
345     public boolean equals(Object obj) {
346 	if (this == obj) {
347 	    return true;
348 	} else if (!(obj instanceof VerifyingProxyPreparer)) {
349 	    return false;
350 	}
351 	VerifyingProxyPreparer other = (VerifyingProxyPreparer) obj;
352 	if (type != other.type ||
353 	    loader != other.loader ||
354 	    contextElements.length != other.contextElements.length ||
355 	    (principals == null) != (other.principals == null) ||
356 	    (principals != null &&
357 	     principals.length != other.principals.length) ||
358 	    permissions.length != other.permissions.length)
359 	{
360 	    return false;
361 	}
362 	Object[] otherElts = (Object[]) other.contextElements.clone();
363     context:
364 	for (int i = contextElements.length; --i >= 0; ) {
365 	    Object elt = contextElements[i];
366 	    for (int j = i; j >= 0; j--) {
367 		if (elt.equals(otherElts[j])) {
368 		    otherElts[j] = otherElts[i];
369 		    continue context;
370 		}
371 	    }
372 	    return false;
373 	}
374 	if (principals != null) {
375 	    Principal[] otherPrins = (Principal[]) other.principals.clone();
376 	prins:
377 	    for (int i = principals.length; --i >= 0; ) {
378 		Principal p = principals[i];
379 		for (int j = i; j >= 0; j--) {
380 		    if (p.equals(otherPrins[j])) {
381 			otherPrins[j] = otherPrins[i];
382 			continue prins;
383 		    }
384 		}
385 		return false;
386 	    }
387 	}
388 	Permission[] otherPerms = (Permission[]) other.permissions.clone();
389     perms:
390 	for (int i = permissions.length; --i >= 0; ) {
391 	    Permission p = permissions[i];
392 	    for (int j = i; j >= 0; j--) {
393 		if (p.equals(otherPerms[j])) {
394 		    otherPerms[j] = otherPerms[i];
395 		    continue perms;
396 		}
397 	    }
398 	    return false;
399 	}
400 	return true;
401     }

403     /** Returns a hash code value for this object. */
404     public int hashCode() {
405 	int hash = type;
406 	if (loader != null) {
407 	    hash += loader.hashCode();
408 	}
409 	for (int i = contextElements.length; --i >= 0; ) {
410 	    hash += contextElements[i].hashCode();
411 	}
412 	if (principals != null) {
413 	    for (int i = principals.length; --i >= 0; ) {
414 		hash += principals[i].hashCode();
415 	    }
416 	}
417 	for (int i = permissions.length; --i >= 0; ) {
418 	    hash += permissions[i].hashCode();
419 	}
420 	return hash;
421     }
