19  package net.jini.security.proxytrust;
20  
21  /**
22   * Defines an interface for checking that an object (that is not yet known to
23   * be trusted) is equivalent in trust, content, and function to a known
24   * trusted object. This interface can be useful in implementing a
25   * {@link net.jini.security.TrustVerifier} returned by
26   * {@link ProxyTrust#getProxyVerifier ProxyTrust.getProxyVerifier}.
27   * <p>
28   * An implementation of this interface must always compare components of the
29   * two objects by invoking comparison methods (such as <code>equals</code>
30   * or {@link #checkTrustEquivalence checkTrustEquivalence}) on components of
31   * the known trusted object, and those comparison methods must in turn
32   * compare subcomponents in the same fashion, always invoking comparison
33   * methods on components of <code>this</code>.
34   * <p>
35   * For example, in the case of a smart proxy wrapped around an inner proxy,
36   * where the inner proxy results from exporting a remote object, the inner
37   * proxy could be required to be an instance of a class that implements this
38   * interface. The verifier could contain a canonical instance of the inner
39   * proxy, and use its <code>checkTrustEquivalence</code> method to check
40   * that a candidate smart proxy contains the correct inner proxy.
41   *
42   * @author Sun Microsystems, Inc.
43   * @since 2.0
44   */
45  public interface TrustEquivalence {
46      /**
47       * Returns <code>true</code> if the specified object (that is not yet
48       * known to be trusted) is equivalent in trust, content, and function to
49       * this known trusted object, and returns <code>false</code> otherwise.
50       * This method is permitted to delete content of the specified object
51       * if that content is cached state that might be stale (relative to the
52       * known trusted object's cached state), cannot otherwise be checked,
53       * and will be recreated on demand (from content that has been checked)
54       * during subsequent operations on the object.
55       *
56       * @param obj object to check that is not yet known to be trusted
57       * @return <code>true</code> if the specified object (that is not yet
58       * known to be trusted) is equivalent in trust, content, and function to
59       * this known trusted object, and returns <code>false</code> otherwise
60       */
61      boolean checkTrustEquivalence(Object obj);
62  }
