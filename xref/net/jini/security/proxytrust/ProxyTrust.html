19  package net.jini.security.proxytrust;
20  
21  import java.rmi.Remote;
22  import java.rmi.RemoteException;
23  import net.jini.security.TrustVerifier;
24  
25  /**
26   * Defines a remote interface for obtaining a proxy trust verifier. A service
27   * that uses proxies that will not directly be considered trusted by clients
28   * can implement its remote object, or a bootstrap remote object, to support
29   * this interface to allow clients to verify that the proxies can be trusted
30   * as proxies for the service. The client typically configures the
31   * {@link ProxyTrustVerifier} trust verifier for use with
32   * {@link net.jini.security.Security#verifyObjectTrust
33   * Security.verifyObjectTrust}; given a service proxy,
34   * <code>ProxyTrustVerifier</code> obtains from it a bootstrap proxy (which
35   * must be an instance of both <code>ProxyTrust</code> and
36   * {@link net.jini.core.constraint.RemoteMethodControl}), and after verifying
37   * that the bootstrap proxy (or a derivative) is trusted by the client, calls
38   * the <code>getProxyVerifier</code> method of the bootstrap proxy (or
39   * derivative) to obtain a verifier, and then uses that
40   * verifier to determine if the original service proxy can be trusted. Other
41   * trust verifiers may also make use of <code>ProxyTrust</code>.
42   *
43   * @author Sun Microsystems, Inc.
44   * 
45   * @since 2.0
46   */
47  @Deprecated
48  public interface ProxyTrust extends Remote {
49  
50      /**
51       * Returns a <code>TrustVerifier</code> which can be used to verify that
52       * a proxy can be trusted as a proxy for the service; that is, the
53       * {@link TrustVerifier#isTrustedObject isTrustedObject} method of the
54       * returned verifier can be called with a candidate proxy. The verifier
55       * should be able to verify all proxies for the service, including
56       * proxies for resources (such as leases and registrations).
57       *
58       * @return a <code>TrustVerifier</code> which can be used to verify that
59       * a proxy can be trusted as a proxy for the service
60       * @throws RemoteException if a communication-related exception occurs
61       */
62      TrustVerifier getProxyVerifier() throws RemoteException;
63  }
