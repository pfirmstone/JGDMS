34  /**
35   * Invocation handler for remote objects, supporting proxy trust verification
36   * by clients using {@link ProxyTrustVerifier}. This invocation handler
37   * contains both an underlying main proxy and a bootstrap proxy; the main
38   * proxy is not expected to be considered trusted directly by clients, but
39   * the bootstrap proxy is. The main proxy must be an instance of both
40   * {@link RemoteMethodControl} and {@link TrustEquivalence}, and the bootstrap
41   * proxy must be an instance of {@link ProxyTrust},
42   * <code>RemoteMethodControl</code>, and <code>TrustEquivalence</code>. This
43   * invocation handler handles most method invocations by delegating to the
44   * main proxy. The bootstrap proxy is produced by the iterator returned by
45   * the {@link #getProxyTrustIterator getProxyTrustIterator} method, as
46   * required by <code>ProxyTrustVerifier</code>.
47   *
48   * @author Sun Microsystems, Inc.
49   * @since 2.0
50   */
51  @Deprecated
52  @AtomicSerial
53  public final class ProxyTrustInvocationHandler
54  		 implements InvocationHandler, TrustEquivalence, Serializable
55  {
56      private static final long serialVersionUID = -3270029468290295063L;
57      private static final Class[] consArgs =
58  					new Class[]{InvocationHandler.class};

60      /**
61       * The main proxy.
62       *
63       * @serial
64       */
65      private final RemoteMethodControl main;
66      /**
67       * The bootstrap proxy.
68       *
69       * @serial
70       */
71      private final ProxyTrust boot;

73      /**
74       * Creates an instance with the specified main proxy and bootstrap proxy.
75       *
76       * @param main the main proxy
77       * @param boot the bootstrap proxy
78       * @throws NullPointerException if any argument is <code>null</code>
79       * @throws IllegalArgumentException if the main proxy is not an instance
80       * of {@link TrustEquivalence}, or the bootstrap proxy is not an instance
81       * of <code>RemoteMethodControl</code> or <code>TrustEquivalence</code>
82       */
83      public ProxyTrustInvocationHandler(RemoteMethodControl main,
84  				       ProxyTrust boot)
85      {
86  	this(main, boot, check(main, boot));
87      }
88      
89      public ProxyTrustInvocationHandler(GetArg arg) throws InvalidObjectException, IOException
90      {
91  	this(true,
92  	     arg.get("main", null, RemoteMethodControl.class),
93  	     arg.get("boot", null, ProxyTrust.class)
94  	);
95      }
96      
97      private ProxyTrustInvocationHandler(boolean atomicSerial,
98  					RemoteMethodControl main,
99  				        ProxyTrust boot) throws InvalidObjectException
100     {
101 	this(main, boot, verify(main, boot));
102     }
103     
104     private ProxyTrustInvocationHandler(RemoteMethodControl main,
105 					ProxyTrust boot,
106 					boolean check)
107     {
108 	this.main = main;
109 	this.boot = boot;
110     }
111     
112     private static boolean check(RemoteMethodControl main,
113 			 ProxyTrust boot)
114     {
115 	if (main == null || boot == null) {
116 	    throw new NullPointerException("arguments cannot be null");
117 	} else if (!(main instanceof TrustEquivalence)) {
118 	    throw new IllegalArgumentException(
119 			   "main proxy must implement TrustEquivalence");
120 	} else if (!(boot instanceof RemoteMethodControl)) {
121 	    throw new IllegalArgumentException(
122 			 "bootstrap proxy must implement RemoteMethodControl");
123 	} else if (!(boot instanceof TrustEquivalence)) {
124 	    throw new IllegalArgumentException(
125 			   "bootstrap proxy must implement TrustEquivalence");
126 	}
127 	return true;
128     }

130     /**
131      * Executes the specified method with the specified arguments on the
132      * specified proxy, and returns the return value, if any.
133      * <p>
134      * If the specified method is <code>Object.equals</code>, returns
135      * <code>true</code> if the argument (<code>args[0]</code>) is an
136      * instance of a dynamic proxy class (that is, a class generated by
137      * {@link Proxy}) that implements the same interfaces as the specified
138      * proxy and this invocation handler is equal to the invocation handler
139      * of that argument, and returns <code>false</code> otherwise.
140      * <p>
141      * If the specified method is <code>Object.toString</code>, returns
142      * a string representation of the specified proxy object.
143      * <p>
144      * If the specified method is <code>Object.hashCode</code>, returns
145      * a hash code for the specified proxy object.
146      * <p>
147      * If the specified method is {@link RemoteMethodControl#setConstraints
148      * RemoteMethodControl.setConstraints}, returns a new proxy (an instance
149      * of the same class as the specified proxy) containing an instance of
150      * this class with a new main proxy and the same bootstrap proxy from
151      * this handler. The new main proxy is obtained by delegating to the
152      * existing main proxy of this handler (as described below). An exception
153      * is thrown if the specified proxy is not an instance of a dynamic
154      * proxy class containing this invocation handler.
155      * <p>
156      * If the specified method is
157      * {@link TrustEquivalence#checkTrustEquivalence
158      * TrustEquivalence.checkTrustEquivalence}, returns <code>true</code> if
159      * the argument (<code>args[0]</code>) is an instance of a dynamic proxy
160      * class that implements the same interfaces as the specified proxy and
161      * calling the {@link #checkTrustEquivalence checkTrustEquivalence} method
162      * of this invocation handler with the invocation handler of that argument
163      * returns <code>true</code>, and returns <code>false</code> otherwise.
164      * <p>
165      * For all other methods, returns the object obtained by delegating to
166      * the main proxy: the specified method is reflectively invoked on the
167      * main proxy with the specified arguments, unless the method's declaring
168      * class is not public but the main proxy is an instance of that
169      * declaring class and the main proxy's class is public, in which case
170      * the corresponding method of the main proxy's class is reflectively
171      * invoked instead.
172      * <p>
173      * The semantics of this method are unspecified if the arguments could
174      * not have been produced by an instance of some valid dynamic proxy
175      * class containing this invocation handler.
176      *
177      * @param proxy the proxy object
178      * @param method the method being invoked
179      * @param args the arguments to the specified method
180      * @return the value returned by executing the specified method on
181      * the specified proxy with the specified arguments, or <code>null</code>
182      * if the method has <code>void</code> return type
183      * @throws Throwable the exception thrown by executing the specified
184      * method
185      * @throws IllegalArgumentException if the declaring class of the
186      * specified method is not public and either the main proxy is not an
187      * instance of that declaring class or the main proxy's class is not
188      * public
189      */
190     public Object invoke(Object proxy, Method method, Object[] args)
191 	throws Throwable
192     {
193 	Class decl = method.getDeclaringClass();
194 	if (decl == Object.class) {
195 	    String name = method.getName();
196 	    if (name.equals("equals")) {
197 		Object obj = args[0];
198 		if (proxy == obj ||
199 		    (obj != null && sameProxyClass(proxy, obj) &&
200 		     equals(Proxy.getInvocationHandler(obj))))
201 		{
202 		    return Boolean.TRUE;
203 		}
204 		return Boolean.FALSE;
205 	    } else if (name.equals("toString")) {
206 		return proxyToString(proxy);
207 	    } else if (name.equals("hashCode")) {
208 		return Integer.valueOf(hashCode());
209 	    }
210 	    throw new IllegalArgumentException("unexpected Object method");
211 	} else if (decl == RemoteMethodControl.class &&
212 		   method.getName().equals("setConstraints"))
213 	{
214 	    if (Proxy.getInvocationHandler(proxy) != this) {
215 		throw new IllegalArgumentException("wrong invocation handler");
216 	    }
217 	    InvocationHandler handler =
218 		new ProxyTrustInvocationHandler(
219 			   (RemoteMethodControl) invoke0(method, args), boot);
220 	    return Proxy.newProxyInstance(proxy.getClass().getClassLoader(),
221 					  proxy.getClass().getInterfaces(),
222 					  handler);
223 	} else if (decl == TrustEquivalence.class) {
224 	    if (!method.getName().equals("checkTrustEquivalence")) {
225 		throw new AssertionError("unknown TrustEquivalence method");
226 	    }
227 	    Object obj = args[0];
228 	    if (proxy == obj ||
229 		(obj != null && sameProxyClass(proxy, obj) &&
230 		 checkTrustEquivalence(Proxy.getInvocationHandler(obj))))
231 	    {
232 		return Boolean.TRUE;
233 	    }
234 	    return Boolean.FALSE;
235 	}
236 	return invoke0(method, args);
237     }

239     /**
240      * Reflectively invoke the method on the main proxy, unless the method's
241      * declaring class is not public but the main proxy's class is public, in
242      * which case invoke the corresponding method from the main proxy's class
243      * instead.
244      */
245     private Object invoke0(Method m, Object[] args) throws Throwable {
246 	Class iface = m.getDeclaringClass();
247 	if (!Modifier.isPublic(iface.getModifiers())) {
248 	    Class impl = main.getClass();
249 	    if (Modifier.isPublic(impl.getModifiers()) &&
250 		iface.isInstance(main))
251 	    {
252 		try {
253 		    m = impl.getMethod(m.getName(), m.getParameterTypes());
254 		} catch (Exception e) {
255 		}
256 	    }
257 	}
258 	try {
259 	    return m.invoke(main, args);
260 	} catch (InvocationTargetException e) {
261 	    throw e.getTargetException();
262 	} catch (IllegalAccessException e) {
263 	    throw new IllegalArgumentException().initCause(e);
264 	}
265     }

267     /**
268      * Returns <code>true</code> if the argument is an instance of this
269      * class, and calling the <code>checkTrustEquivalence</code> method on
270      * the main proxy of this invocation handler, passing the main proxy of
271      * the argument, returns <code>true</code>, and calling the
272      * <code>checkTrustEquivalence</code> method on the bootstrap proxy of
273      * this invocation handler, passing the bootstrap proxy of the argument,
274      * returns <code>true</code>, and returns <code>false</code> otherwise.
275      */
276     public boolean checkTrustEquivalence(Object obj) {
277 	if (this == obj) {
278 	    return true;
279 	} else if (!(obj instanceof ProxyTrustInvocationHandler)) {
280 	    return false;
281 	}
282 	ProxyTrustInvocationHandler oh = (ProxyTrustInvocationHandler) obj;
283 	return (((TrustEquivalence) main).checkTrustEquivalence(oh.main) &&
284 		((TrustEquivalence) boot).checkTrustEquivalence(oh.boot));
285     }

287     /**
288      * Returns <code>true</code> if the argument is an instance of this
289      * class with the same main proxy and the same bootstrap proxy, and
290      * <code>false</code> otherwise.
291      */
292     public boolean equals(Object obj) {
293 	if (this == obj) {
294 	    return true;
295 	} else if (!(obj instanceof ProxyTrustInvocationHandler)) {
296 	    return false;
297 	}
298 	ProxyTrustInvocationHandler oh = (ProxyTrustInvocationHandler) obj;
299 	return main.equals(oh.main) && boot.equals(oh.boot);
300     }

317     /**
318      * Returns a string representation for a proxy that uses this invocation
319      * handler.
320      */
321     private String proxyToString(Object proxy) {
322 	Class[] interfaces = proxy.getClass().getInterfaces();
323 	Class iface = null;
324 	for (int i = interfaces.length; --i >= 0; iface = interfaces[i]) {
325 	    if (interfaces[i] == RemoteMethodControl.class) {
326 		break;
327 	    }
328 	}
329 	if (iface == null) {
330 	    return "Proxy[" + this + "]";
331 	}
332 	String n = iface.getName();
333 	int dot = n.lastIndexOf('.');
334 	if (dot >= 0) {
335 	    n = n.substring(dot + 1);
336 	}
337 	return "Proxy[" + n + "," + this + "]";
338     }

351     /**
352      * Verifies that the main proxy is an instance of
353      * {@link TrustEquivalence}, and that the bootstrap proxy is an instance
354      * of both {@link RemoteMethodControl} and <code>TrustEquivalence</code>.
355      *
356      * @throws InvalidObjectException if the main proxy is not an instance of
357      * <code>TrustEquivalence</code>, or the bootstrap proxy is not an
358      * instance of both <code>RemoteMethodControl</code> and
359      * <code>TrustEquivalence</code>
360      */
361     private void readObject(ObjectInputStream s)
362 	throws IOException, ClassNotFoundException
363     {
364 	s.defaultReadObject();
365 	verify(main, boot);
366     }
367       
368     private static boolean verify(RemoteMethodControl main,
369 				  ProxyTrust boot) throws InvalidObjectException
370     {
371 	if (!(main instanceof TrustEquivalence)) {
372 	    throw new InvalidObjectException(
373 			   "main proxy must implement TrustEquivalence");
374 	} else if (!(boot instanceof RemoteMethodControl)) {
375 	    throw new InvalidObjectException(
376 			"bootstrap proxy must implement RemoteMethodControl");
377 	} else if (!(boot instanceof TrustEquivalence)) {
378 	    throw new InvalidObjectException(
379 			   "bootstrap proxy must implement TrustEquivalence");
380 	}
381 	return true;
382     }

384     /**
385      * Returns true if proxy2 is a generated Proxy (proxy1 is assumed to
386      * be one) and the classes of both proxies implement the same ordered
387      * list of interfaces, and returns false otherwise.
388      */
389     private static boolean sameProxyClass(Object proxy1, Object proxy2) {
390 	return (proxy1.getClass() == proxy2.getClass() ||
391 		(Proxy.isProxyClass(proxy2.getClass()) &&
392 		 equalInterfaces(proxy1, proxy2)));
393     }
394 
395     /**
396      * Returns true if the interfaces implemented by obj1's class
397      * are the same (and in the same order) as obj2's class.
398      */
399     private static boolean equalInterfaces(Object obj1, Object obj2) {
400 	Class[] intf1 = obj1.getClass().getInterfaces();
401 	Class[] intf2 = obj2.getClass().getInterfaces();
402 	if (intf1.length != intf2.length) {
403 	    return false;
404 	} else {
405 	    for (int i = 0; i < intf1.length; i++) {
406 		if (intf1[i] != intf2[i]) {
407 		    return false;
408 		}
409 	    }
410 	    return true;
411 	}
412     }
