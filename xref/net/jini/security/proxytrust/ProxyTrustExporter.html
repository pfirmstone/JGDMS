41  /**
42   * Contains the information necessary to export a remote object that has a
43   * proxy that will not directly be considered trusted by clients, such that
44   * the proxy can be trusted by clients using {@link ProxyTrustVerifier}. The
45   * remote object to be exported (called the main remote object) must be an
46   * instance of {@link ServerProxyTrust}. In addition to exporting the main
47   * remote object, this exporter also exports a second remote object
48   * (called the bootstrap remote object, which is created internally by this
49   * exporter) that implements the {@link ProxyTrust} interface by delegating
50   * the {@link ProxyTrust#getProxyVerifier getProxyVerifier} method to the
51   * corresponding {@link ServerProxyTrust#getProxyVerifier getProxyVerifier}
52   * method of the main remote object.  Refer to <code>
53   * net.jini.jeri.ProxyTrustILFactory</code>
54   *
55   * @author Sun Microsystems, Inc.
56   * @since 2.0
57   */
58  @Deprecated
59  public class ProxyTrustExporter implements Exporter {
60      /** Permission required to use class loader of main proxy's class */
61      private static final Permission loaderPermission =
62  				      new RuntimePermission("getClassLoader");
63      /** Executor that executes tasks in pooled system threads. */
64      private static final Executor systemThreadPool =
65  	(Executor) Security.doPrivileged(new GetThreadPoolAction(false));
66      /** Holds strong refs to WeakRefs until they are cleared */
67      private static final Set refs = new HashSet();
68      /** Reference queue for WeakRefs */
69      private static final ReferenceQueue queue = new ReferenceQueue();
70      /** WeakRef reaper, if any */
71      private static Reaper reaper = null;
72  
73      /** The main exporter, for the main remote object.*/
74      private final Exporter mainExporter;
75      /** The bootstrap exporter, for the ProxyTrust remote object. */
76      private final Exporter bootExporter;
77      /** The class loader to define the proxy class in, or null */
78      private final ClassLoader loader;
79      /** WeakRef to impl */
80      private WeakRef ref = null;

82      /**
83       * Creates an instance with the specified main exporter (which will be
84       * used to export the main remote object) and the specified bootstrap
85       * exporter (which will be used to export the bootstrap remote object).
86       * The main exporter must produce a proxy (called the main proxy) that is
87       * an instance of both {@link RemoteMethodControl} and
88       * {@link TrustEquivalence}, and if the main proxy's class is not
89       * <code>public</code>, the direct superinterfaces of that class and all
90       * of its superclasses must be <code>public</code>. The bootstrap
91       * exporter, when used to export a remote object that is an instance of
92       * {@link ProxyTrust}, must produce a proxy (called the bootstrap proxy)
93       * that is an instance of <code>ProxyTrust</code>,
94       * <code>RemoteMethodControl</code>, and <code>TrustEquivalence</code>,
95       * and should satisfy the bootstrap proxy requirements of
96       * <code>ProxyTrustVerifier</code>.
97       * The dynamic proxy class generated at export will be defined
98       * by the same class loader as the main proxy's class.
99       *
100      * @param mainExporter the main exporter, for the main remote object
101      * @param bootExporter the bootstrap exporter, for the bootstrap remote
102      * object
103      * @throws NullPointerException if any argument is <code>null</code>
104      */
105     public ProxyTrustExporter(Exporter mainExporter, Exporter bootExporter) {
106 	this(mainExporter, bootExporter, null);
107     }

109     /**
110      * Creates an instance with the specified main exporter (which will be
111      * used to export the main remote object), the specified bootstrap
112      * exporter (which will be used to export the bootstrap remote object),
113      * and the specified class loader (in which the generated dynamic proxy
114      * class will be defined). The main exporter must produce a proxy (called
115      * the main proxy) that is an instance of both {@link RemoteMethodControl}
116      * and {@link TrustEquivalence}, and if the main proxy's class is not
117      * <code>public</code>, the direct superinterfaces of that class and all
118      * of its superclasses must be <code>public</code>. The bootstrap
119      * exporter, when used to export a remote object that is an instance of
120      * {@link ProxyTrust}, must produce a proxy (called the bootstrap proxy)
121      * that is an instance of <code>ProxyTrust</code>,
122      * <code>RemoteMethodControl</code>, and <code>TrustEquivalence</code>,
123      * and should satisfy the bootstrap proxy requirements of
124      * <code>ProxyTrustVerifier</code>.
125      * If the specified class loader is <code>null</code>, the
126      * dynamic proxy class generated at export will be defined by the same
127      * class loader as the main proxy's class.
128      *
129      * @param mainExporter the main exporter, for the main remote object
130      * @param bootExporter the bootstrap exporter, for the bootstrap remote
131      * object
132      * @param loader the class loader to define the proxy class in, or
133      * <code>null</code>
134      * @throws NullPointerException if either exporter argument is
135      * <code>null</code>
136      */
137     public ProxyTrustExporter(Exporter mainExporter,
138 			      Exporter bootExporter,
139 			      ClassLoader loader)
140     {
141 	if (mainExporter == null || bootExporter == null) {
142 	    throw new NullPointerException("exporter is null");
143 	}
144 	this.mainExporter = mainExporter;
145 	this.bootExporter = bootExporter;
146 	this.loader = loader;
147     }

149     /**
150      * Exports the specified main remote object and returns a dynamic proxy
151      * for the object. The main remote object must be an instance of
152      * {@link ServerProxyTrust}. The main remote object is exported using the
153      * main exporter (specified at the construction of this exporter),
154      * returning a proxy called the main proxy. The main proxy must be an
155      * instance of both {@link RemoteMethodControl} and
156      * {@link TrustEquivalence}, and if the main proxy's class is not
157      * <code>public</code>, the direct superinterfaces of that class and all
158      * of its superclasses must be <code>public</code>. A bootstrap remote
159      * object that is an instance of {@link ProxyTrust} is created and
160      * exported using the bootstrap exporter (also specified at the
161      * construction of this exporter), returning a proxy called the bootstrap
162      * proxy. The bootstrap proxy must be an instance of
163      * <code>ProxyTrust</code>, <code>RemoteMethodControl</code>, and
164      * <code>TrustEquivalence</code>. The bootstrap remote object will remain
165      * reachable as long as the main remote object is reachable and the
166      * {@link #unexport unexport} method of this exporter has not returned
167      * <code>true</code>. A {@link Proxy} class is generated that implements
168      * the direct superinterfaces of the main proxy's class and all of its
169      * superclasses, in the following order: the direct superinterfaces of a
170      * class immediately follow the direct superinterfaces of its direct
171      * superclass; the direct superinterfaces of a class are in declaration
172      * order (the order in which they are declared in the class's
173      * <code>implements</code> clause); and if an interface appears more than
174      * once, only the first instance is retained. If a non-<code>null</code>
175      * class loader was specified at the construction of this exporter, the
176      * generated class is defined by that class loader, otherwise it is
177      * defined by the class loader of the main proxy's class. The dynamic
178      * proxy returned by this method is an instance of that generated class,
179      * containing a {@link ProxyTrustInvocationHandler} instance created with
180      * the main proxy and the bootstrap proxy.
181      *
182      * @throws ExportException if the export of either remote object throws
183      * <code>ExportException</code>, or if the export the bootstrap remote
184      * object throws <code>IllegalArgumentException</code>, or if the main
185      * proxy is not an instance of both <code>RemoteMethodControl</code> and
186      * <code>TrustEquivalence</code>, or if the main proxy's class is not
187      * <code>public</code> and it or a superclass has a
188      * non-<code>public</code> direct superinterface, or if the bootstrap
189      * proxy is not an instance of <code>ProxyTrust</code>,
190      * <code>RemoteMethodControl</code>, and <code>TrustEquivalence</code>,
191      * or if any of the superinterfaces of the main proxy's class are not
192      * visible through the class loader specified at the construction of this
193      * exporter
194      * @throws IllegalArgumentException if the specified remote object is not
195      * an instance of <code>ServerProxyTrust</code>, or if the export of the
196      * main remote object throws <code>IllegalArgumentException</code>
197      * @throws IllegalStateException if the export of either remote object
198      * throws <code>IllegalStateException</code>
199      * @throws SecurityException if a non-<code>null</code> class loader was
200      * not specified at the construction of this exporter and the calling
201      * context does not have
202      * {@link RuntimePermission}<code>("getClassLoader")</code> permission
203      */
204     public synchronized Remote export(Remote impl) throws ExportException {
205 	if (impl != null && !(impl instanceof ServerProxyTrust)) {
206 	    throw new IllegalArgumentException(
207 					   "must implement ServerProxyTrust");
208 	}
209 	if (loader == null) {
210 	    SecurityManager sm = System.getSecurityManager();
211 	    if (sm != null) {
212 		sm.checkPermission(loaderPermission);
213 	    }
214 	}
215 	Object main = mainExporter.export(impl);
216 	Class c = main.getClass();
217 	LinkedList ifaces = new LinkedList();
218 	Object boot = null;
219 	boolean ok = false;
220 	try {
221 	    if (!(main instanceof RemoteMethodControl)) {
222 		throw new ExportException(
223 			  "main proxy must implement RemoteMethodControl");
224 	    } else if (!(main instanceof TrustEquivalence)) {
225 		throw new ExportException(
226 			     "main proxy must implement TrustEquivalence");
227 	    }
228 	    boolean needPub = !Modifier.isPublic(c.getModifiers());
229 	    for (Class sup = c; sup != null; sup = sup.getSuperclass()) {
230 		Class[] ifs = sup.getInterfaces();
231 		for (int i = ifs.length; --i >= 0; ) {
232 		    if (needPub && !Modifier.isPublic(ifs[i].getModifiers())) {
233 			throw new ExportException(
234 				"main proxy implements non-public interface");
235 		    }
236 		    ifaces.remove(ifs[i]);
237 		    ifaces.addFirst(ifs[i]);
238 		}
239 	    }
240 	    ref = new WeakRef(impl);
241 	    synchronized (refs) {
242 		if (reaper == null) {
243 		    reaper = new Reaper();
244 		    systemThreadPool.execute(reaper,
245 					     "ProxyTrustExporter.Reaper");
246 		}
247 		refs.add(ref);
248 	    }
249 	    boot = bootExporter.export(ref.boot);
250 	    if (!(boot instanceof ProxyTrust)) {
251 		throw new ExportException(
252 			     "bootstrap proxy must implement ProxyTrust");
253 	    } else if (!(boot instanceof RemoteMethodControl)) {
254 		throw new ExportException(
255 			 "bootstrap proxy must implement RemoteMethodControl");
256 	    } else if (!(boot instanceof TrustEquivalence)) {
257 		throw new ExportException(
258 			    "bootstrap proxy must implement TrustEquivalence");
259 	    }
260 	    Remote proxy = (Remote) Proxy.newProxyInstance(
261 				loader != null ? loader : c.getClassLoader(),
262 				(Class[]) ifaces.toArray(
263 						    new Class[ifaces.size()]),
264 				new ProxyTrustInvocationHandler(
265 						 (RemoteMethodControl) main,
266 						 (ProxyTrust) boot));
267 	    ok = true;
268 	    return proxy;
269 	} catch (IllegalArgumentException e) {
270 	    throw new ExportException("export failed", e);
271 	} finally {
272 	    if (!ok) {
273 		if (ref != null) {
274 		    ref.enqueue();
275 		}
276 		if (boot != null) {
277 		    bootExporter.unexport(true);
278 		}
279 		mainExporter.unexport(true);
280 	    }
281 	}
282     }

284     /**
285      * Unexports the remote objects that were previously exported via this
286      * exporter. The <code>unexport</code> method of the main exporter is
287      * called with the specified argument and if that returns
288      * <code>true</code>, the <code>unexport</code> method of the bootstrap
289      * exporter is called with <code>true</code>. The result of the main
290      * <code>unexport</code> call is returned by this method. Any exception
291      * thrown by either <code>unexport</code> call is rethrown by this
292      * method.
293      *
294      * @throws IllegalStateException if the unexport of either remote object
295      * throws <code>IllegalStateException</code>
296      */
297     public synchronized boolean unexport(boolean force) {
298 	if (!mainExporter.unexport(force)) {
299 	    return false;
300 	}
301 	bootExporter.unexport(true);
302 	if (ref != null) {
303 	    ref.enqueue();
304 	}
305 	return true;
306     }

308     /**
309      * Weak reference to the main remote object with strong reference
310      * to the bootstrap remote object.
311      */
312     private static class WeakRef extends WeakReference {
313 	/** The bootstrap remote object */
314 	ProxyTrust boot;
315 
316 	/** Create an instance registered with queue */
317 	WeakRef(Remote impl) {
318 	    super(impl, queue);
319 	    this.boot = new ProxyTrustImpl(this);
320 	}
321 
322 	/** Clear both references */
323 	public void clear() {
324 	    super.clear();
325 	    boot = null;
326 	}
327     }
328 
329     /** WeakRef reaper */
330     private static class Reaper implements Runnable {
331 	Reaper() {
332 	}
333 
334 	/**
335 	 * Keep removing refs from the queue, clearing them, and removing
336 	 * them from the refs list, until the refs list is empty.
337 	 */
338 	public void run() {
339 	    try {
340 		while (true) {
341 		    Reference ref = queue.remove();
342 		    ref.clear();
343 		    synchronized (refs) {
344 			refs.remove(ref);
345 			if (refs.isEmpty()) {
346 			    reaper = null;
347 			    return;
348 			}
349 		    }
350 		}
351 	    } catch (InterruptedException e) {
352 	    }
353 	}
354     }

356     /** ProxyTrust impl class */
357     private static class ProxyTrustImpl implements ProxyTrust {
358 	/** Weak reference to the main remote object */
359 	protected final Reference ref;
360 
361 	ProxyTrustImpl(Reference ref) {
362 	    this.ref = ref;
363 	}
364 
365 	/** Delegate to the main remote object */
366 	public TrustVerifier getProxyVerifier() throws RemoteException {
367 	    ServerProxyTrust impl = (ServerProxyTrust) ref.get();
368 	    if (impl == null) {
369 		throw new UnsupportedOperationException("impl is gone");
370 	    }
371 	    return impl.getProxyVerifier();
372 	}
373     }
