46  /**
47   * Permission required to dynamically grant permissions by security policy
48   * providers which implement the {@link DynamicPolicy} interface.  Each
49   * <code>GrantPermission</code> instance contains a set of permissions that can
50   * be granted by code authorized with the <code>GrantPermission</code>.  When
51   * the {@link DynamicPolicy#grant DynamicPolicy.grant} method is invoked, the
52   * <code>checkPermission</code> method of the installed security manager (if
53   * any) is called with a <code>GrantPermission</code> containing the
54   * permissions to grant; if the calling context does not have any permissions
55   * which imply the <code>GrantPermission</code>, then the grant operation will
56   * fail.
57   * <p>
58   * In addition to authorizing granting of contained permissions, each
59   * <code>GrantPermission</code> also authorizes granting of
60   * <code>GrantPermission</code>s for contained permissions, as well as granting
61   * of permissions contained within nested <code>GrantPermission</code>s.  For
62   * example, if <code>GrantPermission g1</code> contains <code>Permission
63   * p</code>, <code>g1</code> authorizes granting of both <code>p</code> and
64   * <code>GrantPermission(p)</code>; if <code>GrantPermission g2</code> contains
65   * <code>GrantPermission(p)</code>, then <code>g2</code> also authorizes
66   * granting of both <code>p</code> and <code>GrantPermission(p)</code>.
67   * <p>
68   * The name (also referred to as the "target name") of each
69   * <code>GrantPermission</code> instance carries a string representation of the
70   * permissions contained by the <code>GrantPermission</code>, while the actions
71   * string of each <code>GrantPermission</code> is always the empty string.  If
72   * a <code>GrantPermission</code> is serialized, only its name string is sent
73   * (i.e., contained permissions are not themselves serialized).  Upon
74   * deserialization, the set of contained permissions is reconstituted based on
75   * information in the name string.  <code>GrantPermission</code>s constructed
76   * explicitly with {@link UnresolvedPermission}s (through either the {@link
77   * #GrantPermission(Permission)} or {@link #GrantPermission(Permission[])}
78   * constructor) will have incomplete target names that cannot be used to
79   * instantiate other <code>GrantPermission</code>s, and will not be
80   * serializable--attempting to serialize such a <code>GrantPermission</code>
81   * will cause a <code>java.io.NotSerializableException</code> to be thrown.
82   * <p>
83   * The syntax of the target name approximates that used for specifying
84   * permissions in the default security policy file; it is listed below using
85   * the same grammar notation employed by <i>The Java(TM) Language
86   * Specification</i>:
87   * <div>
88   * <i>Target</i>:<br>
89   * &nbsp;&nbsp;<i>DelimiterDeclaration</i><sub>opt</sub> <i>Permissions</i> ;<sub>opt</sub><br>
90   * <br>
91   * <i>DelimiterDeclaration</i>:<br>
92   * &nbsp;&nbsp;delim = <i>DelimiterCharacter</i><br>
93   * <br>
94   * <i>Permissions</i>:<br>
95   * &nbsp;&nbsp;<i>Permission</i><br>
96   * &nbsp;&nbsp;<i>Permissions</i> ; <i>Permission</i><br>
97   * <br>
98   * <i>Permission</i>:<br>
99   * &nbsp;&nbsp;<i>PermissionClassName</i><br>
100  * &nbsp;&nbsp;<i>PermissionClassName Name</i><br>
101  * &nbsp;&nbsp;<i>PermissionClassName Name</i> , <i>Actions</i><br>
102  * <br>
103  * <i>PermissionClassName</i>:<br>
104  * &nbsp;&nbsp;<i>ClassName</i><br>
105  * <br>
106  * <i>Name</i>:<br>
107  * &nbsp;&nbsp;<i>DelimitedString</i><br>
108  * <br>
109  * <i>Actions</i>:<br>
110  * &nbsp;&nbsp;<i>DelimitedString</i><br>
111  * </div>
112  * The production for <i>ClassName</i> is the same as that used in <i>The
113  * Java Language Specification</i>.  <i>DelimiterCharacter</i> can be any
114  * unquoted non-whitespace character other than ';' (single and
115  * double-quote characters themselves are allowed).  If
116  * <i>DelimiterCharacter</i> is not specified, then the double-quote
117  * character is the default delimiter.  <i>DelimitedString</i> is the same
118  * as the <i>StringLiteral</i> production in <i>The Java Language
119  * Specification</i>, except that it is delimited by the
120  * <i>DelimiterDeclaration</i>-specified (or default) delimiter character
121  * instead of the double-quote character exclusively.
122  * <p>
123  * Note that if the double-quote character is used as the delimiter and the
124  * name or actions strings of specified permissions themselves contain nested
125  * double-quote characters, then those characters must be escaped (or in some
126  * cases doubly-escaped) appropriately.  For example, the following policy file
127  * entry would yield a <code>GrantPermission</code> containing a
128  * <code>FooPermission</code> in which the target name would include the word
129  * "quoted" surrounded by double-quote characters:
130  * <pre>
131  * permission net.jini.security.GrantPermission
132  *     "FooPermission \"a \\\"quoted\\\" string\"";
133  * </pre>
134  * For comparison, the following policy file entry which uses a custom
135  * delimiter would yield an equivalent <code>GrantPermission</code>:
136  * <pre>
137  * permission net.jini.security.GrantPermission
138  *     "delim=| FooPermission |a \"quoted\" string|";
139  * </pre>
140  * Some additional example policy file permissions:
141  * <pre>
142  * // allow granting of permission to listen for and accept connections
143  * permission net.jini.security.GrantPermission
144  *     "java.net.SocketPermission \"localhost:1024-\", \"accept,listen\"";
145  *
146  * // allow granting of permissions to read files under /foo, /bar directories
147  * permission net.jini.security.GrantPermission 
148  *     "delim=' java.io.FilePermission '/foo/-', 'read'; java.io.FilePermission '/bar/-', 'read'";
149  *
150  * // allow granting of permission for client authentication as jack, with or without delegation, to any server
151  * permission net.jini.security.GrantPermission
152  *     "delim=| net.jini.security.AuthenticationPermission |javax.security.auth.x500.X500Principal \"CN=jack\"|, |delegate|";
153  * </pre>
154  *
155  * @author	Sun Microsystems, Inc.
156  * @see		DynamicPolicy#grant(Class, Principal[], Permission[])
157  * @since 2.0
158  */

235      * Returns a newly created empty mutable permission collection for
236      * <code>GrantPermission</code> instances.  The <code>implies</code> method
237      * of the returned <code>PermissionCollection</code> instance is defined as
238      * follows: for a given <code>GrantPermission g</code>, let
239      * <code>c(g)</code> denote the set of all permissions contained within

243      * g</code> is implied by the <code>PermissionCollection pc</code> if and
244      * only if each permission in <code>c(g)</code> is implied by the union of
245      * <code>c(p)</code> for all <code>p</code> in <code>pc</code>.  
246      * <p>

249      * <code>UnresolvedPermission p1</code> is taken to imply another
250      * <code>UnresolvedPermission p2</code> if and only if the serialized

252      *

260     /**
261      * Returns <code>true</code> if the given permission is a
262      * <code>GrantPermission</code> implied by this permission, or
263      * <code>false</code> otherwise.  Implication is defined as follows: for a
264      * given <code>GrantPermission g</code>, let <code>c(g)</code> denote the
265      * set of all permissions contained within <code>g</code> or within
266      * arbitrarily nested <code>GrantPermission</code>s inside <code>g</code>,
267      * excluding nested <code>GrantPermission</code>s themselves.  Then, a
268      * <code>GrantPermission g1</code> is implied by another
269      * <code>GrantPermission g2</code> if and only if each permission in
270      * <code>c(g1)</code> is implied by <code>c(g2)</code>.
271      * <p>
272      * Implication of contained
273      * <code>java.security.UnresolvedPermission</code>s is special-cased: an
274      * <code>UnresolvedPermission p1</code> is taken to imply another
275      * <code>UnresolvedPermission p2</code> if and only if the serialized
276      * representations of <code>p1</code> and <code>p2</code> are identical.
277      * 
278      * @param	permission permission to check
279      * @return	<code>true</code> if given permission is implied by this
280      *		permission, <code>false</code> otherwise
281      */
282     public boolean implies(Permission permission) {
283 	if (!(permission instanceof GrantPermission)) {
284 	    return false;
285 	}
286 	// perm -> perm implies infrequent, so construct implier lazily
287 	if (implier == null) {
288 	    Implier imp = new Implier();
289 	    imp.add(this);
290 	    implier = imp;
291 	}
292 	return implier.implies(permission);
293     }

295     /**
296      * Returns <code>true</code> if the given object is a
297      * <code>GrantPermission</code> which both implies and is implied by this
298      * permission; returns <code>false</code> otherwise.
299      * 
300      * @param	obj object to compare against
301      * @return	<code>true</code> if given object is a
302      * 		<code>GrantPermission</code> which both implies and is implied
303      * 		by this permission, <code>false</code> otherwise
304      */
305     public boolean equals(Object obj) {
306 	if (this == obj) {
307 	    return true;
308 	}
309 	if (obj instanceof GrantPermission) {
310 	    GrantPermission gp = (GrantPermission) obj;
311 	    return this.implies(gp) && gp.implies(this);
312 	}
313 	return false;
314     }

323     /**
324      * Returns hash code computed by summing hash codes of each distinct
325      * permission class name.
326      */
327     private int computeHashCode() {
328 	int sum = 0;
329 	HashSet set = new HashSet(grants.length);
330 	for (int i = 0; i < grants.length; i++) {
331 	    Permission p = grants[i];
332 	    String pcn = p.getClass().getName();
333 	    if (p instanceof AllPermission) {
334 		return pcn.hashCode();
335 	    } else if (p instanceof UnresolvedPermission) {
336 		pcn += ":" + p.getName();    // add name of unresolved class
337 	    }
338 	    if (!set.contains(pcn)) {
339 		set.add(pcn);
340 		sum += pcn.hashCode();
341 	    }
342 	}
343 	return sum;
344     }

388     /**
389      * Initializes GrantPermission to contain permissions described in the
390      * given name.  Throws an IllegalArgumentException if the name is
391      * misformatted, or specifies an invalid permission class.  Throws a
392      * SecurityException if access to the class is not permitted.
393      */
394     private static Permission[] initFromName(String name) {
395 	PermissionInfo[] pia = parsePermissions(name);
396 	ArrayList l = new ArrayList();
397 	for (int i = 0; i < pia.length; i++) {
398 	    PermissionInfo pi = pia[i];
399 
400 	    SecurityManager sm = System.getSecurityManager();
401 	    if (sm != null) {
402 		int d = pi.type.lastIndexOf('.');
403 		if (d != -1) {
404 		    sm.checkPackageAccess(pi.type.substring(0, d));
405 		}
406 	    }
407 	    Class cl;
408 	    try {
409 		cl = Class.forName(pi.type);
410 	    } catch (ClassNotFoundException ex) {
411 		l.add(new UnresolvedPermission(
412 		    pi.type, pi.name, pi.actions, null));
413 		continue;
414 	    }
415 	    if (!Permission.class.isAssignableFrom(cl)) {
416 		throw new IllegalArgumentException(
417 		    "not a permission class: " + cl);
418 	    }
419 	    if (!Modifier.isPublic(cl.getModifiers())) {
420 		throw new IllegalArgumentException(
421 		    "non-public permission class: " + cl);
422 	    }
423 	    
424 	    if (pi.name == null) {
425 		try {
426 		    Constructor c = cl.getConstructor(PARAMS0);
427 		    l.add(c.newInstance(new Object[0]));
428 		    continue;
429 		} catch (Exception ex) {
430                     if (ex instanceof RuntimeException) throw (RuntimeException) ex;
431 		}
432 	    } 
433 	    if (pi.actions == null) {
434 		try {
435 		    Constructor c = cl.getConstructor(PARAMS1);
436 		    l.add(c.newInstance(new Object[]{ pi.name }));
437 		    continue;
438 		} catch (Exception ex) {
439                     if (ex instanceof RuntimeException) throw (RuntimeException) ex;
440 		}
441 	    } 
442 	    try {
443 		Constructor c = cl.getConstructor(PARAMS2);
444 		l.add(c.newInstance(new Object[]{ pi.name, pi.actions }));
445 		continue;
446 	    } catch (Exception ex) {
447                 if (ex instanceof RuntimeException) throw (RuntimeException) ex;
448 	    }
449 	    throw new IllegalArgumentException(
450 		"uninstantiable permission class: " + cl);
451 	}
452 	return flatten((Permission[]) l.toArray(new Permission[l.size()]));
453     }

455     /**
456      * Parses permission information from given GrantPermission name string.
457      * Throws an IllegalArgumentException if the name string is misformatted.
458      */
459     private static PermissionInfo[] parsePermissions(String s) {
460 	try {
461 	    ArrayList l = new ArrayList();
462 	    StreamTokenizer st = createTokenizer(s);
463 	    char delim = '"';
464 
465 	    if (st.nextToken() == StreamTokenizer.TT_WORD &&
466 		st.sval.equals("delim"))
467 	    {
468 		if (st.nextToken() == '=') {
469 		    if (st.nextToken() == StreamTokenizer.TT_WORD) {
470 			if (st.sval.length() > 1) {
471 			    throw new IllegalArgumentException(
472 				"excess delimiter characters");
473 			}
474 			delim = st.sval.charAt(0);
475 		    } else {
476 			delim = (char) st.ttype;
477 		    }
478 		    if (delim == ';') {
479 			throw new IllegalArgumentException(
480 			    "illegal delimiter ';'");
481 		    }
482 		} else {	// rewind
483 		    st = createTokenizer(s);
484 		}
485 		st.nextToken();
486 	    }
487 	    st.quoteChar(delim);
488 
489 	    do {
490 		String type, name = null, actions = null;
491 
492 		if (st.ttype != StreamTokenizer.TT_WORD) {
493 		    throw new IllegalArgumentException(
494 			"expected permission type");
495 		}
496 		type = st.sval;
497 		
498 		// REMIND: allow unquoted name/actions?
499 		st.nextToken();
500 		if (st.ttype == StreamTokenizer.TT_EOF || st.ttype == ';') {
501 		    l.add(new PermissionInfo(type, null, null));
502 		    continue;
503 		} else if (st.ttype == delim) {
504 		    name = st.sval;
505 		} else {
506 		    throw new IllegalArgumentException(
507 			"expected permission name or ';'");
508 		}
509 		
510 		st.nextToken();
511 		if (st.ttype == StreamTokenizer.TT_EOF || st.ttype == ';') {
512 		    l.add(new PermissionInfo(type, name, null));
513 		    continue;
514 		} else if (st.ttype != ',') {
515 		    throw new IllegalArgumentException("expected ',' or ';'");
516 		}
517 
518 		if (st.nextToken() != delim) {
519 		    throw new IllegalArgumentException(
520 			"expected permission actions");
521 		}
522 		actions = st.sval;
523 		
524 		st.nextToken();
525 		if (st.ttype == StreamTokenizer.TT_EOF || st.ttype == ';') {
526 		    l.add(new PermissionInfo(type, name, actions));
527 		    continue;
528 		} else {
529 		    throw new IllegalArgumentException("expected ';'");
530 		}
531 
532 	    } while (st.nextToken() != StreamTokenizer.TT_EOF);
533 
534 	    return (PermissionInfo[]) l.toArray(new PermissionInfo[l.size()]);
535 	} catch (IOException ex) {
536 	    throw (Error) new InternalError().initCause(ex);
537 	}
538     }

540     /**
541      * Returns tokenizer for parsing given string.  The tokenizer is configured
542      * similarly to that used by sun.security.provider.PolicyParser, except
543      * that comments are disabled and no quote character is set (yet).
544      */
545     private static StreamTokenizer createTokenizer(String s) {
546 	StreamTokenizer st = new StreamTokenizer(new StringReader(s));
547 	st.resetSyntax();
548 	st.wordChars('a', 'z');
549 	st.wordChars('A', 'Z');
550 	st.wordChars('.', '.');
551 	st.wordChars('0', '9');
552 	st.wordChars('_', '_');
553 	st.wordChars('$', '$');
554 	st.wordChars(128 + 32, 255);
555 	st.whitespaceChars(0, ' ');
556 	st.lowerCaseMode(false);
557 	st.ordinaryChar('/');
558 	st.slashSlashComments(false);
559 	st.slashStarComments(false);
560 	return st;
561     }

563     /**
564      * Constructs GrantPermission name/target string appropriate for given list
565      * of permissions.
566      */
567     private static String constructName(Permission[] pa) {
568 	StringBuilder sb = new StringBuilder(60);
569 	for (int i = 0, l = pa.length; i < l; i++) {
570 	    Permission p = pa[i];
571 	    if (p instanceof UnresolvedPermission) {
572 		sb.append(p)
573 		  .append("; ");
574 	    } else {
575 		Class cl = p.getClass();
576 		int nargs = maxConsArgs(cl);
577 		String t = cl.getName(), n = p.getName(), a = p.getActions();
578 		if (nargs == 2 && a != null) {
579 		    // REMIND: handle null name?
580 		    sb.append(t)
581 		      .append(" ")
582 		      .append(quote(n))
583 		      .append(", ")
584 		      .append(quote(a))
585 		      .append("; ");
586 		} else if (nargs >= 1 && n != null) {
587 		    sb.append(t)
588 		      .append(" ")
589 		      .append(quote(n))
590 		      .append("; ");
591 		} else {
592 		    sb.append(t)
593 		      .append("; ");
594 		}
595 	    }
596 	}
597 	return sb.toString().trim();
598     }

600     /**
601      * Returns the maximum number of String parameters (up to 2) accepted by a
602      * constructor of the given class.  Returns -1 if no matching constructor
603      * (including no-arg constructor) is defined by given class.
604      */
605     private static int maxConsArgs(Class cl) {
606 	try {
607 	    cl.getConstructor(PARAMS2);
608 	    return 2;
609 	} catch (Exception ex) {
610 	}
611 	try {
612 	    cl.getConstructor(PARAMS1);
613 	    return 1;
614 	} catch (Exception ex) {
615 	}
616 	try {
617 	    cl.getConstructor(PARAMS0);
618 	    return 0;
619 	} catch (Exception ex) {
620 	}
621 	return -1;
622     }

624     /**
625      * Returns quoted string literal that, if parsed by
626      * java.io.StreamTokenizer, would yield the given string.  This method is
627      * essentially a copy of org.apache.river.config.ConfigUtil.stringLiteral; the
628      * two methods are kept separate since ConfigUtil.stringLiteral could
629      * conceivably escape unicode characters, while such escaping would be
630      * incorrect for GrantPermission.
631      */
632     private static String quote(String s) {
633 	StringBuffer sb = new StringBuffer(s.length() + 2);
634 	sb.append('"');
635 	char[] ca = s.toCharArray();
636 	for (int i = 0; i < ca.length; i++) {
637 	    char c = ca[i];
638 	    if (c == '\\' || c == '"') {
639 		sb.append("\\").append(c);
640 	    } else if (c == '\n') {
641 		sb.append("\\n");
642 	    } else if (c == '\r') {
643 		sb.append("\\r");
644 	    } else if (c == '\t') {
645 		sb.append("\\t");
646 	    } else if (c == '\f') {
647 		sb.append("\\f");
648 	    } else if (c == '\b') {
649 		sb.append("\\b");
650 	    } else if (c < 0x20) {
651 		sb.append("\\").append(Integer.toOctalString(c));
652 	    } else {
653 		sb.append(c);
654 	    }
655 	}
656 	return sb.append('"').toString();
657     }

659     /**
660      * Returns an array containing all non-GrantPermission permissions in the
661      * given permission array, including those contained in nested
662      * GrantPermissions in the array.
663      */
664     private static Permission[] flatten(Permission[] pa) {
665 	List l = new ArrayList(pa.length);
666 	for (int i = 0; i < pa.length; i++) {
667 	    Permission p = pa[i];
668 	    if (p instanceof GrantPermission) {
669 		l.addAll(Arrays.asList(((GrantPermission) p).grants));
670 	    } else {
671 		l.add(p);
672 	    }
673 	}
674 	return (Permission[]) l.toArray(new Permission[l.size()]);
675     }

693     /**
694      * Class for checking implication of contained permissions.
695      */
696     private static class Implier {
697 	
698 	private final PermissionCollection perms = new Permissions();
699 	private final ArrayList unresolved = new ArrayList();
700 
701 	void add(GrantPermission gp) {
702 	    for (int i = 0; i < gp.grants.length; i++) {
703 		Permission p = gp.grants[i];
704 		if (!impliesContained(p)) {
705 		    perms.add(p);
706 		    if (p instanceof UnresolvedPermission) {
707 			unresolved.add(p);
708 		    }
709 		}
710 	    }
711 	}
712 	
713 	boolean implies(Permission p) {
714 	    if (!(p instanceof GrantPermission)) {
715 		return false;
716 	    }
717 	    Permission[] pa = ((GrantPermission) p).grants;
718 	    for (int i = 0; i < pa.length; i++) {
719 		if (!impliesContained(pa[i])) {
720 		    return false;
721 		}
722 	    }
723 	    return true;
724 	}
725 
726 	private boolean impliesContained(Permission p) {
727 	    if (p instanceof UnresolvedPermission) {
728 		for (Iterator i = unresolved.iterator(); i.hasNext();) {
729 		    if (implies((UnresolvedPermission) i.next(),
730 				(UnresolvedPermission) p))
731 		    {
732 			return true;
733 		    }
734 		}
735 		return false;
736 	    } else {
737 		return perms.implies(p);
738 	    }
739 	}
740 	
741 	private static boolean implies(UnresolvedPermission p1,
742 				       UnresolvedPermission p2)
743 	{
744 	    if (p1 == p2) {
745 		return true;
746 	    }
747 	    // REMIND: use UnresolvedPermission.equals() once 4513737 fixed
748 	    try {
749 		ByteArrayOutputStream bout = new ByteArrayOutputStream();
750 		ObjectOutputStream oout = new ObjectOutputStream(bout);
751 
752 		bout.reset();
753 		oout.writeObject(p1);
754 		oout.flush();
755 		byte[] b1 = bout.toByteArray();
756 		
757 		oout.reset();
758 		bout.reset();
759 		oout.writeObject(p2);
760 		oout.flush();
761 		byte[] b2 = bout.toByteArray();
762 		
763 		return Arrays.equals(b1, b2);
764 	    } catch (IOException ex) {
765 		throw (Error) new InternalError().initCause(ex);
766 	    }
767 	}
768     }

770     /**
771      * PermissionCollection variant returned by newPermissionCollection().
772      *
773      * @serial include
774      */
775     static class GrantPermissionCollection extends PermissionCollection {
776 
777 	private static final long serialVersionUID = 8227621799817733985L;
778 
779 	/**
780 	 * @serialField perms List The permissions.
781 	 */
782 	private static final ObjectStreamField[] serialPersistentFields = {
783 	    new ObjectStreamField("perms", List.class, true)
784 	};
785         
786         // Serial form.
787 	private List<Permission> perms = new ArrayList<Permission>();
788 	private Implier implier = new Implier();
789 
790 	public synchronized void add(Permission p) {
791 	    if (!(p instanceof GrantPermission)) {
792 		throw new IllegalArgumentException("invalid permission: " + p);
793 	    }
794 	    if (isReadOnly()) {
795 		throw new SecurityException(
796 		    "can't add to read-only PermissionCollection");
797 	    }
798             // Cannot use TreeSet to ensure correctness, just don't
799             // add twice, in other words check must be external.
800             // Stack overflow may occur if permissions added without checking
801             perms.add(p);
802             implier.add((GrantPermission) p);
803 	    
804 	}
805 	
806 	public synchronized Enumeration<Permission> elements() {
807 	    return Collections.enumeration(perms);
808 	}
809 	
810 	public synchronized boolean implies(Permission p) {
811 	    return implier.implies(p);
812         }
813 
814 	public synchronized void setReadOnly() {
815 	    super.setReadOnly();
816 	}
817 
818 	public synchronized boolean isReadOnly() {
819 	    return super.isReadOnly();
820 	}
821 
822 	/**
823 	 * Writes the permissions list.
824 	 */
825 	private synchronized void writeObject(ObjectOutputStream s)
826 	    throws IOException
827 	{
828 	    s.defaultWriteObject();
829 	}
830 
831 	/**
832 	 * Verifies the permissions list.
833 	 *
834 	 * @throws InvalidObjectException if the list is
835 	 * 	   <code>null</code> or any element is not an instance of
836 	 * 	   <code>GrantPermission</code>
837 	 */
838 	private void readObject(ObjectInputStream s)
839 	    throws IOException, ClassNotFoundException
840 	{
841 	    s.defaultReadObject();
842 	    if (perms == null) {
843 		throw new InvalidObjectException("list cannot be null");
844 	    }
845 	    if (!perms.getClass().equals(ArrayList.class)) {
846 		perms = new ArrayList(perms);
847 	    }
848 	    if (perms.contains(null)) {
849 		throw new InvalidObjectException(
850 		    "element must be a GrantPermission");
851 	    }
852 
853 	    GrantPermission[] pa;
854 	    try {
855 		pa = (GrantPermission[]) 
856 		    perms.toArray(new GrantPermission[perms.size()]);
857 	    } catch (ArrayStoreException e) {
858 		throw new InvalidObjectException(
859 		    "element must be a GrantPermission");
860 	    }
861 
862 	    implier = new Implier();
863 	    for (int i = 0; i < pa.length; i++) {
864 		implier.add(pa[i]);
865 	    }
866 	}
