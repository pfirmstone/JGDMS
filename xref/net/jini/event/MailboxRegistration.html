18  package net.jini.event;
19  
20  import java.rmi.RemoteException;
21  
22  import net.jini.core.event.RemoteEventListener;
23  import net.jini.core.lease.Lease;
24  
25  
26  /**
27   * The <code>MailboxRegistration</code> defines the interface through which
28   * a client manages its registration and its notification processing. 
29   * Event mailbox clients use this interface to:
30   * <UL>
31   * <LI>Manage the lease for this registration
32   * <LI>Obtain a <code>RemoteEventListener</code> reference that
33   *     can be registered with event generators. This listener will
34   *     then store any received events for this registration.
35   * <LI>Enable or disable the delivery of any stored events for this
36   *     registration.
37   * </UL>
38   *
39   * @author Sun Microsystems, Inc.
40   *
41   * @since 1.1
42   */
43  public interface MailboxRegistration {
44  
45      /**
46       * Returns the <code>Lease</code> object associated
47       * with this registration. The client can renew or 
48       * cancel the registration with the mailbox service
49       * through this lease object.
50       *
51       * @return The lease object associated with this registration
52       */
53      public Lease getLease();
54  
55      /**
56       * Returns the <code>RemoteEventListener</code> associated
57       * with this registration. This listener can then be submitted
58       * as the <code>RemoteEventListener</code> argument to an
59       * event generator's registration method(s).
60       *
61       * @return The <code>RemoteEventListener</code> associated
62       *         with this registration.
63       */
64      public RemoteEventListener getListener();
65  
66      /**
67       * Initiates delivery of stored notifications to the supplied
68       * <code>target</code> listener, if any. If a target listener
69       * already exists, then it will be replaced with the specified
70       * target listener. Passing <code>null</code> as the
71       * <code>target</code> parameter has the same effect as calling
72       * the <code>disableDelivery</code> method.
73       *
74       * @param target The listener to be notified of stored events, if any.
75       *
76       * @throws IllegalArgumentException if the supplied <code>target</code> 
77       *         parameter is a listener object that was generated by the
78       *         mailbox service itself.
79       *
80       * @throws java.rmi.RemoteException if there is
81       *  a communication failure between the client and the service.
82       */
83      public void enableDelivery(RemoteEventListener target)
84  		    throws RemoteException;
85  
86      /**
87       * Ceases delivery of stored notifications to the existing
88       * target listener, if any. It is acceptable to call this
89       * method even if no target listener is currently enabled.
90       *
91       * @throws java.rmi.RemoteException if there is
92       *  a communication failure between the client and the service.
93       */
94      public void disableDelivery() throws RemoteException;
95  }
