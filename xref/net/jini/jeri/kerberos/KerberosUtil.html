145     static boolean canGetSubject() {
146 	try {

168     /**
169      * Test whether the specified constraint can possibly be supported
170      * by this provider.
171      *
172      * @param c the constraint to be tested
173      * @return true if the specified constraints can possibly be
174      *         supported, false otherwise.
175      */
176     static boolean isSupportableConstraint(InvocationConstraint c) {
177 
178 	if (c instanceof ConstraintAlternatives) {
179 	    Set alts = ((ConstraintAlternatives) c).elements();
180 	    Class type = null;
181 	    for (Iterator iter = alts.iterator(); iter.hasNext(); ) {
182 		InvocationConstraint alt= (InvocationConstraint) iter.next();
183 		if (type == null) {
184 		    type = alt.getClass();
185 		} else if (type != alt.getClass()) {
186 		    return false; // does not support heterogenous alternatives
187 		}
188 		if (isSupportableConstraint(alt))
189 		    return true;
190 	    }
191 	    return false;
192 	}
193 
194 	if (!isSupportedConstraintType(c))
195 	    return false; // unsupported constraint type
196 
197 	if (c instanceof Integrity) {
198 	    return c == Integrity.YES;
199 	} else if (c instanceof ClientAuthentication) {
200 	    return c == ClientAuthentication.YES;
201 	} else if (c instanceof ServerAuthentication) {
202 	    return c == ServerAuthentication.YES;
203 	} else if (c instanceof ClientMinPrincipal) {
204 	    Set elems = ((ClientMinPrincipal) c).elements();
205 	    if (elems.size() > 1) {
206 		return false; // can only authenticate as one principal
207 	    } else { // elems contains at least one element
208 		return elems.iterator().next() instanceof KerberosPrincipal;
209 	    }
210 	} else if (c instanceof ClientMinPrincipalType) {
211 	    Set elems = ((ClientMinPrincipalType) c).elements();
212 	    if (elems.size() > 1) {
213 		return false; // can only support one type
214 	    } else { // elems contains at least one element
215 		return elems.contains(KerberosPrincipal.class);
216 	    }
217 	} else if (c instanceof ClientMaxPrincipal) {
218 	    Set elems = ((ClientMaxPrincipal) c).elements();
219 	    for (Iterator iter = elems.iterator(); iter.hasNext(); ) {
220 		if (iter.next() instanceof KerberosPrincipal)
221 		    return true;
222 	    }
223 	    return false;
224 	} else if (c instanceof ClientMaxPrincipalType) {
225 	    Set elems = ((ClientMaxPrincipalType) c).elements();
226 	    return elems.contains(KerberosPrincipal.class);
227 	} else if (c instanceof ServerMinPrincipal) {
228 	    Set elems = ((ServerMinPrincipal) c).elements();
229 	    if (elems.size() > 1) {
230 		return false; // can only authenticate as one principal
231 	    } else { // elems contains at least one element
232 		return elems.iterator().next() instanceof KerberosPrincipal;
233 	    }
234 	}
235 
236 	return true;
237     }

239     /**
240      * Test whether the specified configuration is satisfiable by the
241      * given constraint.
242      *
243      * @param config configuration to be tested
244      * @param c the constraint to be tested
245      * @return true if the specified configuration is allowed by
246      *         the given constraint, false otherwise.
247      */
248     static boolean isSatisfiable(Config config, InvocationConstraint c) {
249 
250 	/* Note that though some of the checks done here have already
251 	   been done in isSupportedConstraint(c), they have to be
252 	   repeated here to support ConstraintAlternatives. */
253 
254 	if (c instanceof ConstraintAlternatives) {
255 	    Set elems = ((ConstraintAlternatives) c).elements();
256 	    for (Iterator iter = elems.iterator(); iter.hasNext(); ) {
257 		InvocationConstraint elem =
258 		    (InvocationConstraint) iter.next();
259 		if (isSatisfiable(config, elem))
260 		    return true;
261 	    }
262 	    return false;
263 	}
264 
265 	if (!isSupportedConstraintType(c))
266 	    return false; // unsupported constraint type
267 
268 	if (c instanceof Integrity) {
269 	    return c == Integrity.YES;
270 	} else if (c instanceof Confidentiality) {
271 	    return config.encry == (c == Confidentiality.YES);
272 	} else if (c instanceof ClientAuthentication) {
273 	    return c == ClientAuthentication.YES;
274 	} else if (c instanceof ServerAuthentication) {
275 	    return c == ServerAuthentication.YES;
276 	} else if (c instanceof Delegation) {
277 	    return config.deleg == (c == Delegation.YES);
278 //	    if ( config.deleg == (c == Delegation.YES)){
279 //		try {
280 //		    Class.forName("com.sun.security.jgss.GSSUtil");
281 //		    return true;
282 //		} catch (ClassNotFoundException ex){} // Ignore
283 //	    }
284 //	    return false;
285 	} else if (c instanceof ClientMinPrincipal) {
286 	    Set elems = ((ClientMinPrincipal) c).elements();
287 	    if (elems.size() > 1) {
288 		return false; // can only authenticate as one principal
289 	    } else { // elems contains at least one element
290 		return elems.contains(config.clientPrincipal);
291 	    }
292 	} else if (c instanceof ClientMinPrincipalType) {
293 	    Set elems = ((ClientMinPrincipalType) c).elements();
294 	    if (elems.size() > 1) {
295 		return false; // can only support one type
296 	    } else { // elems contains at least one element
297 		return elems.contains(KerberosPrincipal.class);
298 	    }
299 	} else if (c instanceof ClientMaxPrincipal) {
300 	    Set elems = ((ClientMaxPrincipal) c).elements();
301 	    return elems.contains(config.clientPrincipal);
302 	} else if (c instanceof ClientMaxPrincipalType) {
303 	    Set elems = ((ClientMaxPrincipalType) c).elements();
304 	    return elems.contains(KerberosPrincipal.class);
305 	} else if (c instanceof ServerMinPrincipal) {
306 	    Set elems = ((ServerMinPrincipal) c).elements();
307 	    if (elems.size() > 1) {
308 		return false; // can only authenticate as one principal
309 	    } else { // elems contains at least one element
310 		return elems.contains(config.serverPrincipal);
311 	    }
312 	}
313 	return true;
314     }

316     /**
317 
318      * Collect all client principal candidates from the given
319      * constraint.  This method assumes homogeneous alternatives.
320      *
321      * @param c the given constraint
322      * @param cpCandidates the set of candidates satisfiable by the
323      * constraints previously checked, which new principals should be
324      * added to.  This set contains no principals if no client
325      * principal constraint has been checked yet.
326      * @return false if the passed in constraint is {@link
327      * ClientMinPrincipal} or {@link ClientMaxPrincipal}, or {@link
328      * ConstraintAlternatives} whose elements are of those types, and
329      * is not satisfiable regarding to the given set of candidates,
330      * true other wise.
331      */
332     static boolean collectCpCandidates(
333 	InvocationConstraint c, Set cpCandidates)
334     {
335 	boolean isPrincipalConstraint = false;
336 	HashSet cpset = new HashSet();
337 	if (c instanceof ConstraintAlternatives) {
338 	    Set alts = ((ConstraintAlternatives) c).elements();
339 	    for (Iterator iter = alts.iterator(); iter.hasNext(); ) {
340 		c = (InvocationConstraint) iter.next();
341 		if (c instanceof ClientMinPrincipal) {
342 		    isPrincipalConstraint = true;
343 		    Set elems = ((ClientMinPrincipal) c).elements();
344 		    Object cp = elems.iterator().next();
345 		    if (elems.size() > 1 || !(cp instanceof KerberosPrincipal))
346 			continue; // constraint unsupportable
347 		    cpset.add(cp);
348 		} else if (c instanceof ClientMaxPrincipal) {
349 		    isPrincipalConstraint = true;
350 		    Set elems = ((ClientMaxPrincipal) c).elements();
351 		    for (Iterator jter = elems.iterator(); jter.hasNext(); ) {
352 			Object elem = jter.next();
353 			if (elem instanceof KerberosPrincipal)
354 			    cpset.add(elem);
355 		    }
356 		}
357 	    }	    
358 	} else if (c instanceof ClientMinPrincipal) {
359 	    isPrincipalConstraint = true;
360 	    Set elems = ((ClientMinPrincipal) c).elements();
361 	    Object cp = elems.iterator().next();
362 	    if (elems.size() > 1 || !(cp instanceof KerberosPrincipal))
363 		return false; // constraint unsupportable
364 	    cpset.add(cp);
365 	} else if (c instanceof ClientMaxPrincipal) {
366 	    isPrincipalConstraint = true;
367 	    Set elems = ((ClientMaxPrincipal) c).elements();
368 	    for (Iterator iter = elems.iterator(); iter.hasNext(); ) {
369 		Object elem = iter.next();
370 		if (elem instanceof KerberosPrincipal)
371 		    cpset.add(elem);
372 	    }
373 	}
374 
375 	if (isPrincipalConstraint) {
376 	    if (cpCandidates.size() == 0) {
377 		// this constraint is the 1st principal constraint checked
378 		if (cpset.size() > 0) {
379 		    cpCandidates.addAll(cpset);
380 		    return true;
381 		} else {
382 		    return false;
383 		}
384 	    } else { // seen other principal constraints before
385 		cpCandidates.retainAll(cpset);
386 		return cpCandidates.size() > 0;
387 	    }
388 	} else {
389 	    return true; // no say if not principal constraint
390 	}
391     }

394      * Check whether the caller has the AuthenticationPermission with
395      * the specified principals and action.
396      *
397      * @param local local principal of the
398      *        <code>AuthenticationPermission</code>, cannot be *
399      *        <code>null<code>.
400      * @param peer peer principal of the
401      *        <code>AuthenticationPermission</code>.
402      * @param action action of the
403      *        <code>AuthenticationPermission</code>, valid values
404      *        include: * "connect", "delegate", "listen", and

406      * @throws SecurityException if the caller does not have the
407      *         checked permission
408      */
409     static void checkAuthPermission(KerberosPrincipal local,
410 				    KerberosPrincipal peer,
411 				    String action)
412     {
413 	SecurityManager sm = System.getSecurityManager();
414 	if (sm != null) {
415 	    Set localps = Collections.singleton(local);
416 	    Set peerps = null;
417 	    if (peer != null)
418 		peerps = Collections.singleton(peer);
419 	    AuthenticationPermission perm = 
420 		new AuthenticationPermission(localps, peerps, action);
421 	    sm.checkPermission(perm);
422 	}
423     }

440      * Check whether the given set of constraints contains the
441      * candidate constraint.
442      *
443      * @param constraints the constraints to be checked
444      * @param candidate candidate constraint
445      * @return true if the candidate constraint is found in the give
446      *         set of constraints, false otherwise.
447      */
448     static boolean containsConstraint(
449 	Set constraints, InvocationConstraint candidate)
450     {
451 	for (Iterator iter = constraints.iterator(); iter.hasNext(); ) {
452 	    InvocationConstraint c = (InvocationConstraint) iter.next();
453 	    if (c instanceof ConstraintAlternatives) {
454 		Set elems = ((ConstraintAlternatives) c).elements();
455 		return elems.contains(candidate);
456 	    } else if (c.equals(candidate)) {
457 		return true;
458 	    }
459 	}
460 	return false;
461     }

463     /**
464      * Get the GSSCredential corresponding to the given principal from
465      * the given <code>Subject</code>, whose usage type is governed by
466      * the usage parameter.
467      * 
468      * @param subj the subject from which the TGT or
469      *        <code>KerberosKey</code> will be extracted to construct
470      *        the GSSCredential, can not be null
471      * @param principal the principal whose name will be used to
472      *        construct the GSSCredential. If <code>null</code>, then
473      *        a <code>null</code> name will be passed to the
474      *        <code>manager</code> to allow it to choose a default.
475      * @param manager the GSSManager instance that will be used to
476      *        construct the GSSCredential, can not be null
477      * @param usage intended usage for the GSScredential. The value of
478      *        this parameter must be one of: {@link
479      *        GSSCredential#INITIATE_AND_ACCEPT}, {@link
480      *        GSSCredential#ACCEPT_ONLY}, and {@link
481      *        GSSCredential#INITIATE_ONLY}.
482      * @return the requested GSSCredential
483      * @throws UnsupportedConstraintException if failed to get the
484      *         requested <code>GSSCredential</code>
485      */
486     static GSSCredential getGSSCredential (
487 	final Subject subj, final KerberosPrincipal principal,
488 	final GSSManager manager, final int usage)
489 	throws GSSException
490     {
491 	try {
492 	    return (GSSCredential) Subject.doAs(
493                 subj, new PrivilegedExceptionAction() {
494 		    public Object run() throws GSSException {
495 		        GSSName name = manager.createName(principal.getName(),
496 							  krb5NameType);
497 			return manager.createCredential(
498 		            name, GSSCredential.INDEFINITE_LIFETIME,
499 			    krb5MechOid, usage);
500 		    }
501 	        });
502 	} catch (PrivilegedActionException pe) {
503 	    throw (GSSException) pe.getException();
504 	}
505     }

507     /**
508      * Only throw non-generic exception if caller has getSubject
509      * permission.
510      *
511      * @param detailedException the real
512      *        <code>UnsupportedConstraintException</code> or
513      *        <code>SecurityException</code> to be thrown if caller
514      *        has the "getSubject" <code>AuthPermission</code>.
515      * @param genericException the generic
516      *        <code>UnsupportedConstraintException</code> to be thrown
517      *        if caller does not have the "getSubject"
518      *        <code>AuthPermission</code>.
519      */
520     static void secureThrow(Exception detailedException,
521 			    UnsupportedConstraintException genericException)
522 	throws UnsupportedConstraintException
523     {
524 	if (KerberosUtil.canGetSubject()) { // has "getSubject" permission
525 	    if (detailedException instanceof SecurityException) {
526 		throw (SecurityException) detailedException;
527 	    } else {
528 		throw (UnsupportedConstraintException) detailedException;
529 	    }
530 	} else {
531 	    throw genericException;
532 	}
533     }

604     /** An iterator returns all possible configs */
605     static final class ConfigIter {
606 
607 	private final Set clientPrincipals;
608 	private final KerberosPrincipal serverPrincipal;
609 	private Iterator cpIter;
610 	private final boolean canDeleg; // true if delegation allowed
611 	private int configId;
612 	private int numConfigs;
613 
614 	ConfigIter(Set clientPrincipals, KerberosPrincipal serverPrincipal,
615 		   boolean canDeleg)
616 	{
617 	    this.clientPrincipals = clientPrincipals;
618 	    this.serverPrincipal = serverPrincipal;
619 	    this.canDeleg = canDeleg;
620 	    configId = 0;
621 	    numConfigs = clientPrincipals.size() * 2;
622 	    if (canDeleg)
623 		numConfigs *= 2;
624 	}
625 
626 	boolean hasNext() {
627 	    return configId < numConfigs;
628 	}
629 
630 	Config next() {
631 	    if (configId >= numConfigs)
632 		throw new java.util.NoSuchElementException();
633 
634 	    if (configId % clientPrincipals.size() == 0)
635 		cpIter = clientPrincipals.iterator();
636 
637 	    KerberosPrincipal cp = (KerberosPrincipal) cpIter.next();
638 	    int encryId = (configId / clientPrincipals.size()) % 2;
639 	    Config config;
640 	    if (canDeleg) {
641 		int delegId = configId / clientPrincipals.size() / 2;
642 		config = new Config(cp, serverPrincipal, BOOL_TABLE[encryId],
643 				    BOOL_TABLE[delegId]);
644 	    } else {
645 		config = new Config(cp, serverPrincipal, BOOL_TABLE[encryId],
646 				    false);
647 	    }
648 	    ++configId;
649 	    return config;
650 	}
651     }

715 	/**
716 	 * Wrap the content of the buffer into a GSS token and write
717 	 * it out to the underlying socket.
718 	 * 
719 	 * @param buf the buffer whose content will be send out
720 	 * @param offset offset marks the start of the content to be
721 	 *        sent out
722 	 * @param len number of bytes to be sent out
723 	 * @throws IOException if problems encountered
724 	 */
725 	void write(byte[] buf, int offset, int len) throws IOException {
726 	    MessageProp prop;
727 	    if (doEncryption) {
728 		prop = new MessageProp(PRIVACY_QOP, true);
729 	    } else { // 2 means "integrity using DES MAC of MD5 of plaintext"
730 		prop = new MessageProp(INTEGRITY_QOP, false);
731 	    }
732 	    byte[] token = null;
733 	    try {
734 		try {
735 		    synchronized (gssContext) {
736 			token = gssContext.wrap(buf, offset, len, prop);
737 		    }
738 		} catch (GSSException ge) {
739 		    IOException ioe = new IOException(
740 			"Failed to wrap buf into GSS token.");
741 		    ioe.initCause(ge);
742 		    throw ioe;
743 		}
744 
745 		if (doEncryption != prop.getPrivacy()) {
746 		    throw new IOException(
747 			"Returned token encryption property is: " +
748 			prop.getPrivacy() + ",\nwhile connection " +
749 			"encryption requirement is: " + doEncryption);
750 		}
751 	    
752 		if (connectionLogger.isLoggable(Level.FINEST)) {
753 		    connectionLogger.log(
754 			Level.FINEST, "wrapped " + len + " bytes (" +
755 			(doEncryption ? "" : "not ") + "encrypted) " +
756 			"into a " + token.length + " bytes token and " +
757 			"sending it over the network");
758 		}
759 		
760 		dos.writeInt(token.length);
761 		dos.write(token);
762 	    } catch (IOException ioe) {
763 		if (connectionLogger.isLoggable(Levels.FAILED)) {
764 		    logThrow(connectionLogger, Levels.FAILED,
765 			     this.getClass(), "write",
766 			     "failed to wrap buf of size {0} into a GSS " +
767 			     "token,\nconnection is {1},\nthrows ",
768 			     new Object[] {Integer.valueOf(len), this}, ioe);
769 		}
770 		throw ioe;
771 	    }
772 	}
773 
774 	/** Flush the output stream used for send. */
775 	void flush() throws IOException {
776 	    dos.flush();
777 	}

779 	/**
780 	 * Block until a complete GSS token has been received, unwrap
781 	 * it, and return its content.
782 	 * 
783 	 * @return byte array of the unwrapped GSS token
784 	 * @throws IOException if problems encountered
785 	 */
786 	byte[] read() throws IOException {
787 	    try {
788 		MessageProp prop = new MessageProp(0, false);
789 		byte[] token = new byte[dis.readInt()];
790 		dis.readFully(token);
791 		
792 		byte[] bytes;
793 		try {
794 		    synchronized (gssContext) {
795 			bytes = gssContext.unwrap(
796 			    token, 0, token.length, prop);
797 		    }
798 		} catch (GSSException e) {
799 		    IOException ioe = new IOException(
800 			"Failed to unwrap a GSS token of length " +
801 			token.length);
802 		    ioe.initCause(e);
803 		    throw ioe;
804 		}
805 		
806 		/* this state of the connection can changed by
807 		   every incoming token */
808 		doEncryption = prop.getPrivacy();
809 		
810 		if (connectionLogger.isLoggable(Level.FINEST)) {
811 		    connectionLogger.log(
812 			Level.FINEST,  "received a " + token.length +
813 			" bytes token (" + (doEncryption ? "" : "not ") +
814 			"encrypted), " + bytes.length + " bytes when " +
815 			"unwrapped");
816 		}
817 		return bytes;
818 	    } catch (IOException ioe) {
819 		if (connectionLogger.isLoggable(Levels.FAILED)) {
820 		    logThrow(connectionLogger, Levels.FAILED,
821 			     this.getClass(), "read",
822 			     "read fails on connection {0}, throws",
823 			     new Object[] {this}, ioe);
824 		}
825 		throw ioe;
826 	    }
827 	}

894     /**
895      * Output stream returned by getOutputStream() of client or server
896      * connection
897      */
898     static class ConnectionOutputStream extends OutputStream {
899 	// 8k is chosen because it might be close to a page size
900 	private static final int bufSize = 8000; // buf + overhead < 8192 ?
901 	private final byte[] buf;
902 	private int curLen; // current content length of the internal buffer
903 	private final Connection connection;
904 
905 	/** Construct an instance of ConnectionOutputStream */
906 	ConnectionOutputStream(Connection connection) {
907 	    buf = new byte[bufSize];
908 	    curLen = 0; // init buf as empty
909 	    this.connection = connection;
910 	}
911 
912 	// This method's javadoc is inherited from OutStream
913 	public synchronized void write(int b) throws IOException {
914 	    if (curLen == bufSize) {
915 		connection.write(buf, 0, curLen);
916 		curLen = 0;
917 	    }
918 	    buf[curLen++] = (byte) b;
919 	}
920 
921 	// This method's javadoc is inherited from OutStream
922 	@Override
923 	public synchronized void write(byte[] b, int off, int len)
924 	    throws IOException
925 	{
926 	    if (b == null) {
927 		throw new NullPointerException();
928 	    } else if (off < 0 || len < 0 || (off + len) > b.length) {
929 		throw new IndexOutOfBoundsException();
930 	    }
931 
932 	    if ((curLen + len) >= bufSize) {
933 		int count = bufSize - curLen;
934 		System.arraycopy(b, off, buf, curLen, count);
935 		off += count;
936 		len -= count;
937 		connection.write(buf, 0, bufSize);
938 		curLen = 0;
939 	    }
940 
941 	    while (len > bufSize) {
942 		connection.write(b, off, bufSize);
943 		off += bufSize;
944 		len -= bufSize;
945 	    }
946 
947 	    System.arraycopy(b, off, buf, curLen, len);
948 	    curLen += len;
949 	}
950 
951 	// This method's javadoc is inherited from OutStream
952 	@Override
953 	public synchronized void flush() throws IOException {
954 	    if (curLen > 0) {
955 		connection.write(buf, 0, curLen);
956 		curLen = 0;
957 	    }
958 	    connection.flush();
959 	}
960 
961 	/**
962 	 * Flush this stream, then close the DataOutputStream of the
963 	 * enclosed connection.
964 	 */
965 	@Override
966 	public void close() throws IOException {
967 	    try {
968 		flush();
969 	    } finally {
970 		connection.dis.close();
971 	    }
972 	}

975     /**
976      * A synchronized hash map that only maintains soft reference to
977      * its value objects. It can be configured to have a limited
978      * capacity. LRU replacement policy is used when capacity limit is
979      * reached.
980      */
981     static class SoftCache {
982 
983 	/** Internal hash map used to store the actual key value pairs */
984 	private final LRUHashMap hash;
985 
986 	/** Reference queue for cleared ValueCells */
987 	private ReferenceQueue queue = new ReferenceQueue();
988 
989 	/**
990 	 * Construct an instance of the SoftCache, using a default
991 	 * capacity of 8.
992 	 */
993 	SoftCache() {
994 	    this(Integer.MAX_VALUE, 8); // init cache size as unlimited
995 	}
996 
997 	/**
998 	 * Construct an instance of the SoftCache with the given
999 	 * size limit.
1000 	 *
1001 	 * @param maxCacheSize maximum number of entries allowed in
1002 	 *        this cache
1003 	 */
1004 	SoftCache(int maxCacheSize) {
1005 	    this(maxCacheSize, 8);
1006 	}
1007 
1008 	/**
1009 	 * Construct an instance of the SoftCache with the given
1010 	 * size limit and initial capacity.
1011 	 *
1012 	 * @param maxCacheSize maximum number of entries allowed in
1013 	 *        this cache
1014 	 * @param initialCapacity initial capacity of the cache
1015 	 */
1016 	SoftCache(int maxCacheSize, int initialCapacity) {
1017 	    hash = new LRUHashMap(maxCacheSize, initialCapacity);
1018 	}
1019 
1020 	/**
1021 	 * Associates the specified value with the specified key in
1022 	 * this cache. Only a soft reference is maintained to each
1023 	 * value object stored in the cache. If the map previously
1024 	 * contained a mapping for this key, and the old value object
1025 	 * has not been garbage collected, the old value will be
1026 	 * returned to the caller.  This method is synchronized.
1027          *
1028          * @param key key with which the specified value is to be
1029          *        associated.
1030          * @param value - value to be associated with the specified
1031          *        key, only a soft reference is maintained to value in
1032          *        this cache.
1033          * @return previous value associated with specified key, if an
1034          *         old mapping exists and the old value has not been
1035          *         garbage collected, or null if there was no
1036          *         mapping for key. A null return can also indicate
1037          *         that the HashMap previously associated null with
1038          *         the specified key.
1039 	 */
1040 	public synchronized Object put(Object key, Object value) {
1041 	    processQueue();
1042 	    ValueCell vc = ValueCell.create(key, value, queue);
1043 	    return ValueCell.strip(hash.put(key, vc), true);
1044 	}
1045 
1046 	/**
1047 	 * If there is a mapping in this cache for the specified key,
1048 	 * and the softly referenced value of the mapping has not been
1049 	 * garbage collected, return the value, other wise, return
1050 	 * null.  This method is synchronized.
1051 	 *
1052 	 * @param key - key whose associated value is to be returned.
1053 	 * @return the value to which this map maps the specified key.
1054 	 */
1055 	public synchronized Object get(Object key) {
1056 	    processQueue();
1057 	    return ValueCell.strip(hash.get(key), false);
1058 	}
1059 
1060 	/**
1061 	 * Removes the mapping for this key from this cache if it
1062 	 * still exists.  This method is synchronized.
1063 	 *
1064 	 * @param key key whose mapping is to be removed from the cache.
1065 	 * @return previous value associated with the specified key
1066 	 *         that has not been garbage collected, or null. A
1067 	 *         null return can also indicate that the HashMap
1068 	 *         previously associated null with the specified key.
1069  	 */
1070 	public synchronized Object remove(Object key) {
1071 	    processQueue();
1072 	    return ValueCell.strip(hash.remove(key), true);
1073 	}
1074 	
1075 	/**
1076 	 * Removes all entries in this cache.  This method is
1077 	 * synchronized.
1078 	 */
1079 	public synchronized void clear() {
1080 	    processQueue();
1081 	    hash.clear();
1082 	}
1083 
1084 	/** Process the internal ReferenceQueue */
1085 	private void processQueue() {
1086 	    ValueCell vc;
1087 	    while ((vc = (ValueCell) queue.poll()) != null) {
1088 		/*
1089 		 * vc.isValid() is false, then the vc has been
1090 		 * dropped, and the value cell currently in hash
1091 		 * corresponding to vc.key, if exists, must be one
1092 		 * that has been newly inserted using the same key =>
1093 		 * can not do hash.remove(vc.key)
1094 		 */
1095 		if (vc.isValid())
1096 		    hash.remove(vc.key);
1097 	    }
1098 	}

1100 	/** A linked hash map that implements LRU replacement policy */
1101 	private static class LRUHashMap extends LinkedHashMap {
1102             private static final long serialVersionUID = 1L;
1103 
1104 	    private int maxCacheSize;
1105 
1106 	    /**
1107 	     * Construct an instance of the hash map.
1108 	     *
1109 	     * @param maxCacheSize maximum number of entries allowed
1110 	     *        in this map
1111 	     * @param initialCapacity initial capacity of the map
1112 	     * @throws IllegalArgumentException if maxCacheSize is
1113 	     *         negative
1114 	     */
1115 	    LRUHashMap(int maxCacheSize, int initialCapacity) {
1116 		super(initialCapacity, 0.75f, true); // using access-order
1117 		if (maxCacheSize < 0)
1118 		    throw new IllegalArgumentException("negative cache size");
1119 		this.maxCacheSize = maxCacheSize;
1120 	    }
1121 
1122 	    // This method's javadoc is inherited from LinkedHashMap
1123 	    protected boolean removeEldestEntry(Map.Entry eldest) {
1124 		if (size() > maxCacheSize) {
1125 		    /*
1126 		     * clear the soft ref and mark internal key as
1127 		     * invalid, LRUHashMap will take care of the
1128 		     * removing part
1129 		     */
1130 		    ValueCell.strip(eldest.getValue(), true);
1131 		    return true;
1132 		}
1133 		return false;
1134 	    }
1135 	}

1137 	/**
1138 	 * An instance of this class maintains a reference to a key,
1139 	 * and a soft reference to the value the key maps to.
1140 	 */
1141 	private static class ValueCell extends SoftReference {
1142 
1143 	    static private Object INVALID_KEY = new Object();
1144 	    private Object key;
1145 
1146 	    private ValueCell(Object key, Object value, ReferenceQueue queue) {
1147 		super(value, queue);
1148 		this.key = key;
1149 	    }
1150 
1151 	    private static ValueCell create(Object key, Object value,
1152 					    ReferenceQueue queue)
1153 	    {
1154 		if (value == null) return null;
1155 		return new ValueCell(key, value, queue);
1156 	    }
1157 
1158 	    /**
1159 	     * Extract the encapsulated value if the passed in object
1160 	     * is an instance of ValueCell, clear the soft reference
1161 	     * and mark the cell as invalid if drop is true.
1162 	     */
1163 	    private static Object strip(Object val, boolean drop) {
1164 		if (val == null)
1165 		    return null;
1166 		ValueCell vc = (ValueCell)val;
1167 		Object o = vc.get();
1168 		if (drop)
1169 		    vc.drop();
1170 		return o;
1171 	    }
1172 
1173 	    /** 
1174 	     * Return true if this cell has not been dropped, false
1175 	     * otherwise
1176 	     */
1177 	    private boolean isValid() {
1178 		return (key != INVALID_KEY);
1179 	    }
1180 
1181 	    /** Clear the soft reference, and mark the cell as invalid */
1182 	    private void drop() {
1183 		clear();
1184 		key = INVALID_KEY;
1185 	    }
1186 	}
