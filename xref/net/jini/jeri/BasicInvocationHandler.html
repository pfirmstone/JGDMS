71  /**
72   * A basic implementation of the <code>InvocationHandler</code> interface.
73   * This invocation handler implements Java(TM) Remote Method Invocation
74   * (Java RMI) call semantics when handling
75   * a remote invocation to a remote object.
76   *
77   * <p><code>BasicInvocationHandler</code> instances contain an
78   * <code>ObjectEndpoint</code>, optional client constraints, and
79   * optional server constraints.  The client and server constraints
80   * control the handling of remote methods, and they are represented as
81   * {@link MethodConstraints} objects that map remote methods to
82   * corresponding per-method constraints.
83   *
84   * <p>Invocation requests sent via the {@link #invoke invoke} method use
85   * the protocol defined in that method.  This invocation handler also
86   * assumes that the return value conforms to the protocol outlined in the
87   * {@link BasicInvocationDispatcher#dispatch
88   * BasicInvocationDispatcher.dispatch} method.
89   *
90   * @author	Sun Microsystems, Inc.
91   * @see		BasicInvocationDispatcher
92   * @since 2.0
93   *
94   * 
95   *
96   * This implementation's {@link #invoke invoke} method throws {@link
97   * IllegalArgumentException} if a remote invocation is to be made and
98   * the <code>proxy</code> argument is an instance of an interface
99   * whose binary name is
100  * <code>javax.management.MBeanServerConnection</code> or any of the
101  * names produced by the following procedure:
102  *
103  * <blockquote>
104  *
105  * For each resource named
106  * <code>org/apache/river/proxy/resources/InvocationHandler.moreProhibitedProxyInterfaces</code>
107  * that is visible to the system class loader, the contents of the
108  * resource are parsed as UTF-8 text to produce a list of interface
109  * names.  The resource must contain a list of binary names of
110  * interfaces, one per line.  Space and tab characters surrounding
111  * each name, as well as blank lines, are ignored.  The comment
112  * character is <tt>'#'</tt>; on each line, all characters starting
113  * with the first comment character are ignored.
114  *
115  * </blockquote>
116  *
117  * <p>This implementation uses the {@link Logger} named
118  * <code>net.jini.jeri.BasicInvocationHandler</code> to log
119  * information at the following levels:
120  *
121  * <table summary="Describes what is logged by BasicInvocationHandler at
122  *        various logging levels" border=1 cellpadding=5>
123  *
124  * <tr> <th> Level <th> Description
125  *
126  * <tr> <td> {@link Levels#FAILED FAILED} <td> exception thrown from final
127  * attempt to communicate a remote call
128  * 
129  * <tr> <td> {@link Levels#HANDLED HANDLED} <td> exception caught in
130  * attempt to communicate a remote call
131  *
132  * <tr> <td> {@link Level#FINE FINE} <td> remote method being invoked
133  *
134  * <tr> <td> {@link Level#FINE FINE} <td> successful return of remote call
135  *
136  * <tr> <td> {@link Level#FINEST FINEST} <td> more detailed information on
137  * the above (for example, actual argument and return values) 
138  *
139  * </table>
140  **/

146     // Allowing atomic deserialization of this class may allow
147     // an attacker to deserialize a reflective proxy that uses
148     // standard Java serialization via an atomic stream, then use
149     // that proxy as an attack vector.  So this class isn't annotated with
150     // @AtomicSerial for that reason.
151     // A permission check cannot be used to controll access as there are no
152     // ProtectionDomain's on the stack to represent the untrusted principal
153     // of the serialized input.

155     // AtomicInvocationHander to utilise, these invariants belong

247 	Object clientConstraints = arg.get("clientConstraints", null);
248 	Object serverConstraints = arg.get("serverConstraints", null);
249 	if (clientConstraints != null && !(clientConstraints instanceof MethodConstraints)){
250 	    throw new InvalidObjectException("clientConstraints must be an instance of MethodConstraints");
251 	}
252 	if (serverConstraints != null && !(serverConstraints instanceof MethodConstraints)){
253 	    throw new InvalidObjectException("serverConstraints must be an instance of MethodConstraints");
254 	}
255 	return true;
256     }

296     /**
297      * Processes a method invocation made on the encapsulating
298      * proxy instance, <code>proxy</code>, and returns the result.
299      * This method is invoked when a method is invoked on a proxy
300      * instance that this handler is associated with.
301      *
302      * <p><code>BasicInvocationHandler</code> implements this method
303      * as follows:
304      *
305      * <p>If <code>method</code> is one of the following methods, it
306      * is processed as described below:
307      *
308      * <ul>
309      *
310      * <li>{@link Object#hashCode Object.hashCode}: Returns the hash
311      * code value for the proxy.
312      *
313      * <li>{@link Object#equals Object.equals}: Returns
314      * <code>true</code> if the argument (<code>args[0]</code>) is an
315      * instance of a dynamic proxy class that implements the same
316      * ordered set of interfaces as <code>proxy</code> and this
317      * invocation handler is equal to the invocation handler of that
318      * argument, and returns <code>false</code> otherwise.
319      *
320      * <li>{@link Object#toString Object.toString}: Returns a string
321      * representation of the proxy.
322      *
323      * <li>{@link RemoteMethodControl#setConstraints
324      * RemoteMethodControl.setConstraints}: Returns a new proxy
325      * instance of the same class as <code>proxy</code> containing a
326      * new invocation handler with the specified new client
327      * constraints.  The new invocation handler is created by invoking
328      * the {@link #setClientConstraints setClientConstraints} method
329      * of this object with the specified client constraints
330      * (<code>args[0]</code>).  An exception is thrown if
331      * <code>proxy</code> is not an instance of a dynamic proxy class
332      * containing this invocation handler.
333      *
334      * <li>{@link RemoteMethodControl#getConstraints
335      * RemoteMethodControl.getConstraints}: Returns this
336      * <code>BasicInvocationHandler</code>'s client constraints.
337      *
338      * <li>{@link TrustEquivalence#checkTrustEquivalence
339      * TrustEquivalence.checkTrustEquivalence}: Returns
340      * <code>true</code> if the argument (<code>args[0]</code>) is an
341      * instance of a dynamic proxy class that implements the same
342      * ordered set of interfaces as <code>proxy</code> and invoking
343      * the {@link #checkTrustEquivalence checkTrustEquivalence} method
344      * of this object with the invocation handler of that argument
345      * returns <code>true</code>, and returns <code>false</code>
346      * otherwise.
347      *
348      * </ul>
349      *
350      * <p>Otherwise, a remote call is made as follows:
351      *
352      * <p>The object endpoint's {@link ObjectEndpoint#newCall newCall}
353      * method is invoked to obtain an {@link OutboundRequestIterator},
354      * passing constraints obtained by combining the client and server
355      * constraints for the specified remote method and making them
356      * absolute.  If the returned iterator's {@link
357      * OutboundRequestIterator#hasNext hasNext} method returns
358      * <code>false</code>, then this method throws a {@link
359      * ConnectIOException}.  Otherwise, the iterator is used to make
360      * one or more attempts to communicate the remote call.  Each
361      * attempt proceeds as follows:
362      *
363      * <blockquote>
364      *
365      * The iterator's {@link OutboundRequestIterator#next next} method
366      * is invoked to obtain an {@link OutboundRequest OutboundRequest}
367      * for the current attempt.  If <code>next</code> returns
368      * normally, the request's {@link
369      * OutboundRequest#getUnfulfilledConstraints
370      * getUnfulfilledConstraints} method is invoked, and if the
371      * returned requirements or preferences include {@link
372      * Integrity#YES Integrity.YES}, object integrity is enforced for
373      * the current remote call attempt.  If the returned requirements
374      * include any constraint other than an {@link Integrity}
375      * constraint, an {@link UnsupportedConstraintException} is
376      * generated and, as described below, wrapped and handled like any
377      * other <code>Exception</code> thrown from a remote call attempt.
378      * Otherwise, the marshalling of the remote call proceeds with the
379      * following steps in order:
380      *
381      * <ul>
382      *
383      * <li>A byte of value <code>0x00</code> is written to the request
384      * output stream of the <code>OutboundRequest</code> to indicate
385      * the version of the marshalling protocol being used.
386      *
387      * <li>A byte is written to specify object integrity enforcement:
388      * <code>0x01</code> if object integrity is being enforced for
389      * this remote call attempt, and <code>0x00</code> otherwise.
390      *
391      * <li>A client context collection is created containing an {@link
392      * IntegrityEnforcement} element that reflects whether or not
393      * object integrity is being enforced for this remote call
394      * attempt.
395      *
396      * <li>The {@link #createMarshalOutputStream
397      * createMarshalOutputStream} method is invoked, passing
398      * <code>proxy</code>, <code>method</code>, the
399      * <code>OutboundRequest</code>, and the client context, to create
400      * the marshal output stream for marshalling the remote call.
401      *
402      * <li>The {@link #marshalMethod marshalMethod} method of this
403      * invocation handler is invoked with <code>proxy</code>,
404      * <code>method</code>, the marshal output stream, and the client
405      * context.
406      *
407      * <li>The {@link #marshalArguments marshalArguments} method of
408      * this invocation handler is invoked with <code>proxy</code>,
409      * <code>method</code>, <code>args</code>, the marshal output
410      * stream, and the client context.
411      *
412      * <li>The marshal output stream is closed.
413      *
414      * </ul>
415      *
416      * <p>Then the object endpoint's {@link ObjectEndpoint#executeCall
417      * executeCall} method is invoked with the
418      * <code>OutboundRequest</code>.  If <code>executeCall</code>
419      * returns a <code>RemoteException</code>, then this method throws
420      * that exception (and thus the remote call attempt iteration
421      * terminates).  If <code>executeCall</code> returns
422      * <code>null</code>, then the unmarshalling of the call response
423      * proceeds as follows:
424      *
425      * <p>A byte is read from the response input stream of the
426      * <code>OutboundRequest</code>:
427      *
428      * <ul>
429      *
430      * <li>If the byte is <code>0x00</code>, indicating a marshalling
431      * protocol version mismatch, a {@link ProtocolException} is
432      * generated and, as described below, wrapped and handled like any
433      * other <code>Exception</code> thrown from a remote call attempt.
434      *
435      * <li>If the byte is <code>0x01</code>, indicating a normal
436      * return, the {@link #createMarshalInputStream
437      * createMarshalInputStream} method is invoked, passing
438      * <code>proxy</code>, <code>method</code>, the
439      * <code>OutboundRequest</code>, a <code>boolean</code> indicating
440      * whether or not object integrity is being enforced, and the
441      * client context, to create the marshal input stream for
442      * unmarshalling the response, and the {@link #unmarshalReturn
443      * unmarshalReturn} method of this invocation handler is invoked
444      * with <code>proxy</code>, <code>method</code>, the marshal input
445      * stream, and the client context.  This method returns the value
446      * returned by <code>unmarshalReturn</code> (and thus the remote
447      * call attempt iteration terminates).
448      *
449      * <li>If the byte is <code>0x02</code>, indicating an exceptional
450      * return, a marshal input stream is created by calling the
451      * <code>createMarshalInputStream</code> method as described for
452      * the previous case, and the {@link #unmarshalThrow
453      * unmarshalThrow} method of this invocation handler is invoked
454      * with <code>proxy</code>, <code>method</code>, the marshal input
455      * stream, and the client context.  This method throws the
456      * exception returned by <code>unmarshalThrow</code> (and thus the
457      * remote call attempt iteration terminates).
458      *
459      * <li>If the byte is any other value, a
460      * <code>ProtocolException</code> is generated and, as described
461      * below, wrapped and handled like any other
462      * <code>Exception</code> thrown from a remote call attempt.
463      *
464      * </ul>
465      *
466      * <p>If an <code>IOException</code> is thrown during the attempt
467      * to communicate the remote call, then it is wrapped in a
468      * <code>RemoteException</code> as follows:
469      *
470      * <ul>
471      *
472      * <li>If <code>marshalMethod</code> was not invoked for this
473      * attempt, or if an invocation of {@link
474      * OutboundRequest#getDeliveryStatus getDeliveryStatus} on the
475      * <code>OutboundRequest</code> returns <code>false</code>, or if
476      * a marshalling protocol version mismatch was detected, then
477      *
478      * <ul>
479      *
480      * <li>if the <code>IOException</code> is a {@link
481      * java.net.UnknownHostException java.net.UnknownHostException},
482      * it is wrapped in a {@link java.rmi.UnknownHostException
483      * java.rmi.UnknownHostException};
484      *
485      * <li>if it is a {@link java.net.ConnectException
486      * java.net.ConnectException}, it is wrapped in a {@link
487      * java.rmi.ConnectException java.rmi.ConnectException};
488      *
489      * <li>if it is any other <code>IOException</code>, it is wrapped
490      * in a {@link ConnectIOException}.
491      *
492      * </ul>
493      *
494      * <li>Otherwise, if <code>executeCall</code> was not invoked for
495      * this attempt, the <code>IOException</code> is wrapped in a
496      * {@link MarshalException}, and if <code>executeCall</code> was
497      * invoked, it is wrapped in an {@link UnmarshalException}.
498      *
499      * </ul>
500      *
501      * <p>If a {@link ClassNotFoundException} is thrown
502      * during the unmarshalling, then it is wrapped in an {@link
503      * UnmarshalException}.
504      *
505      * <p>In all cases, either the request output stream and the
506      * response input stream will be closed or the
507      * <code>OutboundRequest</code> will be aborted before this
508      * attempt completes.
509      *
510      * </blockquote>
511      *
512      * <p>If an attempt to communicate the remote call throws an
513      * <code>Exception</code> (other than an exception returned by
514      * <code>executeCall</code> or <code>unmarshalThrow</code>, which
515      * terminates the remote call attempt iteration), then if
516      * <code>marshalMethod</code> was not invoked for the attempt or
517      * if an invocation of <code>getDeliveryStatus</code> on the
518      * <code>OutboundRequest</code> returns <code>false</code>, then
519      * if the iterator's <code>hasNext</code> method returns
520      * <code>true</code>, then another attempt is made.  Otherwise,
521      * this method throws the <code>Exception</code> thrown by the
522      * last attempt (possibly wrapped as described above).
523      *
524      * <p>Note that invoking a remote method on a remote object via this
525      * invoke method preserves "at-most-once" call semantics.  At-most-once
526      * call semantics guarantees that the remote call will either a) not
527      * execute, b) partially execute, or c) execute exactly once at the remote
528      * site.  With Java RMI's at-most-once call semantics, arguments may be
529      * marshalled more than once for a given remote call.
530      *
531      * <p>A subclass can override this method to handle the methods of
532      * any additional non-remote interfaces implemented by the proxy
533      * or to otherwise control invocation handling behavior.
534      *
535      * <p>The semantics of this method are unspecified if the
536      * arguments could not have been produced by an instance of some
537      * valid dynamic proxy class containing this invocation handler.
538      * This method throws {@link IllegalArgumentException} if
539      * <code>proxy</code> is an instance of
540      * <code>InvocationHandler</code> or, if a remote call is to be
541      * made, any of the superinterfaces of <code>proxy</code>'s class
542      * have a method with the same name and parameter types as
543      * <code>method</code> but that does not declare
544      * <code>RemoteException</code> or a superclass of
545      * <code>RemoteException</code> in its <code>throws</code> clause
546      * (even if such a method is not a member of any of the direct
547      * superinterfaces of <code>proxy</code>'s class because of
548      * overriding).
549      *
550      * @throws	Throwable {@inheritDoc}
551      *
552      * @see	java.lang.reflect.UndeclaredThrowableException
553      **/
554     public Object invoke(Object proxy, Method method, Object[] args)
555 	throws Throwable
556     {
557 	if (proxy instanceof InvocationHandler) {
558 	    throw new IllegalArgumentException(
559 				    "proxy cannot be an invocation handler");
560 	} else if (method.getDeclaringClass() == Object.class) {
561 	    return invokeObjectMethod(proxy, method, args);
562 	} else if (RemoteMethodControl.class.isAssignableFrom(method.getDeclaringClass())//  == RemoteMethodControl.class
563 		) {
564 	    /*
565 	     * REMIND: This optimization (testing the identity of the
566 	     * method's declaring class) fails if the proxy's class
567 	     * implements, instead of RemoteMethodControl directly, a
568 	     * subinterface of RemoteMethodControl that overrides the
569 	     * setConstraints method.  This problem could be fixed by
570 	     * using a more expensive Class.isAssignableFrom check,
571 	     * but even that would fail if the proxy class implements,
572 	     * prior to a subinterface of RemoteMethodControl, another
573 	     * interface that also declares a method with the same
574 	     * signature as RemoteMethodControl.setConstraints (this
575 	     * case does seem less unlikely).  It seems that to cover
576 	     * all cases, we should be testing the signature (name and
577 	     * parameter types) of the method directly.
578 	     */
579 	    return invokeRemoteMethodControlMethod(proxy, method, args);
580 	} else if (TrustEquivalence.class.isAssignableFrom(method.getDeclaringClass())// == TrustEquivalence.class
581 		) {
582 	    /*
583 	     * REMIND: Ditto.
584 	     */
585 	    return invokeTrustEquivalenceMethod(proxy, method, args);
586 	} else {
587 	    return invokeRemoteMethod(proxy, method, args);
588 	}
589     }

591     /**
592      * Handles java.lang.Object methods.
593      **/
594     private Object invokeObjectMethod(Object proxy,
595 				      Method method,
596 				      Object[] args)
597     {
598 	String name = method.getName();
599 
600 	if (name.equals("hashCode")) {
601 	    return Integer.valueOf(hashCode());
602 
603 	} else if (name.equals("equals")) {
604 	    Object obj = args[0];

738     /**
739      * Make one attempt to invoke a remote method.
740      *
741      * If this method returns an object that is not a Failure
742      * instance, then the overall remote invocation should return that
743      * object.  If this method throws an exception, then the overall
744      * remote invocation should throw that exception (no further
745      * retry).
746      *
747      * If this method returns a Failure instance, then this attempt to
748      * invoke the remote method failed due a communication failure.
749      * If the retry flag of the Failure instance is true, then the
750      * failure is safe to retry (in light of "at most once" execution
751      * semantics) and possibly worth retrying.
752      **/
753     private Object invokeRemoteMethodOnce(Object proxy,
754 					  Method method,
755 					  Object[] args,
756 					  OutboundRequestIterator iter,
757 					  InvocationConstraints constraints)
758 	throws Throwable
759     {
760 	/*
761 	 * Initiate remote call request.
762 	 */
763 	OutboundRequest request;
764 	try {
765 	    request = iter.next();
766 	} catch (Exception e) {
767 	    if (e instanceof IOException) {
768 		e = wrapSafeIOException((IOException) e, oe);
769 	    }
770 	    return new Failure(e, true);
771 	}
772 
773 	/*
774 	 * Marshal method and arguments.
775 	 */
776 	boolean ok = false;
777 	boolean integrity = false;
778 	boolean atomicValidation = false;
779 	boolean wroteMethod = false;
780 	Collection context;
781 	try {
782 	    /*
783 	     * Check the unfulfilled constraints.  If the unfulfilled
784 	     * requirements include Integrity.YES, then we must verify
785 	     * codebase integrity at this level.  If there are any
786 	     * non-Integrity unfulfilled requirements, we cannot
787 	     * satisfy them, so this request attempt must fail.
788 	     */
789 	    InvocationConstraints unfulfilled =
790 		request.getUnfulfilledConstraints();
791 	    for (Iterator i = unfulfilled.requirements().iterator();
792 		 i.hasNext();)
793 	    {
794 		InvocationConstraint c = (InvocationConstraint) i.next();
795 		if (c == Integrity.YES) {
796 		    integrity = true;
797 		} else if ( c == AtomicInputValidation.YES){
798 		    atomicValidation = true;
799 		} else if (!(c instanceof Integrity) && !(c instanceof AtomicInputValidation)) {
800 		    throw new UnsupportedConstraintException(
801 			"cannot satisfy unfulfilled constraint: " + c);
802 		}
803 		// REMIND: support ConstraintAlternatives containing Integrity?
804 	    }
805 
806 	    /*
807 	     * Even if Integrity.YES wasn't a requirement, we will
808 	     * satisfy a preference for it.
809 	     */
810 	    if (!integrity) {
811 		for (Iterator i = unfulfilled.preferences().iterator();
812 		     i.hasNext();)
813 		{
814 		    InvocationConstraint c = (InvocationConstraint) i.next();
815 		    if (c == Integrity.YES) {
816 			integrity = true;
817 //			break;	// no need to examine preferences further
818 		    } else if (c == AtomicInputValidation.YES){
819 			atomicValidation = true;
820 		    }
821 		    // NYI: support ConstraintAlternatives containing Integrity
822 		}
823 	    }
824 
825 	    OutputStream ros = request.getRequestOutputStream();
826 	    // Use new protocol version if atomicValidation is required or preferred.
827 	    if (atomicValidation){
828 		ros.write(0x01);			// marshalling protocol version
829 		ros.write(integrity ? 0x01 : 0x00);	// integrity
830 		ros.write(atomicValidation ? 0x01 : 0x00);	// atomicValidation
831 	    } else { // Use old version for compatibility
832 	    ros.write(0x00);			// marshalling protocol version
833 	    ros.write(integrity ? 0x01 : 0x00);	// integrity
834 	    }
835 	    
836 	    context = new ArrayList(3);
837 	    if (clientConstraints != null) context.add(clientConstraints);
838 	    
839 	    Util.populateContext(context, integrity, atomicValidation);
840 
841 
842 	    ObjectOutputStream out =
843 		createMarshalOutputStream(proxy, method, request, context);
844 
845 	    wroteMethod = true;
846 	    marshalMethod(proxy, method, out, context);
847 
848 	    args = (args == null) ? new Object[0] : args;
849 	    marshalArguments(proxy, method, args, out, context);
850 
851 	    out.close();
852 	    ok = true;
853 	} catch (IOException e){
854 	    if (wroteMethod && request.getDeliveryStatus()) {
855 		e = new MarshalException("error marshalling arguments", e);
856 	    } else {
857 		e = wrapSafeIOException((IOException) e, oe);
858 	    }
859 	    return new Failure(e, !wroteMethod || !request.getDeliveryStatus());
860 	} catch (RuntimeException e){
861 	    throw e;
862 	} catch (Exception e) {
863 	    return new Failure(e, !wroteMethod || !request.getDeliveryStatus());
864 	} finally {
865 	    if (!ok) {
866 		request.abort();
867 	    }
868 	}
869 
870 	/*
871 	 * Execute call and unmarshal return value or exception.
872 	 */
873 	ok = false;
874 	boolean versionMismatch = false;
875 	Object returnValue = null;
876 	Throwable throwable = null;
877 	try {
878 	    throwable = oe.executeCall(request);
879 	    if (throwable == null) {
880 		InputStream ris = request.getResponseInputStream();
881 		
882 		int responseCode = ris.read();
883 		if (responseCode == -1) {
884 		    throw new EOFException("connection closed by server");
885 		} else if (responseCode == 0x00) {
886 		    versionMismatch = true;
887 		    throw new ProtocolException(
888 			"marshalling protocol version mismatch");
889 		}
890 		
891 		ObjectInputStream in =
892 		    createMarshalInputStream(proxy, method, request,
893 					     integrity, context);
894 		
895 		switch (responseCode) {
896 		case 0x01:
897 		    returnValue = unmarshalReturn(proxy, method, in, context);
898 		    if (logger.isLoggable(Level.FINE)) {
899 			logReturn(method, returnValue);
900 		    }
901 		    break;
902 
903 		case 0x02:
904 		    throwable = unmarshalThrow(proxy, method, in, context);
905 		    break;
906 
907 		default:
908 		    throw new ProtocolException(
909 			"invalid response code " + responseCode);
910 		}
911 
912 		in.close();
913 	    }
914 	    ok = true;
915 	} catch (Exception e) {
916 	    boolean retry = false;
917 	    if (e instanceof IOException) {
918 		if (!versionMismatch && request.getDeliveryStatus()) {
919 		    e = new UnmarshalException(
920 			 "exception unmarshalling response", e);
921 		} else {
922 		    e = wrapSafeIOException((IOException) e, oe);
923 		    retry = !versionMismatch;
924 		}
925 	    } else if (e instanceof ClassNotFoundException) {
926 		e = new UnmarshalException("error unmarshalling response", e);
927 	    }
928 	    return new Failure(e,
929 		!versionMismatch || !request.getDeliveryStatus());
930 	} finally {
931 	    if (!ok) {
932 		request.abort();
933 	    }
934 	}
935 
936 	/*
937 	 * Throw exception or return the return value.
938 	 */
939 	if (throwable != null) {
940 	    if (logger.isLoggable(Levels.FAILED)) {
941 		logThrow(Levels.FAILED, method, throwable, true);
942 	    }
943 	    throw throwable;
944 	} else {
945 	    return returnValue;
946 	}
947     }

1306     /**
1307      * Unmarshals the return value for the specified remote method from the
1308      * incoming response stream, <code>in</code>.  In the case that a value is
1309      * returned from the invocation on the remote object, this method is
1310      * called to unmarshal that return value.
1311      *
1312      * <p><code>BasicInvocationHandler</code> implements this method
1313      * as follows:
1314      *
1315      * <p>If the return type of the method is void, then no return value is
1316      * read from the stream and <code>null</code> is returned.  If the return
1317      * type is a primitive type, then the primitive value is read from the
1318      * stream (for example, if the type is <code>int.class</code>, then the

1327      *
1328      * @param	proxy the proxy instance that the method was invoked on
1329      * @param   method the <code>Method</code> instance corresponding
1330      *          to the interface method invoked on the proxy
1331      *          instance.  The declaring class of the
1332      *          <code>Method</code> object will be the interface that
1333      *          the method was declared in, which may be a
1334      *          superinterface of the proxy interface that the proxy
1335      *          class inherits the method through.
1336      * @param	in the incoming result stream for the remote call
1337      * @param	context the client context
1338      * @return  the unmarshalled return value of the method invocation on 
1339      *		the proxy instance.  If the declared return value of the
1340      *          interface method is a primitive type, then the
1341      *          value returned by <code>invoke</code> will be an
1342      *          instance of the corresponding primitive wrapper
1343      *          class; otherwise, it will be a type assignable to
1344      *		the declared return type.
1345      * @throws	IOException if an I/O exception occurs
1346      * @throws  ClassNotFoundException if a class could not be found during
1347      *          unmarshalling
1348      * @throws	NullPointerException if any argument is <code>null</code>
1349      **/
1350     protected Object unmarshalReturn(Object proxy,
1351 				     Method method,
1352 				     ObjectInputStream in,
1353 				     Collection context)
1354 	throws IOException, ClassNotFoundException
1355     {
1356 	if (proxy == null || in == null || context == null) {
1357 	    throw new NullPointerException();
1358 	}
1359 	Class returnType = method.getReturnType();
1360 	Object returnValue = null;
1361 	if (returnType != void.class) {
1362 	    returnValue = Util.unmarshalValue(returnType, in);
1363 	}
1364 	return returnValue;

1367     /**
1368      * Unmarshals the throwable for the specified remote method from the
1369      * incoming response stream, <code>in</code>, and returns the result.  In
1370      * the case that an exception was thrown as a result of the
1371      * invocation on the remote object, this method is called to unmarshal
1372      * that throwable.
1373      *
1374      * <p><code>BasicInvocationHandler</code> implements this method
1375      * to return the throwable unmarshalled from the stream using the
1376      * <code>readObject</code> method.  If the unmarshalled throwable
1377      * is a checked exception that is not assignable to any exception
1378      * in the <code>throws</code> clause of the method implemented by the
1379      * <code>proxy</code>'s class, then: if there is no public member
1380      * method of <code>proxy</code>'s class with the same name and
1381      * parameter types as <code>method</code> an {@link
1382      * IllegalArgumentException} is thrown, otherwise that exception
1383      * is wrapped in an {@link UnexpectedException} and the wrapped
1384      * exception is returned.
1385      *
1386      * <p>A subclass can override this method to unmarshal the return
1387      * value in an alternative context, perform post-processing on the
1388      * return value, unmarshal additional implicit data, or otherwise
1389      * control how the return value is unmarshalled.
1390      *
1391      * @param	proxy the proxy instance that the method was invoked on
1392      * @param   method the <code>Method</code> instance corresponding
1393      *          to the interface method invoked on the proxy
1394      *          instance.  The declaring class of the
1395      *          <code>Method</code> object will be the interface that
1396      *          the method was declared in, which may be a
1397      *          superinterface of the proxy interface that the proxy
1398      *          class inherits the method through.
1399      * @param	context the client context
1400      * @param	in the incoming result stream for the remote call
1401      * @return	the unmarshalled exception to throw from the method
1402      *		invocation on the proxy instance.  The exception's type
1403      *		must be assignable either to any of the exception types
1404      *		declared in the <code>throws</code> clause of the interface
1405      *		method or to the unchecked exception types
1406      *		<code>java.lang.RuntimeException</code> or
1407      *		<code>java.lang.Error</code>.
1408      * @throws	IOException if an I/O exception occurs
1409      * @throws  ClassNotFoundException if a class could not be found during
1410      *          unmarshalling
1411      * @throws	NullPointerException if any argument is <code>null</code>
1412      **/
1413     protected Throwable unmarshalThrow(Object proxy,
1414 				       Method method,
1415 				       ObjectInputStream in,
1416 				       Collection context)
1417 	throws IOException, ClassNotFoundException
1418     {
1419 	if (proxy == null || method == null || context == null) {
1420 	    throw new NullPointerException();
1421 	}
1422 	Throwable t = (Throwable) in.readObject();
1423 	Util.exceptionReceivedFromServer(t);
1424 	if (!(t instanceof RuntimeException || t instanceof Error)) {
1425 	    Class cl = proxy.getClass();
1426 	    try {
1427 		method = cl.getMethod(method.getName(),
1428 				      method.getParameterTypes());
1429 	    } catch (NoSuchMethodException e) {
1430 		throw (IllegalArgumentException)
1431 		    new IllegalArgumentException().initCause(e);
1432 	    }
1433 	    Class[] exTypes = method.getExceptionTypes();
1434 	    Class thrownType = t.getClass();
1435 	    for (int i = 0; i < exTypes.length; i++) {
1436 		if (exTypes[i].isAssignableFrom(thrownType)) {
1437 		    return t;
1438 		}
1439 	    }
1440 	    UnexpectedException wrapper =
1441 		new UnexpectedException("unexpected exception", (Exception) t);
1442 	    t = wrapper;
1443 	}
1444 	return t;
1445     }
