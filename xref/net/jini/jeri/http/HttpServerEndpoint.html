57  /**
58   * An implementation of the {@link ServerEndpoint} abstraction that
59   * uses HTTP messages sent over TCP sockets (instances of {@link
60   * ServerSocket}) for the underlying communication mechanism.
61   *
62   * <p><code>HttpServerEndpoint</code> instances contain a host name
63   * and a TCP port number, as well as an optional {@link
64   * ServerSocketFactory} for customizing the type of
65   * <code>ServerSocket</code> to use and an optional {@link
66   * SocketFactory} for customizing the type of {@link Socket} that
67   * client endpoints will use.  The port number is the local TCP port
68   * to bind to when listening for incoming socket connections.  If the
69   * port number is zero, then each listen operation will bind to a free
70   * (non-zero) port, which will be the port number contained in the
71   * resulting {@link HttpEndpoint}.  The host name contained in an
72   * <code>HttpServerEndpoint</code> controls the host name that will be
73   * contained in the <code>HttpEndpoint</code> instances produced when
74   * {@link #enumerateListenEndpoints enumerateListenEndpoints} is
75   * invoked to listen on the <code>HttpServerEndpoint</code> (this host
76   * name does not affect the behavior of listen operations themselves).
77   * If the host name in an <code>HttpServerEndpoint</code> is
78   * <code>null</code>, then the host name in the
79   * <code>HttpEndpoint</code> instances that it produces will be the IP
80   * address string obtained from {@link InetAddress#getLocalHost
81   * InetAddress.getLocalHost} when {@link #enumerateListenEndpoints
82   * enumerateListenEndpoints} is invoked.
83   *
84   * <p><code>HttpServerEndpoint</code> instances map incoming HTTP
85   * messages to requests; when possible, underlying TCP connections are
86   * persisted to accommodate multiple non-overlapping incoming
87   * requests.  Inbound request data is received as the
88   * <code>entity-body</code> of an HTTP POST request; outbound response
89   * data is sent as the <code>entity-body</code> of the corresponding
90   * HTTP return message.  For information on HTTP, refer to <a
91   * href="http://www.ietf.org/rfc/rfc2616.txt">RFC 2616</a>.  Note that
92   * providing socket factories that produce SSL sockets does not result
93   * in endpoints that are fully HTTPS capable.
94   *
95   * <p>A <code>ServerSocketFactory</code> used with an
96   * <code>HttpServerEndpoint</code> must implement {@link Object#equals
97   * Object.equals} to obey the guidelines that are specified for
98   * <code>equals</code> methods of {@link
99   * net.jini.jeri.ServerEndpoint.ListenEndpoint ListenEndpoint}
100  * instances.  A <code>SocketFactory</code> used with a
101  * <code>HttpServerEndpoint</code> should be serializable and must
102  * implement <code>Object.equals</code> to obey the guidelines that
103  * are specified for <code>equals</code> methods of {@link Endpoint}
104  * instances.
105  *
106  * 
107  * @see HttpEndpoint
108  * @since 2.0
109  **/
110 public final class HttpServerEndpoint implements ServerEndpoint {

333     /**
334      * Passes the {@link net.jini.jeri.ServerEndpoint.ListenEndpoint
335      * ListenEndpoint} for this <code>HttpServerEndpoint</code> to
336      * <code>listenContext</code>, which will ensure an active listen
337      * operation on the endpoint, and returns an
338      * <code>HttpEndpoint</code> instance corresponding to the listen
339      * operation chosen by <code>listenContext</code>.
340      *
341      * <p>If this <code>HttpServerEndpoint</code>'s host name is not
342      * <code>null</code>, then the returned <code>HttpEndpoint</code>
343      * will contain that host name.  If this
344      * <code>HttpServerEndpoint</code>'s host name is
345      * <code>null</code>, then this method invokes {@link
346      * InetAddress#getLocalHost InetAddress.getLocalHost} to obtain an
347      * <code>InetAddress</code> for the local host.  If
348      * <code>InetAddress.getLocalHost</code> throws an {@link
349      * UnknownHostException}, this method throws an
350      * <code>UnknownHostException</code>.  The returned
351      * <code>HttpEndpoint</code>'s host name will be the string
352      * returned by invoking {@link InetAddress#getHostAddress
353      * getHostAddress} on that <code>InetAddress</code>.  If there is
354      * a security manager, its {@link
355      * SecurityManager#checkConnect(String,int) checkConnect} method
356      * will be invoked with the string returned by invoking {@link
357      * InetAddress#getHostName getHostName} on that same
358      * <code>InetAddress</code> as the host argument and
359      * <code>-1</code> as the port argument; this could result in a
360      * <code>SecurityException</code>.
361      *
362      * <p>This method invokes <code>addListenEndpoint</code> on
363      * <code>listenContext</code> once, passing a
364      * <code>ListenEndpoint</code> as described below.  If
365      * <code>addListenEndpoint</code> throws an exception, then this
366      * method throws that exception.  Otherwise, this method returns
367      * an <code>HttpEndpoint</code> instance with the host name
368      * described above, the TCP port number bound by the listen
369      * operation represented by the {@link
370      * net.jini.jeri.ServerEndpoint.ListenHandle ListenHandle}
371      * returned by <code>addListenEndpoint</code>, and the same
372      * <code>SocketFactory</code> as this
373      * <code>HttpServerEndpoint</code>.
374      *
375      * <p>The <code>ListenEndpoint</code> passed to
376      * <code>addListenEndpoint</code> represents the TCP port number
377      * and <code>ServerSocketFactory</code> of this
378      * <code>HttpServerEndpoint</code>.  Its methods behave as
379      * follows:
380      *
381      * <p>{@link net.jini.jeri.ServerEndpoint.ListenEndpoint#listen
382      * ListenHandle listen(RequestDispatcher)}:
383      *
384      * <blockquote>
385      *
386      * Listens for requests received on this endpoint's TCP port,
387      * dispatching them to the supplied <code>RequestDispatcher</code>
388      * in the form of {@link InboundRequest} instances.
389      *
390      * <p>When the implementation of this method needs to create a new
391      * <code>ServerSocket</code>, it will do so by invoking one of the
392      * <code>createServerSocket</code> methods that returns a bound
393      * server socket on the contained <code>ServerSocketFactory</code>
394      * if non-<code>null</code>, or it will create a
395      * <code>ServerSocket</code> directly otherwise.
396      *
397      * <p>If there is a security manager, its {@link
398      * SecurityManager#checkListen checkListen} method will be invoked
399      * with this endpoint's TCP port; this could result in a
400      * <code>SecurityException</code>.  Furthermore, before a given
401      * <code>InboundRequest</code> gets dispatched to the supplied
402      * request dispatcher, the security manager's {@link
403      * SecurityManager#checkAccept checkAccept} method must have been
404      * successfully invoked in the security context of this
405      * <code>listen</code> invocation with the remote IP address and
406      * port of the <code>Socket</code> used to receive the request.
407      * The <code>checkPermissions</code> method of the dispatched
408      * <code>InboundRequest</code> also performs this latter security
409      * check.  (Note that in some cases, the implementation may carry
410      * out these security checks indirectly, such as through
411      * invocations of <code>ServerSocket</code>'s constructors or
412      * <code>accept</code> method.)
413      *
414      * <p>Requests will be dispatched in a {@link PrivilegedAction}
415      * wrapped by a {@link SecurityContext} obtained when this method
416      * was invoked, with the {@link AccessControlContext} of that
417      * <code>SecurityContext</code> in effect.
418      *
419      * <p>Dispatched requests will implement {@link
420      * InboundRequest#populateContext populateContext} to populate the
421      * supplied collection with context information representing the
422      * request.
423      *
424      * <p>Throws {@link IOException} if an I/O exception occurs while
425      * performing this operation, such as if the TCP port is already
426      * in use.
427      *
428      * <p>Throws {@link SecurityException} if there is a security
429      * manager and the invocation of its <code>checkListen</code>
430      * method fails.
431      *
432      * <p>Throws {@link NullPointerException} if
433      * <code>requestDispatcher</code> is <code>null</code>
434      *
435      * </blockquote>
436      *
437      * <p>{@link
438      * net.jini.jeri.ServerEndpoint.ListenEndpoint#checkPermissions
439      * void checkPermissions()}:
440      *
441      * <blockquote>
442      *
443      * Verifies that the calling context has all of the security
444      * permissions necessary to listen for requests on this endpoint.
445      *
446      * <p>If there is a security manager, its <code>checkListen</code>
447      * method will be invoked with this endpoint's TCP port; this
448      * could result in a <code>SecurityException</code>.
449      *
450      * <p>Throws {@link SecurityException} if there is a security
451      * manager and the invocation of its <code>checkListen</code>
452      * method fails.
453      *
454      * </blockquote>
455      *
456      * <p>{@link Object#equals boolean equals(Object)}:
457      *
458      * <blockquote>
459      *
460      * Compares the specified object with this
461      * <code>ListenEndpoint</code> for equality.
462      *
463      * <p>This method returns <code>true</code> if and only if
464      *
465      * <ul>
466      *
467      * <li>the specified object is also a <code>ListenEndpoint</code>
468      * produced by an <code>HttpServerEndpoint</code>,
469      *
470      * <li>the port in the specified object is equal to the port in
471      * this object, and
472      *
473      * <li>either this object and the specified object both have no
474      * <code>ServerSocketFactory</code> or the
475      * <code>ServerSocketFactory</code> in the specified object has
476      * the same class and is equal to the one in this object.
477      *
478      * </ul>
479      *
480      * </blockquote>
481      *
482      * @param listenContext the <code>ListenContext</code> to pass
483      * this <code>HttpServerEndpoint</code>'s
484      * <code>ListenEndpoint</code> to
485      *
486      * @return the <code>HttpEndpoint</code> instance for sending
487      * requests to this <code>HttpServerEndpoint</code>'s endpoint
488      * being listened on
489      *
490      * @throws UnknownHostException if this
491      * <code>HttpServerEndpoint</code>'s host name is
492      * <code>null</code> and <code>InetAddress.getLocalHost</code>
493      * throws an <code>UnknownHostException</code>
494      *
495      * @throws IOException if an I/O exception occurs while performing
496      * this operation, such as if the TCP port is already in use
497      *
498      * @throws SecurityException if there is a security manager and
499      * either the invocation of its <code>checkListen</code> method
500      * fails or this <code>HttpServerEndpoint</code>'s host name is
501      * <code>null</code> and the invocation of the security manager's
502      * <code>checkConnect</code> method fails
503      *
504      * @throws IllegalArgumentException {@inheritDoc}
505      *
506      * @throws NullPointerException {@inheritDoc}
507      **/
508     public Endpoint enumerateListenEndpoints(ListenContext listenContext)
509 	throws IOException
510     {
511 	if (listenContext == null) {
512 	    throw new NullPointerException();
513 	}
514 
515 	String localHost = LOCAL_HOST.check(host, this);
516 	
517 	LE listenEndpoint = new LE(); // REMIND: needn't be new?
518 	ListenCookie listenCookie =
519 	    listenContext.addListenEndpoint(listenEndpoint);
520 
521 	if (!(listenCookie instanceof LE.Cookie)) {
522 	    throw new IllegalArgumentException();
523 	}
524 	LE.Cookie cookie = (LE.Cookie) listenCookie;
525 	if (!listenEndpoint.equals(cookie.getLE())) {
526 	    throw new IllegalArgumentException();
527 	}
528 
529 	return HttpEndpoint.getInstance(localHost, cookie.getPort(), sf);
530     }

619 	public ListenHandle listen(RequestDispatcher requestDispatcher)
620 	    throws IOException
621 	{
622 	    if (requestDispatcher == null) {
623 		throw new NullPointerException();
624 	    }
625 
626 	    ServerSocket serverSocket;
627 	    if (ssf != null) {
628 		serverSocket = ssf.createServerSocket(port);
629 	    } else {
630 		serverSocket = new ServerSocket(port);
631 	    }
632 
633 	    if (logger.isLoggable(Level.FINE)) {
634 		logger.log(Level.FINE, 
635 			   (ssf == null ? "created server socket {0}" :
636 			    "created server socket {0} using factory {1}"),
637 			   new Object[] { serverSocket, ssf });
638 	    }
639 
640 	    Cookie cookie = new Cookie(serverSocket.getLocalPort());
641 	    final LH listenHandle = new LH(requestDispatcher, serverSocket,
642 					   Security.getContext(), cookie);
643 	    listenHandle.startAccepting();
644 	    return listenHandle;
645 	}

691     /**
692      * ListenHandle implementation: represents a listen operation.
693      **/
694     private static class LH implements ListenHandle {
695 
696 	private final RequestDispatcher requestDispatcher;
697 	private final ServerSocket serverSocket;
698 	private final SecurityContext context;
699 	private final ListenCookie cookie;
700 
701 	private long acceptFailureTime = 0L;	// local to accept thread
702 	private int acceptFailureCount;		// local to accept thread
703 
704 	private final Object lock = new Object();
705 	private boolean closed = false;
706 	private final Set conns = new HashSet();
707 
708 	LH(RequestDispatcher requestDispatcher,
709 	   ServerSocket serverSocket,
710 	   SecurityContext context,
711 	   ListenCookie cookie)
712 	{
713 	    this.requestDispatcher = requestDispatcher;
714 	    this.serverSocket = serverSocket;
715 	    this.context = context;
716 	    this.cookie = cookie;
717 	}
718 
719 	/**
720 	 * Starts the accept loop.
721 	 **/
722 	void startAccepting() {
723 	    systemThreadPool.execute(new Runnable() {
724                 @Override
725 		public void run() {
726 		    try {
727 			executeAcceptLoop();
728 		    } finally {
729 			/*
730 			 * The accept loop is only started once, so
731 			 * after no more socket accepts will occur,
732 			 * ensure that the server socket is no longer
733 			 * listening.
734 			 */
735 			try {
736 			    serverSocket.close();
737 			} catch (IOException e) {
738 			}
739 		    }
740 		}
741 	    }, toString() + " accept loop");
742 	}
743 
744 	/**
745 	 * Runs the accept loop in the access control context
746 	 * preserved by LE.listen.
747 	 **/
748 	private void executeAcceptLoop() {
749 	    AccessController.doPrivileged(context.wrap(new PrivilegedAction() {
750                 @Override
751 		public Object run() {
752 		    executeAcceptLoop0();
753 		    return null;
754 		}
755 	    }), context.getAccessControlContext());
756 	}
757 
758 	/**
759 	 * Executes the accept loop.
760 	 **/
761 	private void executeAcceptLoop0() {
762 	    while (true) {
763 		Socket socket = null;
764 		try {
765 		    socket = serverSocket.accept();
766 		    if (logger.isLoggable(Level.FINE)) {
767 			logger.log(Level.FINE, 
768 			    "accepted socket {0} from server socket {1}",
769 			    new Object[]{ socket, serverSocket });
770 		    }
771 
772 		    setSocketOptions(socket);
773 
774 		    Connection con = new Connection(socket);
775                     con.start();
776 
777 		} catch (Throwable t) {
778 		    try {
779 			/*
780 			 * If this listen operation has been stopped,
781 			 * then we expect the socket accept to throw
782 			 * an exception, so just terminate normally.
783 			 */
784 			synchronized (lock) {
785 			    if (closed) {
786 				break;
787 			    }
788 			}
789 
790 			try {
791 			    if (logger.isLoggable(Level.WARNING)) {
792 				LogUtil.logThrow(logger, Level.WARNING,
793 				    HttpServerEndpoint.class,
794 				    "executeAcceptLoop",
795 				    "accept loop for {0} throws",
796 				    new Object[] { serverSocket }, t);
797 			    }
798 			} catch (Throwable tt) {
799 			}
800 		    } finally {
801 			/*
802 			 * Always close the accepted socket (if any)
803 			 * if an exception occurs, but only after
804 			 * logging an unexpected exception.
805 			 */
806 			if (socket != null) {
807 			    try {
808 				socket.close();
809 			    } catch (IOException e) {
810 			    }
811 			}
812 		    }
813 		    
814 		    if (!(t instanceof SecurityException)) {
815 			try {
816 			    Object[] snapshot;
817 			    synchronized (lock) {
818 				snapshot = closed ? null : conns.toArray();
819 			    }
820 			    if (snapshot != null) {
821 				for (int i = 0; i < snapshot.length; i++) {
822 				    ((Connection) snapshot[i]).shutdown(false);
823 				}
824 			    }
825                         // TODO: This looks dodgy, if not completely bonkers!
826                         // Review to decide if this is really what we should do?
827 			} catch (OutOfMemoryError e) { 
828 			} catch (Exception e) {
829 			}
830 		    }
831 
832 		    /*
833 		     * A NoClassDefFoundError can occur if no file
834 		     * descriptors are available, in which case this
835 		     * loop should not terminate.
836 		     */
837 		    boolean knownFailure =
838 			t instanceof Exception ||
839 			t instanceof OutOfMemoryError ||
840 			t instanceof NoClassDefFoundError;
841 		    if (knownFailure) {
842 			if (continueAfterAcceptFailure(t)) {
843 			    continue;
844 			} else {
845 			    return;
846 			}
847 		    }
848 
849 		    throw (Error) t;
850 		}
851 	    }
852 	}
853 
854 	/**
855 	 * Stops this listen operation.
856 	 **/
857         @Override
858 	public void close() {
859 	    synchronized (lock) {
860 		if (closed) {
861 		    return;
862 		}
863 		closed = true;
864 	    }
865 
866 	    try {
867 		serverSocket.close();
868 	    } catch (IOException e) {
869 	    }
870 	    if (logger.isLoggable(Level.FINE)) {
871 		logger.log(Level.FINE,
872 			   "closed server socket {0}", serverSocket);
873 	    }
874 
875 	    /*
876 	     * Iterating over conns without synchronization is safe
877 	     * at this point because no other thread will access it
878 	     * without verifying that closed is false in a
879 	     * synchronized block first.
880 	     */
881 	    for (Iterator i = conns.iterator(); i.hasNext(); ) {
882 		((Connection) i.next()).shutdown(true);
883 	    }
884 	}

899 	/**
900 	 * Throttles the accept loop after ServerSocket.accept throws
901 	 * an exception, and decides whether to continue at all.  The
902 	 * current code is borrowed from the JRMP implementation; it
903 	 * always continues, but it delays the loop after bursts of
904 	 * failed accepts.
905 	 **/
906 	private boolean continueAfterAcceptFailure(Throwable t) {
907 	    /*
908 	     * If we get a burst of NFAIL failures in NMSEC milliseconds,
909 	     * then wait for ten seconds.  This is to ensure that individual
910 	     * failures don't cause hiccups, but sustained failures don't
911 	     * hog the CPU in futile accept-fail-retry looping.
912 	     */
913 	    final int NFAIL = 10;
914 	    final int NMSEC = 5000;
915 	    long now = System.currentTimeMillis();
916 	    if (acceptFailureTime == 0L ||
917 		(now - acceptFailureTime) > NMSEC)
918 	    {
919 		// failure time is very old, or this is first failure
920 		acceptFailureTime = now;
921 		acceptFailureCount = 0;
922 	    } else {
923 		// failure window was started recently
924 		acceptFailureCount++;
925 		if (acceptFailureCount >= NFAIL) {
926 		    try {
927 			Thread.sleep(10000);
928 		    } catch (InterruptedException ignore) {
929 		    }
930 		    // no need to reset counter/timer
931 		}
932 	    }
933 	    return true;
934 	}

936 	/**
937 	 * HttpServerConnection subclass.
938 	 **/
939 	private class Connection extends HttpServerConnection {
940 
941 	    private final Socket socket;
942 	    private final Object connLock = new Object();
943 	    private boolean connClosed;
944 
945 	    Connection(Socket socket) throws IOException {
946 		super(socket, requestDispatcher, serverManager);
947 		this.socket = socket;
948 	    }
949             
950             @Override
951             public void start(){
952 		boolean needShutdown = false;
953 		synchronized (lock) {
954 		    if (closed) {
955 			needShutdown = true; // shutdown after releasing lock
956 		    } else {
957 			conns.add(this);
958 		    }
959 		}
960 		if (needShutdown) {
961 		    shutdown(true);
962 		} else {
963 		    super.start();
964 		}
965             }
966 
967             @Override
968 	    public boolean shutdown(boolean force) {
969 		synchronized (connLock) {
970 		    if (connClosed) {
971 			return true;
972 		    }
973 		    connClosed = super.shutdown(force);
974 		    if (!connClosed) {
975 			return false;
976 		    }
977 		}
978 
979 		connTimer.cancelTimeout(this);
980 		synchronized (lock) {
981 		    if (!closed) {	// must not mutate set after closed
982 			conns.remove(this);
983 		    }
984 		}
985 
986 		if (logger.isLoggable(Level.FINE)) {
987 		    logger.log(Level.FINE,
988 			       "shut down connection on socket {0}", socket);
989 		}
990 		return true;
991 	    }
992 
993             @Override
994 	    protected void checkPermissions() {
995 		SecurityManager sm = System.getSecurityManager();
996 		if (sm != null) {
997 		    sm.checkAccept(socket.getInetAddress().getHostAddress(),
998 				   socket.getPort());
999 		}
1000 	    }
1001 
1002             @Override
1003 	    protected InvocationConstraints checkConstraints(
1004 		InvocationConstraints constraints)
1005 		throws UnsupportedConstraintException
1006 	    {
1007 		/*
1008 		 * The transport layer aspects of all constraints
1009 		 * supported by this transport provider are always
1010 		 * satisfied by all requests on the server side, so
1011 		 * this method can have the same static behavior as
1012 		 * ServerCapabilities.checkConstraints does.
1013 		 * (Otherwise, this operation would need to be
1014 		 * parameterized by this connection or the request.)
1015 		 */
1016 		return Constraints.check(constraints, true);
1017 	    }

1019 	    /**
1020 	     * Populates the context collection with information representing
1021 	     * the connection.
1022 	     **/
1023             @Override
1024 	    protected void populateContext(Collection context) {
1025 	        Util.populateContext(context, socket.getInetAddress());
1026 	    }

1028             @Override
1029 	    protected void idle() {
1030 		connTimer.scheduleTimeout(this, false);
1031 	    }
1032 
1033             @Override
1034 	    protected void busy() {

1036 	    }
