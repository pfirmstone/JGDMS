19  package net.jini.jeri.connection;
20  
21  import java.io.IOException;
22  import java.util.Collection;
23  import net.jini.jeri.Endpoint;
24  
25  /**
26   * Represents a remote communication endpoint to establish connections
27   * to.  For example, a TCP-based <code>ConnectionEndpoint</code>
28   * implementation typically contains the remote host address and TCP
29   * port to connect to.
30   *
31   * <p><code>ConnectionEndpoint</code> is implemented by
32   * connection-based {@link Endpoint} implementations that use {@link
33   * ConnectionManager} for managing connections.  A
34   * <code>ConnectionManager</code> is created for a given
35   * <code>ConnectionEndpoint</code>, and it uses that
36   * <code>ConnectionEndpoint</code> to choose an established connection
37   * for a given request or to establish a new connection for a given
38   * request.
39   *
40   * <p>An instance of this interface should implement {@link
41   * Object#equals Object.equals} to obey the guidelines that are
42   * specified for <code>equals</code> methods of {@link Endpoint}
43   * instances.
44   *
45   * @author Sun Microsystems, Inc.
46   * @since 2.0
47   **/
48  public interface ConnectionEndpoint {
49  
50      /**
51       * Returns a new connection that can be used to send a request for
52       * the specified handle.
53       *
54       * <p>The actual network connection might not be completely
55       * established when this method returns; connection establishment
56       * (including any permission checks required) may proceed
57       * asynchronously.
58       *
59       * <p>Either this method or the returned connection must
60       * eventually check and throw a <code>SecurityException</code> if
61       * the calling thread (at the point of the check) does not have
62       * the requisite permissions to send an outbound request for the
63       * specified handle.  If an exception is thrown, data written to
64       * the connection's output stream must not have been transmitted
65       * to the server, and the client's identity must not have been
66       * revealed to the server.
67       *
68       * <p>Either this method or the returned connection must
69       * eventually check and throw an <code>IOException</code> if the
70       * client or server does not have the requisite principals and
71       * credentials to allow the client to send an outbound request for
72       * the specified handle.  If an exception is thrown, data written
73       * to the connection's output stream must not have been
74       * transmitted to the server.
75       *
76       * @param handle the handle for the request
77       *
78       * @return a new connection that can be used to send a request for
79       * the specified handle
80       *
81       * @throws IOException if an I/O exception occurs
82       *
83       * @throws SecurityException if a security exception occurs
84       *
85       * @throws IllegalArgumentException if the specified handle was
86       * not created for use with an equivalent connection endpoint
87       *
88       * @throws NullPointerException if <code>handle</code> is
89       * <code>null</code>
90       **/
91      Connection connect(OutboundRequestHandle handle)
92  	throws IOException;
93  
94      /**
95       * Returns an existing or new connection that can be used to send
96       * a request for the specified handle, or <code>null</code> if a
97       * new connection needs to be created in a way that requires
98       * synchronous I/O.
99       *
100      * <p>This method is passed any existing connections, both active
101      * and idle, that might be suitable for use.  The active
102      * connections have other requests in progress; the idle
103      * connections do not.  All other things being equal, an active
104      * connection should be chosen over an idle one, and an idle
105      * connection should be chosen over creating a new one.  An active
106      * (or idle) connection, however, might be less suitable than an
107      * idle (or new) one for various reasons, such as being too
108      * expensive relative to the constraints that need to be
109      * satisfied.
110      *
111      * <p>This method is permitted to alter the state of an idle
112      * connection (for example, to renegotiate the constraints in
113      * force), but any I/O for that purpose must be performed either
114      * asynchronously or at subsequent I/O operations on the
115      * connection, and it must be completed before any data written to
116      * the connection's output stream is actually transmitted.  An
117      * implementation that performs any such I/O must distinguish it
118      * from I/O performed on the connection's streams.
119      *
120      * <p>This method is permitted to return a new connection, but it
121      * must not perform any synchronous I/O to establish the
122      * connection; such I/O must be performed either asynchronously or
123      * at subsequent I/O operations on the connection.  If synchronous
124      * I/O is required to create a new connection, this method should
125      * return <code>null</code>.
126      *
127      * <p>This method should not assume that the collections are
128      * modifiable and should not assume that their contents will
129      * remain valid after this method returns.
130      *
131      * <p>Either this method or the returned connection must
132      * eventually check and throw a <code>SecurityException</code> if
133      * the calling thread (at the point of the check) does not have
134      * the requisite permissions to send an outbound request for the
135      * specified handle.  If an exception is thrown, data written to
136      * the connection's output stream must not have been transmitted
137      * to the server, and the client's identity must not have been
138      * revealed to the server.
139      *
140      * <p>The returned connection must eventually check and throw an
141      * <code>IOException</code> if the client or server does not have
142      * the requisite principals and credentials to allow the client to
143      * send an outbound request for the specified handle.  If an
144      * exception is thrown, data written to the connection's output
145      * stream must not have been transmitted to the server.
146      *
147      * <p>If an existing active connection is returned, this method
148      * must ensure that the security and credential checks for the
149      * current request will not cause the checks for existing active
150      * requests to unnecessarily fail or to incorrectly succeed, and
151      * vice versa.  Therefore, in practice, this method should only
152      * return an existing active connection if all of the security and
153      * credential checks are made before this method returns.
154      *
155      * <p>Note that a <code>ConnectionManager</code> never makes
156      * concurrent invocations of this method; implementations should
157      * take that into consideration when deciding if and how long this
158      * method should block.
159      *
160      * @param handle the handle for the request
161      *
162      * @param active the connections with requests in progress
163      *
164      * @param idle the connections with no requests in progress
165      *
166      * @return a connection that can be used to send an outbound
167      * request for the specified handle, or <code>null</code>
168      *
169      * @throws SecurityException if a security exception occurs
170      *
171      * @throws IllegalArgumentException if the specified handle was
172      * not created for use with an equivalent connection endpoint, or
173      * a connection in either collection was not created by an
174      * equivalent connection endpoint
175      *
176      * @throws NullPointerException if any argument is
177      * <code>null</code> or any element of either collection is
178      * <code>null</code>
179      **/
180     Connection connect(OutboundRequestHandle handle,
181 		       Collection active,
182 		       Collection idle);
183 }
