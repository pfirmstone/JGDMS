19  package net.jini.jeri.ssl;
20  
21  import java.io.IOException;
22  import java.io.Serializable;
23  import net.jini.core.constraint.Confidentiality;
24  import net.jini.core.constraint.InvocationConstraint;
25  import org.apache.river.api.io.AtomicSerial;
26  import org.apache.river.api.io.AtomicSerial.GetArg;
27  
28  /**
29   * Represents a constraint that, if confidentiality of message contents is
30   * ensured, the specified strength of confidentiality be used. <p>
31   *
32   * The use of an instance of this constraint does not directly imply a {@link
33   * Confidentiality#YES} constraint; that must be specified separately to ensure
34   * that confidentiality is actually ensured. <p>
35   *
36   * Serialization for this class is guaranteed to produce instances that are
37   * comparable with <code>==</code>.  For future security, this may be converted
38   * to an enum, which is a breaking change, in order to honor the preceding statement.<p>
39   *
40   * This constraint is supported by the endpoints defined in this package. <p>
41   *
42   * The {@link SslTrustVerifier} trust verifier may be used for establishing
43   * trust in remote proxies that use instances of this class.
44   *
45   * 
46   * @see SslEndpoint
47   * @see SslServerEndpoint
48   * @see HttpsEndpoint
49   * @see HttpsServerEndpoint
50   * @see SslTrustVerifier
51   * @since 2.0
52   */
53  @AtomicSerial
54  public final class ConfidentialityStrength
55      implements InvocationConstraint, Serializable
56  {
57      /* -- Fields -- */
58  
59      private static final long serialVersionUID = -5413316999614306469L;
60  
61      /**
62       * RFC 7525 Current best practice, if confidentiality of message contents 
63       * is ensured, then use strong confidentiality for message contents. <p>
64       * STRONG confidentiality isn't possible, unless both client and server
65       * are authenticated.
66       * <p>
67       * Guidance:
68       * <ul>
69       * <li> Keys used for strong encryption must not be shared by WEAK encryption connections.
70       * <li> Public Keys lengths used must follow the guidance of RFC 7525.
71       * <li> DHE_DSS and ECDHE_ECDSA key exchanges are allowed in addition to
72       * those recommended in RFC 7525.
73       * </ul>
74       *
75       * For the endpoints in this package, this constraint is supported by
76       * cipher suites with the following cipher algorithms:
77       *
78       * <ul>
79       * <li> AES_128_GCM
80       * <li> AES_256_GCM
81       * </ul>
82       * 
83       * Key exchange is limited to the following ephemeral protocols with forward secrecy.
84       * 
85       * <ul>
86       * <li> ECDHE_RSA
87       * <li> DHE_RSA
88       * <li> DHE_DSS
89       * <li> ECDHE_ECDSA
90       * </ul>
91       */
92      public static final ConfidentialityStrength STRONG =
93  	new ConfidentialityStrength(true);
94  
95      /**
96       * If confidentiality of message contents is ensured, then use weak
97       * confidentiality for message contents. <p>
98       * All protocols allowed by WEAK are known to be vulnerable to attack.
99       * <p>
100      * Note that in all previous versions of Apache River and Jini, that the
101      * following protocols are considered STRONG.
102      * <p>
103      * For the endpoints in this package, this constraint is supported by
104      * cipher suites with the following cipher algorithms:
105      *
106      * <ul>
107      * <li> AES_128_CBC
108      * <li> AES_256_CBC
109      * <li> 3DES_EDE_CBC
110      * <li> RC4_128
111      * </ul>
112      */
113     public static final ConfidentialityStrength WEAK =
114 	new ConfidentialityStrength(false);
115 
116     /**
117      * <code>true</code> for <code>STRONG</code>, <code>false</code> for
118      * <code>WEAK</code>
119      *
120      * @serial
121      */
122     private final boolean value;
123 
124     /* -- Methods -- */
125     
126     public ConfidentialityStrength(GetArg arg) throws IOException{
127 	this(arg.get("value", true));
128     }
129 
130     /**
131      * Simple constructor.
132      *
133      * @param value <code>true</code> for <code>STRONG</code>, <code>false</code>
134      *	      for <code>WEAK</code>
135      */
136     private ConfidentialityStrength(boolean value) {
137 	this.value = value;
138     }
139 
140     /** Returns a string representation of this object. */
141     @Override
142     public String toString() {
143 	return value
144 	    ? "ConfidentialityStrength.STRONG"
145 	    : "ConfidentialityStrength.WEAK"; 
146     }
147 
148     /** Canonicalize so that <code>==</code> can be used. */
149     private Object readResolve() {
150 	return value ? STRONG : WEAK;
151     }
152 }
