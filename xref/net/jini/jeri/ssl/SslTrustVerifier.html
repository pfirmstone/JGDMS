19  package net.jini.jeri.ssl;
20  
21  import java.rmi.RemoteException;
22  import javax.net.SocketFactory;
23  import javax.security.auth.x500.X500Principal;
24  import net.jini.security.Security;
25  import net.jini.security.TrustVerifier;
26  import aQute.bnd.annotation.headers.RequireCapability;
27  import aQute.bnd.annotation.headers.ProvideCapability;
28  
29  /**
30   * Trust verifier for the {@link SslEndpoint}, {@link HttpsEndpoint}, and
31   * {@link ConfidentialityStrength} classes. Also trusts principals of type
32   * {@link X500Principal}. This class is intended to be specified in a resource
33   * to configure the operation of {@link Security#verifyObjectTrust
34   * Security.verifyObjectTrust}.
35   *
36   * 
37   * @since 2.0
38   * @see SslEndpoint
39   * @see HttpsEndpoint
40   * @see ConfidentialityStrength
41   */
42  @RequireCapability(
43  	ns="osgi.extender",
44  	filter="(osgi.extender=osgi.serviceloader.registrar)")
45  @ProvideCapability(
46  	ns="osgi.serviceloader",
47  	name="net.jini.security.TrustVerifier")
48  public final class SslTrustVerifier implements TrustVerifier {
49  
50      /** Creates an instance of this class. */
51      public SslTrustVerifier() { }
52  
53      /**
54       * Returns <code>true</code> if the object is an instance of {@link
55       * SslEndpoint} or {@link HttpsEndpoint}, and it's {@link SocketFactory} is
56       * either <code>null</code> or trusted by the specified
57       * <code>TrustVerifier.Context</code>; or if the object is an instance of
58       * {@link ConfidentialityStrength} or {@link X500Principal}; and returns
59       * <code>false</code> otherwise.
60       *
61       * @throws RemoteException if a communication-related exception occurs
62       *	       when verifying a socket factory
63       * @throws SecurityException if a security exception occurs when verifying
64       *	       a socket factory
65       * @throws NullPointerException {@inheritDoc}
66       */
67      public boolean isTrustedObject(Object obj, TrustVerifier.Context ctx)
68  	throws RemoteException
69      {
70  	SocketFactory factory;
71  	if (obj == null || ctx == null) {
72  	    throw new NullPointerException();
73  	} else if (obj instanceof ConfidentialityStrength ||
74  		   obj instanceof X500Principal)
75  	{
76  	    return true;
77  	} else if (obj instanceof SslEndpoint) {
78  	    factory = ((SslEndpoint) obj).getSocketFactory();
79  	} else if (obj instanceof HttpsEndpoint) {
80  	    factory = ((HttpsEndpoint) obj).getSocketFactory();
81  	} else {
82  	    return false;
83  	}
84  	return factory == null || ctx.isTrustedObject(factory);
85      }
86  }
