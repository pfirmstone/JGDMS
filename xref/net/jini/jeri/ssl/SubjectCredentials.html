49      /**
50       * Retrieves the X.509 CertPath for a credential name.  Returns null if the
51       * chain associated with the credential name is not found.  Does not check
52       * if either principal or private key associated with the chain are
53       * present.
54       *
55       * @param subject the Subject containing the credentials or null
56       * @param name the name of the credentials
57       * @return the certificate chain or null
58       */
59      static CertPath getCertificateChain(Subject subject, String name) {
60  	if (subject == null) {
61  	    return null;
62  	}
63  	CertificateMatcher matcher = CertificateMatcher.create(name);
64  	if (matcher != null) {
65  	    Set publicCreds = subject.getPublicCredentials();
66  	    synchronized (publicCreds) {
67  		for (Iterator it = publicCreds.iterator(); it.hasNext(); ) {
68  		    Object cred = it.next();
69  		    if (isX509CertificateChain(cred)) {
70  			CertPath chain = (CertPath) cred;
71  			if (matcher.matches(firstX509Cert(chain))) {
72  			    return chain;
73  			}
74  		    }
75  		}
76  	    }
77  	}
78  	return null;
79      }

91      /**
92       * Provides utilities for converting between X.509 certificates and unique
93       * certificate names.
94       */
95      private static class CertificateMatcher {
96  	private final BigInteger serialNumber;
97  	private final String issuerName;
98  
99  	/**
100 	 * Creates an object that can be compared with an X.509 certificate.
101 	 * Returns null if the argument is not a valid certificate name.
102 	 */
103 	static CertificateMatcher create(String certificateName) {
104 	    if (certificateName == null) {
105 		return null;
106 	    }
107 	    int atSignPosition = certificateName.indexOf('@');
108 	    if (atSignPosition < 0) {
109 		return null;
110 	    }
111 	    BigInteger serialNumber;
112 	    try {
113 		serialNumber = new BigInteger(
114 		    certificateName.substring(0, atSignPosition),
115 		    16);
116 	    } catch (NumberFormatException e) {
117 		return null;
118 	    }
119 	    String issuerName = certificateName.substring(atSignPosition + 1);
120 	    return new CertificateMatcher(serialNumber, issuerName);
121 	}
122 
123 	private CertificateMatcher(BigInteger serialNumber, String issuerName)
124 	{
125 	    this.serialNumber = serialNumber;
126 	    this.issuerName = issuerName;
127 	}
128 
129 	/** Returns the unique certificate name for an X.509 certificate */
130 	static String getName(X509Certificate certificate) {
131 	    /*
132 	     * Use the certificate serial number, which is unique for all
133 	     * certificates from a given issuer, plus the issuer name, to get a
134 	     * unique name for the certificate.
135 	     */
136 	    return certificate.getSerialNumber().toString(16) + "@" +
137 		getIssuerName(certificate);
138 	}
139 
140 	/**
141 	 * Returns true if an X.509 certificate matches the certificate name
142 	 * specified in the constructor.
143 	 */
144 	boolean matches(X509Certificate certificate) {
145 	    return certificate.getSerialNumber().equals(serialNumber)
146 		&& getIssuerName(certificate).equals(issuerName);
147 	}
148 
149 	/** Returns the canonical issuer name for an X.509 certificate. */
150 	private static String getIssuerName(X509Certificate certificate) {
151 	    return certificate.getIssuerX500Principal().getName(
152 		X500Principal.CANONICAL);
153 	}
154     }

156     /**
157      * Returns the X.509 CertPaths stored in the public credentials of the
158      * subject.  Does not check if the associated principals or private keys
159      * are present.  Returns null if none are found.
160      *
161      * @param subject the subject containing the X.509 CertPaths or null
162      * @return List of the X.509 CertPaths in the subject
163      */
164     static List getCertificateChains(Subject subject) {
165 	List result = null;
166 	if (subject != null) {
167 	    Set publicCreds = subject.getPublicCredentials();
168 	    synchronized (publicCreds) {
169 		for (Iterator it = publicCreds.iterator(); it.hasNext(); ) {
170 		    Object cred = it.next();
171 		    if (isX509CertificateChain(cred)) {
172 			if (result == null) {
173 			    result = new ArrayList(publicCreds.size());
174 			}
175 			result.add(cred);
176 		    }
177 		}
178 	    }
179 	}
180 	return result;
181     }

183     /**
184      * Checks if the subject's public credentials contain a certificate chain
185      * that starts with a certificate with the same subject and public key, and
186      * returns the certificate chain if it does.  Does not check the validity
187      * of the certificate chain, or for associated private credentials or
188      * principal.
189      *
190      * @param cert the certificate
191      * @return the certificate chain starting with an equivalent certificate,
192      *	       if present, otherwise null
193      */
194     static CertPath getCertificateChain(Subject subject, X509Certificate cert)
195     {
196 	if (subject != null) {
197 	    Principal subjectDN = null;
198 	    PublicKey key = null;
199 	    Set publicCreds = subject.getPublicCredentials();
200 	    synchronized (publicCreds) {
201 		for (Iterator it = publicCreds.iterator(); it.hasNext(); ) {
202 		    Object cred = it.next();
203 		    if (!isX509CertificateChain(cred)) {
204 			continue;
205 		    }
206 		    CertPath chain = (CertPath) cred;
207 		    X509Certificate start = firstX509Cert(chain);
208 		    if (cert.equals(start)) {
209 			return chain;
210 		    }
211 		    if (subjectDN == null) {
212 			subjectDN = cert.getSubjectDN();
213 			key = cert.getPublicKey();
214 		    }
215 		    if (subjectDN.equals(start.getSubjectDN())
216 			&& key.equals(start.getPublicKey()))
217 		    {
218 			return chain;
219 		    }
220 		}
221 	    }
222 	}
223 	return null;
224     }

226     /**
227      * Retrieves the principals in the subject with X.509 CertPaths which use
228      * the specified key algorithm and, optionally, have associated private
229      * credentials.  Uses the specified private credentials rather than getting
230      * them from the Subject to permit callers to cache them and avoid the cost
231      * of repeated permission checks to access them.
232      *
233      * @param subject the Subject containing the principals
234      * @param keyAlgorithms the permitted key algorithms, an OR of any of
235      *	      DSA_KEY_ALGORITHM and RSA_KEY_ALGORITHM
236      * @param privateCredentials the available private credentials, or null if
237      *	      private credentials are not required
238      * @return set of matching principals
239      */
240     static Set getPrincipals(Subject subject,
241 			     int keyAlgorithms,
242 			     X500PrivateCredential[] privateCredentials)
243     {
244 	Set result = new HashSet(subject.getPrincipals().size());
245 	List certPaths = getCertificateChains(subject);
246 	if (certPaths != null) {
247 	    for (int i = certPaths.size(); --i >= 0; ) {
248 		CertPath chain = (CertPath) certPaths.get(i);
249 		X509Certificate cert = firstX509Cert(chain);
250 		String alg = cert.getPublicKey().getAlgorithm();
251 		if (!permittedKeyAlgorithm(alg, keyAlgorithms)) {
252 		    continue;
253 		}
254 		X500Principal principal = getPrincipal(subject, cert);
255 		if (principal != null) {
256 		    boolean pcOK = privateCredentials == null;
257 		    if (!pcOK) {
258 			for (int j = privateCredentials.length; --j >= 0; ) {
259 			    X500PrivateCredential xpc = privateCredentials[j];
260 			    if (cert.equals(xpc.getCertificate())) {
261 				pcOK = true;
262 				break;
263 			    }
264 			}
265 		    }
266 		    if (pcOK) {
267 			result.add(principal);
268 		    }
269 		}
270 	    }
271 	}
272 	return result;
273     }

275     /**
276      * A privileged action that gets the private credentials for an X.509
277      * certificate.
278      */
279     static class GetPrivateCredentialAction implements PrivilegedAction {
280 	private final Subject subject;
281 	private final X509Certificate cert;
282 
283 	GetPrivateCredentialAction(Subject subject, X509Certificate cert) {
284 	    this.subject = subject;
285 	    this.cert = cert;
286 	}
287 
288 	public Object run() {
289 	    return SubjectCredentials.getPrivateCredential(subject, cert);
290 	}
291     }

293     /**
294      * A privileged action that returns all the X.500 private credentials for a
295      * subject as an X500PrivateCredential array.  Assumes that the subject is
296      * non-null.
297      */
298     static class GetAllPrivateCredentialsAction implements PrivilegedAction {
299 	private final Subject subject;
300 
301 	GetAllPrivateCredentialsAction(Subject subject) {
302 	    this.subject = subject;
303 	}
304 
305 	public Object run() {
306 	    Set pcs = subject.getPrivateCredentials();
307 	    List xpcs = new ArrayList(pcs.size());
308 	    synchronized (pcs) {
309 		/*
310 		 * XXX: Include this synchronization to work around BugID
311 		 * 4892913, Subject.getPrivateCredentials not thread-safe
312 		 * against changes to principals.  -tjb[22.Jul.2003]
313 		 *
314 		 * synchronized (subject.getPrincipals()) {
315 		 */
316 		for (Iterator iter = pcs.iterator(); iter.hasNext(); ) {
317 		    Object pc = iter.next();
318 		    if (pc instanceof X500PrivateCredential) {
319 			xpcs.add(pc);
320 		    }
321 		}
322 	    }
323 	    return xpcs.toArray(new X500PrivateCredential[xpcs.size()]);
324 	}
325     }

327     /**
328      * Returns the X500PrivateCredential for an X.509 certificate.  Returns
329      * null if the associated private credential is missing from the subject.
330      * Does not check if the public credential or principal are present.
331      * Assumes that the subject is non-null.  The caller should check for
332      * AuthenticationPermission and then call this method from within
333      * AccessController.doPrivileged to give it private credential permissions.
334      *
335      * @param subject the Subject containing the credentials
336      * @param cert the X.509 certificate
337      * @return the X500PrivateCredential or null
338      */
339     static X500PrivateCredential getPrivateCredential(Subject subject,
340 						      X509Certificate cert)
341     {
342 	X500PrivateCredential result = null;
343 	Set privateCreds = subject.getPrivateCredentials();
344 	synchronized (privateCreds) {
345 	    /*
346 	     * XXX: Include this synchronization to work around BugID 4892913,
347 	     * Subject.getPrivateCredentials not thread-safe against changes to
348 	     * principals.  -tjb[18.Jul.2003]
349 	     *
350 	     * synchronized (subject.getPrincipals()) {
351 	     */
352 	    for (Iterator it = privateCreds.iterator(); it.hasNext(); ) {
353 		Object cred = it.next();
354 		if (cred instanceof X500PrivateCredential) {
355 		    X500PrivateCredential xpc =
356 			(X500PrivateCredential) cred;
357 		    if (cert.equals(xpc.getCertificate())) {
358 			result = xpc;
359 			break;
360 		    }
361 		}
362 	    }
363 	}
364 	return result;
365     }

367     /**
368      * Returns the subject principal matching the X.509 certificate.  Returns
369      * null if the principal is not found.  Does not check if the associated
370      * private key is present.  Assumes that the subject is non-null.
371      *
372      * @param subject the Subject containing the credentials 
373      * @param cert the X.509 certificate
374      * @return the X.500 principal or null
375      */
376     static X500Principal getPrincipal(Subject subject, X509Certificate cert) {
377 	X500Principal x500 = cert.getSubjectX500Principal();
378 	String name = x500.getName(X500Principal.CANONICAL);
379 	Set principals = subject.getPrincipals();
380 	synchronized (principals) {
381 	    for (Iterator i = principals.iterator(); i.hasNext(); ) {
382 		Object next = i.next();
383 		if (!(next instanceof X500Principal)) {
384 		    continue;
385 		}
386 		X500Principal principal = (X500Principal) next;
387 		if (principal.getName(X500Principal.CANONICAL).equals(name)) {
388 		    return principal;
389 		}
390 	    }
391 	}
392 	return null;
393     }

395     /**
396      * Returns a String that describes the credentials in the subject.
397      *
398      * @param subject the Subject containing the credentials
399      * @return a String describing the credentials
400      * @throws NullPointerException if the subject is null
401      */
402     static String credentialsString(Subject subject) {
403 	List certPaths = getCertificateChains(subject);
404 	if (certPaths == null) {
405 	    return "";
406 	}
407 	StringBuffer buf = new StringBuffer();
408 	for (int i = certPaths.size(); --i >= 0; ) {
409 	    CertPath chain = (CertPath) certPaths.get(i);
410 	    X509Certificate cert = firstX509Cert(chain);
411 	    X500Principal principal = getPrincipal(subject, cert);
412 	    if (principal != null) {
413 		buf.append("  Principal: ").append(principal).append('\n');
414 		buf.append("    Public key: ");
415 		appendKeyString(cert.getPublicKey(), buf);
416 		buf.append('\n');
417 		buf.append("    Private key: ");
418 		try {
419 		    X500PrivateCredential cred =
420 			(X500PrivateCredential) Security.doPrivileged(
421 			    new GetPrivateCredentialAction(subject, cert));
422 		    PrivateKey privateKey =
423 			cred != null ? cred.getPrivateKey() : null;
424 		    if (privateKey == null) {
425 			buf.append("Not found");
426 		    } else {
427 			appendKeyString(privateKey, buf);
428 		    }
429 		} catch (SecurityException e) {
430 		    buf.append("No permission");
431 		}
432 	    }
433 	}
434 	return buf.toString();
435     }

437     /** Appends information about a key to a StringBuffer. */
438     private static void appendKeyString(Key key, StringBuffer buf) {
439 	String className = key.getClass().getName();
440 	buf.append(className.substring(className.lastIndexOf('.') + 1));
441 	buf.append('@');
442 	buf.append(Integer.toHexString(System.identityHashCode(key)));
443     }

445     /**
446      * Determines if the argument is an X.509 certificate CertPath.  Returns
447      * true if the argument is a non-null CertPath, has at least one
448      * certificate, and has type X.509.
449      */
450     private static boolean isX509CertificateChain(Object credential) {
451 	if (!(credential instanceof CertPath)) {
452 	    return false;
453 	}
454 	CertPath certPath = (CertPath) credential;
455 	if (certPath.getCertificates().isEmpty()) {
456 	    return false;
457 	} else if (!certPath.getType().equals("X.509")) {
458 	    return false;
459 	}
460 	return true;
461     }
