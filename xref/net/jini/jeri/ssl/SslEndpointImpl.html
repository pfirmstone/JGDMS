177     /** Implements Endpoint.newRequest */
178     final OutboundRequestIterator newRequest(
179 	InvocationConstraints constraints)
180     {
181 	if (constraints == null) {
182 	    throw new NullPointerException("Constraints cannot be null");
183 	}
184 	try {
185 	    return newRequest(getCallContext(constraints));
186 	} catch (UnsupportedConstraintException e) {
187 	    return new ExceptionOutboundRequestIterator(e);
188 	} catch (SecurityException e) {
189 	    return new ExceptionOutboundRequestIterator(e);
190 	}
191     }

205      * An outbound request iterator that throws an IOException or a
206      * SecurityException.
207      */
208     private static final class ExceptionOutboundRequestIterator
209 	implements OutboundRequestIterator
210     {
211 	private final Exception exception;
212 	private boolean done = false;
213 
214 	ExceptionOutboundRequestIterator(Exception exception) {
215 	    this.exception = exception;
216 	}
217 
218 	public synchronized boolean hasNext() {
219 	    return !done;
220 	}
221 
222 	public synchronized OutboundRequest next() throws IOException {
223 	    if (done) {
224 		throw new NoSuchElementException();
225 	    }
226 	    done = true;
227 	    if (exception instanceof SecurityException) {
228 		throw (SecurityException) exception;
229 	    } else {
230 		throw (IOException) exception;
231 	    }
232 	}
233     }

240     /** Returns the connection manager for this endpoint. */
241     private ConnManager getConnectionManager() {
242 	synchronized (connectionMgrs) {
243 	    if (connectionManager == null) {
244 		Reference ref = (Reference) connectionMgrs.get(this);
245 		connectionManager =
246 		    (ref != null) ? (ConnManager) ref.get() : null;
247 		if (connectionManager == null) {
248 		    connectionManager = connectionManagerFactory.create(this);
249 		    connectionMgrs.put(
250 			this, new WeakReference(connectionManager));
251 		}
252 	    }
253 	    return connectionManager;
254 	}

257     /**
258      * Returns a context for making a remote call with the specified
259      * constraints and the current subject. This method does not perform
260      * communication with the remote server.
261      *
262      * Throws a SecurityException if lack of authentication permissions
263      * needed for requirements prevent the use of all contexts.
264      *
265      * Throws UnsupportedConstraintException if some requirements cannot
266      * be satisfied.
267      *
268      * Returns a CallContext if throwing an exception or returning null
269      * would reveal information about the current subject that the
270      * caller does not have permission to know.
271      */
272     private CallContext getCallContext(InvocationConstraints constraints)
273 	throws UnsupportedConstraintException
274     {
275 	final AccessControlContext acc = AccessController.getContext();
276 	Subject clientSubject = (Subject) AccessController.doPrivileged(
277 	    new PrivilegedAction() {
278 		public Object run() {
279 		    return Subject.getSubject(acc);
280 		}
281 	    });
282 	Set clientPrincipals = getClientPrincipals(constraints.requirements());
283 	boolean requiredClient = clientPrincipals != null;
284 	boolean constrainedServer = getServerPrincipals(constraints) != null;
285 	Boolean subjectPrivilege = null;
286 	if (!requiredClient) {
287 	    /* Try using principals from Subject instead */
288 	    if (clientSubject == null) {
289 		clientPrincipals = Collections.EMPTY_SET;
290 	    } else {
291 		/*
292 		 * XXX: Work around BugID 4892841, Subject.getPrincipals(Class)
293 		 * not thread-safe against changes to principals.
294 		 * -tjb[18.Jul.2003]
295                  * 
296                  * This was fixed in Java 1.5 which is now our minimum
297                  * supported version.
298 		 */
299 		synchronized (clientSubject.getPrincipals()) {
300 		    clientPrincipals =
301 			clientSubject.getPrincipals(X500Principal.class);
302 		}
303 	    }
304 	    if (clientPrincipals.isEmpty()) {
305 		subjectPrivilege = getSubjectPermitted();
306 		if (subjectPrivilege.equals(Boolean.FALSE)) {
307 		    /* Don't reveal that the client Subject has no principals.
308 		     * Provide a dummy Principal (no credentials) which cannot
309 		     * be authenticated in order to follow the same code path
310 		     * as if there were a Subject principal.
311 		     */
312 		    clientPrincipals =
313 			Collections.singleton(UNKNOWN_PRINCIPAL);
314 		}
315 	    }
316 	}
317 
318 	/* Compute a list of ConnectionContexts between principals (as above):
319 	 * (a) principals named in client constraints, if any; otherwise
320 	 * (b) principals named in the client Subject, if any; otherwise
321 	 * (c) a dummy principal or no principal
322 	 * and the principals named in server constraints (if any). These
323 	 * contexts will satisfy most constraints.
324 	 */
325 	List contexts = new CopyOnRemoveList(
326 	    getConnectionContexts(constraints, clientPrincipals));
327 	if (constrainedServer) {
328 	    /* Server principals were named in constraints.  Remove from the
329 	     * context list any ConnectionContexts for which there is no
330 	     * permission to authenticate the context's client principal with
331 	     * the context's server principal.
332 	     */
333 	    try {
334 		contexts = checkAuthenticationPermissions(contexts);
335 	    } catch (SecurityException e) {
336 		/* This SecurityException indicates that there are no
337 		 * remaining valid ConnectionContexts, either because there
338 		 * are no client constraint principals and no client subject
339 		 * principals, or because none of these principals was given
340 		 * permission to authenticate with any server constraint
341 		 * principal.
342 		 *
343 		 * If there were no client constraint principals, then
344 		 * throwing an exception here would reveal that at least one
345 		 * client subject principal existed and did not have
346 		 * authentication permission.
347 		 *
348 		 * If there were client constraint principals, or if the
349 		 * caller could determine Subject principals for itself, then
350 		 * pass on the SecurityException.
351 		 */
352 		if (!requiredClient && subjectPrivilege == null) {
353 		    subjectPrivilege = getSubjectPermitted();
354 		}
355 		if (requiredClient || subjectPrivilege) {
356 		    if (logger.isLoggable(Levels.FAILED)) {
357 			logThrow(
358 			    logger, Levels.FAILED,
359 			    SslEndpointImpl.class, "getCallContext",
360 			    "new request for {0}\nwith {1}\nand {2}\nthrows",
361 			    new Object[] { endpoint, constraints,
362 					   subjectString(clientSubject) },
363 			    e);
364 		    }
365 		    throw e;
366 		} else {
367 		    /* Don't reveal that the client Subject has no principals.
368 		     * Provide a dummy Principal (no credentials) which cannot
369 		     * be authenticated in order to follow the same code path
370 		     * as if there were a Subject principal.
371 		     */
372 		    CallContext result = createCallContext(
373 			getConnectionContexts(
374 			    constraints,
375 			    Collections.singleton(UNKNOWN_PRINCIPAL)),
376 			null);
377 		    if (logger.isLoggable(Levels.FAILED)) {
378 			logThrow(
379 			    logger, Levels.FAILED,
380 			    SslEndpointImpl.class, "getCallContext",
381 			    "new request for {0}\nwith {1}\nand {2}\n" +
382 			    "will fail but cannot throw " +
383 			    "because caller has no subject access\n" +
384 			    "returns {3}\n" +
385 			    "caught exception",
386 			    new Object[] { endpoint, constraints,
387 					   subjectString(clientSubject),
388 					   result },
389 			    e);
390 		    }
391 		    return result;
392 		}
393 	    }
394 	}
395 	UnsupportedConstraintException unsupported = null;
396 	if (contexts.isEmpty()) {
397 	    unsupported = new UnsupportedConstraintException(
398 		"Constraints not supported: " + constraints);
399 	} else {
400 	    boolean checkSubject;
401 	    if (constrainedServer) {
402 		checkSubject = true;
403 	    } else {
404 		if (subjectPrivilege == null) {
405 		    subjectPrivilege = getSubjectPermitted();
406 		}
407 		checkSubject = (subjectPrivilege == Boolean.TRUE);
408 	    }
409 	    if (checkSubject) {
410 		/* Check subject if caller has any access */
411 		try {
412 		    contexts = checkSubject(contexts, clientSubject,
413 					    constrainedServer, constraints);
414 		} catch (UnsupportedConstraintException e) {
415 		    unsupported = e;
416 		}
417 	    }
418 	}
419 	if (unsupported != null) {
420 	    if (logger.isLoggable(Levels.FAILED)) {
421 		logThrow(logger, Levels.FAILED,
422 			 SslEndpointImpl.class, "getCallContext",
423 			 "new request for {0}\nwith {1}\nand {2}\nthrows",
424 			 new Object[] { endpoint, constraints,
425 					subjectString(clientSubject) },
426 			 unsupported);
427 	    }
428 	    throw unsupported;
429 	} else {
430 	    CallContext result = createCallContext(contexts, clientSubject);
431 	    if (logger.isLoggable(Level.FINE)) {
432 		logger.log(Level.FINE,
433 			   "new request for {0}\nwith {1}\nand {2}\n" +
434 			   "returns {3}",
435 			   new Object[] {
436 			       endpoint, constraints,
437 			       subjectString(clientSubject), result
438 			   });
439 	    }
440 	    return result;
441 	}
442     }

444     /** Convert connection contexts to a call context */
445     private CallContext createCallContext(List contexts,
446 					  Subject clientSubject)
447     {
448 	boolean clientAuthRequired = true;
449 	boolean clientAuthPermitted = false;
450 	Set clientPrincipals = null;
451 	Set serverPrincipals = null;
452 	List suites = new ArrayList();
453 	boolean integrityRequired = false;
454 	boolean integrityPreferred = false;
455 	boolean atomicityRequired = false;
456 	boolean atomicityPreferred = false;
457 	long connectionTimeout = -1;
458 	int max = contexts.size();
459 	for (int i = 0; i < max; i++) {
460 	    ConnectionContext context = (ConnectionContext) contexts.get(i);
461 	    if (context.client == null) {
462 		clientAuthRequired = false;
463 	    } else {
464 		clientAuthPermitted = true;
465 		if (clientPrincipals == null) {
466 		    clientPrincipals = new HashSet();
467 		}
468 		/*
469 		 * Don't include the unknown principal, but still should record
470 		 * that principals were specified by making clientPrincipals
471 		 * non-null.
472 		 */
473 		if (context.client != UNKNOWN_PRINCIPAL) {
474 		    clientPrincipals.add(context.client);
475 		}
476 	    }
477 	    if (context.server != null &&
478 		context.server != UNKNOWN_PRINCIPAL)
479 	    {
480 		if (serverPrincipals == null) {
481 		    serverPrincipals = new HashSet();
482 		}
483 		serverPrincipals.add(context.server);
484 	    }
485 	    if (!suites.contains(context.cipherSuite)) {
486 		suites.add(context.cipherSuite);
487 	    }
488 	    if (context.getIntegrityRequired()) {
489 		integrityRequired = true;
490 	    } else if (context.getIntegrityPreferred()) {
491 		integrityPreferred = true;
492 	    }
493 	    if (context.getAtomicityRequired()) {
494 		atomicityRequired = true;
495 	    } else if (context.getAtomicityPreferred()) {
496 		atomicityPreferred = true;
497 	    }
498 	    if (context.getConnectionTime() != -1 &&
499 		(connectionTimeout == -1 ||
500 		 connectionTimeout > context.getConnectionTime()))
501 	    {
502 		connectionTimeout = context.getConnectionTime();
503 	    }
504 	}
505 	return new CallContext(
506 	    endpoint, this,
507 	    clientAuthPermitted ? clientSubject : null,
508 	    clientAuthRequired, clientPrincipals, serverPrincipals, suites,
509 	    integrityRequired, integrityPreferred, atomicityRequired, 
510 		atomicityPreferred, connectionTimeout);
511     }

513     /**
514      * Returns a list of the contexts which are supported by principals and
515      * credentials in the Subject.  Throws an UnsupportedConstraintException if
516      * none of the contexts are supported, otherwise returns a non-empty list.
517      */
518 
519     private static List checkSubject(List contexts,
520 				     Subject clientSubject,
521 				     boolean constrainedServer,
522 				     InvocationConstraints constraints)
523 	throws UnsupportedConstraintException
524     {
525 	Map publicCreds = getPublicCredentials(clientSubject);
526 	X500PrivateCredential[] privateCreds =
527 	    (clientSubject != null && constrainedServer)
528 	    ? (X500PrivateCredential[]) AccessController.doPrivileged(
529 		new SubjectCredentials.GetAllPrivateCredentialsAction(
530 		    clientSubject))
531 	    : new X500PrivateCredential[0];
532 	Set missingPublic = new HashSet();
533 	Set missingPrivate = new HashSet();
534 	/* Check principals and credentials */
535       top:
536 	for (int i = contexts.size(); --i >= 0; ) {
537 	    ConnectionContext context = (ConnectionContext) contexts.get(i);
538 	    if (context.client == null) {
539 		continue;			/* Anonymous client OK */
540 	    }
541 	    Collection certs = (Collection) publicCreds.get(context.client);
542 	    if (certs == null) {
543 		logger.log(Levels.HANDLED,
544 			   "missing principal or public credentials: {0}",
545 			   context.client);
546 		contexts.remove(i);
547 		missingPublic.add(context.client);
548 	    } else if (constrainedServer) {
549 		/*
550 		 * Checked authentication permissions, so OK to check private
551 		 * credentials.
552 		 */
553 		for (int j = privateCreds.length; --j >= 0; ) {
554 		    X509Certificate cert = privateCreds[j].getCertificate();
555 		    if (cert != null) {		/* null if destroyed */
556 			if (certs.contains(cert)) {
557 			    continue top;	/* Private credentials OK */
558 			}
559 		    }
560 		}
561 		logger.log(Levels.HANDLED, "missing private credentials: {0}",
562 			   context.client);
563 		contexts.remove(i);
564 		missingPrivate.add(context.client);
565 	    }
566 	}
567 	if (!contexts.isEmpty()) {
568 	    return contexts;
569 	} else {
570 	    throw new UnsupportedConstraintException(
571 		"Constraints not supported: " + constraints + ";" +
572 		(missingPublic.isEmpty() ? ""
573 		 : ("\nmissing principals or public credentials: " +
574 		    missingPublic)) +
575 		(missingPrivate.isEmpty() ? ""
576 		 : "\nmissing private credentials: " + missingPrivate));
577 	}
578     }

580     /**
581      * Checks if the caller has permission to get the current subject,
582      * returning Boolean.TRUE or FALSE.
583      */
584     private static Boolean getSubjectPermitted() {
585 	SecurityManager sm = System.getSecurityManager();
586 	if (sm != null) {
587 	    try {
588 		sm.checkPermission(GET_SUBJECT_PERMISSION);
589 	    } catch (SecurityException e) {
590 		return Boolean.FALSE;
591 	    }
592 	}
593 	return Boolean.TRUE;
594     }

596     /**
597      * Removes the contexts for which the client does not have authentication
598      * permission.  Throws SecurityException if lack of permission prevents any
599      * contexts from being used.
600      */
601     private static List checkAuthenticationPermissions(List contexts) {
602 	if (contexts.isEmpty()) {
603 	    return contexts;
604 	}
605 	SecurityManager sm = System.getSecurityManager();
606 	if (sm == null) {
607 	    return contexts;
608 	}
609 	Map perms = new HashMap();
610 	Set exceptions = new HashSet();
611 	for (int i = contexts.size(); -- i >= 0; ) {
612 	    ConnectionContext context = (ConnectionContext) contexts.get(i);
613 	    if (context.client == null) {
614 		/* Client anonymous -- OK */
615 		continue;
616 	    } else if (context.server == UNKNOWN_PRINCIPAL) {
617 		/* Server not known -- can't check any permissions */
618 		break;
619 	    }
620 	    AuthenticationPermission p = new AuthenticationPermission(
621 		Collections.singleton(context.client),
622 		Collections.singleton(context.server),
623 		"connect");
624 	    Object value = perms.get(p);
625 	    if (Boolean.FALSE.equals(value)) {
626 		contexts.remove(i);
627 	    } else if (!Boolean.TRUE.equals(value)) {
628 		try {
629 		    sm.checkPermission(p);
630 		    perms.put(p, Boolean.TRUE);
631 		} catch (SecurityException e) {
632 		    logger.log(
633 			Levels.HANDLED,
634 			"check authentication permission caught exception",
635 			e);
636 		    perms.put(p, Boolean.FALSE);
637 		    exceptions.add(e);
638 		    contexts.remove(i);
639 		}
640 	    }
641 	}
642 	if (!contexts.isEmpty()) {
643 	    return contexts;
644 	} else if (exceptions.size() == 1) {
645 	    throw (SecurityException) exceptions.iterator().next();
646 	} else {
647 	    throw new SecurityException(exceptions.toString());
648 	}
649     }

651     /**
652      * Returns a map that maps each principal in the subject to a set of the
653      * associated X.500 public credentials.
654      */
655     private static Map getPublicCredentials(Subject subject) {
656 	Map publicCreds = new HashMap();
657 	List certPaths = SubjectCredentials.getCertificateChains(subject);
658 	if (certPaths != null) {
659 	    for (int i = certPaths.size(); --i >= 0; ) {
660 		CertPath chain = (CertPath) certPaths.get(i);
661 		X509Certificate cert = SubjectCredentials.firstX509Cert(chain);
662 		X500Principal p =
663 		    SubjectCredentials.getPrincipal(subject, cert);
664 		if (p != null) {
665 		    Collection certs = (Collection) publicCreds.get(p);
666 		    if (certs == null) {
667 			certs = new ArrayList(1);
668 			publicCreds.put(p, certs);
669 		    }
670 		    certs.add(cert);
671 		}
672 	    }
673 	}
674 	return publicCreds;
675     }

677     /**
678      * A List that supports removing items by making a copy of the underlying
679      * list.
680      */
681     private static final class CopyOnRemoveList extends AbstractList {
682 	private List list;
683 	private boolean modified;
684 
685 	CopyOnRemoveList(List list) {
686 	    this.list = list;
687 	}
688 
689 	public Object get(int index) {
690 	    return list.get(index);
691 	}
692 
693 	public int size() {
694 	    return list.size();
695 	}
696 
697 	public Object remove(int index) {
698 	    if (!modified) {

700 		modified = true;
701 	    }
702 	    return list.remove(index);
703 	}

707      * Defines a structure to cache a ConnectionContexts for specific
708      * constraints and client principals.
709      */
710     private static final class ConnectionContextCache {
711 	final InvocationConstraints constraints;
712 	final Set clientPrincipals;
713 	final List connectionContexts;
714 
715 	ConnectionContextCache(InvocationConstraints constraints,
716 			       Set clientPrincipals,
717 			       List connectionContexts)
718 	{
719 	    this.constraints = constraints;
720 	    this.clientPrincipals = clientPrincipals;
721 	    this.connectionContexts = connectionContexts;
722 	}
723     }

725     /**
726      * Gets an unmodifiable list of the ConnectionContexts for the
727      * specified constraints and client principals.
728      */
729     private List getConnectionContexts(InvocationConstraints constraints,
730 				       Set clientPrincipals)
731     {
732 	synchronized (connectionContextCache) {
733 	    for (int i = CACHE_SIZE; --i >= 0; ) {
734 		ConnectionContextCache cache = connectionContextCache[i];
735 		if (cache != null &&
736 		    cache.constraints.equals(constraints) &&
737 		    cache.clientPrincipals.equals(clientPrincipals))
738 		{
739 		    logger.log(Level.FINEST, "used connection cache");
740 		    return cache.connectionContexts;
741 		}
742 	    }
743 	}
744 	Set serverPrincipals = getServerPrincipals(constraints);
745 	if (serverPrincipals == null) {
746 	    serverPrincipals = Collections.singleton(UNKNOWN_PRINCIPAL);
747 	}
748 	List contexts = Collections.unmodifiableList(
749 	    computeConnectionContexts(
750 		getSupportedCipherSuites(), clientPrincipals, serverPrincipals,
751 		constraints));
752 	synchronized (connectionContextCache) {
753 	    connectionContextCache[cacheNext] =
754 		new ConnectionContextCache(
755 		    constraints, clientPrincipals, contexts);
756 	    if (cacheNext == 0) {
757 		cacheNext = CACHE_SIZE;
758 	    }
759 	    cacheNext--;
760 	}
761 	return contexts;
762     }

764     /** Used for sorting ConnectionContexts by preferences and suite order. */
765     private static final class ComparableConnectionContext
766 	implements Comparable
767     {
768 	final ConnectionContext context;
769 	private final int suiteIndex;
770 
771 	ComparableConnectionContext(ConnectionContext context,
772 				    int suiteIndex)
773 	{
774 	    this.context = context;
775 	    this.suiteIndex = suiteIndex;
776 	}
777 
778 	public int compareTo(Object object) {
779 	    ComparableConnectionContext other =
780 		(ComparableConnectionContext) object;
781 	    /* Lower value for more preferences */
782 	    int result =
783 		other.context.getPreferences() - context.getPreferences();
784 	    if (result == 0) {
785 		/* Lower value for lower index */
786 		result = suiteIndex - other.suiteIndex;
787 	    }
788 	    return result;
789 	}
790 	
791 	@Override
792 	public boolean equals(Object o){
793 	    if (!(o instanceof ComparableConnectionContext)) return false;
794 	    ComparableConnectionContext that = (ComparableConnectionContext) o;
795 	    if (this.suiteIndex != that.suiteIndex) return false;
796 	    return this.context.equals(that.context);
797 	}
798 
799 	@Override
800 	public int hashCode() {
801 	    int hash = 3;
802 	    hash = 97 * hash + (this.context != null ? this.context.hashCode() : 0);
803 	    hash = 97 * hash + this.suiteIndex;
804 	    return hash;
805 	}
806 
807 	public String toString() {
808 	    StringBuffer sb = new StringBuffer("ComparableConnectionContext[");
809 	    context.fieldsToString(sb);
810 	    sb.append(", index: ").append(suiteIndex);
811 	    sb.append("]");
812 	    return sb.toString();
813 	}
814     }

816     /**
817      * Computes a list of ConnectionContexts for the specified set of
818      * suites, client and server principals, and constraints, sorted by
819      * preferences and suite order.
820      */
821     private static List computeConnectionContexts(
822 	String[] suites,
823 	Set clients,
824 	Set servers,
825 	InvocationConstraints constraints)
826     {
827 	List result = new ArrayList(
828 	    suites.length * (clients.size() + 1) * (servers.size() + 1));
829 	for (int suiteIndex = suites.length; --suiteIndex >= 0; ) {
830 	    String suite = suites[suiteIndex];
831 	    Iterator cIter = clients.iterator();
832 	    Principal client;
833 	    do {
834 		if (cIter.hasNext()) {
835 		    client = (Principal) cIter.next();
836 		    assert client != null;
837 		} else {
838 		    client = null;
839 		}
840 		Iterator sIter = servers.iterator();
841 		Principal server;
842 		do {
843 		    if (sIter.hasNext()) {
844 			server = (Principal) sIter.next();
845 			assert server != null;
846 		    } else {
847 			server = null;
848 		    }
849 		    for (int i = 2; --i >= 0;) {
850 			// First loop checks lower level constraints only
851 			// Second loop checks upper layer constraints as well.
852 			boolean upperLayerConstraints = i == 0;
853 			ConnectionContext context =
854 			    ConnectionContext.getInstance(suite, client, server, upperLayerConstraints,
855 				true /* clientSide */, constraints);
856 			if (context != null) {
857 			    result.add(
858 				new ComparableConnectionContext(
859 				    context, suiteIndex));
860 			}
861 		    }
862 		} while (server != null);
863 	    } while (client != null);
864 	}
865 	Collections.sort(result);
866 	logger.log(Level.FINEST, "compute connection contexts produces {0}",
867 		   result);
868 	for (int i = result.size(); --i >= 0; ) {
869 	    ComparableConnectionContext ccc =
870 		(ComparableConnectionContext) result.get(i);
871 	    result.set(i, ccc.context);
872 	}
873 	return result;
874     }

887     /** Chooses a connection from existing connections. */
888     public Connection connect(OutboundRequestHandle handle,
889 			      Collection active,
890 			      Collection idle)
891     {
892 	CallContext context = CallContext.coerce(handle, endpoint);
893 	if (active == null || idle == null) {
894 	    throw new NullPointerException("Arguments cannot be null");
895 	}
896 	Connection result = null;
897 	/*
898 	 * Choose the first connection with an appropriate subject, an active
899 	 * suite that is one of the requested suites, and one for which all the
900 	 * requested suites better than the active one were also better for its
901 	 * call context.  That insures that we've gotten the best connection we
902 	 * can, assuming a new handshake were to make similar decisions.
903 	 */
904 	boolean checkedResolvePermission = false;
905 	for (Iterator iter = new ConnectionsIterator(endpoint, active, idle);
906 	     iter.hasNext(); )
907 	{
908 	    SslConnection connection = (SslConnection) iter.next();
909 	    if (connection.useFor(context)) {
910 		String phost = connection.getProxyHost();
911 		boolean usingProxy = (phost.length() != 0);
912 
913 		if (usingProxy) {
914 		    SecurityManager sm = System.getSecurityManager();
915 		    if (sm != null) {
916 			try {
917 			    sm.checkConnect(serverHost, port);
918 			} catch (SecurityException e) {
919 			    if (logger.isLoggable(Levels.FAILED)) {
920 				logThrow(logger, Levels.FAILED,
921 					 SslEndpointImpl.class, "connect",
922 					 "choose connection for {0}\nthrows",
923 					 new Object[] { this }, e);
924 			    }
925 			    throw e;
926 			}
927 		    }
928 		} else {
929 		    if (!checkedResolvePermission) {
930 			try {
931 			    checkResolvePermission();
932 			} catch (SecurityException e) {
933 			    if (logger.isLoggable(Levels.FAILED)) {
934 				LogUtil.logThrow(logger, Levels.FAILED,
935 					   SslEndpointImpl.class, "connect",
936 					   "exception resolving host {0}",
937 					   new Object[] { serverHost }, e);
938 			    }
939 			    throw e;
940 			}
941 			checkedResolvePermission = true;
942 		    }
943 		    try {
944 			if (!connection.checkConnectPermission()) {
945 			    continue;
946 			}
947 		    } catch (SecurityException e) {
948 			if (logger.isLoggable(Levels.HANDLED)) {
949 			    LogUtil.logThrow(logger, Levels.HANDLED,
950 				SslEndpointImpl.class, "nextRequest",
951 				"access to reuse connection {0} denied",
952 				new Object[] { connection }, e);
953 			}
954 			continue;
955 		    }
956 		}
957 		result = connection;
958 		break;
959 	    }
960 	}
961 	if (logger.isLoggable(Level.FINE)) {
962 	    logger.log(Level.FINE,
963 		       "choose connection for {0}\nwith active {1}\n" +
964 		       "and idle {2}\nreturns {3}",
965 		       new Object[] { handle, active, idle, result });
966 	}
967 	return result;
968     }
969 
970     private void checkResolvePermission() {
971 	SecurityManager sm = System.getSecurityManager();
972 	if (sm != null) {
973 	    sm.checkConnect(serverHost, -1);
974 	}
975     }

977     /**
978      * Defines an iterator over active and idle connections which performs
979      * error checking on connections.
980      */
981     private static final class ConnectionsIterator implements Iterator {
982 	private final Endpoint endpoint;
983 	private Collection active;	/* Set to null when exhausted */
984 	private final Collection idle;
985 	private Iterator iter;
986 
987 	ConnectionsIterator(Endpoint endpoint,
988 			    Collection active,
989 			    Collection idle)
990 	{
991 	    this.endpoint = endpoint;
992 	    this.active = active;
993 	    this.idle = idle;
994 	    iter = active.iterator();
995 	    if (!iter.hasNext()) {
996 		this.active = null;
997 		iter = idle.iterator();
998 	    }
999 	}
1000 
1001 	public boolean hasNext() {
1002 	    return iter.hasNext();
1003 	}
1004 
1005 	public Object next() {
1006 	    if (!hasNext()) {
1007 		throw new NoSuchElementException();
1008 	    }
1009 	    Object next = iter.next();
1010 	    if (next == null) {
1011 		throw new NullPointerException("Connection cannot be null");
1012 	    } else if (!(next instanceof SslConnection)) {
1013 		throw new IllegalArgumentException(
1014 		    "Connection must be of type SslConnection: " + next);
1015 	    }
1016 	    SslConnection result = (SslConnection) next;
1017 	    if (!endpoint.equals(result.callContext.endpoint)) {
1018 		throw new IllegalArgumentException(
1019 		    "Connection has wrong endpoint: found " +
1020 		    result.callContext.endpoint + ", expected " + endpoint);
1021 	    }
1022 	    if (!iter.hasNext() && active != null) {
1023 		active = null;
1024 		iter = idle.iterator();
1025 	    }
1026 	    return result;
1027 	}
1028 
1029 	public void remove() {
1030 	    throw new UnsupportedOperationException();
1031 	}
