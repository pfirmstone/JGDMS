19  package net.jini.jeri;
20  
21  import java.rmi.Remote;
22  import java.rmi.server.ExportException;
23  import net.jini.export.Exporter;
24  
25  /**
26   * A factory for creating a compatible proxy and invocation dispatcher for
27   * a remote object being exported.  An <code>InvocationLayerFactory</code>
28   * is used in conjunction with the {@link BasicJeriExporter} class to
29   * customize remote invocation and dispatch behavior for remote objects.
30   *
31   * @author	Sun Microsystems, Inc.
32   * @since 2.0
33   **/
34  public interface InvocationLayerFactory {
35  
36      /**
37       * Returns a compatible proxy and invocation dispatcher for a remote
38       * object being exported.  The proxy and invocation dispatcher are
39       * returned in an {@link Instances} container object.  The proxy sends
40       * calls to the remote object using the supplied
41       * <code>ObjectEndpoint</code>.
42       *
43       * <p>The returned proxy implements an implementation-specific set of
44       * remote interfaces of <code>impl</code> and may implement additional
45       * implementation-specific interfaces.
46       *
47       * <p>A given {@link Exporter} implementation should only call this
48       * method once per export.  An invocation dispatcher constructed for a
49       * previous export should not be reused.
50       *
51       * @param	impl the remote object that the proxy is being
52       *		created for
53       * @param	oe the object endpoint used to communicate with
54       *		the remote object
55       * @param	caps the transport capabilities of the server
56       * @return	a proxy and invocation dispatcher contained in an
57       *		<code>Instances</code> object
58       * @throws	ExportException if there is a problem creating the proxy or
59       *		dispatcher 
60       * @throws	NullPointerException if any argument is <code>null</code>
61       **/
62      Instances createInstances(Remote impl,
63  			      ObjectEndpoint oe,
64  			      ServerCapabilities caps)
65  	throws ExportException;
66  
67      /**
68       * A container for the proxy and invocation dispatcher instances
69       * returned by {@link InvocationLayerFactory#createInstances
70       * InvocationLayerFactory.createInstances}.
71       *
72       * @since 2.0
73       **/
74      public class Instances {
75  
76  	private final Remote proxy;
77  	private final InvocationDispatcher dispatcher;
78  
79  	/**
80  	 * Creates a container for a proxy instance and an invocation
81  	 * dispatcher instance. 
82  	 *
83  	 * @param   proxy a proxy
84  	 * @param   dispatcher an invocation dispatcher
85  	 * @throws  NullPointerException if any argument is
86  	 *	    <code>null</code> 
87  	 **/
88  	public Instances(Remote proxy, InvocationDispatcher dispatcher) {
89  	    if (proxy == null || dispatcher == null) {
90  		throw new NullPointerException();
91  	    }
92  	    this.proxy = proxy;
93  	    this.dispatcher = dispatcher;
94  	}
95  
96  	/**
97  	 * Returns the proxy.
98  	 *
99  	 * @return the proxy
100 	 */
101 	public Remote getProxy() {
102 	    return proxy;
103 	}
104 
105 	/**
106 	 * Returns the invocation dispatcher.
107 	 *
108 	 * @return the invocation dispatcher
109 	 */
110 	public InvocationDispatcher getInvocationDispatcher() {
111 	    return dispatcher;
112 	}
113     }
114 }
