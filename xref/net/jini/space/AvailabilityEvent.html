18  package net.jini.space;
19  
20  import java.io.IOException;
21  import java.io.InvalidObjectException;
22  import java.io.ObjectInputStream;
23  import java.io.ObjectOutputStream;
24  import java.rmi.MarshalledObject;
25  import net.jini.core.entry.Entry;
26  import net.jini.core.entry.UnusableEntryException;
27  import net.jini.core.event.RemoteEvent;
28  import org.apache.river.api.io.AtomicSerial;
29  import org.apache.river.api.io.AtomicSerial.GetArg;
30  
31  /**
32   * A <code>RemoteEvent</code> marking the transition of an
33   * <code>Entry</code> from {@link
34   * JavaSpace05#registerForAvailabilityEvent <em>unavailable</em> to
35   * <em>available</em>}.<p>
36   * 
37   * Note, by the time the event is delivered, the
38   * <code>Entry</code> whose transition triggered this event may
39   * have transitioned to a state where it is no longer visible
40   * and/or available.<p>
41   *
42   * @see JavaSpace05
43   * @since 2.1 
44   */
45  @AtomicSerial
46  public abstract class AvailabilityEvent extends RemoteEvent {
47      private static final long serialVersionUID = 1L;
48  
49      /** 
50       * <code>true</code> if this event signals a
51       * transition from invisible to visible as well
52       * as unavailable to available.
53       *
54       * @serial 
55       */
56      private final boolean visibilityTransition;
57  
58      private static boolean check(GetArg arg) throws IOException{
59  	// If this class is not in the heirarchy of classes the stream
60  	// may have been tampered with, see the Serialization Specification
61  	// section 3.5, this is equivalent to readObjectNoData().
62  	Class [] serialClasses = arg.serialClasses();
63  	for (Class serialClass : serialClasses){
64  	    if (serialClass.equals(AvailabilityEvent.class)){
65  		RemoteEvent sup = new RemoteEvent(arg);
66  		if (sup.getSource() == null)
67  		    throw new InvalidObjectException("null source reference");
68  		if (!(sup.getSource() instanceof JavaSpace)) 
69  		    throw new InvalidObjectException("source is not a JavaSpace");
70  		return arg.get("visibilityTransition", false);
71  	    }
72  	}
73  	throw new InvalidObjectException(
74  		"AvailabilityEvent should always have data");
75      }
76      
77      private AvailabilityEvent(GetArg arg, boolean visibilityTransition) throws IOException{
78  	super(arg);
79  	this.visibilityTransition = visibilityTransition;
80      }
81      
82      protected AvailabilityEvent(GetArg arg) throws IOException {
83  	this(arg, check(arg));
84      }
85      
86      
87  
88      /**
89       * Create a new <code>AvailabilityEvent</code> instance.
90       * @param source    the event source
91       * @param eventID   the event identifier
92       * @param seqNum    the event sequence number
93       * @param handback  the handback object
94       * @param visibilityTransition <code>true</code> if this event
95       *                  must also signal a transition from
96       *                  invisible to visible
97       * @throws NullPointerException if <code>source</code> is
98       *        <code>null</code>
99       */
100     protected AvailabilityEvent(JavaSpace        source,
101 				long             eventID,
102 				long             seqNum,
103 				MarshalledObject handback,
104 				boolean          visibilityTransition) 
105     {
106 	super(source, eventID, seqNum, handback);
107 	this.visibilityTransition = visibilityTransition;
108     }
109 
110     private void writeObject(ObjectOutputStream out) throws IOException {
111 	out.defaultWriteObject();
112     }
113 
114     /**
115      * @throws InvalidObjectException if {@link #source} is <code>null</code>
116      * or is not a {@link JavaSpace}
117      */
118     private void readObject(ObjectInputStream in)
119 	throws IOException, ClassNotFoundException
120     {
121 	in.defaultReadObject();
122 
123 	if (getSource() == null)
124 	    throw new InvalidObjectException("null source reference");
125 
126 	if (!(getSource() instanceof JavaSpace)) 
127 	    throw new InvalidObjectException("source is not a JavaSpace");
128     }
129 
130     /** 
131      * @throws InvalidObjectException if called
132      */
133     private void readObjectNoData() throws InvalidObjectException {
134 	throw new InvalidObjectException(
135 	    "AvailabilityEvent should always have data");
136     }
137 
138     /**
139      * Returns a copy of the {@link Entry} whose transition
140      * triggered this event. The returned <code>Entry</code> must
141      * be unmarshalled in accordance with the <a
142      * href=http://www.jini.org/standards/index.html>Jini
143      * Entry Specification</a>.
144      * 
145      * @return a copy of the {@link Entry} whose transition
146      *         triggered this event
147      * @throws UnusableEntryException if the <code>Entry</code>
148      *         can't be unmarshalled in the client. The next call
149      *         must re-attempt unmarshalling the
150      *         <code>Entry</code> 
151      */
152     public abstract Entry getEntry() throws UnusableEntryException;
153 
154     /**
155      * Returns a <em>snapshot</em> of the {@link Entry} whose
156      * transition triggered this event. Snapshots are defined in
157      * section JS.2.6 of the <a
158      * href=http://www.jini.org/standards/index.html>;
159      * JavaSpaces Service Specification</a> and are an
160      * alternative representation of a given <code>Entry</code>
161      * produced by a particular space for use with that same
162      * space. Passing a snapshot to a space is generally more
163      * efficient than passing the original <code>Entry</code>.<p>
164      * 
165      * Any snapshot returned by this method will meet the same
166      * contract as the object returned by passing the result of
167      * {@link #getEntry getEntry} to {@link JavaSpace#snapshot
168      * JavaSpace.snapshot}.<p>
169      *
170      * Generally there is a cost associated with calling the
171      * <code>JavaSpace.snapshot</code> method and thus creating a
172      * snapshot using that method is usually only worthwhile if
173      * the resulting snapshot is used more than once. The cost of
174      * invoking this method should be low and should be worthwhile
175      * even if the resulting snapshot is used only once. <p>
176      * 
177      * @return a <em>snapshot</em> of the {@link Entry} whose
178      *         transition triggered this event 
179      */
180     public abstract Entry getSnapshot();
181 
182     /**
183      * Returns <code>true</code> if the transition that triggered
184      * this event was a transition from {@linkplain
185      * JavaSpace05#registerForAvailabilityEvent <em>invisible to
186      * visible</em>} as well as a transition from unavailable to
187      * available, and <code>false</code> otherwise. <p>
188      *
189      * @return <code>true</code> if the transition that triggered
190      * this event was a transition from invisible to visible as
191      * well as a transition from unavailable to available, and
192      * <code>false</code> otherwise 
193      */
194     public boolean isVisibilityTransition() {
195 	return visibilityTransition;
196     }
197 }
