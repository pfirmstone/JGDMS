18  package net.jini.space;
19  
20  import java.rmi.RemoteException;
21  import java.rmi.NoSuchObjectException;
22  import net.jini.core.entry.Entry;
23  import net.jini.core.entry.UnusableEntryException;
24  import net.jini.core.lease.Lease;
25  import net.jini.core.lease.UnknownLeaseException;
26  import net.jini.core.constraint.RemoteMethodControl;
27  
28  /**
29   * A collection of {@link Entry} instances to be incrementally
30   * returned from a JavaSpaces service. Some operations on a
31   * space must return more entries than can be conveniently
32   * returned by a single call, generally because returning all the
33   * entries in one result would consume too many resources in the
34   * client or introduce too much latency before the first entry
35   * could be processed. In these cases, match sets are returned to
36   * incrementally return the necessary entries. A
37   * <code>MatchSet</code> instance is a proxy object that can be
38   * used to access a match set created by a space.<p>
39   *
40   * Typically match sets are created by some factory method on the
41   * space (for example, {@link JavaSpace05#contents
42   * JavaSpace05.contents}) which returns a <code>MatchSet</code>
43   * instance that will serve as a proxy to the match set. The match
44   * set will initially contain some population of entries specified
45   * by the operation that created it. These entries can be
46   * retrieved by calling {@link #next next}. A successful call to
47   * <code>next</code> will remove the returned <code>Entry</code>
48   * from the match set.  Match sets can end up in one of two
49   * terminal states, <em>exhausted</em> or <em>invalidated</em>.
50   * Under what conditions a match set enters one of these states is
51   * specified by the operation that created it. An exhausted match
52   * set is empty and will have no more entries added. Calling
53   * <code>next</code> on an exhausted match set must always return
54   * <code>null</code>. Calling <code>next</code> on an invalidated
55   * match set may return a non-<code>null</code> value, or it may
56   * throw one of the allowed exceptions.  In particular it may
57   * throw {@link NoSuchObjectException} to indicate that the match
58   * set has been invalidated. Once <code>next</code> throws a
59   * <code>NoSuchObjectException</code>, all future
60   * <code>next</code> calls on that instance must also throw
61   * <code>NoSuchObjectException</code>. Calling <code>next</code>
62   * on an invalidated match set must never return
63   * <code>null</code>. Implementations must not add entries to an
64   * invalidated match set. <p>
65   *
66   * Between the time a match set is created and the time it reaches a
67   * terminal state, entries may be added by the space. However, an
68   * <code>Entry</code> that is removed by a <code>next</code> call
69   * must not be added back to a match set (though if there is a
70   * distinct but equivalent entry in the space it may be
71   * added). The space may also remove entries independent of
72   * <code>next</code> calls. The conditions under which entries
73   * will be removed independent of <code>next</code> calls or added
74   * after the initial creation of the match set are specified by
75   * the operation that created the match set. <p>
76   *
77   * If there is a possibility that a match set may become
78   * invalidated, it must be leased and an appropriate proxy must be
79   * returned by the {@link #getLease getLease} method. If there is
80   * no possibility that the match set will become invalidated,
81   * implementations should not lease the match set. If a match set
82   * is not leased, then <code>getLease</code> must return
83   * <code>null</code>. <p>
84   *
85   * An active lease on a match set serves as a hint to the space
86   * that the client is still interested in the match set, and as a
87   * hint to the client that the match set is still functioning.
88   * However, implementations are allowed to invalidate match sets
89   * associated with active leases and to unilaterally cancel leases
90   * associated with functioning match sets. If a match set is
91   * leased and the lease is active, implementations should, to the
92   * best of their ability, maintain the match set and not invalidate
93   * it. There are cases, however, where this may not be possible in
94   * particular, it is not expected that implementations will
95   * maintain match sets across crashes. If the lease expires or is
96   * canceled, an implementation should invalidate the match
97   * set. An implementation must unilaterally cancel a match set's
98   * lease if the match set is invalidated. An implementation may
99   * unilaterally cancel the lease at other times without necessarily
100  * invalidating the match set. Clients should <em>not</em> assume
101  * that the resources associated with a leased match set will be
102  * freed if the match set reaches the exhausted state, and should
103  * instead cancel the lease.
104  *
105  * This interface is not a remote interface; though in general a
106  * <code>MatchSet</code> is a proxy for some remote match set,
107  * only the <code>next</code> method is considered to be a remote
108  * method, and as outlined in its {@linkplain #next description}
109  * it deviates in a number of ways from normal Java(TM) Remote
110  * Method Invocation remote method semantics.
111  *
112  * @since 2.1 */
113 public interface MatchSet {
114     /**
115      * Removes one <code>Entry</code> from the match set and
116      * returns a copy to the caller. Returns <code>null</code> if
117      * the match set is empty. The returned <code>Entry</code>
118      * must be unmarshalled in accordance with the <a 
119      * href=http://www.jini.org/standards/index.html>Jini Entry
120      * Specification</a>.<p>
121      *
122      * A given invocation of this method may perform remote
123      * communications, but generally the <code>next</code> method
124      * is not expected to have {@linkplain
125      * net.jini.core.constraint remote method constraints} that
126      * can vary from invocation to invocation. Instead the set of
127      * constraints associated with the <code>next</code> method
128      * will be fixed at the time the match set was created, even
129      * if this object implements an interface like {@link
130      * RemoteMethodControl} that would otherwise allow the set of
131      * constraints associated with the <code>next</code> method to
132      * be altered.<p>
133      *
134      * @return an <code>Entry</code> from the match set, or
135      *         <code>null</code> if the match set is empty
136      * @throws UnusableEntryException if the entry removed from
137      *         the match set could not be unmarshalled
138      * @throws RemoteException if a communication error occurs. If a
139      *         <code>RemoteException</code> is thrown, no
140      *         <code>Entry</code> was removed from the match set
141      *         because of this call 
142      */
143     public Entry next() throws RemoteException, UnusableEntryException;
144 
145     /**
146      * Returns a proxy to the {@link Lease} associated with this
147      * match set, or <code>null</code> if this match set is not
148      * leased.
149      * @return a proxy for the match set's lease, or
150      * <code>null</code> if there is no lease associated with this
151      * match set  
152      */
153     public Lease getLease();
154 
155     /**
156      * Returns a <em>snapshot</em> of the {@link Entry} removed by
157      * the last call to {@link #next next}. Snapshots are defined
158      * in section JS.2.6 of the <a
159      * href=http://www.jini.org/standards/index.html>;
160      * JavaSpaces Service Specification</a> and are an
161      * alternative representation of a given <code>Entry</code>
162      * produced by a particular space for use with that same
163      * space. Passing a snapshot to a space is generally more
164      * efficient than passing the original representation.<p>
165      * 
166      * Any snapshot returned by this method will meet the same
167      * contract as the object returned by passing the result of
168      * the last <code>next</code> invocation to {@link
169      * JavaSpace#snapshot JavaSpace.snapshot}.<p>
170      *
171      * Generally there is a cost associated with calling the
172      * <code>JavaSpace.snapshot</code> method. Thus creating a
173      * snapshot using that method is usually only worthwhile if
174      * the resulting snapshot is used more than once. The cost of
175      * invoking this method should be low and should be worthwhile
176      * even if the resulting snapshot is used only once. <p>
177      *
178      * @return a <em>snapshot</em> of the {@link Entry} removed
179      *         from the match set by the last call to {@link #next next}
180      * @throws IllegalStateException if the last call to
181      *         <code>next</code> did not remove an
182      *         <code>Entry</code> from the match set, or no call
183      *         to <code>next</code> has been made
184      */
185     public Entry getSnapshot();
186 }
