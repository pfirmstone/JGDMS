18  package net.jini.lookup.ui;
19  
20  /**
21   * UI role interface implemented by About UIs, which enable users to
22   * view (or in some way experience) information about a service.
23   * If a UI descriptor's UI factory produces a UI that implements
24   * this interface (i.e., produces a About UI), the UI descriptor's
25   * <CODE>role</CODE> field must reference a <CODE>String</CODE> with the value
26   * <CODE>"net.jini.lookup.ui.AboutUI"</CODE>.
27   *
28   * <P>
29   * The first parameter of any factory method declared in a UI factory type is an
30   * object called the "role object."
31   * Any factory method that produces an About UI must accept as the role object the
32   * service item (the <CODE>net.jini.core.lookup.ServiceItem</CODE>) of the service
33   * with which the About UI is associated.
34   *
35   * <P>
36   * About UIs should allow clients to configure them before they
37   * begin. For example, About UIs produced by <CODE>FrameFactory</CODE>,
38   * <CODE>JFrameFactory</CODE>, <CODE>WindowFactory</CODE>
39   * and <CODE>JWindowFactory</CODE> (all members of the <CODE>net.jini.lookup.ui.factory</CODE> package)
40   * should not be visible when they are returned from the factory method. This allows clients to set
41   * the UI's position and size, for example, before making the UI
42   * visible by invoking <CODE>setVisible(true)</CODE> on the UI.
43   *
44   * <P>
45   * A client should be
46   * able to invoke a About UI factory method multiple times sequentially. In other words, if a user
47   * uses a service via a About UI, then says exit, then double clicks once again on the service icon,
48   * the client can just simply invoke a UI factory method again, and get another About UI for the same
49   * service. About UIs, therefore, should be written so that they work no matter what state
50   * the service object happens to be in when the About UI is created.
51   *
52   * <P>
53   * It is recommended that clients use multiple About UIs for the same service only sequentially, and
54   * avoid creating multiple About UIs for the same service that operate concurrently with one another.
55   * But because some clients may create and use multiple About UIs at the same time for the same service,
56   * providers of services and About UIs should program defensively, to ensure that multiple About UIs
57   * for the same service at the same time will all work together concurrently.
58   *
59   * @author Bill Venners
60   */
61  
62  public interface AboutUI {
63  
64      /**
65       * Convenience constant to use in the <CODE>role</CODE>
66       * field of <CODE>UIDescriptor</CODE>s for AboutUI role UIs.
67       * The value of this constant is <CODE>"net.jini.lookup.ui.AboutUI"</CODE>.
68       */
69      String ROLE = "net.jini.lookup.ui.AboutUI";
70  }
