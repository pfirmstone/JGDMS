18  package net.jini.core.transaction.server;
19  
20  import net.jini.core.transaction.*;
21  import net.jini.core.lease.LeaseDeniedException;
22  import java.rmi.RemoteException;
23  
24  /**
25   * The interface used for managers of the two-phase commit protocol for
26   * nestable transactions.  All nestable transactions must have a
27   * transaction manager that runs this protocol.
28   *
29   * @author Sun Microsystems, Inc.
30   *
31   * @see NestableServerTransaction
32   * @see TransactionParticipant
33   *
34   * @since 1.0
35   */
36  public interface NestableTransactionManager extends TransactionManager {
37      /**
38       * Begin a nested transaction, with the specified transaction as parent.
39       *
40       * @param parentMgr the manager of the parent transaction
41       * @param parentID the id of the parent transaction
42       * @param lease the requested lease time for the transaction
43       * @return the created transaction and the lease granted
44       * 
45       * @throws UnknownTransactionException if the transaction 
46       *         is unknown to the transaction manager, either 
47       *         because the transaction ID is incorrect or because the 
48       *         transaction is complete and its state has been 
49       *         discarded by the manager.
50       * @throws CannotJoinException if the transaction is known 
51       *         to the manager but is no longer active.
52       * @throws LeaseDeniedException if this manager is unwilling to 
53       *         grant the requested lease time
54       * @throws RemoteException if there is a communication error
55       *
56       */
57      TransactionManager.Created create(NestableTransactionManager parentMgr,
58  				      long parentID, long lease)
59          throws UnknownTransactionException, CannotJoinException,
60  	       LeaseDeniedException, RemoteException;
61   
62      /**
63       * Promote the listed participants into the specified transaction.
64       * This method is for use by the manager of a subtransaction when the
65       * subtransaction commits.  At this point, all participants of the
66       * subtransaction must become participants in the parent transaction.
67       * Prior to this point, the subtransaction's manager was a participant
68       * of the parent transaction, but after a successful promotion it need
69       * no longer be one (if it was not itself a participant of the
70       * subtransaction), and so it may specify itself as a participant to
71       * drop from the transaction.  Otherwise, participants should not be
72       * dropped out of transactions.  For each promoted participant, the
73       * participant's crash count is stored in the corresponding element of
74       * the <code>crashCounts</code> array.
75       *
76       * @param id the id of the parent transaction
77       * @param parts the participants being promoted to the parent
78       * @param crashCounts the crash counts of the participants
79       * @param drop the manager to drop out, if any
80       *
81       * @throws CrashCountException if the crash count provided 
82       *         for at least one participant differs from the crash  
83       *         count in a previous invocation of the same pairing  
84       *         of participant and transaction
85       * @throws UnknownTransactionException if a transaction 
86       *         is unknown to the transaction manager, either 
87       *         because the transaction ID is incorrect or because the 
88       *         transaction is complete and its state has been 
89       *         discarded by the manager.
90       * @throws CannotJoinException if a transaction is known 
91       *         to the manager but is no longer active.
92       * @throws RemoteException if there is a communication error
93       * @see TransactionManager#join
94       */
95      void promote(long id, TransactionParticipant[] parts,
96  		 long[] crashCounts, TransactionParticipant drop)
97         	throws UnknownTransactionException, CannotJoinException,
98  	       CrashCountException, RemoteException;
99  }
