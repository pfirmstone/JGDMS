19  package net.jini.core.constraint;
20  
21  /**
22   * Represents an invariant constraint on {@link java.io.ObjectInput} and 
23   * {@link java.io.ObjectOutput}, covering data transmitted in band as part
24   * of the remote call itself. 
25   * If an invariant violation on in-band data is detected during a remote
26   * call, a {@link java.io.InvalidObjectException} will be thrown, construction of the Object
27   * in question will fail atomically
28   * (in the client or in the server, depending on which side detected the violation).
29   * Invariant validation is fine grained, which means every Object in the 
30   * graph must have its invariants satisfied, prior to java.lang.Object's 
31   * constructor being called.  This ensures that an object instance cannot be
32   * created if invariants are not satisfied.
33   * <p>
34   * Java's standard {@link java.io.ObjectOutputStream} and {@link java.io.ObjectInputStream}
35   * check invariants after construction, circular links and finalizers allow
36   * an attacker to obtain references to partially constructed objects before
37   * invariants have been checked.
38   * <p>
39   * Serialization for this class is guaranteed to produce instances that are
40   * comparable with <code>==</code>.
41   *
42   * @since 3.0
43   */
44  public enum AtomicInputValidation implements InvocationConstraint {
45      YES, 
46      NO;
47      
48      @Override
49      public String toString(){
50  	return "AtomicInputValidation." + super.toString();
51      }
52  }
