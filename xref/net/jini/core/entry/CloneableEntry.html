19  package net.jini.core.entry;
20  
21  /**
22   * Required by service utilities to avoid sharing internal state. All cloned
23   * instances should be deep copies.
24   * 
25   * @since 3.0.0
26   */
27  public interface CloneableEntry extends Entry, Cloneable {
28  
29      /**
30       * Clone has been implemented in AbstractEntry to allow utilities such as
31       * <code> net.jini.lookup.ServiceDiscoveryManager </code> to avoid sharing 
32       * internally stored mutable Entry instances with client code.
33       * 
34       * Entry's that have mutable fields, for example arrays or collections,
35       * should override this method, call super.clone(), then safely copy
36       * any mutable fields before returning the Entry clone.
37       * 
38       * @return a clone of the original Entry
39       * @since 3.0.0
40       */
41      public Entry clone();
42      
43  }
