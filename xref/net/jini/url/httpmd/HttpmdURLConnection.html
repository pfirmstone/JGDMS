19  package net.jini.url.httpmd;
20  
21  import java.io.IOException;
22  import java.io.InputStream;
23  import java.net.HttpURLConnection;
24  import java.net.MalformedURLException;
25  import java.net.URL;
26  import java.security.MessageDigest;
27  import java.security.NoSuchAlgorithmException;
28  
29  /**
30   * An HTTP URL connection for HTTPMD URLs.
31   *
32   * @author Sun Microsystems, Inc.
33   * 
34   */
35  class HttpmdURLConnection extends DelegatingHttpURLConnection {
36  
37      /** The message digest algorithm. */ 
38      private final String algorithm;
39  
40      /** The expected message digest for the URL data. */
41      private final byte[] expectedDigest;
42  
43      /** The URL specifying the location of the data. */
44      private final URL content;
45  
46      /** Creates an HTTP URL connection for an HTTPMD URL. */
47      HttpmdURLConnection(URL url) throws IOException, MalformedURLException {
48  	super(url);
49  
50  	/* Check syntax of message digest parameter */
51  	String path = url.getPath() == null ? "" : url.getPath();
52  	int semiIndex = path.lastIndexOf(';');
53  	if (semiIndex < 0) {
54  	    throw new MalformedURLException(
55  		"Message digest parameter is missing");
56  	}
57  	int equalsIndex = path.indexOf('=', semiIndex);
58  	if (equalsIndex < 0) {
59  	    throw new MalformedURLException(
60  		"Message digest parameter is missing a '='");
61  	}
62  	algorithm = path.substring(semiIndex + 1, equalsIndex);
63  	try {
64  	    MessageDigest.getInstance(algorithm);
65  	} catch (NoSuchAlgorithmException e) {
66  	    MalformedURLException t = new MalformedURLException(
67  		"Message digest parameter algorithm is not found: " +
68  		algorithm);
69  	    t.initCause(e);
70  	    throw t;
71  	}
72  	String digest = path.substring(equalsIndex + 1);
73  	int comment = digest.indexOf(',');
74  	if (comment >= 0) {
75  	    for (int i = digest.length(); --i > comment; ) {
76  		char c = digest.charAt(i);
77  		if (!HttpmdUtil.commentChar(c)) {
78  		    throw new MalformedURLException(
79  			"Comment contains illegal character: '" + c + "'");
80  		}
81  	    }
82  	    digest = digest.substring(0, comment);
83  	}
84  	int length = digest.length();
85  	if (length == 0) {
86  	    throw new MalformedURLException(
87  		"Message digest parameter digest is empty");
88  	}
89  	try {
90  	    expectedDigest = HttpmdUtil.stringDigest(digest);
91  	} catch (NumberFormatException e) {
92  	    MalformedURLException t = new MalformedURLException(
93  		"Message digest parameter has invalid format for digest: " +
94  		digest);
95  	    t.initCause(e);
96  	    throw t;
97  	}
98  	try {
99  	    content = new URL(
100 		"http", url.getHost(), url.getPort(),
101 		path.substring(0, semiIndex) +
102 		(url.getQuery() == null ? "" : url.getQuery()) +
103 		(url.getRef() == null ? "" : url.getRef()));
104 	} catch (MalformedURLException e) {
105 	    MalformedURLException t = new MalformedURLException(
106 		"Problem with content location");
107 	    t.initCause(e);
108 	    throw t;
109 	}
110 
111 	delegateConnection = (HttpURLConnection) content.openConnection();
112     }
113 
114     /** Returns our URL, not the one for the HTTP connection. */
115     public URL getURL() {
116 	return url;
117     }
118 
119     /**
120      * Returns an input stream that uses MdInputStream to check that the input
121      * has the expected message digest.
122      */
123     public InputStream getInputStream() throws IOException {
124 	try {
125 	    return new MdInputStream(url,
126 				     delegateConnection.getInputStream(),
127 				     MessageDigest.getInstance(algorithm),
128 				     expectedDigest);
129 	} catch (NoSuchAlgorithmException e) {
130 	    IOException t = new IOException(
131 		"Message digest algorithm not found: " + algorithm);
132 	    t.initCause(e);
133 	    throw t;
134 	}
135     }
136 }
