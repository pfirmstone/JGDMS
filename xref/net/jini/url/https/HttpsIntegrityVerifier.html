19  package net.jini.url.https;
20  
21  import java.net.URL;
22  import net.jini.security.IntegrityVerifier;
23  import aQute.bnd.annotation.headers.RequireCapability;
24  import aQute.bnd.annotation.headers.ProvideCapability;
25  
26  /**
27   * Integrity verifier for HTTPS URLs. This class is intended to be specified
28   * in a resource to configure the operation of
29   * {@link net.jini.security.Security#verifyCodebaseIntegrity
30   * Security.verifyCodebaseIntegrity}.
31   * <p>
32   * This verifier assumes the HTTPS URL protocol handler is configured to
33   * provide adequate data integrity and server authentication.
34   *
35   * @author Sun Microsystems, Inc.
36   * @since 2.0
37   */
38  @RequireCapability(
39  	ns="osgi.extender",
40  	filter="(osgi.extender=osgi.serviceloader.registrar)")
41  @ProvideCapability(
42  	ns="osgi.serviceloader",
43  	name="net.jini.security.IntegrityVerifier")
44  public class HttpsIntegrityVerifier implements IntegrityVerifier {
45      /**
46       * Returns <code>true</code> if the specified URL uses the "https"
47       * protocol; returns <code>false</code> otherwise.
48       *
49       * @throws NullPointerException {@inheritDoc}
50       */
51      public boolean providesIntegrity(URL url) {
52  	return "https".equals(url.getProtocol());
53      }
54  }
