16  package net.jini.loader;
17  
18  import java.io.IOException;
19  import java.util.Collection;
20  import java.util.Iterator;
21  import net.jini.export.CodebaseAccessor;
22  import net.jini.io.MarshalledInstance;
23  import net.jini.io.context.IntegrityEnforcement;
24  
25  /**
26   * A provider used by @{link org.apache.river.api.io.AtomicMarshalInputStream}
27   * to allow third party smart proxy's found in a stream to have 
28   * {@link net.jini.core.constraint.MethodConstraints} applied by the client
29   * to {@link CodebaseAccessor} methods and allow the client to obtain
30   * current information about the code base
31   * required by the smart proxy {@link MarshalledInstance} and 
32   * provide it with a ClassLoader, unique
33   * to it's {@link java.lang.reflect.InvocationHandler} instance.
34   * 
35   * This interface also exists to avoid code base annotation loss when smart 
36   * proxy instances are re-serialized, by consulting the original
37   * export node, using {@link CodebaseAccessor}.
38   * 
39   * @see org.apache.river.api.io.AtomicMarshalOutputStream
40   * @see org.apache.river.api.io.AtomicMarshalInputStream
41   * @see org.apache.river.api.io.AtomicMarshalledInstance
42   * 
43   * @author peter
44   */
45  public interface ProxyCodebaseSpi {
46  
47      /**
48       * 
49       * @param bootstrapProxy an instance of {@link net.jini.core.constraint.RemoteMethodControl} and
50       * {@link net.jini.security.proxytrust.TrustEquivalence}
51       * @param proxy marshalled proxy to be deserialized.
52       * @param parentLoader default loader of the current stream.
53       * @param verifierLoader verifier loader of the current stream.
54       * @param context the {@link net.jini.io.ObjectStreamContext} may contain
55       * client {@link net.jini.core.constraint.MethodConstraints} to apply to
56       * {@link CodebaseAccessor} methods.  The context should also be passed
57       * to the {@link MarshalledInstance} get method.
58       * @return unmarshalled proxy instance.
59       * @throws IOException if an 
60       *         <code>IOException</code> occurs while deserializing the
61       *         object from its internal representation
62       * @throws ClassNotFoundException if any classes necessary
63       *         for reconstructing the contained object can not
64       *         be found or if context contains integrity constraints
65       *		and the integrity of the contained object's codebase
66       *		cannot be confirmed.
67       */
68      public Object resolve(
69  	    CodebaseAccessor bootstrapProxy,
70  	    MarshalledInstance proxy,
71  	    ClassLoader parentLoader,
72  	    ClassLoader verifierLoader,
73  	    Collection context) throws IOException, ClassNotFoundException;
74      
75      /**
76       * Used in modular environments when class visibility is expected to
77       * be identical at the local and remote endpoints.
78       * @param serviceClass the class to check for visibility.
79       * @param streamLoader ClassLoader to check class visibility from.
80       * @return true if the class should be substituted, false if not.
81       */
82      public boolean substitute(Class serviceClass, ClassLoader streamLoader);
83      
84      public static class Util {
85  	
86  	/**
87  	 * Utility method that tests whether to check for Integrity.
88  	 * 
89  	 * @param context the object stream context.
90  	 * @return true if codebase integrity should be verified.
91  	 */
92  	public static boolean verifyCodebaseIntegrity(Collection context) {
93  	    if (context != null){
94  		Iterator it = context.iterator();
95  		while (it.hasNext()){
96  		    Object next = it.next();
97  		    if (next instanceof IntegrityEnforcement){
98  			return ((IntegrityEnforcement) next).integrityEnforced();
99  		    }
100 		}
101 	    }
102 	    return false;
103 	}
104     }
