74       * Determines if the URL is pointing to a directory.

76      private static boolean isDirectory(URL url) {
77          String file = url.getFile();
78          return (file.length() > 0 && file.charAt(file.length() - 1) == File.separatorChar);
79      }

81      @Override
82      public Object resolve(CodebaseAccessor bootstrapProxy,
83  			MarshalledInstance smartProxy,
84  			final ClassLoader parent,
85  			ClassLoader verifier,
86  			Collection context) 
87  	    throws IOException, ClassNotFoundException
88      {
89  	if (context == null) throw new NullPointerException(
90  		"stream context cannot be null");
91  	Iterator it = context.iterator();
92  	MethodConstraints mc = null;
93  	IntegrityEnforcement integrityEnforcement = null;
94  	while(it.hasNext()){
95  	    Object o = it.next();
96  	    if (o instanceof MethodConstraints){
97  		mc = (MethodConstraints) o;
98  	    } else if (o instanceof IntegrityEnforcement){
99  		integrityEnforcement = (IntegrityEnforcement) o;
100 	    }
101 	}
102 	if (mc != null || integrityEnforcement != null){
103 	    if (!(bootstrapProxy instanceof RemoteMethodControl)) 
104 		throw new IOException(
105 		    "bootstrap proxy must be instance of RemoteMethodControl for client to apply method constraints");
106 	    bootstrapProxy = (CodebaseAccessor) 
107 		((RemoteMethodControl) bootstrapProxy).setConstraints(mc); // MinPrincipal happens here.
108 	}   
109 	ClassLoader loader;
110 	final String path = bootstrapProxy.getClassAnnotation();
111 	String loaderPath = PreferredClassProvider.getLoaderAnnotation(parent, false, null);
112 	if (path != null && path.equals(loaderPath)){
113 	    /**
114 	     * This check prevents us accidentally creating a new ClassLoader for
115 	     * a returned proxy that can be resolved by the parent ClassLoader
116 	     * but where that loader hasn't been cached.
117 	     */
118 	    loader = parent;
119 	} else {
120 	    Uri [] codebases = PreferredClassProvider.pathToURIs(path);
121 	    Key loaderKey = new Key(
122 				Proxy.getInvocationHandler(bootstrapProxy),
123 				Arrays.asList(codebases)
124 			    );
125 	    loader = cache.get(loaderKey);
126 	    if (loader == null){
127 		byte [] encodedCerts = bootstrapProxy.getEncodedCerts();
128 		if ((encodedCerts == null 
129 		    || encodedCerts.length == 0 )
130 		    && integrityEnforcement != null
131 		    && integrityEnforcement.integrityEnforced())
132 		{
133 		    Security.verifyCodebaseIntegrity(path, verifier);
134 		} else if (encodedCerts != null && encodedCerts.length > 0) {
135 		    // Although we trust the bootstrapProxy now, if we require validation,
136 		    // we must check the jar file has been signed.
137 		    try {
138 			URL [] codebase = PreferredClassProvider.asURL(codebases);
139 			String certFactoryType = bootstrapProxy.getCertFactoryType();
140 			String certPathEncoding = bootstrapProxy.getCertPathEncoding();
141 			CertificateFactory factory =
142 				CertificateFactory.getInstance(certFactoryType);
143 			CertPath certPath = factory.generateCertPath(
144 				new ByteArrayInputStream(encodedCerts), certPathEncoding);
145 			Collection<? extends Certificate> certs = certPath.getCertificates();
146 			for (int i = 0, l = codebase.length; i < l; i++){
147 			    URL searchURL = createSearchURL(codebase[i]);
148 			    URL jarURL = ((JarURLConnection) searchURL
149 				.openConnection()).getJarFileURL();
150 			    JarURLConnection juc = (JarURLConnection) new URL(
151 				    "jar", "", //$NON-NLS-1$ //$NON-NLS-2$
152 				    jarURL.toExternalForm() + "!/").openConnection(); //$NON-NLS-1$
153 			    juc.connect();
154 			    InputStream in = juc.getInputStream();
155 			    byte [] bytes = new byte[1024];
156 			    int bytesRead = 0;
157 			    // reading in the entire jar file will check it's validity.
158 			    // it will also be cached.
159 			    do { // keep reading until we reach end of stream.
160 				bytesRead = in.read(bytes);
161 			    } while (bytesRead == 1024);
162 			    // We should be able to read certs now, confirming the jar 
163 			    // has been verified.
164 			    Certificate [] certificates = juc.getCertificates();
165 			    if (certs == null){
166 				throw new SecurityException("jar file invalid");
167 			    }
168 			    // Check our certs match.
169 			    HashSet<Certificate> actualCerts 
170 				    = new HashSet<Certificate>(Arrays.asList(certificates));
171 			    HashSet<Certificate> requiredCerts = new HashSet<Certificate>(certs);
172 			    if (!actualCerts.containsAll(requiredCerts)){
173 				throw new SecurityException("certificates don't match");
174 			    }
175 			}
176 		    } catch (CertificateException ex) {
177 			throw new IOException("Problem creating signer certificates", ex);
178 		    } 
179 		}
180 
181 		try {
182 		    loader = AccessController.doPrivileged(
183 			    new PrivilegedExceptionAction<ClassLoader>(){
184 
185 				public ClassLoader run() throws IOException {
186 				    Thread currentThread = Thread.currentThread();
187 				    ClassLoader context = currentThread.getContextClassLoader();
188 				    try {
189 					currentThread.setContextClassLoader(parent);
190 					return ClassLoading.getClassLoader(path);
191 				    } finally {
192 					if (context != null) currentThread.setContextClassLoader(context);
193 				    }
194 
195 				}
196 			    }
197 		    );
198 		} catch (PrivilegedActionException ex) {
199 		    Exception e = ex.getException();
200 		    if (e instanceof IOException) throw (IOException ) e;
201 		    if (e instanceof SecurityException) throw (SecurityException) e;
202 		    if (e instanceof RuntimeException) throw (RuntimeException) e;
203 		}    
204 		ClassLoader existed = cache.putIfAbsent(loaderKey, loader);
205 		if (existed != null) loader = existed;
206 	    }
207 	}
208 	Object sp = smartProxy.get(loader, true, verifier, context);
209 	// Don't set constraints on smart proxy.
210 //	if (sp instanceof RemoteMethodControl && mc != null)
211 //	    sp = ((RemoteMethodControl)sp).setConstraints(mc);
212 	return sp;
213     }

215     /**
216      * Returns an URL that will be checked if it contains the class or resource.
217      * If the file component of the URL is not a directory, a Jar URL will be
218      * created.
219      *
220      * @return java.net.URL a test URL
221      */
222     private URL createSearchURL(URL url) throws MalformedURLException {
223         if (url == null) return url;
224         String protocol = url.getProtocol();
225         if (isDirectory(url) || protocol.equals("jar")) { //$NON-NLS-1$
226             return url;
227         }
228 	return new URL("jar", "", //$NON-NLS-1$ //$NON-NLS-2$
229 		-1, url.toString() + "!/"); //$NON-NLS-1$
230     }

249     private static class Key {
250 	private final InvocationHandler handler;
251 	private final List<Uri> codebase;
252 	private final int hashcode;
253 	
254 	Key(InvocationHandler h, List<Uri> codebase){
255 	    this.handler = h;
256 	    this.codebase = codebase;
257 	    int hash = 5;
258 	    hash = 73 * hash + (this.handler != null ? this.handler.hashCode() : 0);
259 	    hash = 73 * hash + (this.codebase != null ? this.codebase.hashCode() : 0);
260 	    this.hashcode = hash;
261 	}
262 
263 	@Override
264 	public int hashCode() {
265 	    return hashcode;
266 	}
267 	
268 	@Override
269 	public boolean equals(Object o){
270 	    if (!(o instanceof Key)) return false;
271 	    if (!handler.equals(((Key)o).handler)) return false;
272 	    return codebase.equals(((Key)o).codebase);
273 	}
