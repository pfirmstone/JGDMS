19  package net.jini.loader.pref;
20  
21  import java.rmi.server.RMIClassLoader;
22  import net.jini.loader.DownloadPermission;
23  import aQute.bnd.annotation.headers.RequireCapability;
24  import aQute.bnd.annotation.headers.ProvideCapability;
25  
26  /**
27   * An <code>RMIClassLoader</code> provider that supports preferred
28   * classes and enforces {@link DownloadPermission}.
29   *
30   * <p>See the {@link RMIClassLoader} specification for information
31   * about how to install and configure the <code>RMIClassLoader</code>
32   * service provider.
33   *
34   * @author Sun Microsystems, Inc.
35   * @since 2.0
36   **/
37  @RequireCapability(
38  	ns="osgi.extender",
39  	filter="(osgi.extender=osgi.serviceloader.registrar)")
40  @ProvideCapability(
41  	ns="osgi.serviceloader",
42  	name="java.rmi.server.RMIClassLoaderSpi")
43  public class RequireDlPermProvider extends PreferredClassProvider {
44  
45      /**
46       * Creates a new <code>RequireDlPermProvider</code>.
47       *
48       * <p>This constructor is used by the {@link RMIClassLoader}
49       * service provider location mechanism when
50       * <code>RequireDlPermProvider</code> is configured as the
51       * <code>RMIClassLoader</code> provider class.
52       *
53       * <p>This constructor passes <code>true</code> to the superclass
54       * constructor that has a <code>boolean</code> parameter.
55       **/
56      public RequireDlPermProvider() {
57  	super(true);
58      }
59  }
