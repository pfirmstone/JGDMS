67  /**
68   * A class loader that supports preferred classes.
69   *
70   * <p>A preferred class is a class that is to be loaded by a class
71   * loader without the loader delegating to its parent class loader
72   * first.  Resources may also be preferred.
73   *
74   * <p>Like {@link java.net.URLClassLoader},
75   * <code>PreferredClassLoader</code> loads classes and resources from
76   * a search path of URLs.  If a URL in the path ends with a
77   * <code>'/'</code>, it is assumed to refer to a directory; otherwise,
78   * the URL is assumed to refer to a JAR file.
79   *
80   * <p>The location of the first URL in the path can contain a
81   * <i>preferred list</i> for the entire path.  A preferred list
82   * declares names of certain classes and other resources throughout
83   * the path as being <i>preferred</i> or not.  When a
84   * <code>PreferredClassLoader</code> is asked to load a class or
85   * resource that is preferred (according to the preferred list) and
86   * the class or resource exists in the loader's path of URLs, the
87   * loader will not delegate first to its parent class loader as it
88   * otherwise would do; instead, it will attempt to load the class or
89   * resource from its own path of URLs only.
90   *
91   * <p>The preferred list for a path of URLs, if one exists, is located
92   * relative to the first URL in the path.  If the first URL refers to
93   * a JAR file, then the preferred list is the contents of the file
94   * named <code>"META-INF/PREFERRED.LIST"</code> within that JAR file.
95   * If the first URL refers to a directory, then the preferred list is
96   * the contents of the file at the location
97   * <code>"META-INF/PREFERRED.LIST"</code> relative to that directory
98   * URL.  If there is no preferred list at the required location, then
99   * no classes or resources are preferred for the path of URLs.  A
100  * preferred list at any other location (such as relative to one of
101  * the other URLs in the path) is ignored.
102  *
103  * <p>Note that a class or resource is only considered to be preferred
104  * if the preferred list declares the name of the class or resource as
105  * being preferred and the class or resource actually exists in the
106  * path of URLs.
107  *
108  * <h3>Preferred List Syntax</h3>
109  *
110  * A preferred list is a UTF-8 encoded text file, with lines separated
111  * by CR&nbsp;LF, LF, or CR (not followed by an LF).  Multiple
112  * whitespace characters in a line are equivalent to a single
113  * whitespace character, and whitespace characters at the beginning or
114  * end of a line are ignored.  If the first non-whitespace character
115  * of a line is <code>'#'</code>, the line is a comment and is
116  * equivalent to a blank line.
117  *
118  * <p>The first line of a preferred list must contain a version
119  * number in the following format:
120  *
121  * <pre>
122  *     PreferredResources-Version: 1.<i>x</i>
123  * </pre>
124  *
125  * This specification defines only version 1.0, but
126  * <code>PreferredClassLoader</code> will parse any version
127  * 1.<i>x</i>, <i>x</i>&lt;=0 with the format and semantics specified
128  * here.
129  *
130  * <p>After the version number line, a preferred list comprises an
131  * optional default preferred entry followed by zero or more named
132  * preferred entries.  A preferred list must contain either a default
133  * preferred entry or at least one named preferred entry.  Blank lines
134  * are allowed before and after preferred entries, as well as between
135  * the lines of a named preferred entry.
136  *
137  * <p>A default preferred entry is a single line in the following
138  * format:
139  *
140  * <pre>
141  *     Preferred: <i>preferred-setting</i>
142  * </pre>
143  *
144  * where <i>preferred-setting</i> is a non-empty sequence of
145  * characters.  If <i>preferred-setting</i> equals <code>"true"</code>
146  * (case insensitive), then resource names not matched by any of the
147  * named preferred entries are by default preferred; otherwise,
148  * resource names not matched by any of the named preferred entries
149  * are by default not preferred.  If there is no default preferred
150  * entry, then resource names are by default not preferred.
151  *
152  * <p>A named preferred entry is two lines in the following format:
153  *
154  * <pre>
155  *     Name: <i>name-expression</i>
156  *     Preferred: <i>preferred-setting</i>
157  * </pre>
158  *
159  * where <i>name-expression</i> and <i>preferred-setting</i> are
160  * non-empty sequences of characters.  If <i>preferred-setting</i>
161  * equals <code>"true"</code> (case insensitive), then resource names
162  * that are matched by <i>name-expression</i> (and not any more
163  * specific named preferred entries) are preferred; otherwise,
164  * resource names that are matched by <i>name-expression</i> (and not
165  * any more specific named preferred entries) are not preferred.
166  * 
167  * <p>If <i>name-expression</i> ends with <code>".class"</code>, it
168  * matches a class whose binary name is <i>name-expression</i> without
169  * the <code>".class"</code> suffix and with each <code>'/'</code>
170  * character replaced with a <code>'.'</code>.  It also matches any
171  * class whose binary name starts with that same value followed by a
172  * <code>'$'</code>; this rule is intended to match nested classes
173  * that have an enclosing class of that name, so that the preferred
174  * settings of a class and all of its nested classes are the same by
175  * default.  It is possible, but strongly discouraged, to override the
176  * preferred setting of a nested class with a named preferred entry
177  * that explicitly matches the nested class's binary name.
178  *
179  * <p><i>name-expression</i> may match arbitrary resource names as
180  * well as class names, with path elements separated by
181  * <code>'/'</code> characters.
182  *
183  * <p>If <i>name-expression</i> ends with <code>"/"</code> or
184  * <code>"/*"</code>, then the entry is a directory wildcard entry
185  * that matches all resources (including classes) in the named
186  * directory.  If <i>name-expression</i> ends with <code>"/-"</code>,
187  * then the entry is a namespace wildcard entry that matches all
188  * resources (including classes) in the named directory and all of its
189  * subdirectories.
190  *
191  * <p>When more than one named preferred entry matches a class or
192  * resource name, then the most specific entry takes precedence.  A
193  * non-wildcard entry is more specific than a wildcard entry.  A
194  * directory wildcard entry is more specific than a namespace wildcard
195  * entry.  A namespace wildcard entry with more path elements is more
196  * specific than a namespace wildcard entry with fewer path elements.
197  * Given two non-wildcard entries, the entry with the longer
198  * <i>name-expression</i> is more specific (this rule is only
199  * significant when matching a class).  The order of named preferred
200  * entries is insignificant.
201  *
202  * <h3>Example Preferred List</h3>
203  *
204  * <p>Following is an example preferred list:
205  *
206  * <pre>
207  *     PreferredResources-Version: 1.0
208  *     Preferred: false
209  *
210  *     Name: com/foo/FooBar.class
211  *     Preferred: true
212  *
213  *     Name: com/foo/*
214  *     Preferred: false
215  *
216  *     Name: com/foo/-
217  *     Preferred: true
218  *
219  *     Name: image-files/*
220  *     Preferred: mumble
221  * </pre>
222  *
223  * <p>The class <code>com.foo.FooBar</code> is preferred, as well as
224  * any nested classes that have it as an enclosing class.  All other
225  * classes in the <code>com.foo</code> package are not preferred
226  * because of the directory wildcard entry.  Classes in subpackages of
227  * <code>com.foo</code> are preferred because of the namespace
228  * wildcard entry.  Resources in the directory <code>"com/foo/"</code>
229  * are not preferred, and resources in subdirectories of
230  * <code>"com/foo/"</code> are preferred.  Resources in the directory
231  * <code>"image-files/"</code> are not preferred because preferred
232  * settings other than <code>"true"</code> are interpreted as false.
233  * Classes that are in a package named <code>com.bar</code> are not
234  * preferred because of the default preferred entry.
235  *
236  * @author Sun Microsystems, Inc.
237  * @since 2.0
238  **/

319     /**
320      * Creates a new <code>PreferredClassLoader</code> that loads
321      * classes and resources from the specified path of URLs,
322      * delegates to the specified parent class loader, and uses the
323      * specified {@link URLStreamHandlerFactory} when creating new URL
324      * objects.  This constructor passes <code>factory</code> to the
325      * superclass constructor that has a
326      * <code>URLStreamHandlerFactory</code> parameter.
327      *
328      * <p>If <code>exportAnnotation</code> is not <code>null</code>,
329      * then it will be used as the return value of the loader's {@link
330      * #getClassAnnotation getClassAnnotation} method.  If
331      * <code>exportAnnotation</code> is <code>null</code>, the
332      * loader's <code>getClassAnnotation</code> method will return a
333      * space-separated list of the URLs in the specified path.  The
334      * <code>exportAnnotation</code> parameter can be used to specify
335      * so-called "export" URLs, from which other parties should load
336      * classes defined by the loader and which are different from the
337      * "import" URLs that the classes are actually loaded from.
338      *
339      * <p>If <code>requireDlPerm</code> is <code>true</code>, the
340      * loader's {@link #getPermissions getPermissions} method will
341      * require that the {@link CodeSource} of any class defined by the
342      * loader is granted {@link DownloadPermission}.
343      *
344      * @param urls the path of URLs to load classes and resources from
345      *
346      * @param parent the parent class loader for delegation
347      *
348      * @param exportAnnotation the export class annotation string to
349      * use for classes defined by this loader, or <code>null</code>
350      *
351      * @param requireDlPerm if <code>true</code>, the loader will only
352      * define classes with a {@link CodeSource} that is granted {@link
353      * DownloadPermission}
354      *
355      * @param factory the <code>URLStreamHandlerFactory</code> to use
356      * when creating new URL objects, or <code>null</code>
357      *
358      * @throws SecurityException if there is a security manager and an
359      * invocation of its {@link SecurityManager#checkCreateClassLoader
360      * checkCreateClassLoader} method fails
361      *
362      * @since 2.1
363      **/
364     public PreferredClassLoader(URL[] urls,
365 				ClassLoader parent,
366 				String exportAnnotation,
367 				boolean requireDlPerm,
368 				URLStreamHandlerFactory factory)
369     {
370 	super(urls, parent, factory);
371 	firstURL = (urls.length > 0 ? urls[0] : null);
372 	if (exportAnnotation != null) {
373 	    this.exportAnnotation = exportAnnotation;
374 	} else {
375 	    /*
376 	     * Caching the value of class annotation string here
377 	     * assumes that the protected method addURL() is never
378 	     * called on this class loader.
379 	     */
380 	    this.exportAnnotation = urlsToPath(urls);
381 	}
382 	this.requireDlPerm = requireDlPerm;
383 	if (factory != null) {
384 	    jarHandler = factory.createURLStreamHandler("jar");
385 	} else {
386 	    jarHandler = null;
387 	}
388 
389 	acc = AccessController.getContext();
390 	    
391 	/*
392 	 * Precompute the permissions required to access the loader.
393 	 */
394 	PermissionCollection perm = new Permissions();
395 	addPermissionsForURLs(urls, perm, false);
396         /*
397          * If a preferred list exists relative to the first URL of this
398          * loader's path, sets this loader's PreferredResources according
399          * to that preferred list.  If no preferred list exists relative
400          * to the first URL, leaves this loader's PreferredResources null.
401          *
402          * Throws IOException if an I/O exception occurs from which the
403          * existence of a preferred list relative to the first URL cannot
404          * be definitely determined.
405          *
406          * This method must only be invoked while synchronized on this
407          * PreferredClassLoader, and it must not be invoked again after it
408          * has completed successfully.
409          * 
410          * This was called from privileged context when it as part of a method.  
411          * Note that InputStream is not subject to deserialization attacks 
412          * like ObjectInputStream.
413          * 
414          * Also synchronization is not required as it is called from within
415          * the constructor now, this change was made to remove any possiblity
416          * of deadlock by minimising locking.
417          */
418         IOException except = null;
419         PreferredResources pref = null;
420 	Permission [] perms = null;
421 	if (firstURL != null) {
422             try {
423                 pref = AccessController.doPrivileged(
424                     new PreferredResourcesPrivilegedExceptionAction(firstURL, jarHandler)
425                 );
426             } catch (PrivilegedActionException ex) {
427                 Exception e = ex.getException();
428                 if (e instanceof IOException){
429                     except = (IOException) e;
430                 } else {
431                     except = new IOException(e);
432                 }
433             }
434 	    /*
435 	    * Use parent ClassLoader, if permission is not visible from parent
436 	    * it will remain unresolved and later be resolved by the policy provider.
437 	    */
438 	    perms = AccessController.doPrivileged(
439 		new PreferredPermissionsPrivilegedAction(
440 			urls, jarHandler, parent));
441 	}
442         exceptionWhileLoadingPreferred = except;
443         preferredResources = pref;
444 	advisoryPermissions = perms;
445         this.permissions = new LinkedList<Permission>();
446         Enumeration<Permission> en = perm.elements();
447         while(en.hasMoreElements()){
448             this.permissions.add(en.nextElement());
449         }
450     }

452     /**
453      * Convert an array of URL objects into a corresponding string
454      * containing a space-separated list of URLs.
455      *
456      * Note that if the array has zero elements, the return value is
457      * null, not the empty string.
458      */
459     static String urlsToPath(URL[] urls) {
460 	if (urls.length == 0) {
461 	    return null;
462 	} else if (urls.length == 1) {
463 	    return urls[0].toExternalForm();
464 	} else {
465 	    StringBuilder path = new StringBuilder(urls[0].toExternalForm());
466 	    for (int i = 1; i < urls.length; i++) {
467 		path.append(' ');
468 		path.append(urls[i].toExternalForm());
469 	    }
470 	    return path.toString();
471 	}
472     }

474     /**
475      * Creates a new instance of <code>PreferredClassLoader</code>
476      * that loads classes and resources from the specified path of
477      * URLs and delegates to the specified parent class loader.
478      *
479      * <p>The <code>exportAnnotation</code> and
480      * <code>requireDlPerm</code> parameters have the same semantics
481      * as they do for the constructors.
482      *
483      * <p>The {@link #loadClass loadClass} method of the returned
484      * <code>PreferredClassLoader</code> will, if there is a security
485      * manager, invoke its {@link SecurityManager#checkPackageAccess
486      * checkPackageAccess} method with the package name of the class
487      * to load before attempting to load the class; this could result
488      * in a <code>SecurityException</code> being thrown from
489      * <code>loadClass</code>.
490      *
491      * @param urls the path of URLs to load classes and resources from
492      *
493      * @param parent the parent class loader for delegation
494      *
495      * @param exportAnnotation the export class annotation string to
496      * use for classes defined by this loader, or <code>null</code>
497      *
498      * @param requireDlPerm if <code>true</code>, the loader will only
499      * define classes with a {@link CodeSource} that is granted {@link
500      * DownloadPermission}
501      *
502      * @return the new <code>PreferredClassLoader</code> instance
503      *
504      * @throws SecurityException if the current security context does
505      * not have the permissions necessary to connect to all of the
506      * URLs in <code>urls</code>
507      **/
508     public static PreferredClassLoader
509 	newInstance(final URL[] urls,
510 		    final ClassLoader parent,
511 		    final String exportAnnotation,
512 		    final boolean requireDlPerm)
513     {
514 	/* ensure caller has permission to access all urls */
515 	PermissionCollection perms = new Permissions();
516 	addPermissionsForURLs(urls, perms, false);
517 	checkPermissions(perms);
518 
519 	AccessControlContext acc = getLoaderAccessControlContext(urls);
520 	/* Use privileged status to return a new class loader instance */
521 	return
522 	    AccessController.doPrivileged(new PrivilegedAction<PreferredClassLoader>() {
523 		@Override
524 		public PreferredClassLoader run() {
525 		    return new PreferredFactoryClassLoader(urls, parent,
526 			exportAnnotation, requireDlPerm);
527 		}
528 	    }, acc);
529     }

533     /**
534      * Returns an InputStream from which the preferred list relative
535      * to the specified URL can be read, or null if the there is
536      * definitely no preferred list relative to the URL.  If the URL's
537      * path ends with "/", then the preferred list is sought at the
538      * location "META-INF/PREFERRED.LIST" relative to the URL;
539      * otherwise, the URL is assumed to refer to a JAR file, and the
540      * preferred list is sought within that JAR file, as the entry
541      * named "META-INF/PREFERRED.LIST".
542      *
543      * Throws IOException if an I/O exception occurs from which the
544      * existence of a preferred list relative to the specified URL
545      * cannot be definitely determined.
546      **/
547     private static InputStream getPreferredInputStream(URL firstURL, String resource, URLStreamHandler jarHandler)
548 	throws IOException
549     {
550 	URL prefListURL;
551 	try {
552 	    URL baseURL;	// base URL to load PREF_NAME relative to
553 	    if (firstURL.getFile().endsWith("/")) { // REMIND: track 4915051
554 		baseURL = firstURL;
555 	    } else {
556 		/*
557 		 * First try to get a definite answer about the existence of a
558 		 * PREFERRED.LIST that doesn't by-pass a JAR file cache, if
559 		 * any. If that fails we determine if the JAR file exists by
560 		 * attempting to access it directly, without using a "jar:" URL,
561 		 * because the "jar:" URL handler can mask the distinction
562 		 * between definite lack of existence and less definitive
563 		 * errors. Unfortunately, this direct access circumvents the JAR
564 		 * file caching done by the "jar:" handler, so it ends up
565 		 * causing duplicate requests of the JAR file on first use when
566 		 * our first attempt fails. (For HTTP-protocol URLs, the initial
567 		 * request will use HEAD instead of GET.)
568 		 *
569 		 * After determining that the JAR file exists, attempt to
570 		 * retrieve the preferred list using a "jar:" URL, like
571 		 * URLClassLoader uses to load resources from a JAR file.
572 		 */
573 		if (jarExists(firstURL, jarHandler)) {
574 		    baseURL = getBaseJarURL(firstURL, jarHandler);
575 		} else {
576 		    return null;
577 		}
578 	    }
579 	    prefListURL = new URL(baseURL, resource);
580 	    URLConnection preferredConnection =
581 		getPreferredConnection(prefListURL, false);
582 	    if (preferredConnection != null) {
583 		return preferredConnection.getInputStream();
584 	    } else {
585 		return null;
586 	    }
587 	} catch (IOException e) {
588 	    /*
589 	     * Assume that any IOException thrown while attempting to
590 	     * access a "file:" URL and any FileNotFoundException
591 	     * implies that there is definitely no preferred list
592 	     * relative to the specified URL.
593 	     */
594 	    if (firstURL.getProtocol().equals("file") ||
595 		e instanceof FileNotFoundException)
596 	    {
597 		return null;
598 	    } else {
599 		throw e;
600 	    }
601 	}
602     }

607     /*
608      * Determine if a jar file in a given URL location exists.  If the
609      * jar exists record the jar file's URL in a cache of URL strings.
610      *
611      * Recording the existence of the jar prevents the need to
612      * re-determine the jar's existence on subsequent downloads of the
613      * jar in potentially different preferred class loaders.
614      */
615     private static boolean jarExists(URL firstURL, URLStreamHandler jarHandler) throws IOException {
616 	boolean exists;
617         
618 	synchronized (existSet) {
619             // The comment says in a cache of URL strings, URL in Set, bad.
620 	    exists = existSet.contains(firstURL.toString());
621 	}
622 
623 	if (!exists) {
624 	    /*
625 	     * first try to get a definite answer of the existence of a JAR
626 	     * file, if no IOException is thrown when obtaining it through the
627 	     * "jar:" protocol we can safely assume the JAR file is locally
628 	     * available upon the attempt (elsewhere) to obtain the preferred
629 	     * list
630 	     */
631             URL baseURL = null;
632 	    try {
633                 baseURL = getBaseJarURL(firstURL, jarHandler);
634 		((JarURLConnection) baseURL.openConnection()).getManifest();
635 		exists = true;
636 	    } catch (IOException e) {
637 		// we still have no definite answer on whether the JAR file
638 		// and therefore the PREFERRED.LIST exists
639 	    } catch (NullPointerException e){
640                 // Sun Bug ID: 6536522
641                 // NullPointerException is thrown instead of MalformedURLException
642                 // Case is the same as above, we have no definite answer on
643                 // whether the JAR file and therefore the PREFERRED.LIST exists.
644                 System.err.println("NPE thrown while trying to open connection :" +
645                         baseURL);
646                 e.printStackTrace(System.err);
647             }
648 
649 	    if (!exists) {
650 		exists = (getPreferredConnection(firstURL, true) != null);
651 	    }
652 	    if (exists) {
653 		synchronized (existSet) {
654 		    existSet.add(firstURL.toString());
655 		}
656 	    }
657 	}
658 	return exists;
659     }

686     /**
687      * Obtain a url connection from which an input stream that
688      * contains a preferred list can be obtained.
689      *
690      * For http urls, attempts to use http response codes to
691      * determine if a preferred list exists or is definitely not
692      * found.  Simply attempts to open a connection to other kinds
693      * of non-file urls.  If the attempt fails, an IOException is
694      * thrown to user code.
695      *
696      * Returns null if the preferred list definitely does not
697      * exist.  Rethrows all indefinite IOExceptions generated
698      * while trying to open a connection to the preferred list.
699      *
700      * The caller has the option to close the connection after the
701      * resource has been detected (as will happen when probing for a
702      * PREFERRED.LIST).
703      */
704     private static URLConnection getPreferredConnection(URL url, boolean closeAfter)
705 	throws IOException
706     {
707 	if (url.getProtocol().equals("file")) {
708 	    return url.openConnection();
709 	}
710 	URLConnection closeConn = null;
711 	URLConnection conn = null;
712 	try {
713 	    closeConn = url.openConnection();
714 	    conn = closeConn;
715 						 
716 	    /* check status of http urls  */
717 	    if (conn instanceof HttpURLConnection) {
718 		HttpURLConnection hconn = (HttpURLConnection) conn;
719 		if (closeAfter) {
720 		    hconn.setRequestMethod("HEAD");
721 		}
722 		int responseCode = hconn.getResponseCode();//NPE
723 		
724 		switch (responseCode) {
725 		case HttpURLConnection.HTTP_OK:
726 		case HttpURLConnection.HTTP_NOT_AUTHORITATIVE:
727 		    /* the preferred list exists */
728 		    break;
729 
730 		    /* 404, not found appears to be handled by
731 		     * HttpURLConnection (FileNotFoundException is
732 		     * thrown), but to be safe do the right thing here as
733 		     * well.
734 		     */
735 		case HttpURLConnection.HTTP_NOT_FOUND:
736 		case HttpURLConnection.HTTP_FORBIDDEN:
737 		case HttpURLConnection.HTTP_GONE:
738 		    /* list definitely does not exist */
739 		    conn = null;
740 		    break;
741 		default:
742 		    /* indefinite response code */
743 		    throw new IOException("Indefinite http response for " +
744 			"preferred list request:" +
745 			hconn.getResponseMessage());
746 		}
747 	    }
748         } catch (RuntimeException e){
749             if ( e instanceof NullPointerException || e.getCause() instanceof NullPointerException) {
750                 // Sun Bug ID: 6536522
751                 throw new IOException(url.toString(), e);
752             } else {
753                 throw e;
754             }
755 	} finally {
756 	    if (closeAfter && (closeConn != null)) {
757 		/* clean up after... */
758 		try {
759 		    closeConn.getInputStream().close();//RTE NPE
760 		} catch (IOException e) {
761 		} catch (RuntimeException e){
762                     if ( e instanceof NullPointerException || e.getCause() 
763 			    instanceof NullPointerException) {
764                         // Sun Bug ID: 6536522
765                         // swallow
766                         e.printStackTrace(System.err);
767                     } else {
768                         throw e;
769                     }
770                 }
771 	    }
772 	}
773 
774 	return conn;
775     }

836     /*
837      * Perform the work to determine if a resource name is preferred.
838      * 
839      * Synchronized removed to avoid possible ClassLoader deadlock.
840      */
841     private boolean isPreferredResource0(String name,
842 						      boolean isClass)
843 	throws IOException
844     {
845         if (exceptionWhileLoadingPreferred != null) throw exceptionWhileLoadingPreferred;
846 	if (preferredResources == null) {
847 	    return false;	// no preferred list: nothing is preferred
848 	}
849 
850 	String resourceName = name;
851 	if (isClass) {
852 	    /* class name -> resource name */
853 	    resourceName = name.replace('.', '/') + ".class";
854 	}
855 
856 	/*
857 	 * Determine if the class name is preferred. Making this
858 	 * distinction is somewhat tricky because we need to cache the
859 	 * preferred state (i.e. if the name is preferred and its
860 	 * resource exists) in a way that avoids duplication of
861 	 * preferred information - state information is stored back
862 	 * into the preferred resources object for this class loader
863 	 * and not held in a separate preferred settings cache.
864 	 */
865 	boolean resourcePreferred = false;
866 
867 	int state = preferredResources.getNameState(resourceName, isClass);
868 	switch (state) {
869 	case PreferredResources.NAME_NOT_PREFERRED:
870 	    resourcePreferred = false;
871 	    break;
872 
873 	case PreferredResources.NAME_PREFERRED_RESOURCE_EXISTS:
874 	    resourcePreferred = true;
875 	    break;
876 
877 	case PreferredResources.NAME_NO_PREFERENCE:
878 	    Boolean wildcardPref =
879 		preferredResources.getWildcardPreference(resourceName);
880 	    if (wildcardPref == null) {
881 		/* preferredDefault counts as a wild card */
882 		wildcardPref = preferredResources.getDefaultPreference();
883 	    }
884 	    if (wildcardPref.booleanValue()) {
885 		resourcePreferred =
886 		    findResourceUpdateState(name, resourceName);
887 	    }
888 	    break;
889 
890 	case PreferredResources.NAME_PREFERRED:
891 	    resourcePreferred =
892 		findResourceUpdateState(name, resourceName);
893 	    if (!resourcePreferred) {
894 		throw new IOException("no resource found for " +
895 				      "complete preferred name");
896 	    }
897 	    break;
898 
899 	default:
900 	    throw new Error("unknown preference state");
901 	}
902 	return resourcePreferred;
903     }

925      * Loads a class with the specified name.
926      *
927      * <p><code>PreferredClassLoader</code> implements this method as
928      * follows:
929      *
930      * <p>This method first invokes {@link #findLoadedClass
931      * findLoadedClass} with <code>name</code>; if
932      * <code>findLoadedClass</code> returns a non-<code>null</code>
933      * <code>Class</code>, then this method returns that
934      * <code>Class</code>.

936      * <p>Otherwise, this method invokes {@link #isPreferredResource

939      *

942      * <li>If <code>isPreferredResource</code> throws an
943      * <code>IOException</code>, then this method throws a
944      * <code>ClassNotFoundException</code> containing the

948      * <code>true</code>, then this method invokes {@link #findClass
949      * findClass} with <code>name</code>.  If <code>findClass</code>
950      * throws an exception, then this method throws that exception.
951      * Otherwise, this method returns the <code>Class</code> returned
952      * by <code>findClass</code>, and if <code>resolve</code> is
953      * <code>true</code>, {@link #resolveClass resolveClass} is
954      * invoked with the <code>Class</code> before returning.
955      *
956      * <li>If <code>isPreferredResource</code> returns
957      * <code>false</code>, then this method invokes the superclass
958      * implementation of {@link ClassLoader#loadClass(String,boolean)
959      * loadClass} with <code>name</code> and <code>resolve</code> and
960      * returns the result.  If the superclass's <code>loadClass</code>
961      * throws an exception, then this method throws that exception.
962      *
963      * </ul>
964      *
965      * @param name the binary name of the class to load
966      *
967      * @param resolve if <code>true</code>, then {@link #resolveClass
968      * resolveClass} will be invoked with the loaded class before
969      * returning
970      *
971      * @return the loaded class
972      *
973      * @throws ClassNotFoundException if the class could not be found
974      **/
975     @Override
976     protected Class loadClass(String name, boolean resolve)
977 	throws ClassNotFoundException
978     {
979 	// First, check if the class has already been loaded
980 	Class c = findLoadedClass(name);
981 	if (c == null) {
982 	    boolean preferred;
983 	    try {
984 		preferred = isPreferredResource(name, true);
985 	    } catch (IOException e) {
986 		throw new ClassNotFoundException(name +
987 		    " (could not determine preferred setting; " +
988 		    (firstURL != null ?
989 		     "first URL: \"" + firstURL + "\"" : "no URLs") +
990 		    ")", e);
991 	    }
992 	    if (preferred) {
993                 synchronized (this){
994                     // Double check again in case the class has been loaded.
995                     c = findLoadedClass(name);
996                     if (c == null){
997                         c = findClass(name);
998                         if (resolve) resolveClass(c);
999                     }
1000                 }
1001 	    } else {
1002 		return super.loadClass(name, resolve);
1003 	    }
1004 	}
1005 	return c;
1006     }

1221     /**
1222      * Return the access control context that a loader for the given
1223      * codebase URL path should execute with.
1224      */
1225     static AccessControlContext getLoaderAccessControlContext(URL[] urls)
1226     {
1227 	/*
1228 	 * The approach used here is taken from the similar method
1229 	 * getAccessControlContext() in the sun.applet.AppletPanel class.
1230 	 */
1231 	// We don't need to consult the policy here as the ProtectionDomain
1232 	// does so during the permission check, see comment further below.
1233 	PermissionCollection perms = new Permissions();

1235 	perms.add(new RuntimePermission("createClassLoader"));
1236 
1237 	// add permissions to read any "java.*" property
1238 	perms.add(new java.util.PropertyPermission("java.*","read"));
1239 
1240 	// add permissions required to load from codebase URL path
1241 	addPermissionsForURLs(urls, perms, true);

1245 	 * protection domain with only the permissions calculated above.
1246          * Comment added 7th May 2010 by Peter Firmstone:

1248          * ProtectionDomain to not consult the Policy, this
1249          * had the effect of not allowing Dynamic Permission changes to be
1250          * effected by the Policy.  It doesn't affect the existing
1251          * DynamicPolicy implementation as it returns the Permissions
1252          * allowing the ProtectionDomain domain combiner to combine
1253          * cached permissions with those from the Policy.
1254          * ProtectionDomain(CodeSource, PermissionCollection)
1255          * By utilising this earlier constructor it also prevents
1256          * RevokeableDynamicPolicy, hence the constructor change.

1258 	ProtectionDomain pd = new ProtectionDomain(
1259 	    new CodeSource((urls.length > 0 ? urls[0] : null),
1260 			   (Certificate[]) null), perms, null, null);

1266      * necessary to load classes from a loader with the specified URL
1267      * path; if "forLoader" is true, also adds URL-specific
1268      * permissions necessary for the security context that such a
1269      * loader operates within, such as permissions necessary for
1270      * granting automatic permissions to classes defined by the
1271      * loader.  A given permission is only added to the collection if
1272      * it is not already implied by the collection.
1273      * @param urls
1274      * @param perms
1275      * @param forLoader
1276      **/
1277     static void addPermissionsForURLs(URL[] urls,
1278 				      PermissionCollection perms,
1279 				      boolean forLoader)
1280     {
1281 	for (int i = 0, l = urls.length; i < l; i++) {
1282 	    URL url = urls[i];
1283 	    try {
1284 		URLConnection urlConnection = url.openConnection();
1285 		Permission p = urlConnection.getPermission();
1286 		if (p != null) {
1287 		    if (p instanceof FilePermission) {
1288 			/*
1289 			 * If the codebase is a file, the permission required
1290 			 * to actually read classes from the codebase URL is
1291 			 * the permission to read all files beneath the last
1292 			 * directory in the file path, either because JAR
1293 			 * files can refer to other JAR files in the same
1294 			 * directory, or because permission to read a
1295 			 * directory is not implied by permission to read the
1296 			 * contents of a directory, which all that might be
1297 			 * granted.
1298 			 */
1299 			String path = p.getName();
1300 			int endIndex = path.lastIndexOf(File.separatorChar);
1301 			if (endIndex != -1) {
1302 			    path = path.substring(0, endIndex+1);
1303 			    if (path.endsWith(File.separator)) {
1304 				path += "-";
1305 			    }
1306 			    Permission p2 = new FilePermission(path, "read");
1307 			    if (!perms.implies(p2)) {
1308 				perms.add(p2);
1309 			    }
1310 			} else {
1311 			    /*
1312 			     * No directory separator: use permission to
1313 			     * read the file.
1314 			     */
1315 			    if (!perms.implies(p)) {
1316 				perms.add(p);
1317 			    }
1318 			}
1319 		    } else {
1320 			if (!perms.implies(p)) {
1321 			    perms.add(p);
1322 			}
1323 
1324 	                /*
1325 			 * If the purpose of these permissions is to grant
1326 			 * them to an instance of a URLClassLoader subclass,
1327 			 * we must add permission to connect to and accept
1328 			 * from the host of non-"file:" URLs, otherwise the
1329 			 * getPermissions() method of URLClassLoader will
1330 			 * throw a security exception.
1331 			 */
1332 			if (forLoader) {
1333                             // URLPermission is required for JDK1.8 and JDK1.9
1334                             try {
1335                                 perms.add(
1336                                     new URLPermission(
1337                                             Uri.urlToUri(url).toString(),
1338                                             "GET:"
1339                                     )
1340                                 );
1341                             } catch (URISyntaxException ex) {
1342                                 Logger.getLogger(PreferredClassLoader.class.getName()).log(Level.FINE, "Unable to grant URLPermission", ex);
1343                             } catch (IllegalArgumentException ex){
1344                                 Logger.getLogger(PreferredClassLoader.class.getName()).log(Level.FINE, "Unable to grant URLPermission", ex);
1345                             }
1346 			    // get URL with meaningful host component
1347 			    URL hostURL = url;
1348 			    for (URLConnection conn = urlConnection;
1349 				 conn instanceof JarURLConnection;)
1350 			    {
1351 				hostURL =
1352 				    ((JarURLConnection) conn).getJarFileURL();
1353 				conn = hostURL.openConnection();
1354 			    }
1355 			    String host = hostURL.getHost();
1356 			    if (host != null &&
1357 				p.implies(new SocketPermission(host,
1358 							       "resolve")))
1359 			    {
1360 				Permission p2 =
1361 				    new SocketPermission(host,
1362 							 "connect,accept");
1363 				if (!perms.implies(p2)) {
1364 				    perms.add(p2);
1365 				}
1366 			    }
1367 			}
1368 		    }
1369 		}
1370 	    } catch (IOException e) {
1371 		/*
1372 		 * This shouldn't happen, although it is declared to be
1373 		 * thrown by openConnection() and getPermission().  If it
1374 		 * does, don't bother granting or requiring any permissions
1375 		 * for this URL.
1376 		 */
1377 	    } catch (NullPointerException e){
1378                 // Sun Bug ID: 6536522
1379             }
1380 	}
1381     }

1383     @Override
1384     public Permission[] getPermissions() {
1385 	if (advisoryPermissions == null) {
1386 	    return AdvisoryDynamicPermissions.DEFAULT_PERMISSIONS;
1387 	}
1388 	return advisoryPermissions.clone();
1389     }

1391     private static class PreferredPermissionsPrivilegedAction
1392 	    implements PrivilegedAction<Permission[]> {
1393 
1394 	private static final String resource = "META-INF/PERMISSIONS.LIST";
1395 	private final URL[] urls;
1396 	private final URLStreamHandler jarHandler;
1397 	private final ClassLoader loader;
1398 
1399 	PreferredPermissionsPrivilegedAction(
1400 		URL[] urls,
1401 		URLStreamHandler jarHandler,
1402 		ClassLoader loader) 
1403 	{
1404 	    this.urls = urls;
1405 	    this.jarHandler = jarHandler;
1406 	    this.loader = loader;
1407 	}
1408 
1409 	@Override
1410 	public Permission[] run() {
1411 	    Permission[] permissions = null;
1412 	    List<Permission> perms = new LinkedList<Permission>();
1413 	    InputStream in = null;
1414 	    for (int i = 0, l = urls.length; i < l; i++) {
1415 		try {
1416 		    in = getPreferredInputStream(urls[i], resource, jarHandler);
1417 		    if (in != null) {
1418 			BufferedReader reader = new BufferedReader(
1419 				    new InputStreamReader(in, "UTF-8"));
1420 			try { 
1421 			    for (String line = reader.readLine(); line != null; line = reader.readLine()) {
1422 				String trim = line.trim();
1423 				if (trim.startsWith("#") || trim.startsWith("//")
1424 					|| (trim.length() == 0)) {
1425 				    continue;
1426 				}
1427 				perms.add(AdvisoryPermissionParser.parse(line, loader));
1428 			    }
1429 			    permissions = perms.toArray(new Permission[perms.size()]);
1430 			} finally {
1431 			    reader.close();
1432 			}
1433 		    }
1434 		} catch (Exception ex) {
1435 		    Logger.getLogger(PreferredClassLoader.class.getName()).log(Level.CONFIG, "No advisory permissions: " + urls[i].toString(), ex);
1436 		} finally {
1437 		    if (in != null) {
1438 			try {
1439 			    in.close();
1440 			} catch (IOException ex) {
1441 			} // Ignore.
1442 		    }
1443 		}
1444 	    }
1445 	    return permissions;
1446 	}
1447 
1448     }
