36  /**
37   * Provides static methods for loading classes using {@link
38   * RMIClassLoaderSpi} with optional verification that the codebase URIs
39   * used to load classes provide content integrity (see {@link
40   * Security#verifyCodebaseIntegrity
41   * Security.verifyCodebaseIntegrity}).
42   * <p>
43   * Traditionally a class extending {@link RMIClassLoaderSpi} is determined by setting
44   * the system property "java.rmi.server.RMIClassLoaderSpi", or alternatively,
45   * {@link RMIClassLoaderSpi} may also be defined by {@link RMIClassLoader}
46   * using a provider visible to the {@link ClassLoader} returned by 
47   * {@link ClassLoader#getSystemClassLoader} with {@link Service}.
48   * </p><p>
49   * As explained in River-336 this isn't always practical for IDE's or other 
50   * frameworks.  To solve River-336, ClassLoading now uses {@link Service}
51   * to determine a {@link RMIClassLoaderSpi} provider, however unlike 
52   * {@link RMIClassLoader}, by default it uses ClassLoading's {@link ClassLoader#getResources} 
53   * instance to find providers.
54   * </p><p>
55   * To define a new RMIClassLoaderSpi for River to utilize, create a file in
56   * your providers jar file called:
57   * </p><p>
58   * META-INF/services/java.rmi.server.RMIClassLoaderSpi
59   * </p><p>
60   * This file should contain a single line with the fully qualified name of
61   * your RMIClassLoaderSpi implementation.
62   * </p><p>
63   * ClassLoading will iterate through all RMIClassLoaderSpi implementations found
64   * until it finds one defined by the system property:
65   * </p><p>
66   * System.getProperty("net.jini.loader.ClassLoading.provider");
67   * </p><p>
68   * If this System property is not defined, ClassLoading will load 
69   * <code>net.jini.loader.pref.PreferredClassProvider</code>, alternatively
70   * <code>java.rmi.server.RMIClassLoader</code> delegates all calls to {@link
71   * RMIClassLoader}.
72   * </p><p>
73   * If a provider is not found, it will not be updated.
74   * </p>
75   * <h1>History</h1>
76   * <p>Gregg Wonderly originally reported River-336 and provided a patch
77   * containing a new CodebaseAccessClassLoader to replace {@link RMIClassLoader}, 
78   * later Sim Isjkes created RiverClassLoader that utilized ServiceLoader.
79   * Both implementations contained methods identical to {@link RMIClassLoaderSpi},
80   * however new implementations were required to extend new provider 
81   * implementations, creating a compatibility issue with existing implementations
82   * extending {@link RMIClassLoaderSpi}.  For backward compatibility with existing
83   * implementations, {@link RMIClassLoaderSpi} has been retained as the provider,
84   * avoiding the need to recompile client code.  The abilities of both
85   * implementations, to use ServiceLoader, or to define a provider using a method
86   * call have been retained, with the restriction that implementations are to be
87   * obtained via ServiceLoader.
88   * </p><p>
89   * Instead, all that is required for utilization of existing service provider
90   * {@link RMIClassLoaderSpi} implementations is to set the system property
91   * "net.jini.loader.ClassLoading.provider".
92   * </p>
93   * @author Sun Microsystems, Inc.
94   * @since 2.0
95   **/
96  public final class ClassLoading {
97      private final static Logger logger = Logger.getLogger(ClassLoading.class.getName());
98      private static volatile RMIClassLoaderSpi provider;
99      private static final Object lock = new Object();
100     private static final Guard permission = new RuntimePermission("setFactory");

102     static {
103         provider(null, ClassLoading.class.getClassLoader());
104     }

106     /**
107      * The current RMIClassLoaderSpi provider in use by ClassLoading.
108      * 
109      * @return currently installed Provider, may be null.
110      * @throws SecurityException if caller doesn't have RuntimePermission "getFactory"
111      */
112     public static RMIClassLoaderSpi getProvider(){
113         permission.checkGuard(null);
114         return provider;
115     }
116     
117     private static boolean provider(
118             final String providerName, 
119             final ClassLoader providerLoader)
120     {
121         RMIClassLoaderSpi newProvider = AccessController.doPrivileged(
122         new PrivilegedAction<RMIClassLoaderSpi>(){
123             @Override
124             public RMIClassLoaderSpi run() {
125                 String name = providerName;
126                 if (name == null){
127                     name = System.getProperty(
128                             "net.jini.loader.ClassLoading.provider");
129                     if (name == null) {
130                         name = "net.jini.loader.pref.PreferredClassProvider";
131                     } else if ("java.rmi.server.RMIClassLoader".equals(name)){
132                         return null;
133                     }
134                 }
135                 Iterator<RMIClassLoaderSpi> iter = 
136 			Service.providers(RMIClassLoaderSpi.class, providerLoader);
137                 RMIClassLoaderSpi spi;
138                 try {
139                     while ( iter.hasNext() ) {
140                         try {
141                             spi = iter.next();
142                             if (spi != null) {
143                                 if (!name.equals(spi.getClass().getName()))
144                                     continue;
145                                 logger.log(Level.CONFIG, "loaded: {0}", name);
146                                 return spi;
147                             }
148                         } catch (Throwable e) {
149                             logger.log( 
150                                     Level.CONFIG, 
151                                     "error loading RMIClassLoaderSpi: {0}",
152                                     new Object[]{e}
153                             );
154                         }
155                     }
156                 } catch (Throwable t) { }
157                 logger.log(Level.CONFIG, "uable to find {0}" , name);
158                 return null;
159             }
160         });
161         if (newProvider != null) {
162             provider = newProvider;
163             return true;
164         } else if (providerName == null) {
165             provider = null;
166             logger.log(Level.CONFIG, "loaded: java.rmi.server.RMIClassLoader");
167         }
168         return false;
169     }

171     /**
172      * Installs a new RMIClassLoaderSpi provider with the ClassLoader 
173      * provided.
174      * 
175      * @param providerName fully defined class name of the provider, if null,
176      * a new provider instance will be determined by system properties.
177      * @param providerLoader The class loader to be used to load 
178      * provider-configuration files and provider classes, or null if the 
179      * system class loader (or, failing that, the bootstrap class loader) 
180      * is to be used.
181      * @return true if successful.
182      * @throws SecurityException if caller doesn't have RuntimePermission "getFactory"
183      */
184     public static boolean installNewProvider(
185             String providerName, 
186             ClassLoader providerLoader)
187     {
188         permission.checkGuard(null);
189         synchronized (lock){
190             return provider(providerName, providerLoader);
191         }
192     }

196      * (soft) sets of codebase values that have been verified (to

199     private static final ThreadLocal perThreadCache = new ThreadLocal() {

203     /**
204      * Returns a class loader that loads classes from the given codebase
205      * RFC3986 compliant URI path.
206      *
207      * <p>This method delegates to the
208      * {@link RMIClassLoaderSpi#getClassLoader(String)} method
209      * of the provider instance, passing <code>codebase</code> as the argument.
210      *
211      * <p>If there is a security manger, its <code>checkPermission</code>
212      * method will be invoked with a
213      * <code>RuntimePermission("getClassLoader")</code> permission;
214      * this could result in a <code>SecurityException</code>.
215      * The provider implementation of this method may also perform further
216      * security checks to verify that the calling context has permission to
217      * connect to all of the URIs in the codebase URI path.
218      *
219      * @param   codebase the list of URIs (space-separated) from which
220      * the returned class loader will load classes from, or <code>null</code>
221      *
222      * @return a class loader that loads classes from the given codebase URI
223      * path
224      *
225      * @throws  MalformedURLException if <code>codebase</code> is
226      * non-<code>null</code> and contains an non RFC3986 compliant URI, or
227      * if <code>codebase</code> is <code>null</code> and a provider-specific
228      * URL used to identify the class loader is invalid
229      *
230      * @throws  SecurityException if there is a security manager and the
231      * invocation of its <code>checkPermission</code> method fails, or
232      * if the caller does not have permission to connect to all of the
233      * URIs in the codebase URI path
234      * @since 3.0
235      */
236     public static ClassLoader getClassLoader(String codebase)
237         throws MalformedURLException, SecurityException
238     {
239         if (provider != null) return provider.getClassLoader(codebase);
240         return RMIClassLoader.getClassLoader(codebase);
241     }

266     /**
267      * Loads a class using {@link
268      * RMIClassLoaderSpi#loadClass(String,String,ClassLoader)}, 
269      * optionally verifying that the RFC3986 compliant
270      * codebase URIs provide content integrity.
271      *
272      * <p>If <code>verifyCodebaseIntegrity</code> is <code>true</code>
273      * and <code>codebase</code> is not <code>null</code>, then this
274      * method invokes {@link Security#verifyCodebaseIntegrity
275      * Security.verifyCodebaseIntegrity} with <code>codebase</code> as
276      * the first argument and <code>verifierLoader</code> as the
277      * second argument (this invocation may be skipped if a previous
278      * invocation of this method or {@link #loadProxyClass
279      * loadProxyClass} has already invoked
280      * <code>Security.verifyCodebaseIntegrity</code> with the same
281      * value of <code>codebase</code> and the same effective value of
282      * <code>verifierLoader</code> as arguments without it throwing an
283      * exception).  If <code>Security.verifyCodebaseIntegrity</code>
284      * throws a <code>SecurityException</code>, then this method
285      * proceeds as if <code>codebase</code> were <code>null</code>.
286      * If <code>Security.verifyCodebaseIntegrity</code> throws any
287      * other exception, then this method throws that exception.
288      *
289      * <p>This method then invokes {@link
290      * RMIClassLoaderSpi#loadClass(String,String,ClassLoader)
291      * RMIClassLoaderSpi.loadClass} with <code>codebase</code> as the
292      * first argument (or <code>null</code> if in the previous step
293      * <code>Security.verifyCodebaseIntegrity</code> was invoked and
294      * it threw a <code>SecurityException</code>), <code>name</code>
295      * as the second argument, and <code>defaultLoader</code> as the
296      * third argument.  If <code>RMIClassLoaderSpi.loadClass</code>
297      * throws a <code>ClassNotFoundException</code>, then this method
298      * throws a <code>ClassNotFoundException</code>; if
299      * <code>RMIClassLoaderSpi.loadClass</code> throws any other
300      * exception, then this method throws that exception; otherwise,
301      * this method returns the <code>Class</code> returned by
302      * <code>RMIClassLoaderSpi.loadClass</code>.
303      *
304      * @param codebase the list of URLs (separated by spaces) to load
305      * the class from, or <code>null</code>
306      *
307      * @param name the name of the class to load
308      *
309      * @param defaultLoader the class loader value (possibly
310      * <code>null</code>) to pass as the <code>defaultLoader</code>
311      * argument to <code>RMIClassLoaderSpi.loadClass</code>
312      *
313      * @param verifyCodebaseIntegrity if <code>true</code>, verify
314      * that the RFC3986 compliant codebase URIs provide content integrity
315      *
316      * @param verifierLoader the class loader value (possibly
317      * <code>null</code>) to pass to
318      * <code>Security.verifyCodebaseIntegrity</code>, if
319      * <code>verifyCodebaseIntegrity</code> is <code>true</code>
320      *
321      * @return the <code>Class</code> object representing the loaded
322      * class
323      *
324      * @throws MalformedURLException if
325      * <code>Security.verifyCodebaseIntegrity</code> or
326      * <code>RMIClassLoaderSpi.loadClass</code> throws a
327      * <code>MalformedURLException</code>
328      *
329      * @throws ClassNotFoundException if
330      * <code>RMIClassLoaderSpi.loadClass</code> throws a
331      * <code>ClassNotFoundException</code>
332      *
333      * @throws NullPointerException if <code>name</code> is
334      * <code>null</code>
335      **/
336     public static Class<?> loadClass(String codebase,
337 				  String name,
338 				  ClassLoader defaultLoader,
339 				  boolean verifyCodebaseIntegrity,
340 				  ClassLoader verifierLoader)
341 	throws MalformedURLException, ClassNotFoundException
342     {
343 	SecurityException verifyException = null;
344 	if (verifyCodebaseIntegrity && codebase != null) {
345 	    try {
346 		verifyIntegrity(codebase, verifierLoader);
347 	    } catch (SecurityException e) {
348 		verifyException = e;
349 		codebase = null;
350 	    }
351 	}
352 	try {
353             if (provider != null) 
354                 return provider.loadClass(codebase, name, defaultLoader);
355 	    return RMIClassLoader.loadClass(codebase, name, defaultLoader);
356 	} catch (ClassNotFoundException e) {
357 	    if (verifyException != null) {
358 		// assume that the verify exception is more important
359 		throw new ClassNotFoundException(e.getMessage(),
360 						 verifyException);
361 	    } else {
362                 e.fillInStackTrace();
363 		throw e;
364 	    }
365 	}
366     }

368     /**
369      * Loads a dynamic proxy class using {@link
370      * RMIClassLoaderSpi#loadProxyClass(String,String[],ClassLoader)},
371      * optionally verifying that the RFC3986 compliant
372      * codebase URIs provide content integrity.
373      *
374      * <p>If <code>verifyCodebaseIntegrity</code> is <code>true</code>
375      * and <code>codebase</code> is not <code>null</code>, then this
376      * method invokes {@link Security#verifyCodebaseIntegrity
377      * Security.verifyCodebaseIntegrity} with <code>codebase</code> as
378      * the first argument and <code>verifierLoader</code> as the
379      * second argument (this invocation may be skipped if a previous
380      * invocation of this method or {@link #loadClass loadClass} has
381      * already invoked <code>Security.verifyCodebaseIntegrity</code>
382      * with the same value of <code>codebase</code> and the same
383      * effective value of <code>verifierLoader</code> as arguments
384      * without it throwing an exception).  If
385      * <code>Security.verifyCodebaseIntegrity</code> throws a
386      * <code>SecurityException</code>, then this method proceeds as if
387      * <code>codebase</code> were <code>null</code>.  If
388      * <code>Security.verifyCodebaseIntegrity</code> throws any other
389      * exception, then this method throws that exception.
390      *
391      * <p>This method invokes {@link
392      * RMIClassLoaderSpi#loadProxyClass(String,String[],ClassLoader)} 
393      * with <code>codebase</code> as
394      * the first argument (or <code>null</code> if in the previous
395      * step <code>Security.verifyCodebaseIntegrity</code> was invoked
396      * and it threw a <code>SecurityException</code>),
397      * <code>interfaceNames</code> as the second argument, and
398      * <code>defaultLoader</code> as the third argument.  If
399      * <code>RMIClassLoaderSpi.loadProxyClass</code> throws a
400      * <code>ClassNotFoundException</code>, then this method throws a
401      * <code>ClassNotFoundException</code>; if
402      * <code>RMIClassLoaderSpi.loadProxyClass</code> throws any other
403      * exception, then this method throws that exception; otherwise,
404      * this method returns the <code>Class</code> returned by
405      * <code>RMIClassLoaderSpi.loadProxyClass</code>.
406      *
407      * @param codebase the list of URLs (separated by spaces) to load
408      * classes from, or <code>null</code>
409      *
410      * @param interfaceNames the names of the interfaces for the proxy
411      * class to implement
412      *
413      * @param defaultLoader the class loader value (possibly
414      * <code>null</code>) to pass as the <code>defaultLoader</code>
415      * argument to <code>RMIClassLoader.loadProxyClass</code>
416      *
417      * @param verifyCodebaseIntegrity if <code>true</code>, verify
418      * that the codebase URLs provide content integrity
419      *
420      * @param verifierLoader the class loader value (possibly
421      * <code>null</code>) to pass to
422      * <code>Security.verifyCodebaseIntegrity</code>, if
423      * <code>verifyCodebaseIntegrity</code> is <code>true</code>
424      *
425      * @return the <code>Class</code> object representing the loaded
426      * dynamic proxy class
427      *
428      * @throws MalformedURLException if
429      * <code>Security.verifyCodebaseIntegrity</code> or
430      * <code>RMIClassLoaderSpi.loadProxyClass</code> throws a
431      * <code>MalformedURLException</code>
432      *
433      * @throws ClassNotFoundException if
434      * <code>RMIClassLoaderSpi.loadProxyClass</code> throws a
435      * <code>ClassNotFoundException</code>
436      *
437      * @throws NullPointerException if <code>interfaceNames</code> is
438      * <code>null</code> or if any element of
439      * <code>interfaceNames</code> is <code>null</code>
440      **/
441     public static Class<?> loadProxyClass(String codebase,
442 				       String[] interfaceNames,
443 				       ClassLoader defaultLoader,
444 				       boolean verifyCodebaseIntegrity,
445 				       ClassLoader verifierLoader)
446 	throws MalformedURLException, ClassNotFoundException
447     {
448 	SecurityException verifyException = null;
449 	if (verifyCodebaseIntegrity && codebase != null) {
450 	    try {
451 		verifyIntegrity(codebase, verifierLoader);
452 	    } catch (SecurityException e) {
453 		verifyException = e;
454 		codebase = null;
455 	    }
456 	}
457 	try {
458             if (provider != null) return 
459                 provider.loadProxyClass(codebase, interfaceNames, defaultLoader);
460 	    return RMIClassLoader.loadProxyClass(codebase, interfaceNames,
461 						 defaultLoader);
462 	} catch (ClassNotFoundException e) {
463 	    if (verifyException != null) {
464 		// assume that the verify exception is more important
465 		throw new ClassNotFoundException(e.getMessage(),
466 						 verifyException);
467 	    } else {
468                 e.fillInStackTrace();
469 		throw e;
470 	    }
471 	}
472     }

474     /**
475      * Wraps Security.verifyCodebaseIntegrity with caching for
476      * performance.  (Perhaps such caching should be done by
477      * Security.verifyCodebaseIntegrity instead.)
478      **/
479     private static void verifyIntegrity(String codebase,
480 					ClassLoader verifierLoader)
481 	throws MalformedURLException
482     {
483 	/*
484 	 * Check if we've already verified the same codebase in this
485 	 * thread using the same verifierLoader value.
486 	 */
487 	Map verifierLoaderCache = (Map) perThreadCache.get();
488 	// defend against varying context class loader value of thread
489 	ClassLoader verifierLoaderKey =
490 	    (verifierLoader != null ? verifierLoader :
491 	     (ClassLoader) AccessController.doPrivileged(
492 		new PrivilegedAction() {
493 		    public Object run() {
494 			return Thread.currentThread().getContextClassLoader();
495 		    }
496 		}));
497 	Map verifiedCodebases =
498 	    (Map) verifierLoaderCache.get(verifierLoaderKey);
499 	if (verifiedCodebases != null &&
500 	    verifiedCodebases.containsKey(codebase))
501 	{
502 	    return;
503 	}
504 
505 	Security.verifyCodebaseIntegrity(codebase, verifierLoader);
506 
507 	/*
508 	 * Remember that we've verified this codebase in this thread
509 	 * with the given verifierLoader value.
510 	 */
511 	if (verifiedCodebases == null) {
512 	    verifiedCodebases = new WeakHashMap();
513 	    verifierLoaderCache.put(verifierLoaderKey, verifiedCodebases);
514 	}
515 	verifiedCodebases.put(codebase, new SoftReference(codebase));
516 	return;
517     }
