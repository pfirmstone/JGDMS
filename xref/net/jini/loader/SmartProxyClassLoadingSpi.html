16  package net.jini.loader;
17  
18  import java.io.IOException;
19  import java.util.Collection;
20  import java.util.Iterator;
21  import net.jini.export.CodebaseAccessor;
22  import net.jini.io.MarshalledInstance;
23  import net.jini.io.context.IntegrityEnforcement;
24  
25  /**
26   * A provider used by @{link org.apache.river.api.io.AtomicMarshalInputStream}
27   * to allow third party smart proxy's found in a stream to have 
28   * {@link net.jini.core.constraint.InvocationConstraints} applied by the client
29   * and allow the client to provide the smart proxy with a ClassLoader, unique
30   * to it's {@link java.lang.reflect.InvocationHandler} instance and dynamically
31   * grant permissions.
32   * 
33   * This interface also exists to avoid codebase annotation loss when smart 
34   * proxy instances are re-serialized, by consulting the original
35   * export node, using {@link CodebaseAccessor}.
36   * 
37   * @see org.apache.river.api.io.AtomicMarshalOutputStream
38   * @see org.apache.river.api.io.AtomicMarshalInputStream
39   * @see org.apache.river.api.io.AtomicMarshalledInstance
40   * 
41   * @author peter
42   */
43  public interface SmartProxyClassLoadingSpi {
44  
45      /**
46       * 
47       * @param bootstrapProxy
48       * @param smartProxy
49       * @param context
50       * @return
51       * @throws IOException 
52       */
53      public Object resolve(
54  	    CodebaseAccessor bootstrapProxy, 
55  	    MarshalledInstance smartProxy, 
56  	    Collection context) throws IOException;
57      
58      
59      
60      public static class Util {
61  	
62  	public static boolean verifyCodebaseIntegrity(Collection context) {
63  	    if (context != null){
64  		Iterator it = context.iterator();
65  		while (it.hasNext()){
66  		    Object next = it.next();
67  		    if (next instanceof IntegrityEnforcement){
68  			return ((IntegrityEnforcement) next).integrityEnforced();
69  		    }
70  		}
71  	    }
72  	    return false;
73  	}
74      }
75      
76  }
