20  import java.lang.reflect.*;
21  
22  import java.util.ArrayList;
23  import java.util.WeakHashMap;
24  import net.jini.core.entry.CloneableEntry;
25  import net.jini.core.entry.Entry;

27  /**
28   * An abstract implementation of {@link Entry} that provides useful
29   * implementations of <code>equals</code>, <code>hashCode</code>, and
30   * <code>toString</code>. Implementations of the <code>Entry</code>
31   * interface may, but are not required to, extend this class. <p>
32   *
33   * The methods of this class consult the <em>entry fields</em> of the
34   * entries they process. The entry fields of an <code>Entry</code> are
35   * its public, non-primitive, non-static, non-transient, non-final
36   * fields.<p>
37   *
38   * @author Sun Microsystems, Inc.
39   * @see Entry 
40   */
41  public abstract class AbstractEntry implements CloneableEntry {
42      static final long serialVersionUID = 5071868345060424804L;
43  
44      /**
45       * Creates an instance of this class.
46       */
47      protected AbstractEntry() { }
48      
49      /**
50       * Clone has been implemented to allow utilities such as
51       * <code> net.jini.lookup.ServiceDiscoveryManager </code> to avoid sharing 
52       * internally stored instances with client code.
53       * 
54       * Entry's that have mutable fields, for example arrays or collections,
55       * should override this method, call super.clone(), then safely copy
56       * any mutable fields before returning.
57       * 
58       * @return a clone of the original Entry
59       * @since 3.0.0
60       */
61      public Entry clone() 
62      {
63          try {
64              Entry clone = (Entry) super.clone();
65              return clone;
66          } catch (CloneNotSupportedException ex) {
67              throw new AssertionError();
68          }
69      }
70      
71      /**
72       * Compares this <code>AbstractEntry</code> to the specified
73       * object. If <code>other</code> is <code>null</code> or not an
74       * instance of {@link Entry} returns <code>false</code>,
75       * otherwise returns the result of calling {@link #equals(Entry,Entry)
76       * AbstractEntry.equals(this, (Entry) other)}.
77       * @param other the object to compare this <code>AbstractEntry</code>
78       *              against
79       * @return <code>true</code> if this object is equivalent
80       *         to <code>other</code> and <code>false</code> otherwise
81       */
82      public boolean equals(Object other) {
83  	if (!(other instanceof Entry))
84  	    return false;
85  	return equals(this, (Entry) other);
86      }
87  
88      /**
89       * Returns <code>true</code> if the two arguments are of the same
90       * class and for each entry field <em>F</em>, the arguments'
91       * values for <em>F</em> are either both <code>null</code> or the
92       * invocation of <code>equals</code> on one argument's value for
93       * <em>F</em> with the other argument's value for <em>F</em> as
94       * its parameter returns <code>true</code>. Will also return
95       * <code>true</code> if both arguments are <code>null</code>. In
96       * all other cases an invocation of this method will return
97       * <code>false</code>.<p>
98       *
99       * @param e1  an entry object to compare to e2
100      * @param e2  an entry object to compare to e1
101      * @return <code>true</code> if the two arguments are equivalent 
102      */
103     public static boolean equals(Entry e1, Entry e2) {
104 	if (e1 == e2)
105 	    return true;
106 
107 	// Note, if both e1 and e2 are null the previous test would
108 	// have returned true.
109 	if (e1 == null || e2 == null)
110 	    return false;
111 
112 	if (e1.getClass() != e2.getClass())
113 	    return false;
114 
115 	Field[] fields = fieldInfo(e1);
116 	try {
117 	    // compare each field
118 	    for (int i = 0; i < fields.length; i++) {
119 
120 		// f works for other since other is the same type as this
121 		Field f = fields[i];
122 		Object ov = f.get(e1);
123 		Object tv = f.get(e2);
124 
125 		if (tv == ov)			// same obj or both null is OK
126 		    continue;
127 		if (tv == null || ov == null)	// if only one is null, not OK
128 		    return false;
129 		if (!tv.equals(ov))		// not equals is not OK
130 		    return false;
131 	    }
132 	    return true;
133 	} catch (IllegalAccessException e) {
134 	    // should never happen, all entry fields are public
135 	    throw new AssertionError(e);
136 	}
137     }
138 
139     /**
140      * Returns the result of calling {@link #hashCode(Entry)
141      * AbstractEntry.hashCode(this)}.
142      * @return the result of calling {@link #hashCode(Entry)
143      * AbstractEntry.hashCode(this)}
144      */
145     public int hashCode() {
146 	return hashCode(this);
147     }
148 
149     /**
150      * Returns zero XORed with the result of invoking
151      * <code>hashCode</code> on each of the argument's
152      * non-<code>null</code> entry fields. Returns <code>0</code> if
153      * the argument is <code>null</code>.
154      *
155      * @param entry the <code>Entry</code> for which to generate a
156      *              hash code
157      * @return a hash code formed by XORing the hash codes of
158      *         <code>entry</code>'s non-<code>null</code> entry field,
159      *         or <code>0</code> if <code>entry</code> is
160      *         <code>null</code> 
161      */
162     public static int hashCode(Entry entry) {
163 	if (entry == null)
164 	    return 0;
165 	
166 	int hash = 0;
167 	Field[] fields = fieldInfo(entry);
168 	try {
169 	    for (int i = 0; i < fields.length; i++) {
170 		Object tv = fields[i].get(entry);
171 		if (tv != null)
172 		    hash ^= tv.hashCode();
173 	    }
174 	    return hash;
175 	} catch (IllegalAccessException e) {
176 	    // should never happen, all entry fields are public
177 	    throw new AssertionError(e);
178 	}
179     }
180 
181     /**
182      * Returns the result of calling {@link #toString(Entry)
183      * AbstractEntry.toString(this)}.
184      * @return the result of calling {@link #toString(Entry)
185      * AbstractEntry.toString(this)}
186      */
187     public String toString() {
188 	return toString(this);
189     }
190 
191     /**
192      * Returns a <code>String</code> representation of its argument
193      * that will contain the name of the argument's class and a
194      * representation of each of the argument's entry fields. The
195      * representation of each entry field will include the field's
196      * name and a representation of its value. If passed
197      * <code>null</code> will return the string <code>"null"</code>. <p>
198      *
199      * @param entry  an entry to represent as a String
200      * @return a <code>String</code> representation of
201      *         <code>entry</code> that contains the name of the
202      *         <code>entry</code>'s class and a representation of each
203      *         of <code>entry</code>'s entry fields
204      */
205     public static String toString(Entry entry) {
206 	if (entry == null)
207 	    return "null";
208 
209 	Field[] fields = fieldInfo(entry);
210 	StringBuffer str = new StringBuffer(entry.getClass().getName());
211 	str.append('(');	// later matched with a ')'
212 	for (int i = 0; i < fields.length; i++) {
213 	    try {
214 		Field f = fields[i];
215 		if (i > 0)
216 		    str.append(',');
217 		str.append(f.getName());
218 		str.append('=');
219 		str.append(f.get(entry));
220 	    } catch (IllegalAccessException e) {
221 		// should never happen, all entry fields are public
222 		throw new AssertionError(e);
223 	    }
224 	}
225 	// now add the ending '('
226 	str.append(')');
227 	return str.toString();
228     }

230     private static WeakHashMap fieldArrays;
231 
232     /**
233      * Calculate the list of usable fields for this type
234      */
235     private static Field[] fieldInfo(Entry entry) {
236 	Field[] fields = null;
237 
238 	synchronized (AbstractEntry.class) {
239 	    if (fieldArrays == null)
240 		fieldArrays = new WeakHashMap();
241 	    else {
242 		fields = (Field[]) fieldArrays.get(entry.getClass());
243 		if (fields != null)
244 		    return fields;
245 	    }
246 	}
247 
248 	/*
249 	 * Scan the array to see if we can use it or if we must build up
250 	 * a smaller array because we must skip some fields.  If so, we
251 	 * create an ArrayList and add the unskippable fields to it, and
252 	 * then fetch the array back out of it.
253 	 */
254 	final int SKIP_MODIFIERS =
255 	    (Modifier.STATIC | Modifier.TRANSIENT | Modifier.FINAL);
256 	fields = entry.getClass().getFields();
257 	ArrayList usable = null;
258 	for (int i = 0; i < fields.length; i++) {
259 	    // exclude this one?
260 	    if ((fields[i].getModifiers() & SKIP_MODIFIERS) != 0 ||
261 		(fields[i].getType().isPrimitive())) 
262 	    {
263 		if (usable == null) {		// first excluded: set up for it
264 		    usable = new ArrayList();	// allocate the list of usable
265 		    for (int j = 0; j < i; j++)	// earlier fields are usable
266 			usable.add(fields[j]);
267 		}
268 	    } else {				// not excluded
269 		if (usable != null)		// tracking usable fields?
270 		    usable.add(fields[i]);
271 	    }
272 	}
273 	if (usable != null)
274 	    fields = (Field[]) usable.toArray(new Field[usable.size()]);
275 
276 	synchronized (AbstractEntry.class) {
277 	    // We could check to make sure someone else 
278 	    // has not already stuck a value for entry.getClass() in,
279 	    // fieldArrays but there should be no harm in an overwrite
280 	    // and if anything likely to be less efficient
281 	    fieldArrays.put(entry.getClass(), fields);
282 	}
283 
284 	return fields;
285     }
