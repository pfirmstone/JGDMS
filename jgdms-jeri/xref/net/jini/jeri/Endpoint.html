19  package net.jini.jeri;
20  
21  import net.jini.core.constraint.InvocationConstraints;
22  import net.jini.security.proxytrust.TrustEquivalence;
23  
24  /**
25   * Represents a remote communication endpoint to send requests to.
26   *
27   * <p>An <code>Endpoint</code> instance contains the information
28   * necessary to send requests to the remote endpoint.  For example, a
29   * TCP-based <code>Endpoint</code> implementation typically contains
30   * the remote host address and TCP port to connect to.
31   *
32   * <p>The {@link #newRequest newRequest} method can be used to send a
33   * request to the remote endpoint that this object represents.
34   *
35   * <p>An instance of this interface should be serializable and should
36   * implement {@link Object#equals Object.equals} to return
37   * <code>true</code> if and only if the argument is equivalent to the
38   * instance in trust, content, and function.  That is, the
39   * <code>equals</code> method should be a sufficient substitute for
40   * {@link TrustEquivalence#checkTrustEquivalence
41   * TrustEquivalence.checkTrustEquivalence}.  But unlike
42   * <code>checkTrustEquivalence</code>, the <code>equals</code> method
43   * cannot assume that the implementations of any of the invoked
44   * instance's pluggable component objects are trusted (whether
45   * pluggable through public APIs or deserialization).  Therefore, the
46   * <code>equals</code> method should not invoke comparison methods
47   * (such as <code>equals</code>) on any such component without first
48   * verifying that the component's implementation is at least as
49   * trusted as the implementation of the corresponding component in the
50   * <code>equals</code> argument (such as by verifying that the
51   * corresponding component objects have the same actual class).  If
52   * any such verification fails, the <code>equals</code> method should
53   * return <code>false</code> without invoking a comparison method on
54   * the component.  Furthermore, these guidelines should be recursively
55   * obeyed by the comparison methods of each such component for its
56   * subcomponents.
57   *
58   * <p>If an <code>Endpoint</code> is an instance of {@link
59   * TrustEquivalence}, then its <code>equals</code> method must obey
60   * the above guidelines and its <code>checkTrustEquivalence</code>
61   * method must not return <code>true</code> if its <code>equals</code>
62   * method would not return <code>true</code> for the same argument.
63   *
64   * <p>All aspects of the underlying communication mechanism that are not
65   * specified here are defined by the particular implementation of this
66   * interface.
67   *
68   * @author Sun Microsystems, Inc.
69   * @see ServerEndpoint
70   * @since 2.0
71   **/
72  public interface Endpoint {
73  
74      /**
75       * Returns an <code>OutboundRequestIterator</code> to use to send
76       * a new request to this remote endpoint using the specified
77       * constraints.
78       *
79       * <p>The constraints must be the complete, absolute constraints
80       * for the request.
81       *
82       * @param constraints the complete, absolute constraints
83       *
84       * @return an <code>OutboundRequestIterator</code> to use to send
85       * a new request to this remote endpoint
86       *
87       * @throws NullPointerException if <code>constraints</code> is
88       * <code>null</code>
89       **/
90      OutboundRequestIterator newRequest(InvocationConstraints constraints);
91  }
