129     /** Computes the principals in the subject available for authentication */
130     private static Set computePrincipals(Subject subject) {
131 	if (subject == null) {
132 	    return null;
133 	}
134 	/* Get principals from the subject */
135 	X500PrivateCredential[] credentials =
136 	    (X500PrivateCredential[]) AccessController.doPrivileged(
137 		new SubjectCredentials.GetAllPrivateCredentialsAction(
138 		    subject));
139 	Set result = SubjectCredentials.getPrincipals(
140 	    subject, ANY_KEY_ALGORITHM, credentials);
141 	/* Remove ones we don't have authentication listen permission for. */
142 	SecurityManager sm = System.getSecurityManager();
143 	if (sm != null) {
144 	    for (Iterator iter = result.iterator(); iter.hasNext(); ) {
145 		Principal p = (Principal) iter.next();
146 		try {
147 		    sm.checkPermission(
148 			new AuthenticationPermission(
149 			    Collections.singleton(p), null, "listen"));
150 		} catch (SecurityException e) {
151 		    logger.log(Levels.HANDLED,
152 			       "compute principals for server endpoint " +
153 			       "caught exception",
154 			       e);
155 		    iter.remove();
156 		}
157 	    }
158 	}
159 	return result.isEmpty() ? null : result;
160     }

162     /**
163      * Checks that principals is not empty and contains no nulls, and returns
164      * it as a set.  Returns null if no principals are specified.
165      */
166     private static Set<X500Principal> checkPrincipals(X500Principal[] principals) {
167 	if (principals.length == 0) {
168 	    return null;
169 	}
170 	Set<X500Principal> result = new HashSet<X500Principal>(principals.length);
171 	for (int i = principals.length; --i >= 0; ) {
172 	    X500Principal p = principals[i];
173 	    if (p == null) {
174 		throw new NullPointerException(
175 		    "Server principal cannot be null");
176 	    }
177 	    result.add(p);
178 	}
179 	return result;
180     }

204     /**
205      * Two instances of this class are equal if they have the same actual
206      * class; have server subjects that compare equal using ==; have server
207      * principals that are either both null or are equal when compared as the
208      * elements of a Set; have the same server host and port; have socket
209      * factories that are either both null, or have the same actual class and
210      * are equal; and have server socket factories that are either both null,
211      * or have the same actual class and are equal.
212      */
213     public boolean equals(Object object) {
214 	if (object == null || object.getClass() != getClass()) {
215 	    return false;
216 	}
217 	SslServerEndpointImpl other = (SslServerEndpointImpl) object;
218 	return listenEndpoint.serverSubject == other.listenEndpoint.serverSubject
219 	    && safeEquals(getServerPrincipals(), other.getServerPrincipals())
220 	    && safeEquals(getServerHost(), other.getServerHost())
221 	    && getPort() == other.getPort()
222 	    && Util.sameClassAndEquals(getSocketFactory(), other.getSocketFactory())
223 	    && Util.sameClassAndEquals(getServerSocketFactory(), other.getServerSocketFactory());
224     }

251     final InvocationConstraints checkConstraints(
252 	InvocationConstraints constraints)
253 	throws UnsupportedConstraintException
254     {
255 	try {
256 	    listenEndpoint.checkListenPermissions(false);
257 	} catch (SecurityException e) {
258 	    if (logger.isLoggable(Levels.FAILED)) {
259 		logThrow(logger, Levels.FAILED,
260 			 SslServerEndpoint.class, "checkConstraints",
261 			 "check constraints for {0}\nwith {1}\nthrows",
262 			 new Object[] { this, constraints },
263 			 e);
264 	    }
265 	    throw e;
266 	}
267 	Set clientPrincipals = getClientPrincipals(constraints);
268 	if (clientPrincipals == null) {
269 	    clientPrincipals = Collections.singleton(UNKNOWN_PRINCIPAL);
270 	}
271 	Map serverKeyTypes = new HashMap();
272 	List certPaths =
273 	    SubjectCredentials.getCertificateChains(listenEndpoint.serverSubject);
274 	if (certPaths != null) {
275 	    for (int i = certPaths.size(); --i >= 0; ) {
276 		CertPath chain = (CertPath) certPaths.get(i);
277 		X509Certificate cert = SubjectCredentials.firstX509Cert(chain);
278 		X500Principal principal = SubjectCredentials.getPrincipal(
279 		    listenEndpoint.serverSubject, cert);
280 		if (principal != null) {
281 		    Collection keyTypes =
282 			(Collection) serverKeyTypes.get(principal);
283 		    if (keyTypes == null) {
284 			keyTypes = new ArrayList(1);
285 			serverKeyTypes.put(principal, keyTypes);
286 		    }
287 		    keyTypes.add(cert.getPublicKey().getAlgorithm());
288 		}
289 	    }
290 	}
291 	String[] suites = getSupportedCipherSuites();
292 	for (int suiteIndex = suites.length; --suiteIndex >= 0; ) {
293 	    String suite = suites[suiteIndex];
294 	    String suiteKeyType = getKeyAlgorithm(suite);
295 	    Iterator sIter =
296 		(getServerPrincipals() == null
297 		 ? Collections.EMPTY_SET : getServerPrincipals()).iterator();
298 	    X500Principal server;
299 	    do {
300 		if (sIter.hasNext()) {
301 		    server = (X500Principal) sIter.next();
302 		    assert server != null;
303 		    Collection keyTypes =
304 			(Collection) serverKeyTypes.get(server);
305 		    if (keyTypes == null || !keyTypes.contains(suiteKeyType)) {
306 			continue;
307 		    }
308 		} else {
309 		    server = null;
310 		}
311 		Iterator cIter = clientPrincipals.iterator();
312 		Principal client;
313 		do {
314 		    if (cIter.hasNext()) {
315 			client = (Principal) cIter.next();
316 			assert client != null;
317 		    } else {
318 			client = null;
319 		    }
320 		    InvocationConstraints unfulfilledConstraints =
321 			getUnfulfilledIntegrityOrAtomicityConstraints(
322 			    suite, client, server, constraints);
323 		    if (unfulfilledConstraints != null) {
324 			if (logger.isLoggable(Level.FINE)) {
325 			    logger.log(Level.FINE,
326 				       "check constraints for {0}\n" +
327 				       "with {1}\nreturns {2}",
328 				       new Object[] {
329 					   serverEndpoint, constraints,
330 					   unfulfilledConstraints
331 				       });
332 			}
333 			return unfulfilledConstraints;
334 		    }
335 		} while (client != null);
336 	    } while (server != null);
337 	}
338 	UnsupportedConstraintException uce =
339 	    new UnsupportedConstraintException(
340 	    "Constraints are not supported: " + constraints);
341 	if (logger.isLoggable(Levels.FAILED)) {
342 	    logThrow(
343 		logger, Levels.FAILED,
344 		SslServerEndpoint.class, "checkConstraints",
345 		"check constraints for {0}\nwith {1}\nthrows",
346 		new Object[] { serverEndpoint, constraints },
347 		uce);
348 	}
349 	throw uce;
350     }

352     /**
353      * Returns null if the constraints are not supported, else any integrity
354      * constraints required or preferred by the arguments.
355      */
356     static InvocationConstraints getUnfulfilledConstraints(
357 	String cipherSuite,
358 	Principal client,
359 	Principal server,
360 	InvocationConstraints constraints)
361     {
362 	boolean supported = false;
363 	for (int i = 2; --i >= 0; ) {
364 	    boolean integrity = i == 0;
365 	    ConnectionContext context = ConnectionContext.getInstance(cipherSuite, client, server, integrity,
366 		false /* clientSide */, constraints);
367 	    if (context != null) {
368 		if (context.getIntegrityRequired()) {
369 		    return INTEGRITY_REQUIRED;
370 		} else if (context.getIntegrityPreferred()) {
371 		    return INTEGRITY_PREFERRED;
372 		} else {
373 		    supported = true;
374 		}
375 	    }
376 	}
377 	return supported ? InvocationConstraints.EMPTY : null;
378     }

380     /**
381      * Returns null if the constraints are not supported, else any integrity
382      * constraints required or preferred by the arguments.
383      */
384     static InvocationConstraints getUnfulfilledIntegrityOrAtomicityConstraints(
385 	String cipherSuite,
386 	Principal client,
387 	Principal server,
388 	InvocationConstraints constraints)
389     {
390 	List<InvocationConstraint> requiredConstraints = new ArrayList<InvocationConstraint>(2);
391 	List<InvocationConstraint> preferredConstraints = new ArrayList<InvocationConstraint>(2);
392 	boolean supported = false;
393 	for (int i = 2; --i >= 0; ) {
394 	    boolean upperLayerConstraints = i == 1;
395 	    ConnectionContext context = 
396 		    ConnectionContext.getInstance(cipherSuite,
397 			    client,
398 			    server,
399 			    upperLayerConstraints,
400 			    false /* clientSide */,
401 			    constraints);
402 	    if (context != null) {
403 		if (context.getIntegrityRequired()) {
404 		    requiredConstraints.add(Integrity.YES);
405 		} else if (context.getIntegrityPreferred()) {
406 		    preferredConstraints.add(Integrity.YES);
407 		} else if (context.getAtomicityRequired()){
408 		    requiredConstraints.add(AtomicInputValidation.YES);
409 		} else if (context.getAtomicityPreferred()){
410 		    preferredConstraints.add(AtomicInputValidation.YES);
411 		} else {
412 		    supported = true;
413 		}
414 	    }
415 	}
416 	if (!requiredConstraints.isEmpty() || !preferredConstraints.isEmpty()){
417 	    return new InvocationConstraints(
418 		    requiredConstraints,
419 		    preferredConstraints
420 	    );
421 	}
422 	return supported ? InvocationConstraints.EMPTY : null;
423     }

430         String resolvedHost = LOCAL_HOST.check(this.getServerHost(), this);

436             logger.log(Level.FINE,
437                        "enumerate listen endpoints for {0}\nreturns {1}",
438                        new Object[] { this, result });
439         }
440         return result;

451     /**
452      * Checks that the argument is a valid listen cookie for this server
453      * endpoint.
454      */
455     private SslListenCookie checkCookie(ListenCookie cookie) {
456 	if (!(cookie instanceof SslListenCookie)) {
457 	    throw new IllegalArgumentException(
458 		"Cookie must be of type SslListenCookie: " + cookie);
459 	}
460 	SslListenCookie sslListenCookie = ((SslListenCookie) cookie);
461 	ListenEndpoint cookieListenEndpoint
462 	    = sslListenCookie.getListenEndpoint();
463 
464 	if (!listenEndpoint.equals(cookieListenEndpoint)) {
465 	    throw new IllegalArgumentException(
466 		"Cookie has wrong listen endpoint: found " +
467 		cookieListenEndpoint + ", expected " + listenEndpoint);
468 	}
469 	return sslListenCookie;
470     }

566         SslListenEndpoint(Subject serverSubject,
567                             X500Principal[] serverPrincipals,
568                             String serverHost,
569                             int port,
570                             SocketFactory socketFactory,
571                             ServerSocketFactory serverSocketFactory)
572         {
573             this.serverConnectionManager = defaultServerConnectionManager;
574             this.serverHost = serverHost;
575             this.port = port;
576             this.socketFactory = socketFactory;
577             this.serverSocketFactory = serverSocketFactory;
578             boolean useCurrentSubject = serverSubject == null;
579             if (useCurrentSubject) {
580                 final AccessControlContext acc = AccessController.getContext();
581                 serverSubject = AccessController.doPrivileged(
582                     new PrivilegedAction<Subject>() {
583                         @Override
584                         public Subject run() {
585                             return Subject.getSubject(acc);
586                         }
587                     });
588             }
589             this.serverPrincipals = (serverPrincipals == null)
590                 ? computePrincipals(serverSubject)
591                 : checkPrincipals(serverPrincipals);
592             Permission [] listenPerms;
593             if (this.serverPrincipals == null) {
594                 listenPerms = null;
595             } else {
596                 listenPerms =
597                     new AuthenticationPermission[this.serverPrincipals.size()];
598                 int i = 0;
599                 for (Iterator iter = this.serverPrincipals.iterator();
600                      iter.hasNext();
601                      i++)
602                 {
603                     Principal p = (Principal) iter.next();
604                     listenPerms[i] = new AuthenticationPermission(
605                         Collections.singleton(p), null, "listen");
606                 }
607             }
608             if (this.serverPrincipals == null ||
609                 /* Don't use current subject without any permission */
610                 (useCurrentSubject &&
611                  serverPrincipals != null &&
612                  !hasListenPermissions(listenPerms, port)))
613             {
614                 this.serverSubject = null;
615                 listenPerms = null;
616             } else {
617                 this.serverSubject = serverSubject;
618             }
619             this.listenPermissions = listenPerms;
620             
621         }

629         /**
630          * Returns true if the caller has AuthenticationPermission for listen on
631          * this endpoint.
632          */
633         private static boolean hasListenPermissions(Permission[] listenPerms, int port) {
634             try {
635                 checkListenPermissions(false, listenPerms, port);
636                 return true;
637             } catch (SecurityException e) {
638                 logger.log(Levels.HANDLED,
639                            "check listen permissions for server endpoint " +
640                            "caught exception",
641                            e);
642                 return false;
643             }
644         }

646         /**
647          * Check for permission to listen on this endpoint, but only checking
648          * socket permissions if checkSocket is true.
649          */
650         final void checkListenPermissions(boolean checkSocket) {
651             checkListenPermissions(checkSocket, listenPermissions, port);
652         }
653         
654         /**
655          * Check for permission to listen on this endpoint, but only checking
656          * socket permissions if checkSocket is true.
657          */
658         static void checkListenPermissions(boolean checkSocket, Permission[] listenPerms, int port) {
659             SecurityManager sm = System.getSecurityManager();
660             if (sm != null) {
661                 if (checkSocket) {
662                     sm.checkListen(port);
663                 }
664                 if (listenPerms != null) {
665                     for (int i = listenPerms.length; --i >= 0; ) {
666                         sm.checkPermission(listenPerms[i]);
667                     }
668                 }
669             }
670         }

672         /**
673          * Initializes the sslSocketFactory and authManager fields.  Wait to do
674          * this until needed, because creating the SSLContext requires initializing
675          * the secure random number generator, which can be time consuming.
676          */
677         private void sslInit() {
678             assert Thread.holdsLock(this);
679             SSLContextInfo info = getServerSSLContextInfo(
680                 serverSubject, serverPrincipals);
681             sslSocketFactory = info.sslContext.getSocketFactory();
682             authManager = (ServerAuthManager) info.authManager;
683         }

688                 if (sslSocketFactory == null) {
689                     sslInit();
690                 }
691                 return sslSocketFactory;
692             }
693         }

697             synchronized (this) {
698                 if (authManager == null) {
699                     sslInit();
700                 }
701             return authManager;
702             }

721 	 * Check that the subject has credentials for the principals specified
722 	 * when the server endpoint was created.
723 	 */
724 	private void checkCredentials() throws UnsupportedConstraintException {
725 	    if (serverSubject == null) {
726 		return;
727 	    }
728 	    checkListenPermissions(false, listenPermissions, port);
729 	    Set principals = serverSubject.getPrincipals();
730 	    /* Keep track of progress; remove entry when check is done */
731             boolean nullServerPrincipals = serverPrincipals == null;
732 	    Map<X500Principal,Object> progress 
733 		    = new HashMap<X500Principal,Object>(
734 			    nullServerPrincipals ? 0 : serverPrincipals.size());
735             if (!nullServerPrincipals){
736 		for (X500Principal p : serverPrincipals) {
737 		    if (!principals.contains(p)) {
738 			throw new UnsupportedConstraintException(
739 				"Missing principal: " + p);
740 		    }
741 		    progress.put(p, X500Principal.class);
742 		}
743             }
744 	    X500PrivateCredential[] privateCredentials =
745 		(X500PrivateCredential[]) AccessController.doPrivileged(
746 		    new SubjectCredentials.GetAllPrivateCredentialsAction(
747 			serverSubject));
748 	    List certPaths =
749 		SubjectCredentials.getCertificateChains(serverSubject);
750 	    if (certPaths != null) {
751 		for (int i = certPaths.size(); --i >= 0; ) {
752 		    CertPath chain = (CertPath) certPaths.get(i);
753 		    X509Certificate firstCert = firstX509Cert(chain);
754 		    X500Principal p = firstCert.getSubjectX500Principal();
755 		    if (progress.containsKey(p)) {
756 			try {
757 			    checkValidity(chain, null);
758 			} catch (CertificateException e) {
759 			    progress.put(p, e);
760 			    continue;
761 			}
762 			progress.put(p, CertPath.class);
763 			for (int j = privateCredentials.length; --j >= 0; ) {
764 			    X509Certificate cert =
765 				privateCredentials[j].getCertificate();
766 			    if (firstCert.equals(cert)) {
767 				progress.remove(p);
768 				break;
769 			    }
770 			}
771 		    }
772 		}
773 	    }
774 	    if (!progress.isEmpty()) {
775 		Map.Entry<X500Principal,Object> entry 
776 			= progress.entrySet().iterator().next();
777 		X500Principal p = entry.getKey();
778 		Object result = entry.getValue();
779 		if (result == X500Principal.class) {
780 		    throw new UnsupportedConstraintException(
781 			"Missing public credentials: " + p);
782 		} else if (result == CertPath.class) {
783 		    throw new UnsupportedConstraintException(
784 			"Missing private credentials: " + p);
785 		} else {
786 		    throw new UnsupportedConstraintException(
787 			"Problem with certificates: " + p + "\n" + result,
788 			(CertificateException) result);
789 		}
790 	    }
791 	}

898 	/** Handles new socket connections. */
899 	final void acceptLoop() {
900 	    while (true) {
901 		Socket socket = null;
902 		Throwable exception;
903 		SslServerConnection connection = null;
904 		try {
905 		    socket = serverSocket.accept();
906 
907 		    /* Send data without delay */
908 		    try {
909 			socket.setTcpNoDelay(true);
910 		    } catch (SocketException e) {
911 		    }
912 		    /*
913 		     * Send periodic pings so we can tell if the connection
914 		     * dies.
915 		     */
916 		    try {
917 			socket.setKeepAlive(true);
918 		    } catch (SocketException e) {
919 		    }
920 
921 		    connection = serverConnection(socket);
922 		    synchronized (this) {
923 			if (closed) {
924 			    try {
925 				connection.closeInternal(
926 				    false /* removeFromListener */);
927 			    } catch (IOException e) {
928 			    }
929 			    break;
930 			}
931 			connections.add(connection);
932 		    }
933 		    final SslServerConnection finalConnection = connection;
934 		    AccessController.doPrivileged(
935 			securityContext.wrap(
936 			    new PrivilegedAction() {
937 				public Object run() {
938 				    handleConnection(
939 					finalConnection, requestDispatcher);
940 				    return null;
941 				}
942 			    }),
943 			securityContext.getAccessControlContext());
944 		    continue;
945 		} catch (Exception e) {
946 		    exception = e;
947 		} catch (Error e) {
948 		    exception = e;
949 		}
950 		/* Problem occurred */
951 		boolean closedSync;
952 		synchronized (this) {
953 		    closedSync = closed;
954 		}
955 		if (!closedSync && logger.isLoggable(Level.INFO)) {
956 		    String msg = "handling connection {0} throws";
957 		    Object arg = connection;
958 		    if (connection == null) {
959 			msg = "accepting connection on {0} throws";
960 			arg = this;
961 		    }
962 		    logThrow(logger, Level.INFO,
963 			     SslListenHandle.class, "acceptLoop",
964 			     msg, new Object[] { arg }, exception);
965 		}
966 		if (socket != null) {
967 		    try {
968 			socket.close();
969 		    } catch (IOException e) {
970 		    }
971 		}
972 		if (closedSync) {
973 		    break;
974 		}
975 		boolean knownFailure =
976 		    (exception instanceof Exception ||
977 		     exception instanceof OutOfMemoryError ||
978 		     exception instanceof NoClassDefFoundError);
979 		if (!(knownFailure && continueAfterAcceptFailure(exception))) {
980 		    try {
981 			serverSocket.close();
982 		    } catch (IOException e) {
983 		    }
984 		    if (!knownFailure) {
985 			throw (Error) exception;
986 		    } else {
987 			return;
988 		    }
989 		}
990 	    }
991 	}

993 	/**
994 	 * Throttles the accept loop after ServerSocket.accept throws
995 	 * an exception, and decides whether to continue at all.  The
996 	 * current code is borrowed from the JRMP implementation; it
997 	 * always continues, but it delays the loop after bursts of
998 	 * failed accepts.
999 	 */
1000 	private boolean continueAfterAcceptFailure(Throwable t) {
1001 	    /*
1002 	     * If we get a burst of NFAIL failures in NMSEC milliseconds,
1003 	     * then wait for ten seconds.  This is to ensure that individual
1004 	     * failures don't cause hiccups, but sustained failures don't
1005 	     * hog the CPU in futile accept-fail-retry looping.
1006 	     */
1007 	    final int NFAIL = 10;
1008 	    final int NMSEC = 5000;
1009 	    long now = System.currentTimeMillis();
1010             boolean fail = false;
1011             synchronized (failureLock){
1012                 if (acceptFailureTime == 0L ||
1013                     (now - acceptFailureTime) > NMSEC)
1014                 {
1015                     // failure time is very old, or this is first failure
1016                     acceptFailureTime = now;
1017                     acceptFailureCount = 0;
1018                 } else {
1019                     // failure window was started recently
1020                     acceptFailureCount++;
1021                     if (acceptFailureCount >= NFAIL) {
1022                         fail = true;
1023                     }
1024                 }
1025             }
1026             if (fail) {
1027                 try {
1028                     Thread.sleep(10000);
1029                 } catch (InterruptedException ignore) {
1030                     /* Why are we ignoring the interrupt and not 
1031                      * restoring the interrupted status?
1032                      */
1033                     Thread.currentThread().interrupt();
1034                 }
1035                 // no need to reset counter/timer
1036             }
1037 	    
1038 	    return true;
1039 	}

1404 	    }
1405 	    InvocationConstraints result = getUnfulfilledIntegrityOrAtomicityConstraints(
1406 		cipherSuite, clientPrincipal, serverPrincipal, constraints);
1407 	    if (result == null) {
1408 		UnsupportedConstraintException uce = 
1409 		    new UnsupportedConstraintException(
1410 			"Constraints are not supported: " + constraints);
1411 		if (logger.isLoggable(Levels.FAILED)) {
1412 		    logThrow(logger, Levels.FAILED,
1413 			     SslServerConnection.class, "checkConstraints",
1414 			     "check constraints for {0}\nwith {1}\n" +
1415 			     "throws",
1416 			     new Object[] {
1417 				 SslServerConnection.this, constraints
1418 			     },
1419 			     uce);
1420 		}
1421 		throw uce;
1422 	    }
1423 	    if (logger.isLoggable(Level.FINE)) {
1424 		logger.log(Level.FINE,
1425 			   "check constraints for {0}\nwith {1}\nreturns {2}",
1426 			   new Object[] {
1427 			       SslServerConnection.this, constraints,
1428 			       result
1429 			   });
1430 	    }
1431 	    return result;
1432 	}
