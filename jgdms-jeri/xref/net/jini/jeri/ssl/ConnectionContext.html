44  /**
45   * Records information about a connection used for a remote call and determines
46   * whether the connection could support specific constraints. <p>
47   *
48   * Does not support heterogeneous constraint alternatives.  As a result,
49   * callers can be assured that the choice of possible principals or of
50   * Integrity.YES is independent of the suites, insuring that those items can be
51   * picked before negotiating the cipher suite.
52   *
53   * 
54   */
55  final class ConnectionContext extends Utilities {
56  
57      /** Constraints supported, without integrity or connection timeout */
58      private static final long OK = Long.MAX_VALUE;
59      
60      /** Constraints supported with deserialization atomic input validation */
61      private static final long ATOMICITY = -2;
62  
63      /** Constraints supported with codebase integrity */
64      private static final long INTEGRITY = -3;
65  
66      /** Constraints not supported */
67      private static final long NOT_SUPPORTED = -4;
68  
69      /** The ClientMinPrincipalType supported by the provider. */
70      private static final ClientMinPrincipalType clientMinPrincipalType =
71  	new ClientMinPrincipalType(X500Principal.class);
72  
73      /** The cipher suite */
74      final String cipherSuite;
75  
76      /** The client principal, or null for an anonymous client */
77      final Principal client;
78  
79      /** The server principal, or null for an anonymous server */
80      final Principal server;
81  
82      /** Whether codebase integrity and atomicity should be enforced */
83      private final boolean upperLayerConstraints;
84  
85      /**
86       * Whether the connection is being considered on the client side, which
87       * does not support relative time constraints.
88       */
89      private final boolean clientSide;
90  
91      /** Set to true if the principals and cipher suite conflict */
92      private boolean notSupported;
93  
94      /** Whether the requirements specify Integrity.YES */
95      private boolean integrityRequired;
96  
97      /** Whether the preferences specify Integrity.YES */
98      private boolean integrityPreferred;
99      
100     /** Whether the requirements specify AtomicInputValidation.YES */
101     private boolean atomicityRequired;
102     
103     /** Whether the preferences specify AtomicInputValidation.YES */
104     private boolean atomicityPreferred;
105 
106     /** The absolute connection time, or Long.MAX_VALUE if not specified */
107     private long connectionTime = Long.MAX_VALUE;
108 
109     /** The number of preferences satisfied */
110     private int preferences;

112     /**
113      * Creates an instance that represents using the specified cipher suite,
114      * client and server principals, whether to guarantee codebase integrity,
115      * input validation failure atomicity during deserialization
116      * and constraints.  Null values for the principals mean they are
117      * anonymous.  Non-X.500 principals are permitted to allow specifying a
118      * dummy principal if the principal is unknown.  Returns null if the
119      * constraints are not supported.
120      */
121     static ConnectionContext getInstance(
122 	    String cipherSuite, 
123 	    Principal client, 
124 	    Principal server,
125 	    boolean upperLayerConstraints, 
126 	    boolean clientSide, 
127 	    InvocationConstraints constraints)
128     {
129 	ConnectionContext context = new ConnectionContext(
130 	    cipherSuite, client, server, upperLayerConstraints, clientSide);
131 	return context.supported(constraints) ? context : null;
132     }

134     /** Creates an instance of this class. */
135     private ConnectionContext(String cipherSuite,
136 				Principal client,
137 				Principal server,
138 				boolean upperLayerConstraints,
139 				boolean clientSide)
140     {
141 	this.cipherSuite = cipherSuite;
142 	this.client = client;
143 	this.server = server;
144 	this.upperLayerConstraints = upperLayerConstraints;
145 	this.clientSide = clientSide;
146 	boolean serverAuth = doesServerAuthentication(cipherSuite);
147 	if (serverAuth != (server != null)
148 	    || (client != null && server == null))
149 	{
150 	    notSupported = true;
151 	}
152     }

155 	StringBuffer sb = new StringBuffer("ConnectionContext[");
156 	fieldsToString(sb);
157 	sb.append("]");
158 	return sb.toString();

161     void fieldsToString(StringBuffer sb) {
162 	sb.append(cipherSuite);
163 	if (client != null) {
164 	    sb.append(", client: ").append(client);
165 	}
166 	if (server != null) {
167 	    sb.append(", server: ").append(server);
168 	}
169 	if (integrityRequired) {
170 	    sb.append(", integrity: required");
171 	} else if (integrityPreferred) {
172 	    sb.append(", integrity: preferred");
173 	}
174 	if (atomicityRequired) {
175 	    sb.append(", input validation failure atomicity: required");
176 	} else if (atomicityPreferred) {
177 	    sb.append(", input validation failure atomicity: preferred");
178 	}
179 	if (connectionTime != Long.MAX_VALUE) {
180 	    sb.append(", connectionTime = ").append(connectionTime);
181 	}
182 	sb.append(", preferences: ").append(preferences);
183     }

218     /**
219      * Checks if the specified constraints are supported, computing
220      * integrityRequired, integrityPreferred, connectionTime and preferences as
221      * a side effect.
222      */
223     private boolean supported(InvocationConstraints constraints) {
224 	if (notSupported) {
225 	    return false;
226 	}
227 	for (Iterator i = constraints.requirements().iterator(); i.hasNext(); )
228 	{
229 	    long r = supported((InvocationConstraint) i.next());
230 	    if (r == NOT_SUPPORTED) {
231 		return false;
232 	    }
233 	    if (r == INTEGRITY) {
234 		integrityRequired = true;
235 	    } else if (r == ATOMICITY){
236 		atomicityRequired = true;
237 	    } else if (connectionTime > r) {
238 		connectionTime = r;
239 	    }
240 	}
241 	for (Iterator i = constraints.preferences().iterator(); i.hasNext(); ) {
242 	    long r = supported((InvocationConstraint) i.next());
243 	    if (r == NOT_SUPPORTED) {
244 	      continue;
245 	    }
246 	    preferences++;
247 	    if (r == INTEGRITY) {
248 		if (!integrityRequired) {
249 		    integrityPreferred = true;
250 		}
251 	    } else if (r == ATOMICITY){
252 		if (!atomicityRequired){
253 		    atomicityPreferred = true;
254 		}
255 	    } else if (connectionTime > r) {
256 		connectionTime = r;
257 	    }
258 	}
259 	if (upperLayerConstraints){
260 	    if (!integrityRequired && !integrityPreferred && !atomicityRequired
261 		    && !atomicityPreferred) return false;
262 	}
263 	return true;
264     }

266     /**
267      * Checks if the constraint is supported, returning NOT_SUPPORTED if it is
268      * not supported, INTEGRITY if the constraint is Integrity.YES or
269      * constraint alternatives with elements of type Integrity, the connection
270      * time if the constraint is an instance of ConnectionAbsoluteTime or
271      * constraint alternatives of them, and otherwise OK.
272      */
273     private long supported(InvocationConstraint constraint) {
274 	if (constraint instanceof ConstraintAlternatives) {
275 	    return supported((ConstraintAlternatives) constraint);
276 	} else if (constraint instanceof Integrity) {
277 	    return upperLayerConstraints && constraint == Integrity.YES
278 		? INTEGRITY : NOT_SUPPORTED;
279 	} else if (constraint instanceof AtomicInputValidation){
280 	    return upperLayerConstraints && constraint == AtomicInputValidation.YES
281 		? ATOMICITY : NOT_SUPPORTED;
282 	} else if (constraint instanceof Confidentiality) {
283 	    return ok(doesEncryption(cipherSuite) ==
284 		      (constraint == Confidentiality.YES));
285 	} else if (constraint instanceof ConfidentialityStrength) {
286 	    return ok(!doesEncryption(cipherSuite) ||
287 		      (hasStrongKeyCipherAlgorithms(cipherSuite)
288 		       == (constraint == ConfidentialityStrength.STRONG)));
289 	} else if (constraint instanceof ClientAuthentication) {
290 	    return ok((client != null) ==
291 		      (constraint == ClientAuthentication.YES));
292 	} else if (constraint instanceof ClientMinPrincipalType) {
293 	    return ok(constraint.equals(clientMinPrincipalType));
294 	} else if (constraint instanceof ClientMaxPrincipalType) {
295 	    return ok(((ClientMaxPrincipalType) constraint).elements().contains(
296 			  X500Principal.class));
297 	} else if (constraint instanceof ClientMinPrincipal) {
298 	    Set elements = ((ClientMinPrincipal) constraint).elements();
299 	    return ok(elements.size() == 1 && elements.contains(client));
300 	} else if (constraint instanceof ClientMaxPrincipal) {
301 	    return ok(((ClientMaxPrincipal) constraint).elements().contains(client));
302 	} else if (constraint instanceof Delegation) {
303 	    return ok((constraint == Delegation.NO));
304 	} else if (constraint instanceof DelegationAbsoluteTime) {
305 	    return OK;
306 	} else if (constraint instanceof DelegationRelativeTime) {
307 	    return ok(!clientSide);
308 	} else if (constraint instanceof ServerAuthentication) {
309 	    return ok((server != null) ==
310 		      (constraint == ServerAuthentication.YES));
311 	} else if (constraint instanceof ServerMinPrincipal) {
312 	    Set elements = ((ServerMinPrincipal) constraint).elements();
313 	    return ok(elements.size() == 1 && elements.contains(server));
314 	} else if (constraint instanceof net.jini.core.constraint.ServerMaxPrincipal) {
315 	    return ok(((net.jini.core.constraint.ServerMaxPrincipal) constraint).elements().contains(server));
316 	}else if (constraint instanceof ConnectionAbsoluteTime) {
317 	    return Math.max(((ConnectionAbsoluteTime) constraint).getTime(), 0);
318 	} else if (constraint instanceof ConnectionRelativeTime) {
319 	    return ok(!clientSide);
320 	} else {
321 	    return NOT_SUPPORTED;
322 	}
323     }
324 
325     /** Returns OK if the argument is true, else NOT_SUPPORTED. */
326     private static long ok(boolean ok) {
327 	return ok ? OK : NOT_SUPPORTED;
328     }
329 
330     /**
331      * Checks if the constraint alternatives are supported, returning
332      * NOT_SUPPORTED if the elements have different types or none are
333      * supported, INTEGRITY if the elements are instances of Integrity, the
334      * largest connection time if the elements are instances of
335      * ConnectionAbsoluteTime, and otherwise OK.
336      */
337     private long supported(ConstraintAlternatives constraint) {
338 	Set alts = constraint.elements();
339 	long connectionTime = -1;
340 	Class type = null;
341 	boolean supported = false;
342 	boolean integrity = false;
343 	for (Iterator i = alts.iterator(); i.hasNext(); ) {
344 	    InvocationConstraint alt = (InvocationConstraint) i.next();
345 	    if (type == null) {
346 		type = alt.getClass();
347 	    } else if (type != alt.getClass()) {
348 		return NOT_SUPPORTED;
349 	    }
350 	    long r = supported(alt);
351 	    if (r != NOT_SUPPORTED) {
352 		supported = true;
353 		if (r == INTEGRITY) {
354 		    integrity = true;
355 		} else if (r > connectionTime) {
356 		    connectionTime = r;
357 		}
358 	    }
359 	}
360 	if (!supported) {
361 	    return NOT_SUPPORTED;
362 	} else if (integrity) {
363 	    return INTEGRITY;
364 	} else if (connectionTime >= 0) {
365 	    return connectionTime;
366 	} else {
367 	    return OK;
368 	}
369     }

371     @Override
372     public int hashCode() {
373 	int hash = 7;
374 	hash = 17 * hash + (this.cipherSuite != null ? this.cipherSuite.hashCode() : 0);
375 	hash = 17 * hash + (this.client != null ? this.client.hashCode() : 0);
376 	hash = 17 * hash + (this.server != null ? this.server.hashCode() : 0);
377 	hash = 17 * hash + (this.upperLayerConstraints ? 1 : 0);
378 	hash = 17 * hash + (this.clientSide ? 1 : 0);
379 	hash = 17 * hash + (this.notSupported ? 1 : 0);
380 	hash = 17 * hash + (this.integrityRequired ? 1 : 0);
381 	hash = 17 * hash + (this.integrityPreferred ? 1 : 0);
382 	hash = 17 * hash + (this.atomicityRequired ? 1 : 0);
383 	hash = 17 * hash + (this.atomicityPreferred ? 1 : 0);
384 	hash = 17 * hash + (int) (this.connectionTime ^ (this.connectionTime >>> 32));
385 	hash = 17 * hash + this.preferences;
386 	return hash;
387     }

389     @Override
390     public boolean equals(Object o){
391 	if (!(o instanceof ConnectionContext)) return false;
392 	ConnectionContext that = (ConnectionContext) o;
393 	if (this.atomicityPreferred != that.atomicityPreferred) return false;
394 	if (this.atomicityRequired != that.atomicityRequired) return false;
395 	if (this.clientSide != that.clientSide) return false;
396 	if (this.integrityPreferred != that.integrityPreferred) return false;
397 	if (this.integrityRequired != that.integrityRequired) return false;
398 	if (this.notSupported != that.notSupported) return false;
399 	if (this.upperLayerConstraints != that.upperLayerConstraints) return false;
400 	if (this.connectionTime != that.connectionTime) return false;
401 	if (this.preferences != that.preferences) return false;
402 	if (!this.cipherSuite.equals(that.cipherSuite)) return false;
403 	if (!this.client.equals(that.client)) return false;
404 	return this.server.equals(that.server);
405     }
