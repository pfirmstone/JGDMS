23  import java.util.Set;

32  /**
33   * Records information needed to make a remote call.  The information consists
34   * of the endpoint, client subject, requirement for client authentication,
35   * permitted client and server principals, cipher suites, and integrity and
36   * connection time constraints.
37   *
38   * 
39   */
40  class CallContext extends Utilities implements OutboundRequestHandle {
41  
42      /** The endpoint. */
43      final Endpoint endpoint;
44  
45      /** The associated endpoint implementation delegate. */
46      final SslEndpointImpl endpointImpl;
47  
48      /** The client subject. */
49      final Subject clientSubject;
50  
51      /** Whether client authentication is required. */
52      final boolean clientAuthRequired;
53  
54      /**
55       * The principals in the client subject that may be used for
56       * authentication, or null if the client is anonymous.
57       */
58      final Set clientPrincipals;
59  
60      /**
61       * The principals that will be accepted for authentication of the server
62       * subject, or null if all principals are permitted or if the server is
63       * anonymous.
64       */
65      final Set serverPrincipals;
66  
67      /** An ordered list of cipher suites to use. */
68      final String[] cipherSuites;
69  
70      /** Whether codebase integrity is required. */
71      final boolean integrityRequired;
72  
73      /** Whether codebase integrity is preferred. */
74      final boolean integrityPreferred;
75      
76      /** Whether deserialization atomic input validation is required. */
77      final boolean atomicityRequired;
78  
79      /** Whether deserialization atomic input validation is preferred. */
80      final boolean atomicityPreferred;
81  
82      /**
83       * The absolute time by which a new connection must be completed, or
84       * Long.MAX_VALUE for no restriction.
85       */
86      final long connectionTime;
87  
88      /**
89       * Converts an OutboundRequestHandle intended for the specified endpoint
90       * into a CallContext.  Throws NullPointerException if the handle is null.
91       * Throws IllegalArgumentException if the instance is not associated with
92       * the endpoint.
93       */
94      static CallContext coerce(OutboundRequestHandle handle, Endpoint endpoint) {
95  	if (handle == null) {
96  	    throw new NullPointerException("Handle cannot be null");
97  	} else if (!(handle instanceof CallContext)) {
98  	    throw new IllegalArgumentException(
99  		"Handle must be of type CallContext: " + handle);
100 	}
101 	CallContext context = (CallContext) handle;
102 	if (!endpoint.equals(context.endpoint)) {
103 	    throw new IllegalArgumentException(
104 		"Handle has wrong endpoint -- was " + context.endpoint +
105 		", should be " + endpoint);
106 	}
107 	return context;
108     }
109 
110     /** Creates a CallContext. */
111     CallContext(Endpoint endpoint,
112 		SslEndpointImpl endpointImpl,
113 		Subject clientSubject,
114 		boolean clientAuthRequired,
115 		Set clientPrincipals,
116 		Set serverPrincipals,
117 		List cipherSuites,
118 		boolean integrityRequired,
119 		boolean integrityPreferred,
120 		boolean atomicityRequired,
121 		boolean atomicityPreferred,
122 		long connectionTime)
123     {
124 	this.endpoint = endpoint;
125 	this.endpointImpl = endpointImpl;
126 	this.clientSubject = clientSubject;
127 	this.clientAuthRequired = clientAuthRequired;
128 	this.clientPrincipals = clientPrincipals;
129 	this.serverPrincipals = serverPrincipals;
130 	this.cipherSuites =
131 	    (String[]) cipherSuites.toArray(new String[cipherSuites.size()]);
132 	this.integrityRequired = integrityRequired;
133 	this.integrityPreferred = integrityPreferred;
134 	this.atomicityRequired = atomicityRequired;
135 	this.atomicityPreferred = atomicityPreferred;
136 	this.connectionTime = connectionTime;
137     }
138 
139     public String toString() {
140 	StringBuffer buff = new StringBuffer("CallContext[");
141 	buff.append("\n  ").append(endpoint);
142 	buff.append("\n  clientSubject=");
143 	if (clientSubject == null) {
144 	    buff.append("null");
145 	} else {
146 	    buff.append("Subject@");
147 	    buff.append(
148 		Integer.toHexString(
149 		    System.identityHashCode(clientSubject)));
150 	}
151 	buff.append("\n  clientAuthRequired=").append(clientAuthRequired);
152 	buff.append("\n  clientPrincipals=").append(clientPrincipals);
153 	buff.append("\n  serverPrincipals=").append(serverPrincipals);
154 	buff.append("\n  cipherSuites=").append(toString(cipherSuites));
155 	if (integrityRequired) {
156 	    buff.append("\n  integrity=required");
157 	} else if (integrityPreferred) {
158 	    buff.append("\n  integrity=preferred");
159 	}
160 	if (atomicityRequired) {
161 	    buff.append("\n  deserialization input validation failure atomicity=required");
162 	} else if (atomicityPreferred) {
163 	    buff.append("\n  deserialization input validation failure atomicity=preferred");
164 	}
165 	if (connectionTime != Long.MAX_VALUE) {
166 	    buff.append("\n  connectionTime=").append(connectionTime);
167 	}
168 	buff.append("\n]");
169 	return buff.toString();
170     }

172     /**
173      * Returns any constraints that must be partially or fully implemented by
174      * higher layers for this outbound request.
175      */
176     InvocationConstraints getUnfulfilledConstraints() {
177 	List<InvocationConstraint> required = new ArrayList<InvocationConstraint>(2);
178 	List<InvocationConstraint> preferred = new ArrayList<InvocationConstraint>(2);
179 	if (integrityRequired) {
180 	    required.add(Integrity.YES);
181 	} else if (integrityPreferred) {
182 	    preferred.add(Integrity.YES);
183 	}
184 	if (atomicityRequired){
185 	    required.add(AtomicInputValidation.YES);
186 	} else if (atomicityPreferred){
187 	    preferred.add(AtomicInputValidation.YES);   
188 	}
189 	if (!required.isEmpty() || !preferred.isEmpty()){
190 	    return new InvocationConstraints(required, preferred);
191 	}
192 	return InvocationConstraints.EMPTY;
193     }
