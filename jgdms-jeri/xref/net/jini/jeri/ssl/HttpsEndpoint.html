97  /**
98   * An implementation of {@link Endpoint} that uses HTTPS (HTTP over TLS/SSL) to
99   * support invocation constraints for communication through firewalls. <p>
100  *
101  * Instances of this class are intended to be created by the {@link
102  * BasicJeriExporter} class when it calls {@link
103  * ServerEndpoint#enumerateListenEndpoints enumerateListenEndpoints} on
104  * instances of {@link HttpsServerEndpoint}. <p>
105  *
106  * The {@link SslTrustVerifier} trust verifier may be used for establishing
107  * trust in remote proxies that use instances of this class. <p>
108  *
109  * This class supports at least the following constraints, possibly limited by
110  * the available cipher suites: <p>
111  *
112  * <ul>
113  * <li> {@link ClientAuthentication}
114  * <li> {@link ClientMaxPrincipal}, when it contains an {@link X500Principal}
115  * <li> {@link ClientMaxPrincipalType}, when it contains
116  *	<code>X500Principal</code>
117  * <li> {@link ClientMinPrincipal}, when it contains a single
118  *	<code>X500Principal</code> only
119  * <li> {@link ClientMinPrincipalType}, when it contains
120  *	<code>X500Principal</code> only
121  * <li> {@link Confidentiality}
122  * <li> {@link ConfidentialityStrength}, a provider-specific constraint for
123  *	specifying weak or strong confidentiality
124  * <li> {@link ConnectionAbsoluteTime}
125  * <li> {@link ConstraintAlternatives}, if the elements all have the same
126  *	actual class and at least one element is supported
127  * <li> {@link Delegation#NO}
128  * <li> {@link Delegation#YES}, trivially, for anonymous clients
129  * <li> {@link DelegationAbsoluteTime}, trivially, when delegation is not
130  *	supported
131  * <li> {@link Integrity#YES}
132  * <li> {@link ServerAuthentication}
133  * <li> {@link ServerMinPrincipal}, when it contains a single
134  *	<code>X500Principal</code> only
135  * </ul> <p>
136  *
137  * Note that {@link ConnectionRelativeTime} and {@link DelegationRelativeTime}
138  * constraints may be used at higher levels, but should be converted to the
139  * associated absolute time constraints for use by this class. <p>
140  *
141  * This class authenticates as a single {@link Principal} if the following
142  * items are present in the current {@link Subject}: <p>
143  *
144  * <ul>
145  * <li> One or more principals of type <code>X500Principal</code>
146  * <li> For each principal, one or more certificate chains, stored as public
147  *      credentials, and represented by instances of {@link CertPath}, whose
148  *      <code>getType</code> method returns "X.509", and for which calling
149  *      <code>getSubjectDN</code> on the certificate chain's first element
150  *      returns that principal's name
151  * <li> For each certificate chain, an instance of {@link
152  *      X500PrivateCredential}, stored as a private credential, whose
153  *      <code>getCertificate</code> method returns a value equal to the first
154  *      element of the certificate chain, and whose <code>getPrivateKey</code>
155  *      method returns the associated private key
156  * </ul> <p>
157  *
158  * In addition, this class's {@link #newRequest newRequest} method will only
159  * authenticate as a given principal if the caller has been granted {@link
160  * AuthenticationPermission} with that principal as the local principal, the
161  * principal representing the authenticated identity of the server as the peer
162  * principal, and the <code>connect</code> action. <p>
163  *
164  * This class supports remote connections between authenticated servers and
165  * authenticated or anonymous clients, and between anonymous servers and
166  * anonymous clients. Connections between anonymous servers and authenticated
167  * clients are not supported. Because of the suites available in the TLS/SSL
168  * protocol, support for {@link Confidentiality#NO} requires the server to
169  * authenticate with an RSA public key. <p>
170  *
171  * This class permits specifying a {@link SocketFactory} for creating the
172  * {@link Socket} instances that it uses to make remote connections. These
173  * socket factories should not be instances of {@link SSLSocketFactory} or
174  * return instances {@link SSLSocket}; it is the responsibility of the
175  * implementation to establish a TLS/SSL connection over the socket it obtains
176  * from the socket factory. <p>
177  *
178  * A <code>SocketFactory</code> used with instances of this class should be
179  * serializable, and must implement {@link Object#equals Object.equals} to
180  * obey the guidelines that are specified for <code>equals</code> methods of
181  * {@link Endpoint} instances. <p>
182  *
183  * This class recognizes the following system properties: <p>
184  * 
185  * <ul>
186  * <li> https.proxyHost - The host name for the secure proxy server. The
187  *	default is to use no proxy server.
188  * <li> https.proxyPort - The port for the secure proxy server. The default is
189  *	443.
190  * <li> http.nonProxyHosts - The names of hosts for which direct connections
191  *	should be made rather than using the proxy server. Each host name may
192  *	contain '<code>*</code>' wildcard characters in any position to match
193  *	zero or more of any characters within the name. Multiple host names may
194  *	be specified by separating the names with '<code>|</code>'
195  *	characters. The default is for all connections to use the proxy server
196  *	if one is specified.
197  * </ul>
198  *
199  * 
200  * @see HttpsServerEndpoint
201  * @see ConfidentialityStrength
202  * @see SslTrustVerifier
203  * @since 2.0
204  *
205  *  <!-- Implementation Specifics -->
206  *
207  * This implementation uses the <a
208  * href="http://java.sun.com/j2se/1.4/docs/guide/security/jsse/JSSERefGuide.html"
209  * target="_top">Java(TM) Secure Socket Extension (JSSE)</a>. <p>
210  *
211  * This implementation uses the {@link ConnectionManager} class to manage
212  * connections. <p>
213  *
214  * This implementation uses the following {@link Logger} instances in the
215  * <code>net.jini.jeri.ssl</code> namespace: <p>
216  *
217  * <ul>
218  * <li> <a href="#init_logger">init</a> - problems during initialization
219  * <li> <a href="#client_logger">client</a> - information about
220  *	client-side connections
221  * </ul> <p>
222  *
223  * <a name="init_logger"></a>
224  * <table border="1" cellpadding="5" summary="Describes logging to the init
225  *	  logger performed by the HttpsEndpoint class at different logging
226  *	  levels">
227  *
228  * <caption><b><code>
229  *	    net.jini.jeri.ssl.init</code></b></caption>
230  *     
231  * <tr> <th scope="col"> Level <th scope="col"> Description
232  *
233  * <tr> <td> {@link Level#WARNING WARNING} <td> problems with initializing JSSE
234  *
235  * </table> <p>
236  *
237  * <a name="client_logger"></a>
238  * <table border="1" cellpadding="5" summary="Describes logging to the client
239  *	  logger performed by the HttpsEndpoint class at different logging
240  *	  levels">
241  *
242  * <caption><b><code>
243  *	    net.jini.jeri.ssl.client</code></b></caption>
244  *     
245  * <tr> <th scope="col"> Level <th scope="col"> Description
246  *
247  * <tr> <td> {@link Levels#FAILED FAILED} <td> problems with outbound requests
248  *
249  * <tr> <td> {@link Levels#HANDLED HANDLED} <td> exceptions caught involving
250  * authentication
251  *

254  *
255  * <tr> <td> {@link Level#FINEST FINEST} <td> low level operation tracing
256  *
257  * </table> <p>
258  *
259  * This implementation uses the following security providers: <p>
260  *
261  * <ul>
262  * <li> {@link SSLContext}, with the protocol specified by the
263  *	<code>org.apache.river.jeri.ssl.sslProtocol</code> system property, or
264  *	<code>"TLS"</code> if that property is not defined, to provide the
265  *	TLS/SSL implementation. The {@link SSLContext#init SSLContext.init}
266  *	method is called with <code>null</code> for the <code>random</code>
267  *	parameter to use the default {@link SecureRandom} implementation.
268  * <li> {@link CertificateFactory}, with type <code>"X.509"</code>, to generate
269  *	<code>CertPath</code> instances from X.509 certificate chains
270  * <li> {@link TrustManagerFactory}, with the algorithm specified by the
271  *	<code>org.apache.river.jeri.ssl.trustManagerFactoryAlgorithm</code> system
272  *	property, or the default algorithm if that property is not defined, to
273  *	implement trust management for the TLS/SSL implementation. The factory
274  *	must return trust managers that implement {@link X509TrustManager}.
275  * </ul> <p>
276  * 
277  * See the documentation on <a
278  * href="http://java.sun.com/j2se/1.4/docs/guide/security/CryptoSpec.html#ProviderInstalling"
279  * target="_top">installing security providers</a> and <a
280  * href="http://java.sun.com/j2se/1.4/docs/guide/security/jsse/JSSERefGuide.html#ProviderCust"
281  * target="_top">configuring JSSE</a> for information on configuring these
282  * providers. <p>
283  *
284  * The <a
285  * href="http://java.sun.com/j2se/1.4/docs/guide/security/jsse/JSSERefGuide.html#Customization"
286  * target="_top">JSSE documentation</a> also describes the system properties
287  * for configuring the location, type, and password of the truststore that this
288  * implementation uses, through JSSE, to make decisions about what certificate
289  * chains should be trusted. <p>
290  * 
291  * This implementation recognizes the following system properties: <p>
292  * 
293  * <ul>
294  * <li> <code>org.apache.river.jeri.ssl.maxClientSessionDuration</code> - The
295  *	maximum number of milliseconds a client-side TLS/SSL session should be
296  *	used. The default is 23.5 hours. The value should be smaller than the
297  *	maximum server session duration to allow the client to negotiate a new
298  *	session before the server timeout occurs.
299  * <li> <code>org.apache.river.jeri.ssl.sslProtocol</code> - The secure socket
300  *	protocol used when obtaining {@link SSLContext} instances. The default
301  *	is <code>"TLS"</code>.
302  * <li> <code>org.apache.river.jeri.ssl.trustManagerFactoryAlgorithm</code> - The
303  *	algorithm used when obtaining {@link TrustManagerFactory}
304  *	instances. The default is the value returned by {@link
305  *	TrustManagerFactory#getDefaultAlgorithm
306  *	TrustManagerFactory.getDefaultAlgorithm}.
307  * <li> <code>org.apache.river.jeri.ssl.cipherSuites</code> - The TLS/SSL cipher
308  *	suites that should be used for communication. The default is the list
309  *	of suites supported by the JSSE implementation. The value should
310  *	specify the suite names, separated by commas. The value will be ignored
311  *	if it contains no suites or specifies suites that are not supported by
312  *	the JSSE implementation. Suites appearing earlier in the list will be
313  *	preferred to ones appearing later for suites that support the same
314  *	requirements and preferences.
315  * <li> <code>org.apache.river.jeri.https.idleConnectionTimeout</code> - The number
316  *	of milliseconds to retain idle client-side HTTPS connections before
317  *	closing them. The default is <code>15000</code>.
318  * <li> <code>org.apache.river.jeri.https.responseAckTimeout</code> - The number of
319  *	milliseconds to keep track of acknowledgments that have not yet been
320  *	sent for {@link AcknowledgmentSource} instances. The default is
321  *	<code>15000</code>.
322  * <li> <code>org.apache.river.jeri.https.pingProxyConnections</code> - If
323  *      the value is case-insensitive equal to <code>true</code>, then if an
324  *      HTTP proxy is being used, ping the server endpoint to verify whether
325  *      it is alive and reachable. The ping occurs before the first request
326  *      and before each subsequent request which follows the expiration of
327  *      the ping proxy timeout period (below) following the previous ping.
328  *      When using an HTTP proxy it is often impossible to distinguish
329  *      between inability to reach the server endpoint (such as because the
330  *      server process refused a connection by the HTTP proxy) and the lack
331  *      of response from a delivered request (which might result in an
332  *      UnmarshalException). The ping increases the likelihood that the
333  *      inability to reach the server endpoint can be explicitly identified.
334  *	The default value is <code>false</code>, and no pings are done.
335  * <li> <code>org.apache.river.jeri.https.pingProxyConnectionTimeout</code> - The
336  *      number of milliseconds from the time a server endpoint was last
337  *      pinged before a ping will precede the next request. The default is
338  *      <code>Long.MAX_VALUE</code> (essentially meaning, ping only before
339  *      the first request).
340  * </ul>

514     /**
515      * {@inheritDoc} <p>
516      *
517      * <p>The returned <code>OutboundRequestIterator</code>'s {@link
518      * OutboundRequestIterator#next next} method behaves as follows:
519      *
520      * <blockquote>
521      *
522      * Initiates an attempt to communicate the request to this remote
523      * endpoint.
524      *
525      * <p>When the implementation of this method needs to create a new
526      * <code>Socket</code>, it will do so by invoking one of the
527      * <code>createSocket</code> methods on the
528      * <code>SocketFactory</code> of this <code>HttpsEndpoint</code>
529      * (which produced this iterator) if non-<code>null</code>, or it
530      * will create a <code>Socket</code> directly otherwise.
531      *
532      * <p>When the implementation needs to connect a
533      * <code>Socket</code>, if the host name to connect to (if an HTTP
534      * proxy is to be used for the communication, the proxy's host
535      * name; otherwise, this <code>HttpsEndpoint</code>'s host name)
536      * resolves to multiple addresses (according to {@link
537      * InetAddress#getAllByName InetAddress.getAllByName}), it
538      * attempts to connect to the first resolved address; if that
539      * attempt fails with an <code>IOException</code> or (as is
540      * possible in the case that an HTTP proxy is not to be used) a
541      * <code>SecurityException</code>, it then attempts to connect to
542      * the next address; and this iteration continues as long as there
543      * is another resolved address and the attempt to connect to the
544      * previous address fails with an <code>IOException</code> or a
545      * <code>SecurityException</code>.  If the host name resolves to
546      * just one address, the implementation makes one attempt to
547      * connect to that address.  If the host name does not resolve to
548      * any addresses (<code>InetAddress.getAllByName</code> would
549      * throw an <code>UnknownHostException</code>), the implementation
550      * still makes an attempt to connect the <code>Socket</code> to
551      * that host name, which could result in an
552      * <code>UnknownHostException</code>.  If the final connection
553      * attempt fails with an <code>IOException</code> or a
554      * <code>SecurityException</code>, then if any connection attempt
555      * failed with an <code>IOException</code>, this method throws an
556      * <code>IOException</code>, and otherwise (if all connection
557      * attempts failed with a <code>SecurityException</code>), this
558      * method throws a <code>SecurityException</code>.
559      *
560      * <p>If there is a security manager and an HTTP proxy is to be
561      * used for the communication, the security manager's {@link
562      * SecurityManager#checkConnect(String,int) checkConnect} method
563      * is invoked with this <code>HttpsEndpoint</code>'s host and port;
564      * if this results in a <code>SecurityException</code>, this
565      * method throws that exception.
566      * 
567      * <p>If there is a security manager and an HTTP proxy is not to
568      * be used for the communication:
569      *
570      * <ul>
571      *
572      * <li>If a new connection is to be created, the security
573      * manager's {@link SecurityManager#checkConnect(String,int)
574      * checkConnect} method is invoked with this
575      * <code>HttpsEndpoint</code>'s host and <code>-1</code> for the
576      * port; if this results in a <code>SecurityException</code>, this
577      * method throws that exception.  <code>checkConnect</code> is
578      * also invoked for each connection attempt, with the remote IP
579      * address (or the host name, if it could not be resolved) and
580      * port to connect to; this could result in a
581      * <code>SecurityException</code> for that attempt.  (Note that
582      * the implementation may carry out these security checks
583      * indirectly, such as through invocations of
584      * <code>InetAddress.getAllByName</code> or <code>Socket</code>'s
585      * constructors or <code>connect</code> method.)
586      *
587      * <li><p>In order to reuse an existing connection for the
588      * communication, the current security context must have all of
589      * the permissions that would be necessary if the connection were
590      * being created.  Specifically, it must be possible to invoke
591      * <code>checkConnect</code> in the current security context with
592      * this <code>HttpsEndpoint</code>'s host and <code>-1</code> for
593      * the port without resulting in a <code>SecurityException</code>,
594      * and it also must be possible to invoke
595      * <code>checkConnect</code> with the remote IP address and port
596      * of the <code>Socket</code> without resulting in a
597      * <code>SecurityException</code> (if the remote socket address is
598      * unresolved, its host name is used instead).  If no existing
599      * connection satisfies these requirements, then this method must
600      * behave as if there are no existing connections.
601      *
602      * </ul>
603      *
604      * <p>Throws {@link NoSuchElementException} if this iterator does
605      * not support making another attempt to communicate the request
606      * (that is, if <code>hasNext</code> would return
607      * <code>false</code>).
608      *
609      * <p>Throws {@link IOException} if an I/O exception occurs while
610      * performing this operation, such as if a connection attempt
611      * timed out or was refused or there are unsupported or conflicting
612      * constraints.
613      *
614      * <p>Throws {@link SecurityException} if there is a security
615      * manager and an invocation of its <code>checkConnect</code>
616      * method fails, or if the caller does not have the appropriate
617      * <code>AuthenticationPermission</code>.
618      *
619      * </blockquote>
620      *
621      * @throws NullPointerException {@inheritDoc}
622      */
623     public OutboundRequestIterator newRequest(
624 	InvocationConstraints constraints)
625     {
626 	return impl.newRequest(constraints);
627     }

741 	 * Implements Endpoint.newRequest when the constraints are supported.
742 	 */
743 	OutboundRequestIterator newRequest(final CallContext callContext) {
744 	    return new OutboundRequestIterator() {
745 		private boolean done;
746 		public synchronized boolean hasNext() {
747 		    return !done;
748 		}
749 		public synchronized OutboundRequest next() throws IOException {
750 		    if (!hasNext()) {
751 			throw new NoSuchElementException();
752 		    }

754 		    return getOutboundRequest(callContext);

759 	/** Returns an outbound request for the specified call context. */
760 	OutboundRequest getOutboundRequest(CallContext callContext)
761 	    throws IOException
762 	{
763 	    HttpsConnection connection = null;
764 	    EndpointInfo info;
765 	    synchronized (endpointMap) {
766 		info = (EndpointInfo) endpointMap.get(this);
767 		if (info == null) {
768 		    info = new EndpointInfo(this);
769 		}
770 	    }
771 	    connection = info.connect(callContext);
772 	    boolean ok = false;
773 	    try {
774 
775 		// If using a connection through a proxy, and if the
776 		// connectionTimeout has passed, then ping the server
777 		// endpoint to verify the connection.  This reduces the
778 		// likelihood that a connection dropped by the server
779 		// won't be noticed until after some or all of the data
780 		// destined for the server has already been sent to the
781 		// proxy.
782 
783 		if (connection.usesHttpProxy()) {
784 		    HttpSettings settings = getHttpSettings();
785 		    if (settings.getPingProxyConnections()) {
786 			long timeout = settings.getPingProxyConnectionTimeout();
787 			long now = System.currentTimeMillis();
788 			if (now - timeLastVerified > timeout) {
789 			    pingEndpoint(connection);
790 			    // The ping succeeded, but the connection
791 			    // cannot be reused as it was marked idle.
792 			    connection = info.connect(callContext);
793 			    timeLastVerified = System.currentTimeMillis();
794 			}
795 		    }
796 		}
797 		OutboundRequest request = connection.newRequest(callContext);
798 		ok = true;
799 		logger.log(Level.FINE, "using {0}", connection);
800 		return request;
801 	    } finally {
802 		if (!ok) {
803 		    info.noteClosed(connection);
804 		}
805 	    }
806 	}

1084     /**
1085      * Manages the open connections associated with endpoints equal to a
1086      * representative endpoint.
1087      */
1088     private static final class EndpointInfo {
1089 
1090 	/**
1091 	 * A representative endpoint that equals the endpoint for all the
1092 	 * connections.
1093 	 */
1094 	private final HttpsEndpointImpl endpoint;
1095 
1096 	/** The proxy host, or empty string if using a direct connection. */
1097 	private String proxyHost = "";
1098 
1099 	/** The proxy port, ignored if using a direct connection. */
1100 	private int proxyPort = 0;
1101 
1102 	/** The idle connections for the endpoint. */
1103 	private final List idle = new ArrayList(1);
1104 
1105 	/** The connections that are in use for the endpoint. */
1106 	private final List inUse = new ArrayList(1);
1107 
1108 	EndpointInfo(HttpsEndpointImpl endpoint) {
1109 	    this.endpoint = endpoint;
1110 	}
1111 
1112 	/**
1113 	 * Chooses and returns an idle connection that satisfies the
1114 	 * constraints, removing the connection from the list of idle
1115 	 * connections and adding it to the list of ones in use, else returns
1116 	 * null.
1117 	 */
1118 	HttpsConnection connect(CallContext context)
1119 	    throws IOException
1120 	{
1121 	    HttpSettings settings = getHttpSettings();
1122 	    String phost = settings.getProxyHost(endpoint.serverHost);
1123 	    int pport = (phost.length() == 0) ? 0 : settings.getProxyPort();
1124 	    HttpsConnection result;
1125 	    synchronized (this) {
1126 		List reap = new ArrayList(0);
1127 		if (!(proxyHost.equals(phost) && proxyPort == pport)) {
1128 		    proxyHost = phost;
1129 		    proxyPort = pport;
1130 		    reap.addAll(idle);
1131 		    idle.clear();
1132 		} else {
1133 		    checkIdle(System.currentTimeMillis(), reap);
1134 		}
1135 		for (int i = reap.size(); --i >= 0; ) {
1136 		    try {
1137 			((HttpsConnection) reap.get(i)).close();
1138 		    } catch (IOException e) {
1139 		    }
1140 		}
1141 		result = (HttpsConnection) endpoint.connect(
1142 		    context, Collections.EMPTY_SET, idle);
1143 		if (result != null) {
1144 		    idle.remove(result);
1145 		} else {
1146 		    result = new HttpsConnection(endpoint, context,
1147 						 endpoint.serverHost,
1148 						 endpoint.port, phost, pport,
1149 						 endpoint.socketFactory);
1150 		    result.establishCallContext();
1151 		}
1152 	    }
1153 	    synchronized (endpointMap) {
1154 		EndpointInfo info = (EndpointInfo) endpointMap.get(endpoint);
1155 		if (info == null) {
1156 		    if (endpointMap.isEmpty()) {
1157 			systemExecutor.execute(
1158 			    new Reaper(), "HttpsEndpointReaper");
1159 		    }
1160 		    endpointMap.put(endpoint, this);
1161 		    info = this;
1162 		}
1163 		info.noteInUse(result);
1164 	    }
1165 	    return result;
1166 	}

1168 	/** Adds a connection to the set of connections in use. */
1169 	synchronized void noteInUse(HttpsConnection connection) {
1170 	    inUse.add(connection);
1171 	}
1172 
1173 	/** Removes a connection from the set of connections in use. */
1174 	synchronized void noteClosed(HttpsConnection connection) {
1175 	    inUse.remove(connection);
1176 	}
1177 
1178 	/**
1179 	 * Removes a connection from the set of connections in use and adds it
1180 	 * to the set of idle connections.
1181 	 */
1182 	synchronized void noteIdle(HttpsConnection connection) {
1183 	    inUse.remove(connection);
1184 	    if (proxyHost.equals(connection.proxyHost) &&
1185 		proxyPort == connection.proxyPort)
1186 	    {
1187 		idle.add(connection);
1188 	    } else {
1189 		try {
1190 		    connection.close();
1191 		} catch (IOException e) {
1192 		}
1193 	    }
1194 	}
1195 
1196 	/**
1197 	 * For each connection, calls checkIdle on the connection and, if that
1198 	 * returns true, removes the connection and adds it to the reap list.
1199 	 * Returns true if no in-use or idle connections remain.
1200 	 */
1201 	synchronized boolean checkIdle(long now, List reap) {
1202 	    for (int i = idle.size(); --i >= 0; ) {
1203 		HttpsConnection connection = (HttpsConnection) idle.get(i);
1204 		if (connection.checkIdle(now)) {
1205 		    reap.add(connection);
1206 		    idle.remove(connection);
1207 		}
1208 	    }
1209 	    return idle.isEmpty() && inUse.isEmpty();
1210 	}
