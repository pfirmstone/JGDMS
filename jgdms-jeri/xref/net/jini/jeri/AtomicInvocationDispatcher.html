40      public AtomicInvocationDispatcher(Collection methods, ServerCapabilities serverCapabilities, MethodConstraints serverConstraints, Class permissionClass, ClassLoader loader) throws ExportException {
41  	super(methods, serverCapabilities, serverConstraints, permissionClass, loader);
42      }

44      /**
45       * Returns a new marshal input stream to use to read objects from the
46       * request input stream obtained by invoking the {@link
47       * InboundRequest#getRequestInputStream getRequestInputStream} method
48       * on the given <code>request</code>.
49       *
50       * <p><code>AtomicInvocationDispatcher</code> implements this method as
51       * follows:
52       *
53       * <p>First, a class loader is selected to use as the
54       * <code>defaultLoader</code> and the <code>verifierLoader</code> for
55       * the marshal input stream instance.  If the class loader specified at
56       * construction is not <code>null</code>, the selected loader is that
57       * loader.  Otherwise, if a security manager exists, its {@link
58       * SecurityManager#checkPermission checkPermission} method is invoked
59       * with the permission <code>{@link
60       * RuntimePermission}("getClassLoader")</code>; this invocation may
61       * throw a <code>SecurityException</code>.  If the above security check
62       * succeeds, the selected loader is the class loader of
63       * <code>impl</code>'s class.
64       *
65       * <p>This method returns a new {@link MarshalInputStream} instance
66       * constructed with the input stream (obtained from the
67       * <code>request</code> as specified above) for the input stream
68       * <code>in</code>, the selected loader for <code>defaultLoader</code>
69       * and <code>verifierLoader</code>, the boolean <code>integrity</code>
70       * for <code>verifyCodebaseIntegrity</code>, and an unmodifiable view
71       * of <code>context</code> for the <code>context</code> collection.
72       * The {@link MarshalInputStream#useCodebaseAnnotations
73       * useCodebaseAnnotations} method is invoked on the created stream
74       * before it is returned.
75       *
76       * <p>A subclass can override this method to control how the marshal input
77       * stream is created or implemented.
78       *
79       * @param	impl the remote object
80       * @param	request the inbound request
81       * @param	integrity <code>true</code> if object integrity is being
82       * 		enforced for the remote call, and <code>false</code> otherwise
83       * @param	context the server context
84       * @return	a new marshal input stream for unmarshalling a call request
85       * @throws	IOException if an I/O exception occurs
86       * @throws	NullPointerException if any argument is <code>null</code>
87       **/
88      @Override
89      protected ObjectInputStream
90          createMarshalInputStream(Object impl,
91  				 final InboundRequest request,
92  				 final boolean integrity,
93  				 Collection context)
94  	throws IOException
95      {
96  	final ClassLoader streamLoader = getStreamLoader(impl);
97  	
98  	final Collection unmodContext = Collections.unmodifiableCollection(context);
99  	try {
100 	    return AccessController.doPrivileged(new PrivilegedExceptionAction<ObjectInputStream>(){
101 		
102 		@Override
103 		public ObjectInputStream run() throws Exception {
104 		    return AtomicMarshalInputStream.createObjectInputStream(
105 			    request.getRequestInputStream(),
106 			    streamLoader, integrity,
107 			    streamLoader, unmodContext);
108 		}
109     
110 	    });
111 	} catch (PrivilegedActionException ex) {
112 	    Exception cause = ex.getException();
113 	    if (cause instanceof IOException) throw (IOException) cause;
114 	    if (cause instanceof RuntimeException) throw (RuntimeException) cause;
115 	    throw new IOException ("Unable to create ObjectOutputStream ",ex);
116 	}
117     }
118     
119     /**
120      * Returns a new marshal output stream to use to write objects to the
121      * response output stream obtained by invoking the {@link
122      * InboundRequest#getResponseOutputStream getResponseOutputStream}
123      * method on the given <code>request</code>.
124      *
125      * <p>This method will be called with a <code>null</code>
126      * <code>method</code> argument if an <code>IOException</code> occurred
127      * when reading method information from the incoming call stream.
128      *
129      * <p><code>BasicInvocationDispatcher</code> implements this method to
130      * return a new {@link MarshalOutputStream} instance constructed with
131      * the output stream obtained from the <code>request</code> as
132      * specified above and an unmodifiable view of the given
133      * <code>context</code> collection.
134      *
135      * <p>A subclass can override this method to control how the marshal output
136      * stream is created or implemented.
137      *
138      * @param	impl the remote object
139      * @param   method the possibly-<code>null</code> <code>Method</code>
140      *		instance corresponding to the interface method invoked on
141      *		the remote object
142      * @param	request the inbound request
143      * @param	context the server context
144      * @return	a new marshal output stream for marshalling a call response
145      * @throws	IOException if an I/O exception occurs
146      * @throws	NullPointerException if <code>impl</code>,
147      *		<code>request</code>, or <code>context</code> is
148      *		<code>null</code>
149      **/
150     @Override
151     protected ObjectOutputStream
152         createMarshalOutputStream(final Object impl,
153 				  Method method,
154 				  final InboundRequest request,
155 				  final Collection context)
156 	throws IOException
157     {
158 	if (impl == null) {
159 	    throw new NullPointerException();
160 	}
161 	try {
162 	    return AccessController.doPrivileged(new PrivilegedExceptionAction<ObjectOutputStream>(){
163 		
164 		@Override
165 		public ObjectOutputStream run() throws IOException {
166 		    return new AtomicMarshalOutputStream(
167 			    request.getResponseOutputStream(), 
168 			    getStreamLoader(impl), 
169 			    Collections.unmodifiableCollection(context), 
170 			    true
171 		    );
172 		}
173 							  
174 	    });
175 	} catch (PrivilegedActionException ex) {
176 	    Exception cause = ex.getException();
177 	    if (cause instanceof IOException) throw (IOException) cause;
178 	    if (cause instanceof RuntimeException) throw (RuntimeException) cause;
179 	    throw new IOException ("Unable to create ObjectOutputStream ",ex);
180 	}
181 	
182 	
183     }
