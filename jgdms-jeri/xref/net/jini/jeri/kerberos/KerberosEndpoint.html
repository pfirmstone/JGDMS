93  /**
94   * An {@link Endpoint} implementation that uses Kerberos as the
95   * underlying network security protocol to support security related
96   * invocation constraints its caller specified for the corresponding
97   * remote request.  Instances of this class are referred to as the
98   * endpoints of the Kerberos provider, while instances of {@link
99   * KerberosServerEndpoint} are referred to as the server endpoints of
100  * the provider.<p>
101  *
102  * Instances of this class are intended to be created by the {@link
103  * net.jini.jeri.BasicJeriExporter} class when it calls {@link
104  * net.jini.jeri.ServerEndpoint#enumerateListenEndpoints
105  * enumerateListenEndpoints} on instances of
106  * <code>KerberosServerEndpoint</code>. <p>
107  *
108  * The {@link KerberosTrustVerifier} may be used for establishing
109  * trust in remote proxies that use instances of this class. <p>
110  *
111  * This class supports at least the following standard constraints: <p>
112  *
113  * <ul>
114  * <li>{@link net.jini.core.constraint.Integrity#YES}
115  * <li>{@link net.jini.core.constraint.Confidentiality}
116  * <li>{@link net.jini.core.constraint.ClientAuthentication#YES}
117  * <li>{@link net.jini.core.constraint.ConnectionAbsoluteTime}
118  * <li>{@link net.jini.core.constraint.ServerAuthentication#YES}
119  * <li>{@link net.jini.core.constraint.ClientMaxPrincipal}, when it
120  *     contains at least one {@link KerberosPrincipal}
121  * <li>{@link net.jini.core.constraint.ClientMaxPrincipalType}, when
122  *     it contains the <code>KerberosPrincipal</code> class
123  * <li>{@link net.jini.core.constraint.ClientMinPrincipal}, when it
124  *     contains exactly one <code>KerberosPrincipal</code>
125  * <li>{@link net.jini.core.constraint.ClientMinPrincipalType}, when
126  *     it contains only the <code>KerberosPrincipal</code> class
127  * <li>{@link net.jini.core.constraint.ServerMinPrincipal}, when it
128  *     contains exactly one <code>KerberosPrincipal</code>
129  * <li>{@link net.jini.core.constraint.Delegation}
130  * <li>{@link net.jini.core.constraint.ConstraintAlternatives}, if the
131  *     elements all have the same actual class and at least one
132  *     element is supported
133  * </ul> <p>
134  *
135  * An endpoint of this provider uses Kerberos Ticket Granting Tickets
136  * (TGTs), which are stored as private credentials in the {@link
137  * Subject} associated with the access control context of the current
138  * thread, to authenticate the caller of the remote request to the
139  * server. <p>
140  *
141  * A TGT is an instance of {@link KerberosTicket} whose server
142  * principal is of the form <code>"krbtgt/REALM1@REALM2"</code>.  The
143  * client principal of the TGT indicates what principal the endpoint
144  * can use the TGT to authenticate the caller as to the server. <p>
145  *
146  * Instances of this class contain a server host name, a server TCP
147  * port number, a server principal, as well as an optional {@link
148  * SocketFactory} for customizing the type of <code>Socket</code> to
149  * use. <p>
150  *
151  * A <code>SocketFactory</code> used with instances of this class
152  * should be serializable, and should implement {@link Object#equals
153  * Object.equals} to return <code>true</code> when passed an instance
154  * that represents the same (functionally equivalent) socket
155  * factory. <p>
156  *
157  * To make a remote request through an endpoint of this provider, the
158  * caller or initiator of the request supplies the endpoint with a set
159  * of <code>InvocationConstraints</code>.  The client principal used
160  * for the request is a <code>KerberosPrincipal</code> that appears in
161  * the principal set of the current subject, allowable by the given
162  * set of constraints, whose corresponding TGT is in the private
163  * credential set of the subject, and the <code>connect</code>
164  * <code>AuthenticationPermission</code> has been granted to the
165  * current access control context with the client principal as the
166  * <code>local</code> principal and the <code>serverPrincipal</code>
167  * contained in this endpoint as the <code>peer</code> principal. <p>
168  *
169  * If the set of constraints for the request specifies that encryption
170  * and/or delegation is required or preferred, they will be enforced
171  * by the provider.  If encryption is unspecified in the constraints,
172  * the provider may or may not encrypt messages exchanged with the
173  * server.  If delegation is unspecified in the constraints, the
174  * provider will not do delegation for the request.  For delegation to
175  * happen, the TGT for the request has to be forwardable. <p>
176  *
177  * This class uses the <a
178  * href="../connection/doc-files/mux.html">Jini extensible remote
179  * invocation (Jini ERI) multiplexing protocol</a> to map outgoing
180  * requests to the underlying secure connection streams. <p>
181  *
182  * The secure connection streams in this provider are implemented
183  * using the Kerberos Version 5 GSS-API Mechanism, defined in <a
184  * href="http://www.ietf.org/rfc/rfc1964.txt">RFC 1964</a>, over
185  * socket connections between client and server endpoints. <p>
186  *
187  * Note that, because Kerberos inherently requires client authentication,
188  * this transport provider does not support distributed garbage collection
189  * (DGC); if DGC is enabled using {@link net.jini.jeri.BasicJeriExporter},
190  * all DGC remote calls through this provider will silently fail.
191  *
192  *  <!-- Implementation Specifics -->
193  *
194  * This Kerberos provider implementation uses the <a
195  * href="http://www.ietf.org/rfc/rfc2853.txt"> Java(TM) GSS-API</a> to
196  * provide the underlying Kerberos network authentication protocol
197  * support. <p>
198  *
199  * The implementation does not automatically renew any renewable TGTs
200  * in the <code>Subject</code> corresponding to any outbound
201  * request. The assumption is that an endpoint of this provider should
202  * merely be a consumer of the principals and credentials of the
203  * <code>Subject</code>, and never change its content. But if new TGTs
204  * are added into the <code>Subject</code> or old TGTs in the
205  * <code>Subject</code> are renewed by means outside this provider,
206  * the endpoint will pick up and use these new TGTs for new requests
207  * after the old ones have expired. <p>
208  *
209  * This class uses the following {@link Logger} to log information at
210  * the following logging levels: <p>
211  *
212  * <table border="1" cellpadding="5" summary="Describes logging to the
213  *     client logger performed by endpoint classes in this package at
214  *     different logging levels">
215  * 
216  *     <caption><b><code>
217  * 	net.jini.jeri.kerberos.client</code></b></caption>
218  * 
219  *     <tr> <th scope="col"> Level <th scope="col"> Description
220  * 
221  *     <tr> <td> {@link java.util.logging.Level#WARNING WARNING}
222  *          <td> failure to register with discovery provider
223  *     <tr> <td> {@link org.apache.river.logging.Levels#FAILED FAILED}
224  *          <td> problem to support constraint requirements, connect to
225  *               server through socket, establish {@link
226  *               org.ietf.jgss.GSSContext} to server over established
227  *               connections, or wrap/unwrap GSS tokens
228  *     <tr> <td> {@link org.apache.river.logging.Levels#HANDLED HANDLED}
229  *          <td> exceptions caught attempting to set TCP no delay or keep
230  *               alive properties on sockets, connect a socket, or reuse
231  *               a connection
232  *     <tr> <td> {@link java.util.logging.Level#FINE FINE}
233  *          <td> endpoint creation, {@link
234  *               net.jini.jeri.Endpoint#newRequest newRequest}
235  *               invocation, request handle creation, connection
236  *               configuration decesions, socket creation, connection
237  *               open/close, connection reuse decesions,
238  *               <code>GSSContext</code> establishment
239  *     <tr> <td> {@link java.util.logging.Level#FINEST FINEST}
240  *          <td> data message encoding/decoding using
241  *               <code>GSSContext</code>
242  * </table> <p>
243  *
244  * Instances of this class recognize the following system properties:
245  * <p>
246  * 
247  * <ul>
248  * <li>org.apache.river.jeri.kerberos.KerberosEndpoint.minGssContextLifetime -
249  *     Minimum number of seconds of remaining lifetime a {@link
250  *     GSSContext} of an existing connection has to have before it can
251  *     be considered as a candidate connection to be chosen for a new
252  *     request. The default is 30.
253  * <li>org.apache.river.jeri.kerberos.KerberosEndpoint.maxGssContextRetries -
254  *     <a href="http://www.ietf.org/rfc/rfc1510.txt">RFC 1510</a>
255  *     specifies that if a KDC or server receives two authenticators
256  *     with the same client and server pair and timestamps of the
257  *     same microsecond, the second will be considered a replay
258  *     and will be rejected.  This means if multiple session ticket
259  *     requests of the same client and server principal pair and
260  *     microsecond timestamps are received at a KDC, only the first
261  *     one will succeed, and the rest will be considered replays
262  *     and will be rejected by the KDC.  For this reason, the Kerberos
263  *     provider catches the "replay" exception and retries the
264  *     corresponding <code>GSSContext</code> initialization
265  *     handshake.  This system property controls the maximum number
266  *     of retries a <code>KerberosEndpoint</code> will conduct.  The
267  *     default is 3.
268  * </ul> <p>
269  *
270  * 
271  * @see KerberosServerEndpoint
272  * @see KerberosTrustVerifier
273  * @since 2.0
274  */

391     /**
392      * Creates an endpoint of this Kerberos provider.
393      *
394      * @param serverHost the name or ip address of the server host
395      *        this endpoint will connect to
396      * @param serverPort the server port
397      * @param serverPrincipal principal the server can authenticate as
398      * @param csf the <code>SocketFactory</code> to use for this
399      *        <code>KerberosEndpoint</code>, or <code>null</code>
400      * @throws NullPointerException if <code>serverHost</code> or
401      *         <code>serverPrincipal</code> is <code>null</code>
402      * @throws IllegalArgumentException if <code>serverPort</code> is
403      *         not in the range of <code>1</code> to

405      */
406     private KerberosEndpoint(String serverHost, int serverPort,
407 			     KerberosPrincipal serverPrincipal,
408 			     SocketFactory csf)
409     {
410 	if (serverHost == null)
411 	    throw new NullPointerException("serverHost is null");
412 
413 	if (serverPort <= 0 || serverPort > 0xFFFF) {
414 	    throw new IllegalArgumentException(
415 	        "server port number out of range 1-65535: serverPort = " +
416 		serverPort);
417 	}
418 
419 	if (serverPrincipal == null)
420 	    throw new NullPointerException("serverPrincipal is null");
421 
422 	this.serverHost = serverHost;
423 	this.serverPort = serverPort;
424 	this.serverPrincipal = serverPrincipal;
425 	this.csf = csf;
426 	logger.log(Level.FINE, "created {0}", toString()); //previously allowed "this" to escape during construction.
427     }

529     /**
530      * {@inheritDoc}
531      *
532      * <p>The returned <code>OutboundRequestIterator</code>'s {@link
533      * OutboundRequestIterator#next next} method behaves as follows:
534      *
535      * <blockquote>
536      *
537      * Initiates an attempt to communicate the request to this remote
538      * endpoint.
539      *
540      * <p>When the implementation of this method needs to create a new
541      * <code>Socket</code>, it will do so by invoking one of the
542      * <code>createSocket</code> methods on the
543      * <code>SocketFactory</code> of this <code>KerberosEndpoint</code>
544      * (which produced this iterator) if non-<code>null</code>, or it
545      * will create a <code>Socket</code> directly otherwise.
546      *
547      * <p>When the implementation needs to connect a
548      * <code>Socket</code>, if the host name to connect to (this
549      * <code>KerberosEndpoint</code>'s host name) resolves to multiple
550      * addresses (according to {@link InetAddress#getAllByName
551      * InetAddress.getAllByName}), it attempts to connect to the first
552      * resolved address; if that attempt fails with an
553      * <code>IOException</code> or a <code>SecurityException</code>,
554      * it then attempts to connect to the next address; and this
555      * iteration continues as long as there is another resolved
556      * address and the attempt to connect to the previous address
557      * fails with an <code>IOException</code> or a
558      * <code>SecurityException</code>.  If the host name resolves to
559      * just one address, the implementation makes one attempt to
560      * connect to that address.  If the host name does not resolve to
561      * any addresses (<code>InetAddress.getAllByName</code> would
562      * throw an <code>UnknownHostException</code>), the implementation
563      * still makes an attempt to connect the <code>Socket</code> to
564      * that host name, which could result in an
565      * <code>UnknownHostException</code>.  If the final connection
566      * attempt fails with an <code>IOException</code> or a
567      * <code>SecurityException</code>, then if any connection attempt
568      * failed with an <code>IOException</code>, this method throws an
569      * <code>IOException</code>, and otherwise (if all connection
570      * attempts failed with a <code>SecurityException</code>), this
571      * method throws a <code>SecurityException</code>.
572      *
573      * <p>If there is a security manager:
574      *
575      * <ul>
576      *
577      * <li>If a new connection is to be created, the security
578      * manager's {@link SecurityManager#checkConnect(String,int)
579      * checkConnect} method is invoked with this
580      * <code>KerberosEndpoint</code>'s host and <code>-1</code> for the
581      * port; if this results in a <code>SecurityException</code>, this
582      * method throws that exception.  <code>checkConnect</code> is
583      * also invoked for each connection attempt, with the remote IP
584      * address (or the host name, if it could not be resolved) and
585      * port to connect to; this could result in a
586      * <code>SecurityException</code> for that attempt.  (Note that
587      * the implementation may carry out these security checks
588      * indirectly, such as through invocations of
589      * <code>InetAddress.getAllByName</code> or <code>Socket</code>'s
590      * constructors or <code>connect</code> method.)
591      *
592      * <li><p>In order to reuse an existing connection for the
593      * communication, the current security context must have the
594      * <code>KerberosPrincipal</code>, and permissions required to use it,
595      * that would be necessary if the the connection were being created.  In
596      * addition, it must be possible to invoke <code>checkConnect</code> in the
597      * current security context with this <code>KerberosEndpoint</code>'s host
598      * and <code>-1</code> for the port without resulting in a
599      * <code>SecurityException</code>, and it also must be possible to invoke
600      * <code>checkConnect</code> with the remote IP address and port
601      * of the <code>Socket</code> without resulting in a
602      * <code>SecurityException</code> (if the remote socket address is
603      * unresolved, its host name is used instead).  If no existing
604      * connection satisfies these requirements, then this method must
605      * behave as if there are no existing connections.
606      *
607      * </ul>
608      *
609      * <p>Throws {@link java.util.NoSuchElementException}
610      * if this iterator does not support making another attempt to communicate
611      * the request (that is, if <code>hasNext</code> would return
612      * <code>false</code>).
613      *
614      * <p>Throws {@link IOException} if an I/O exception occurs while
615      * performing this operation, such as if a connection attempt
616      * timed out or was refused or there are unsupported or conflicting
617      * constraints.
618      *
619      * <p>Throws {@link SecurityException} if there is a security
620      * manager and an invocation of its <code>checkConnect</code>
621      * method fails.  Also, a <code>SecurityException</code> may be thrown if
622      * the caller does not have the appropriate
623      * <code>AuthenticationPermission</code>.
624      *
625      * </blockquote>
626      *
627      * @throws NullPointerException {@inheritDoc}
628      **/
629     public OutboundRequestIterator newRequest(
630 	InvocationConstraints constraints)
631     {
632         if (constraints == null)
633 	    throw new NullPointerException("constraints cannot be null");
634 
635 	logger.log(Level.FINE, "newRequest requested with constraints:\n" +
636 		   "{0}", constraints);
637 
638 	Subject clientSubject = (Subject) Security.doPrivileged(
639 	    new PrivilegedAction() {
640 		    public Object run() {
641 			return Subject.getSubject(
642 			    AccessController.getContext());
643 		    }
644 		});
645 
646 	CacheKey key = new CacheKey(clientSubject, constraints);
647 	RequestHandleImpl handle = (RequestHandleImpl) softCache.get(key);
648 
649 	if (handle == null || !handle.reusable(clientSubject)) {
650 	    handle = new RequestHandleImpl(clientSubject, constraints);
651 	    
652 	    if (logger.isLoggable(Level.FINE)) {
653 		logger.log(Level.FINE, "new request handle has been " +
654 			   "constructed:\n{0}", new Object[] {handle});
655 	    }
656 	    
657 	    softCache.put(key, handle);
658 	}
659 
660 	return connManager.newRequest(handle);
661     }

871     /** <code>OutboundRequestHandle</code> implementation */
872     private final class RequestHandleImpl implements OutboundRequestHandle {
873 
874 	/** Subject from which private credentials will be extracted */
875 	private Subject clientSubject;
876 
877 	/** Constraints of this request handle */
878 	private InvocationConstraints constraints;
879 
880 	/** True if the subject is readonly when this handle is instantiated */
881 	private boolean subjectReadOnly;
882 
883 	/**
884 	 * In case of subject is not readonly, snapshot its Kerberos
885 	 * principals
886 	 */
887 	private Set subjectClientPrincipals;
888 
889 	/**
890 	 * The set of Kerberos principals allowed by the constraint
891 	 * requirements and found in the principal set of the subject.
892 	 */
893 	private Set clientPrincipals;
894 
895 	/**
896 	 * Error code of this request handle.  A request will not
897 	 * succeed if its handle's errorCode != NO_ERROR.
898 	 */
899 	private int errorCode = NO_ERROR;
900 
901 	/**
902 	 * The message explains the reason of the failure, later on an
903 	 * <code>UnsupportedConstraintException</code> will be
904 	 * instantiated using this message and conditionally thrown to
905 	 * the caller, depending on whether the caller has the
906 	 * "getSubject" <code>AuthPermission</code>.
907 	 */
908 	private String detailedExceptionMsg;
909 
910 	/**
911 	 * Set of configurations that can satisfy the given set of
912 	 * constraints using principals in the current subject.  The
913 	 * set is composed without checking the private credentials of
914 	 * the subject and AuthenticationPermissions of the caller.
915 	 */
916 	private Config[] configs;
917 
918 	/**
919 	 * Constraints that must be partially or fully implemented by
920 	 * higher layers for an outbound request using this handle.
921 	 */
922 	private InvocationConstraints unfulfilledConstraints;
923 
924 	/** Absolute time by when the connection must be established */
925         long connectionAbsoluteTime;
926 
927 	/**
928 	 * Construct a <code>RequestHandleImpl</code>. <p>
929 	 *
930 	 * For each outgoing request, the computation took to
931          * determine the {@link KerberosUtil.Config} for the {@link
932          * ConnectionImpl Connection} carrying it is divided into two
933          * phases.  The first phase includes steps that no
934          * <code>AuthenticationPermission</code> checks are needed,
935          * which are done once for each constraints and subject pair
936          * in this constructor.  The second phase contains steps that
937          * require <code>AuthenticationPermission</code> checks, which
938          * are done for each request in {@link
939          * RequestHandleImpl#getConfigs}.  All problems, even
940          * encountered in the first phase, are reported in the second
941          * phase in <code>getConfigs</code>. <p>
942 	 *
943 	 * The computation steps taken in phase one are listed as the
944 	 * following:
945 	 *
946 	 * <ul>
947 	 * <li> Check for unsupportable requirements
948 	 * <li> Determine client principal candidate set based on
949 	 *      client principal constraints in requirements and
950 	 *      current subject
951 	 * <li> Generate all possible <code>Configs</code> based on
952 	 *      the set of client principal candidates, and whether
953 	 *      encryption and delegation are mentioned/allowed by the
954 	 *      constraints
955 	 * <li> Pass each <code>Config</code> through the constraints
956 	 *      and filter out those that conflict with requirements
957 	 * <li> Reorder the remaining <code>Config</code> list by
958 	 *      preferences
959 	 * </ul> <p>
960 	 *
961 	 * @param clientSubject the client subject that contains
962 	 *        client principals and TGTs, can not be
963 	 *        <code>null</code>.
964 	 * @param constraints the security constraint set, can not be
965 	 *        <code>null</code>
966 	 */
967 	RequestHandleImpl(Subject clientSubject,
968 			  InvocationConstraints constraints)
969 	{
970 	    /* unsupportable constraint has to be checked before any
971 	       other security sensitive things are checked, so a
972 	       detailed exception regarding to it can always be thrown
973 	       regardless whether the caller has the getSubject
974 	       permission */
975 	    for (Iterator iter = constraints.requirements().iterator();
976 		 iter.hasNext(); )
977 	    {
978 		InvocationConstraint c = (InvocationConstraint) iter.next();
979 		if (!KerberosUtil.isSupportableConstraint(c)) {
980 		    errorCode = UNSUPPORTABLE_CONSTRAINT_REQUIRED;
981 		    detailedExceptionMsg = "A constraint unsupportable by " +
982 			"this endpoint has been required: " + c;
983 		    return;
984 		}
985 	    }
986 
987 	    /* All Kerberos principals allowed by the constraints.  If
988 	       the resulting set is empty, it means no client min/max
989 	       principal constraints found in the constraints, instead
990 	       of no principals allowed. */
991 	    clientPrincipals = new HashSet();
992 	    for (Iterator iter = constraints.requirements().iterator();
993 		 iter.hasNext(); )
994 	    {
995 		if (!KerberosUtil.collectCpCandidates(
996 		    (InvocationConstraint) iter.next(),
997 		    clientPrincipals))
998 		{
999 		    errorCode = UNSUPPORTABLE_CONSTRAINT_REQUIRED;
1000 		    detailedExceptionMsg = "Client principal constraint " +
1001 			"related conflicts found in the given set of " +
1002 			"constraints: " + constraints;
1003 		    return;
1004 		}
1005 	    }
1006 
1007 	    if (clientSubject == null) {
1008 		errorCode = NULL_SUBJECT;
1009 		detailedExceptionMsg = "JAAS login has not been done " +
1010 		    "properly, the subject associated with the current " +
1011 		    "AccessControlContext is null.";
1012 		return;
1013 	    }
1014 
1015 	    this.clientSubject = clientSubject;
1016 	    this.constraints = constraints;
1017 
1018 	    subjectReadOnly = clientSubject.isReadOnly();
1019 	    subjectClientPrincipals = getClientPrincipals(clientSubject);
1020 
1021 	    if (subjectClientPrincipals.isEmpty()) {
1022 		errorCode = NO_CLIENT_PRINCIPAL;
1023 		detailedExceptionMsg = "JAAS login has not been done " +
1024 		    "properly, the subject associated with the current " +
1025 		    "AccessControlContext contains no KerberosPrincipal.";
1026 		return;
1027 	    }
1028 
1029 	    if (clientPrincipals.size() > 0) {
1030 		clientPrincipals.retainAll(subjectClientPrincipals);
1031 	    } else {
1032 		clientPrincipals = subjectClientPrincipals;
1033 	    }
1034 
1035 	    boolean canDeleg = false;
1036 	    if (KerberosUtil.containsConstraint(
1037 		constraints.requirements(), Delegation.YES) ||
1038 		KerberosUtil.containsConstraint(
1039 		    constraints.preferences(), Delegation.YES))
1040 	    {
1041 		canDeleg = true;
1042 	    }
1043 
1044 	    // enumerate all possible configs and filter them by constraints
1045 	    ArrayList configArr = new ArrayList();
1046 	  outer:
1047 	    for (ConfigIter citer = new ConfigIter(
1048 		     clientPrincipals, serverPrincipal, canDeleg);
1049 		 citer.hasNext(); )
1050 	    {
1051 		Config config = citer.next();
1052 		for (Iterator jter = constraints.requirements().iterator();
1053 		     jter.hasNext(); )
1054 		{
1055 		    InvocationConstraint c =
1056 			(InvocationConstraint) jter.next();
1057 		    if (!KerberosUtil.isSatisfiable(config, c))
1058 			continue outer;
1059 		}
1060 		configArr.add(config);
1061 	    }
1062 
1063 	    if (configArr.isEmpty()) {
1064 		errorCode = UNSATISFIABLE_CONSTRAINT_REQUIRED;
1065 		detailedExceptionMsg = "Constraints unsatisfiable by this " +
1066 		    "endpoint with the current subject have been required: " +
1067 		    constraints + ", while the KerberosPrincipal set of " +
1068 		    "the subject is: " + subjectClientPrincipals;
1069 		return;
1070 	    }
1071 
1072 	    configs = (Config[]) configArr.toArray(
1073 		new Config[configArr.size()]);
1074 
1075 	    // reorder configs by the num of preferences a config can satisfy
1076 	    for (int i = 0, l = configs.length; i < l; i++) {
1077 		for (Iterator iter = constraints.preferences().iterator();
1078 		     iter.hasNext(); )
1079 	        {
1080 		    InvocationConstraint c =
1081 			(InvocationConstraint) iter.next();
1082 		    if (KerberosUtil.isSatisfiable(configs[i], c))
1083 			configs[i].prefCount++;
1084 		}
1085 	    }
1086 	    Arrays.sort(configs, new Comparator() {
1087 		    public int compare(Object o1, Object o2) {
1088 			Config config1 = (Config) o1;
1089 			Config config2 = (Config) o2;
1090 			// sort to descending order by prefCount
1091 			return config2.prefCount - config1.prefCount;
1092 		    }
1093 		});
1094 
1095 	    if (KerberosUtil.containsConstraint(
1096 		constraints.requirements(), Integrity.YES))
1097 	    {
1098 		unfulfilledConstraints =
1099 		    KerberosUtil.INTEGRITY_REQUIRED_CONSTRAINTS;
1100 	    } else if (KerberosUtil.containsConstraint(
1101 		constraints.preferences(), Integrity.YES))
1102 	    {
1103 		unfulfilledConstraints =
1104 		    KerberosUtil.INTEGRITY_PREFERRED_CONSTRAINTS;
1105 	    } else {
1106 		unfulfilledConstraints = InvocationConstraints.EMPTY;
1107 	    }
1108 
1109 	    connectionAbsoluteTime = Math.min(
1110 		computeConnectionTimeLimit(constraints.requirements()),
1111 		computeConnectionTimeLimit(constraints.preferences()));
1112 	}

1157 	/**
1158 	 * Get a list of satisfiable configurations.  Elements of the
1159 	 * list is from the configs array of this handle, with those
1160 	 * who failed their corresponding AuthenticationPermission
1161 	 * and private credential (TGT) checks filtered out.  The
1162 	 * returned list is ordered by decreasing preference.
1163 	 *
1164 	 * @return a list of satisfiable configurations in decreasing
1165 	 *         preference order.
1166 	 * @throws UnsupportedConstraintException if the caller has
1167 	 *         required unsupported constraints, or there are
1168 	 *         conflicts or unsatisfiable constraint in the
1169 	 *         requirements, or the JAAS login has not been done
1170 	 *         (Subject.getSubject(AccessController.getContext())
1171 	 *         returns <code>null</code>), or no appropriate
1172 	 *         Kerberos principal and corresponding TGT allowed by
1173 	 *         the requirements can be found in the current
1174 	 *         subject.  If the caller has not been granted
1175 	 *         <code>javax.security.auth.AuthPermission("getSubject")
1176 	 *         </code> and cause is not unsupported constraints
1177 	 *         being required, the exception message will be
1178 	 *         generic and enumerate all these possible causes.
1179 	 *         Otherwise, the message will spell out the reason
1180 	 *         caused the exception.
1181 	 * @throws SecurityException if there is a security manager
1182 	 *         and the caller has
1183 	 *         <code>javax.security.auth.AuthPermission("getSubject")
1184 	 *         </code> but not any
1185 	 *         <code>AuthenticationPermission</code> whose local
1186 	 *         principal is a member of the client principal
1187 	 *         candidate set.  The action of the
1188 	 *         <code>AuthenticationPermission</code> is either
1189 	 *         <code>connect</code> or <code>delegate</code>,
1190 	 *         determined by the requirements of the constraints.
1191 	 */
1192         List getConfigs() throws UnsupportedConstraintException {
1193 
1194 	    if (errorCode != NO_ERROR) {
1195 		throw new UnsupportedConstraintException(
1196 		    detailedExceptionMsg);
1197 	    }
1198 
1199 	    KerberosTicket[] tickets =
1200 		(KerberosTicket[]) AccessController.doPrivileged(
1201 		    new PrivilegedAction() {
1202 			    public Object run() {
1203 				return getTickets();
1204 			    }
1205 			});
1206 	    
1207 	    GSSCredential[] gssCredentials;
1208 	    if (tickets.length == 0){
1209 		gssCredentials = AccessController.doPrivileged(
1210 		    new PrivilegedAction<GSSCredential[]>(){
1211 			public GSSCredential[] run() {
1212 			    return getCredential();
1213 			}  
1214 		    }
1215 		);
1216 	    } else {
1217 		gssCredentials = new GSSCredential[0];
1218 	    }
1219 
1220 	    ArrayList configList = new ArrayList(configs.length);
1221 
1222 	    /* This illustrates how a detailed failure msg is derived:
1223 	     *
1224 	     *               |<-- stepsFromSuccess -->|
1225 	     *
1226 	     *                               TGT.forwardable
1227 	     *                      TGT.yes
1228 	     *            perm.yes           TGT.unforwardable

1827 				    new Object[] { c.sock }, e);
1828 			    }
1829 			}
1830 		    }
1831 		}
1832 	    }
1833 
1834 	    for (Iterator i = configList.iterator(); i.hasNext(); ) {
1835 		Config config = (Config) i.next();
1836 		for (Iterator j = idle.iterator(); j.hasNext(); ) {
1837 		    ConnectionImpl c = checkConnection(j.next());
1838 		    if (c.switchTo(config)) {
1839 			if (logger.isLoggable(Level.FINE)) {
1840 			    logger.log(Level.FINE, "found an idle " +
1841 				       "connection for reusing:\n{0}\n{1}",
1842 				       new Object[] {c, config});
1843 			}
1844 			if (!checkedResolvePermission) {
1845 			    try {
1846 				checkResolvePermission();
1847 			    } catch (SecurityException e) {
1848 				if (logger.isLoggable(Levels.FAILED)) {
1849 				    LogUtil.logThrow(logger, Levels.FAILED,
1850 					ConnectionEndpointImpl.class, "connect",
1851 					"exception resolving host {0}",
1852 					new Object[] { serverHost }, e);
1853 				}
1854 				throw e;
1855 			    }
1856 			    checkedResolvePermission = true;
1857 			}
1858 			try {
1859 			    c.checkConnectPermission();
1860 			    return c;
1861 			} catch (SecurityException e) {
1862 			    if (logger.isLoggable(Levels.HANDLED)) {
1863 				LogUtil.logThrow(logger, Levels.HANDLED,
1864 				    ConnectionEndpointImpl.class, "connect",
1865 				    "access to reuse connection {0} denied",
1866 				    new Object[] { c.sock }, e);
1867 			    }
1868 			}
1869 		    }
1870 		}
1871 	    }
1872 
1873 	    return null;
1874 	}

1877 	    SecurityManager sm = System.getSecurityManager();
1878 	    if (sm != null) {
1879 		sm.checkConnect(serverHost, -1);
1880 	    }
1881 	}

2069 	/**
2070 	 * Exchange handshake messages with server to establish the
2071 	 * session or context of the connection.
2072 	 */
2073 	private void establishContext() throws IOException, GSSException {
2074 
2075 	    synchronized (classLock) {
2076 		if (gssManager == null) {
2077 		    gssManager = GSSManager.getInstance();
2078 		}
2079 	    }
2080 
2081 	    GSSName clientName = gssManager.createName(
2082 		clientPrincipal.getName(), KerberosUtil.krb5NameType);
2083 
2084 	    GSSCredential clientCred = gssManager.createCredential(
2085 		clientName, GSSCredential.INDEFINITE_LIFETIME,
2086 		KerberosUtil.krb5MechOid, GSSCredential.INITIATE_ONLY);
2087 
2088 	    GSSName serverName = gssManager.createName(
2089 		serverPrincipal.getName(), KerberosUtil.krb5NameType);
2090 
2091 	    for (int i = maxGssContextRetries; i > 0; i--) {
2092 		gssContext = gssManager.createContext(
2093 		    serverName, KerberosUtil.krb5MechOid, clientCred,
2094 		    GSSContext.DEFAULT_LIFETIME);
2095 		/*
2096 		 * Set the desired optional features on the
2097 		 * context. The client chooses these options.
2098 		 */
2099 		gssContext.requestMutualAuth(true); // Mutual authentication
2100 		gssContext.requestInteg(true);      // Will use integrity later
2101 		gssContext.requestConf(true);       // always enable encryption
2102 		gssContext.requestCredDeleg(doDelegation);
2103 		gssContext.requestReplayDet(true);
2104 		gssContext.requestSequenceDet(true);
2105 		
2106 		// Do the context establishment loop
2107 		byte[] token = new byte[0];
2108 		try {
2109 		    while (true) {
2110 			// token is ignored on the first call
2111 			token = gssContext.initSecContext(
2112 			    token, 0, token.length);
2113 			/*
2114 			 * Send a token to the server if one was generated by
2115 			 * initSecContext
2116 			 */
2117 			if (token != null) {
2118 			    dos.writeInt(token.length);
2119 			    dos.write(token);
2120 			    dos.flush();
2121 			}
2122 			
2123 			if (gssContext.isEstablished()) {
2124 			    break;
2125 			} else {
2126 			    token = new byte[dis.readInt()];
2127 			    dis.readFully(token);
2128 			}
2129 		    }
2130 		    break; // done gssContext establishment
2131 		} catch (GSSException ge) {
2132 		    if ((ge.getMessage().contains("34") ||
2133 			 ge.getMajor() == GSSException.DUPLICATE_TOKEN ||
2134 			 ge.getMinor() == KRB_AP_ERR_REPEAT ||
2135 			 ge.getMessage().contains("Request is a replay"))
2136 			&& i != 1) // will throw ge if loop has terminated
2137 		    {
2138 			/* this could caused by our authenticator
2139                          * being time stamped to the same micro-second
2140                          * as those of other concurrent connections,
2141                          * retry for better luck */
2142 			continue;
2143 		    }
2144 		    throw ge;
2145 		}
2146 	    }
2147 
2148 	    if (!gssContext.getIntegState() ||
2149 		(doEncryption && !gssContext.getConfState()) ||
2150 		(doDelegation && !gssContext.getCredDelegState()) ||
2151 		!gssContext.getTargName().toString().
2152 		equals(serverPrincipal.getName()))
2153 	    {
2154 		throw new IOException("Failed to establish gss context " +
2155 				      "for connection");
2156 	    }
2157 	    logger.log(Level.FINE, "GSSContext established for {0}", this);
2158 	}
