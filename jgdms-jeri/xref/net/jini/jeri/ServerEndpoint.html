21  import java.io.IOException;
22  import javax.security.auth.Subject;

24  /**
25   * Represents one or more communication endpoints on the current
26   * (local) host to listen for and receive requests on and a template
27   * for producing an <code>Endpoint</code> instance to send requests to
28   * those communication endpoints.
29   *
30   * <p>A <code>ServerEndpoint</code> instance contains the information
31   * necessary to listen for requests on the communication endpoints and
32   * to produce <code>Endpoint</code> instances.  For example, a
33   * TCP-based <code>ServerEndpoint</code> implementation typically
34   * contains the TCP port to listen on and the host address to put in
35   * the <code>Endpoint</code> instances it produces.  An implementation
36   * that supports authentication typically contains the {@link Subject}
37   * (if any) to use for server authentication.
38   *
39   * <p>The {@link #enumerateListenEndpoints enumerateListenEndpoints}
40   * method can be invoked to
41   *
42   * <ul>
43   *
44   * <li>discover each of the discrete communication endpoints
45   * represented by this <code>ServerEndpoint</code>, which are
46   * individually represented as {@link ListenEndpoint ListenEndpoint}
47   * instances,
48   *
49   * <li>ensure an active <i>listen operation</i> (see {@link
50   * ListenEndpoint#listen ListenEndpoint.listen}) on each discrete
51   * communication endpoint by choosing, for each endpoint, either to
52   * start a new listen operation or to use an existing one, and
53   *
54   * <li>obtain an <code>Endpoint</code> instance that corresponds to
55   * the chosen listen operations
56   *
57   * </ul>
58   *
59   * The obtained <code>Endpoint</code> instance can then be used to
60   * send requests to the communication endpoints being listened on as a
61   * result of that {@link #enumerateListenEndpoints
62   * enumerateListenEndpoints} invocation.
63   *
64   * <p>Typically, a <code>ServerEndpoint</code> is just used to specify
65   * the transport layer implementation to use when exporting a remote
66   * object; for example, some constructors of {@link BasicJeriExporter
67   * BasicJeriExporter} have a <code>ServerEndpoint</code> parameter to
68   * specify the transport layer implementation.  The exporter
69   * implementation is then responsible for using the supplied
70   * <code>ServerEndpoint</code> to manage listen operations as
71   * necessary and to obtain an <code>Endpoint</code> instance for
72   * putting in the client-side proxy.
73   *
74   * <p>All aspects of the underlying communication mechanism that are
75   * not specified here are defined by the particular implementation of
76   * this interface.
77   *
78   * @author Sun Microsystems, Inc.
79   * @since 2.0
80   **/
81  public interface ServerEndpoint extends ServerCapabilities {
82  
83      /**
84       * Enumerates the communication endpoints represented by this
85       * <code>ServerEndpoint</code> by passing the
86       * <code>ListenEndpoint</code> for each of them to
87       * <code>listenContext</code>, which will ensure an active listen
88       * operation on each endpoint, and returns an
89       * <code>Endpoint</code> instance corresponding to the listen
90       * operations chosen by <code>listenContext</code>.
91       *
92       * <p>This method uses <code>listenContext</code> to cooperate
93       * with the caller to ensure the appropriate listen operations.
94       * In this cooperation,
95       *
96       * <ul>
97       *
98       * <li>the <code>ServerEndpoint</code> is responsible for
99       * declaring to the caller the discrete communication endpoints
100      * represented by this <code>ServerEndpoint</code> as
101      * <code>ListenEndpoint</code> instances passed to
102      * <code>listenContext</code>'s {@link
103      * ListenContext#addListenEndpoint addListenEndpoint} method, and
104      *
105      * <li>the caller is responsible for declaring the active listen
106      * operations to be used for this
107      * <code>ServerEndpoint.enumerateListenEndpoints</code> invocation
108      * corresponding to each discrete communication endpoint as {@link
109      * ListenCookie ListenCookie} instances returned from
110      * <code>listenContext</code>'s <code>addListenEndpoint</code>
111      * method.
112      *
113      * </ul>
114      *
115      * For each <code>ListenEndpoint</code>, the caller (through
116      * <code>listenContext</code>) may choose to start a new listen
117      * operation, or it may choose to reuse a previously started
118      * listen operation that it has a <code>ListenCookie</code> for.
119      *
120      * <p>This method sequentially invokes
121      * <code>addListenEndpoint</code> on <code>listenContext</code>
122      * once for each discrete communication endpoint represented by
123      * this <code>ServerEndpoint</code>, passing the
124      * <code>ListenEndpoint</code> representing that communication
125      * endpoint.  If any of the invocations of
126      * <code>ListenContext.addListenEndpoint</code> throws an
127      * exception, this method throws that exception.  Otherwise, this
128      * method returns an <code>Endpoint</code> instance that sends
129      * requests to be received by the listen operations chosen by
130      * <code>listenContext</code>.
131      *
132      * @param listenContext the <code>ListenContext</code> to pass
133      * this <code>ServerEndpoint</code>'s <code>ListenEndpoint</code>
134      * instances to
135      *
136      * @return the <code>Endpoint</code> instance for sending requests
137      * to this <code>ServerEndpoint</code>'s communication endpoints
138      * being listened on
139      *
140      * @throws IOException if an I/O exception occurs while attempting
141      * to listen for requests on the communication endpoints
142      * represented by this <code>ServerEndpoint</code>.  This could
143      * occur, for example, if an I/O resource associated with one of
144      * the communication endpoints is already in exclusive use, or if
145      * there are insufficient I/O resources for the operation.
146      *
147      * @throws SecurityException if the current security context does
148      * not have the permissions necessary to listen for requests on
149      * one of the communication endpoints represented by this
150      * <code>ServerEndpoint</code>
151      *
152      * @throws IllegalArgumentException if an invocation of the
153      * <code>addListenEndpoint</code> method on the supplied
154      * <code>ListenContext</code> returns a <code>ListenCookie</code>
155      * that does not correspond to the <code>ListenEndpoint</code>
156      * that was passed to it
157      *
158      * @throws NullPointerException if <code>listenContext</code> is
159      * <code>null</code>
160      **/
161     Endpoint enumerateListenEndpoints(ListenContext listenContext)
162 	throws IOException;
163 
164     /**
165      * A callback object for passing to {@link
166      * ServerEndpoint#enumerateListenEndpoints
167      * ServerEndpoint.enumerateListenEndpoints} to receive the
168      * enumerated {@link ServerEndpoint.ListenEndpoint ListenEndpoint}
169      * instances and to choose an active listen operation for each of
170      * them on behalf of the caller of
171      * <code>enumerateListenEndpoints</code>.
172      **/
173     interface ListenContext {
174 
175 	/**
176 	 * Adds <code>listenEndpoint</code> to this
177 	 * <code>ListenContext</code>'s collection of
178 	 * <code>ListenEndpoint</code> instances for the
179 	 * <code>ServerEndpoint</code> it was passed to, starts a
180 	 * listen operation on <code>listenEndpoint</code> if
181 	 * necessary, and returns the <code>ListenCookie</code> for an
182 	 * active listen operation on <code>listenEndpoint</code>.
183 	 *
184 	 * <p>The returned <code>ListenCookie</code> must have been
185 	 * obtained from a <code>ListenHandle</code> returned from
186 	 * some invocation of {@link
187 	 * ServerEndpoint.ListenEndpoint#listen ListenEndpoint.listen}
188 	 * on a <code>ListenEndpoint</code> equivalent to
189 	 * <code>listenEndpoint</code> by {@link Object#equals
190 	 * Object.equals}.
191 	 *
192 	 * <p>This method may start a new listen operation on
193 	 * <code>listenEndpoint</code> by invoking its
194 	 * <code>listen</code> method and returning the
195 	 * <code>ListenCookie</code> from the resulting
196 	 * <code>ListenHandle</code>, or it may return a
197 	 * <code>ListenCookie</code> for a listen operation previously
198 	 * started (but still active) on an equivalent
199 	 * <code>ListenEndpoint</code>.  If this method does invoke
200 	 * <code>listen</code> on <code>listenEndpoint</code> and it
201 	 * throws an exception, then this method throws that
202 	 * exception.
203 	 *
204 	 * <p>The implementation of this method may invoke {@link
205 	 * ServerEndpoint.ListenEndpoint#checkPermissions
206 	 * checkPermissions} on <code>listenEndpoint</code> to verify
207 	 * that a party that it is operating on behalf of has all of
208 	 * the security permissions necessary to listen for requests
209 	 * on <code>listenEndpoint</code>.
210 	 *
211 	 * @param listenEndpoint the <code>ListenEndpoint</code> to
212 	 * add to this <code>ListenContext</code> and to return a
213 	 * <code>ListenCookie</code> for
214 	 *
215 	 * @return a <code>ListenCookie</code> that represents an
216 	 * active listen operation on <code>listenEndpoint</code>
217 	 *
218 	 * @throws IOException if an invocation of <code>listen</code>
219 	 * on <code>listenEndpoint</code> throws an
220 	 * <code>IOException</code>
221 	 *
222 	 * @throws SecurityException if an invocation of
223 	 * <code>checkPermissions</code> or <code>listen</code> on
224 	 * <code>listenEndpoint</code> throws a
225 	 * <code>SecurityException</code>
226 	 *
227 	 * @throws IllegalStateException if this method is invoked
228 	 * unexpectedly, such as before being passed to
229 	 * <code>ServerEndpoint.enumerateListenEndpoints</code> or
230 	 * after the invocation of
231 	 * <code>ServerEndpoint.enumerateListenEndpoints</code> that
232 	 * it was created for has returned
233 	 *
234 	 * @throws NullPointerException if <code>listenEndpoint</code>
235 	 * is <code>null</code>
236 	 **/
237 	ListenCookie addListenEndpoint(ListenEndpoint listenEndpoint)
238 	    throws IOException;
239     }

241     /**
242      * Represents a communication endpoint on the current (local) host
243      * to listen for and receive requests on.
244      *
245      * <p>A <code>ListenEndpoint</code> instance contains the
246      * information necessary to listen for requests on the
247      * communication endpoint.  For example, a TCP-based
248      * <code>ListenEndpoint</code> implementation typically contains
249      * the TCP port to listen on.  An implementation that supports
250      * authentication typically contains the {@link Subject} (if any)
251      * to use for server authentication.
252      *
253      * <p>The {@link #listen listen} method can be used to start a
254      * <i>listen operation</i> on the communication endpoint that it
255      * represents, during which requests received on the endpoint will
256      * be dispatched to a supplied {@link RequestDispatcher}.
257      *
258      * <p><code>ListenEndpoint</code> instances make up the discrete
259      * communication endpoints represented by a
260      * <code>ServerEndpoint</code>, and they are passed to a {@link
261      * ServerEndpoint.ListenContext ListenContext} as part of a {@link
262      * ServerEndpoint#enumerateListenEndpoints
263      * ServerEndpoint.enumerateListenEndpoints} invocation.
264      *
265      * <p>An instance of this interface should implement {@link
266      * Object#equals Object.equals} to return <code>true</code> if and
267      * only if the argument is equivalent to the instance in trust,
268      * content, and function.  The <code>equals</code> method should
269      * not invoke comparison methods (such as <code>equals</code>) on
270      * any pluggable component without first verifying that the
271      * component's implementation is at least as trusted as the
272      * implementation of the corresponding component in the
273      * <code>equals</code> argument (such as by verifying that the
274      * corresponding component objects have the same actual class).
275      * If any such verification fails, the <code>equals</code> method
276      * should return <code>false</code> without invoking a comparison
277      * method on the component.  Furthermore, these guidelines should
278      * be recursively obeyed by the comparison methods of each such
279      * component for its subcomponents.  To avoid opening a security
280      * hole, implementations should only compare object identity
281      * (<code>==</code>) of <code>Subject</code> instances, rather
282      * than comparing their contents.
283      *
284      * <p>The equivalence relation of a <code>ListenEndpoint</code>
285      * should not be a function of any state in an associated
286      * <code>ServerEndpoint</code> that only applies to the
287      * <code>ServerEndpoint</code>'s template for producing
288      * <code>Endpoint</code> instances.
289      **/
290     interface ListenEndpoint {
291 
292 	/**
293 	 * Verifies that the current security context has all of the
294 	 * security permissions necessary to listen for requests on
295 	 * this <code>ListenEndpoint</code>.
296 	 *
297 	 * <p>This method should be used when an invocation of
298 	 * <code>ListenEndpoint.listen</code> is used to receive
299 	 * requests for a variety of interested parties, each with
300 	 * potentially different security permissions possibly more
301 	 * limited than those granted to the code managing the shared
302 	 * endpoint, so that the managing code can enforce proper
303 	 * access control for each party.
304 	 *
305 	 * @throws SecurityException if the current security context
306 	 * does not have the permissions necessary to listen for
307 	 * requests on this <code>ListenEndpoint</code>
308 	 *
309 	 * @see InboundRequest#checkPermissions
310 	 **/
311 	void checkPermissions();
312 
313 	/**
314 	 * Listens for requests received on the communication endpoint
315 	 * represented by this <code>ListenEndpoint</code>,
316 	 * dispatching them to the supplied
317 	 * <code>RequestDispatcher</code> in the form of {@link
318 	 * InboundRequest} instances.
319 	 *
320 	 * <p>This method starts a continuing <i>listen operation</i>
321 	 * and then immediately returns a <code>ListenHandle</code>
322 	 * that represents the listen operation that was started.  For
323 	 * the duration of the listen operation, all requests received
324 	 * on the communication endpoint will be dispatched to the
325 	 * supplied <code>RequestDispatcher</code> as
326 	 * <code>InboundRequest</code> instances.  The returned
327 	 * <code>ListenHandle</code> can be used to stop the listen
328 	 * operation and to obtain a {@link
329 	 * ServerEndpoint.ListenCookie ListenCookie} to identify the
330 	 * listen operation as the return value of the {@link
331 	 * ServerEndpoint.ListenContext#addListenEndpoint
332 	 * ListenContext.addListenEndpoint} method.
333 	 *
334 	 * <p>Typically, this method is invoked by a
335 	 * <code>ListenContext</code> implementation when its
336 	 * <code>addListenEndpoint</code> method is called as part of
337 	 * the execution of a
338 	 * <code>ServerEndpoint.enumerateListenEndpoints</code>
339 	 * invocation.  The <code>Endpoint</code> instance that can be
340 	 * used to send requests to the communication endpoints
341 	 * represented by the <code>ServerEndpoint</code> (including
342 	 * this <code>ListenEndpoint</code>) is produced by the
343 	 * <code>ServerEndpoint</code> implementation given, in part,
344 	 * the <code>ListenCookie</code> obtained from the
345 	 * <code>ListenHandle</code> returned by this method.
346 	 *
347 	 * <p>This method verifies that the current security context
348 	 * has all of the security permissions necessary to listen for
349 	 * requests on this <code>ListenEndpoint</code>, as
350 	 * appropriate to the implementation of this interface (see
351 	 * {@link #checkPermissions}).  Note that in addition to this
352 	 * security check, the implementation of this interface may
353 	 * also perform a further security check per request received
354 	 * (when the origin of a given request is known), to verify
355 	 * that the same security context also has all of the
356 	 * permissions necessary to receive requests from that
357 	 * particular origin (see {@link
358 	 * InboundRequest#checkPermissions}).  This interface does not
359 	 * provide an API for determining when such a security check
360 	 * has failed; the behavior will be as if the associated
361 	 * request never occurred.
362 	 *
363 	 * <p>Implementations of this method should provide robust
364 	 * behavior (such as continuing to listen for requests) in the
365 	 * event that the supplied <code>RequestDispatcher</code>'s
366 	 * <code>dispatch</code> method throws an unchecked exception.
367 	 *
368 	 * <p>Requests may be dispatched in separate threads as
369 	 * necessary for requests received on this
370 	 * <code>ListenEndpoint</code> while others are still being
371 	 * processed by earlier <code>dispatch</code> invocations.
372 	 * The implementation of this interface must assume that there
373 	 * may be arbitrary execution dependencies between the
374 	 * processing of such concurrently received requests, and thus
375 	 * it must not attempt any particular serialization of their
376 	 * processing; therefore, a request received must be either
377 	 * dispatched or rejected within a reasonable period of time,
378 	 * rather than be queued indefinitely waiting for the return
379 	 * of earlier <code>dispatch</code> invocations.
380 	 *
381 	 * <p>Implementations of this method should generally dispatch
382 	 * a request in a daemon thread with a security context at
383 	 * least as restrictive as the one in which this method was
384 	 * invoked, and without holding visible synchronization locks.
385 	 *
386 	 * @param requestDispatcher the <code>RequestDispatcher</code>
387 	 * to use to dispatch incoming requests received on this
388 	 * communication endpoint
389 	 *
390 	 * @return a <code>ListenHandle</code> that represents the
391 	 * listen operation that was started
392 	 *
393 	 * @throws IOException if an I/O exception occurs while
394 	 * attempting to listen for requests on this
395 	 * <code>ListenEndpoint</code>.  This could occur, for
396 	 * example, if an I/O resource associated with this
397 	 * communication endpoint is already in exclusive use, or if
398 	 * there are insufficient I/O resources for the operation.
399 	 *
400 	 * @throws SecurityException if the current security context
401 	 * does not have the permissions necessary to listen for
402 	 * requests on this <code>ListenEndpoint</code>
403 	 *
404 	 * @throws NullPointerException if
405 	 * <code>requestDispatcher</code> is <code>null</code>
406 	 **/
407 	ListenHandle listen(RequestDispatcher requestDispatcher)
408 	    throws IOException;
409     }
410 
411     /**
412      * Represents a listen operation that has been started on a {@link
413      * ServerEndpoint.ListenEndpoint ListenEndpoint}.
414      *
415      * <p>A <code>ListenHandle</code> is returned from a successful
416      * <code>ListenEndpoint.listen</code> invocation to represent the
417      * listen operation started by that invocation.  This object can
418      * be used to stop the listen operation and to obtain a
419      * <code>ListenCookie</code> to identify the listen operation as
420      * the return value of the {@link
421      * ServerEndpoint.ListenContext#addListenEndpoint
422      * ListenContext.addListenEndpoint} method.
423      **/
424     interface ListenHandle {
425 
426 	/**
427 	 * Stops listening for requests on the associated
428 	 * <code>ListenEndpoint</code>.
429 	 *
430 	 * <p>After this method has returned, no more requests will be
431 	 * dispatched to the <code>RequestDispatcher</code> for the
432 	 * listen operation represented by this
433 	 * <code>ListenHandle</code>, and the listen operation is no
434 	 * longer considered active.  This method frees any resources
435 	 * associated with the listen operation.
436 	 *
437 	 * <p>Invoking this method terminates any requests that have
438 	 * been received because of the listen operation and
439 	 * dispatched to the associated <code>RequestDispatcher</code>
440 	 * but have not yet had their response output stream closed
441 	 * (see {@link InboundRequest#abort InboundRequest.abort});
442 	 * subsequent I/O operations on such requests will fail with
443 	 * an <code>IOException</code>, except some operations that
444 	 * may succeed because they only affect data in local I/O
445 	 * buffers.
446 	 **/
447 	void close();
448 
449 	/**
450 	 * Returns a <code>ListenCookie</code> to identify the listen
451 	 * operation as the return value of the {@link
452 	 * ServerEndpoint.ListenContext#addListenEndpoint
453 	 * ListenContext.addListenEndpoint} method.
454 	 *
455 	 * @return a <code>ListenCookie</code> to identify the listen
456 	 * operation
457 	 **/
458 	ListenCookie getCookie();
459     }
