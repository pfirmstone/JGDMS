64  /**
65   * An implementation of the {@link Endpoint} abstraction that uses TCP
66   * sockets (instances of {@link Socket}) for the underlying
67   * communication mechanism.
68   *
69   * <p><code>TcpEndpoint</code> instances contain a host name and a TCP
70   * port number, as well as an optional {@link SocketFactory} for
71   * customizing the type of <code>Socket</code> to use.  The host name
72   * and port number are used as the remote address to connect to when
73   * making socket connections.
74   *
75   * <p><code>TcpEndpoint</code> uses the <a
76   * href="../connection/doc-files/mux.html">Jini extensible remote
77   * invocation (Jini ERI) multiplexing protocol</a> to map outgoing
78   * requests to socket connections.
79   *
80   * <p>A <code>SocketFactory</code> used with a
81   * <code>TcpEndpoint</code> should be serializable and must implement
82   * {@link Object#equals Object.equals} to obey the guidelines that are
83   * specified for <code>equals</code> methods of {@link Endpoint}
84   * instances.
85   *
86   * 
87   * @see TcpServerEndpoint
88   * @since 2.0
89   **/

192     /**
193      * Returns canonical instance equivalent to given instance.
194      **/
195     private static TcpEndpoint intern(TcpEndpoint endpoint) {
196 	synchronized (internTable) {
197 	    Reference<TcpEndpoint> ref = (WeakReference) internTable.get(endpoint);
198 	    if (ref != null) {
199 		TcpEndpoint canonical = ref.get();
200 		if (canonical != null) {
201 		    return canonical;
202 		}
203 	    }
204 	    endpoint.connectionManager =
205 		new ConnectionManager(
206                         new ConnectionEndpointImpl(
207                                 endpoint.getHost(),
208                                 endpoint.getPort(),
209                                 endpoint.getSocketFactory()
210                         )
211                 );
212 	    internTable.put(endpoint, new WeakReference<TcpEndpoint>(endpoint));
213 	    return endpoint;
214 	}
215     }

229     /**
230      *  Invariant checks.
231      */
232     private static int check(String host, int port){
233 	if (host == null) {
234 	    throw new NullPointerException("null hostname");
235 	}
236 	if (port < 1 || port > 0xFFFF) {
237 	    throw new IllegalArgumentException(
238 	        "port number out of range: " + port);
239 	}
240 	return port;
241     }

298     /**
299      * {@inheritDoc}
300      *
301      * <p>The returned <code>OutboundRequestIterator</code>'s {@link
302      * OutboundRequestIterator#next next} method behaves as follows:
303      *
304      * <blockquote>
305      *
306      * Initiates an attempt to communicate the request to this remote
307      * endpoint.
308      *
309      * <p>When the implementation of this method needs to create a new
310      * <code>Socket</code>, it will do so by invoking one of the
311      * <code>createSocket</code> methods on the
312      * <code>SocketFactory</code> of this <code>TcpEndpoint</code>
313      * (which produced this iterator) if non-<code>null</code>, or it
314      * will create a <code>Socket</code> directly otherwise.
315      *
316      * <p>When the implementation needs to connect a
317      * <code>Socket</code>, if the host name to connect to (this
318      * <code>TcpEndpoint</code>'s host name) resolves to multiple
319      * addresses (according to {@link InetAddress#getAllByName
320      * InetAddress.getAllByName}), it attempts to connect to the first
321      * resolved address; if that attempt fails with an
322      * <code>IOException</code> or a <code>SecurityException</code>,
323      * it then attempts to connect to the next address; and this
324      * iteration continues as long as there is another resolved
325      * address and the attempt to connect to the previous address
326      * fails with an <code>IOException</code> or a
327      * <code>SecurityException</code>.  If the host name resolves to
328      * just one address, the implementation makes one attempt to
329      * connect to that address.  If the host name does not resolve to
330      * any addresses (<code>InetAddress.getAllByName</code> would
331      * throw an <code>UnknownHostException</code>), the implementation
332      * still makes an attempt to connect the <code>Socket</code> to
333      * that host name, which could result in an
334      * <code>UnknownHostException</code>.  If the final connection
335      * attempt fails with an <code>IOException</code> or a
336      * <code>SecurityException</code>, then if any connection attempt
337      * failed with an <code>IOException</code>, this method throws an
338      * <code>IOException</code>, and otherwise (if all connection
339      * attempts failed with a <code>SecurityException</code>), this
340      * method throws a <code>SecurityException</code>.
341      *
342      * <p>If there is a security manager:
343      *
344      * <ul>
345      *
346      * <li>If a new connection is to be created, the security
347      * manager's {@link SecurityManager#checkConnect(String,int)
348      * checkConnect} method is invoked with this
349      * <code>TcpEndpoint</code>'s host and <code>-1</code> for the
350      * port; if this results in a <code>SecurityException</code>, this
351      * method throws that exception.  <code>checkConnect</code> is
352      * also invoked for each connection attempt, with the remote IP
353      * address (or the host name, if it could not be resolved) and
354      * port to connect to; this could result in a
355      * <code>SecurityException</code> for that attempt.  (Note that
356      * the implementation may carry out these security checks
357      * indirectly, such as through invocations of
358      * <code>InetAddress.getAllByName</code> or <code>Socket</code>'s
359      * constructors or <code>connect</code> method.)
360      *
361      * <li><p>In order to reuse an existing connection for the
362      * communication, the current security context must have all of
363      * the permissions that would be necessary if the connection were
364      * being created.  Specifically, it must be possible to invoke
365      * <code>checkConnect</code> in the current security context with
366      * this <code>TcpEndpoint</code>'s host and <code>-1</code> for
367      * the port without resulting in a <code>SecurityException</code>,
368      * and it also must be possible to invoke
369      * <code>checkConnect</code> with the remote IP address and port
370      * of the <code>Socket</code> without resulting in a
371      * <code>SecurityException</code> (if the remote socket address is
372      * unresolved, its host name is used instead).  If no existing
373      * connection satisfies these requirements, then this method must
374      * behave as if there are no existing connections.
375      *
376      * </ul>
377      *
378      * <p>Throws {@link NoSuchElementException} if this iterator does
379      * not support making another attempt to communicate the request
380      * (that is, if <code>hasNext</code> would return
381      * <code>false</code>).
382      *
383      * <p>Throws {@link IOException} if an I/O exception occurs while
384      * performing this operation, such as if a connection attempt
385      * timed out or was refused.
386      *
387      * <p>Throws {@link SecurityException} if there is a security
388      * manager and an invocation of its <code>checkConnect</code>
389      * method fails.
390      *
391      * </blockquote>
392      *
393      * @throws NullPointerException {@inheritDoc}
394      **/
395     public OutboundRequestIterator
396 	newRequest(final InvocationConstraints constraints)
397     {
398 	if (constraints == null) {
399 	    throw new NullPointerException();
400 	}
401 
402 	try {
403 	    Constraints.Distilled distilled =
404 		Constraints.distill(constraints, false);
405 	    return connectionManager.newRequest(new Handle(distilled));
406 
407 	} catch (final UnsupportedConstraintException e) {
408 	    return new OutboundRequestIterator() {
409 		private boolean nextCalled = false;
410 		public boolean hasNext() { return !nextCalled; }
411 		public OutboundRequest next() throws IOException {
412 		    if (!hasNext()) { throw new NoSuchElementException(); }
413 		    nextCalled = true;
414 		    e.fillInStackTrace(); // REMIND: is this cool?
415 		    throw e;
416 		}
417 	    };
418 	}
419     }

465 	    port == other.port &&

556     /**
557      * ConnectionEndpoint implementation.
558      *
559      * Instances of this class should never get exposed to anything
560      * other than our ConnectionManager, which we trust to operate
561      * correctly, so we do not bother to validate request handles and
562      * connections passed in.
563      **/
564     private static class ConnectionEndpointImpl implements ConnectionEndpoint {
565         private final String host;
566         private final int port;
567         private final SocketFactory sf;
568 
569 	ConnectionEndpointImpl(String host, int port, SocketFactory sf) {
570             this.host = host;
571             this.port = port;
572             this.sf = sf;
573         }
574 
575 	/**
576 	 * Invoked by ConnectionManager to create a new connection.
577 	 **/
578 	public Connection connect(OutboundRequestHandle handle)
579 	    throws IOException
580 	{
581 	    Handle h = (Handle) handle;
582 	    Constraints.Distilled distilled = h.getDistilledConstraints();
583 
584 	    Socket socket = connectToHost(distilled);
585 
586 	    if (logger.isLoggable(Level.FINE)) {
587 		logger.log(Level.FINE, "connected socket {0}", socket);
588 	    }
589 
590 	    setSocketOptions(socket);
591 
592 	    return new ConnectionImpl(socket);
593 	}
594 
595 	/**
596 	 * Returns a socket connected to this endpoint's host and
597 	 * port, according the specified constraints.  If the host
598 	 * name resolves to multiple addresses, attempts to connect to
599 	 * each of them in order until one succeeds.
600 	 **/
601 	private Socket connectToHost(Constraints.Distilled distilled)
602 	    throws IOException
603 	{
604 	    InetAddress[] addresses;
605 	    try {
606 		addresses = InetAddress.getAllByName(host);
607 	    } catch (UnknownHostException uhe) {
608 		try {
609 		    /*
610 		     * Creating the InetSocketAddress attempts to
611 		     * resolve the host again; in J2SE 5.0, there is a
612 		     * factory method for creating an unresolved
613 		     * InetSocketAddress directly.
614 		     */
615 		    return connectToSocketAddress(
616 			new InetSocketAddress(host, port), distilled);
617 		} catch (IOException e) {
618 		    if (logger.isLoggable(Levels.FAILED)) {
619 			LogUtil.logThrow(logger, Levels.FAILED,
620 			    ConnectionEndpointImpl.class, "connectToHost",
621 			    "exception connecting to unresolved host {0}",
622 			    new Object[] { host + ":" + port }, e);
623 		    }
624 		    throw e;
625 		} catch (SecurityException e) {
626 		    if (logger.isLoggable(Levels.FAILED)) {
627 			LogUtil.logThrow(logger, Levels.FAILED,
628 			    ConnectionEndpointImpl.class, "connectToHost",
629 			    "exception connecting to unresolved host {0}",
630 			    new Object[] { host + ":" + port }, e);
631 		    }
632 		    throw e;
633 		}
634 	    } catch (SecurityException e) {
635 		if (logger.isLoggable(Levels.FAILED)) {
636 		    LogUtil.logThrow(logger, Levels.FAILED,
637 			ConnectionEndpointImpl.class, "connectToHost",
638 			"exception resolving host {0}",
639 			new Object[] { host }, e);
640 		}
641 		throw e;
642 	    }
643 	    IOException lastIOException = null;
644 	    SecurityException lastSecurityException = null;
645 	    for (int i = 0; i < addresses.length; i++) {
646 		SocketAddress socketAddress =
647 		    new InetSocketAddress(addresses[i], port);
648 		try {
649 		    return connectToSocketAddress(socketAddress, distilled);
650 		} catch (IOException e) {
651 		    if (logger.isLoggable(Levels.HANDLED)) {
652 			LogUtil.logThrow(logger, Levels.HANDLED,
653 			    ConnectionEndpointImpl.class, "connectToHost",
654 			    "exception connecting to {0}",
655 			    new Object[] { socketAddress }, e);
656 		    }
657 		    lastIOException = e;
658 		    if (e instanceof SocketTimeoutException) {
659 			break;
660 		    }
661 		} catch (SecurityException e) {
662 		    if (logger.isLoggable(Levels.HANDLED)) {
663 			LogUtil.logThrow(logger, Levels.HANDLED,
664 			    ConnectionEndpointImpl.class, "connectToHost",
665 			    "exception connecting to {0}",
666 			    new Object[] { socketAddress }, e);
667 		    }
668 		    lastSecurityException = e;
669 		}
670 	    }
671 	    if (lastIOException != null) {
672 		if (logger.isLoggable(Levels.FAILED)) {
673 		    LogUtil.logThrow(logger, Levels.FAILED,
674 			ConnectionEndpointImpl.class, "connectToHost",
675 			"exception connecting to {0}",
676 			new Object[] { host + ":" + port },
677 			lastIOException);
678 		}
679 		throw lastIOException;
680 	    }
681 	    assert lastSecurityException != null;
682 	    if (logger.isLoggable(Levels.FAILED)) {
683 		LogUtil.logThrow(logger, Levels.FAILED,
684 		    ConnectionEndpointImpl.class, "connectToHost",
685 		    "exception connecting to {0}",
686 		    new Object[] { host + ":" + port },
687 		    lastSecurityException);
688 	    }
689 	    throw lastSecurityException;
690 	}

692 	/**
693 	 * Returns a socket connected to the specified address, with a
694 	 * timeout governed by the specified constraints.
695 	 **/
696 	private Socket connectToSocketAddress(SocketAddress socketAddress,
697 					      Constraints.Distilled distilled)
698 	    throws IOException
699 	{
700 	    int timeout;
701 	    if (distilled.hasConnectDeadline()) {
702 		long now = System.currentTimeMillis();
703 		long deadline = distilled.getConnectDeadline();
704 		if (deadline <= now) {
705 		    throw new SocketTimeoutException(
706 			"deadline past before connect attempt");
707 		}
708 		assert now > 0; // if not, we could overflow
709 		long delta = deadline - now;
710 		// assume that no socket connect will last over 24 days
711 		timeout = (delta > Integer.MAX_VALUE ? 0 : (int) delta);
712 	    } else {
713 		timeout = 0;
714 	    }
715 
716 	    Socket socket = newSocket();
717 	    boolean ok = false;
718 	    try {
719 		socket.connect(socketAddress, timeout);
720 		ok = true;
721 		return socket;
722 	    } finally {
723 		if (!ok) {
724 		    try {
725 			socket.close();
726 		    } catch (IOException e) {
727 		    }
728 		}
729 	    }
730 	}

732 	/**
733 	 * Returns a new unconnected socket, using this endpoint's
734 	 * socket factory if non-null.
735 	 **/
736 	private Socket newSocket() throws IOException {
737 	    Socket socket;
738 	    if (sf != null) {
739 		socket = sf.createSocket();
740 	    } else {
741 		if (useNIO) {
742 		    socket = SocketChannel.open().socket();
743 		} else {
744 		    socket = new Socket();
745 		}
746 	    }
747 
748 	    if (logger.isLoggable(Level.FINE)) {
749 		logger.log(Level.FINE,
750 			   (sf == null ? "created socket {0}" :
751 			    "created socket {0} using factory {1}"),
752 			   new Object[] { socket, sf });
753 	    }
754 
755 	    return socket;
756 	}

758 	/**
759 	 * Invoked by ConnectionManager to reuse an existing
760 	 * connection.
761 	 **/
762 	public Connection connect(OutboundRequestHandle handle,
763 				  Collection active,
764 				  Collection idle)
765 	{
766 	    if (active == null || idle == null) {
767 		throw new NullPointerException();
768 	    }
769 
770 	    /*
771 	     * The transport level aspects of all constraints
772 	     * supported by this transport provider are always
773 	     * satisfied by all open connections, so we don't need to
774 	     * consider constraints here.
775 	     */
776 
777 	    boolean checkedResolvePermission = false;
778 	    for (Iterator i = active.iterator(); i.hasNext();) {
779 		ConnectionImpl c = (ConnectionImpl) i.next();
780 		if (!checkedResolvePermission) {
781 		    try {
782 			checkResolvePermission();
783 		    } catch (SecurityException e) {
784 			if (logger.isLoggable(Levels.FAILED)) {
785 			    LogUtil.logThrow(logger, Levels.FAILED,
786 				ConnectionEndpointImpl.class, "connect",
787 				"exception resolving host {0}",
788 				new Object[] { host }, e);
789 			}
790 			throw e;
791 		    }
792 		    checkedResolvePermission = true;
793 		}
794 		try {
795 		    c.checkConnectPermission();
796 		    if (logger.isLoggable(Level.FINE)) {
797 			logger.log(Level.FINE,
798 				   "reusing connection {0}", c.getSocket());
799 		    }
800 		    return c;
801 		} catch (SecurityException e) {
802 		    if (logger.isLoggable(Levels.HANDLED)) {
803 			LogUtil.logThrow(logger, Levels.HANDLED,
804 			    ConnectionEndpointImpl.class, "connect",
805 			    "access to reuse connection {0} denied",
806 			    new Object[] { c.getSocket() }, e);
807 		    }
808 		}
809 	    }
810 	    for (Iterator i = idle.iterator(); i.hasNext();) {
811 		ConnectionImpl c = (ConnectionImpl) i.next();
812 		if (!checkedResolvePermission) {
813 		    try {
814 			checkResolvePermission();
815 		    } catch (SecurityException e) {
816 			if (logger.isLoggable(Levels.FAILED)) {
817 			    LogUtil.logThrow(logger, Levels.FAILED,
818 				ConnectionEndpointImpl.class, "connect",
819 				"exception resolving host {0}",
820 				new Object[] { host }, e);
821 			}
822 			throw e;
823 		    }
824 		    checkedResolvePermission = true;
825 		}
826 		try {
827 		    c.checkConnectPermission();
828 		    if (logger.isLoggable(Level.FINE)) {
829 			logger.log(Level.FINE,
830 				   "reusing connection {0}", c.getSocket());
831 		    }
832 		    return c;
833 		} catch (SecurityException e) {
834 		    if (logger.isLoggable(Levels.HANDLED)) {
835 			LogUtil.logThrow(logger, Levels.HANDLED,
836 			    ConnectionEndpointImpl.class, "connect",
837 			    "access to reuse connection {0} denied",
838 			    new Object[] { c.getSocket() }, e);
839 		    }
840 		}
841 	    }
842 	    return null;
843 	}
844 
845 	private void checkResolvePermission() {
846 	    SecurityManager sm = System.getSecurityManager();
847 	    if (sm != null) {
848 		sm.checkConnect(host, -1);
849 	    }
850 	}
851     }

859      * passed in.
860      **/
861     private static class ConnectionImpl implements Connection {
862 
863 	private final Socket socket;

866 	private final InetSocketAddress inetSocketAddress;
867 	private final int socketPort;
868 
869 	ConnectionImpl(Socket socket) {
870 	    assert socket.isConnected();
871 
872 	    this.socket = socket;
873 	    inetSocketAddress =
874 		(InetSocketAddress) socket.getRemoteSocketAddress();
875 	    socketPort = socket.getPort();
876 	}

937 	void checkConnectPermission() {
938 	    SecurityManager sm = System.getSecurityManager();
939 	    if (sm != null) {
940 		if (inetSocketAddress.isUnresolved()) {
941 		    sm.checkConnect(inetSocketAddress.getHostName(),
942 				    socketPort);
943 		} else {
944 		    sm.checkConnect(
945 			inetSocketAddress.getAddress().getHostAddress(),
946 			socketPort);
947 		}
948 	    }

952     /**
953      * Attempts to set desired socket options for a connected socket
954      * (TCP_NODELAY and SO_KEEPALIVE); ignores SocketException.
955      **/
956     private static void setSocketOptions(Socket socket) {
957 	try {
958 	    socket.setTcpNoDelay(true);

963 				 "exception setting TCP_NODELAY on socket {0}",
964 				 new Object[] { socket }, e);
965 	    }
966 	}
967 	try {
968 	    socket.setKeepAlive(true);
969 	} catch (SocketException e) {

971 		LogUtil.logThrow(logger, Levels.HANDLED,
972 				 TcpEndpoint.class, "setSocketOptions",
