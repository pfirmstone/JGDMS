46  /**
47   * Constraint support for this transport provider.
48   *
49   * This code makes some significant simplifying assumptions:
50   *
51   * - The transport layer aspects of all constraints supported by this
52   *   provider are always satisfied by all open connections and
53   *   requests.
54   *
55   * - No combination of individual constraints supported by this
56   *   provider can contain conflicting constraints.
57   *
58   * 
59   **/
60  class Constraints {
61  
62      /**
63       * indicates that this provider does not support implementing (or
64       * does not understand how to implement) the transport layer
65       * aspects of satisfying a given constraint
66       **/
67      private static final int NO_SUPPORT = 0;
68  
69      /**
70       * indicates that this provider supports implementing all aspects
71       * of satisfying a given constraint
72       **/
73      private static final int FULL_SUPPORT = 1;
74  
75      /**
76       * indicates that this provider supports implementing the
77       * transport layer aspects of satisfying a given constraint, but
78       * at least partial implementation by higher layers is also needed
79       * in order to fully satisfy the constraint
80       **/
81      private static final int PARTIAL_SUPPORT = 2;
82  
83      /**
84       * maps constraint values that are supported to Boolean indicating
85       * whether or not they must be at least partially implemented by
86       * higher layers to be fully satisfied
87       **/
88      private static final Map supportedValues = new HashMap();
89      static {
90  	supportedValues.put(Integrity.NO,		Boolean.TRUE);
91  	supportedValues.put(Confidentiality.NO,		Boolean.FALSE);
92  	supportedValues.put(ClientAuthentication.NO,	Boolean.FALSE);
93  	supportedValues.put(ServerAuthentication.NO,	Boolean.FALSE);
94  	supportedValues.put(Delegation.NO,		Boolean.FALSE);
95      }
96  
97      /**
98       * maps constraint classes that are supported to Boolean
99       * indicating whether or not such constraints must be at least
100      * partially implemented by higher layers to be fully satisfied
101      **/
102     private static final Map supportedClasses = new HashMap();
103     static {
104 	// ConstraintAlternatives is supported but handled specially in code
105 	supportedClasses.put(ConnectionAbsoluteTime.class,	Boolean.FALSE);
106 	supportedClasses.put(ConnectionRelativeTime.class,	Boolean.FALSE);
107 	/*
108 	 * The following classes are (trivially) supported just
109 	 * because ClientAuthentication.YES, ServerAuthentication.YES,
110 	 * and Delegation.YES are not supported.
111 	 */
112 	supportedClasses.put(ClientMaxPrincipal.class,		Boolean.FALSE);
113 	supportedClasses.put(ClientMaxPrincipalType.class,	Boolean.FALSE);
114 	supportedClasses.put(ClientMinPrincipal.class,		Boolean.FALSE);
115 	supportedClasses.put(ClientMinPrincipalType.class,	Boolean.FALSE);
116 	supportedClasses.put(ServerMinPrincipal.class,		Boolean.FALSE);
117 	supportedClasses.put(DelegationAbsoluteTime.class,	Boolean.FALSE);
118 	supportedClasses.put(DelegationRelativeTime.class,	Boolean.FALSE);
119     }
120 
121     /**
122      * Returns this provider's general support for the given
123      * constraint.
124      **/
125     private static int getSupport(InvocationConstraint c) {
126 	Boolean support = (Boolean) supportedValues.get(c);
127 	if (support == null) {
128 	    support = (Boolean) supportedClasses.get(c.getClass());
129 	}
130 	return support == null ? NO_SUPPORT :
131 	    support.booleanValue() ? PARTIAL_SUPPORT : FULL_SUPPORT;
132     }
133 
134     /**
135      * Checks that we support at least the transport layer aspects of
136      * the given requirements (and throws an
137      * UnsupportedConstraintException if not), and returns the
138      * requirements that must be at least partially implemented by
139      * higher layers and the supported preferences that must be at
140      * least partially implemented by higher layers.
141      *
142      * [If this provider supported constraints whose transport layer
143      * aspects were not always satisfied by open connections or
144      * requests, then we would need a variant of this method that
145      * checks the given constraints against an open connection or
146      * request.  If this provider supported constraints that could
147      * conflict with each other then (when not checking against an
148      * open connection or request) we would need to check for possible
149      * conflicts.]
150      **/
151     static InvocationConstraints check(InvocationConstraints constraints,
152 				       boolean relativeOK)
153 	throws UnsupportedConstraintException
154     {
155 	return distill(constraints, relativeOK).getUnfulfilledConstraints();
156     }
157 
158     /**
159      * Distills the given constraints to a form more directly usable
160      * by this provider.  Throws an UnsupportedConstraintException if
161      * we do not support at least the transport layer aspects of the
162      * requirements.
163      **/
164     static Distilled distill(InvocationConstraints constraints,
165 			     boolean relativeOK)
166 	throws UnsupportedConstraintException
167     {
168 	return new Distilled(constraints, relativeOK);
169     }
170 
171     private Constraints() { throw new AssertionError(); }
172 
173     /**
174      * A distillation of constraints to a form more directly usable by
175      * this provider.
176      **/
177     static class Distilled {
178 
179 	/**
180 	 * true if relative time constraints are allowed (in other
181 	 * words, not for client-side use)
182 	 */
183 	private final boolean relativeOK;
184 
185 	private Collection unfulfilledRequirements = null; // lazily created
186 	private Collection unfulfilledPreferences = null; // lazily created
187 
188 	private boolean hasConnectDeadline = false;
189 	private long connectDeadline;
190 
191 	Distilled(InvocationConstraints constraints, boolean relativeOK)
192 	    throws UnsupportedConstraintException
193 	{
194 	    this.relativeOK = relativeOK;
195 	    for (Iterator i = constraints.requirements().iterator();
196 		 i.hasNext();)
197 	    {
198 		addConstraint((InvocationConstraint) i.next(), true);
199 	    }
200 	    for (Iterator i = constraints.preferences().iterator();
201 		 i.hasNext();)
202 	    {
203 		addConstraint((InvocationConstraint) i.next(), false);
204 	    }
205 	}
206 
207 	/**
208 	 * Returns the requirements and supported preferences that
209 	 * must be at least partially implemented by higher layers.
210 	 **/
211 	InvocationConstraints getUnfulfilledConstraints() {
212 	    if (unfulfilledRequirements == null &&
213 		unfulfilledPreferences == null)
214 	    {
215 		return InvocationConstraints.EMPTY;
216 	    } else {
217 		return new InvocationConstraints(unfulfilledRequirements,
218 						 unfulfilledPreferences);
219 	    }
220 	}
221 
222 	/**
223 	 * Returns true if a there is a socket connect deadline.
224 	 **/
225 	boolean hasConnectDeadline() {
226 	    return hasConnectDeadline;
227 	}
228 
229 	/**
230 	 * Returns the absolute time of the socket connect deadline.
231 	 **/
232 	long getConnectDeadline() {
233 	    assert hasConnectDeadline;
234 	    return connectDeadline;
235 	}
236 
237 	/**
238 	 * If "isRequirement" is true, throws an
239 	 * UnsupportedConstraintException if we do not support at
240 	 * least the transport layer aspects of the given constraint.
241 	 *
242 	 * If we do support at least the transport layer aspects of
243 	 * the given constraint, then if appropriate, adds it to the
244 	 * collection of requirements or preferences that must be at
245 	 * least partially implemented by higher layers.
246 	 **/
247 	private void addConstraint(InvocationConstraint constraint,
248 				   boolean isRequirement)
249 	    throws UnsupportedConstraintException
250 	{
251 	    if (!(constraint instanceof ConstraintAlternatives)) {
252 		int support = getSupport(constraint);
253 		if (support == NO_SUPPORT ||
254 		    (!relativeOK &&
255 		     constraint instanceof RelativeTimeConstraint))
256 		{
257 		    if (isRequirement) {
258 			throw new UnsupportedConstraintException(
259 			    "cannot satisfy constraint: " + constraint);
260 		    } else {
261 			return;
262 		    }
263 		}
264 		if (support == PARTIAL_SUPPORT) {
265 		    if (isRequirement) {
266 			if (unfulfilledRequirements == null) {
267 			    unfulfilledRequirements = new ArrayList();
268 			}
269 			unfulfilledRequirements.add(constraint);
270 		    } else {
271 			if (unfulfilledPreferences == null) {
272 			    unfulfilledPreferences = new ArrayList();
273 			}
274 			unfulfilledPreferences.add(constraint);
275 		    }
276 		}
277 		if (constraint instanceof ConnectionAbsoluteTime) {
278 		    // REMIND: only bother with this on client side?
279 		    addConnectDeadline(
280 			((ConnectionAbsoluteTime) constraint).getTime());
281 		}
282 	    } else {
283 		addAlternatives((ConstraintAlternatives) constraint,
284 				isRequirement);
285 	    }
286 	}
287 
288 	/**
289 	 * If "isRequirement" is true, throws an
290 	 * UnsupportedConstraintException if we do not support at
291 	 * least the transport layer aspects of at least one of the
292 	 * constraints in the given alternatives.
293 	 *
294 	 * If we do support at least the transport layer aspects of at
295 	 * least one of the constraints in the given alternatives,
296 	 * then if appropriate, adds a ConstraintAlternatives of the
297 	 * supported alternatives to the collection of requirements or
298 	 * preferences that must be at least partially implemented by
299 	 * higher layers.
300 	 *
301 	 * If all of the supported alternatives need at least partial
302 	 * implementation by higher layers, then adds a
303 	 * ConstraintAlternatives with all of the supported
304 	 * alternatives to the unfulfilled collection or preferences,
305 	 * because higher layers must support at least one of them.
306 	 * But if at least one of the supported alternatives can be
307 	 * fully satisfied by the transport layer, then add nothing to
308 	 * the unfulfilled collection, because it is possible that
309 	 * higher layers need not support any of them (and there is no
310 	 * way to express no constraint).
311 	 *
312 	 * The weakest connect deadline (with no deadline being the
313 	 * the weakest possibility) is chosen among alternatives.
314 	 **/
315 	private void addAlternatives(ConstraintAlternatives constraint,
316 				     boolean isRequirement)
317 	    throws UnsupportedConstraintException
318 	{
319 	    Collection alts = constraint.elements();
320 	    boolean supported = false;
321 	    long maxConnectDeadline = Long.MIN_VALUE;
322 	    Collection unfulfilledAlts = null; // lazily created
323 	    boolean forgetUnfulfilled = false;
324 	    for (Iterator i = alts.iterator(); i.hasNext();) {
325 		InvocationConstraint c = (InvocationConstraint) i.next();
326 
327 		// nested ConstraintAlternatives not allowed
328 		int support = getSupport(c);
329 		if (support == NO_SUPPORT ||
330 		    (!relativeOK && c instanceof RelativeTimeConstraint))
331 		{
332 		    continue;
333 		}
334 		supported = true;	// we support at least one
335 		if (!forgetUnfulfilled) {
336 		    if (support == PARTIAL_SUPPORT) {
337 			if (unfulfilledAlts == null) {
338 			    unfulfilledAlts = new ArrayList();
339 			}
340 			unfulfilledAlts.add(c);
341 		    } else {
342 			assert support == FULL_SUPPORT;
343 			unfulfilledAlts = null;
344 			forgetUnfulfilled = true;
345 		    }
346 		}
347 		if (c instanceof ConnectionAbsoluteTime) {
348 		    assert support == FULL_SUPPORT; // else more care required
349 		    maxConnectDeadline =
350 			Math.max(maxConnectDeadline,
351 				 ((ConnectionAbsoluteTime) c).getTime());
352 		} else {
353 		    maxConnectDeadline = Long.MAX_VALUE;
354 		}
355 	    }
356 	    if (!supported) {
357 		if (isRequirement) {
358 		    throw new UnsupportedConstraintException(
359 			"cannot satisfy constraint: " + constraint);
360 		} else {
361 		    return; // maxConnectDeadline is bogus in this case
362 		}
363 	    }
364 	    if (!forgetUnfulfilled && unfulfilledAlts != null) {
365 		if (isRequirement) {
366 		    if (unfulfilledRequirements == null) {
367 			unfulfilledRequirements = new ArrayList();
368 		    }
369 		    unfulfilledRequirements.add(
370 			ConstraintAlternatives.create(unfulfilledAlts));
371 		} else {
372 		    if (unfulfilledPreferences == null) {
373 			unfulfilledPreferences = new ArrayList();
374 		    }
375 		    unfulfilledPreferences.add(
376 			ConstraintAlternatives.create(unfulfilledAlts));
377 		}
378 	    }
379 	    if (maxConnectDeadline < Long.MAX_VALUE) {
380 		assert maxConnectDeadline != Long.MIN_VALUE;
381 		addConnectDeadline(maxConnectDeadline);
382 	    }
383 	}
384 
385 	/**
386 	 * Adds the given connect deadline to this object's state.
387 	 * The earliest connect deadline is what gets remembered.
388 	 **/
389 	private void addConnectDeadline(long deadline) {
390 	    if (!hasConnectDeadline) {
391 		hasConnectDeadline = true;
392 		connectDeadline = deadline;
393 	    } else {
394 		connectDeadline = Math.min(connectDeadline, deadline);
395 	    }
396 	}
397     }
