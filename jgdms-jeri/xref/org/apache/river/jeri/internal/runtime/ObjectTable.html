48      /**
49       * lock to serialize request dispatcher reservation per export, so
50       * that a partial export will not cause another export to fail
51       * unnecessarily
52       **/
53      private final Object requestDispatchersLock;
54  
55      /** table of references to impls exported with DGC */
56      private final ImplRefManager implRefManager;
57  
58      /** number of objects exported with keepAlive == true */
59      private final JvmLifeSupport keepAliveCount;
60  
61      /** maps client ID to Lease (lock guards leaseChecker too) */
62      private final ConcurrentMap<Uuid,Lease> leaseTable;
63  
64      /** thread to check for expired leases */
65      private Thread leaseChecker;
66      
67      /** expired lease thread guard */
68      private final Object runLock;
69      
70      /** flag to indicate expired lease thread started */
71      private volatile boolean running;

73      ObjectTable() { 
74          requestDispatchersLock = new Object();
75          implRefManager = new ImplRefManager();
76          keepAliveCount = new JvmLifeSupport();
77          leaseTable = new ConcurrentHashMap<Uuid,Lease>(256);//Plenty of capacity to reduce resizing.
78          leaseChecker = null;
79          running = false;
80          runLock = new Object();
81      }

84  	return new DgcRequestDispatcher(unrefCallback, this);
85      }

87      boolean isReferenced(RequestDispatcher requestDispatcher) {
88  	return getRD(requestDispatcher).isReferenced();
89      }

91      DgcServer getDgcServer(DgcRequestDispatcher dgdRD){
92          return new DgcServerImpl(dgdRD);
93      }
94  
95      Target export(Remote impl,
96  		  RequestDispatcher[] requestDispatchers,
97  		  boolean allowDGC,
98  		  boolean keepAlive,
99  		  Uuid id)
100         throws ExportException
101     {
102 	DgcRequestDispatcher[] rds = new DgcRequestDispatcher[requestDispatchers.length];
103 	for (int i = 0; i < requestDispatchers.length; i++) {
104 	    rds[i] = getRD(requestDispatchers[i]);
105 	}
106         SecurityContext securityContext = Security.getContext();
107         ClassLoader ccl = Thread.currentThread().getContextClassLoader();
108         Target t = null;
109         t = new Target(id, rds, allowDGC, keepAlive,this,
110                 securityContext, ccl, keepAliveCount);
111         synchronized (requestDispatchersLock){
112             t.procRequestDispatchers();
113         }
114         ImplRef implRef = implRefManager.getImplRef(impl, t);
115         t.setImplRef(implRef);
116         t.setExported();
117 	return t;
118     }

120     private DgcRequestDispatcher getRD(RequestDispatcher requestDispatcher) {
121 	/*
122 	 * The following cast will throw a ClassCastException if we were
123 	 * passed a RequestDispatcher that was not returned by this class's
124 	 * createRequestDispatcher method:
125 	 */
126 	DgcRequestDispatcher rd = (DgcRequestDispatcher) requestDispatcher;
127 	if (!rd.forTable(this)) {
128 	    throw new IllegalArgumentException(
129 		"request dispatcher for different object table");
130 	}
131 	return rd;
132     }
133 
134     void registerTarget(Target target, Uuid clientID) {
135         Lease lease = leaseTable.get(clientID);
136         if (lease == null) {
137             target.leaseExpired(clientID);
138         } else {
139             boolean added = lease.add(target);
140             if ( added == false){
141                 // lease has been locked because it has expired
142                 // prior to removal
143                 target.leaseExpired(clientID);
144             }
145         }
146     }
147 
148     void unregisterTarget(Target target, Uuid clientID) {
149         Lease lease = leaseTable.get(clientID);
150         if (lease != null) {
151             lease.remove(target);
152         }
153     }

164         public long dirty(Uuid clientID,
165                           long sequenceNum,
166                           Uuid[] ids)
167         {
168             if (logger.isLoggable(Level.FINEST)) {
169                 logger.log(Level.FINEST,
170                     "clientID={0}, sequenceNum={1}, ids={2}",
171                     new Object[] {
172                         clientID, Long.valueOf(sequenceNum), Arrays.asList(ids)
173                     });
174             }
175 
176             long duration = Jeri.leaseValue;
177 
178             Lease lease = leaseTable.get(clientID);
179             if (lease == null) {
180                 lease = new Lease(clientID, duration);
181                 Lease existed = leaseTable.putIfAbsent(clientID,lease);
182                 if (existed != null){
183                     assert clientID.equals(existed.getClientID());
184                     boolean renewed = existed.renew(duration);
185                     if (!renewed){
186                         /* The probability of getting here is low,
187                          * it indicates a lease with an extremely short 
188                          * expiry and a very small lease table.
189                          */               
190                         if (logger.isLoggable(Level.WARNING)) {
191                             logger.log(Level.WARNING,
192                                 "Problem with lease table, try a longer " +
193                                 "lease duration clientID={0}, " +
194                                 "sequenceNum={1}, ids={2}",
195                                 new Object[] {
196                                     clientID, Long.valueOf(sequenceNum), Arrays.asList(ids)
197                                 });
198                         }                          
199                     }
200                 }
201             } else {
202                 assert clientID.equals(lease.getClientID());
203                 boolean renewed = lease.renew(duration);
204                 if (!renewed){
205                     // Indicates an expired lease in the table.  A lease
206                     // always becomes expired prior to removal, it is 
207                     // never removed prior to expiry, in case it is
208                     // renewed by another thread, which would risk a renewed
209                     // lease being removed from the table.  
210                     // An expired lease must be replaced.
211                     leaseTable.remove(clientID, lease); // Another thread could remove it first.
212                     lease = new Lease(clientID, duration);
213                     Lease existed = leaseTable.putIfAbsent(clientID, lease);
214                     if (existed != null){
215                         lease = existed;
216                         assert clientID.equals(lease.getClientID());
217                         renewed = lease.renew(duration);
218                         if (!renewed){
219                             /* The probability of getting here is low,
220                              * it indicates a lease of extremely short 
221                              * duration and a very small lease table.
222                              */               
223                             if (logger.isLoggable(Level.WARNING)) {
224                                 logger.log(Level.WARNING,
225                                     "Problem with lease table, try a longer " +
226                                     "lease duration clientID={0}, " +
227                                     "sequenceNum={1}, ids={2}",
228                                     new Object[] {
229                                         clientID, Long.valueOf(sequenceNum), Arrays.asList(ids)
230                                     });
231                             }                          
232                         }  
233                     }
234                 }
235             }
236             /* FIXED: River-142: 
237              * In the server-side DGC implementation's thread that check's for
238              * lease expirations 
239              * (org.apache.river.jeri.internal.runtime.ObjectTable.LeaseChecker.run),
240              * it checks for them while synchronized on the overall lease table,
241              * but it delays notifying the expired leases' individual registered
242              * Targets about the expirations until after it has released the
243              * lease table lock. This approach was taken from the 
244              * JRMP implementation, which is that way because of the fix 
245              * for 4118056 (a previous deadlock bug-- but now, I'm thinking 
246              * that the JRMP implementation has this bug too).
247              *
248              * The problem seems to be that after releasing the lease table 
249              * lock, it is possible for another lease renewal/request to 
250              * come in (from the same DGC client and for the same remote object)
251              * that would then be invalidated by the subsequent Target 
252              * notification made by the lease expiration check thread-- and 
253              * thus the client's lease renewal (for that remote object) will 
254              * be forgotten. It would appear that the synchronization approach 
255              * here needs to be reconsidered.
256              * 
257              * ( Comments note: )
258              * In addition to the basic problem of the expired-then-renewed 
259              * client being removed from the referenced set, there is also 
260              * the problem of the sequence table entry being forgotten-- which 
261              * prevents detection of a "late clean call".
262              * Normally, late clean calls are not a problem because sequence 
263              * numbers are retained while the client is in the referenced set 
264              * (and there is no such thing as a "strong dirty"). 
265              * But in this case, with the following order of events on 
266              * the server side:
267              *
268              *   1. dirty, seqNo=2
269              *   2. (lease expiration)
270              *   3. clean, seqNo=1
271              *
272              * The primary bug here is that the first two events will leave 
273              * the client missing from the referenced set. But the secondary 
274              * bug is that even if that's fixed, with the sequence number 
275              * forgotten, the third event (the "late clean call") will still 
276              * cause the client to be removed from the referenced set.
277              * 
278              * FIX:
279              * This issue has been fixed by making the Lease responsible for
280              * it's own state, which is protected internally by synchronization.
281              * The lease checker passes the time to the Lease, which checks 
282              * itself and notifies the Targets in the event of expiry.
283              * 
284              * Because the notification is not delayed, the client id and
285              * sequence number will not be not be removed by the LeaseChecker
286              * thread after being updated by the second dirty call (when
287              * the lock was cleared as described).
288              * 
289              * The client id and sequence number are added to the Target sequence
290              * table by the second dirty call, after the Lease removes
291              * them immediately upon expiry being invoked by the LeaseChecker.
292              * 
293              * Then because the late clean call sequence number is less than the 
294              * second dirty call and exists, it is correctly recognised.
295              */
296             // This must be performed after changing the leaseTable
297             if (!running){ // double checked
298                 synchronized (runLock){
299                     if (!running) {
300                         leaseChecker =
301                             (Thread) AccessController.doPrivileged(
302                                 new NewThreadAction(new LeaseChecker(),
303                                     "DGC Lease Checker", true));
304                         leaseChecker.start();
305                         running = true;
306                     }
307                 }
308             }
309             for (int i = 0; i < ids.length; i++) {
310                 Target target = dgcRequestDispatcher.get(ids[i]);
311                 if (target != null) {
312                     target.referenced(clientID, sequenceNum);
313                 }
314             }
315             return duration;
316         }

318         public void clean(Uuid clientID,
319                           long sequenceNum,
320                           Uuid[] ids,
321                           boolean strong)
322         {
323             if (logger.isLoggable(Level.FINEST)) {
324                 logger.log(Level.FINEST,
325                     "clientID={0}, sequenceNum={1}, ids={2}, strong={3}",
326                     new Object[] {
327                         clientID, Long.valueOf(sequenceNum),
328                         Arrays.asList(ids), Boolean.valueOf(strong)
329                     });
330             }
331 
332             for (int i = 0; i < ids.length; i++) {
333                 Target target = dgcRequestDispatcher.get(ids[i]);
334                 if (target != null) {
335                     target.unreferenced(clientID, sequenceNum, strong);
336                 }
337             }
338         }
339     }

341     private class LeaseChecker implements Runnable {
342 
343 	public void run() {
344 	    boolean done = false;
345             try {
346                 do {
347                     Thread.sleep(Jeri.leaseCheckInterval);
348                     long now = System.currentTimeMillis();		
349                     for (Iterator i = leaseTable.values().iterator();
350                          i.hasNext();)
351                     {
352                         Lease lease = (Lease) i.next();
353                         boolean expired = lease.notifyIfExpired(now);
354                         if (expired) {			    
355                             i.remove();
356                         }
357                     }
358                     if (leaseTable.isEmpty()) {
359                         done = true;
360                     }		
361                 } while (!done);
362             } catch (InterruptedException e) {
363                 // REMIND: shouldn't happen, OK to ignore?
364                 // No, restore the interrupted status
365                  Thread.currentThread().interrupt();
366             } finally {
367                 // This is always executed and returns the lease checker
368                 // to the non running state, such that if the application
369                 // has not exited, another thread will be started eventually.
370                 synchronized (runLock){
371                     leaseChecker = null;
372                     running = false;               
373                 }
374             }
375 	}
376     }
