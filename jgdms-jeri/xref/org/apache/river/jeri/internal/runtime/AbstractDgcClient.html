41  /**
42   * AbstractDgcClient implements the client-side behavior of RMI's
43   * distributed garbage collection system abstractly with respect to
44   * the types used to represent transport endpoints, object IDs, and
45   * live remote references.  The actual types used for endpoints,
46   * object IDs, and live references depends on the concrete subclass.
47   *
48   * The entry point into the machinery of AbstractDgcClient is the
49   * "registerRefs" method: when a live reference enters the scope of
50   * this AbstractDgcClient (the current virtual machine, for example),
51   * it should be registered with that method in order for it to
52   * participate in distributed garbage collection.
53   *
54   * When the first live reference to a particular remote object is
55   * registered, a "dirty" call is made to the server-side distributed
56   * garbage collector at the remote object's endpoint, which, if
57   * successful, returns a lease guaranteeing that the server-side DGC
58   * will not collect the remote object for a certain period of time.
59   * While live references to remote objects at a particular endpoint
60   * exist, this AbstractDgcClient will continue to make more "dirty"
61   * calls to renew its lease on the referenced remote objects.
62   *
63   * This AbstractDgcClient tracks the local reachability of registered
64   * live references (using phantom references).  When all of the live
65   * reference instances for a particular remote object become garbage
66   * collected locally, a "clean" call is made to the server-side
67   * distributed garbage collector, indicating that the server no longer
68   * needs to keep the remote object alive for this client.
69   *
70   * Internally, AbstractDgcClient holds and manipulates transport
71   * endpoints, object IDs, and live references with references of type
72   * java.lang.Object; it is assumed that their actual classes define
73   * "equals" and "hashCode" in a meaningful way.
74   *
75   * Concrete subclasses must provide additional behavior for the actual
76   * endpoint, object ID, and live reference types by implementing the
77   * abstract protected methods of this class (see below).  In
78   * particular, the "getDgcProxy" method should return an object that
79   * implements the actual protocol for DGC "dirty" and "clean" calls
80   * for the endpoint type being used, the "freeEndpoint" method may
81   * make use of the indication that a particular endpoint no longer has
82   * references, and the "getRefEndpoint" and "getRefObjectID" methods
83   * should return the endpoint and object ID contained in a particular
84   * live reference object.  A concrete subclass should also provide a
85   * type-safe equivalent of "registerRefs" that delegates to this
86   * class's "registerRefs" method.
87   *
88   * @author Sun Microsystems, Inc.
89   **/
90  abstract class AbstractDgcClient {

92      /** lease duration to request (usually ignored by server) */
93      private static final long leaseValue =		// default 10 minutes
94  	( AccessController.doPrivileged(new GetLongAction(
95  	    "org.apache.river.jeri.dgc.leaseValue", 600000)))
96  	    .longValue();

98      /** maximum interval between retries of failed clean calls */
99      private static final long cleanInterval =		// default 3 minutes
100 	( AccessController.doPrivileged(new GetLongAction(

102 	    .longValue();

105     private static final long minimumDuration =		// default 5 seconds
106 	( AccessController.doPrivileged(new GetLongAction(
107 	    "org.apache.river.jeri.dgc.minimumDuration", 5000)))
108 	    .longValue();

131     /**
132      * A DgcProxy is a proxy for invoking DGC operations on a server-side
133      * DGC implementation at a particular endpoint.  A DgcProxy instance
134      * for a given endpoint is obtained from concrete class (using the
135      * getDgcProxy method) and used by the abstract implementation.
136      */
137     protected interface DgcProxy {
138 	long dirty(long sequenceNum, Object[] ids, long duration)
139 	    throws RemoteException;
140 	void clean(long sequenceNum, Object[] ids, boolean strong)
141 	    throws RemoteException;
142     }
143     /** Returns a proxy for making DGC calls to the given endpoint. */
144     protected abstract DgcProxy getDgcProxy(Object endpoint);
145     /** Indicates that resources for the given endpoint may be freed. */
146     protected abstract void freeEndpoint(Object endpoint);
147     /** Returns the endpoint in the given live reference. */
148     protected abstract Object getRefEndpoint(Object ref);
149     /** Returns the object ID in the given live reference. */
150     protected abstract Object getRefObjectID(Object ref);

152     /**
153      * Registers the live reference instances in the supplied collection to
154      * participate in distributed garbage collection.
155      *
156      * All of the live references in the list must be for remote objects at
157      * the given endpoint.
158      */
159     protected final void registerRefs(Object endpoint, Collection refs) {
160 	/*
161 	 * Look up the given endpoint and register the refs with it.
162 	 * The retrieved entry may get removed from the global endpoint
163 	 * table before EndpointEntry.registerRefs() is able to acquire
164 	 * its lock; in this event, it returns false, and we loop and
165 	 * try again.
166 	 */
167 	EndpointEntry epEntry;
168 	do {
169 	    epEntry = getEndpointEntry(endpoint);
170 	} while (!epEntry.registerRefs(refs));
171     }
172 
173     /**
174      * Gets the next sequence number to be used for a dirty or clean
175      * operation from this AbstractDgcClient.  This method should only be
176      * called while synchronized on the EndpointEntry whose data structures
177      * the operation affects.
178      */
179     private static synchronized long getNextSequenceNum() {
180 	return nextSequenceNum++;
181     }
182 
183     /**
184      * Given the length of a lease and the time that it was granted,
185      * computes the absolute time at which it should be renewed, giving
186      * room for reasonable computational and communication delays.
187      */
188     private static long computeRenewTime(long grantTime, long duration) {
189 	/*
190 	 * REMIND: This algorithm should be more sophisticated, waiting
191 	 * a longer fraction of the lease duration for longer leases.
192 	 */
193 	return grantTime + (duration / 2);
194     }

196     /**
197      * Looks up the EndpointEntry for the given endpoint.  An entry is
198      * created if one does not already exist.
199      */
200     private EndpointEntry getEndpointEntry(Object endpoint) {
201 	synchronized (endpointTable) {
202 	    EndpointEntry entry = (EndpointEntry) endpointTable.get(endpoint);
203 	    if (entry == null) {
204 		entry = new EndpointEntry(endpoint);
205                 entry.start();
206 		endpointTable.put(endpoint, entry);
207 		/*
208 		 * If the endpoint table was previously empty, we are now
209 		 * interested in special assistance from the local garbage
210 		 * collector for aggressively discovering unreachable live
211 		 * remote references (by notifying our phantom references),
212 		 * so that DGC "clean" calls can be sent in a timely fashion.
213 		 *
214 		 * Without guaranteed access to something like the
215 		 * sun.misc.GC API, however, we currently have no
216 		 * practical way of getting such special assistance.
217 		 */
218 	    }
219 	    return entry;
220 	}
221     }

223     /**
224      * EndpointEntry encapsulates the client-side DGC information specific
225      * to a particular endpoint.  Of most significance is the table that
226      * maps live reference objects to RefEntry objects and the renew/clean
227      * thread that handles asynchronous client-side DGC operations.
228      */
229     private final class EndpointEntry {
230 
231 	/** the endpoint that this EndpointEntry is for */
232 	private final Object endpoint;
233 	/** synthesized reference to the remote server-side DGC */
234 	private final DgcProxy dgcProxy;
235 	/** renew/clean thread for handling lease renewals and clean calls */
236 	private final Thread renewCleanThread;
237 	/** reference queue for phantom references */
238 	private final ReferenceQueue refQueue = new ReferenceQueue();
239 
240 	/* mutable instance state (below) is guarded by this object's lock */
241 
242 	/** true if this entry has been removed from the global table */
243 	private volatile boolean removed = false;
244 
245 	/** table of refs held for endpoint: maps object ID to RefEntry */
246 	private final Map refTable = new HashMap(5);
247 	/** set of RefEntry instances from last (failed) dirty call */
248 	private final Set invalidRefs = new HashSet(5);
249 
250 	/** absolute time to renew current lease to this endpoint */
251 	private long renewTime = Long.MAX_VALUE;
252 	/** absolute time current lease to this endpoint will expire */
253 	private long expirationTime = Long.MIN_VALUE;
254 	/** count of recent dirty calls that have failed */
255 	private int dirtyFailures = 0;
256 	/** absolute time of first recent failed dirty call */
257 	private long dirtyFailureStartTime;
258 	/** (average) elapsed time for recent failed dirty calls */
259 	private long dirtyFailureDuration;
260 
261 	/** true if renew/clean thread may be interrupted */
262 	private boolean interruptible = false;
263 
264 	/** set of clean calls that need to be made, changed Set to an
265          * underlying ConcurrentHashMap because no lock is held while 
266          * iterating and processing pending Clean Requests*/
267 	private final Set<CleanRequest> pendingCleans 
268                 = Collections.newSetFromMap(new ConcurrentHashMap<CleanRequest,Boolean>(5));
269 
270 	private EndpointEntry(final Object endpoint) {
271 	    this.endpoint = endpoint;
272 	    dgcProxy = getDgcProxy(endpoint);
273 	    renewCleanThread = (Thread)	AccessController.doPrivileged(
274 		new NewThreadAction(new RenewCleanThread(),
275 				    "RenewClean-" + endpoint, true));
276 	}
277         
278         private void start(){
279             renewCleanThread.start();
280         }
281         
282         boolean pendingCleanCalls(){
283             return !pendingCleans.isEmpty();
284         }
285         
286         boolean removed(){
287             return removed;
288         }

290 	/**
291 	 * Registers the live reference instances in the supplied list to
292 	 * participate in distributed garbage collection.
293 	 *
294 	 * This method returns false if this entry was removed from the
295 	 * global endpoint table (because it was empty) before these refs
296 	 * could be registered.  In that case, a new EndpointEntry needs
297 	 * to be looked up.
298 	 *
299 	 * This method must NOT be invoked while synchronized on this
300 	 * EndpointEntry.
301 	 */
302 	boolean registerRefs(Collection refs) {
303 	    assert !Thread.holdsLock(this);
304 
305 	    Set refsToDirty = null;	// entries for refs needing dirty
306 	    long sequenceNum;		// sequence number for dirty call
307 
308 	    synchronized (this) {
309 		if (removed) {
310 		    return false;
311 		}
312 
313 		Iterator iter = refs.iterator();
314 		while (iter.hasNext()) {
315 		    Object ref = iter.next();
316 		    assert getRefEndpoint(ref).equals(endpoint);
317 
318 		    Object objectID = getRefObjectID(ref);
319 		    RefEntry refEntry = (RefEntry) refTable.get(objectID);
320 		    if (refEntry == null) {
321 			refEntry = new RefEntry(objectID);
322 			refTable.put(objectID, refEntry);
323 			if (refsToDirty == null) {
324 			    refsToDirty = new HashSet(5);
325 			}
326 			refsToDirty.add(refEntry);
327 		    }
328 
329 		    refEntry.addInstanceToRefSet(ref);
330 		}
331 
332 		if (refsToDirty == null) {
333 		    return true;
334 		}
335 
336 		refsToDirty.addAll(invalidRefs);
337 		invalidRefs.clear();
338 
339 		sequenceNum = getNextSequenceNum();
340 	    }
341 
342 	    makeDirtyCall(refsToDirty, sequenceNum);
343 	    return true;
344 	}
345 
346 	/**
347 	 * Removes the given RefEntry from the ref table.  If that makes
348 	 * the ref table empty, remove this entry from the global endpoint
349 	 * table.
350 	 *
351 	 * This method must ONLY be invoked while synchronized on this
352 	 * EndpointEntry.
353 	 */
354 	private synchronized void removeRefEntry(RefEntry refEntry) {
355 	    assert !removed;
356 	    assert refTable.containsKey(refEntry.getObjectID());
357 
358 	    refTable.remove(refEntry.getObjectID());
359 	    invalidRefs.remove(refEntry);
360 	    if (refTable.isEmpty()) {
361 		synchronized (endpointTable) {
362 		    endpointTable.remove(endpoint);
363 		    freeEndpoint(endpoint);
364 		    /*
365 		     * If the endpoint table is now empty, we are no longer
366 		     * interested in special assistance from the local garbage
367 		     * collector for aggressively discovering unreachable
368 		     * live remote references, if we had been getting such
369 		     * special assistance in the first place.
370 		     */
371 		}
372 		removed = true;
373 	    }
374 	}

376 	/**
377 	 * Makes a DGC dirty call to this entry's endpoint, for the
378 	 * object IDs corresponding to the given set of refs and with
379 	 * the given sequence number.
380 	 *
381 	 * This method must NOT be invoked while synchronized on this
382 	 * EndpointEntry.
383 	 */
384 	private void makeDirtyCall(Set refEntries, long sequenceNum) {
385 	    assert !Thread.holdsLock(this);
386 
387 	    Object[] ids;
388 	    if (refEntries != null) {
389 		ids = createObjectIDArray(refEntries);
390 	    } else {
391 		ids = emptyObjectArray;
392 	    }
393 
394 	    long startTime = System.currentTimeMillis();
395 	    try {
396 		long duration = dgcProxy.dirty(sequenceNum, ids, leaseValue);
397 
398 		synchronized (this) {
399 		    dirtyFailures = 0;
400 
401 		    if (duration < 0) {
402 			setRenewTime(Long.MAX_VALUE);
403 			invalidRefs.addAll(refTable.values());
404 		    } else {
405 			setRenewTime(
406 			    computeRenewTime(startTime,
407 					     Math.max(duration,
408 						      minimumDuration)));
409 			expirationTime = startTime + duration;
410 		    }
411 		}
412 
413 	    } catch (NoSuchObjectException e) {
414 		synchronized (this) {
415 		    setRenewTime(Long.MAX_VALUE);
416 		    invalidRefs.addAll(refTable.values());
417 		}
418 	    } catch (Exception e) {
419 		long endTime = System.currentTimeMillis();
420 
421 		synchronized (this) {
422 		    dirtyFailures++;
423 
424 		    if (dirtyFailures == 1) {
425 			/*
426 			 * If this was the first recent failed dirty call,
427 			 * reschedule another one immediately, in case there
428 			 * was just a transient network problem, and remember
429 			 * the start time and duration of this attempt for
430 			 * future calculations of the delays between retries.
431 			 */
432 			dirtyFailureStartTime = startTime;
433 			dirtyFailureDuration = endTime - startTime;
434 			setRenewTime(endTime);
435 		    } else {
436 			/*
437 			 * For each successive failed dirty call, wait for a
438 			 * (binary) exponentially increasing delay before
439 			 * retrying, to avoid network congestion.
440 			 */
441 			int n = dirtyFailures - 2;
442 			if (n == 0) {
443 			    /*
444 			     * Calculate the initial retry delay from the
445 			     * average time elapsed for each of the first
446 			     * two failed dirty calls.  The result must be
447 			     * at least 1000ms, to prevent a tight loop.
448 			     */
449 			    dirtyFailureDuration =
450 				Math.max((dirtyFailureDuration +
451 					  (endTime - startTime)) >> 1, 1000);
452 			}
453 			long newRenewTime =
454 			    endTime + (dirtyFailureDuration << n);
455 
456 			/*
457 			 * Continue if the last known held lease has not
458 			 * expired, or else at least a fixed number of times,
459 			 * or at least until we've tried for a fixed amount
460 			 * of time (the default lease value we request).
461 			 */
462 			if (newRenewTime < expirationTime ||
463 			    dirtyFailures < dirtyFailureRetries ||
464 			    newRenewTime < dirtyFailureStartTime + leaseValue)
465 			{
466 			    setRenewTime(newRenewTime);
467 			} else {
468 			    /*
469 			     * Give up: postpone lease renewals until next
470 			     * ref is registered for this endpoint.
471 			     */
472 			    setRenewTime(Long.MAX_VALUE);
473 			}
474 		    }
475 
476 		    if (refEntries != null) {
477 			/*
478 			 * Add all of these refs to the set of refs for this
479 			 * endpoint that may be invalid (this AbstractDgcClient
480 			 * may not be in the server's referenced set), so that
481 			 * we will attempt to explicitly dirty them again in
482 			 * the future.
483 			 */
484 			invalidRefs.addAll(refEntries);
485 			
486 			/*
487 			 * Record that a dirty call has failed for all of these
488 			 * refs, so that clean calls for them in the future
489 			 * will be strong.
490 			 */
491 			Iterator iter = refEntries.iterator();
492 			while (iter.hasNext()) {
493 			    RefEntry refEntry = (RefEntry) iter.next();
494 			    refEntry.markDirtyFailed();
495 			}
496 		    }
497 
498 		    /*
499 		     * If the last known held lease will have expired before
500 		     * the next renewal, all refs might be invalid.
501 		     */
502 		    if (renewTime >= expirationTime) {
503 			invalidRefs.addAll(refTable.values());
504 		    }
505 		}
506 	    }
507 	}

509 	/**
510 	 * Sets the absolute time at which the lease for this entry should
511 	 * be renewed.
512 	 *
513 	 * This method must ONLY be invoked while synchronized on this
514 	 * EndpointEntry.
515 	 */
516 	private void setRenewTime(long newRenewTime) {
517 	    assert Thread.holdsLock(this);
518 
519 	    if (newRenewTime < renewTime) {
520 		renewTime = newRenewTime;
521 		if (interruptible) {
522 		    AccessController.doPrivileged(new PrivilegedAction() {
523 			public Object run() {
524 			    renewCleanThread.interrupt();
525 			    return null;
526 			}
527 		    });
528 		}
529 	    } else {
530 		renewTime = newRenewTime;
531 	    }
532 	}

534 	/**
535 	 * RenewCleanThread handles the asynchronous client-side DGC activity
536 	 * for this entry: renewing the leases and making clean calls.
537 	 */
538 	private class RenewCleanThread implements Runnable {
539 
540 	    public void run() {
541 		do {
542 		    long timeToWait;
543 		    RefEntry.PhantomLiveRef phantom = null;
544 		    boolean needRenewal = false;
545 		    Set refsToDirty = null;
546 		    long sequenceNum = Long.MIN_VALUE;
547 
548 		    synchronized (EndpointEntry.this) {
549 			/*
550 			 * Calculate time to block (waiting for phantom
551 			 * reference notifications).  It is the time until the
552 			 * lease renewal should be done, bounded on the low
553 			 * end by 1 ms so that the reference queue will always
554 			 * get processed, and if there are pending clean
555 			 * requests (remaining because some clean calls
556 			 * failed), bounded on the high end by the maximum
557 			 * clean call retry interval.
558 			 */
559 			long timeUntilRenew =
560 			    renewTime - System.currentTimeMillis();
561 			timeToWait = Math.max(timeUntilRenew, 1);
562 			if (pendingCleanCalls()) {
563 			    timeToWait = Math.min(timeToWait, cleanInterval);
564 			}
565 
566 			/*
567 			 * Set flag indicating that it is OK to interrupt this
568 			 * thread now, such as if a earlier lease renewal time
569 			 * is set, because we are only going to be blocking
570 			 * and can deal with interrupts.
571 			 */
572 			interruptible = true;
573 		    }
574 
575 		    try {
576 			/*
577 			 * Wait for the duration calculated above for any of
578 			 * our phantom references to be enqueued.
579 			 */
580 			phantom = (RefEntry.PhantomLiveRef)
581 			    refQueue.remove(timeToWait);
582 		    } catch (InterruptedException e) {
583 		    }
584 
585 		    synchronized (EndpointEntry.this) {
586 			/*
587 			 * Set flag indicating that it is NOT OK to interrupt
588 			 * this thread now, because we may be undertaking I/O
589 			 * operations that should not be interrupted (and we
590 			 * will not be blocking arbitrarily).
591 			 */
592 			interruptible = false;
593 			Thread.interrupted();	// clear interrupted state
594 
595 			/*
596 			 * If there was a phantom reference enqueued, process
597 			 * it and all the rest on the queue, generating
598 			 * clean requests as necessary.
599 			 */
600 			if (phantom != null) {
601 			    processPhantomRefs(phantom);
602 			}
603 
604 			/*
605 			 * Check if it is time to renew this entry's lease.
606 			 */
607 			long currentTime = System.currentTimeMillis();
608 			if (currentTime > renewTime) {
609 			    needRenewal = true;
610 			    if (currentTime >= expirationTime) {
611 				invalidRefs.addAll(refTable.values());
612 			    }
613 			    if (!invalidRefs.isEmpty()) {
614 				refsToDirty = new HashSet(invalidRefs);
615 				invalidRefs.clear();
616 			    }
617 			    sequenceNum = getNextSequenceNum();
618 			}
619 		    }
620 
621 		    if (needRenewal) {
622 			makeDirtyCall(refsToDirty, sequenceNum);
623 		    }
624 
625 		    if (pendingCleanCalls()) {
626 			makeCleanCalls();
627 		    }
628 		} while (!removed() || pendingCleanCalls());
629 	    }
630 	}
631 
632 	/**
633 	 * Processes the notification of the given phantom reference and any
634 	 * others that are on this entry's reference queue.  Each phantom
635 	 * reference is removed from its RefEntry's ref set.  All ref
636 	 * entries that have no more registered instances are collected
637 	 * into up to two batched clean call requests: one for refs
638 	 * requiring a "strong" clean call, and one for the rest.
639 	 *
640 	 * This method must ONLY be invoked while synchronized on this
641 	 * EndpointEntry.
642 	 */
643 	private void processPhantomRefs(RefEntry.PhantomLiveRef phantom) {
644 	    assert Thread.holdsLock(this);
645 
646 	    Set strongCleans = null;
647 	    Set normalCleans = null;
648 
649 	    do {
650 		RefEntry refEntry = phantom.getRefEntry();
651 		refEntry.removeInstanceFromRefSet(phantom);
652 		if (refEntry.isRefSetEmpty()) {
653 		    if (refEntry.hasDirtyFailed()) {
654 			if (strongCleans == null) {
655 			    strongCleans = new HashSet(5);
656 			}
657 			strongCleans.add(refEntry);
658 		    } else {
659 			if (normalCleans == null) {
660 			    normalCleans = new HashSet(5);
661 			}
662 			normalCleans.add(refEntry);
663 		    }
664 		    removeRefEntry(refEntry);
665 		}
666 	    } while ((phantom =
667 		(RefEntry.PhantomLiveRef) refQueue.poll()) != null);
668 
669 	    if (strongCleans != null) {
670 		pendingCleans.add(
671 		    new CleanRequest(getNextSequenceNum(),
672 				     createObjectIDArray(strongCleans),
673 				     true));
674 	    }
675 	    if (normalCleans != null) {
676 		pendingCleans.add(
677 		    new CleanRequest(getNextSequenceNum(),
678 				     createObjectIDArray(normalCleans),
679 				     false));
680 	    }
681 	}

683 	/**
684 	 * Makes all of the clean calls described by the clean requests in
685 	 * this entry's set of "pending cleans".  Clean requests for clean
686 	 * calls that succeed are removed from the "pending cleans" set.
687 	 *
688 	 * This method must NOT be invoked while synchronized on this
689 	 * EndpointEntry.
690 	 */
691 	private void makeCleanCalls() {
692 	    assert !Thread.holdsLock(this);
693 
694 	    Iterator iter = pendingCleans.iterator();
695 	    while (iter.hasNext()) {
696 		CleanRequest request = (CleanRequest) iter.next();
697 		try {
698 		    dgcProxy.clean(request.sequenceNum, request.objectIDs,
699 				  request.strong);
700 		    iter.remove();
701 		} catch (NoSuchObjectException e) {
702 		    iter.remove();
703 		} catch (Exception e) {
704 		    if (e instanceof ConnectException ||
705 			e instanceof ConnectIOException)
706 		    {
707 			/*
708 			 * If we get a ConnectException, the target DGC likely
709 			 * has gone away, in which case we shouldn't bother
710 			 * retrying this clean request forever.  Then again,
711 			 * the server could just be heavily loaded, so we will
712 			 * give a finite number of retry opportunities to
713 			 * clean requests that fail this way.
714 			 *
715 			 * A similar (but different) argument can be made for
716 			 * ConnectIOException.
717 			 */
718 			if (request.connectFailures.incrementAndGet() >= cleanConnectRetries) {
719 			    iter.remove();
720 			}
721 		    } else {
722 			// possible transient failure, retain clean request
723 		    }
724 		}
725 	    }
726 	}
727 
728 	/**
729 	 * Creates an array of object IDs (needed for the DGC remote calls)
730 	 * from the ids in the given set of refs.
731 	 */
732 	private Object[] createObjectIDArray(Set refEntries) {
733 	    Object[] ids = new Object[refEntries.size()];
734 	    Iterator iter = refEntries.iterator();
735 	    for (int i = 0; i < ids.length; i++) {
736 		ids[i] = ((RefEntry) iter.next()).getObjectID();
737 	    }
738 	    return ids;
739 	}

741 	/**
742 	 * RefEntry encapsulates the client-side DGC information specific to
743 	 * a particular object ID of an endpoint (a unique live reference
744 	 * value).
745 	 *
746 	 * In particular, it contains a set of phantom references to all of
747 	 * the live reference instances for the given object ID and endpoint
748 	 * in this VM that have been registered with this AbstractDgcClient
749 	 * (but not yet garbage collected locally).
750 	 */
751 	private class RefEntry {
752 
753 	    /**
754 	     * the object ID that this RefEntry is for (the endpoint is
755 	     * implied by the outer EndpointEntry instance)
756 	     */
757 	    private final Object objectID;
758 
759 	    /*
760 	     * mutable instance state (below) is guarded by outer
761 	     * EndpointEntry's lock
762 	     */
763 
764 	    /** set of phantom references to registered instances */
765 	    private final Set refSet = new HashSet(5);
766 	    /** true if a dirty call containing this ref has failed */
767 	    private boolean dirtyFailed = false;
768 
769 	    RefEntry(Object objectID) {
770 		this.objectID = objectID;
771 	    }
772 
773 	    /**
774 	     * Returns the object ID that this entry is for.
775 	     */
776 	    Object getObjectID() {
777 		return objectID;
778 	    }
779 
780 	    /**
781 	     * Adds a live reference to the set of registered instances for
782 	     * this entry.
783 	     *
784 	     * This method must ONLY be invoked while synchronized on this
785 	     * RefEntry's EndpointEntry.
786 	     */
787 	    void addInstanceToRefSet(Object ref) {
788 		assert Thread.holdsLock(EndpointEntry.this);
789 		assert getRefObjectID(ref).equals(objectID);
790 
791 		/*
792 		 * Only keep a phantom reference to the registered instance,
793 		 * so that it can be garbage collected normally (and we can be
794 		 * notified when that happens).
795 		 */
796 		refSet.add(new PhantomLiveRef(ref));
797 	    }
798 
799 	    /**
800 	     * Removes a PhantomLiveRef from the set of registered instances.
801 	     *
802 	     * This method must ONLY be invoked while synchronized on this
803 	     * RefEntry's EndpointEntry.
804 	     */
805 	    void removeInstanceFromRefSet(PhantomLiveRef phantom) {
806 		assert Thread.holdsLock(EndpointEntry.this);
807 		assert refSet.contains(phantom);
808 		refSet.remove(phantom);
809 	    }
810 
811 	    /**
812 	     * Returns true if there are no registered live reference
813 	     * instances for this entry still reachable in this VM.
814 	     *
815 	     * This method must ONLY be invoked while synchronized on this
816 	     * RefEntry's EndpointEntry.
817 	     */
818 	    boolean isRefSetEmpty() {
819 		assert Thread.holdsLock(EndpointEntry.this);
820 		return refSet.size() == 0;
821 	    }
822 
823 	    /**
824 	     * Records that a dirty call that explicitly contained this
825 	     * entry's ref value has failed.
826 	     *
827 	     * This method must ONLY be invoked while synchronized on this
828 	     * RefEntry's EndpointEntry.
829 	     */
830 	    void markDirtyFailed() {
831 		assert Thread.holdsLock(EndpointEntry.this);
832 		dirtyFailed = true;
833 	    }
834 
835 	    /**
836 	     * Returns true if a dirty call that explicitly contained this
837 	     * entry's ref value has failed (and therefore a clean call for
838 	     * the ref value needs to be marked "strong").
839 	     *
840 	     * This method must ONLY be invoked while synchronized on this
841 	     * RefEntry's EndpointEntry.
842 	     */
843 	    boolean hasDirtyFailed() {
844 		assert Thread.holdsLock(EndpointEntry.this);
845 		return dirtyFailed;
846 	    }
847 
848 	    /**
849 	     * PhantomLiveRef is a PhantomReference to a live reference
850 	     * instance, used to detect when the particular live reference
851 	     * becomes permanently unreachable in this VM.
852 	     */
853 	    class PhantomLiveRef extends PhantomReference {
854 	
855 		PhantomLiveRef(Object ref) {
856 		    super(ref, EndpointEntry.this.refQueue);
857 		}
858 
859 		RefEntry getRefEntry() {
860 		    return RefEntry.this;
861 		}
862 	    }
863 	}
864     }

866     /**
867      * CleanRequest holds the data for the arguments of a clean call
868      * that needs to be made.
869      */
870     private static class CleanRequest {
871 
872 	final long sequenceNum;
873 	final Object[] objectIDs;
874 	final boolean strong;
875 
876 	/** how many times this request has failed with ConnectException */
877 	final AtomicInteger connectFailures = new AtomicInteger();
878 
879 	CleanRequest(long sequenceNum, Object[] objectIDs, boolean strong) {
880 	    this.sequenceNum = sequenceNum;
881 	    this.objectIDs = objectIDs;
882 	    this.strong = strong;
883 	}
