19  package org.apache.river.jeri.internal.http;
20  
21  import java.io.EOFException;
22  import java.io.IOException;
23  import java.io.InputStream;
24  import java.net.HttpURLConnection;
25  import java.util.StringTokenizer;
26  
27  /**
28   * Class for reading HTTP messages.  Each instance reads a single HTTP message.
29   *
30   * @author Sun Microsystems, Inc.
31   * 
32   */
33  class MessageReader {
34      
35      /* state values */
36      private static final int START   = 0;
37      private static final int HEADER  = 1;
38      private static final int CONTENT = 2;
39      private static final int DONE    = 3;
40      
41      private static final byte[] sink = new byte[256];
42  
43      private final InputStream in;
44      private final boolean noContent;
45      private int state = START;
46      private StartLine sline;
47      private InputStream cin;
48  
49      /**
50       * Creates new reader on top of given input stream.  If noContent is true,
51       * incoming message is assumed to be bodiless (e.g., a HEAD response).
52       */
53      MessageReader(InputStream in, boolean noContent) {
54  	this.in = in;
55  	this.noContent = noContent;
56      }
57      
58      /**
59       * Reads in HTTP message start line.
60       */
61      StartLine readStartLine() throws IOException {
62  	updateState(START, HEADER);
63  	sline = new StartLine(in);
64  	return sline;
65      }
66      
67      /**
68       * Reads in HTTP message header.
69       */
70      Header readHeader() throws IOException {
71  	updateState(HEADER, CONTENT);
72  	Header header = new Header(in);
73  
74  	if (!noContent && contentIndicated(sline, header)) {
75  	    String clen;
76  	    if (header.containsValue("Transfer-Encoding", "chunked", true)) {
77  		cin = new ChunkedInputStream();
78  	    } else if ((clen = header.getField("Content-Length")) != null) {
79  		int len;
80  		try {
81  		    len = Integer.parseInt(clen);
82  		} catch (Exception ex) {
83  		    throw new HttpParseException("invalid content length");
84  		}
85  		if (len < 0) {
86  		    throw new HttpParseException("invalid content length");
87  		}
88  		cin = new BoundedInputStream(len);
89  	    } else if (sline.isRequest) {
90  		throw new HttpParseException("request length undeclared");
91  	    } else {
92  		cin = in;
93  	    }
94  	} else {
95  	    cin = new BoundedInputStream(0);
96  	}
97  
98  	return header;
99      }
100     
101     /**
102      * Reads message content.
103      */
104     int readContent(byte[] b, int off, int len) throws IOException {
105 	updateState(CONTENT, CONTENT);
106 	return cin.read(b, off, len);
107     }
108     
109     /**
110      * Returns count of available message content.
111      */
112     int availableContent() throws IOException {
113 	updateState(CONTENT, CONTENT);
114 	return cin.available();
115     }
116     
117     /**
118      * Reads in message trailer after consuming any unread content data.
119      * Returns null if message doesn't have a trailer.
120      */
121     Header readTrailer() throws IOException {
122 	updateState(CONTENT, CONTENT);
123 	while (cin.read(sink) != -1) {
124 	}
125 
126 	updateState(CONTENT, DONE);
127 	if (cin instanceof ChunkedInputStream) {
128 	    Header trailer = new Header(in);
129 	    return (trailer.size() > 0) ? trailer : null;
130 	} else {
131 	    return null;
132 	}
133     }
134 
135     /**
136      * Reads and returns next line from stream, or null if at end of stream.
137      * Expects ASCII lines terminated by the HTTP end-of-line sequence "\r\n".
138      */
139     static String readLine(InputStream in) throws IOException {
140 	// REMIND: use Charset?
141 	int c = in.read();
142 	if (c == -1) {
143 	    return null;
144 	}
145 	StringBuffer sbuf = new StringBuffer();
146 	do {
147 	    if (c == '\r') {
148 		if ((c = in.read()) == '\n') {
149 		    break;
150 		}
151 		sbuf.append('\r');
152 	    } else {
153 		sbuf.append((char) c);
154 		c = in.read();
155 	    }
156 	} while (c != -1);
157 	return sbuf.toString();
158     }
159 
160     private void updateState(int oldState, int newState) {
161 	if (state != oldState) {
162 	    throw new IllegalStateException();
163 	}
164 	state = newState;
165     }
166     
167     /**
168      * Returns true if given start line and header indicate a content body.
169      */
170     private static boolean contentIndicated(StartLine sline, Header header) {
171 	if (!sline.isRequest && 
172 	    ((sline.status / 100) == 1 ||
173 	     sline.status == HttpURLConnection.HTTP_NO_CONTENT ||
174 	     sline.status == HttpURLConnection.HTTP_NOT_MODIFIED))
175 	{
176 	    return false;
177 	}
178 	
179 	if (header.getField("Transfer-Encoding") != null ||
180 	    header.getField("Content-Length") != null)
181 	{
182 	    return true;
183 	}
184 
185 	return !sline.isRequest;
186     }
187     
188     /**
189      * Input stream for reading bounded content data.
190      */
191     private class BoundedInputStream extends InputStream {
192 	
193 	private int bound;
194 	
195 	BoundedInputStream(int bound) {
196 	    this.bound = bound;
197 	}
198 	
199 	public int read() throws IOException {
200 	    byte[] b = new byte[1];
201 	    return (read(b, 0, 1) != -1) ? b[0] & 0xFF : -1;
202 	}
203 	
204 	public int read(byte[] b, int off, int len) throws IOException {
205 	    if (bound == 0) {
206 		return -1;
207 	    } else {
208 		int n = in.read(b, off, Math.min(bound, len));
209 		if (n != -1) {
210 		    bound -= n;
211 		}
212 		return n;
213 	    }
214 	}
215 	
216 	public int available() throws IOException {
217 	    return Math.min(bound, in.available());
218 	}
219     }
220     
221     /**
222      * Input stream for reading chunked content data.
223      */
224     private class ChunkedInputStream extends InputStream {
225 
226 	private byte[] buf;
227 	private int pos = 0;
228 	private int lim = 0;
229 	
230 	ChunkedInputStream() {}
231 
232 	public int read() throws IOException {
233 	    byte[] b = new byte[1];
234 	    return (read(b, 0, 1) != -1) ? b[0] & 0xFF : -1;
235 	}
236 
237 	public int read(byte[] b, int off, int len) throws IOException {
238 	    while (pos >= lim) {
239 		refill();
240 	    }
241 	    if (pos < 0) {
242 		return -1;
243 	    }
244 	    int n = Math.min(lim - pos, len);
245 	    System.arraycopy(buf, pos, b, off, n);
246 	    pos += n;
247 	    return n;
248 	}
249 	
250 	public int available() throws IOException {
251 	    while (pos >= lim) {
252 		refill();
253 	    }
254 	    return (pos >= 0) ? (lim - pos) : 0;
255 	}
256 
257 	private void refill() throws IOException {
258 	    int newlim = 0;
259 	    try {
260 		String line = readLine(in);
261 		StringTokenizer tok = new StringTokenizer(line, " ;\t");
262 		newlim = Integer.parseInt(tok.nextToken(), 16);
263 	    } catch (NumberFormatException ex) {
264 		throw new HttpParseException("error parsing chunk length");
265             }
266 	    if (newlim < 0) {
267 		throw new HttpParseException("illegal chunk length");
268 	    } else if (newlim == 0) {
269 		pos = -1;
270 	    } else {
271 		// REMIND: place upper limit on chunk length?
272 		if (buf == null || newlim > buf.length) {
273 		    buf = new byte[newlim];
274 		}
275 		for (int i = 0; i < newlim; ) {
276 		    int n = in.read(buf, i, newlim - i);
277 		    if (n < 0) {
278 			throw new EOFException("incomplete chunk");
279 		    }
280 		    i += n;
281 		}
282 		String blank = readLine(in);
283 		if (blank == null || blank.length() > 0) {
284 		    throw new HttpParseException("illegal chunk tail");
285 		}
286 		pos = 0;
287 		lim = newlim;
288 	    }
289 	}
290     }
