18  package org.apache.river.tool.envcheck;
19  
20  import java.util.HashSet;
21  import java.util.ResourceBundle;
22  import java.util.StringTokenizer;
23  
24  /**
25   * A class which provides a standard way to report the results of 
26   * a test. 
27   */
28  public abstract class Reporter {
29  
30      /** the level at which information messages are generated */
31      public final static int INFO = 0;
32  
33      /** the level at which warning messages are generated */
34      public final static int WARNING = 1;
35  
36      /** the level at which error messages are generated */
37      public final static int ERROR = 2;
38  
39      /** the header label for the source record */
40      private static String sourceString;
41  
42      /** the header labels for the test result records */
43      private static String[] msgHeaders = new String[3];
44  
45      /** flag controlling display of explanation text */
46      private static boolean explain = false;
47  
48      /** the display reportingLevel */
49      private static int reportingLevel = WARNING;
50  
51      /** the warning count */
52      private static int warningCount = 0;
53  
54      /** the error count */
55      private static int errorCount = 0;
56  
57      /** flag controlling printing stack traces */
58      private static boolean traces = false;
59  
60      /** table of explanations which have been output,  to inhibit duplicates */
61      private static HashSet explanationsGiven = new HashSet();
62  
63      /** initialize the localized header labels */
64      static {
65  	ResourceBundle bundle = Util.getResourceBundle(EnvCheck.class);
66  	msgHeaders[0] = Util.getString("reporter.info", bundle);
67  	msgHeaders[1] = Util.getString("reporter.warning", bundle);
68  	msgHeaders[2] = Util.getString("reporter.error", bundle);
69  	sourceString = Util.getString("reporter.source", bundle);
70      }
71  
72      // inhibit instantiation
73      private Reporter(){
74      }
75  
76      /** 
77       * Set the reporting level. No validity checks are done.
78       *
79       * @param level the max level to generate output
80       */
81      public static void setLevel(int level) {
82          reportingLevel = level;
83      }
84  
85      /**
86       * Set the explanation generation flag;
87       *
88       * @param explain the value to set the flag to
89       */
90      public static void setExplanation(boolean explain) {
91  	Reporter.explain = explain;
92      }
93  
94      /**
95       * Set the flag controlling printing of stack traces.
96       *
97       * @param traces if <code>true</code>, print stack traces
98       */
99      public static void setPrintTraces(boolean traces) {
100 	Reporter.traces = traces;
101     }
102 
103     /**
104      * Generate the output for <code>message</code> . This method is silent
105      * if the message level is less than <code>reportLevel</code>.
106      * If <code>explain</code> is <code>true</code>, the explanation text 
107      * will be output only the first time it is encountered.  
108      *
109      * @param message the <code>Message</code> to print
110      */
111     public static void print(Message message) {
112 	print(message, null);
113     }
114     
115     /**
116      * Generate the output for <code>message.</code> This method is silent if
117      * the message level is less than <code>Reporter.level</code>.  If
118      * <code>Reporter.explain</code> is <code>true</code>, the explanation text
119      * will be output only the first time it is encountered. If
120      * <code>source</code> is <code>null</code>, the source record is not
121      * output. If the message level is error or warning, the appropriate counter
122      * is updated.
123      *
124      * @param message the <code>Message</code> to print
125      * @param source additional information identifying the component under test
126      */
127     public static void print(Message message, String source) {
128 	int messageLevel = message.getLevel();
129 	if (reportingLevel > messageLevel) {
130 	    return;
131 	}
132 	if (messageLevel == WARNING) {
133 	    warningCount++;
134 	} else if (messageLevel == ERROR) {
135 	    errorCount++;
136 	}
137 	System.out.println();
138 	System.out.println(msgHeaders[messageLevel] 
139 			   + " " 
140 			   + message.getMessage());
141 	if (source != null) {
142 	    System.out.println(sourceString + " " + source);
143 	}
144 	if (explain) {
145 	    printExplanation(message);
146 	}
147 	if (traces) {
148 	    message.printStackTrace(); // noop if no exception
149 	}
150     }
151     
152     /**
153      * Output the explanation text supplied by <code>message.</code> If there is
154      * no explanation text, or if the text has been output previously, this
155      * method returns. Otherwise the text is output. Formatting is applied if
156      * <code>message.formatExplanation()</code> returns <code>true</code>.
157      *
158      * @param message the <code>Message</code> to explain
159      */
160     private static void printExplanation(Message message) {
161 	String exp = message.getExplanation();
162 	if (exp == null) {
163 	    return;
164 	}
165 	if (explanationsGiven.contains(exp)) {
166 	    return;
167 	}
168 	explanationsGiven.add(exp);
169 	if (message.formatExplanation()) {
170 	    String indent = "    ";
171 	    int lineMax = 70;
172 	    StringBuffer buf = new StringBuffer();
173 	    StringTokenizer tok = new StringTokenizer(exp);
174 	    int lineStart = 0;
175 	    while (tok.hasMoreTokens()) {
176 		String nextWord = tok.nextToken();
177 		int lineLength = 
178 		    buf.length() + nextWord.length() + 1 - lineStart; 
179 		if (lineLength > lineMax) {
180 		    buf.append("\n");
181 		    lineStart = buf.length();
182 		}
183 		if (lineStart == buf.length()) {
184 		    buf.append(indent);
185 		} else {
186 		    buf.append(" ");
187 		}
188 		buf.append(nextWord);
189 	    }
190 	    if (buf.length() > lineStart) {
191 		buf.append("\n");
192 	    }
193 	    exp = buf.toString();
194 	    if (exp.length() > 0) {
195 		System.out.println();
196 	    }
197 	}
198 	System.out.println(exp);
199     }

201     /**
202      * Get the total number of warning records that were generated.
203      *
204      * @return the warning count
205      */
206     public static int getWarningCount() {
207 	return warningCount;
208     }
209 
210     /**
211      * Get the total number of error records that were generated.
212      *
213      * @return the error count
214      */
215     public static int getErrorCount() {
216 	return errorCount;
217     }

219     /**
220      * A container of message information to be processed by
221      * <code>Reporter</code>.
222      */
223     public static class Message {
224 
225 	/** optional exception associated with the message */
226 	private Throwable thrownException = null;
227 
228 	/** the brief message text */
229 	private String message = null;
230 
231 	/** the full explanation text */
232 	private String explanation = null;
233 
234 	/** the message level, initialized to an illegal value */
235 	private int level = -1;
236 
237 	/** flag to format the explanation */
238 	private boolean formatExplanation = true;
239 
240 	/**
241 	 * Construct a <code>Message</code> having the given level
242 	 * and text. 
243 	 *
244 	 * @param level the message level
245 	 * @param message the short message text
246 	 * @param explanation the explanation text
247 	 */
248 	public Message(int level, String message, String explanation) {
249 	    this.level = level;
250 	    this.message = message;
251 	    this.explanation = explanation;
252 	}
253 
254 	/**
255 	 * Construct a <code>Message</code> having the given level,
256 	 * text, and exception data.
257 	 *
258 	 * @param level the message level
259 	 * @param message the short message text
260 	 * @param thrownException the exception associated with the message
261 	 * @param explanation the explanation text
262 	 */
263 	public Message(int level,
264 		       String message, 
265 		       Throwable thrownException,
266 		       String explanation) 
267 	{
268 	    this.level = level;
269 	    this.message = message;
270 	    this.thrownException = thrownException;
271 	    this.explanation = explanation;
272 	}
273 
274 	/**
275 	 * Returns the message level.
276 	 * 
277 	 * @return the level
278 	 */
279 	protected int getLevel() {
280 	    if (level == -1) {
281 		throw new IllegalStateException("Bad Level");
282 	    }
283 	    return level;
284 	}
285 
286 	/**
287 	 * Returns the short message text. If a non-<code>null</code> exception
288 	 * is associated with this message, the exception message is
289 	 * appended to the short message.
290 	 *
291 	 * @return the short message
292 	 */
293 	protected String getMessage() {
294 	    String buf = "";
295 	    if (message != null) {
296 		buf =  message;
297 		if (thrownException != null) {
298 		    buf += ": ";
299 		}
300 	    }
301 	    if (thrownException != null) {
302 		buf += thrownException.getMessage();
303 	    } 
304 	    if (buf.length() == 0) {
305 		throw new IllegalStateException("No Message");
306 	    }
307 	    return buf;
308 	}
309 
310 	/**
311 	 * Returns the explanation text.
312 	 *
313 	 * @return the explanation text
314 	 */
315 	protected String getExplanation() {
316 	    return explanation;
317 	}
318 
319 	/**
320 	 * Prints a stack trace for the exception associated with this
321 	 * message. If there is no such exception this method returns silently.
322 	 */
323 	protected void printStackTrace() {
324 	    if (thrownException != null) {
325 		thrownException.printStackTrace();
326 	    }
327 	}
328 
329 	/**
330 	 * Set the flag indicating whether to format the explanation text.
331 	 *
332 	 * @param b if <code>true</code>, format the text
333 	 */
334 	protected void setFormatExplanation(boolean b) {
335 	    formatExplanation = b;
336 	}
337 
338 	/**
339 	 * Returns the flag indicating whether to format the explanation text.
340 	 *
341 	 * @return <code>true</code> if the text is to be formatted.
342 	 */
343 	protected boolean formatExplanation() {
344 	    return formatExplanation;
345 	}
346     }
