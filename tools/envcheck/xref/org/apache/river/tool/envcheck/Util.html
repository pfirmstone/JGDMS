58      /**
59       * Validate the accessibility of the non-directory file identified by
60       * <code>name</code>. The file must exist, must not be a directory, and must
61       * be readable.  If any check fails, an error message is returned. If all
62       * checks are successful, <code>null</code> is returned.
63       *
64       * @param name the file name
65       * @param desc a short descriptive string which describes the file, suitable
66       *             for inclusion in the error message
67       * @return and error message, or <code>null</code> if all checks pass
68       */
69      public static String checkFileName(String name, String desc) {
70  	File file = new File(name);
71  	String canonicalPath = null;
72  	try {
73  	    canonicalPath = file.getCanonicalPath();
74  	} catch (Exception e) {
75  	    return getString("util.cantresolve", bundle, desc, name);
76  	}
77  	if (!file.exists()) {
78  	    return getString("util.notexist", bundle, desc, canonicalPath);
79  	}
80  	if (file.isDirectory()) {
81  	    return getString("util.isdir", bundle, desc, canonicalPath);
82  	}
83  	if (!file.canRead()) {
84  	    return getString("util.cantread", bundle, desc, canonicalPath);
85  	}
86  	return null;
87      }

89      /**
90       * Check the accessibility of the given <code>URL</code>. If the
91       * url is a <code>file:</code> url, the usual file access checks are
92       * performed. For any other url a stream is opened, and a non-exceptional
93       * return is considered a success.
94       *
95       * @param url the <code>URL</code> to check
96       * @param desc a description of the source of the url
97       * @return an error message, or <code>null</code> if the access check passes
98       */
99      public static String checkURL(URL url, String desc) {
100         if ("file".equals(url.getProtocol())) {
101             String path = url.getFile().replace('/', File.separatorChar);
102             return checkFileName(path, desc);
103         } else {
104 	    try {
105 		url.openStream().close();
106 		return null;
107 	    } catch (IOException e) {
108 		return getString("util.cantreadURL", 
109 				 bundle, 
110 				 url.toString(),
111 				 desc);
112 	    }
113         }
114     }

116     /**
117      * Validate the accessibility of the non-directory file identified by the
118      * system property <code>prop</code>. The system property must have a
119      * non-<code>null</code> value. The file identified by the value must exist,
120      * must not be a directory, and must be readable.  If any check fails, an
121      * error message is returned. If all checks are successful,
122      * <code>null</code> is returned.
123      *
124      * @param prop name of a system property whose value must be a file name
125      * @param desc a short descriptive string which describes the file, suitable
126      *             for inclusion in the error message
127      * @return an error message, or <code>null</code> if all checks pass
128      */
129     public static String checkSystemPropertyFile(String prop, String desc) {
130 	String name = System.getProperty(prop);
131 	if (name == null) {
132 	    return getString("util.undef", bundle, desc);
133 	}
134 	return checkFileName(name, desc);
135     }

137     /**
138      * Get the resource bundle associated with class <code>clazz</code>.  The
139      * resource bundle name is constructed by the class name to lower case and
140      * inserting <code>.resources</code> in front of the name. Thus, if
141      * <code>clazz.getName()</code> returned <code>a.b.Foo</code> then the
142      * associated resource bundle name would be <code>a.b.resources.foo</code>.
143      * If no resource bundle having the associated name is found, a stack trace
144      * is printed and <code>null</code> is returned. The resource bundle is
145      * loaded using the class loader for the given class, and is cached so that
146      * the search is performed only once.
147      * 
148      * @param clazz the class for which to obtain a resource bundle
149      * @return the resource bundle
150      */
151     public static ResourceBundle getResourceBundle(Class clazz) {
152 	if (resourceMap.containsKey(clazz)) {
153 	    return (ResourceBundle) resourceMap.get(clazz);
154 	}
155 	String className = clazz.getName();
156 	int lastDot = className.lastIndexOf(".");
157 	if (lastDot < 0) {
158 	    throw new IllegalStateException("Class is in default package");
159 	}
160 	String pkgName = className.substring(0, lastDot);
161 	// includes leading '.'
162 	String resourceName = className.substring(lastDot).toLowerCase();
163 	String bundleName = pkgName + ".resources" + resourceName;
164 	ResourceBundle bundle = null;
165 	try {
166 	    bundle = ResourceBundle.getBundle(bundleName, 
167 					      Locale.getDefault(),
168 					      clazz.getClassLoader());
169 	} catch (MissingResourceException e) {
170 	    e.printStackTrace(); //XXX should failures be silent?
171 	}
172 	resourceMap.put(clazz, bundle);
173 	return bundle;
174     }

176     /**
177      * Get the format string associated with <code>key</code> in the
178      * given resource bundle. If <code>key</code> is not present in
179      * the bundle, a default format string is returned which will
180      * identify the missing key when printed.
181      *
182      * @param key the key of the format string to retrieve
183      * @param bundle the bundle to retrieve the format string from
184      * @return the format string.
185      */
186     private static String getFormat(String key, ResourceBundle bundle) {
187 	String fmt = "no text found: \"" + key + "\" {0}";
188 	if (bundle != null) {
189 	    try {
190 		fmt = bundle.getString(key);
191 	    } catch (MissingResourceException e) {
192 	    }
193 	}
194 	return fmt;
195     }
