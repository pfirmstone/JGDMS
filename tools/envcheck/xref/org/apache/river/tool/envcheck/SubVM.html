18  package org.apache.river.tool.envcheck;
19  
20  import java.io.ObjectOutputStream;
21  import java.io.PrintStream;
22  
23  /**
24   * A container for a subtask which runs in a separate and returns a
25   * single object by writing that object to <code>System.out</code>.
26   */
27  public class SubVM {
28  
29      private static PrintStream origOut;
30  
31      /**
32       * Entry point for the subtask. The first element of <code>args</code> must
33       * name a <code>SubVMTask</code> to run; the remaining elements comprise the
34       * arguments for the subtask.  That task is instantiated, and its
35       * <code>run(String[] args)</code> method called passing an array containing
36       * only the subtask arguments. The object returned by the <code>run</code>
37       * method is written to <code>System.out</code>. If the <code>run</code>
38       * method throws an exception, then that exception is written to
39       * <code>System.out</code> instead.
40       *
41       * @param args the command line arguments
42       */
43      public static void main(String[] args) {
44          origOut = System.out;
45          System.setOut(System.err);
46  	try {
47  	    if (args.length == 0) {
48  		String msg = "Missing SubVMTask to instantiate in arg 0";
49  		writeResponse(new IllegalStateException(msg));
50  		System.exit(1);
51  	    }
52  	    Class subVMTaskClass = Class.forName(args[0]);
53  	    SubVMTask task = (SubVMTask) subVMTaskClass.newInstance();
54  	    String[] newArgs = new String[args.length - 1];
55  	    System.arraycopy(args, 1, newArgs, 0, newArgs.length);
56  	    writeResponse(task.run(newArgs));
57  	    System.exit(0);
58  	} catch (Throwable t) {
59  	    try {
60  		writeResponse(t);
61  		
62  	    } catch (Throwable fatal) {
63  		fatal.printStackTrace();
64  	    }
65  	    System.exit(1);
66  	}
67      }
68  
69      /**
70       * Write the serialized form of the given object to <code>System.out</code>.
71       *
72       * @param obj the object to write
73       */
74      private static void writeResponse(Object obj) throws Throwable {
75  	ObjectOutputStream os = new ObjectOutputStream(origOut);
76  	os.writeObject(obj);
77  	os.close();
78      }
79  }
