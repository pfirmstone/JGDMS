55  /**
56   * A simple HTTP server, for serving up JAR and class files.
57   * <p>
58   * The following items are discussed below:
59   * <ul>
60   * <li>{@linkplain #main Command line options}

65   * <h3><a name="logging">Logging</a></h3>
66   * <p>
67   *
68   * This implementation uses the {@link Logger} named
69   * <code>org.apache.river.tool.ClassServer</code> to log information at the
70   * following logging levels:
71   * <p>
72   * <table border="1" cellpadding="5"
73   *         summary="Describes logging performed by ClassServer at different
74   *	   logging levels">
75   * <caption><b><code>
76   *    org.apache.river.tool.ClassServer</code></b></caption>
77   *
78   * <tr> <th scope="col">Level</th> <th scope="col">Description</th> </tr>
79   * <tr>
80   *   <td>{@link Level#SEVERE SEVERE}</td>
81   *   <td>failure to accept an incoming connection</td>
82   * </tr>
83   * <tr>
84   *   <td>{@link Level#WARNING WARNING}</td>
85   *   <td>failure to read the contents of a requested file,
86   *       failure to find the message resource bundle, failure while
87   *       executing the <code>-stop</code> option

112  * <p>
113  * <h3><a name="running">Examples for running ClassServer</a></h3>
114  * <p>
115  *
116  * This server can be run directly from the 
117  * {@linkplain #main command line}
118  * or as a nonactivatable service under the 
119  * <code> org.apache.river.start.ServiceStarter</code>.
120  * <p>
121  * An example of running directly from the command line is:
122  * <blockquote><pre>
123  * % java -jar <var><b>install_dir</b></var>/lib/classserver.jar \
124  *        -port 8081 -dir <var><b>install_dir</b></var>/lib-dl -verbose
125  * </pre></blockquote>
126  * where <var><b>install_dir</b></var>
127  * is the directory where the JGDMS release is installed.
128  * This command places the class server on the (non-default) port 
129  * 8081, which  serves out the files under the (non-default) directory  
130  * <var><b>install_dir</b></var>/lib-dl. The <code>-verbose</code> option
131  * also causes download attempts to be logged.
132  * <p>
133  * An example of running under the Service Starter is:
134  * <blockquote><pre>
135  * % java -Djava.security.policy=<var><b>start_policy</b></var> \
136  *        -jar <var><b>install_dir</b></var>/lib/start.jar \
137  *        <a href="#config">httpd.config</a>
138  * </pre></blockquote>
139  * <p>
140  * where <var><b>start_policy</b></var> is the name of a security
141  * policy file (not provided), and <code>httpd.config</code> is the
142  * following configuration file:
143  * <a name="config"></a>
144  * <blockquote><pre>
145  * import org.apache.river.start.NonActivatableServiceDescriptor;
146  * import org.apache.river.start.ServiceDescriptor;
147  * 
148  * org.apache.river.start {
149  * 
150  *   serviceDescriptors = new ServiceDescriptor[]{
151  *     new NonActivatableServiceDescriptor(
152  *       "",
153  *       "<var><b>httpd_policy</b></var>",
154  *       "<var><b>install_dir</b></var>/lib/classserver.jar",
155  *       "org.apache.river.tool.ClassServer",
156  *       new String[]{"-port", "8081", "-dir", "<var><b>install_dir</b></var>/lib-dl", "-verbose"})
157  *     };
158  * }
159  * </pre></blockquote>
160  * where <var><b>httpd_policy</b></var> is the name of a security
161  * policy file (not provided).
162  *
163  * @author Sun Microsystems, Inc.
164  *
165  */

239     private static class Initializer {
240         int port;
241         String dirlist;
242         boolean trees;
243         boolean verbose;
244         boolean stoppable;
245         LifeCycle lifeCycle;
246         
247         Initializer(LifeCycle lifeCycle, String[] args){
248             port = DEFAULT_PORT;
249             dirlist = DEFAULT_DIR;
250             if (File.separatorChar == '\\') dirlist = DEFAULT_WIN_DIR;
251             trees = false;
252             verbose = false;
253             stoppable = false;
254             for (int i = 0; i < args.length ; i++ ) {
255                 String arg = args[i];
256                 if (arg.equals("-port")) {
257                     i++;
258                     port = Integer.parseInt(args[i]);
259                 } else if (arg.equals("-dir") || arg.equals("-dirs")) {
260                     i++;
261                     dirlist = args[i];
262                 } else if (arg.equals("-verbose")) {
263                     verbose = true;
264                 } else if (arg.equals("-trees")) {
265                     trees = true;
266                 } else if (arg.equals("-stoppable")) {
267                     stoppable = true;
268                 } else {
269                     throw new IllegalArgumentException(arg);
270                 }
271             }
272         }
273     }

279     /**
280      * Do the real work of the constructor.
281      */
282     private ClassServer(int port,
283 		      String dirlist,
284 		      boolean trees,
285 		      boolean verbose,
286 		      boolean stoppable,
287 		      LifeCycle lifeCycle)
288 	throws IOException
289     {
290 	StringTokenizer st = new StringTokenizer(dirlist, File.pathSeparator);
291 	dirs = new String[st.countTokens()];
292 	perms = new FilePermission[dirs.length];
293 	for (int i = 0; st.hasMoreTokens(); i++) {
294 	    String dir = st.nextToken();
295 	    if (!dir.endsWith(File.separator))
296 		dir = dir + File.separatorChar;
297 	    dirs[i] = dir;
298 	    perms[i] = new FilePermission(dir + '-', "read");
299 	}
300 	this.verbose = verbose;
301 	this.stoppable = stoppable;
302 	this.lifeCycle = lifeCycle;
303         server = new ServerSocket();
304         server.setReuseAddress(true);
305         server.setSoTimeout(300000); // 5 minutes
306         try {
307             server.bind(new InetSocketAddress(port));
308         } catch( BindException be ) {
309             throw new IOException( "failure to bind to port: "+port, be );
310         }
311 	if (!trees) {
312             map = null;
313 	    return;
314         }
315 	map = new HashMap<String, JarFile[]>();
316 	Map<String,JarFile> jfmap = new HashMap<String,JarFile>();
317 	for (int i = 0; i < dirs.length; i++) {
318 	    String[] files = new File(dirs[i]).list();
319 	    if (files == null)
320 		continue;
321 	    for (int j = 0; j < files.length; j++) {
322 		String jar = files[j];
323 		if (!jar.endsWith(".jar") && !jar.endsWith(".zip"))
324 		    continue;
325 		String name = jar.substring(0, jar.length() - 4);
326 		if (map.containsKey(name))
327 		    continue;
328 		List<JarFile> jflist = new ArrayList<JarFile>(1);
329 		addJar(jar, jflist, jfmap);
330 		map.put(name, jflist.toArray(new JarFile[jflist.size()]));
331 	    }
332 	}
333     }

357     /** Add transitive Class-Path JARs to jflist. */
358     private void addJar(String jar, List<JarFile> jflist, Map<String,JarFile> jfmap)
359 	throws IOException
360     {
361 	JarFile jf = jfmap.get(jar);
362 	if (jf != null) {
363 	    if (jflist.contains(jf)) {
364 		return;
365 	    }
366 	} else {
367 	    for (int i = 0; i < dirs.length; i++) {
368 		File f = new File(dirs[i] + jar).getCanonicalFile();
369 		if (f.exists()) {
370 		    jf = new JarFile(f);
371 		    jfmap.put(jar, jf);
372 		    if (verbose)
373 			print("classserver.jar", f.getPath());
374 		    logger.config(f.getPath());
375 		    break;
376 		}
377 	    }
378 	    if (jf == null) {
379 		if (verbose)
380 		    print("classserver.notfound", jar);
381 		logger.log(Level.CONFIG, "{0} not found", jar);
382 		return;
383 	    }
384 	}
385 	jflist.add(jf);
386 	Manifest man = jf.getManifest();
387 	if (man == null)
388 	    return;
389 	Attributes attrs = man.getMainAttributes();
390 	if (attrs == null)
391 	    return;
392 	String val = attrs.getValue(Attributes.Name.CLASS_PATH);
393 	if (val == null)
394 	    return;
395 	for (StringTokenizer st = new StringTokenizer(val);
396 	     st.hasMoreTokens(); )
397 	{
398 	    String elt = st.nextToken();
399 	    String path = decode(elt);
400 	    if (path == null) {
401 		if (verbose)
402 		    print("classserver.notfound", elt);
403 		logger.log(Level.CONFIG, "{0} not found", elt);
404 	    }
405 	    if ('/' != File.separatorChar) {
406 		path = path.replace('/', File.separatorChar);
407 	    }
408 	    addJar(path, jflist, jfmap);
409 	}
410     }

412     /** Just keep looping, spawning a new thread for each incoming request. */
413     public void run() {
414 	logger.log(Level.INFO, "ClassServer started [{0}, port {1}]",
415 		   new Object[]{Arrays.asList(dirs),
416 				Integer.toString(getPort())});
417 	try {
418 	    while (!isInterrupted()) {
419                 try {
420                     new Task(server.accept()).start();
421                 } catch (SocketTimeoutException e){
422                     // This happens every 5 minutes, it allows ClassServer to
423                     // be interrupted if necessary.
424                 } catch ( SecurityException e){
425                     logger.log(Level.SEVERE, "Permission denied: ", e);
426                     interrupt();
427                 }
428 	    }
429 	} catch (IOException e) {
430             if (verbose) {
431                 e.printStackTrace();
432             }
433             if (!server.isClosed())
434                 logger.log(Level.SEVERE, "accepting connection", e);
435         } finally {
436             terminate();
437         }
438     }

440     /** Close the server socket, causing the thread to terminate. */
441     public synchronized void terminate() {
442 	verbose = false;
443 	try {
444 	    server.close();
445 	} catch (IOException e) {
446 	}
447 	if (lifeCycle != null)
448 	    lifeCycle.unregister(this);
449 	logger.log(Level.INFO, "ClassServer terminated [port {0}]",
450 		   Integer.toString(getPort()));
451     }

458     /** Read up to CRLF, return false if EOF */
459     private static boolean readLine(InputStream in, StringBuffer buf)
460 	throws IOException
461     {
462 	while (true) {
463 	    int c = in.read();
464 	    if (c < 0)
465 		return buf.length() > 0;
466 	    if (c == '\r') {
467 		in.mark(1);
468 		c = in.read();
469 		if (c != '\n')
470 		    in.reset();
471 		return true;
472 	    }
473 	    if (c == '\n')
474 		return true;
475 	    buf.append((char) c);
476 	}
477     }

479     /** Parse % HEX HEX from s starting at i */
480     private static char decode(String s, int i) {
481 	return (char) Integer.parseInt(s.substring(i + 1, i + 3), 16);
482     }
483 
484     /** Decode escape sequences */
485     private static String decode(String path) {
486 	try {
487 	    for (int i = path.indexOf('%');
488 		 i >= 0;
489 		 i = path.indexOf('%', i + 1))
490 	    {
491 		char c = decode(path, i);
492 		int n = 3;
493 		if ((c & 0x80) != 0) {
494 		    switch (c >> 4) {
495 		    case 0xC:
496 		    case 0xD:
497 			n = 6;
498 			c = (char)(((c & 0x1F) << 6) |
499 				   (decode(path, i + 3) & 0x3F));
500 			break;
501 		    case 0xE:
502 			n = 9;
503 			c = (char)(((c & 0x0f) << 12) |
504 				   ((decode(path, i + 3) & 0x3F) << 6) |
505 				   (decode(path, i + 6) & 0x3F));
506 			break;
507 		    default:
508 			return null;
509 		    }
510 		}
511 		path = path.substring(0, i) + c + path.substring(i + n);
512 	    }
513 	} catch (Exception e) {
514 	    return null;
515 	}
516 	return path;
517     }

519     /** Read the request/response and return the initial line. */
520     private static String getInput(Socket sock, boolean isRequest)
521 	throws IOException
522     {
523 	BufferedInputStream in =
524 	    new BufferedInputStream(sock.getInputStream(), 256);
525 	StringBuffer buf = new StringBuffer(80);
526 	do {
527 	    if (!readLine(in, buf))
528 		return null;
529 	} while (isRequest && buf.length() == 0);
530 	String initial = buf.toString();
531 	do {
532 	    buf.setLength(0);
533 	} while (readLine(in, buf) && buf.length() > 0);
534 	return initial;
535     }

537     /** Simple daemon task thread */
538     private class Task extends Thread {
539 	/** Socket for the incoming request */
540 	private Socket sock;
541 
542 	/** Simple constructor */
543 	public Task(Socket sock) {
544 	    this.sock = sock;
545 	    setDaemon(true);
546 	}

548 	/** Read specified number of bytes and always close the stream. */
549 	private byte[] getBytes(InputStream in, long length)
550 	    throws IOException
551 	{
552 	    DataInputStream din = new DataInputStream(in);
553 	    byte[] bytes = new byte[(int)length];
554 	    try {
555 		din.readFully(bytes);
556 	    } finally {
557 		din.close();
558 	    }
559 	    return bytes;
560 	}

562 	/** Canonicalize the path */
563 	private String canon(String path) {
564 	    if (path.regionMatches(true, 0, "http://", 0, 7)) {
565 		int i = path.indexOf('/', 7);
566 		if (i < 0)
567 		    path = "/";
568 		else
569 		    path = path.substring(i);
570 	    }
571 	    path = decode(path);
572 	    if (path == null || path.length() == 0 || path.charAt(0) != '/')
573 		return null;
574 	    return path.substring(1);
575 	}
576 
577 	/** Return the bytes of the requested file, or null if not found. */
578 	private byte[] getBytes(String path) throws IOException {
579 	    if (map != null) {
580 		int i = path.indexOf('/');
581 		if (i > 0) {
582 		    JarFile[] jfs = (JarFile[])map.get(path.substring(0, i));
583 		    if (jfs != null) {
584 			String jpath = path.substring(i + 1);
585 			for (i = 0; i < jfs.length; i++) {
586 			    JarEntry je = jfs[i].getJarEntry(jpath);
587 			    if (je != null)
588 				return getBytes(jfs[i].getInputStream(je),
589 						je.getSize());
590 			}
591 		    }
592 		}
593 	    }
594 	    if ('/' != File.separatorChar) {
595 		path = path.replace('/', File.separatorChar);
596 	    }
597 	    for (int i = 0; i < dirs.length; i++) {
598 		File f = new File(dirs[i] + path);
599 		if (perms[i].implies(new FilePermission(f.getPath(), "read")))
600 		{
601 		    try {
602 			return getBytes(new FileInputStream(f), f.length());
603 		    } catch (FileNotFoundException e) {
604 		    }
605 		}
606 	    }
607 	    return null;
608 	}

610 	/** Process the request */
611 	public void run() {
612 	    try {
613 		DataOutputStream out =
614 		    new DataOutputStream(sock.getOutputStream());
615 		String req;
616 		try {
617 		    req = getInput(sock, true);
618 		} catch (Exception e) {
619 		    if (verbose) {
620 			print("classserver.inputerror",
621 			      new String[]{sock.getInetAddress().getHostName(),
622 					   Integer.toString(sock.getPort())});
623 			e.printStackTrace();
624 		    }
625 		    logger.log(Levels.HANDLED, "reading request", e);
626 		    return;
627 		}
628 		if (req == null)
629 		    return;
630 		if (req.startsWith("SHUTDOWN *")) {
631 		    if (verbose)
632 			print("classserver.shutdown",
633 			      new String[]{sock.getInetAddress().getHostName(),
634 					   Integer.toString(sock.getPort())});
635 		    boolean ok = stoppable;
636 		    try {
637 			new ServerSocket(0, 1, sock.getInetAddress());
638 		    } catch (IOException e) {
639 			ok = false;
640 		    }
641 		    if (!ok) {
642 			out.writeBytes("HTTP/1.0 403 Forbidden\r\n\r\n");
643 			out.flush();
644 			return;
645 		    }
646 		    try {
647 			out.writeBytes("HTTP/1.0 200 OK\r\n\r\n");
648 			out.flush();
649 		    } catch (Exception e) {
650 			if (verbose)
651 			    e.printStackTrace();
652 			logger.log(Levels.HANDLED, "writing response", e);
653 		    }
654 		    terminate();
655 		    return;
656 		}
657 		String[] args = null;
658 		if (verbose || logger.isLoggable(Level.FINE))
659 		    args = new String[]{req,
660 					sock.getInetAddress().getHostName(),
661 					Integer.toString(sock.getPort())};
662 		boolean get = req.startsWith("GET ");
663 		if (!get && !req.startsWith("HEAD ")) {
664 		    if (verbose)
665 			print("classserver.badrequest", args);
666 		    logger.log(Level.FINE,
667 			       "bad request \"{0}\" from {1}:{2}", args);
668 		    out.writeBytes("HTTP/1.0 400 Bad Request\r\n\r\n");
669 		    out.flush();
670 		    return;
671 		}
672 		String path = req.substring(get ? 4 : 5);
673 		int i = path.indexOf(' ');
674 		if (i > 0)
675 		    path = path.substring(0, i);
676 		path = canon(path);
677 		if (path == null) {
678 		    if (verbose)
679 			print("classserver.badrequest", args);
680 		    logger.log(Level.FINE,
681 			       "bad request \"{0}\" from {1}:{2}", args);
682 		    out.writeBytes("HTTP/1.0 400 Bad Request\r\n\r\n");
683 		    out.flush();
684 		    return;
685 		}
686 		if (args != null)
687 		    args[0] = path;
688 		if (verbose) {
689 		    print(get ? "classserver.request" : "classserver.probe",
690 			  args);
691 		}
692 		logger.log(Level.FINER,
693 			   get ?
694 			   "{0} requested from {1}:{2}" :
695 			   "{0} probed from {1}:{2}",
696 			   args);
697 		byte[] bytes;
698 		try {
699 		    bytes = getBytes(path);
700 		} catch (Exception e) {
701 		    if (verbose)
702 			e.printStackTrace();
703 		    logger.log(Level.WARNING, "getting bytes", e);
704 		    out.writeBytes("HTTP/1.0 500 Internal Error\r\n\r\n");
705 		    out.flush();
706 		    return;
707 		}
708 		if (bytes == null) {
709 		    if (verbose)
710 			print("classserver.notfound", path);
711 		    logger.log(Level.FINE, "{0} not found", path);
712 		    out.writeBytes("HTTP/1.0 404 Not Found\r\n\r\n");
713 		    out.flush();
714 		    return;
715 		}
716 		out.writeBytes("HTTP/1.0 200 OK\r\n");
717 		out.writeBytes("Content-Length: " + bytes.length + "\r\n");
718 		out.writeBytes("Content-Type: application/java\r\n\r\n");
719 		if (get)
720 		    out.write(bytes);
721 		out.flush();
722 		if (get)
723 		    fileDownloaded(path, sock.getInetAddress());
724 	    } catch (Exception e) {
725 		if (verbose)
726 		    e.printStackTrace();
727 		logger.log(Levels.HANDLED, "writing response", e);
728 	    } finally {
729 		try {
730 		    sock.close();
731 		} catch (IOException e) {
732 		}
733 	    }
734 	}
735     }

782     /**
783      * Command line interface for creating an HTTP server.
784      * The command line options are:
785      * <pre>
786      * [-port <var>port</var>] [-dir <var>dirlist</var>] [-dirs <var>dirlist</var>] [-stoppable] [-verbose] [-trees]
787      * </pre>
788      * The default port is 8080; the default can be overridden with
789      * the <code>-port</code> option.  The default directory on Windows is
790      * <code>J:</code> and the default on other systems is
791      * <code>/vob/jive/lib-dl</code>; the default can be overridden with the
792      * <code>-dir</code> or <code>-dirs</code> option, providing one or more
793      * directories separated by the {@linkplain File#pathSeparatorChar
794      * path-separator character}. All files under these directories (including
795      * all subdirectories) are served up via HTTP.  If the pathname of a file
796      * is <var>path</var> relative to one of the top-level directories, then
797      * the file can be downloaded using the URL
798      * <pre>
799      * http://<var>host</var>:<var>port</var>/<var>path</var>;
800      * </pre>
801      * If a relative <var>path</var> matches a file under more than one
802      * top-level directory, the file under the first top-level directory
803      * with a match is used. No caching of directory contents or file contents
804      * is performed. <p>
805      *
806      * If the <code>-stoppable</code> option is given, the HTTP server can be
807      * shut down with a custom HTTP <code>SHUTDOWN</code> request originating
808      * from the local host. The command line options for stopping an existing
809      * HTTP server are:
810      * <pre>
811      * [-port <var>port</var>] -stop
812      * </pre>
813      * <p>
814      *
815      * If the <code>-verbose</code> option is given, then all attempts to
816      * download files are output. <p>
817      * 
818      * The <code>-trees</code> option can be used to serve up individual files
819      * stored within JAR and zip files in addition to the files that are
820      * served up as described above. If the option is used, the server finds
821      * all JAR and zip files in the top-level directories (not in
822      * subdirectories).  If the name of the JAR or zip file is
823      * <var>name</var><code>.jar</code> or <var>name</var><code>.zip</code>,
824      * then any individual file named <var>file</var> within it (or within the
825      * JAR or zip files referenced transitively in <code>Class-Path</code>
826      * manifest attributes), can be downloaded using a URL of the form:
827      * <pre>
828      * http://<var>host</var>:<var>port</var>/<var>name</var>/<var>file</var>;
829      * </pre>
830      * If multiple top-level directories have JAR or zip files with the same
831      * <var>name</var>, the file under the first top-level directory with a
832      * match is used. If a <code>Class-Path</code> element matches a file under
833      * more than one top-level directory, the file under the first top-level
834      * directory with a match is used. When this option is used, an open file
835      * descriptor and cached information is held for each JAR or zip file, for
836      * the life of the process.
837      */
838     public static void main(String[] args) {
839 	int port = DEFAULT_PORT;
840 	String dirlist = DEFAULT_DIR;
841 	if (File.separatorChar == '\\')
842 	    dirlist = DEFAULT_WIN_DIR;
843 	boolean trees = false;
844 	boolean verbose = false;
845 	boolean stoppable = false;
846 	boolean stop = false;
847 	for (int i = 0; i < args.length ; i++ ) {
848 	    String arg = args[i];
849 	    if (arg.equals("-port")) {
850 		i++;
851 		port = Integer.parseInt(args[i]);
852 	    } else if (arg.equals("-dir") || arg.equals("-dirs")) {
853 		i++;
854 		dirlist = args[i];
855 	    } else if (arg.equals("-verbose")) {
856 		verbose = true;
857 	    } else if (arg.equals("-trees")) {
858 		trees = true;
859 	    } else if (arg.equals("-stoppable")) {
860 		stoppable = true;
861 	    } else if (arg.equals("-stop")) {
862 		stop = true;
863 	    } else {
864 		print("classserver.usage", (String)null);
865 		return;
866 	    }
867 	}
868 	try {
869 	    if (stop) {
870 		Socket sock = new Socket(LocalHostLookup.getLocalHost(), port);
871 		try {
872 		    DataOutputStream out =
873 			new DataOutputStream(sock.getOutputStream());
874 		    out.writeBytes("SHUTDOWN *\r\n\r\n");
875 		    out.flush();
876 		    String status = getInput(sock, false);
877 		    if (status != null && status.startsWith("HTTP/")) {
878 			status = status.substring(status.indexOf(' ') + 1);
879 			if (status.startsWith("403 ")) {
880 			    print("classserver.forbidden", status);
881 			} else if (!status.startsWith("200 ") &&
882 				   status.indexOf(' ') == 3)
883 			{
884 			    print("classserver.status",
885 				  new String[]{status.substring(0, 3),
886 					       status.substring(4)});
887 			}
888 		    }
889 		} finally {
890 		    try {
891 			sock.close();
892 		    } catch (IOException e) {
893 		    }
894 		}
895 	    } else {
896 		new ClassServer(port, dirlist, trees, verbose,
897 				stoppable).start();
898 	    }
899 	} catch (IOException e) {
900 	    logger.log(Level.WARNING, "requesting shutdown", e);
901 	}
902     }
