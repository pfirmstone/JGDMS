20  import java.io.File;
21  import java.io.Serializable;
22  import java.lang.reflect.Field;
23  import java.lang.reflect.Modifier;
24  import java.util.ResourceBundle;
25  import java.util.MissingResourceException;
26  import java.text.MessageFormat;
27  
28  
29  /**
30   * Tool to check for serializable classes that do not have explicit
31   * <code>serialVersionUID</code> fields.
32   *
33   * @author Sun Microsystems, Inc.
34   */
35  public class CheckSer {
36  
37      private static final int MASK = Modifier.STATIC | Modifier.FINAL;
38  
39      /**
40       * Checks class file directory hierarchies for serializable classes
41       * that do not have explicit <code>serialVersionUID</code> fields,
42       * and prints the names of such classes to the standard output stream.
43       * The only options are zero or more filenames that specify the roots
44       * of directory hierarchies; if no filenames are specified, the single
45       * root <code>/vob/jive/classes</code> is used. In those hierarchies,
46       * each file with a name ending in the suffix <code>.class</code> is
47       * treated as a class file; the corresponding class name is obtained
48       * from the filename by stripping off both the original prefix root
49       * filename and the <code>.class</code> suffix, and replacing each file
50       * separator character with a period (<code>.</code>). Each such class
51       * is loaded from the class loader of this tool. If the class is not an
52       * interface, directly or indirectly implements {@link Serializable},
53       * and does not have a declared <code>static</code> <code>final</code>
54       * field named <code>serialVersionUID</code>, then the name of the class
55       * is printed to the standard output stream.
56       *
57       * @param args the roots of directory hierarchies
58       */
59      public static void main(String[] args) {
60  	if (args.length == 0)
61  	    args = new String[]{"/vob/jive/classes"};
62  	for (int i = 0; i < args.length; i++) {
63  	    check(args[i], args[i].length() + 1);
64  	}
65      }
66  
67      /**
68       * Checks the class file directory hierarchy starting from the specified
69       * directory. In the hierarchy, each file with a name ending in the
70       * suffix <code>.class</code> is treated as a class file; the
71       * corresponding class name is obtained from the filename by stripping
72       * off the first <code>strip</code> characters of prefix and the
73       * <code>.class</code> suffix, and replacing each file separator
74       * character with a period (<code>.</code>). Each such class is loaded
75       * from the class loader of this tool. If the class is not an interface,
76       * directly or indirectly implements {@link Serializable}, and does not
77       * have a declared <code>static</code> <code>final</code> field named
78       * <code>serialVersionUID</code>, then the name of the class is printed
79       * to the standard output stream.
80       *
81       * @param dir directory hierarchy root
82       * @param strip number of characters of prefix to strip from each
83       * class file name
84       */
85      public static void check(String dir, int strip) {
86  	String[] files = new File(dir).list();
87  	if (files == null) {
88  	    print("checkser.nodir", dir);
89  	    return;
90  	}
91  	for (int i = 0; i < files.length; i++) {
92  	    String file = dir + File.separatorChar + files[i];
93  	    if (file.endsWith(".class"))
94  		checkClass(file, strip);
95  	    else if (new File(file).isDirectory())
96  		check(file, strip);
97  	}
98      }
99  
100     private static void checkClass(String file, int strip) {
101 	file = file.substring(strip, file.length() - 6);
102 	file = file.replace(File.separatorChar, '.');
103 	try {
104 	    Class c = Class.forName(file);
105 	    if (c.isInterface() || !Serializable.class.isAssignableFrom(c))
106 		return;
107 	    Field f = c.getDeclaredField("serialVersionUID");
108 	    if ((f.getModifiers() & MASK) != MASK)
109 		System.out.println(file);
110 	} catch (ClassNotFoundException e) {
111 	    print("checkser.failed", file);
112 	} catch (NoClassDefFoundError e) {
113 	    print("checkser.failed", file);
114 	} catch (NoSuchFieldException e) {
115 	    System.out.println(file);
116 	}
117     }
118 
119     private static ResourceBundle resources;
120     private static boolean resinit = false;
121 
122     private static synchronized String getString(String key) {
123 	if (!resinit) {
124 	    try {
125 		resources = ResourceBundle.getBundle(
126 				     "org.apache.river.tool.resources.checkser");
127 		resinit = true;
128 	    } catch (MissingResourceException e) {
129 		e.printStackTrace();
130 	    }
131 	}
132 	try {
133 	    return resources.getString(key);
134 	} catch (MissingResourceException e) {
135 	    return null;
136 	}
137     }
138 
139     private static void print(String key, String val) {
140 	String fmt = getString(key);
141 	if (fmt == null)
142 	    fmt = "no text found: \"" + key + "\" {0}";
143 	System.out.println(MessageFormat.format(fmt, new String[]{val}));
144     }
