187     private static class Exceptional<String> implements Callable<String> {
188         private final AtomicInteger tries = new AtomicInteger(0);
189         @Override
190         public String  call() throws Exception {
191             System.out.println("Task called at:");
192             System.out.println(System.currentTimeMillis());
193             int tri = tries.incrementAndGet();
194             if (tri < 7) throw new RemoteException("Dummy communication problem");
195             return (String) "success";
196         }
