48      static class TestParser extends DefaultPolicyParser {
49  
50          PermissionGrant[] content;
51  
52          public TestParser(PermissionGrant[] content) {
53              this.content = content;
54          }
55  
56          public Collection<PermissionGrant> parse(URL location, Properties system)
57              throws Exception {
58              if (content != null) {
59                  return Arrays.asList(content);
60              }
61              throw new RuntimeException();
62          }
63      }

65      /**
66       * Tests that policy is really resetted on refresh().
67       * @throws PolicyInitializationException if test fails.
68       */
69      public void testRefresh() throws PolicyInitializationException {
70          Permission sp = new SecurityPermission("sdf");
71          PermissionGrantBuilder pgb = PermissionGrantBuilder.newBuilder();
72          PermissionGrant[] pe = new PermissionGrant[] { 
73              pgb.uri(null).principals(null)
74                 .permissions(new Permission[] { sp })
75                 .context(PermissionGrantBuilder.URI)
76                 .build()
77          };
78          TestParser tp = new TestParser(pe);
79          ConcurrentPolicyFile policy = new ConcurrentPolicyFile(tp, new PermissionComparator());
80          CodeSource cs = new CodeSource(null, (Certificate[])null);
81          ProtectionDomain pd = new ProtectionDomain(cs, null, null, null);
82          assertTrue(policy.getPermissions(pd).implies(sp));
83  
84          tp.content = new PermissionGrant[0];
85          policy.refresh();
86          assertFalse(policy.getPermissions(pd).implies(sp));
87  
88          tp.content = null;
89          policy.refresh();
90          assertFalse(policy.getPermissions(pd).implies(sp));
91      }
92  
93      /**
94       * Tests that refresh() does not fail on failing parser.
95       * @throws PolicyInitializationException if test fails.
96       */
97      public void testRefresh_Failure() throws PolicyInitializationException {
98          CodeSource cs = new CodeSource(null, (Certificate[])null);
99          ConcurrentPolicyFile policy = new ConcurrentPolicyFile(new TestParser(null), new PermissionComparator());
100         policy.refresh();
101         assertFalse(policy.getPermissions(cs).elements().hasMoreElements());
102     }

104     /**
105      * Tests proper policy evaluation for CodeSource parameters.
106      * @throws java.lang.Exception if test fails.
107      */
108     public void testGetPermissions_CodeSource() throws Exception {
109         PermissionGrantBuilder pgb = PermissionGrantBuilder.newBuilder();
110         pgb.context(PermissionGrantBuilder.URI);
111         CodeSource cs = new CodeSource(null, (Certificate[])null);
112         CodeSource cs2 = new CodeSource(new URL("http://a.b.c"),
113             (Certificate[])null);
114         Permission sp1 = new SecurityPermission("aaa");
115         Permission sp2 = new SecurityPermission("bbb");
116         Permission sp3 = new SecurityPermission("ccc");
117         PermissionGrant pe1 = pgb.uri(null)
118                 .permissions(new Permission[] { sp1 })
119                 .build();
120         pgb.reset().context(PermissionGrantBuilder.URI);
121         PermissionGrant pe2 = pgb.uri("http://a.b.c")
122                 .principals(new Principal[0])
123                 .permissions(new Permission[] { sp2 })
124                 .build();
125         pgb.reset().context(PermissionGrantBuilder.URI);
126         PermissionGrant pe3 = pgb.uri(null)
127                 .principals( new Principal[] {new FakePrincipal("qqq") })
128                 .permissions(new Permission[] { sp3 })
129                         .build();
130         PermissionGrant[] peArray = new PermissionGrant[] { pe1, pe2, pe3};
131         ConcurrentPolicyFile policy = new ConcurrentPolicyFile(new TestParser(peArray), new PermissionComparator());
132         
133         ProtectionDomain pd = new ProtectionDomain(cs, null, null, null);
134         ProtectionDomain pd2 = new ProtectionDomain(cs2, null, null, null);
135 
136         assertTrue(policy.getPermissions(pd).implies(sp1));
137         assertFalse(policy.getPermissions(pd).implies(sp2));
138         assertFalse(policy.getPermissions(pd).implies(sp3));
139 
140         assertTrue(policy.getPermissions(pd2).implies(sp1));
141         assertTrue(policy.getPermissions(pd2).implies(sp2));
142         assertFalse(policy.getPermissions(pd2).implies(sp3));
143     }

145     /**
146      * Tests proper policy evaluation for ProtectionDomain parameters.
147      * @throws java.lang.Exception if test fails.
148      */
149     public void testGetPermissions_ProtectionDomain() throws Exception {
150         PermissionGrantBuilder pgb = PermissionGrantBuilder.newBuilder();
151         pgb.context(PermissionGrantBuilder.URI);
152         Permission sp1 = new SecurityPermission("aaa");
153         Permission sp2 = new SecurityPermission("bbb");
154         Permission sp3 = new SecurityPermission("ccc");
155         Permission sp4 = new SecurityPermission("ddd");
156         Permission spZ = new SecurityPermission("zzz");
157         PermissionCollection pcZ = spZ.newPermissionCollection();
158         pcZ.add(spZ);
159         CodeSource cs = new CodeSource(null, (Certificate[])null);
160         CodeSource cs2 = new CodeSource(new URL("http://a.b.c"),
161             (Certificate[])null);
162         ProtectionDomain pd1 = new ProtectionDomain(cs, null);
163         ProtectionDomain pd2 = new ProtectionDomain(cs2, pcZ, null,
164             new Principal[] { new FakePrincipal("qqq") });
165         
166         PermissionGrant pe1 = pgb.uri(null)
167                 .permissions(new Permission[] { sp1 })
168                 .build();
169         PermissionGrant pe2 = pgb.uri(cs2.getLocation().toString())
170                 .principals(new Principal[] { new UnresolvedPrincipal(
171                 UnresolvedPrincipal.WILDCARD, UnresolvedPrincipal.WILDCARD) })
172                 .permissions(new Permission[] { sp2 })
173                 .build();
174         PermissionGrant pe3 = pgb.uri(null)
175                 .principals(new Principal[] { new UnresolvedPrincipal(
176                 FakePrincipal.class.getName(), "qqq") })
177                 .permissions(new Permission[] { sp3 })
178                 .build();
179         PermissionGrant pe4 = pgb.uri(cs2.getLocation().toString())
180                 .principals(new Principal[] { new UnresolvedPrincipal(
181                 FakePrincipal.class.getName(), "ttt") })
182                 .permissions(new Permission[] { sp4 })
183                 .build();
184         PermissionGrant[] peArray = new PermissionGrant[]{
185             pe1, pe2, pe3, pe4 };
186         ConcurrentPolicyFile policy = new ConcurrentPolicyFile(new TestParser(peArray), new PermissionComparator());
187 
188         assertTrue(policy.getPermissions(pd1).implies(sp1));
189         assertFalse(policy.getPermissions(pd1).implies(sp2));
190         assertFalse(policy.getPermissions(pd1).implies(sp3));
191         assertFalse(policy.getPermissions(pd1).implies(sp4));
192 
193         assertTrue(policy.getPermissions(pd2).implies(sp1));
194         assertTrue(policy.getPermissions(pd2).implies(sp2));
195         assertTrue(policy.getPermissions(pd2).implies(sp3));
196         assertFalse(policy.getPermissions(pd2).implies(sp4));
197     }
