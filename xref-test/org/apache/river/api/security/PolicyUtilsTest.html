67      /**
68       * Tests valid expansion of ${key} entries.
69       * @throws Exception if the test fails.
70       */
71      public void testExpand() throws Exception {
72          String[] input = new String[] { "${key.1}", "abcd${key.1}",
73                  "a ${key.1} b ${$key$}${key.2}", "$key.1", "${}" };
74          String[] output = new String[] { "value1", "abcdvalue1",
75                  "a value1 b ${${${${${${value.2", "$key.1", "" };
76          Properties props = new Properties();
77          props.put("key.1", "value1");
78          props.put("key.2", "value.2");
79          props.put("$key$", "${${${${${${");
80          props.put("", "");
81          for (int i = 0; i < output.length; i++) {
82              assertEquals(output[i], PolicyUtils.expand(input[i], props));
83          }
84      }

111     /**
112      * Tests valid expansion of ${{protocol:data}} entries.
113      * @throws Exception if the test fails.
114      */
115     public void testExpandGeneral() throws Exception {
116         String[] input = new String[] { "${{a:b}}", "a ${{self}}${{a: made}}",
117                 "${{}}" };
118         String[] output = new String[] { "b", "a this made", "" };
119         PolicyUtils.GeneralExpansionHandler handler = new PolicyUtils.GeneralExpansionHandler() {
120 
121             public String resolve(String protocol, String data) {
122                 if ("a".equals(protocol)) {
123                     return data;
124                 }
125                 if ("self".equals(protocol)) {
126                     return "this";
127                 }
128                 if ("".equals(protocol)) {
129                     return protocol;
130                 }
131                 return null;
132             }
133         };
134         for (int i = 0; i < output.length; i++) {
135             assertEquals(output[i], PolicyUtils
136                     .expandGeneral(input[i], handler));
137         }
138     }

165         final String key = "policy.expandProperties";
166         String OLD = Security.getProperty(key);

168             Security.setProperty(key, "true");
169             assertTrue(PolicyUtils.canExpandProperties());
170             Security.setProperty(key, "false");
171             assertFalse(PolicyUtils.canExpandProperties());
172             Security.setProperty(key, "");
173             assertTrue(PolicyUtils.canExpandProperties());
174             Security.setProperty(key, "laejhg");
175             assertTrue(PolicyUtils.canExpandProperties());
176         }
177         finally {
178             Security.setProperty(key, OLD);
179         }

206             result = PolicyUtils.getPolicyURLs(arg, KEY, "");

209             assertEquals(new URL("http://foo.bar.com"), result[0]);

259     /**
260      * Tests expansion in system and security URLs.
261      * @throws Exception if the test fails.
262      */
263     public void testGetPolicyURLs03() throws Exception {
264         final String KEY_DYNAMIC = "policy.allowSystemProperty";
265         final String OLD_DYNAMIC = Security.getProperty(KEY_DYNAMIC);
266         final String KEY_EXP = "policy.expandProperties";
267         final String OLD_EXP = Security.getProperty(KEY_EXP);
268         final String PREFIX = "testGetPolicyURLs03.";
269         String[] OLD = new String[5];
270         for (int i = 0; i < OLD.length; i++) {
271             OLD[i] = Security.getProperty(PREFIX + i);
272         }
273 
274         final String KEY = "dsfvdf";
275         Properties arg = new Properties();
276         arg.put(KEY, "file://${foo.path}/${foo.name}");
277         arg.put("foo.path", "path");
278         arg.put("foo.name", "name");
279         arg.put("foo", "acme");
280         Security.setProperty(KEY_DYNAMIC, "true");
281         Security.setProperty(KEY_EXP, "true");
282         Security.setProperty(PREFIX + 1, "http://foo0.${foo}.org");
283         Security.setProperty(PREFIX + 2, "http://${bar}.com");
284         Security.setProperty(PREFIX + 3,
285                 "http://foo2.bar.com/${foo.path}/${foo.name}");
286         try {
287 
288             URL[] result = PolicyUtils.getPolicyURLs(arg, KEY, PREFIX);
289             assertNotNull(result);
290             assertEquals(3, result.length);
291             assertEquals(new URL("http://foo0.acme.org"), result[0]);
292             assertEquals(new URL("http://foo2.bar.com/path/name"), result[1]);
293             assertEquals(new URL("file://path/name"), result[2]);
294 
295             //expansion here cannot be switched off
296             Security.setProperty(KEY_EXP, "false");
297             result = PolicyUtils.getPolicyURLs(arg, KEY, PREFIX);
298             assertNotNull(result);
299             assertEquals(3, result.length);
300             assertEquals(new URL("http://foo0.acme.org"), result[0]);
301             assertEquals(new URL("http://foo2.bar.com/path/name"), result[1]);
302             assertEquals(new URL("file://path/name"), result[2]);
303         }
304         finally {
305             Security.setProperty(KEY_DYNAMIC, OLD_DYNAMIC);
306             Security.setProperty(KEY_EXP, OLD_EXP);
307             for (int i = 0; i < OLD.length; i++) {
308                 Security
309                         .setProperty(PREFIX + i, (OLD[i] == null) ? "" : OLD[i]);
310             }
311         }
312     }

328     /** 
329      * Tests various combinations of arrays:
330      * null/empty/containing null/containing real objects. 
331      */
332     public void testMatchSubset() {
333         assertTrue(PolicyUtils.matchSubset(null, null));
334         assertTrue(PolicyUtils.matchSubset(new Object[] {}, null));
335         assertTrue(PolicyUtils.matchSubset(new Object[] { null }, null));
336         assertTrue(PolicyUtils.matchSubset(new Object[] {},
337                 new Object[] { null }));
338         assertTrue(PolicyUtils.matchSubset(new Object[] { "1", "2" },
339                 new Object[] { "3", "2", "1" }));
340         assertTrue(PolicyUtils.matchSubset(new Object[] { "1", null },
341                 new Object[] { "3", "2", "1" }));
342         assertFalse(PolicyUtils.matchSubset(new Object[] { "1", null },
343                 new Object[] { "3", "2", }));
344     }
