197     public void test_setReadOnly() {
198         // test java.security.permissionCollection.setReadOnly()
199         SecurityPermission permi = new SecurityPermission(
200                 "testing permissionCollection-setReadOnly");
201         PermissionCollection permCollect = permi.newPermissionCollection();
202         assertTrue("readOnly has not been set, but isReadOnly returned true",
203                 !permCollect.isReadOnly());
204         permCollect.setReadOnly();
205         assertTrue("readOnly is set, but isReadonly returned false",
206                 permCollect.isReadOnly());
207     }
208 
209     public void test_toString() {
210         // test java.security.permissionCollection.toString()
211         SecurityPermission permi = new SecurityPermission(
212                 "testing permissionCollection-isREadOnly");
213         assertNotNull("toString should have returned a string of elements",
214                 permi.newPermissionCollection().toString());
215         assertTrue(permi.newPermissionCollection().toString().endsWith("\n"));
216     }

218     // FIXME move me to Support_Resources
219     public static URL getResourceURL(String name) {
220 
221         URL url = ClassLoader.getSystemClassLoader().getResource(name);
222 
223         if (url == null) {
224             throw new RuntimeException("Failed to get resource url: " + name);
225         }
226 
227         return url;
228     }
