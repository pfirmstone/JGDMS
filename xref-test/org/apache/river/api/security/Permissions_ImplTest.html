23  package org.apache.river.api.security;
24  import java.security.AllPermission;
25  import java.security.BasicPermission;
26  import java.security.Permission;
27  import java.security.PermissionCollection;
28  import java.security.SecurityPermission;
29  import java.security.UnresolvedPermission;
30  
31  import junit.framework.TestCase;
32  
33  /**
34   * Tests for <code>Permissions</code>
35   * 
36   */
37  
38  public class Permissions_ImplTest extends TestCase {
39  
40      /**
41       * A permission is implied by this collection, if either of the following is
42       * true:
43       * <ul>
44       * <li>This collection contains AllPermission;
45       * <li>This collection has elements of the same type as the permission
46       * being checked, and they imply it;
47       * <li>This collection has UnresolvedPermissions which can be resolved to
48       * the checked type, and after resolving they imply the checked one.
49       * </ul>
50       * The only exception is UnresolvedPermission itself, which is effectively
51       * implied only by AllPermission
52       */
53      public void testImplies() {
54          ConcurrentPermissions ps = new ConcurrentPermissions();
55          Permission ap = new AllPermission();
56          Permission bp1 = new BasicPermission("jhb23jhg5") {
57          };
58          Permission bp2 = new BasicPermission("&%#&^$HJVH") {
59  
60              public PermissionCollection newPermissionCollection() {
61                  return null;
62              }
63          };
64          Permission sp1 = new SecurityPermission("a.b.c");
65          Permission sp2 = new SecurityPermission("a.b.*");
66          Permission sp3 = new SecurityPermission("a.*");
67          Permission up = new UnresolvedPermission(
68              "java.security.SecurityPermission", "*", null, null);
69  
70          Permission[] arr = new Permission[] {
71              ap, bp1, bp2, sp1, sp2, up };
72          for (int i = 0; i < arr.length; i++) {
73              assertFalse(ps.implies(arr[i]));
74          }
75  
76          ps.add(bp1);
77          assertTrue(ps.implies(bp1));
78          assertFalse(ps.implies(bp2));
79          assertFalse(ps.implies(ap));
80          assertFalse(ps.implies(sp1));
81  
82          ps.add(sp2);
83          assertTrue(ps.implies(sp1));
84          assertTrue(ps.implies(sp2));
85          assertFalse(ps.implies(sp3));
86  
87          ps.add(up);
88          assertFalse(ps.implies(up));
89          assertTrue(ps.implies(sp1));
90          assertTrue(ps.implies(sp2));
91          assertTrue(ps.implies(sp3));
92  
93          ps.add(ap);
94          for (int i = 0; i < arr.length; i++) {
95              assertTrue(ps.implies(arr[i]));
96          }
97      }
98   }
