356 	static void load(InputStream in, Hashtable<String, String> result) throws IOException {
357 		int NONE = 0, SLASH = 1, UNICODE = 2, CONTINUE = 3, DONE = 4, IGNORE = 5;
358 		int mode = NONE, unicode = 0, count = 0, nextChar;
359 		StringBuffer key = new StringBuffer(), value = new StringBuffer(), buffer = key;
360 		boolean firstChar = true;
361 
362 		while ((nextChar = in.read()) != -1) {
363 			if (mode == UNICODE) {
364 				int digit = Character.digit((char) nextChar, 16);
365 				if (digit >= 0) {
366 					unicode = (unicode << 4) + digit;
367 					if (++count < 4) {
368                         continue;
369                     }
370 				}
371 				mode = NONE;
372 				buffer.append((char) unicode);
373 				if (nextChar != '\n') {
374                     continue;
375                 }
376 			}
377 			if (mode == SLASH) {
378 				mode = NONE;
379 				switch (nextChar) {
380 				case '\r':
381 					mode = CONTINUE; // Look for a following \n
382 					continue;
383 				case '\n':
384 					mode = IGNORE; // Ignore whitespace on the next line
385 					continue;
386 				case 'b':
387 					nextChar = '\b';
388 					break;
389 				case 'f':
390 					nextChar = '\f';
391 					break;
392 				case 'n':
393 					nextChar = '\n';
394 					break;
395 				case 'r':
396 					nextChar = '\r';
397 					break;
398 				case 't':
399 					nextChar = '\t';
400 					break;
401 				case 'u':
402 					mode = UNICODE;
403 					unicode = count = 0;
404 					continue;
405 				}
406 			} else {
407 				switch (nextChar) {
408 				case '#':
409 				case '!':
410 					if (firstChar) {
411 						while ((nextChar = in.read()) != -1) {
412                             if (nextChar == '\r' || nextChar == '\n') {
413                                 break;
414                             }
415                         }
416 						continue;
417 					}
418 					break;
419 				case '\n':
420 					if (mode == CONTINUE) { // Part of a \r\n sequence
421 						mode = IGNORE; // Ignore whitespace on the next line
422 						continue;
423 					}
424 					// fall into the next case
425 				case '\r':
426 					mode = NONE;
427 					firstChar = true;
428 					if (key.length() > 0 || buffer == value) {
429                         result.put(key.toString(), value.toString());
430                     }
431 					key.setLength(0);
432 					value.setLength(0);
433 					buffer = key;
434 					continue;
435 				case '\\':
436 					mode = SLASH;
437 					continue;
438 				case ':':
439 				case '=':
440 					if (buffer == key) {
441 						buffer = value;
442 						continue;
443 					}
444 					break;
445 				}
446 				char c = (char) nextChar;
447 				if ((c >= 0x1c && c <= 0x20) || (c >= 0x9 && c <= 0xd)) {
448 					if (mode == CONTINUE) {
449                         mode = IGNORE;
450                     }
451 					if (buffer.length() == 0 || mode == IGNORE) {
452                         continue;
453                     }
454 					if (buffer == key) {
455 						mode = DONE;
456 						continue;
457 					}
458 				}
459 				if (mode == IGNORE || mode == CONTINUE) {
460                     mode = NONE;
461                 }
462 			}
463 			firstChar = false;
464 			if (mode == DONE) {
465 				buffer = value;
466 				mode = NONE;
467 			}
468 			buffer.append((char) nextChar);
469 		}
470 		if (key.length() > 0 || buffer == value) {
471             result.put(key.toString(), value.toString());
472         }
473 	}
