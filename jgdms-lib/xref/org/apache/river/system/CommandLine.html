33  /**
34   * <code>CommandLine</code> is a base class for command line parsing
35   * classes.  It provides several useful methods for the implementor
36   * of the subclass and for the programmer.  It also defines exceptions
37   * that can be generated during parsing commands.
38   * <p>
39   * <code>CommandLine</code> provides you several methods to get input
40   * streams from the command line.  If these do not suffice for your
41   * particular needs, you can get the argument as a <code>String</code>
42   * and do your own processing.
43   * <p>
44   * You can use these methods to parse your own operand arguments.  For
45   * example, if your operands are integers, you may want to use
46   * <code>parseInt</code> to parse them to provide consistency to the
47   * user.
48   *
49   * @author Sun Microsystems, Inc.
50   *
51   * @see POSIXCommandLine
52   * @see MultiCommandLine
53   */
54  public abstract class CommandLine {
55      /**
56       * Return the result of parsing the given string from the command
57       * line.  If <code>str</code> is <code>null</code>, return
58       * <code>defaultValue</code>.
59       */
60      public static String parseString(String str, String defaultValue) {
61  	return (str != null ? str : defaultValue);
62      }
63  
64      /**
65       * Return the result of parsing the given <code>int</code> from the
66       * command line.  If <code>str</code> is <code>null</code>, return
67       * <code>defaultValue</code>.  The value is parsed using to
68       * understand leading <code>0x</code> or <code>#</code> to
69       * introduce a hexidecimal number, leading <code>0</code> to
70       * introduce an octal number, and anything else to be a decimal
71       * number.
72       */
73      public static int parseInt(String str, int defaultValue)
74  	throws NumberFormatException
75      {
76  	if (str == null)
77  	    return defaultValue;
78  	if (str.startsWith("0x"))
79  	    return Integer.parseInt(str.substring(2), 16);
80  	else if (str.startsWith("#"))
81  	    return Integer.parseInt(str.substring(1), 16);
82  	else if (str.startsWith("0"))
83  	    return Integer.parseInt(str.substring(1), 8);
84  	return Integer.parseInt(str);
85      }
86  
87      /**
88       * Return the result of parsing the given <code>long</code> from
89       * the command line.  If <code>str</code> is <code>null</code>,
90       * return <code>defaultValue</code>.  The value is parsed using to
91       * understand leading <code>0x</code> or <code>#</code> to
92       * introduce a hexidecimal number, leading <code>0</code> to
93       * introduce an octal number, and anything else to be a decimal
94       * number.
95       */
96      public static long parseLong(String str, long defaultValue)
97  	throws NumberFormatException
98      {
99  	if (str == null)
100 	    return defaultValue;
101 	if (str.startsWith("0x"))
102 	    return Long.parseLong(str.substring(2), 16);
103 	else if (str.startsWith("#"))
104 	    return Long.parseLong(str.substring(1), 16);
105 	else if (str.startsWith("0"))
106 	    return Long.parseLong(str.substring(1), 8);
107 	return Long.parseLong(str);
108     }
109 
110     /**
111      * Return the result of parsing the given <code>double</code> from
112      * the command line.  If <code>str</code> is <code>null</code>,
113      * return <code>defaultValue</code>.  The value is parsed using
114      * <code>Double.value</code>.
115      */
116     public static double parseDouble(String str, double defaultValue)
117 	throws NumberFormatException
118     {
119 	if (str == null)
120 	    return defaultValue;
121 	return Double.parseDouble(str);
122     }

124     /**
125      * Return a <code>Writer</code> that is the result of creating a
126      * new <code>FileWriter</code> object for the file named by the
127      * given <code>path</code>.  If the argument is `<code>-</code>' then
128      * an <code>OutputStreamWriter</code> for <code>System.out</code>
129      * is returned.  If <code>path</code> is <code>null</code>, return
130      * <code>defaultValue</code>.
131      */
132     public static Writer parseWriter(String path, Writer defaultValue)
133 	throws IOException
134     {
135 	if (path == null)
136 	    return defaultValue;
137 	if (path.equals("-"))
138 	    return new OutputStreamWriter(System.out, "UTF-8");
139 	return new OutputStreamWriter(new FileOutputStream(path), "UTF-8");
140     }

147      * is returned.  If <code>path</code> is <code>null</code>, the
148      * string <code>path</code> is used as the file name.

313 	    super(str);
314 	}

317     /**
318      * Signal that the user specified an option in an improper way.
319      * The most common errors are not providing an argument to an
320      * option that needed one or specifying an unknown option.
321      */
322     public static class BadInvocationException extends Exception {
323 	static final long serialVersionUID = 4503820475450471907L;
324 
325 	/** Create an "Argument required" exception for the given option. */
326 	public BadInvocationException(Object opt) {
327 	    super("Argument required for '" + opt + "'");
328 	}
329 	/** Create an exception with the given detail string. */
330 	public BadInvocationException(String str) {
331 	    super(str);
332 	}
333     }
