19  package org.apache.river.system;
20  
21  import java.io.File;
22  import java.util.ArrayList;
23  import java.util.Observable;
24  import java.util.Observer;
25  
26  /**
27   * Observer class that gets notified when a change has been noted,
28   * that is a file has been found, by an Observable object. 
29   * This class is normally passed into a class that extends 
30   * <code>Observable</code>
31   *
32   * @author Sun Microsystems, Inc.
33   *
34   * @see java.util.Observable
35   * @see java.util.Observer
36   * @see org.apache.river.system.FileWalker
37   */
38  public class FileObserver implements Observer {
39      private ArrayList fileList = (new ArrayList());
40      private ArrayList timeList = (new ArrayList());
41      private File currentFile = null;
42      private int lcv = 0;
43      private String suffix;
44  
45      /**
46       * Constructor that allows the user to specify interest in a 
47       * particular suffix. Only files with the provide suffix 
48       * will be saved and returned to the user. All others will fall
49       * through.
50       * <p>
51       * If suffix is an empty string then all files will be matched.
52       */
53      public FileObserver(String suffix) {
54  	this.suffix = suffix;
55      }
56  
57      /**
58       * Called when an Observer monitors a change.
59       * When a change is noted by the item being observed,
60       * in this case the given root build directory,
61       * this method gets called.
62       */
63      public void update(Observable o,Object arg) {
64  	currentFile = (File)arg;
65  	if (currentFile.getName().endsWith(suffix)) {
66  	    //System.out.println("found file:"+currentFile.getName());
67  	    fileList.add(lcv,currentFile.getAbsolutePath());
68  	    timeList.add(lcv,Long.valueOf(currentFile.lastModified()));
69  	    lcv++;
70  	}
71      }	
72  
73      /**
74       * Get the complete list of found files. The
75       * list contains String representations of the file names.
76       * 
77       * @return ArrayList - List of found files.
78       */
79      public ArrayList getFileList() {
80  	return fileList;
81      }
82  
83      /**
84       * Get the time stamps of the found files. There is a
85       * 1-1 correspondence with the contents of getFileList.
86       * The lit contains Long representations of the file time stamps.
87       *
88       * @see #getFileList
89       * @return ArrayList - List of found files date stamps
90       */
91      public ArrayList getFileTimeList() {
92  	return timeList;
93      }
94  }
