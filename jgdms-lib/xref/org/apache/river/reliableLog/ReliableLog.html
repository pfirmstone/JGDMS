34  /**
35   * This class is a simple implementation of a reliable Log.  The
36   * client of a ReliableLog must provide a set of callbacks (via a
37   * LogHandler) that enables a ReliableLog to read and write snapshots
38   * (checkpoints) and log records.  This implementation ensures that the
39   * data stored (via a ReliableLog) is recoverable after a system crash.
40   * The implementation is unsynchronized; the client must synchronize
41   * externally. <p>
42   *
43   * The secondary storage strategy is to record values in files using a
44   * representation of the caller's choosing.  Two sorts of files are
45   * kept: snapshots and logs.  At any instant, one snapshot is current.
46   * The log consists of a sequence of updates that have occurred since
47   * the current snapshot was taken.  The current stable state is the
48   * value of the snapshot, as modified by the sequence of updates in
49   * the log.  From time to time, the client of a ReliableLog instructs
50   * the package to make a new snapshot and clear the log.  A ReliableLog
51   * arranges disk writes such that updates are stable (as long as the
52   * changes are force-written to disk) and atomic: no update is lost,
53   * and each update either is recorded completely in the log or not at
54   * all.  Making a new snapshot is also atomic. <p>
55   *
56   * Normal use for maintaining the recoverable store is as follows: The
57   * client maintains the relevant data structure in virtual memory.  As
58   * updates happen to the structure, the client informs the ReliableLog
59   * (call it "log") by calling log.update.  Periodically, the client
60   * calls log.snapshot to provide the current complete contents of the
61   * data.  On restart, the client calls log.recover to obtain the
62   * latest snapshot and the following sequences of updates; the client
63   * applies the updates to the snapshot to obtain the state that
64   * existed before the crash. <p>
65   *
66   * @author Sun Microsystems, Inc.
67   *
68   * @see LogHandler
69   *
70   */
71  
72  public class ReliableLog {
73  
74      private static final String snapshotPrefix = "Snapshot.";
75      private static final String logfilePrefix = "Logfile.";
76      private static final String versionFile = "Version_Number";
77      private static final int MAGIC = 0xf2ecefe7;
78      private static final int FORMAT_UNPADDED = 0;
79      private static final int FORMAT_PADDED = 1;
80      private static final long intBytes = 4;
81  
82      private final File dir;		// base directory
83      private int version = 0;		// current snapshot and log version
84      private int format = FORMAT_UNPADDED;
85      private String logName = null;
86      private RandomAccessFile log = null;
87      private FileDescriptor logFD;
88      private long snapshotBytes = 0;
89      private long logBytes = 0;
90      private final LogHandler handler;
91      private final byte[] intBuf = new byte[4];
92      private final byte[] zeroBuf = new byte[4];
93   
94      /**
95       * Creates a ReliableLog to handle snapshots and logging in a
96       * stable storage directory, and sets up to recover any existing data
97       * from the stable storage directory. If there is no existing data,
98       * snapshot must be called next, otherwise recover must be called next.
99       *
100      * @param dirPath path to the stable storage directory
101      * @param handler the handler for log callbacks
102      *
103      * @throws LogException if the directory cannot be created or
104      * the current version in the directory is corrupted
105      * @throws IOException if any other I/O error occurs
106      */
107     public ReliableLog(String dirPath, LogHandler handler) throws IOException {
108 	dir = new File(dirPath);
109 	if (!(dir.exists() ? dir.isDirectory() : dir.mkdir())) {
110 	    throw new LogException("could not create directory for log: " +
111 				   dirPath);
112 	}
113 	this.handler = handler;
114 	try {
115 	    DataInputStream in =
116 		new DataInputStream(new FileInputStream(fName(versionFile)));
117 	    try {
118 		version = in.readInt();
119 	    } finally {
120 		in.close();
121 	    }
122 	} catch (IOException ex) {
123 	    writeVersionFile();
124 	}
125 	if (version < 0) {
126 	    throw new LogException("corrupted version file");
127 	}
128     }
129 
130     /**
131      * Retrieves the contents of the snapshot file by calling the client
132      * supplied recover callback and then applies the incremental updates
133      * by calling the readUpdate callback for each logged updated.
134      *
135      * @throws LogException if recovery fails due to serious log corruption,
136      * or if an exception is thrown by the recover or readUpdate callbacks
137      * @throws IOException if an other I/O error occurs
138      */
139     public void recover() throws IOException {
140 	if (version == 0) 
141 	    return;
142 	
143 	String fname = versionName(snapshotPrefix);
144 	File file = new File(fname);
145 	InputStream in = new BufferedInputStream(new FileInputStream(file));
146 	try {
147 	    handler.recover(in);
148 	} catch (Exception e) {
149 	    throw new LogException("recovery failed", e);
150 	} finally {
151 	    in.close();
152 	}
153 	snapshotBytes = file.length();
154 	
155 	fname = versionName(logfilePrefix);
156 	file = new File(fname);
157 	DataInputStream din =
158 	    new DataInputStream(new BufferedInputStream(
159 						 new FileInputStream(file)));
160 	long length = file.length();
161 	try {
162 	    int updateLen = din.readInt();
163 	    /* have to worry about no MAGIC in original format */
164 	    if (updateLen == MAGIC) {
165 		format = din.readInt();
166 		if (format != FORMAT_PADDED) {
167 		    throw new LogException("corrupted log: bad log format");
168 		}
169 		logBytes += (intBytes + intBytes);
170 		updateLen = din.readInt();
171 	    }
172 	    while (true) {
173 		if (updateLen == 0) { /* expected termination case */
174 		    break;
175 		}
176 		if (updateLen < 0) { /* serious corruption */
177 		    throw new LogException("corrupted log: bad update length");
178 		}
179 		if (length - logBytes - intBytes < updateLen) {
180 		    /* partial record at end of log; this should not happen
181 		     * if forceToDisk is always true, but might happen if
182 		     * buffered updates are used.
183 		     */
184 		    break;
185 		}
186 		try {
187 		    handler.readUpdate(new LogInputStream(din, updateLen));
188 		} catch (Exception e) {
189 		    throw new LogException("read update failed", e);
190 		}
191 		logBytes += (intBytes + updateLen);
192 		if (format == FORMAT_PADDED) {
193 		    int offset = (int)logBytes & 3;
194 		    if (offset > 0) {
195 			offset = 4 - offset;
196                         // In case fewer than offset bytes are skipped.
197 			logBytes += din.skipBytes(offset);
198 		    }
199 		}
200 		updateLen = din.readInt();
201 	    } /* while */
202 	} catch (EOFException e) {
203 	} finally {
204 	    din.close();
205 	}
206 	/* reopen log file at end */
207 	openLogFile();
208     }

210     /**
211      * Records this update in the log file and forces the update to disk.
212      * The update is recorded by calling the client's writeUpdate callback.
213      * This method must not be called until this log's recover method has
214      * been invoked (and completed).
215      *
216      * @param value the object representing the update
217      *
218      * @throws LogException if an exception is thrown by the writeUpdate
219      * callback, or forcing the update to disk fails
220      * @throws IOException if any other I/O error occurs
221      */
222     public void update(Object value) throws IOException {
223 	update(value, true);
224     }
225     
226     /**
227      * Records this update in the log file and optionally forces the update
228      * to disk.  The update is recorded by calling the client's writeUpdate
229      * callback.  This method must not be called until this log's recover
230      * method has been invoked (and completed).
231      *
232      * @param value the object representing the update
233      * @param forceToDisk true if the update should be forced to disk, false
234      * if the updates should be buffered
235      *
236      * @throws LogException if an exception is thrown by the writeUpdate
237      * callback, or forcing the update to disk fails
238      * @throws IOException if any other I/O error occurs
239      */
240     public void update(Object value, boolean forceToDisk) throws IOException {
241         /* avoid accessing a null log field */
242         if (log == null) {
243             throw new LogException("log file for persistent state is "
244                                    +"inaccessible, it may have been "
245                                    +"corrupted or closed");
246         }
247 	/* note: zero length header for this update was written as part
248 	 * of the previous update, or at initial opening of the log file
249 	 */
250 	try {
251 	    handler.writeUpdate(new LogOutputStream(log), value);
252 	} catch (Exception e) {
253 	    throw new LogException("write update failed", e);
254 	}
255 	if (forceToDisk) {
256 	    /* must force contents to disk before writing real length header */
257 	    try {
258 		logFD.sync();
259 	    } catch (SyncFailedException sfe) {
260 		throw new LogException("sync log failed", sfe);
261 	    }
262 	}
263 	long entryEnd = log.getFilePointer();
264 	long updateLen = entryEnd - logBytes - intBytes;
265 	if (updateLen > Integer.MAX_VALUE) {
266 	    throw new LogException("maximum record length exceeded");
267 	}
268 	/* write real length header */
269 	log.seek(logBytes);
270 	writeInt(log, (int) updateLen);
271 	/* pad out update record so length header does not span disk blocks */
272 	if (format == FORMAT_PADDED) {
273 	    entryEnd = (entryEnd + 3) & ~3L;
274 	}
275 	/* write zero length header for next update */
276 	log.seek(entryEnd);
277 	log.write(zeroBuf);
278 	logBytes = entryEnd;
279 	/* force both length headers to disk */
280 	if (forceToDisk) {
281 	    try {
282 		logFD.sync();
283 	    } catch (SyncFailedException sfe) {
284 		throw new LogException("sync log failed", sfe);
285 	    }
286 	}
287     }

289     /**
290      * Write an int value in single write operation.
291      *
292      * @param out output stream
293      * @param val int value
294      * @throws IOException if any other I/O error occurs
295      */
296     private void writeInt(DataOutput out, int val) throws IOException {
297 	intBuf[0] = (byte) (val >> 24);
298 	intBuf[1] = (byte) (val >> 16);
299 	intBuf[2] = (byte) (val >> 8);
300 	intBuf[3] = (byte) val;
301 	out.write(intBuf);
302     }

304     /**
305      * Records the client-defined current snapshot by invoking the client
306      * supplied snapshot callback, and then empties the log of incremental
307      * updates.
308      *
309      * @throws LogException if the snapshot callback throws an exception
310      * @throws IOException if any other I/O error occurs
311      */
312     public void snapshot() throws IOException {
313 	int oldVersion = version;
314 	version++;
315 
316 	String fname = versionName(snapshotPrefix);
317 	File snapshotFile = new File(fname);
318 	FileOutputStream out = new FileOutputStream(snapshotFile);
319 	try {
320 	    try {
321 		handler.snapshot(out);
322 		/* force contents to disk */
323 		out.getFD().sync();
324 	    } catch (Exception e) {
325 		throw new LogException("snapshot failed", e);
326 	    }
327 	    snapshotBytes = snapshotFile.length();
328 	} finally {
329 	    out.close();
330 	}
331 
332 	logBytes = 0;
333 	openLogFile();
334 	writeVersionFile();
335 	deleteSnapshot(oldVersion);
336 	deleteLogFile(oldVersion);
337     }
338     
339     /**
340      * Closes the stable storage directory in an orderly manner.
341      *
342      * @throws IOException if an I/O error occurs
343      */
344     public void close() throws IOException {
345 	if (log == null) return;
346 	try {
347 	    log.close();
348 	} finally {
349 	    log = null;
350 	}
351     }
352     
353     /**
354      * Closes the incremental update log file, removes all ReliableLog-related
355      * files from the stable storage directory, and deletes the directory.
356      */
357     public void deletePersistentStore() {
358         try {
359 	    close();
360 	} catch (IOException e) {
361         }
362 	try {
363             deleteLogFile(version);
364 	} catch (LogException e) {
365 	}
366 	try {
367             deleteSnapshot(version);
368 	} catch (LogException e) {
369 	}
370 	try {
371             deleteFile(fName(versionFile));
372 	} catch (LogException e) {
373 	}
374 	try {
375 	    /* Delete the directory. The following call to the delete method
376              * will fail only if the directory is not empty or if the Security
377              * Manager's checkDelete() method throws a SecurityException. 
378              * (The Security Manager will throw such an exception if it
379              * determines that the current application is not allowed to
380              * delete the directory.) For either case, upon un-successful
381              * deletion of the directory, take no further action.
382              */
383 	    dir.delete();
384 	} catch (SecurityException e) {
385 	}
386     }

398     /**
399      * Generates a filename prepended with the stable storage directory path.
400      *
401      * @param name the name of the file (sans directory path)
402      */
403     private String fName(String name) {
404 	return dir.getPath() + File.separator + name;
405     }

407     /**
408      * Generates a version filename prepended with the stable storage
409      * directory path with the current version number as a suffix.
410      *
411      * @param name version filename prefix
412      */
413     private String versionName(String name) {
414 	return versionName(name, version);
415     }
416     
417     /**
418      * Generates a version filename prepended with the stable storage
419      * directory path with the given version number as a suffix.
420      *
421      * @param prefix filename prefix
422      * @param ver version number
423      */
424     private String versionName(String prefix, int ver) {
425 	return fName(prefix) + String.valueOf(ver);
426     }

428     /**
429      * Deletes a file.
430      *
431      * @param name the name of the file (complete path)
432      * @throws LogException if file cannot be deleted
433      */
434     private void deleteFile(String name) throws LogException {
435 	if (!new File(name).delete()) {
436 	    throw new LogException("couldn't delete file: " + name); 
437 	}
438     }

440     /**
441      * Removes the snapshot file.
442      *
443      * @param ver the version to remove
444      *
445      * @throws LogException if file cannot be deleted
446      */
447     private void deleteSnapshot(int ver) throws LogException {
448 	if (ver != 0) {
449 	    deleteFile(versionName(snapshotPrefix, ver));
450 	}
451     }

453     /**
454      * Removes the incremental update log file.
455      *
456      * @param ver the version to remove
457      *
458      * @throws LogException if file cannot be deleted
459      */
460     private void deleteLogFile(int ver) throws LogException {
461 	if (ver != 0) {
462 	    deleteFile(versionName(logfilePrefix, ver));
463 	}
464     }

466     /**
467      * Opens the incremental update log file in read/write mode.  If the
468      * file does not exist, it is created.
469      *
470      * @throws IOException if an I/O error occurs
471      */
472     private void openLogFile() throws IOException {
473 	try {
474 	    close();
475 	} catch (IOException e) { /* assume this is okay */
476 	} 
477 	
478 	logName = versionName(logfilePrefix);
479 	log = new RandomAccessFile(logName, "rw");
480 	logFD = log.getFD();
481 
482 	if (logBytes == 0) {
483 	    format = FORMAT_PADDED;
484 	    writeInt(log, MAGIC);
485 	    writeInt(log, format);
486 	    logBytes = (intBytes + intBytes);
487 	} else {
488 	    log.seek(logBytes);
489 	}
490 	log.setLength(logBytes);
491 	/* always start out with zero length header for the next update */
492 	log.write(zeroBuf);
493 	/* force length header to disk */
494 	logFD.sync();
495     }
496     
497     /**
498      * Writes the current version number to the version file.
499      *
500      * @throws IOException if an I/O error occurs
501      */
502     private void writeVersionFile() throws IOException {
503 	RandomAccessFile out = new RandomAccessFile(fName(versionFile), "rw");
504 	try {
505 	    /* write should be atomic (four bytes on one disk block) */
506 	    writeInt(out, version);
507 	    /* force version to disk */
508 	    out.getFD().sync();
509 	} finally {
510 	    out.close();
511 	}
512     }
