66  /**
67   * Supplies objects needed to configure applications, such as {@link Exporter}
68   * or {@link ProxyPreparer} instances, or application-specific objects,
69   * constructed from data in a configuration source and override options, as
70   * well as data supplied in the call to <code>getEntry</code>. The
71   * configuration source is specified with a file or URL location, or as a
72   * character input stream. The contents of the configuration source consist of
73   * optional import statements followed by entries, grouped by component, that
74   * specify configuration objects using a subset of expression syntax in the
75   * Java(TM) programming language. Additional options specify values for
76   * individual entries, overriding any matching entries supplied in the
77   * configuration source. <p>
78   *
79   * Applications should normally use {@link ConfigurationProvider} to obtain
80   * {@link Configuration} instances, rather than referencing this class
81   * directly, so that the interpretation of configuration options can be
82   * customized without requiring code modifications. <p>
83   *
84   * The syntax of a configuration source is as follows, using the same grammar
85   * notation that is used in <i>The Java Language Specification (JLS)</i>:
86   *
87   * <div>
88   * <p><br>
89   * <i>Source</i>:<br>
90   * &nbsp;&nbsp;<i>Imports</i><sub>opt</sub> <i>Components</i><sub>opt</sub><br>
91   * <br>
92   * <i>Imports</i>:<br>
93   * &nbsp;&nbsp;<i>Import</i><br>
94   * &nbsp;&nbsp;<i>Imports</i> <i>Import</i><br>
95   *<br>
96   * <i>Import</i>:<br>
97   * &nbsp;&nbsp;import <i>PackageName</i> . * ;<br>
98   * &nbsp;&nbsp;import <i>PackageName</i> . <i>ClassName</i> . * ;<br>
99   * &nbsp;&nbsp;import <i>PackageName</i> . <i>ClassName</i> ;<br>
100  * <br>
101  * <i>PackageName</i>:<br>
102  * &nbsp;&nbsp;<i>QualifiedIdentifier</i><br>
103  * <br>
104  * <i>ClassName</i>:<br>
105  * &nbsp;&nbsp;<i>QualifiedIdentifier</i><br>
106  * <br>
107  * <i>Components</i>:<br>
108  * &nbsp;&nbsp;<i>Component</i><br>
109  * &nbsp;&nbsp;<i>Components</i> <i>Component</i><br>
110  * <br>
111  * <i>Component</i>:<br>
112  * &nbsp;&nbsp;<i>QualifiedIdentifier</i> { <i>Entries</i><sub>opt</sub> }<br>
113  * <br>
114  * <i>Entries</i>:<br>
115  * &nbsp;&nbsp;<i>Entry</i><br>
116  * &nbsp;&nbsp;<i>Entries</i> <i>Entry</i><br>
117  *<br>
118  * <i>Entry</i>:<br>
119  * &nbsp;&nbsp;<i>EntryModifiers</i><sub>opt</sub> <i>Identifier</i> = <i>Expr</i> ;<br>
120  *<br>
121  * <i>EntryModifiers</i>:<br>
122  * &nbsp;&nbsp;static<br>
123  * &nbsp;&nbsp;private<br>
124  * &nbsp;&nbsp;static private<br>
125  * &nbsp;&nbsp;private static<br>
126  * <br>
127  * <i>Expr</i>:<br>
128  * &nbsp;&nbsp;<i>Literal</i><br>
129  * &nbsp;&nbsp;<i>TypeName</i> . class<br>
130  * &nbsp;&nbsp;<i>EntryName</i><br>
131  * &nbsp;&nbsp;<i>ThisReference</i><br>
132  * &nbsp;&nbsp;<i>FieldName</i><br>
133  * &nbsp;&nbsp;<i>Cast</i><br>
134  * &nbsp;&nbsp;<i>NewExpr</i><br>
135  * &nbsp;&nbsp;<i>MethodCall</i><br>
136  * &nbsp;&nbsp;<i>Data</i><br>
137  * &nbsp;&nbsp;<i>Loader</i><br>
138  * &nbsp;&nbsp;<i>StringConcatenation</i> <br>
139  * <br>
140  * <i>Literal</i>:<br>
141  * &nbsp;&nbsp;<i>IntegerLiteral</i><br>
142  * &nbsp;&nbsp;<i>FloatingPointLiteral</i><br>
143  * &nbsp;&nbsp;<i>BooleanLiteral</i><br>
144  * &nbsp;&nbsp;<i>CharacterLiteral</i><br>
145  * &nbsp;&nbsp;<i>StringLiteral</i><br>
146  * &nbsp;&nbsp;<i>NullLiteral</i><br>
147  * <br>
148  * <i>TypeName</i>:<br>
149  * &nbsp;&nbsp;<i>ClassName</i><br>
150  * &nbsp;&nbsp;<i>ClassName</i> [ ]<br>
151  * &nbsp;&nbsp;<i>PrimitiveType</i><br>
152  * &nbsp;&nbsp;<i>PrimitiveType</i> [ ]<br>
153  * <br>
154  * <i>EntryName</i>:<br>
155  * &nbsp;&nbsp;<i>QualifiedIdentifier</i><br>
156  * <br>
157  * <i>ThisReference</i>:<br>
158  * &nbsp;&nbsp;this<br>
159  * <br>
160  * <i>FieldName</i>:<br>
161  * &nbsp;&nbsp;<i>QualifiedIdentifier</i> . <i>Identifier</i><br>
162  * <br>
163  * <i>Cast</i>:<br>
164  * &nbsp;&nbsp;( <i>TypeName</i> ) <i>Expr</i><br>
165  * <br>
166  * <i>NewExpr</i>:<br>
167  * &nbsp;&nbsp;new <i>QualifiedIdentifier</i> ( <i>ExprList</i><sub>opt</sub> )<br>
168  * &nbsp;&nbsp;new <i>QualifiedIdentifier</i> [ ] { <i>ExprList</i><sub>opt</sub> ,<sub>opt</sub> }<br>
169  * <br>
170  * <i>MethodCall</i>:<br>
171  * &nbsp;&nbsp;<i>StaticMethodName</i> ( <i>ExprList</i><sub>opt</sub> )<br>
172  * <br>
173  * <i>StaticMethodName</i>:<br>
174  * &nbsp;&nbsp;<i>QualifiedIdentifier</i> . <i>Identifier</i><br>
175  * <br>
176  * <i>ExprList</i>:<br>
177  * &nbsp;&nbsp;<i>Expr</i><br>
178  * &nbsp;&nbsp;<i>ExprList</i> , <i>Expr</i><br>
179  * <br>
180  * <i>Data</i>:<br>
181  * &nbsp;&nbsp;$data<br>
182  * <br>
183  * <i>Loader</i>:<br>
184  * &nbsp;&nbsp;$loader<br>
185  * <br>
186  * <i>StringConcatenation</i>:<br>
187  * &nbsp;&nbsp;<i>Expr</i> + <i>Expr</i> <br>
188  * </p></div>
189  *
190  * The syntax of each override option is as follows:
191  *
192  * <div>
193  * <i>Override</i>:<br>
194  * &nbsp;&nbsp;<i>EntryModifiers</i><sub>opt</sub> <i>FullyQualifiedEntryName</i> = <i>Expr</i><br>
195  * <br>
196  * <i>FullyQualifiedEntryName</i>:<br>
197  * &nbsp;&nbsp;<i>QualifiedIdentifier</i> . <i>Identifier</i><br>
198  * </div> <p>
199  *
200  * For example, a simple configuration source file might look like the
201  * following:
202  *
203  * <pre>
204  * import java.util.HashSet;
205  *
206  * com.acme.ContainerUtility {
207  *     container = new HashSet(containerSize);
208  *     containerSize = 33;
209  * }
210  * </pre> <p>
211  *
212  * The productions for <i>BooleanLiteral</i>, <i>CharacterLiteral</i>,
213  * <i>FloatingPointLiteral</i>, <i>Identifier</i>, <i>IntegerLiteral</i>,
214  * <i>NullLiteral</i>, <i>PrimitiveType</i>, <i>QualifiedIdentifier</i>, and
215  * <i>StringLiteral</i> are the same as the ones used in the
216  * JLS. <i>StringLiteral</i>s can refer to the values of system properties by
217  * using the syntax <code>${<i>propertyName</i>}</code> within the
218  * <i>StringLiteral</i>, and can refer to the file name separator character by
219  * using the syntax <code>${/}</code>. System property references cannot be
220  * nested. Expansion of system properties occurs when the entry is evaluated
221  * and, if there is a security manager, will result in its {@link
222  * SecurityManager#checkPropertyAccess checkPropertyAccess} method being called
223  * with the property name as its argument. Both <i>StringLiteral</i>s and
224  * <i>CharacterLiteral</i>s can use character and Unicode escape
225  * sequences. Standard comment syntax can also be used throughout. <p>
226  *
227  * Each <i>Import</i> specifies a class or group of classes which may be
228  * referred to using simple names, as specified in the JLS. Classes in the
229  * <code>java.lang</code> package are imported by default. <p>
230  *
231  * Each <i>Component</i> includes <i>Entries</i> which specify expressions to
232  * evaluate and return when <code>getEntry</code> is called with the associated
233  * component and entry name. More than one <i>Component</i> is allowed to
234  * specify the same name; all contribute entries for that component. For a
235  * given component, each entry name must be unique. If <i>EntryModifiers</i>
236  * contains the <code>static</code> keyword, then the entry is only evaluated
237  * once when it is first referenced. Otherwise, entries are evaluated at each
238  * reference, including each time an entry is referred to by another
239  * entry. Because static entries are only evaluated once (in the access control
240  * context of the first caller), care should be taken when passing instances of
241  * this class to callers with different access control contexts. If
242  * <i>EntryModifiers</i> contains the <code>private</code> keyword, then the
243  * entry may be referred to in other entries, but will not be considered by
244  * calls to <code>getEntry</code>, which will treat the entry name as not being
245  * defined. Entries may have reference, primitive, or <code>null</code>
246  * values. Entry values are converted to the requested type by <i>assignment
247  * conversion</i>, as defined in the JLS, with the restriction that the value
248  * is only considered a constant expression if it is either a
249  * <i>StringLiteral</i> with no system property references, another kind of
250  * <i>Literal</i>, or an <i>EntryName</i> that refers to another entry whose
251  * value is a constant expression. In particular, this restriction means that
252  * narrowing primitive conversions are not applied when the value is a
253  * reference to a static field, even if the field is a constant. <p>
254  *
255  * Override options are specified as the second and following elements of the
256  * <code>options</code> argument in this class's constructors. Each
257  * <i>Override</i> specifies a single entry, using the fully qualified name of
258  * the entry (<code><i>component</i>.<i>name</i></code>). The override replaces
259  * the matching entry in the configuration source, if any, including both its
260  * value and entry modifiers. Each <i>Override</i> option must specify a
261  * different <i>FullyQualifiedEntryName</i>. The contents of the <i>Expr</i>
262  * are evaluated in the context of any <i>Imports</i> defined in the
263  * configuration source. <p>
264  *
265  * The <i>Expr</i> for each <i>Entry</i> may be specified using a subset of the
266  * expression syntax in the Java programming language, supporting literals,
267  * references to static fields, casts, class instance creation (using the
268  * standard method invocation conversion and selection semantics, but not
269  * including creation of anonymous class instances), single dimensional array
270  * creation with an array initializer (but not multi-dimensional arrays or
271  * arrays declared with an explicit size), and static method invocation using a
272  * class name (also using standard method invocation conversion and selection
273  * semantics, but not permitting methods with a <code>void</code> return
274  * type). Expressions are interpreted in the unnamed package, although only
275  * public members may be accessed. The <code>this</code> expression may be used
276  * to refer to the containing <code>ConfigurationFile</code> instance
277  * itself. <p>
278  *
279  * The use of the <code>+</code> operator in a configuration source is also
280  * allowed, but it may be used only for string concatenation, as defined by the
281  * JLS.  Using the <code>+</code> operator in an arithmetic expression results 
282  * in a <code>ConfigurationException</code> being thrown.<p>  
283  *
284  * The <code>ConfigurationFile</code> class provides built-in support for two
285  * <i>special entry expressions</i>, which are <i>Identifier</i>s that
286  * start with <code>'$'</code>. The <code>$data</code> expression, of type
287  * {@link Object}, may be used to refer to the <code>data</code> argument
288  * specified in a call to <code>getEntry</code>. Only non-static entries may
289  * refer to <code>$data</code> or other entries that refer to
290  * <code>$data</code>. Calling <code>getEntry</code> without specifying
291  * <code>$data</code> results in a <code>ConfigurationException</code> being
292  * thrown if the associated entry refers to <code>$data</code>. The
293  * <code>$loader</code> expression, of type {@link ClassLoader}, may be used to
294  * refer to the <code>ClassLoader</code> specified when creating the
295  * <code>ConfigurationFile</code>. If the <code>ConfigurationFile</code> was
296  * created using the context class loader either by not specifying a class
297  * loader or by specifying <code>null</code> for the class loader, then the
298  * caller must be granted {@link
299  * RuntimePermission}<code>("getClassLoader")</code> in order to evaluate an
300  * entry that refers to <code>$loader</code>. Subclasses can provide support
301  * for additional special entry expressions by supplying implementations of
302  * {@link #getSpecialEntryType getSpecialEntryType} and {@link #getSpecialEntry
303  * getSpecialEntry}. <p>
304  *
305  * Entry expressions may also refer to other entries by name, using the simple
306  * entry name for entries within the same component, and the fully qualified
307  * entry name for entries in any component. A fully qualified name for which
308  * there is both an entry and a valid static field is interpreted as referring
309  * to the entry. An unqualified entry name for which there is an entry within
310  * the same component and is specified as a special entry expression will be
311  * interpreted as referring to the entry within that component. <p>
312  *
313  * Calls to the following methods are prohibited in order to avoid incorrect
314  * behavior because of their reliance on determining the
315  * <code>ClassLoader</code> or <code>AccessControlContext</code> of the caller:
316  *
317  * <ul>
318  * <li> <code>java.lang.Class.forName</code>
319  * <li> <code>java.lang.ClassLoader.getSystemClassLoader</code>
320  * <li> <code>java.lang.Package.getPackage</code>
321  * <li> <code>java.lang.Package.getPackages</code>
322  * <li> <code>java.lang.System.load</code>
323  * <li> <code>java.lang.System.loadLibrary</code>
324  * <li> <code>java.security.AccessController.doPrivileged</code>
325  * <li> <code>java.sql.DriverManager.deregisterDriver</code>
326  * <li> <code>java.sql.DriverManager.getConnection</code>
327  * <li> <code>java.sql.DriverManager.getDriver</code>
328  * <li> <code>java.sql.DriverManager.getDrivers</code>
329  * <li> <code>net.jini.security.Security.doPrivileged</code>
330  * </ul>
331  *
332  * Attempting to evaluate an entry that calls any of these methods results in
333  * <code>ConfigurationException</code> being thrown. Additional prohibited
334  * methods may be specified by providing a resource named
335  * "net/jini/config/resources/ConfigurationFile.moreProhibitedMethods". Each
336  * line in the resource file should specify an additional prohibited method,
337  * represented by the fully qualified class name, a <code>'.'</code>, and the
338  * method name for an additional prohibited method, with no spaces. The
339  * resource file must be encoded in UTF-8. <p>
340  *
341  * Any syntax error or problem reading from the configuration source or an
342  * override option results in a <code>ConfigurationException</code> being
343  * thrown. <p>
344  *
345  * If there is a security manager, the configuration source refers to the
346  * members of a class, and the class is in a named package, then this class
347  * calls the security manager's {@link SecurityManager#checkPackageAccess
348  * checkPackageAccess} method with the class's package. Making this call in
349  * <code>ConfigurationFile</code> insures that the check is made despite any
350  * decisions within reflection to skip the check based on the class of the
351  * caller, which in this case will be <code>ConfigurationFile</code> rather
352  * than its caller. Note that implementations are permitted to make calls to
353  * reflection to access class members at arbitrary stack depths relative to
354  * that of the caller of <code>ConfigurationFile</code>; applications using
355  * security managers with custom implementations of the {@link
356  * SecurityManager#checkMemberAccess checkMemberAccess} method should take this
357  * behavior into account.

547 	PushbackStreamTokenizer(Reader reader) {
548 	    super(reader);
549 	}
550 
551 	public int nextToken() throws IOException {
552 	    if (pushedBack) {
553 		/* Use values saved by pushBack */
554 		nval = savedNval;
555 		sval = savedSval;
556 		ttype = savedTtype;
557 		pushedBack = false;
558 	    } else {
559 		/* Save values for pushBack */
560 		savedNval = nval;
561 		savedSval = sval;
562 		savedTtype = ttype;
563 		savedLineno = lineno();
564 		super.nextToken();
565 		gotToken = true;
566 	    }
567 	    return ttype;
568 	}
569 
570 	public void pushBack() {
571 	    if (gotToken && !pushedBack) {
572 		/*
573 		 * Save values for next nextToken call and reestablish previous
574 		 * values.
575 		 */
576 		double tempNval = savedNval;
577 		savedNval = nval;
578 		nval = tempNval;
579 		String tempSval = savedSval;
580 		savedSval = sval;
581 		sval = tempSval;
582 		int tempTtype = savedTtype;
583 		savedTtype = ttype;
584 		ttype = tempTtype;
585 		pushedBack = true;
586 	    }
587 	}
588 
589 	public int lineno() {
590 	    /* Use the previous value if pushed back */
591 	    return pushedBack ? savedLineno : super.lineno();
592 	}

655     /** Represents an entry. */
656     private class Entry extends ParseNode {
657 	final String component;
658 	final String fullName;
659 	private final ParseNode node;
660 	final boolean isPrivate;
661 	final boolean isStatic;
662 	final boolean isOverride;
663 	private Class type;
664 	private boolean resolved;	/* resolve done */
665 	private boolean resolving;	/* resolve in progress */
666 	private boolean isConstant;
667 	private boolean refersToData;	/* computed in parse and resolve */
668 	private boolean evaluated;	/* static eval done */
669 	private boolean evaluating;	/* static eval in progress */
670 	private Object value;
671 
672 	Entry(String component,
673 	      String fullName,
674 	      boolean isPrivate,
675 	      boolean isStatic,
676 	      boolean isOverride,
677 	      int lineno,
678 	      Parser parser)
679 	    throws ConfigurationException, IOException
680 	{
681 	    super(lineno);
682 	    this.component = component;
683 	    this.fullName = fullName;
684 	    this.isPrivate = isPrivate;
685 	    this.isStatic = isStatic;
686 	    this.isOverride = isOverride;
687 	    node = parser.parseExpr(this);
688 	}
689 
690 	boolean getRefersToData() {
691 	    return refersToData;
692 	}
693 
694 	void setRefersToData() {
695 	    refersToData = true;
696 	}
697 
698         @Override
699 	<T> Class<T> resolve(Entry inEntry) throws ConfigurationException {
700 	    /*
701 	     * Grab a single lock when resolving any entry.  Until we know that
702 	     * the entry contains no textually circular references, which is
703 	     * determined during this resolve step, there could be a deadlock
704 	     * if two threads were to lock different entries at the same time
705 	     * that refer to each other.  -tjb[5.Sep.2002]
706 	     */
707 	    synchronized (resolveLock) {
708 		if (!resolved) {
709 		    if (resolving) {
710 			oops("entry with circular reference: " + fullName);
711 		    }
712 		    resolving = true;
713 		    try {
714 			type = node.resolve(this);
715 			isConstant = node.isConstant();
716 		    } finally {
717 			resolving = false;
718 		    }
719 		    resolved = true;
720 		}
721 	    }
722 	    return type;
723 	}
724 
725 	boolean isConstant() throws ConfigurationException {
726 	    resolve(this);
727 	    return isConstant;
728 	}
729 
730 	Object eval(Object data) throws ConfigurationException {
731 	    resolve(this);
732 	    if (!isStatic) {
733 		return node.eval(data);
734 	    }
735 	    /*
736 	     * Grab a separate lock when evaluating each static entry.  Using
737 	     * separate locks permits entries to be evaluated simultaneously if
738 	     * they don't refer to each other, which is important if the entry
739 	     * expressions perform network or other long running operations.
740 	     * Note that there is still the possibility of a deadlock if there
741 	     * is a circular dependency between static entries via recursive
742 	     * calls to getEntry.  -tjb[5.Sep.2002]
743 	     */
744 	    synchronized (this) {
745 		if (!evaluated) {
746 		    if (evaluating) {
747 			oops("entry with circular reference: " + fullName);
748 		    }
749 		    evaluating = true;
750 		    try {
751 			value = node.eval(NO_DATA);
752 		    } finally {
753 			evaluating = false;
754 		    }
755 		    evaluated = true;
756 		}
757                 return value;
758 	    }
759 	}

957 		type = Array.newInstance(type, 0).getClass();
958 	    }

981 		}
982 	    } else if ((type.isInterface() &&

1304 	/** Parses an Import, updating classImports and onDemandImports. */
1305 	private void parseImport() throws ConfigurationException, IOException {
1306 	    if (!entries.isEmpty()) {
1307 		oops("import follows entry");
1308 	    }
1309 	    /* Include asterisks */
1310 	    st.wordChars('*', '*');
1311 	    String value = token("import package or type");
1312 	    st.ordinaryChar('*');
1313 	    boolean onDemand = value.endsWith(".*");
1314 	    if (onDemand) {
1315 		value = value.substring(0, value.length() - 2);
1316 	    }
1317 	    if (!validQualifiedIdentifier(value)) {
1318 		syntax("import package or type");
1319 	    }
1320 	    token(';');
1321 	    if (onDemand) {
1322 		if (!onDemandImports.contains(value)) {
1323 		    Class c = findClassNoImports(value, st.lineno(), override);
1324 		    if (c != null && c.getName().indexOf('.') < 0) {
1325 			oops("import from unnamed package");		    
1326 		    }
1327 		    onDemandImports.add(value);
1328 		}
1329 	    } else {
1330 		int dot = value.lastIndexOf('.');
1331 		if (dot < 0) {
1332 		    oops("import from unnamed package");
1333 		}
1334 		Class c = findClassNoImports(value, st.lineno(), override);
1335 		if (c == null) {
1336 		    oops("class not found: " + value);
1337 		} else if (c.getName().indexOf('.') < 0) {
1338 		    oops("import from unnamed package");
1339 		}
1340 		String simple = value.substring(dot + 1);
1341 		if (classImports.containsKey(simple) &&
1342 		    !classImports.get(simple).equals(value))
1343 		{
1344 		    oops("conflicting imports: " + classImports.get(simple) +
1345 			 " and " + value);
1346 		}
1347 		classImports.put(simple, value);
1348 	    }
1349 	}

1372 	/** Parses an Entry. */
1373 	private void parseEntry(String component, String name)
1374 	    throws ConfigurationException, IOException
1375 	{
1376 	    boolean isPrivate = false;
1377 	    boolean isStatic = false;
1378 	    while (true) {
1379 		if ("private".equals(name)) {
1380 		    if (isPrivate) {
1381 			oops("duplicate 'private'");
1382 		    }
1383 		    isPrivate = true;
1384 		} else if ("static".equals(name)) {
1385 		    if (isStatic) {
1386 			oops("duplicate 'static'");
1387 		    }
1388 		    isStatic = true;
1389 		} else {
1390 		    break;
1391 		}
1392 		name = token("keyword 'static' or 'private', or entry name");
1393 	    }
1394 	    if (!validIdentifier(name)) {
1395 		oops("illegal entry name: " + name);
1396 	    }
1397 	    String fullName = component + '.' + name;
1398 	    if (entries.containsKey(fullName)) {
1399 		oops("duplicate entry name: " + name);
1400 	    }
1401 	    token('=');
1402 	    Entry entry = new Entry(component, fullName, isPrivate, isStatic,
1403 				    false /* isOverride */, st.lineno(), this);
1404 	    token(';');

1508 	/** Parses an Expr. */
1509 	ParseNode parseSubExpr(Entry inEntry)
1510 	    throws ConfigurationException, IOException
1511 	{
1512 	    int t = st.nextToken();
1513 	    String val = st.sval;
1514 	    if (t == '\'') {
1515 		if (val.length() != 1) {
1516 		    oops("invalid character: '" + val + "'");
1517 		}
1518 		return new Literal(Character.TYPE,
1519 				   Character.valueOf(val.charAt(0)), st.lineno());
1520 	    } else if (t == '"') {
1521 		return new StringLiteral(val, st.lineno());
1522 	    } else if (t == '(') {
1523 		int lineno = st.lineno();
1524 		String type = token("type");
1525 		t = st.nextToken();
1526 		boolean isArray = false;
1527 		if (t == '[') {
1528 		    isArray = true;
1529 		    token(']');
1530 		    t = st.nextToken();
1531 		}
1532 		if (t != ')') {
1533 		    syntax("')'");
1534 		}
1535 		return new Cast(type, isArray, parseExpr(inEntry), lineno);
1536 	    }
1537 	    st.pushBack();
1538 	    val = token("expression");
1539 	    if (Character.isDigit(val.charAt(0)) || val.charAt(0) == '-') {
1540 		return getNumber(val);
1541 	    } else if ("true".equals(val)) {
1542 		return new Literal(Boolean.TYPE, Boolean.TRUE, st.lineno());
1543 	    } else if ("false".equals(val)) {
1544 		return new Literal(Boolean.TYPE, Boolean.FALSE, st.lineno());
1545 	    } else if ("null".equals(val)) {
1546 		return new Literal(null, null, st.lineno());
1547 	    } else if ("new".equals(val)) {
1548 		return parseNewInstance(inEntry);
1549 	    } else if ("this".equals(val)) {
1550 		return new ThisRef(st.lineno());
1551 	    } else if (val.endsWith(".class")) {
1552 		val = val.substring(0, val.length() - 6);
1553 		if (findPrimitiveClass(val) == null &&
1554 		    !validQualifiedIdentifier(val))
1555 		{
1556 		    oops("illegal class name: " + val);
1557 		}
1558 		return new ClassLiteral(val, false, st.lineno());
1559 	    }
1560 	    t = st.nextToken();
1561 	    if (t == '[') {
1562 		int lineno = st.lineno();
1563 		token(']');
1564 		if (!".class".equals(token("'.class'"))) {
1565 		    syntax("'.class'");
1566 		} else if (findPrimitiveClass(val) == null &&
1567 			   !validQualifiedIdentifier(val))
1568 		{
1569 		    oops("illegal class name: " + val);
1570 		}
1571 		return new ClassLiteral(val, true, lineno);
1572 	    }
1573 	    st.pushBack();
1574 	    if (!validQualifiedIdentifier(val)) {
1575 		if (t == '(') {
1576 		    oops("illegal method name: " + val);
1577 		} else {
1578 		    oops("illegal field or entry name: " + val);
1579 		}
1580 	    }
1581 	    if (t == '(') {
1582 		return parseMethodCall(inEntry, val);
1583 	    } else {
1584 		if ("$data".equals(val)) {
1585 		    if (inEntry.isStatic) {
1586 			oops("cannot use '$data' in a static entry");
1587 		    }
1588 		    inEntry.setRefersToData();
1589 		}
1590 		return new NameRef(inEntry.component, val, st.lineno());
1591 	    }
1592 	}

1600 	    char c = val.charAt(val.length() - 1);

1614 			Double.TYPE,
1615 			Double.valueOf(val.substring(0, val.length() - 1)),

1620 		{
1621 		    return new Literal(
1622 			Double.TYPE, Double.valueOf(val), st.lineno());

1630 	    }

1708 			( (result.lastIndexOf(".") != result.length() - 1 ) && !st.sval.startsWith(".")))

2701     /**
2702      * Expands properties embedded in a string with ${some.property.name}. Also
2703      * treats ${/} as ${file.separator}.
2704      */
2705     String expandStringProperties(String value, int lineno)
2706 	throws ConfigurationException
2707     {
2708 	int p = value.indexOf("${", 0);
2709 	if (p == -1) {
2710 	    return value;
2711 	}
2712 	int max = value.length();
2713 	StringBuffer sb = new StringBuffer(max);
2714 	int i = 0;		/* Index of last character we copied */
2715 	while (true) {
2716 	    if (p > i) {
2717 		/* Copy in anything before the special stuff */
2718 		sb.append(value.substring(i, p));
2719 		i = p;
2720 	    }
2721 	    int pe = value.indexOf('}', p + 2);
2722 	    if (pe == -1) {
2723 		/* No matching '}' found, just add in as normal text */
2724 		sb.append(value.substring(p, max));
2725 		break;
2726 	    }
2727 	    String prop = value.substring(p + 2, pe);
2728 	    if (prop.equals("/")) {
2729 		sb.append(File.separatorChar);
2730 	    } else {
2731 		try {
2732 		    String val = prop.length() == 0 ? null
2733 			: System.getProperty(prop);
2734 		    if (val != null) {
2735 			sb.append(val);
2736 		    } else {
2737 			oops("problem expanding system property '" + prop + "'",
2738 			     lineno, override);
2739 		    }
2740 		} catch (SecurityException e) {
2741 		    oops("problem expanding system property '" + prop + "'",
2742 			 lineno, override, e);
2743 		}
2744 	    }
2745 	    i = pe + 1;
2746 	    p = value.indexOf("${", i);
2747 	    if (p == -1) {
2748 		/* No more to expand -- copy in any extra. */
2749 		if (i < max) {
2750 		    sb.append(value.substring(i, max));
2751 		} 
2752 		/* Break out of loop */
2753 		break;
2754 	    }
2755 	}
2756 	return sb.toString();
2757     }
