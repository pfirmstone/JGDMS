38  /**
39   * Provides a standard means for obtaining {@link Configuration} instances,
40   * using a configurable provider. This class cannot be instantiated. The
41   * configuration provider can be specified by providing a resource named
42   * "META-INF/services/net.jini.config.Configuration" containing the name of the
43   * provider class. If multiple resources with that name are available, then the
44   * one used will be the last one returned by {@link ClassLoader#getResources
45   * ClassLoader.getResources}. If the resource is not found, the {@link
46   * ConfigurationFile} class is used. <p>
47   *
48   * Downloaded code can specify its own class loader in a call to {@link
49   * #getInstance(String[],ClassLoader) getInstance(String[], ClassLoader)} in
50   * order to use the configuration provider specified in the JAR file from which
51   * it was downloaded. <p>
52   *
53   * The provider class must be a public, non-abstract class that implements
54   * <code>Configuration</code>, and has a public constructor that has
55   * <code>String[]</code> and <code>ClassLoader</code> parameters. The
56   * <code>String[]</code> parameter supplies provider-specific options, for
57   * example specifying a source location or values for particular entries. The
58   * constructor should throw {@link ConfigurationNotFoundException} if the
59   * configuration specified by the <code>String[]</code> options argument cannot
60   * be found, or if <code>null</code> was specified for the options argument and
61   * the class does not provide defaults. The constructor should use the class
62   * loader argument when loading resources and classes, and should interpret a
63   * <code>null</code> class loader argument as signifying the context class
64   * loader. <p>
65   *
66   * The resource file should contain the fully qualified name of the provider
67   * class. Space and tab characters surrounding each name, as well as blank
68   * lines, are ignored. The comment character is <tt>'#'</tt> (<tt>0x23</tt>);
69   * on each line, all characters following the first comment character are
70   * ignored. The resource file must be encoded in UTF-8.
71   *
72   * @author Sun Microsystems, Inc.
73   * @since 2.0
74   *
75   * <!-- Implementation Specifics -->
76   *
77   * This implementation uses the {@link Logger} named
78   * <code>net.jini.config</code> to log information at the following logging
79   * levels: <br>
80   *
81   * <table border="1" cellpadding="5" summary="Describes logging performed by
82   *	  the ConfigurationProvider class at different logging levels">
83   *
84   * <caption><b><code>net.jini.config</code></b></caption>
85   *
86   * <tr> <th scope="col"> Level <th scope="col"> Description
87   *
88   * <tr> <td> {@link Levels#FAILED FAILED} <td> problems getting a configuration
89   *
90   * </table>
91   */
92  public class ConfigurationProvider {
93  
94      private static final String resourceName =
95  	"META-INF/services/" + Configuration.class.getName();
96  
97      /** Config logger. */
98      private static final Logger logger = Logger.getLogger("net.jini.config");

100     /** This class cannot be instantiated. */
101     private ConfigurationProvider() {
102 	throw new AssertionError();
103     }

105     /**
106      * Creates and returns an instance of the configuration provider, using the
107      * specified options. Specifying <code>null</code> for <code>options</code>
108      * uses provider-specific default options, if available. Uses the current
109      * thread's context class loader to load resources and classes, and
110      * supplies <code>null</code> as the class loader to the provider
111      * constructor.
112      *
113      * @param options values to use when constructing the configuration, or
114      * <code>null</code> if default options should be used
115      * @return an instance of the configuration provider constructed with the
116      * specified options and <code>null</code> for the class loader
117      * @throws ConfigurationNotFoundException if <code>options</code> specifies
118      * a source location that cannot be found, or if <code>options</code> is
119      * <code>null</code> and the provider does not supply default options
120      * @throws ConfigurationException if an I/O exception occurs; if there are
121      * problems with the <code>options</code> argument or the format of the
122      * contents of any source location it specifies; if there is a problem with
123      * the contents of the resource file that names the configuration provider;
124      * if the configured provider class does not exist, is not public, is
125      * abstract, does not implement <code>Configuration</code>, or does not
126      * have a public constructor that has <code>String[]</code> and
127      * <code>ClassLoader</code> parameters; if the calling thread does not have
128      * permission to obtain information from the specified source; or if the
129      * provider does not have permission to access the context class
130      * loader. Any <code>Error</code> thrown while creating the provider
131      * instance is propagated to the caller; it is not wrapped in a
132      * <code>ConfigurationException</code>.
133      */
134     public static Configuration getInstance(String[] options)
135 	throws ConfigurationException
136     {
137 	return getInstance(options, null);
138     }

140     /**
141      * Creates and returns an instance of the configuration provider, using the
142      * specified options and class loader. Specifying <code>null</code> for
143      * <code>options</code> uses provider-specific default options, if
144      * available. Uses the specified class loader to load resources and
145      * classes, or the current thread's context class loader if <code>cl</code>
146      * is <code>null</code>. Supplies <code>cl</code> as the class loader to
147      * the provider constructor. <p>
148      *
149      * Downloaded code can specify its own class loader in a call to this
150      * method in order to use the configuration provider specified in the JAR
151      * file from which it was downloaded.
152      *
153      * @param options values to use when constructing the configuration, or
154      * <code>null</code> if default options should be used
155      * @param cl the class loader to load resources and classes, and to pass
156      * when constructing the provider. If <code>null</code>, uses the context
157      * class loader.
158      * @return an instance of the configuration provider constructed with the
159      * specified options and class loader
160      * @throws ConfigurationNotFoundException if <code>options</code> specifies
161      * a source location that cannot be found, or if <code>options</code> is
162      * <code>null</code> and the provider does not supply default options
163      * @throws ConfigurationException if an I/O exception occurs; if there are
164      * problems with the <code>options</code> argument or the format of the
165      * contents of any source location it specifies; if there is a problem with
166      * the contents of the resource file that names the configuration provider;
167      * if the configured provider class does not exist, is not public, is
168      * abstract, does not implement <code>Configuration</code>, or does not
169      * have a public constructor that has <code>String[]</code> and
170      * <code>ClassLoader</code> parameters; if the calling thread does not have
171      * permission to obtain information from the specified source; or if
172      * <code>cl</code> is <code>null</code> and the provider does not have
173      * permission to access the context class loader. Any <code>Error</code>
174      * thrown while creating the provider instance is propagated to the caller;
175      * it is not wrapped in a <code>ConfigurationException</code>.
176      */
177     public static Configuration getInstance(String[] options, ClassLoader cl)
178 	throws ConfigurationException
179     {
180 		/*
181 		  ClassLoader resourceLoader = (cl != null) ? cl :
182 		  (ClassLoader) Security.doPrivileged(
183 		  new PrivilegedAction() {
184 		  public Object run() {
185 		  return Thread.currentThread().getContextClassLoader();
186 		  }
187 		  });
188 		*/
189 		ClassLoader resourceLoader=cl;
190 		if (resourceLoader == null) {
191 			logger.fine("Null class loader provided, fetching context class loader...");
192 			resourceLoader =
193                             (ClassLoader) Security.doPrivileged(
194                                 new PrivilegedAction() {
195                                     public Object run() {
196                                         return Thread.currentThread().getContextClassLoader();
197                                     }
198                                 });
199 			logger.fine("...resource class loader is now: " + resourceLoader);
200 		}
201 	final ClassLoader finalResourceLoader = (resourceLoader == null)
202 	    ? Utilities.bootstrapResourceLoader : resourceLoader;
203 	logger.fine("Final resource class loader is: " + finalResourceLoader);
204 	String cname = null;
205 	ConfigurationException configEx = null;
206 	try {
207 	    cname = (String) Security.doPrivileged(
208 		new PrivilegedExceptionAction() {
209 		    public Object run()
210 			throws ConfigurationException, IOException
211 		    {
212 			URL resource = null;
213 			Enumeration providers = 
214 			    finalResourceLoader.getResources(resourceName);
215 			while (providers.hasMoreElements()) {
216 			    resource = (URL) providers.nextElement();
217 			}
218 			return (resource == null)
219 			    ? null : getProviderName(resource);
220 		    }
221 		});
222 	} catch (PrivilegedActionException e) {
223 	    Exception e2 = e.getException();
224 	    if (e2 instanceof ConfigurationException) {
225 		configEx = (ConfigurationException) e2;
226 	    } else {
227 		configEx = new ConfigurationException(
228 		    "problem accessing provider resources", e2);
229 	    }
230 	} catch (RuntimeException e) {
231 	    configEx = new ConfigurationException(
232 		"problem accessing provider resources", e);
233 	}
234 	if (configEx != null) {
235 	    logger.log(Levels.FAILED, "getting configuration provider throws",
236 		       configEx);
237 	    throw configEx;
238 	}
239 	if (cname == null) {
240 	    return new ConfigurationFile(options, cl);
241 	}
242 	try {
243 	    Class cls = LoadClass.forName(cname, true, resourceLoader);
244 	    if (!Configuration.class.isAssignableFrom(cls)) {
245 		configEx = new ConfigurationException(
246 		    "provider class " + cname +
247 		    " does not implement Configuration");
248 	    } else {
249 		Constructor cons = cls.getConstructor(
250 		    new Class[] { String[].class, ClassLoader.class });
251 		return (Configuration) cons.newInstance(
252 		    new Object[] { options, cl });
253 	    }
254 	} catch (ClassNotFoundException e) {
255 	    configEx = new ConfigurationException(
256 		"provider class " + cname + " not found");
257 	} catch (NoSuchMethodException e) {
258 	    configEx = new ConfigurationException(
259 		"provider class " + cname +
260 		" does not have the right constructor");
261 	} catch (IllegalAccessException e) {
262 	    configEx = new ConfigurationException(
263 		"provider class " + cname + " constructor is not public");
264 	} catch (InstantiationException e) {
265 	    configEx = new ConfigurationException(
266 		"provider class " + cname + " is abstract");
267 	} catch (InvocationTargetException e) {
268 	    Throwable t = e.getTargetException();
269 	    if (t instanceof Error) {
270 		throw (Error) t;
271 	    } else if (t instanceof ConfigurationException) {
272 		configEx = (ConfigurationException) t;
273 	    } else {
274 		configEx = new ConfigurationException(
275 		    "problem with provider class", t);
276 	    }
277 	} catch (RuntimeException e) {
278 	    configEx = new ConfigurationException(
279 		"problem with provider class", e);
280 	}
281 	logger.log(Levels.FAILED, "getting configuration throws", configEx);
282 	throw configEx;
283     }

285     /**
286      * Returns the configuration provider class name specified in the contents
287      * of the URL.
288      */
289     private static String getProviderName(URL url)
290 	throws ConfigurationException, IOException
291     {
292 	InputStream in = null;
293 	try {
294 	    in = url.openStream();
295 	    final BufferedReader reader =
296 		new BufferedReader(new InputStreamReader(in, "utf-8"));
297 	    try {
298 		    String result = null;
299 		    while (true) {
300 			    String line = reader.readLine();
301 			    if (line == null) {
302 				    break;
303 			    }
304 			    int commentPos = line.indexOf('#');
305 			    if (commentPos >= 0) {
306 				    line = line.substring(0, commentPos);
307 			    }
308 			    line = line.trim();
309 			    int len = line.length();
310 			    if (len != 0) {
311 				    if (result != null) {
312 					    throw new ConfigurationException(
313 							    "resource specifies multiple providers");
314 				    }
315 				    result = line;
316 			    }
317 		    }
318 		    if (result == null) {
319 			    throw new ConfigurationException(
320 					    "resource specifies no providers");
321 		    }
322 		    return result;
323 	    } finally {
324 		    reader.close();
325 	    }
326 	} finally {
327 	    if (in != null) {
328 		try {
329 		    in.close();
330 		} catch (IOException e) {
331 		}
332 	    }
333 	}
334     }
