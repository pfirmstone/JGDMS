31  /**
32   * Represents a constraint on the client, such that if the client
33   * authenticates itself, then it may only authenticate itself as principals
34   * that are instances of one or more of the specified classes. The mechanisms
35   * and credentials used to authenticate the client as those principals are not
36   * specified by this constraint. A client can use this constraint to limit how
37   * much of its identity is exposed in remote calls. For example, if the
38   * client's subject contains sufficient information to authenticate as two
39   * distinct principals of different types, the client might wish to limit
40   * authentication to just one of the two types, if the client believes the
41   * other is unnecessary for authorization at the server and wants to avoid
42   * revealing that part of its identity to the server.
43   * <p>
44   * The use of an instance of this class does not directly imply a
45   * {@link ClientAuthentication#YES} constraint; that must be specified
46   * separately to ensure that the client actually authenticates itself.
47   * Because this constraint is conditional on client authentication, it does
48   * not conflict with {@link ClientAuthentication#NO}.
49   *
50   * @author Sun Microsystems, Inc.
51   * @see ClientAuthentication
52   * @see ClientMaxPrincipal
53   * @see ClientMinPrincipal
54   * @see ClientMinPrincipalType
55   * @see net.jini.security.AuthenticationPermission
56   * @since 2.0
57   */
58  @AtomicSerial
59  public final class ClientMaxPrincipalType
60  				implements InvocationConstraint, Serializable
61  {
62      private static final long serialVersionUID = -2521616888337674811L;
63  
64      /**
65       * @serialField classes Class[] The classes.
66       */
67      private static final ObjectStreamField[] serialPersistentFields = {
68          new ObjectStreamField("classes", Class[].class, true)
69      };
70  
71      /**
72       * The classes.
73       */
74      private final Class[] classes;
75  
76      /**
77       * Creates a constraint containing the specified class.  This
78       * constructor is equivalent to calling a constructor with a
79       * single-element array containing the specified class.
80       *
81       * @param clazz the class
82       * @throws NullPointerException if the argument is <code>null</code>
83       * @throws IllegalArgumentException if the argument is a primitive type,
84       * an array type, or a <code>final</code> class that does not have
85       * {@link java.security.Principal} as a superinterface
86       */
87      public ClientMaxPrincipalType(Class clazz) {
88  	this(check(clazz), true);
89      }
90      
91      public ClientMaxPrincipalType(GetArg arg) throws IOException{
92  	this(verify(arg.get("classes", null, Class[].class)), true);
93      }
94      
95      private static Class[] check(Class clazz){
96  	Constraint.verify(clazz);
97  	return new Class[]{clazz};
98      }
99      
100     private static Class[] verify(Class[] classes) throws InvalidObjectException{
101 	classes = classes.clone();
102 	Constraint.verify(classes);
103 	return classes;
104     }
105     
106     private ClientMaxPrincipalType(Class[] clazzs, boolean check){
107 	classes = clazzs;
108     }
109 
110     /**
111      * Creates a constraint containing the specified classes, with redundant
112      * classes removed. Redundant classes are removed as follows: for any two
113      * specified classes <code>c1</code> and <code>c2</code>, if
114      * <code>c1.isAssignableFrom(c2)</code> is <code>true</code>, then
115      * <code>c2</code> is removed. That is, duplicates and subtypes are
116      * removed. The argument passed to the constructor is neither modified
117      * nor retained; subsequent changes to that argument have no effect on
118      * the instance created.
119      *
120      * @param classes the classes
121      * @throws IllegalArgumentException if the argument is empty, or if any
122      * element is a primitive type, an array type, or a <code>final</code>
123      * class that does not have {@link java.security.Principal} as a
124      * superinterface
125      * @throws NullPointerException if the argument is <code>null</code> or
126      * any element is <code>null</code>
127      */
128     public ClientMaxPrincipalType(Class[] classes) {
129 	this(Constraint.reduce(classes, true), true);
130     }
131 
132     /**
133      * Creates a constraint containing the specified classes, with redundant
134      * classes removed. Redundant classes are removed as follows: for any two
135      * specified classes <code>c1</code> and <code>c2</code>, if
136      * <code>c1.isAssignableFrom(c2)</code> is <code>true</code>, then
137      * <code>c2</code> is removed. That is, duplicates and subtypes are
138      * removed. The argument passed to the constructor is neither modified
139      * nor retained; subsequent changes to that argument have no effect on
140      * the instance created.
141      *
142      * @param c the classes
143      * @throws IllegalArgumentException if the argument is empty, or if any
144      * element is not a <code>Class</code>, or is a primitive type, an array
145      * type, or a <code>final</code> class that does not have
146      * {@link java.security.Principal} as a superinterface
147      * @throws NullPointerException if the argument is <code>null</code> or
148      * any element is <code>null</code>
149      */
150     public ClientMaxPrincipalType(Collection c) {
151 	this(Constraint.reduce(c, true), true);
152     }
153 
154     /**
155      * Returns an immutable set of all of the classes. Any attempt to
156      * modify the set results in an {@link UnsupportedOperationException}
157      * being thrown.
158      *
159      * @return an immutable set of all of the classes
160      */
161     public Set elements() {
162 	return new ArraySet(classes);
163     }
164 
165     /**
166      * Returns a hash code value for this object.
167      */
168     public int hashCode() {
169 	return (ClientMaxPrincipalType.class.hashCode() +
170 		Constraint.hash(classes));
171     }
172 
173     /**
174      * Two instances of this class are equal if they have the same classes
175      * (ignoring order).
176      */
177     public boolean equals(Object obj) {
178 	return (obj instanceof ClientMaxPrincipalType &&
179 		Constraint.equal(classes,
180 				 ((ClientMaxPrincipalType) obj).classes));
181     }
182 
183     /**
184      * Returns a string representation of this object.
185      */
186     public String toString() {
187 	return "ClientMaxPrincipalType" + Constraint.toString(classes);
188     }
189 
190     /**
191      * Verifies that there is at least one class, that none of the classes
192      * is <code>null</code>, primitive types, array types, or
193      * <code>final</code> classes that do not have <code>Principal</code> as
194      * a superinterface, and that no class is assignable to any other class.
195      *
196      * @throws java.io.InvalidObjectException if there are no classes, or
197      * any class is <code>null</code>, a primitive type, array type, or
198      * <code>final</code> class that does not have <code>Principal</code> as
199      * a superinterface, or if any class is assignable to any other class
200      * @param s ObjectInputStream
201      * @throws ClassNotFoundException if class not found.
202      * @throws IOException if de-serialization problem occurs.
203      */
204     private void readObject(ObjectInputStream s)
205 	throws IOException, ClassNotFoundException
206     {
207 	s.defaultReadObject();
208 	Constraint.verify(classes);
209     }
