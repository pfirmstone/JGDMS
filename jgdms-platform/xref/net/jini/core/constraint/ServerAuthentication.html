19  package net.jini.core.constraint;
20  
21  import java.io.IOException;
22  import java.io.Serializable;
23  import org.apache.river.api.io.AtomicSerial;
24  import org.apache.river.api.io.AtomicSerial.GetArg;
25  
26  /**
27   * Represents a constraint on authentication of the server to the client.
28   * <p>
29   * Network authentication by a server (to a client) is implementation-specific,
30   * but typically is also scoped and controlled by a <code>Subject</code>. The
31   * server subject normally is contained in the
32   * {@link net.jini.export.Exporter} used to export that remote
33   * object and is taken from the current thread when the exporter is
34   * constructed. However, a server might use one subject to control its local
35   * execution and a different subject to control its network authentication.
36   * A server generally must have permission (such as
37   * {@link net.jini.security.AuthenticationPermission}) to authenticate itself
38   * to clients.
39   * <p>
40   * It is important to understand that specifying
41   * <code>ServerAuthentication.YES</code> as a requirement <i>does not</i>
42   * ensure that a server is to be trusted; it <i>does</i> ensure that the
43   * server authenticates itself as someone, but it does not ensure that the
44   * server authenticates itself as anyone in particular. Without knowing who
45   * the server authenticated itself as, there is no basis for actually
46   * trusting the server. The client generally needs to specify a
47   * {@link ServerMinPrincipal} requirement in addition, or else verify
48   * that the server has specified a satisfactory
49   * <code>ServerMinPrincipal</code> requirement for each of the methods that
50   * the client cares about.
51   * <p>
52   * Serialization for this class is guaranteed to produce instances that are
53   * comparable with <code>==</code>.
54   *
55   * @author Sun Microsystems, Inc.
56   * @see ServerMinPrincipal
57   * @see net.jini.security.AuthenticationPermission
58   * @since 2.0
59   */
60  @AtomicSerial
61  public final class ServerAuthentication
62  				implements InvocationConstraint, Serializable
63  {
64      private static final long serialVersionUID = 2837982503744131014L;
65  
66      /**
67       * Authenticate the server to the client. The mechanisms and credentials
68       * used to authenticate the server are not specified by this constraint.
69       */
70      public static final ServerAuthentication YES =
71  					new ServerAuthentication(true);
72      /**
73       * Do not authenticate the server to the client, so that the server
74       * remains anonymous.
75       */
76      public static final ServerAuthentication NO =
77  					new ServerAuthentication(false);
78  
79      /**
80       * <code>true</code> for <code>YES</code>, <code>false</code> for
81       * <code>NO</code>
82       *
83       * @serial
84       */
85      private final boolean val;
86      
87      public ServerAuthentication(GetArg arg) throws IOException{
88  	this(arg.get("val", true));
89      }
90  
91      /**
92       * Simple constructor.
93       *
94       * @param val <code>true</code> for <code>YES</code>, <code>false</code>
95       * for <code>NO</code>
96       */
97      private ServerAuthentication(boolean val) {
98  	this.val = val;
99      }
100 
101     /**
102      * Returns a string representation of this object.
103      */
104     public String toString() {
105 	return val ? "ServerAuthentication.YES" : "ServerAuthentication.NO";
106     }
107 
108     /**
109      * Canonicalize so that <code>==</code> can be used.
110      * @return true for YES, false for NO.
111      */
112     private Object readResolve() {
113 	return val ? YES : NO;
114     }
115 }
