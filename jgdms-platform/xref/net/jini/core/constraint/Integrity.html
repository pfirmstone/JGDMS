19  package net.jini.core.constraint;
20  
21  import java.io.IOException;
22  import java.io.Serializable;
23  import org.apache.river.api.io.AtomicSerial;
24  import org.apache.river.api.io.AtomicSerial.GetArg;
25  
26  /**
27   * Represents a constraint on the integrity of message contents, covering not
28   * only data transmitted in band as part of the remote call itself, but also
29   * out-of-band downloaded code. If an integrity violation on in-band data is
30   * detected during a remote call, a {@link java.rmi.RemoteException} will be
31   * thrown (in the client or in the server, depending on which side detected
32   * the violation). If an integrity violation on out-of-band data is detected,
33   * an {@link java.io.IOException} will be thrown at the point where the data
34   * is downloaded.
35   * <p>
36   * Although most of the data for a remote call is transmitted in band as part
37   * of the call itself, code is downloaded out of band, based on codebase URLs
38   * that are transmitted in band. For a remote call to have integrity, the
39   * out-of-band code as well as the in-band data must have integrity. A proxy
40   * implementation that provides for integrity must ensure the integrity of
41   * both code and data.
42   * <p>
43   * Code signing is difficult to use for this purpose if the classes span more
44   * than a single package (because individual files are signed rather than the
45   * entire JAR file being signed, and the only automatic enforcement is that
46   * classes in a single package all have the same signers), or if the code
47   * references bundled resources (because there is no way to determine the
48   * signers of a resource). A better technique is to use codebase URLs that
49   * provide content integrity, such as
50   * <a href="../../url/httpmd/package-summary.html#package_description">HTTPMD</a>
51   * or HTTPS URLs. If integrity-protecting codebase URLs are used, and the URLs
52   * themselves are sent as part of the integrity-protected in-band data, the
53   * result is complete object integrity. Because out-of-band communication is
54   * used, integrity-protecting URLs must either contain sufficient information
55   * to independently verify integrity (as is the case with HTTPMD URLs), or
56   * must contain sufficient information to authenticate the origin of the
57   * content and use sufficient means to maintain content integrity in transit
58   * (as is the case with HTTPS URLs).
59   * <p>
60   * Serialization for this class is guaranteed to produce instances that are
61   * comparable with <code>==</code>.
62   *
63   * @author Sun Microsystems, Inc.
64   * @see net.jini.security.Security#verifyCodebaseIntegrity
65   * Security.verifyCodebaseIntegrity
66   * @since 2.0
67   */
68  @AtomicSerial
69  public final class Integrity implements InvocationConstraint, Serializable {
70      private static final long serialVersionUID = 418483423937969897L;
71  
72      /**
73       * Detect when message contents (both requests and replies) have been
74       * altered by third parties, and if detected, refuse to process the
75       * message and throw an exception. The mechanisms used to maintain
76       * integrity are not specified by this constraint.
77       */
78      public static final Integrity YES = new Integrity(true);
79      /**
80       * Do not detect when message contents have been altered by third parties.
81       * Normally this constraint should not be used, as many secure
82       * communication mechanisms have integrity mechanisms that cannot be
83       * disabled.
84       */
85      public static final Integrity NO = new Integrity(false);
86  
87      /**
88       * <code>true</code> for <code>YES</code>, <code>false</code> for
89       * <code>NO</code>
90       *
91       * @serial
92       */
93      private final boolean val;
94  
95      /**
96       * Simple constructor.
97       *
98       * @param val <code>true</code> for <code>YES</code>, <code>false</code>
99       * for <code>NO</code>
100      */
101     private Integrity(boolean val) {
102 	this.val = val;
103     }
104 
105     Integrity(GetArg arg) throws IOException {
106 	this(arg.get("val", true));
107     }
108 
109     /**
110      * Returns a string representation of this object.
111      */
112     public String toString() {
113 	return val ? "Integrity.YES" : "Integrity.NO";
114     }
115 
116     /**
117      * Canonicalize so that <code>==</code> can be used.
118      * @return true for YES, false for NO.
119      */
120     private Object readResolve() {
121 	return val ? YES : NO;
122     }
123 }
