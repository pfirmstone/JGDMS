18  package net.jini.core.transaction.server;
19  
20  import java.io.IOException;
21  import java.rmi.Remote;
22  import java.rmi.RemoteException;
23  import net.jini.core.transaction.*;
24  import org.apache.river.api.io.AtomicSerial;
25  import org.apache.river.api.io.AtomicSerial.GetArg;
26  import org.apache.river.api.io.Valid;
27  
28  
29  /**
30   * Class implementing the <code>Transaction</code> interface, for use with
31   * transaction participants that implement the default transaction semantics.
32   *
33   * @author Sun Microsystems, Inc.
34   *
35   * @see net.jini.core.transaction.Transaction
36   * @see NestableServerTransaction
37   * @see TransactionManager
38   * @see net.jini.core.transaction.TransactionFactory
39   *
40   * @since 1.0
41   */
42  @AtomicSerial
43  public class ServerTransaction implements Transaction, java.io.Serializable {
44      static final long serialVersionUID = 4552277137549765374L;
45  
46      /**
47       * The transaction manager.
48       * @serial
49       */
50      public final TransactionManager mgr;
51      /**
52       * The transaction id.
53       * @serial
54       */
55      public final long id;
56      
57      public ServerTransaction(GetArg arg) throws IOException{
58  	this(Valid.notNull(arg.get("mgr", null, TransactionManager.class), 
59  		"TransactionManager cannot be null"),
60  	     arg.get("id", 0L)
61  	);
62      }
63  
64      /**
65       * Simple constructor.  Clients should not call this directly, but
66       * should instead use <code>TransactionFactory</code>.
67       *
68       * @param mgr the manager for this transaction
69       * @param id the transaction id
70       */
71      public ServerTransaction(TransactionManager mgr, long id) {
72  	this.mgr = mgr;
73  	this.id = id;
74      }
75  
76      // inherit javadoc
77      public int hashCode() {
78          return (int) id ^ mgr.hashCode();
79      }
80  
81      /**
82       * Two instances are equal if they have the same transaction manager
83       * and the same transaction id.
84       */
85      public boolean equals(Object other) {
86  	if (this == other)
87  	   return true;
88  
89  	if (!(other instanceof ServerTransaction)) 
90  	   return false;
91  
92          ServerTransaction t = (ServerTransaction) other;
93          return (id == t.id && mgr.equals(t.mgr));
94      }
95  
96      // inherit javadoc
97      public void commit()
98  	throws UnknownTransactionException, CannotCommitException,
99  	       RemoteException
100     {
101         mgr.commit(id);
102     }
103 
104     // inherit javadoc
105     public void commit(long waitFor)
106         throws UnknownTransactionException, CannotCommitException,
107 	       TimeoutExpiredException, RemoteException
108     {
109 	mgr.commit(id, waitFor);
110     }
111 
112     // inherit javadoc
113     public void abort()
114 	throws UnknownTransactionException, CannotAbortException,
115 	       RemoteException
116     {
117 	mgr.abort(id);
118     }
119 
120     // inherit javadoc
121     public void abort(long waitFor)
122 	throws UnknownTransactionException, CannotAbortException,
123 	       TimeoutExpiredException, RemoteException
124     {
125         mgr.abort(id, waitFor);
126     }
127 
128     /**
129      * Join the transaction. The <code>crashCount</code> marks the state of
130      * the storage used by the participant for transactions. If the
131      * participant attempts to join a transaction more than once, the crash
132      * counts must be the same. Each system crash or other event that
133      * destroys the state of the participant's unprepared transaction storage
134      * must cause the crash count to increase by at least one.
135      *
136      * @param part the participant joining the transaction
137      * @param crashCount the participant's current crash count
138      *
139      * @throws UnknownTransactionException if the transaction 
140      *         is unknown to the transaction manager, either 
141      *         because the transaction ID is incorrect or because the 
142      *         transaction is complete and its state has been 
143      *         discarded by the manager.
144      * @throws CannotJoinException if the transaction is known 
145      *         to the manager but is no longer active.
146      * @throws CrashCountException if the crash count provided 
147      *         for the participant differs from the crash count 
148      *         in a previous invocation of the same pairing of 
149      *         participant and transaction
150      * @throws RemoteException if there is a communication error
151      */
152     public void join(TransactionParticipant part, long crashCount)
153         throws UnknownTransactionException, CannotJoinException,
154 	       CrashCountException, RemoteException
155     {
156         mgr.join(id, part, crashCount);
157     }
158  
159     /**
160      * Returns the current state of the transaction.  The returned
161      * state can be any of the <code>TransactionConstants</code> values.
162      *
163      * @return an <code>int</code> representing the state of the transaction
164      *
165      * @throws UnknownTransactionException if the transaction 
166      *         is unknown to the transaction manager, either 
167      *         because the transaction ID is incorrect or because the 
168      *         transaction is complete and its state has been 
169      *         discarded by the manager.
170      * @throws RemoteException if there is a communication error
171      *
172      * @see TransactionConstants
173      */
174     public int getState() throws UnknownTransactionException, RemoteException {
175 	return mgr.getState(id);
176     }
177 
178     /**
179      * Return true if the transaction has a parent, false if the transaction
180      * is top level.
181      *
182      * @return true if the transaction has a parent, false if the transaction
183      * is top level.
184      */
185     public boolean isNested() {
186 	return false;
187     }
188 
189     // inherit javadoc
190     public String toString() {
191 	return this.getClass().getName() +
192 	    " [manager=" + mgr +
193 	    ", id=" + id +
194 	    "]";
195     }
196 }
