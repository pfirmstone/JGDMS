19  package net.jini.core.event;
20  
21  /**
22   * The RemoteEventListener interface needs to be implemented by any object
23   * that wants to receive a notification of a remote event from some other
24   * object. 
25   * <p>
26   * The object implementing this interface does not need to be the object 
27   * that originally registered interest in the occurrence of an event. To 
28   * allow the notification of an event's occurrence to be sent to an entity
29   * other than the one that made the interest registration, the registration
30   * call needs to accept a destination parameter, which indicates the object
31   * to which the notification should be sent. This parameter must be an
32   * object which supports the RemoteEventListener interface.
33   *
34   * @author Sun Microsystems, Inc.
35   *
36   * @since 1.0
37   */
38  public interface RemoteEventListener
39  			extends java.rmi.Remote, java.util.EventListener
40  {
41  
42      /**
43       * Notify the listener about an event.
44       * <p>
45       * The call to notify is synchronous to allow the party making the call
46       * to know if the call succeeded.  However, it is not part of the
47       * semantics of the call that the notification return can be delayed
48       * while the recipient of the call reacts to the occurrence of the event.
49       * Simply put, the best strategy on the part of the recipient is to note
50       * the occurrence in some way and then return from the notify method as
51       * quickly as possible.
52       * <p>
53       * UnknownEventException is thrown when the recipient does not recognize
54       * the combination of the event identifier and the event source as
55       * something in which it is interested.  Throwing this exception has the
56       * effect of asking the sender to not send further notifications of
57       * this kind of event from this source in the future.
58       *
59       * @param theEvent the remote event that occurred
60       *
61       * @throws UnknownEventException the recipient does not recognize the
62       *         combination of event identifier and event source
63       * @throws java.rmi.RemoteException if a connection problem occurs.
64       */
65      void notify(RemoteEvent theEvent)
66  	throws UnknownEventException, java.rmi.RemoteException;
67  }
