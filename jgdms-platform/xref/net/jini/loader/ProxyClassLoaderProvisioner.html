16  package net.jini.loader;
17  
18  import java.io.IOException;
19  
20  /**
21   * Allows client code to implement provisioning of a ClassLoader for a URI path,
22   * where the use of codebase annotations is unsuitable.
23   * 
24   * The first URI in the path, must be the proxy URI, any additionally appended
25   * URI are dependants, these may not be loaded directly by the returned
26   * ClassLoader, but their classes should still be visible and resolvable from it.
27   * Only the proxy URI classes are guaranteed to be loaded by a returned 
28   * ClassLoader.
29   * Dependant URI are not guaranteed to be loaded if suitably versioned are
30   * already.
31   * 
32   * Some systems, notably OSGi, manage ClassLoader visibility differently than 
33   * Java's typical hierarchical ClassLoader relationships, implementors of this
34   * interface may implement this to provision codebases into a ClassLoader,
35   * prior to deserializing a proxy into the resolved ClassLoader.
36   * 
37   * A proxy may be an instance of {@link java.lang.reflect.Proxy} and have
38   * no associated codebase, in this case, a new ClassLoader should be returned
39   * into which a dynamic proxy class can be provisioned.
40   * 
41   * The implementing class must have {@link java.lang.RuntimePermission}
42   * "getClassLoader" and "createClassLoader".  The implementing class must
43   * ensure that the caller has {@link java.lang.RuntimePermission} 
44   * "createClassLoader" as well.
45   * 
46   */
47  public interface ProxyClassLoaderProvisioner {
48      
49      /**
50       * Create a new ClassLoader, given a space separated list of URL's.
51       * 
52       * The first URL must contain the proxy class.
53       * 
54       * The parent ClassLoader may be ignored, depending on the underlying
55       * infrastructure.
56       * 
57       * If uriPath is null, a ClassLoader is still created to accommodate
58       * any dynamically created classes, with the parent ClassLoader as the 
59       * parent.
60       * 
61       * @param uriPath the codebase URL path as a space-separated list
62       * of URLs, or <code>null</code>
63       * @param parent the default parent ClassLoader.
64       * @return
65       * @throws IOException 
66       * @throws SecurityException if the caller doesn't have {@link java.lang.RuntimePermission} "createClassLoader".
67       */
68      public ClassLoader createClassLoader(String uriPath, ClassLoader parent) throws IOException;
69      
70  }
