65      /** Contains instances of the <code>ProxyReg</code> wrapper class. This
66       *  set acts as what is referred to as the "managed set of registrars"
67       *  or simply, the "managed set". When a registrar is discovered, that
68       *  registrar and its associated "discovered state" is wrapped in an
69       *  instance of <code>ProxyReg</code>, which is then placed in this set,
70       *  which is managed by this <code>LookupDiscoveryManager</code>. As
71       *  registrars are discovered and discarded, and as the discovered state
72       *  of those registrars is modified, the contents of this set are modified
73       *  appropriately.
74       *
75       *  Note that this set is shared across threads; therefore, when
76       *  accessing or modifying the contents of this set, the appropriate
77       *  synchronization must be applied.
78       * 
79       *  A list is used instead of a Set, because hashCode and equals are 
80       *  remote calls.  However we need to avoid duplicates and since this 
81       *  requires list traversal using remote method calls during add, 
82       *  we need a lock to prevent concurrent additions.  Removal while an
83       *  add operation is in progress would be acceptable, since the worst
84       *  case scenario is the iterator snapshot would contain the removed
85       *  ProxyReg.
86       */
87      private final List<ProxyReg> discoveredSet = new CopyOnWriteArrayList<ProxyReg>(); // sync(discoveredSet)
88      private final Lock discoveredSetAddLock = new ReentrantLock();

93       */

116     /** Wrapper class in which each instance corresponds to a lookup service
117      *  that has been discovered via either group discovery, locator discovery,
118      *  or both.
119      *  
120      *  Note that each element of the set of discovered registrars managed by
121      *  this <code>LookupDiscoveryManager</code> (referred to as the "managed
122      *  set of registrars" or simply, the "managed set"), is actually an 
123      *  instance of this class.
124      */
125     final class ProxyReg  {
126         /** The discovered registrar to be managed */
127 	public final ServiceRegistrar proxy; //no sync final
128         /** The groups to which the discovered registrar belongs */
129         public String[] memberGroups; //sync(this)
130         /** Special-purpose flag used in the discard process. This flag is
131          *  only relevant to <code>LocatorDiscoveryListener.discarded</code>.
132          *  This flag is set to <code>true</code> only when the method
133          *  <code>LookupDiscoveryManager.discard</code> is called by the
134          *  client. 
135          *  
136          *  A <code>true</code> value for this flag indicates to 
137          *  <code>LocatorDiscoveryListener.discarded</code> that the registrar
138          *  referenced by a given discarded event was discarded because the
139          *  registrar was determined to be unreachable (a "communication
140          *  discard"). A <code>false</code> value for this flag indicates to 
141          *  <code>LocatorDiscoveryListener.discarded</code> that the registrar
142          *  was discarded due to lost interest in the registrar's locator
143          *  (a "no-interest discard"). 
144          * 
145          *  This flag is necessary because the locators of the discovered
146          *  registrars are not retrieved and stored with the registrars and
147          *  their member groups. The locators are not retrieved because to
148          *  do so would require a remote call which is viewed as undesirable
149          *  here since such a call cannot be guaranteed to complete in a timely
150          *  fashion. Without the locator of the discarded registrar, the
151          *  <code>LocatorDiscoveryListener.discarded</code> method cannot
152          *  determine if the discarded event received from the 
153          *  <code>LookupLocatorDiscovery</code> is a communication discard 
154          *  or a no-interest discard. It is important for that method to
155          *  be able to make such a determination because the action taken
156          *  by that <code>discarded</code> method is dependent on that
157          *  determination. For more information, refer to the API
158          *  documentation (generated by the <code>javadoc</code> tool) of
159          *  the <code>LocatorDiscoveryListener.discarded</code> method.
160          *  
161          *  Note that the mechanism that uses this flag assumes that
162          *  neither the <code>LookupDiscovery</code> utility nor the 
163          *  <code>LookupLocatorDiscovery</code> utility employed by this
164          *  <code>LookupDiscoveryManager</code> are accessible outside of
165          *  this <code>LookupDiscoveryManager</code>. If either were 
166          *  accessible, then there would be no way to guarantee that a 
167          *  value of <code>true</code> in this flag is equivalent to a
168          *  communication discard.
169          */
170         public boolean commDiscard = false; // sync(this)
171         /** Integer restricted to the values 0, 1, 2, and 3. Each value 
172          *  represents a bit (or set of bits) that, when set, indicates the
173          *  mechanism (group discovery, locator discovery, or both) through
174          *  which the registrar referenced by the current instance of
175          *  this class (proxy) has currently been (or not been) discovered.
176          *  That is, if
177          *  from = 0 (no bits set)  ==&gt; discovered by neither group nor locator
178          *       = 1 (bit 0 set)    ==&gt; discovered by only group discovery
179          *       = 2 (bit 1 set)    ==&gt; discovered by only locator discovery
180          *       = 3 (bits 1&amp;2 set) ==&gt; discovered by both group and locator
181          */
182 	private int from; // sync(this)
183         /** Indicates whether the registrar referenced by this class is
184          *  currently in the process of being discarded. This flag is used to
185          *  prevent access, or inadvertent modifications to, the discovered
186          *  state of registrars that are in the process of being discarded,
187          *  but which have not yet been removed from the managed set.
188          */
189 	private boolean bDiscarded = false; //sync(this)
190         /** Constructs an instance of this wrapper class.
191          *
192          *  @param proxy        reference to a registrar that has been newly
193          *                      discovered or re-discovered
194          *  @param memberGroups the groups to which the discovered registrar
195          *                      belongs
196          *  @param from         indicates the mechanism by which the registrar
197          *                      was discovered (group or locator discovery).
198          *                      The only values which are valid for this
199          *                      parameter are FROM_GROUP or FROM_LOCATOR.
200          */
201         public ProxyReg(ServiceRegistrar proxy,
202                         String[] memberGroups,
203                         int from)
204         {
205             if(proxy == null) {
206 	        throw new IllegalArgumentException("proxy cannot be null");
207             }//endif
208             if( (from != FROM_GROUP) && (from != FROM_LOCATOR) ) {
209 	        throw new IllegalArgumentException
210                                 ("invalid discovery mechanism: must be either "
211                                  +"FROM_GROUP or FROM_LOCATOR");
212             }//endif
213 	    this.proxy        = proxy;
214 	    this.memberGroups = memberGroups;
215 	    this.from         = from;
216 	}//end constructor

228         /** Sets the appropriate bit in the 'from' variable to indicate the

240         /** Un-sets the appropriate bit in the 'from' variable to remove
241          *  the indication that the registrar referenced by this class 
242          *  was discovered via the mechanism identified by the input
243          *  parameter. If, after un-setting the appropriate bit, the
244          *  'from' variable is equal to 0 (indicating the registrar was
245          *  previously discovered by neither group nor locator discovery),
246          *  this method returns <code>true</code>; otherwise it returns
247          *  <code>false</code>, which indicates that the registrar was 
248          *  still discovered by the "opposite" mechanism from that identified
249          *  by the input parameter.
250          *  
251          *  This method is typically called during the discard process; and
252          *  the value input should always be either FROM_GROUP or FROM_LOCATOR.
253          */
254 	public synchronized boolean removeFrom(int from) {
255 	    if( from == 0 ) {
256 	        throw new RuntimeException
257                                    ("call to removeFrom with bad argument");
258             }//endif
259 	    this.from &= ~from;
260 	    return (this.from == 0);
261 	}//end removeFrom

268          *  @return <code>int</code> representing either group discovery (1),

275         /** Discards the registrar referenced in this class from either the
276          *  <code>LookupDiscovery</code> or <code>LookupLocatorDiscovery</code>
277          *  utility employed by this <code>LookupDiscoveryManager</code>.
278          * 
279          *  The utility from which that registrar is discarded is dependent on
280          *  whether the registrar was previously discovered via group or both
281          *  group and locator discovery, or via locator discovery alone.
282          *
283          *  This method enables a mechanism for sequentially "chaining" the
284          *  discard process of the <code>LookupDiscovery</code> utility with
285          *  the discard process of the <code>LookupLocatorDiscovery</code>.
286          *  That is, rather than discarding the registrar from both utilities
287          *  at the same time, by invoking this method, the registrar will be
288          *  discarded from only one of those utilities. Then when the discarded
289          *  event is received by the listener registered with the utility from
290          *  which the registrar was discarded, the listener - based on the
291          *  current discovered state of the registrar - determines whether to: 
292          *  discard the registrar from the other utility, send a discarded
293          *  event out to the client's listener, or simply update state and
294          *  do nothing more.
295          *
296          *  This chaining mechanism helps to present a single event
297          *  source to the client listeners. That is, even though the
298          *  two discovery utilities used internally by this 
299          *  <code>LookupDiscoveryManager</code> operate independently,
300          *  sending discarded events to the group discovery listener and/or
301          *  the locator discovery listener, this chaining mechanism enables
302          *  the coordination of those separate events so that only one 
303          *  event is sent to the client listeners, even though multiple 
304          *  events may have been received here.
305          *
306          *  Although this chaining mechanism is helpful in coordinating the
307          *  discarded events received from the <code>LookupDiscovery</code>
308          *  and <code>LookupLocatorDiscovery</code> utilities, the
309          *  actual reason it must be used is due to the fact that 
310          *  <code>LookupDiscovery</code> can send a certain type of
311          *  discarded event - referred to as "passive communication
312          *  discard"; whereas <code>LookupLocatorDiscovery</code> cannot.
313          *  Recall that <code>LookupDiscovery</code> monitors the
314          *  registrars it has discovered for reachability, whereas
315          *  <code>LookupLocatorDiscovery</code> does not. When the
316          *  <code>LookupDiscovery</code> sends a passive communication
317          *  discard because it has determined that one of its registrars
318          *  has become unreachable, if that registrar was also discovered
319          *  via locator discovery, it is necessary to discard the registrar
320          *  from the <code>LookupLocatorDiscovery</code> as well. In
321          *  that case, this method is called to create a discard chain
322          *  which ultimately will result in the registrar being discarded
323          *  from the <code>LookupLocatorDiscovery</code>.
324          */
325 	public void discard() {
326             LookupDiscovery lookD = null;
327             LookupLocatorDiscovery locateD = null;
328             ServiceRegistrar regProxy = null;
329 	    synchronized(this) {
330 		bDiscarded = true;
331                 regProxy = proxy;
332                 if((from & FROM_GROUP) != 0) {
333                     lookD = lookupDisc;
334                 } 
335                 if((from & FROM_LOCATOR) != 0) {
336                     locateD = locatorDisc;
337                 }//endif
338 	    }//end sync
339             /* Give group discovery priority to avoid race condition. When
340              * appropriate, locator discard will eventually occur as a result
341              * of the first group discard.
342              */
343             if (lookD != null) lookD.discard(regProxy);
344             if (locateD != null) locateD.discard(regProxy);
345 	}//end discard

371     /** Class that defines the listener that is registered with the
372      *  <code>LookupLocatorDiscovery</code> that performs locator discovery
373      *  on behalf of this <code>LookupDiscoveryManager</code>.
374      */
375     class LocatorDiscoveryListener implements DiscoveryListener {
376         /** Called by the <code>LookupLocatorDiscovery</code> to
377          *  send a discovered event to this listener when that
378          *  <code>LookupLocatorDiscovery</code> has discovered at least
379          *  one registrar (reg) having a locator equal one of the locators
380          *  desired by this <code>LookupDiscoveryManager</code>.
381          *  The reg(s) referenced in the given discovered event may,
382          *  or may not, have also been previously discovered through group
383          *  discovery.
384          *  
385          *  If this <code>LookupDiscoveryManager</code> has no prior knowledge
386          *  of a particular reg referenced in the given discovered event when
387          *  the <code>LookupLocatorDiscovery</code> invokes this method, then
388          *  this method will send to the listeners registered with this
389          *  <code>LookupDiscoveryManager</code>, a discovered event
390          *  referencing that reg; otherwise, it will simply update state
391          *  and send no event at all.
392          */
393 	public void discovered(DiscoveryEvent e) {
394             ServiceRegistrar[] proxys = (ServiceRegistrar[])e.getRegistrars();
395             Map<ServiceRegistrar, String[]> groupsMap = e.getGroups();
396             int len = proxys.length;
397             Map<ServiceRegistrar, String[]> discoveredGroupsMap 
398                     = new HashMap<ServiceRegistrar, String[]>(len);
399             discoveredSetAddLock.lock();
400             try {
401                 for(int i=0; i<len; i++) {
402                         ProxyReg reg = findReg(proxys[i]);
403                         if(reg == null) {//newly discovered, send event
404                             reg = new ProxyReg(proxys[i], 
405                                                groupsMap.get(proxys[i]),
406                                                FROM_LOCATOR);
407                             discoveredSet.add(reg);
408                             discoveredGroupsMap.put(proxys[i],
409                                                     groupsMap.get(proxys[i]));
410                         } else {//previously discovered, update bit, send no event
411                             reg.addFrom(FROM_LOCATOR);
412                         }//endif
413                 }//end loop
414             } finally {
415                 discoveredSetAddLock.unlock();
416             }
417             /* Will send notification only if map is non-empty from above */
418             notifyListener(discoveredGroupsMap, DISCOVERED);
419 	}//end discovered
420 
421         /** Called by <code>LookupLocatorDiscovery</code> to send a discarded
422          *  event to this listener when one of the following events occurs,
423          *  affecting at least one registrar (reg) that was previously
424          *  discovered through either locator discovery alone, or through
425          *  both group and locator discovery:
426          *  <p><ul>  
427          *   <li> The method <code>LookupLocatorDiscovery.discard</code>
428          *        was called because the reg was determined to be unreachable
429          *        (this is referred to as a "communication discard").
430          *   <li> Either the method <code>setLocators</code> or the method
431          *        <code>removeLocators</code> was called on the
432          *        <code>LookupLocatorDiscovery</code>, and the resulting new
433          *        set of desired locators no longer contains an element
434          *        equal to the locator of the discarded reg (this is referred
435          *        to as a "no-interest discard").
436          *  </ul><p>
437          *  Depending on the type of discarded event received from the
438          *  <code>LookupLocatorDiscovery</code> as described above, and
439          *  depending on whether the discarded reg was previously discovered
440          *  by locator discovery alone, or by both group and locator discovery,
441          *  this method will determine whether or not to pass a discarded
442          *  event on to the client's listener. (Note that because the discarded
443          *  event was sent by the <code>LookupLocatorDiscovery</code>,
444          *  the discarded reg could not have been previously discovered by
445          *  group discovery alone.)
446          *
447          *  The following describes the logic used by this method to determine
448          *  whether or not to send a discarded event to the client listeners
449          *  registered with this <code>LookupDiscoveryManager</code>. Note
450          *  that since the <code>LookupLocatorDiscovery</code> is accessible
451          *  only from within this <code>LookupDiscoveryManager</code>, the
452          *  method <code>LookupLocatorDiscovery.discard</code> is never called
453          *  from "outside" of this <code>LookupDiscoveryManager</code>.
454          *  This fact is exploited to communicate to this method that a
455          *  communication discard has occurred. That is, when
456          *  <code>LookupDiscoveryManager.discard</code> is called, a boolean
457          *  associated with the discarded reg is set to <code>true</code> to
458          *  indicate that a communication discard has occurred.
459          *
460          *  Thus, for a particular reg referenced in the given discarded event,
461          *  <p><ul>  
462          *   <li> if the discarded event is a result of a call to 
463          *        <code>LookupDiscoveryManager.discard</code>, then this
464          *        method will conclude that the discarded event must be a 
465          *        communication discard (indicated by the value of the
466          *        boolean set by <code>LookupDiscoveryManager.discard</code>
467          *        and associated with the discarded reg). When the discard is
468          *        a communication discard, if the reg was previously
469          *        discovered by only locator discovery, this method sends a
470          *        discarded event. But if the reg was previously discovered
471          *        by group discovery as well as locator discovery,
472          *        then rather than sending a discarded event here,
473          *        <code>LookupDiscovery.discard</code> is called (through
474          *        a call to <code>reg.discard</code>), which will send a
475          *        discarded event to <code>GroupDiscoveryListener</code>, which
476          *        then ultimately sends a discarded event to the registered
477          *        listeners of this <code>LookupDiscoveryManager</code>.
478          *   <li> if the discarded event is NOT a result of a call to
479          *        <code>LookupDiscoveryManager.discard</code> (is not a
480          *        communication discard), then this method will conclude
481          *        that the discarded event must be a no-interest discard.
482          *        When the discard is a no-interest discard, if the discarded
483          *        reg was previously discovered by ONLY locator discovery,
484          *        then a discarded event is sent; otherwise, no event is sent
485          *        (see section DU.2.5.1 of the "Jini Discovery Utilities
486          *        Specification").
487          *  </ul><p>
488          *  The logic described above can be collapsed into the following
489          *  decision process:
490          *  <p><pre>
491          *      if(discovered by locator discovery alone) {
492          *          send discarded event
493          *      } else {//discovered by both group and locator discovery
494          *          if(communication discard) {
495          *              call discard on LookupDiscovery
496          *          }
497          *      }
498          *  </pre>
499          */
500 	public void discarded(DiscoveryEvent e) {
501 	    ServiceRegistrar[] proxys = e.getRegistrars();
502             Map<ServiceRegistrar, String[]> groupsMap = e.getGroups();
503             int len = proxys.length;
504             Map<ServiceRegistrar, String[]> discardedGroupsMap 
505                     = new HashMap<ServiceRegistrar, String[]>(len);
506 	    for(int i=0; i<len; i++) {
507                     ProxyReg reg = findReg(proxys[i]);
508                     if(reg != null) {
509                         String[] newGroups = groupsMap.get(reg.proxy);
510                         if ( removeDiscoveredSet(reg,FROM_LOCATOR) ) {
511                             /* Locator discovery only, always send discarded */
512                             discardedGroupsMap.put(reg.proxy,newGroups);
513                         } else {//group and loc discovery
514                             synchronized (reg){
515                                 if( reg.commDiscard ) {//unreachable, send later
516                                     reg.discard();//discard from LookupDiscovery
517                                 }//endif
518                             }
519                         }//endif
520                     }//endif(reg != null)
521             }//end loop
522             /* Will send notification only if map is non-empty from above */
523             notifyListener(discardedGroupsMap, DISCARDED);
524 	}//end discarded

526         /* Convenience method that first attempts to unset the bit in the
527          * discovery mechanism flag of the given ProxyReg based on the value
528          * of the given discovery mechanism parameter (from). Depending on
529          * the value of that parameter, the flag value that results will show
530          * either no group discovery or no locator discovery (or neither
531          * mechanism).
532          *
533          * After unsetting the appropriate bit, if the value of the flag
534          * shows neither group discovery nor locator discovery, then the 
535          * given ProxyReg is removed from the managed set of registrars, and
536          * <code>true</code> is returned; otherwise <code>false</code> is
537          * returned.
538          */
539 	boolean removeDiscoveredSet(ProxyReg reg, int from) {
540 	    boolean bret = reg.removeFrom(from);
541 	    if (bret) discoveredSet.remove(reg);
542 	    return bret;
543 	}//end removeDiscoveredSet
544     }//end class LocatorDiscoveryListener

546     /** Class that defines the listener that is registered with the
547      *  <code>LookupDiscovery</code> that performs group discovery
548      *  on behalf of this <code>LookupDiscoveryManager</code>.
549      */
550     private final class GroupDiscoveryListener
551                                            extends LocatorDiscoveryListener
552                                            implements DiscoveryChangeListener
553     {
554         /** Called by <code>LookupDiscovery</code> to send a discovered event
555          *  to this listener when that <code>LookupDiscovery</code> has
556          *  discovered at least one registrar (reg) belonging to at least one
557          *  of the groups desired by this <code>LookupDiscoveryManager</code>.
558          *  The reg(s) referenced in the given discovered event may,
559          *  or may not, have also been previously discovered through locator
560          *  discovery.
561          *  
562          *  If this <code>LookupDiscoveryManager</code> has no prior knowledge
563          *  of a particular reg referenced in the given discovered event when
564          *  the <code>LookupDiscovery</code> invokes this method, then this
565          *  method will send to the listeners registered with this
566          *  <code>LookupDiscoveryManager</code>, a discovered event
567          *  referencing that reg; otherwise, depending on whether or not
568          *  the set of member groups referenced in the given discovered
569          *  event equal the member groups currently associated with the
570          *  reg from that event, this method may send a changed event, or
571          *  may simply update state and send no event at all.
572          */
573 	public void discovered(DiscoveryEvent e) {
574             ServiceRegistrar[] proxys = (ServiceRegistrar[])e.getRegistrars();
575             Map<ServiceRegistrar, String[]> groupsMap = e.getGroups();
576             int len = proxys.length;
577             Map<ServiceRegistrar, String[]> discoveredGroupsMap 
578                     = new HashMap<ServiceRegistrar, String[]>(len);
579             Map<ServiceRegistrar, String[]> changedGroupsMap    
580                     = new HashMap<ServiceRegistrar, String[]>(len);
581             discoveredSetAddLock.lock();
582             try {
583                 for(int i=0; i<len; i++) {
584                     ProxyReg reg = findReg(proxys[i]);
585                     if(reg == null) {//newly discovered, send discovered event
586                         reg = new ProxyReg(proxys[i],
587                                            groupsMap.get(proxys[i]),
588                                            FROM_GROUP);
589                         discoveredSet.add(reg);
590                         discoveredGroupsMap.put(proxys[i],
591                                                 groupsMap.get(proxys[i]));
592                     } else {//previously discovered by group, by loc or by both
593                         String[] oldGroups = reg.getMemberGroups();
594                         String[] newGroups = groupsMap.get(reg.proxy);
595                         if( groupSetsEqual(oldGroups,newGroups) ) {//by loc
596                             reg.addFrom(FROM_GROUP);//send no event
597                         } else {//by group or by both, send changed event
598                             reg.setMemberGroups(newGroups);//update groups
599                             changedGroupsMap.put(reg.proxy,newGroups);
600                         }//endif
601                     }//endif
602                 }//end loop
603             } finally {
604                 discoveredSetAddLock.unlock();
605             }
606             /* Will send notification only if map is non-empty from above */
607             notifyListener(discoveredGroupsMap, DISCOVERED);
608             notifyListener(changedGroupsMap, CHANGED);
609 	}//end discovered
610 
611         /** Called by <code>LookupDiscovery</code> to send a discarded event
612          *  to this listener when one of the following events occurs  
613          *  affecting at least one registrar (reg) that was previously
614          *  discovered through either group discovery alone, or through
615          *  both group and locator discovery:
616          *  <p><ul>  
617          *   <li> The method <code>LookupDiscovery.discard</code> was
618          *        called because the reg was determined to be unreachable.
619          *        In this case, the discarded event received from the
620          *        <code>LookupDiscovery</code> is referred to as an
621          *        "active communication discard".
622          *   <li> The <code>LookupDiscovery</code> has determined that the
623          *        reg has stopped sending multicast announcements.
624          *        In this case, the discarded event received from the
625          *        <code>LookupDiscovery</code> is referred to as a
626          *        "passive communication discard".
627          *   <li> Either the method <code>setGroups</code> or the method
628          *        <code>removeGroups</code> has been called on the
629          *        <code>LookupDiscovery</code>, and the resulting new set
630          *        of desired groups no longer intersects the set of member
631          *        groups of the discarded reg. In this case, the discarded
632          *        event received from the <code>LookupDiscovery</code> is
633          *        referred to as an "active no-interest discard".
634          *   <li> The member groups of the discarded reg have been changed
635          *        in such a way that the set of member groups associated
636          *        with the discarded reg no longer intersects the set of
637          *        desired groups to discover. In this case, the discarded
638          *        event received from <code>LookupDiscovery</code> is
639          *        referred to as a "passive no-interest discard".
640          *  </ul><p>
641          *  Depending on the type of discarded event received from the
642          *  <code>LookupDiscovery</code> as described above, and depending
643          *  on whether the discarded reg was previously discovered by group
644          *  discovery alone, or by both group and locator discovery, this
645          *  method will determine whether to send a discarded event, a changed
646          *  event, or no event at all. Note that because the discarded event
647          *  was sent by the <code>LookupDiscovery</code>, the discarded reg
648          *  could not have been previously discovered by locator discovery
649          *  alone. Note also that even though a discarded event is received
650          *  by this method, there are conditions in which a changed event
651          *  is ultimately sent to the listeners registered with this
652          *  <code>LookupDiscoveryManager</code> (see below).
653          *
654          *  The following describes the logic used by this method to determine
655          *  whether or not to send an event to the registered listeners of
656          *  this <code>LookupDiscoveryManager</code>, as well as what type
657          *  of event to send. Note that this method concludes that the
658          *  current discarded event must be a communication discard when the
659          *  the set of desired groups still intersects the discarded reg's
660          *  current set of member groups. This is because when interest in
661          *  the reg still exists, <code>LookupDiscovery</code> will discard 
662          *  the reg only when it has determined that the reg is unreachable.
663          *  
664          *  Thus, for a particular reg referenced in the given discarded event,
665          *  <p><ul>  
666          *   <li> if the member groups contained in the discarded event from
667          *        the <code>LookupDiscovery</code> are identical to the
668          *        previously discovered member groups of that reg, and if
669          *        if those member groups still intersect the current set of
670          *        desired groups (still interested in the reg), then this
671          *        method will conclude that the discarded event must be either
672          *        an active or a passive communication discard. When the 
673          *        discard is a communication discard (either active or
674          *        passive), if the reg was previously discovered by only group
675          *        discovery, this method sends a discarded event. But if the
676          *        reg was previously discovered by locator discovery as well
677          *        as group discovery, then rather than sending a discarded
678          *        event here, <code>LookupLocatorDiscovery.discard</code>
679          *        is called (through a call to <code>reg.discard</code>),
680          *        which will send a discarded event to
681          *        <code>LocatorDiscoveryListener</code>, which then
682          *        ultimately sends a discarded event to the registered
683          *        listeners of this <code>LookupDiscoveryManager</code>.
684          *   <li> if the member groups contained in the discarded event from
685          *        the <code>LookupDiscovery</code> are identical to the
686          *        previously discovered member groups of that reg, but those
687          *        member groups NO LONGER intersect the current set of desired
688          *        groups, then this method will conclude that the discarded
689          *        event must be an active no-interest discard. When the 
690          *        discard is an active no-interest discard, if the discarded
691          *        reg was previously discovered by ONLY group discovery, then
692          *        a discarded event is sent; otherwise, no event is sent
693          *        (see section DU.2.4.1 of the "Jini Discovery Utilities
694          *        Specification").
695          *   <li> if the member groups contained in the discarded event from
696          *        the <code>LookupDiscovery</code> neither intersect the
697          *        current set of desired groups (indicating no interest), nor
698          *        are they identical to the previously discovered member
699          *        groups of that reg (the member groups have changed), this
700          *        method will conclude that the discarded event must be a
701          *        passive no-interest discard. When the discard is a passive
702          *        no-interest discard, if the discarded reg was previously
703          *        discovered by locator discovery as well as group discovery,
704          *        a changed event is is sent to the appropriate listeners;
705          *        otherwise a discarded event is sent. A changed event is
706          *        sent when the reg was previously discovered by both group
707          *        and locator discovery because the fact that the reg was
708          *        discovered via locator discovery indicates continued
709          *        interest in the reg, even though the reg was discarded with
710          *        respect to group discovery. In this case, since the member
711          *        groups have actually changed (changed in such a way that the
712          *        <code>LookupDiscovery</code> sent a discarded event), and 
713          *        since there is still interest in that reg, it is more
714          *        appropriate to send a changed event here rather than a
715          *        discarded event.
716          *  </ul><p>
717          *  The logic described above can be collapsed into the following
718          *  decision process:
719          *  <p><pre>
720          *      if(discovered by group discovery alone) {
721          *          send discarded event
722          *      } else {//discovered by both group and locator discovery
723          *          if(NO LONGER interested in member groups) {
724          *              if(member group have changed) {
725          *                  send changed event
726          *              }
727          *          } else {//still interested in member groups
728          *              call discard on LookupLocatorDiscovery
729          *          }
730          *      }
731          *  </pre>
732          */
733 	public void discarded(DiscoveryEvent e) {
734 	    ServiceRegistrar[] proxys = (ServiceRegistrar[])e.getRegistrars();
735             Map<ServiceRegistrar, String[]> groupsMap = e.getGroups();
736             int len = proxys.length;
737             Map<ServiceRegistrar, String[]> discardedGroupsMap 
738                     = new HashMap<ServiceRegistrar, String[]>(len);
739             Map<ServiceRegistrar, String[]> changedGroupsMap   
740                     = new HashMap<ServiceRegistrar, String[]>(len);
741 	    for(int i=0; i<len; i++) {
742                     ProxyReg reg = findReg(proxys[i]);
743                     if(reg != null) {
744                         String[] newGroups = groupsMap.get(reg.proxy);
745                         if( removeDiscoveredSet(reg,FROM_GROUP) ) {
746                             /* group discovery only, always send discarded */
747                             discardedGroupsMap.put(reg.proxy,newGroups);
748                         } else {//discovered by group and locator
749                             String[] desiredGroups = lookupDisc.getGroups();
750                             if( !stillInterested(newGroups,desiredGroups) ) {
751                                 String[] oldGroups = reg.getMemberGroups();
752                                 if( !groupSetsEqual(oldGroups,newGroups) ) {
753                                     reg.setMemberGroups(newGroups);
754                                     changedGroupsMap.put(reg.proxy,newGroups);
755                                 }//endif(!groupSetsEqual ==> groups changed)
756                             } else {//stillInterested ==> communication discard
757                                 reg.discard();//from LookupLocatorDiscovery
758                             }//endif(stillInterested or not)
759                         }//endif(removeDiscoveredSet ==> group or group & loc)
760                     }//endif(reg != null)
761             }//end loop
762             /* Will send notification only if map is non-empty from above */
763             notifyListener(discardedGroupsMap, DISCARDED);
764             notifyListener(changedGroupsMap, CHANGED);
765 	}//end discarded
766 
767         /** Called by <code>LookupDiscovery</code> to send a changed event
768          *  to this listener when that <code>LookupDiscovery</code> has 
769          *  determined that the member groups of a previously discovered
770          *  registrar (reg) have changed. This method extracts and records
771          *  the appropriate information from the given changed event, and
772          *  then sends a changed event to the appropriate listeners registered
773          *  with this <code>LookupDiscoveryManager</code>.
774          */
775         public void changed(DiscoveryEvent e) {
776             /* update the groups of each changed registrar */
777             ServiceRegistrar[] proxys = (ServiceRegistrar[])e.getRegistrars();
778             Map<ServiceRegistrar, String[]> groupsMap = e.getGroups();
779             int len = proxys.length;
780             Map<ServiceRegistrar, String[]> changedGroupsMap 
781                     = new HashMap<ServiceRegistrar, String[]>(len);
782 	    for(int i=0; i<len; i++) {
783                 ProxyReg reg = findReg(proxys[i]);
784                 if(reg != null) {//previously discovered
785                     String[] newGroups = groupsMap.get(reg.proxy);
786                     reg.setMemberGroups(newGroups);
787                     changedGroupsMap.put(reg.proxy,newGroups);
788                 }//endif
789             }//end loop
790             /* Will send notification only if map is non-empty from above */
791             notifyListener(changedGroupsMap, CHANGED);
792         }//end changed
793     }//end class GroupDiscoveryListener

1071      * <p>
1072      * If <code>null</code> is input, this method takes no action. If the

1087      * @see #removeDiscoveryListener

1090         if(listener == null) {
1091             throw new NullPointerException("can't add null listener");
1092         }
1093         
1094         listeners.add(listener);
1095         if (discoveredSet.isEmpty()) return;
1096         Map<ServiceRegistrar, String[]> groupsMap = new HashMap<ServiceRegistrar, String[]>(discoveredSet.size());
1097         Iterator<ProxyReg> it = discoveredSet.iterator();
1098         while (it.hasNext()){
1099             ProxyReg reg = it.next();

1101         } 
1102         notifyListener(listener, groupsMap, DISCOVERED);
1103     }

1223     /**
1224      * Notify all listeners for a discovery event. If a listener's method
1225      * completes abruptly due to a Throwable, it is logged and processing 
1226      * continues.
1227      * @param groupsMap mapping from the elements of the registrars of this
1228      *               event to the member groups in which each registrar is
1229      *               a member.
1230      * @param eventType The type of event.
1231      */
1232     private void notifyListener(Map<ServiceRegistrar, String[]> groupsMap, int eventType) {
1233 	if(groupsMap.isEmpty()) return;
1234 	Iterator<DiscoveryListener> iter = listeners.iterator();
1235 	while(iter.hasNext()) {
1236 	    DiscoveryListener l = iter.next();
1237 	    try {
1238                 notifyListener(l, groupsMap, eventType);
1239 	    } catch (Throwable t) {
1240                 logger.log(Levels.HANDLED, "a discovery listener failed to process a " +
1241                 	(eventType == DISCARDED ? "discard" : eventType == DISCOVERED ? "discover" : "changed") + " event", t);
1242 	    }
1243 	}
1244     }//end notifyListener

1246     /**
1247      * Notify a specific listener for a discovery event. If the listener's
1248      * method throws a Throwable T, this method will also throw T. 
1249      * @param l The listener to notify.
1250      * @param groupsMap mapping from the elements of the registrars of this
1251      *               event to the member groups in which each registrar is
1252      *               a member.
1253      * @param eventType The type of the event.
1254      */
1255     private void notifyListener(DiscoveryListener l,
1256                                 Map<ServiceRegistrar, String[]> groupsMap,
1257                                 int eventType)
1258     {
1259         /* always send discovered and discarded events, not always changed */
1260         if((eventType == CHANGED) && !(l instanceof DiscoveryChangeListener)){
1261             return;
1262         }
1263         DiscoveryEvent evt = new DiscoveryEvent(AbstractLookupDiscoveryManager.this,
1264                                                 deepCopy(groupsMap) );
1265         switch(eventType) {
1266             case DISCOVERED:
1267                 l.discovered(evt);
1268                 break;
1269             case DISCARDED:
1270                 l.discarded(evt);
1271                 break;
1272             case CHANGED:
1273                 ((DiscoveryChangeListener)l).changed(evt);
1274                 break;
1275 	    default:
1276 		throw new IllegalArgumentException("illegal event type");
1277         }//end switch
1278     }//end notifyListener

1280     /** Determines if two sets of registrar member groups have identical
1281      *  contents. Assumes there are no duplicates, and the sets can never
1282      *  be null.
1283      *
1284      * @param groupSet0    <code>String</code> array containing the group
1285      *                     names from the first set used in the comparison
1286      * @param groupSet1    <code>String</code> array containing the group
1287      *                     names from the second set used in the comparison
1288      * 
1289      *  @return <code>true</code> if the contents of each set is identical;
1290      *          <code>false</code> otherwise
1291      */
1292     private static boolean groupSetsEqual(String[] groupSet0,
1293                                           String[] groupSet1)
1294     {
1295         if(groupSet0.length != groupSet1.length) return false;
1296         /* every element of one set contained in the other set? */
1297         iLoop:
1298         for(int i=0;i<groupSet0.length;i++) {
1299             for(int j=0;j<groupSet1.length;j++) {
1300                 if( groupSet0[i].equals(groupSet1[j]) ) {
1301                     continue iLoop;
1302                 }
1303             }
1304             return false;
1305         }
1306         return true;
1307     }//end groupSetsEqual

1309     /** Determines if at least one member group of a given registrar is
1310      *  contained in the given set of desired groups.
1311      *
1312      * @param regGroups     <code>String</code> array containing the member
1313      *                      groups of a given registrar (will never be null)
1314      * @param desiredGroups <code>String</code> array containing the groups
1315      *                      to discover (can be null - ALL_GROUPS)
1316      * 
1317      *  @return <code>true</code> if at least one element of regGroups is
1318      *          contained in desiredGroups; <code>false</code> otherwise
1319      */
1320     private boolean stillInterested(String[] regGroups,String[] desiredGroups){
1321         if(desiredGroups == DiscoveryGroupManagement.ALL_GROUPS) return true;
1322         if(desiredGroups.length == 0) return false;
1323 	for(int i=0;i<regGroups.length;i++) {
1324 	    for (int j=0;j<desiredGroups.length;j++) {
1325                 if(regGroups[i].equals(desiredGroups[j])) return true;
1326             }
1327         }
1328 	return false;
1329     }//end stillInterested

1331     /** Creates and returns a deep copy of the input parameter. This method
1332      *  assumes the input map is a HashMap of the registrar-to-groups mapping;
1333      *  and returns a clone not only of the map, but of each key-value pair
1334      *  contained in the mapping.
1335      *
1336      * @param groupsMap mapping from a set of registrars to the member groups
1337      *                  of each registrar 
1338      * 
1339      *  @return clone of the input map, and of each key-value pair contained
1340      *          in the input map
1341      */
1342     private Map<ServiceRegistrar, String[]> deepCopy(Map<ServiceRegistrar, String[]> groupsMap) {
1343         /* clone the input HashMap */
1344         Map<ServiceRegistrar, String[]> newMap 
1345                 = new HashMap<ServiceRegistrar, String[]>(groupsMap);
1346         /* clone the values of each mapping in place */
1347         Set<Map.Entry<ServiceRegistrar, String[]>> eSet = newMap.entrySet();
1348         for(Iterator<Map.Entry<ServiceRegistrar, String[]>> itr = eSet.iterator(); itr.hasNext(); ) {
1349             Map.Entry<ServiceRegistrar, String[]> pair = itr.next();
1350             /* only need to clone the value of the order pair */
1351             pair.setValue( pair.getValue().clone() );
1352         }
1353         return newMap;
1354     }//end deepCopy
