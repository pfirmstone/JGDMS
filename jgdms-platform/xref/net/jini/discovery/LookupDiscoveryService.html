18  package net.jini.discovery;
19  
20  import java.rmi.MarshalledObject;
21  import java.rmi.RemoteException;
22  import net.jini.core.discovery.LookupLocator;
23  import net.jini.core.event.RemoteEventListener;
24  
25  /**
26   * Defines the interface to the lookup discovery service. This interface is 
27   * not a remote interface; each implementation of this service exports
28   * proxy objects that implement the LookupDiscoveryService interface local to
29   * the client, using an implementation-specific protocol to communicate
30   * with the actual remote server. All of the proxy methods obey normal
31   * RMI remote interface semantics except where explicitly noted. Two
32   * proxy objects are equal if they are proxies for the same lookup discovery
33   * service.
34   *
35   * @author Sun Microsystems, Inc.
36   *
37   * @see net.jini.discovery.LookupDiscoveryRegistration
38   */
39  public interface LookupDiscoveryService {
40  
41      /**
42       * Register with the lookup discovery service. When a client invokes
43       * this method, it requests that the lookup discovery service perform
44       * discovery processing on its behalf.
45       * <p>
46       * An invocation of this method produces an object - referred to as a
47       * registration object (or simply, a registration) - that is mutable.
48       * Because the registration produced by this method is mutable, each
49       * invocation of this method produces a new registration object. Thus,
50       * this method is not idempotent.
51       * <p>
52       * To register with the lookup discovery service, the client must
53       * indicate the lookup services it is interested in discovering. It
54       * does this by submitting two sets of of objects. One set consists
55       * of the names of the groups whose members are lookup services the
56       * client wishes to be discovered. The other set consists of 
57       * LookupLocator objects, each corresponding to a specific lookup
58       * service the client wishes to be discovered. The state information
59       * managed by the lookup discovery service contains no knowledge of
60       * the clients that register. Thus, there is no requirement that the
61       * client identify itself during the registration process.
62       * <p>
63       * Registration with the lookup discovery service includes registration
64       * with the event mechanism of the lookup discovery service. That is,
65       * for each registration created as a result of an invocation of this
66       * method, an event identifier will be generated that uniquely maps
67       * the registration to the listener object and to the set of groups
68       * and locators input to this method. This event identifier is returned
69       * as a part of the registration object, and is unique across all
70       * other active registrations with the lookup discovery service.
71       * <p>
72       * While the registration is in effect, whenever the lookup discovery
73       * service finds a lookup service matching the discovery criteria of
74       * one or more of its registrations, an instance of 
75       * net.jini.discovery.RemoteDiscoveryEvent will be sent to the
76       * listener corresponding to each such registration. The event sent
77       * to each listener will contain the appropriate event identifier.
78       * <p>
79       * Any registration granted as a result of an invocation of this method
80       * is leased. The initial duration of the lease granted to a client
81       * by the lookup discovery service will be less than or equal to the
82       * requested duration input to this method. Any registration with the
83       * lookup discovery service is persistent across restarts (crashes) 
84       * of the lookup discovery service until the lease on the registration
85       * expires or is cancelled.
86       *
87       * @param groups        String array, none of whose elements may be null,
88       *                      consisting of zero or more names of groups to
89       *                      which lookup services to discover belong.
90       *                      A value of null
91       *                      (net.jini.discovery.LookupDiscovery.ALL_GROUPS)
92       *                      is acceptable. If null is passed to this argument,
93       *                      the lookup discovery service will attempt to
94       *                      discover all lookup services located within the
95       *                      multicast radius of the host on which the lookup
96       *                      discovery service is running. If an empty array
97       *                      (net.jini.discovery.LookupDiscovery.NO_GROUPS)
98       *                      is passed to this argument, then no group discovery
99       *                      will be performed for the associated registration
100      *                      until the client, through one of the registration's
101      *                      methods, populates the managed set of groups.
102      *                      
103      * @param locators      array of zero or more non-null LookupLocator
104      *                      objects, each corresponding to a specific lookup
105      *                      service to discover. If either the empty array
106      *                      or null is passed to this argument, then no
107      *                      locator discovery will be performed for the
108      *                      associated registration until the client, through
109      *                      one of the registration's methods, populates the
110      *                      managed set of locators.
111      *                      
112      * @param listener      a non-null instance of RemoteEventListener. This 
113      *                      argument specifies the entity that will receive
114      *                      events notifying the registration that a lookup
115      *                      service of interest has been discovered. A 
116      *                      non-null value must be passed to this argument,
117      *                      otherwise a NullPointerException will be thrown
118      *                      and the registration will not succeed.
119      *                      
120      * @param handback      null or an instance of MarshalledObject. This
121      *                      argument specifies an object that will be 
122      *                      included in the notification event that the
123      *                      lookup discovery service sends to the registered
124      *                      listener.
125      *                      
126      * @param leaseDuration long value representing the amount of time (in
127      *                      milliseconds) for which the resources of the
128      *                      lookup discovery service are being requested.
129      *                      
130      * @return an instance of the LookupDiscoveryRegistration interface.
131      * 
132      * @throws java.rmi.RemoteException typically, this exception occurs when
133      *         there is a communication failure between the client and the
134      *         lookup discovery service. When this exception does occur, the
135      *         registration may or may not have completed successfully.
136      *
137      * @throws java.lang.NullPointerException this exception occurs when
138      *         null is input to the <code>listener</code> parameter, as well
139      *         as when one or more of the elements of the <code>groups</code>
140      *         parameter is null.
141      *
142      * @throws java.lang.IllegalArgumentException this exception occurs when
143      *         the value input to the <code>leaseDuration</code> parameter
144      *         is neither positive, Lease.FOREVER, nor Lease.ANY.
145      * @deprecated use {@link LookupDiscoveryServiceSafely#register(java.lang.String[], net.jini.core.discovery.LookupLocator[], net.jini.core.event.RemoteEventListener, net.jini.io.MarshalledInstance, long) }
146      */
147     @Deprecated 
148     public LookupDiscoveryRegistration register(String[] groups,
149                                                 LookupLocator[] locators,
150                                                 RemoteEventListener listener,
151                                                 MarshalledObject handback,
152                                                 long leaseDuration)
153                                                         throws RemoteException;
154 }
