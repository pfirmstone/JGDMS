18  package net.jini.discovery;
19  
20  import java.io.ByteArrayInputStream;
21  import java.io.DataInputStream;
22  import java.io.IOException;
23  import java.net.DatagramPacket;
24  import java.net.InetAddress;
25  import net.jini.core.lookup.ServiceID;
26  
27  /**
28   * Encapsulate the details of unmarshaling an incoming multicast
29   * request packet.
30   *
31   * @author Sun Microsystems, Inc.
32   *
33   * @see OutgoingMulticastRequest
34   */
35  @Deprecated
36  public class IncomingMulticastRequest {
37      /**
38       * The address to which any responses should go.
39       */
40      protected InetAddress addr;
41      /**
42       * The port to which any responses should connect.
43       */
44      protected int port;
45      /**
46       * The ServiceIDs from which the requestor has already heard.
47       * This set may not be complete.
48       */
49      protected ServiceID[] serviceIDs;
50      /**
51       * The groups in which the requestor is interested.  This set may
52       * not be complete.
53       */
54      protected String[] groups;
55      /**
56       * The current version of the multicast request protocol.
57       */
58      protected static final int protoVersion = 1;
59  
60      /**
61       * Construct a new object, initialized from the contents of the
62       * given multicast request packet.
63       *
64       * @param dgram the packet to unmarshal
65       * @exception IOException a problem occurred while unmarshaling
66       * the packet
67       */
68      public IncomingMulticastRequest(DatagramPacket dgram) throws IOException {
69  	ByteArrayInputStream bs = new ByteArrayInputStream(dgram.getData());
70  	DataInputStream ds = new DataInputStream(bs);
71  	int proto = ds.readInt();
72  	if (proto != protoVersion)
73  	    throw new IOException("unsupported protocol version: " + proto);
74  	port = ds.readInt();
75  	// Valid port ranges are 1 through 65535 for IPv4.
76  	if (port <= 0 || port >= 65536)
77  	    throw new IOException("port number out of range: " + port);
78  	int sidCount = ds.readInt();
79  	// We know that a ServiceID marshals to 8 bytes, so ensure
80  	// that the count can plausibly fit into this packet.
81  	if (sidCount < 0 || sidCount > bs.available() / 8)
82  	    throw new IOException("service ID count invalid: " + sidCount);
83  	serviceIDs = new ServiceID[sidCount];
84  	for (int i = 0; i < serviceIDs.length; i++) {
85  	    serviceIDs[i] = new ServiceID(ds);
86  	}
87  	int grpCount = ds.readInt();
88  	// We know that the minimal length of a string written with
89  	// DataOutput.writeUTF is two bytes (for the empty string), so
90  	// we make some attempt to see if the count can plausibly fit
91  	// into this packet.
92  	if (grpCount < 0 || grpCount > bs.available() / 2)
93  	    throw new IOException("group count invalid: " + grpCount);
94  	groups = new String[grpCount];
95  	for (int i = 0; i < groups.length; i++) {
96  	    groups[i] = ds.readUTF();
97  	}
98  	addr = dgram.getAddress();
99      }
100 
101     /**
102      * Return the address of the host to contact in order to start
103      * unicast discovery.
104      *
105      * @return the address of the host to contact in order to start
106      *         unicast discovery
107      */
108     public InetAddress getAddress() {
109 	return addr;
110     }
111 
112     /**
113      * Return the number of the port to connect to on the remote host
114      * in order to start unicast discovery.
115      *
116      * @return the number of the port to connect to on the remote host
117      *         in order to start unicast discovery
118      */
119     public int getPort() {
120 	return port;
121     }
122 
123     /**
124      * Return the set of groups in which the requestor is interested.
125      * This set may not be complete, but other incoming packets should
126      * contain the rest of the set.
127      *
128      * @return the set of groups in which the requestor is interested.
129      */
130     public String[] getGroups() {
131 	return groups.clone();
132     }
133 
134     /**
135      * Return a set of ServiceIDs from which the requestor has already
136      * heard.  This set may not be complete.
137      *
138      * @return a set of service IDs from which the requestor has already
139      *         heard
140      */
141     public ServiceID[] getServiceIDs() {
142 	return serviceIDs.clone();
143     }
144 
145     public int hashCode() {
146 	byte[] foo = addr.getAddress();
147 	int code = (foo[0] << 3) ^ (foo[1] << 7) ^ (foo[2] << 5) ^ (foo[3] << 1) ^
148 	    ~port;
149 	for (int i = 0; i < groups.length; i++) {
150 	    code ^= groups[i].hashCode() << i;
151 	}
152 	for (int i = 0; i < serviceIDs.length; i++) {
153 	    code ^= serviceIDs[i].hashCode() << i;
154 	}
155 	return code;
156     }
157 
158     /**
159      * Two requests are equal if they have the same address, port,
160      * groups, and service IDs.
161      */
162     public boolean equals(Object o) {
163 	// We assume here that multiple requests from the same machine
164 	// will have the same source IP address.  This is not
165 	// necessarily true.
166 	if (o instanceof IncomingMulticastRequest) {
167 	    IncomingMulticastRequest oo = (IncomingMulticastRequest) o;
168 	    if (oo.addr.equals(addr) && oo.port == port &&
169 		oo.groups.length == groups.length &&
170 		oo.serviceIDs.length == serviceIDs.length)
171 	    {
172 		for (int i = 0; i < groups.length; i++) {
173 		    if (!groups[i].equals(oo.groups[i]))
174 			return false;
175 		}
176 		for (int i = 0; i < serviceIDs.length; i++) {
177 		    if (!serviceIDs[i].equals(oo.serviceIDs[i]))
178 			return false;
179 		}
180 		return true;
181 	    }
182 	}
183 	return false;
184     }
185 }
