33  /**
34   * An extension of <code>ObjectInputStream</code> that implements the
35   * dynamic class loading semantics of Java(TM) Remote Method
36   * Invocation (Java RMI) argument and result
37   * unmarshalling (using {@link ClassLoading}).  A
38   * <code>MarshalInputStream</code> is intended to read data written by
39   * a corresponding {@link MarshalOutputStream}.
40   *
41   * <p><code>MarshalInputStream</code> implements the input side of the
42   * dynamic class loading semantics by overriding {@link
43   * ObjectInputStream#resolveClass resolveClass} and {@link
44   * ObjectInputStream#resolveProxyClass resolveProxyClass} to resolve
45   * class descriptors in the stream using {@link ClassLoading#loadClass
46   * ClassLoading.loadClass} and {@link ClassLoading#loadProxyClass
47   * ClassLoading.loadProxyClass} (which, in turn, use {@link
48   * RMIClassLoaderSpi#loadClass(String,String,ClassLoader)
49   * RMIClassLoaderSpi.loadClass} and {@link
50   * RMIClassLoaderSpi#loadProxyClass(String,String[],ClassLoader)
51   * RMIClassLoaderSpi.loadProxyClass}), optionally with codebase
52   * annotation strings written by a <code>MarshalOutputStream</code>.
53   *
54   * <p>By default, a <code>MarshalInputStream</code> ignores all
55   * codebase annotation strings, instead using a codebase value of
56   * <code>null</code> when loading classes.  Codebase annotation
57   * strings will only be used by a <code>MarshalInputStream</code>
58   * after its {@link #useCodebaseAnnotations useCodebaseAnnotations}
59   * method has been invoked.
60   *
61   * <p><code>MarshalInputStream</code> supports optional verification
62   * that all codebase annotation URLs that are used to load classes
63   * resolved by the stream provide content integrity (see {@link
64   * net.jini.security.Security#verifyCodebaseIntegrity
65   * net.jini.security.Security.verifyCodebaseIntegrity}).  Whether or not a particular
66   * <code>MarshalInputStream</code> instance verifies that codebase
67   * annotation URLs provide content integrity is determined by the
68   * <code>verifyCodebaseIntegrity</code> constructor argument.  See
69   * {@link ClassLoading#loadClass ClassLoading.loadClass} and {@link
70   * ClassLoading#loadProxyClass ClassLoading.loadProxyClass} for
71   * details of how codebase integrity verification is performed.
72   *
73   * <p><code>MarshalInputStream</code> reads class annotations from its
74   * own stream; a subclass can override the {@link #readAnnotation
75   * readAnnotation} method to read the class annotations from a
76   * different location.
77   *
78   * <p>A <code>MarshalInputStream</code> is not guaranteed to be
79   * safe for concurrent use by multiple threads.
80   *
81   * @author Sun Microsystems, Inc.
82   * @since 2.0
83   **/
84  public class MarshalInputStream
85      extends ObjectInputStream implements ObjectStreamContext
86  {

101 	specialClasses.put("float", float.class);
102 	specialClasses.put("double", double.class);
103 	specialClasses.put("void", void.class);
104     }

128     /**
129      * Creates a new <code>MarshalInputStream</code> that reads
130      * marshalled data from the specified underlying
131      * <code>InputStream</code>.
132      *
133      * <p>This constructor passes <code>in</code> to the superclass
134      * constructor that has an <code>InputStream</code> parameter.
135      *
136      * <p><code>defaultLoader</code> will be passed as the
137      * <code>defaultLoader</code> argument to {@link
138      * ClassLoading#loadClass ClassLoading.loadClass} and {@link
139      * ClassLoading#loadProxyClass ClassLoading.loadProxyClass}
140      * whenever those methods are invoked by {@link #resolveClass
141      * resolveClass} and {@link #resolveProxyClass resolveProxyClass}.
142      *
143      * <p>If <code>verifyCodebaseIntegrity</code> is
144      * <code>true</code>, then the created stream will verify that all
145      * codebase annotation URLs that are used to load classes resolved
146      * by the stream provide content integrity, and whenever {@link
147      * net.jini.security.Security#verifyCodebaseIntegrity
148      * net.jini.security.Security.verifyCodebaseIntegrity} is invoked to enforce that
149      * verification, <code>verifierLoader</code> will be passed as the
150      * <code>loader</code> argument.  See {@link
151      * ClassLoading#loadClass ClassLoading.loadClass} and {@link
152      * ClassLoading#loadProxyClass ClassLoading.loadProxyClass} for
153      * details of how codebase integrity verification is performed.
154      *
155      * <p><code>context</code> will be used as the return value of the
156      * created stream's {@link #getObjectStreamContext
157      * getObjectStreamContext} method.
158      *
159      * @param in the input stream to read marshalled data from
160      *
161      * @param defaultLoader the class loader value (possibly
162      * <code>null</code>) to pass as the <code>defaultLoader</code>
163      * argument to <code>ClassLoading</code> methods
164      *
165      * @param verifyCodebaseIntegrity if <code>true</code>, this
166      * stream will verify that codebase annotation URLs used to load
167      * classes resolved by this stream provide content integrity
168      *
169      * @param verifierLoader the class loader value (possibly
170      * <code>null</code>) to pass to
171      * <code>Security.verifyCodebaseIntegrity</code>, if
172      * <code>verifyCodebaseIntegrity</code> is <code>true</code>
173      *
174      * @param context the collection of context information objects to
175      * be returned by this stream's {@link #getObjectStreamContext
176      * getObjectStreamContext} method
177      *
178      * @throws IOException if the superclass's constructor throws an
179      * <code>IOException</code>
180      *
181      * @throws SecurityException if the superclass's constructor
182      * throws a <code>SecurityException</code>
183      *
184      * @throws NullPointerException if <code>in</code> or
185      * <code>context</code> is <code>null</code>
186      **/
187     public MarshalInputStream(InputStream in,
188 			      ClassLoader defaultLoader,
189 			      boolean verifyCodebaseIntegrity,
190 			      ClassLoader verifierLoader,
191 			      Collection context)
192 	throws IOException
193     {	      
194 	super(check(in, context));
195 	this.defaultLoader = defaultLoader;
196 	this.verifyCodebaseIntegrity = verifyCodebaseIntegrity;
197 	this.verifierLoader = verifierLoader;
198 	this.context = context;
199     }

242     private static InputStream check(InputStream in, Collection context){
243 	if (context == null || in == null) throw new NullPointerException();
244 	// SecurityException's propagate, cause them to be thrown prior to
245 	// superclass instantiation.
246 	// Should we bother?  ObjectInputStream is susceptable to a finalizer
247 	// attack.
248 //	AccessController.doPrivileged(new PrivilegedAction<Object>() {
249 //	    @Override
250 //	    public Object run() {
251 //		new SerializablePermission("enableSubclassImplementation").checkGuard(null);
252 //		new SerializablePermission("enableSubstitution").checkGuard(null);
253 //		return null;
254 //	    }
255 //	});
256 	return in;
257     }

259     /**
260      * Enables the use of codebase annotation strings written by the
261      * corresponding <code>MarshalOutputStream</code>.
262      *
263      * <p>If this method has not been invoked on this stream, then the
264      * <code>resolveClass</code> and <code>resolveProxyClass</code>
265      * methods ignore the class annotation strings and just use a
266      * <code>null</code> codebase value when loading classes.  After
267      * this method has been invoked, then the
268      * <code>resolveClass</code> and <code>resolveProxyClass</code>
269      * methods use the class annotation strings as codebase values.
270      **/
271     public void useCodebaseAnnotations() {
272 	usingCodebaseAnnotations = true;
273     }

284     /**
285      * Resolves the appropriate {@link Class} object for the stream
286      * class descriptor <code>classDesc</code>.
287      *
288      * <p><code>MarshalInputStream</code> implements this method as
289      * follows:
290      *
291      * <p>This method first invokes this stream's {@link
292      * #readAnnotation readAnnotation} method to read the annotation
293      * string value (possibly <code>null</code>) for the class
294      * descriptor.  If <code>readAnnotation</code> throws an
295      * exception, then this method throws that exception.  Otherwise,
296      * a codebase value is chosen as follows: if the {@link
297      * #useCodebaseAnnotations useCodebaseAnnotations} method has been
298      * invoked on this stream, then the codebase value chosen is the
299      * value that was returned by <code>readAnnotation</code>;
300      * otherwise, the codebase value chosen is <code>null</code>.
301      * 
302      * <p>
303      * If the name of the class
304      * described by <code>classDesc</code> equals the Java
305      * programming language keyword for a primitive type or
306      * <code>void</code>, then this method returns the
307      * <code>Class</code> corresponding to that primitive type or
308      * <code>void</code> ({@link Integer#TYPE} for <code>int</code>,
309      * {@link Void#TYPE} for <code>void</code>, and so forth).
310      * 
311      * <p>This method then invokes {@link ClassLoading#loadClass
312      * ClassLoading.loadClass} with the chosen codebase value as the
313      * first argument, the name of the class described by
314      * <code>classDesc</code> as the second argument, and the
315      * <code>defaultLoader</code>,
316      * <code>verifyCodebaseIntegrity</code>, and
317      * <code>verifierLoader</code> values that were passed to this
318      * stream's constructor as the third, fourth, and fifth arguments.
319      * 
320      * If <code>ClassLoading.loadClass</code> throws an
321      * exception, this method throws that exception, and if it returns
322      * normally, this method returns the <code>Class</code> returned
323      * by <code>ClassLoading.loadClass</code>.
324      *
325      * @param classDesc the stream class descriptor to resolve
326      *
327      * @return the resolved class
328      *
329      * @throws IOException if <code>readAnnotation</code> throws an
330      * <code>IOException</code>, or if
331      * <code>ClassLoading.loadClass</code> throws a
332      * <code>MalformedURLException</code>
333      *
334      * @throws ClassNotFoundException if <code>readAnnotation</code>
335      * or <code>ClassLoading.loadClass</code> throws a
336      * <code>ClassNotFoundException</code>
337      *
338      * @throws NullPointerException if <code>classDesc</code> is
339      * <code>null</code>
340      **/
341     protected Class resolveClass(ObjectStreamClass classDesc)
342 	throws IOException, ClassNotFoundException
343     {
344 	if (classDesc == null) {
345 	    throw new NullPointerException();
346 	}
347 
348 	// must always consume annotation written by MarshalOutputStream
349 	String annotation = readAnnotation();
350 	String codebase = usingCodebaseAnnotations ? annotation : null;
351 
352 	String name = classDesc.getName();
353 	// Previously special classes were only sent after catching an
354 	// exception, this caused some confusing messages in logging, in
355 	// addition, exceptional conditions should not be used for the
356 	// normal path of code execution.
357 	Class c = (Class) specialClasses.get(name);
358 	if (c != null) return c;
359 	return ClassLoading.loadClass(codebase,
360 				      name,
361 				      defaultLoader,
362 				      verifyCodebaseIntegrity,
363 				      verifierLoader);
364     }

366     /**
367      * Resolves the appropriate {@link Class} object for the proxy
368      * class described by the interface names
369      * <code>interfaceNames</code>.
370      *
371      * <p><code>MarshalInputStream</code> implements this method as
372      * follows:
373      *
374      * <p>This method first invokes this stream's {@link
375      * #readAnnotation readAnnotation} method to read the annotation
376      * string value (possibly <code>null</code>) for the class
377      * descriptor.  If <code>readAnnotation</code> throws an
378      * exception, then this method throws that exception.  Otherwise,
379      * a codebase value is chosen as follows: if the {@link
380      * #useCodebaseAnnotations useCodebaseAnnotations} method has been
381      * invoked on this stream, then the codebase value chosen is the
382      * value that was returned by <code>readAnnotation</code>;
383      * otherwise, the codebase value chosen is <code>null</code>.
384      *
385      * <p>This method then invokes {@link ClassLoading#loadProxyClass
386      * ClassLoading.loadProxyClass} with the chosen codebase value as
387      * the first argument, <code>interfaceNames</code> as the second
388      * argument, and the <code>defaultLoader</code>,
389      * <code>verifyCodebaseIntegrity</code>, and
390      * <code>verifierLoader</code> values that were passed to this
391      * stream's constructor as the third, fourth, and fifth arguments.
392      * If <code>ClassLoading.loadProxyClass</code> throws an
393      * exception, then this method throws that exception.  Otherwise,
394      * this method returns the <code>Class</code> returned by
395      * <code>ClassLoading.loadProxyClass</code>.
396      *
397      * @param interfaceNames the list of interface names that were
398      * deserialized in the proxy class descriptor
399      *
400      * @return the resolved dynamic proxy class
401      *
402      * @throws IOException if <code>readAnnotation</code> throws an
403      * <code>IOException</code>, or if
404      * <code>ClassLoading.loadProxyClass</code> throws a
405      * <code>MalformedURLException</code>
406      *
407      * @throws ClassNotFoundException if <code>readAnnotation</code>
408      * or <code>ClassLoading.loadProxyClass</code> throws a
409      * <code>ClassNotFoundException</code>
410      *
411      * @throws NullPointerException if <code>interfaceNames</code> is
412      * <code>null</code> or if any element of
413      * <code>interfaceNames</code> is <code>null</code>
414      **/
415     protected Class resolveProxyClass(String[] interfaceNames)
416 	throws IOException, ClassNotFoundException
417     {
418 	for (int i = 0; i < interfaceNames.length; i++) {
419 	    if (interfaceNames[i] == null) {
420 		throw new NullPointerException();
421 	    }
422 	}
423 
424 	// must always consume annotation written by MarshalOutputStream
425 	String annotation = readAnnotation();
426 	String codebase = usingCodebaseAnnotations ? annotation : null;
427 
428 	return ClassLoading.loadProxyClass(codebase,
429 					   interfaceNames,
430 					   defaultLoader,
431 					   verifyCodebaseIntegrity,
432 					   verifierLoader);
433     }

435     /**
436      * Reads and returns a class annotation string value (possibly
437      * <code>null</code>) that was written by a corresponding
438      * <code>MarshalOutputStream</code> implementation.
439      *
440      * <p><code>MarshalInputStream</code> implements this method to
441      * just read the annotation value from this stream using {@link
442      * ObjectInputStream#readObject readObject}, and if
443      * <code>readObject</code> returns a non-<code>null</code> value
444      * that is not a <code>String</code>, an {@link
445      * InvalidObjectException} is thrown.
446      *
447      * <p>A subclass can override this method to read the annotation
448      * from a different location.
449      *
450      * @return the class annotation string value read (possibly
451      * <code>null</code>)
452      *
453      * @throws IOException if an I/O exception occurs reading the
454      * annotation
455      *
456      * @throws ClassNotFoundException if a
457      * <code>ClassNotFoundException</code> occurs reading the
458      * annotation
459      **/
460     protected String readAnnotation()
461 	throws IOException, ClassNotFoundException
462     {
463 	try {
464 	    return (String) readObject();
465 	} catch (ClassCastException e) {
466 	    InvalidObjectException ioe =
467 		new InvalidObjectException("annotation not String or null");
468 	    ioe.initCause(e);
469 	    throw ioe;
470 	}
471     }
