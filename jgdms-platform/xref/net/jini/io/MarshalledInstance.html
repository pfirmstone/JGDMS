48  /**
49   * A <code>MarshalledInstance</code> contains an object in serialized
50   * form. The contained object can be deserialized on demand when
51   * explicitly requested. This allows an object to be sent from one VM
52   * to another in a way that allows the receiver to control when and if
53   * the object is deserialized.
54   * <p>
55   * The contained object is specified at construction time and can
56   * either be provided in unserialized or serialized form. If provided
57   * in unserialized form it will be serialized during construction
58   * with the serialization semantics defined by
59   * <code>MarshalOutputStream</code>. In particular, classes are annotated
60   * with a codebase URL from which the class can be loaded (if available).
61   * <p>
62   * If the <code>MarshalledInstance</code> needs to deserialize the
63   * contained object then the contained object will be deserialized with the
64   * deserialization semantics defined by <code>MarshalInputStream</code>.
65   * In particular, the codebase annotations associated with the contained
66   * object may be used to load classes referenced by the contained object.
67   * <p>
68   * <code>MarshalledInstance</code> provides functionality similar to
69   * <code>java.rmi.MarshalledObject</code>, but additionally provides
70   * for the verification of codebase integrity. Unlike
71   * <code>java.rmi.MarshalledObject</code>, it does not perform remote
72   * object-to-stub replacement.
73   *
74   * @author Sun Microsystems, Inc.
75   * @since 2.0
76   */
77  @AtomicSerial
78  public class MarshalledInstance implements Serializable {
79      
80      private static final Guard UNMARSHAL = new DeSerializationPermission("MARSHALL");
81  
82      /**
83       * @serial Bytes of serialized representation.  If <code>objBytes</code> is
84       * <code>null</code> then the object marshalled was a <code>null</code>
85       * reference.
86       */  
87      private final byte[] objBytes;
88   
89      /**
90       * @serial Bytes of location annotations, which are ignored by
91       * <code>equals</code>.  If <code>locBytes</code> is null, there were no
92       * non-<code>null</code> annotations during marshalling.
93       */  
94      private final byte[] locBytes;
95   
96      /**
97       * @serial Stored hash code of contained object.
98       *   
99       * @see #hashCode
100      */  
101     private final int hash;

105     private static boolean check(GetArg arg) throws IOException{
106 	byte [] objBytes = arg.get("objBytes", null, byte[].class);
107 	byte [] locBytes = arg.get("locBytes", null, byte[].class);
108 	int hash = arg.get("hash", 0);
109 	if ((objBytes == null) && ((hash != 13) || (locBytes != null)))
110 	    throw new InvalidObjectException("Bad hash or annotation");
111 	int h = 0;
112 	if (objBytes == null){
113 	    h = 13;
114 	} else {
115 	    for (int i = 0; i < objBytes.length; i++) {
116 		h = 31 * h + objBytes[i];
117 	    }
118 	}
119 	if (h != hash) throw new InvalidObjectException("Bad hash or annotation");
120 	UNMARSHAL.checkGuard(null);
121 	return true;
122     }
123     
124     public MarshalledInstance(GetArg arg) throws IOException{
125 	this(check(arg), arg);
126     }
127     
128     private MarshalledInstance( boolean check, GetArg arg) throws IOException {
129 	objBytes = Valid.copy(arg.get("objBytes", null, byte[].class));
130 	locBytes = Valid.copy(arg.get("locBytes", null, byte[].class));
131 	hash = arg.get("hash", 0);
132     }

145     /**
146      * Creates a new <code>MarshalledInstance</code> that contains the
147      * marshalled representation of the current state of the supplied
148      * object. The object is serialized with the semantics defined by
149      * <code>MarshalOutput</code>.
150      * 
151      * This is designed to allow MarshalledInstance to be extensible 
152      * and utilize other Serialization Frameworks.
153      * 
154      * This constructor calls a chain of internal constructors and static
155      * methods that prevent finalizer attacks.
156      * 
157      * It is advisable for overriding classes to be stateless.
158      *
159      * @param obj The Object to be contained in the new 
160      *          <code>MarshalledInstance</code>
161      * @param context the collection of context information objects or
162      *        <code>null</code>
163      * @param marshalFactory MarshalFactory used to create underlying MarshalInstanceInput
164      * and MarshalInstanceOutput.
165      * @throws IOException if the object cannot be serialized
166      * @throws NullPointerException if context or marshalFactory is null.
167      */
168     protected MarshalledInstance(Object obj, Collection context, MarshalFactory marshalFactory) throws IOException{
169 	this(obj, context, marshalFactory,
170 		obj != null ? new ByteArrayOutputStream() : null,
171 		obj != null ? new ByteArrayOutputStream() : null);
172     }

174     private MarshalledInstance(Object obj, Collection context, MarshalFactory factory, ByteArrayOutputStream bout,
175 	    ByteArrayOutputStream lout) throws IOException
176     {
177 	this(bout, 
178 	    obj != null ?
179 		writeOutRetLocAnnotation(obj, context, factory, bout, lout) 
180 		: null
181 	);
182     }

184     /**
185      * 
186      * @param obj Object to be marshaled.
187      * @param context
188      * @param factory MarshalFactory used to create MarshalIntstanceOutput
189      * @param bout ByteArrayOutputStream to write object serial form.
190      * @param lout ByteArrayOutputStream to write code-base annotation to.
191      * @return byte array of location.
192      * @throws IOException 
193      */
194     private static byte[] writeOutRetLocAnnotation(Object obj, Collection context, MarshalFactory factory,
195 	    ByteArrayOutputStream bout, ByteArrayOutputStream lout) throws IOException
196     {
197 	if (context == null) throw new NullPointerException();
198 	if (factory == null) throw new NullPointerException();
199 	MarshalInstanceOutput out = null;
200 	try {
201 	    out = factory.createMarshalOutput(bout, lout, context);
202 	    out.writeObject(obj);
203 	    out.flush();
204 	    // locBytes is null if no annotations
205 	    return out.hadAnnotations() ? lout.toByteArray() : null;
206 	} finally {
207 	    try {
208 		if (out != null) out.close();
209 	    } catch (IOException e){} // Ignore
210 	}
211     }
212     
213     private MarshalledInstance(ByteArrayOutputStream bout, byte[] locBytes){
214 	this(bout != null ? bout.toByteArray() : null, locBytes);
215     }
216     
217     private MarshalledInstance(byte[] objBytes, byte[] locBytes){
218 	this.objBytes = objBytes;
219 	this.locBytes = locBytes;
220 	// Calculate hash from the marshalled representation of object
221 	// so the hashcode will be comparable when sent between VMs.
222 	//
223 	// Note: This calculation must match the calculation in
224 	//	 java.rmi.MarshalledObject since we use this hash
225 	//	 in the converted MarshalledObject. The reverse is
226 	//	 also true in that we use the MarshalledObject's
227 	//	 hash for our hash. (see the MarshalledInstance(
228 	//	 MarshalledObject) constructor)
229 	//
230 	if (objBytes == null){
231 	    hash = 13; // null hash for java.rmi.MarshalledObject
232 	} else {
233 	    int h = 0;
234 	    for (int i = 0; i < objBytes.length; i++) {
235 		h = 31 * h + objBytes[i];
236 	    }
237 	    hash = h;
238 	}
239     }

282     /**
283      * Creates a new <code>MarshalledInstance</code> from an
284      * existing <code>MarshalledObject</code>. An object equivalent
285      * to the object contained in the passed <code>MarshalledObject</code>
286      * will be contained in the new <code>MarshalledInstance</code>.
287      * <p>
288      * The object contained in the passed <code>MarshalledObject</code>
289      * will not be unmarshalled as part of this call.
290      *
291      * @param mo The <code>MarshalledObject</code> that contains
292      *        the object the new <code>MarshalledInstance</code> should
293      *        contain
294      * @throws NullPointerException if <code>mo</code> is <code>null</code>
295      */
296     public MarshalledInstance(java.rmi.MarshalledObject mo) {
297 
298 	if (mo == null)
299 	    throw new NullPointerException();
300 
301 	// To extract the java.rmi.MarshalledObject's fields we
302 	// convert the mo into a net.jini.io.MarshalledObject.
303 	// (See resolveClass() in FromMOInputStream) The private
304 	// version of MarshalledObject allows access to the needed
305 	// fields.
306 	//
307 	net.jini.io.MarshalledObject privateMO = null;
308 	try {
309 	    ByteArrayOutputStream baos = new ByteArrayOutputStream();
310 	    ObjectOutputStream oos = new ObjectOutputStream(baos);
311 	    oos.writeObject(mo);
312 	    oos.flush();
313 	    byte[] bytes = baos.toByteArray();
314 	    ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
315 	    ObjectInputStream ois = new FromMOInputStream(bais);
316 	    privateMO =
317 		(net.jini.io.MarshalledObject)ois.readObject();
318 	} catch (IOException  ioe) {
319 	    throw new AssertionError(ioe);
320 	} catch (ClassNotFoundException ioe){
321 	    throw new AssertionError(ioe);
322 	}
323 	objBytes = privateMO.objBytes;
324 	locBytes = privateMO.locBytes;
325 	hash = privateMO.hash;
326     }

328     /**
329      * Creates a new <code>MarshalledObject</code> that will
330      * contain an object equivalent to the object contained
331      * in this <code>MarshalledInstance</code> object.
332      * <p>
333      * The object contained in this <code>MarshalledInstance</code>
334      * object will not be unmarshalled as part of this call.
335      * <p>
336      * Some Objects in the Jini api include MarshalledObject as part of their
337      * serial form.
338      * @return A new <code>MarshalledObject</code> which
339      *        contains an object equivalent to the object
340      *        contained in this <code>MarshalledInstance</code>
341      */
342     public java.rmi.MarshalledObject convertToMarshalledObject() {
343 	// To create a java.rmi.MarshalledObject with previously
344 	// serialized data we first create a private
345 	// net.jini.io.MarshalledObject with the
346 	// data and then convert it to the final object by changing
347 	// the class during readObject(). (See resolveClass() in
348 	// ToMOInputStream)
349 	//
350 	net.jini.io.MarshalledObject privateMO =
351 		new net.jini.io.MarshalledObject();
352 
353 	privateMO.objBytes = objBytes;
354 	privateMO.locBytes = locBytes;
355 	privateMO.hash = hash;
356 
357 	java.rmi.MarshalledObject mo = null;
358 	try {
359 	    ByteArrayOutputStream baos = new ByteArrayOutputStream();
360 	    ObjectOutputStream oos = new ObjectOutputStream(baos);
361 	    oos.writeObject(privateMO);
362 	    oos.flush();
363 	    byte[] bytes = baos.toByteArray();
364 	    ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
365 	    ObjectInputStream ois = new ToMOInputStream(bais);
366 	    mo = (java.rmi.MarshalledObject)ois.readObject();
367 	} catch (IOException ioe) {
368 	    throw new AssertionError(ioe);
369 	} catch (ClassNotFoundException ioe){
370 	    throw new AssertionError(ioe);
371 	}
372 	return mo;
373     }

394      * <p><code>MarshalledInstance</code> implements this method by calling
395      * <code>{@link #get(ClassLoader, boolean, ClassLoader, Collection)
396      * get}(null, verifyCodebaseIntegrity, null, null)</code>.
397      *
398      * @param verifyCodebaseIntegrity if <code>true</code> then
399      *        codebase integrity is verified, otherwise code base
400      *        integrity is not verified
401      * @return a new copy of the contained object
402      * @throws IOException if an 
403      *         <code>IOException</code> occurs while deserializing the

405      * @throws ClassNotFoundException if any classes necessary
406      *         for reconstructing the contained object can not

408      *         is <code>true</code> and the integrity of the
409      *         contained object's codebase cannot be confirmed

417     /**
418      * Returns a new copy of the contained object. Deserialization is
419      * performed with the semantics defined by <code>MarshalInputStream</code>.
420      * If <code>context</code> is not <code>null</code>
421      * the input stream used to unmarshal the object implements {@link
422      * ObjectStreamContext} and returns the given collection from its {@link
423      * ObjectStreamContext#getObjectStreamContext getObjectStreamContext}
424      * method.
425      * <p>If <code>context</code> is <code>null</code>
426      * the input stream used to unmarshal the object implements {@link
427      * ObjectStreamContext} and returns a collection from its {@link
428      * ObjectStreamContext#getObjectStreamContext getObjectStreamContext}
429      * method which contains a single element of type {@link
430      * IntegrityEnforcement}; the {@link IntegrityEnforcement#integrityEnforced
431      * integrityEnforced} method of this element returns the specified
432      * <code>verifyCodebaseIntegrity</code> value.
433      *
434      * @param defaultLoader the class loader value (possibly
435      *	      <code>null</code>) to pass as the <code>defaultLoader</code>
436      *        argument to <code>RMIClassLoader</code> methods
437      * @param verifyCodebaseIntegrity if <code>true</code> then
438      *        codebase integrity is verified, otherwise code base
439      *        integrity is not verified
440      * @param verifierLoader the class loader value (possibly
441      *        <code>null</code>) to pass to {@link
442      *        net.jini.security.Security#verifyCodebaseIntegrity
443      *        Security.verifyCodebaseIntegrity}, if
444      *        <code>verifyCodebaseIntegrity</code> is <code>true</code>
445      * @param context the collection of context information objects or
446      *        <code>null</code>
447      * @return a new copy of the contained object
448      * @throws IOException if an 
449      *         <code>IOException</code> occurs while deserializing the
450      *         object from its internal representation
451      * @throws ClassNotFoundException if any classes necessary
452      *         for reconstructing the contained object can not
453      *         be found or if <code>verifyCodebaseIntegrity</code>
454      *         is <code>true</code> and the integrity of the
455      *         contained object's codebase cannot be confirmed
456      */
457     public Object get(final ClassLoader defaultLoader,
458 		      final boolean verifyCodebaseIntegrity,
459 		      final ClassLoader verifierLoader,
460 		      final Collection context)
461 	throws IOException, ClassNotFoundException 
462     {
463 	if (objBytes == null)   // must have been a null object
464 	    return null;
465 	final Collection ctext;
466 	if (context == null) {
467 	    ctext = Collections.singleton( new IntegrityEnforcement(){
468 
469 		@Override
470 		public boolean integrityEnforced() {
471 		    return verifyCodebaseIntegrity;
472 		}
473 		
474 	    });
475 	} else {
476 	    ctext = context;
477 	}
478 	final ByteArrayInputStream bin = new ByteArrayInputStream(objBytes);
479 	// locBytes is null if no annotations
480 	final ByteArrayInputStream lin =
481 	    (locBytes == null ? null : new ByteArrayInputStream(locBytes));
482 	Object obj;	    
483 	
484 	MarshalInstanceInput in = null;
485 	try {
486 	    in = getMarshalFactory().createMarshalInput(
487 		bin, lin, defaultLoader, verifyCodebaseIntegrity,
488 		verifierLoader, ctext);
489 	    in.useCodebaseAnnotations();
490 	    obj = in.readObject();
491 	    return obj;
492 	} finally {
493 	    try {
494 		if (in != null) in.close();
495 	    } catch (IOException e){} // Ignore
496 	}
497     }

499     /**
500      * Compares this <code>MarshalledInstance</code> to another
501      * object. Returns true if and only if the argument refers to an instance
502      * of <code>MarshalledInstance</code> that contains exactly the same
503      * serialized form for its contained object as this object does and
504      * has the same class codebase annotations.
505      *
506      * @param obj the object to compare with this
507      *            <code>MarshalledInstance</code>
508      * @return <code>true</code> if the argument contains an object
509      *         with an equivalent serialized form and codebase;
510      *	       otherwise returns <code>false</code>
511      */
512     public boolean fullyEquals(Object obj) {
513 	if (equals(obj)) {
514 	    MarshalledInstance other = (MarshalledInstance)obj;
515 	    return Arrays.equals(locBytes, other.locBytes);
516 	}
517 	return false;
518     }

520     /**
521      * Compares this <code>MarshalledInstance</code> to another
522      * object. Returns true if and only if the argument refers to an instance
523      * of <code>MarshalledInstance</code> that contains exactly the same
524      * serialized form for its contained object as this object does. The
525      * comparison ignores any class codebase annotations, so that
526      * two objects can be equivalent if they have the same serialized
527      * representation, except for the codebase of each class in the
528      * serialized representation.
529      * <p>
530      * Subclasses should not override this method.
531      * 
532      * @since 3.1.0 MarshalledInstance is extensible and since it is used for hand-back
533      * objects over remote connections and because it's possible for
534      * MarshalledInstance to be converted to MarshalledObject and back again to 
535      * MarshalledInstance, the resulting conversion may loose type compatibility
536      * with the original instance which may be a subclass of MarshalledInstance.  
537      * This equals method allows a caller to determine if the hand-back
538      * instance is equal to the original, based solely on the serial form 
539      * of the contained object.
540      * 
541      * @param obj the object to compare with this
542      *            <code>MarshalledInstance</code>
543      * @return <code>true</code> if the argument contains an object
544      *         with an equivalent serialized form; otherwise returns
545      *         <code>false</code>
546      */
547     @Override
548     public final boolean equals(Object obj) {
549 	if (obj == this)
550 	    return true;
551 
552 	if (obj instanceof MarshalledInstance) {
553 	    MarshalledInstance other = (MarshalledInstance)obj;
554 	    if (hash != other.hash)
555 		return false;
556 	    return Arrays.equals(objBytes, other.objBytes);
557 	}
558 	return false;
559     }

642      * <code>MarshalledInstance</code>.  It places the location annotations
643      * to one side so that two <code>MarshalledInstance</code>s can be
644      * compared for equality if they differ only in location
645      * annotations.  Objects written using this stream should be read back
646      * from a <code>MarshalledInstanceInputStream</code>.
647      *   
648      * @see MarshalledInstanceInputStream
649      */  
650     private static class MarshalledInstanceOutputStream
651         extends MarshalOutputStream implements MarshalInstanceOutput
652     {
653 	/** The stream on which location objects are written. */
654 	private final ObjectOutputStream locOut;
655  
656 	/** <code>true</code> if non-<code>null</code> annotations are
657 	 *  written.
658 	 */
659 	private boolean hadAnnotations;
660 
661 	/**
662 	 * Creates a new <code>MarshalledObjectOutputStream</code> whose
663 	 * non-location bytes will be written to <code>objOut</code> and whose
664 	 * location annotations (if any) will be written to
665 	 * <code>locOut</code>.
666 	 */
667 	public MarshalledInstanceOutputStream(OutputStream objOut,
668 					      OutputStream locOut,
669 					      Collection context)
670 	    throws IOException
671 	{
672 	    super(objOut, context);
673 	    this.locOut = new ObjectOutputStream(locOut);
674 	    hadAnnotations = false;
675 	}
676  
677 	/**
678 	 * Returns <code>true</code> if any non-<code>null</code> location
679 	 * annotations have been written to this stream.
680 	 */
681 	@Override
682 	public boolean hadAnnotations() {
683 	    return hadAnnotations;
684 	}
685  
686 	/**
687 	 * Overrides <code>MarshalOutputStream.writeAnnotation</code>
688 	 * implementation to write annotations to the location stream.
689 	 */
690 	@Override
691 	public void writeAnnotation(String loc) throws IOException {
692 	    hadAnnotations |= (loc != null);
693 	    locOut.writeObject(loc);
694 	}
695 
696 	@Override
697 	public void flush() throws IOException {
698 	    super.flush();
699 	    locOut.flush();
700 	}
701     }
702 
703     /**
704      * The counterpart to <code>MarshalledInstanceOutputStream</code>.
705      *   
706      * @see MarshalledInstanceOutputStream
707      */  
708     private static class MarshalledInstanceInputStream
709         extends MarshalInputStream implements MarshalInstanceInput
710     {
711 	/**
712 	 * The stream from which annotations will be read.  If this is
713 	 * <code>null</code>, then all annotations were <code>null</code>.
714 	 */
715 	private final ObjectInputStream locIn;
716  
717 	/**
718 	 * Creates a new <code>MarshalledObjectInputStream</code> that
719 	 * reads its objects from <code>objIn</code> and annotations
720 	 * from <code>locIn</code>.  If <code>locIn</code> is
721 	 * <code>null</code>, then all annotations will be
722 	 * <code>null</code>.
723 	 */
724 	MarshalledInstanceInputStream(InputStream objIn,
725 				      InputStream locIn,
726 				      ClassLoader defaultLoader,
727 				      boolean verifyCodebaseIntegrity,
728 				      ClassLoader verifierLoader,
729 				      Collection context)
730 	    throws IOException
731 	{
732 	    super(objIn,
733 		  defaultLoader,
734 		  verifyCodebaseIntegrity,
735 		  verifierLoader,
736 		  context);
737 	    this.locIn = (locIn == null ? null : new ObjectInputStream(locIn));
738 	}
739  
740 	/**
741 	 * Overrides <code>MarshalInputStream.readAnnotation</code> to
742 	 * return locations from the stream we were given, or <code>null</code>
743 	 * if we were given a <code>null</code> location stream.
744 	 */
745 	@Override
746 	protected String readAnnotation()
747 	    throws IOException, ClassNotFoundException
748 	{
749 	    return (locIn == null ? null : (String)locIn.readObject());
750 	}
751     }
