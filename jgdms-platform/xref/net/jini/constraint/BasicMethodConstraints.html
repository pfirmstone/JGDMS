37  /**
38   * Basic implementation of {@link MethodConstraints}, allowing limited
39   * wildcard matching on method names and parameter types. Methods can be
40   * specified by exact name and parameter types (matching a single method),
41   * by exact name (matching all methods with that name), by name prefix
42   * (matching all methods with names that start with that prefix), by name
43   * suffix (matching all methods with names that end with that suffix), and
44   * by a default that matches all methods. Normally instances of this class
45   * should be obtained from a
46   * {@link net.jini.config.Configuration} rather than being
47   * explicitly constructed.
48   *
49   * @author Sun Microsystems, Inc.
50   * 
51   * @since 2.0

206 	/**
207 	 * Invariant checks for de-serialization.
208 	 * @param name
209 	 * @param types
210 	 * @param constraints
211 	 * @return
212 	 * @throws InvalidObjectException 
213 	 */
214 	private static boolean checkSerial(
215 		String name, 
216 		Class[] types,
217 		InvocationConstraints constraints) throws InvalidObjectException
218 	{
219 	    if (name == null) {
220 		if (types != null) {
221 		    throw new InvalidObjectException(
222 					 "cannot have types with null name");
223 		}
224 	    } else {
225 		try {
226 		    check(name, types);
227 		} catch (RuntimeException e) {
228 		    rethrow(e);
229 		}
230 	    }
231 	    if (constraints != null && constraints.isEmpty()) {
232 		throw new InvalidObjectException(
233 					     "constraints cannot be empty");
234 	    }
235 	    return true;
236 	}

238 	/**
239 	 * Verifies that the name is a syntactically valid method name, or
240 	 * (if types is null) if the name is a syntactically valid method name
241 	 * with a '*' appended or could be constructed from some syntactically
242 	 * valid method name containing more than two characters by replacing
243 	 * the first character of that name with '*', and verifies that none
244 	 * of the elements of types are null.
245 	 */
246 	private static boolean check(String name, Class[] types) {
247 	    boolean star = types == null;
248 	    int len = name.length();
249 	    if (len == 0) {
250 		throw new IllegalArgumentException(
251 					       "method name cannot be empty");
252 	    }
253 	    char c = name.charAt(0);
254 	    if (!Character.isJavaIdentifierStart(c) &&
255 		!(star && c == '*' && len > 1))
256 	    {
257 		throw new IllegalArgumentException("invalid method name");
258 	    }
259 	    if (star && c != '*' && name.charAt(len - 1) == '*') {
260 		len--;
261 	    }
262 	    while (--len >= 1) {
263 		if (!Character.isJavaIdentifierPart(name.charAt(len))) {
264 		    throw new IllegalArgumentException("invalid method name");
265 		}
266 	    }
267 	    if (types != null) {
268 		for (int i = types.length; --i >= 0; ) {
269 		    if (types[i] == null) {
270 			throw new NullPointerException("class cannot be null");
271 		    }
272 		}
273 	    }
274 	    return true;
275 	}

461 	try {

463 	} catch (RuntimeException e) {
464 	    rethrow(e);
465 	}
466 	return true;

469     /**
470      * Throws IllegalArgumentException if the descriptors array is empty, or
471      * if any descriptor is preceded by another descriptor that matches at
472      * least the same methods. Throws NullPointerException if the array or
473      * any element is null.
474      */
475     private static boolean check(MethodDesc[] descs) {
476 	if (descs.length == 0) {
477 	    throw new IllegalArgumentException(
478 					 "must have at least one descriptor");
479 	}
480 	for (int i = 0; i < descs.length; i++) {
481 	    MethodDesc desc = descs[i];
482 	    String dname = desc.name;
483 	    if (dname == null) {
484 		if (i < descs.length - 1) {
485 		    throw new IllegalArgumentException(
486 					   "default descriptor must be last");
487 		}
488 	    } else if (dname.charAt(0) == '*') {
489 		int dlen = dname.length() + 1;
490 		for (int j = 0; j < i; j++) {
491 		    MethodDesc prev = descs[j];
492 		    String pname = prev.name;
493 		    if (pname.charAt(0) == '*' &&
494 			pname.regionMatches(1, dname, dlen - pname.length(),
495 					    pname.length() - 1))
496 		    {
497 			check(prev, desc);
498 		    }
499 		}
500 	    } else if (dname.charAt(dname.length() - 1) == '*') {
501 		for (int j = 0; j < i; j++) {
502 		    MethodDesc prev = descs[j];
503 		    String pname = prev.name;
504 		    int plen = pname.length() - 1;
505 		    if (pname.charAt(plen) == '*' &&
506 			pname.regionMatches(0, dname, 0, plen))
507 		    {
508 			check(prev, desc);
509 		    }
510 		}
511 	    } else {
512 		for (int j = 0; j < i; j++) {
513 		    MethodDesc prev = descs[j];
514 		    String pname = prev.name;
515 		    int plen = pname.length() - 1;
516 		    if (pname.charAt(0) == '*') {
517 			if (dname.regionMatches(dname.length() - plen,
518 						pname, 1, plen))
519 			{
520 			    check(prev, desc);
521 			}
522 		    } else if (pname.charAt(plen) == '*') {
523 			if (dname.regionMatches(0, pname, 0, plen)) {
524 			    check(prev, desc);
525 			}
526 		    } else {
527 			if (pname.equals(dname)) {
528 			    check(prev, desc);
529 			}
530 		    }
531 		}
532 	    }
533 	}
534 	return true;
535     }

537     /**
538      * Throws IllegalArgumentException if the parameter types of prev cover
539      * those of desc.
540      */
541     private static void check(MethodDesc prev, MethodDesc desc) {
542 	if (prev.types == null || Arrays.equals(prev.types, desc.types)) {
543 	    StringBuffer buf = new StringBuffer();
544 	    prev.toString(buf, false);
545 	    buf.append(" cannot precede ");
546 	    desc.toString(buf, false);
547 	    throw new IllegalArgumentException(buf.toString());
548 	}
549     }

564     /**
565      * Returns the constraints for the specified remote method as a
566      * non-<code>null</code> value. Searches the descriptors in order, and
567      * returns the constraints in the first descriptor that matches the
568      * method, or an empty constraints instance if there is no match.
569      *
570      * @throws NullPointerException {@inheritDoc}
571      */
572     public InvocationConstraints getConstraints(Method method) {
573 	String name = method.getName();
574 	Class[] types = null;
575 	InvocationConstraints sc = null;
576     outer:
577 	for (int i = 0; i < descs.length; i++) {
578 	    MethodDesc desc = descs[i];
579 	    String dname = desc.name;
580 	    if (dname == null) {
581 		sc = desc.constraints;
582 		break;
583 	    } else if (desc.types != null) {
584 		if (!name.equals(dname)) {
585 		    continue;
586 		}
587 		if (types == null) {
588 		    types = method.getParameterTypes();
589 		}
590 		if (types.length != desc.types.length) {
591 		    continue;
592 		}
593 		for (int j = types.length; --j >= 0; ) {
594 		    if (types[j] != desc.types[j]) {
595 			continue outer;
596 		    }
597 		}
598 		sc = desc.constraints;
599 		break;
600 	    } else {
601 		int dlen = dname.length() - 1;
602 		if (dname.charAt(0) == '*') {
603 		    if (name.regionMatches(name.length() - dlen,
604 					   dname, 1, dlen))
605 		    {
606 			sc = desc.constraints;
607 			break;
608 		    }
609 		} else if (dname.charAt(dlen) == '*') {
610 		    if (name.regionMatches(0, dname, 0, dlen)) {
611 			sc = desc.constraints;
612 			break;
613 		    }
614 		} else if (name.equals(dname)) {
615 		    sc = desc.constraints;
616 		    break;
617 		}
618 	    }
619 	}
620 	if (sc == null) {
621 	    sc = InvocationConstraints.EMPTY;
622 	}
623 	return sc;
624     }

643 		}
644 		InvocationConstraints sc = descs[--i].constraints;
645 		if (sc == null) {
646 		    sc = InvocationConstraints.EMPTY;
647 		}
648 		return sc;
649 	    }

694 	return (this == obj ||
695 		(obj instanceof BasicMethodConstraints &&
696 		 Arrays.equals(descs, ((BasicMethodConstraints) obj).descs)));
697     }

699     private void writeObject(ObjectOutputStream out) throws IOException {
700 	out.defaultWriteObject();
701     }

703     /**
704      * Verifies legal descriptor ordering.
705      *
706      * @throws InvalidObjectException if any descriptor is <code>null</code>,
707      * or the descriptors array is empty, or if any descriptor is preceded by
708      * another descriptor that matches at least the same methods
709      * 
710      * Default de-serialization is vulnerable to finalizer and reference
711      * stealer attacks.
712      */
713     private void readObject(ObjectInputStream s)
714 	throws IOException, ClassNotFoundException
715     {
716 	s.defaultReadObject();
717 	checkSerial(descs);
718 	}

720     /**
721      * Returns the sum of the hash codes of all elements of the given array.
722      */
723     private static int hash(Object[] elements) {
724 	int h = 0;
725 	for (int i = elements.length; --i >= 0; ) {
726 	    h += elements[i].hashCode();
727 	}
728 	return h;
729     }

731     /**
732      * If the exception is a NullPointerException or IllegalArgumentException,
733      * wrap it in an InvalidObjectException and throw that, otherwise rethrow
734      * the exception as is.
735      */
736     private static void rethrow(RuntimeException e)
737 	throws InvalidObjectException
738     {
739 	if (e instanceof NullPointerException ||
740 	    e instanceof IllegalArgumentException)
741 	{
742 	    InvalidObjectException ee =
743 		new InvalidObjectException(e.getMessage());
744 	    ee.initCause(e);
745 	    throw ee;
746 	}
747 	throw e;
748     }
