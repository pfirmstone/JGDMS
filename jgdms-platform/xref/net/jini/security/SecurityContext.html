19  package net.jini.security;
20  
21  import java.security.AccessControlContext;
22  import java.security.PrivilegedAction;
23  import java.security.PrivilegedExceptionAction;
24  
25  /**
26   * Interface implemented by objects representing security contexts, returned
27   * from the {@link Security#getContext getContext} method of the {@link
28   * Security} class, which in turn may obtain them from a security manager or
29   * policy provider implementing the
30   * {@link net.jini.security.policy.SecurityContextSource} interface.
31   * Each <code>SecurityContext</code> contains an {@link AccessControlContext}
32   * instance representing the access control context in place when the security
33   * context was snapshotted; this instance can be passed to one of the
34   * <code>doPrivileged</code> methods of the {@link
35   * java.security.AccessController} class to restore the
36   * <code>AccessControlContext</code> portion of the overall security context.
37   * Additional state (if any) carried by the security context can be restored
38   * for the duration of a {@link PrivilegedAction} or {@link
39   * PrivilegedExceptionAction} by passing that action to the appropriate wrap
40   * method of the <code>SecurityContext</code> instance, and then executing the
41   * returned "wrapper" action.  These two operations--restoring the access
42   * control context, and restoring any additional context encapsulated by the
43   * <code>SecurityContext</code> instance--should be performed in conjunction
44   * with a single <code>AccessController.doPrivileged</code> call, as
45   * illustrated below:
46   * <pre>
47   *      // snapshot context
48   *      SecurityContext ctx = Security.getContext();
49   *
50   *      // restore context
51   *      AccessController.doPrivileged(
52   *          ctx.wrap(action), ctx.getAccessControlContext());
53   * </pre>
54   * 
55   * <b>
56   * Implementations must override Object equals and hashCode.
57   * </b>
58   * 
59   * @author Sun Microsystems, Inc.
60   * 
61   * @since 2.0
62   */
63  public interface SecurityContext {
64  
65      /**
66       * Returns a security context-restoring <code>PrivilegedAction</code> that
67       * wraps the given action, or returns the action itself if the security
68       * context does not include any non-<code>AccessControlContext</code> state
69       * to restore.  The <code>run</code> method of the "wrapper" action (if
70       * any) restores the non-<code>AccessControlContext</code> state of the
71       * security context before invoking the <code>run</code> method of the
72       * wrapped action, and unrestores that state after the wrapped action's
73       * <code>run</code> method has completed (normally or otherwise).  The
74       * value returned or exception thrown by the wrapped action's
75       * <code>run</code> method is propagated through the <code>run</code>
76       * method of the wrapper action.
77       *
78       * @param <T> return type of PrivilegedAction
79       * @param action the action to be wrapped
80       * @return security context-restoring action wrapping <code>action</code>,
81       * or <code>action</code> if no wrapping is necessary
82       * @throws NullPointerException if <code>action</code> is <code>null</code>
83       */
84      <T> PrivilegedAction<T> wrap(PrivilegedAction<T> action);
85  
86      /**
87       * Returns a security context-restoring
88       * <code>PrivilegedExceptionAction</code> that wraps the given action, or
89       * returns the action itself if the security context does not include any
90       * non-<code>AccessControlContext</code> state to restore.  The
91       * <code>run</code> method of the "wrapper" action (if any) restores the
92       * non-<code>AccessControlContext</code> state of the security context
93       * before invoking the <code>run</code> method of the wrapped action, and
94       * unrestores that state after the wrapped action's <code>run</code> method
95       * has completed (normally or otherwise).  The value returned or exception
96       * thrown by the wrapped action's <code>run</code> method is propagated
97       * through the <code>run</code> method of the wrapper action.
98       *
99       * @param <T> return type of PrivilegedExceptionAction
100      * @param action the action to be wrapped
101      * @return security context-restoring action wrapping <code>action</code>,
102      * or <code>action</code> if no wrapping is necessary
103      * @throws NullPointerException if <code>action</code> is <code>null</code>
104      */
105     <T> PrivilegedExceptionAction<T> wrap(PrivilegedExceptionAction<T> action);
106 
107     /**
108      * Returns access control context portion of snapshotted security context.
109      *
110      * @return access control context portion of snapshotted security context
111      */
112     AccessControlContext getAccessControlContext();
113 }
