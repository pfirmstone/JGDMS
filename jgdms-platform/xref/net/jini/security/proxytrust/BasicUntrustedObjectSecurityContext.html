19  package net.jini.security.proxytrust;

21  import java.security.AccessControlContext;
22  import java.security.AccessController;
23  import java.security.CodeSource;
24  import java.security.DomainCombiner;
25  import java.security.Permission;
26  import java.security.PermissionCollection;
27  import java.security.Permissions;
28  import java.security.PrivilegedAction;
29  import java.security.ProtectionDomain;
30  import java.security.cert.Certificate;
31  import net.jini.security.Security;

34  /**
35   * A basic trust verifier context element that provides a security
36   * context to use to restrict privileges when invoking methods on untrusted
37   * objects, based on a specified set of permissions.
38   *
39   * @author Sun Microsystems, Inc.
40   * @since 2.1
41   */
42  public final class BasicUntrustedObjectSecurityContext
43      implements UntrustedObjectSecurityContext
44  {
45      /** Empty codesource. */
46      private static final CodeSource emptyCS =
47  	new CodeSource(null, (Certificate[]) null);
48      /** ProtectionDomain containing the empty codesource. */
49      private static final ProtectionDomain emptyPD =
50  	new ProtectionDomain(emptyCS, null, null, null);
51  
52      /** Permissions. */
53      private final Permission[] permissions;
54      /** EmptyPD plus specified permissions. */
55      private final ProtectionDomain restrictedPD;
56  
57      /**
58       * Constructs an instance with the specified permissions. The security
59       * context returned by {@link #getContext getContext} will be restricted
60       * to the permissions granted by default to all code plus the additional
61       * specified permissions. The argument passed to the constructor is
62       * neither modified nor retained; subsequent changes to that argument
63       * have no effect on the instance created.
64       *
65       * @param permissions additional permissions to allow, or
66       * <code>null</code> if no additional permissions should be allowed
67       * @throws NullPointerException if any element of the argument is
68       * <code>null</code>
69       */
70      public BasicUntrustedObjectSecurityContext(Permission[] permissions) {
71  	if (permissions == null || permissions.length == 0) {
72  	    this.permissions = new Permission[0];
73  	    restrictedPD = emptyPD;
74  	} else {
75  	    this.permissions = (Permission[]) permissions.clone();
76  	    PermissionCollection pc = new Permissions();
77  	    for (int i = this.permissions.length; --i >= 0; ) {
78  		Permission p = this.permissions[i];
79  		if (p == null) {
80  		    throw new NullPointerException(
81  					       "permission cannot be null");
82  		}
83  		pc.add(p);
84  	    }
85  	    restrictedPD = new ProtectionDomain(emptyCS, pc, null, null);
86  	}
87      }
88  
89      /**
90       * Returns a security context to use to restrict privileges when
91       * invoking methods on untrusted objects. The returned context is
92       * equivalent to the current security context (as returned by
93       * {@link Security#getContext Security.getContext}) with an additional
94       * protection domain combined into the access control context that
95       * contains an empty {@link CodeSource} (<code>null</code> location and
96       * certificates), a permission collection containing the permissions
97       * used to construct this instance, <code>null</code> class loader, and
98       * <code>null</code> principals.
99       *
100      * @return a security context to use to restrict privileges when
101      * invoking methods on untrusted objects
102      */
103     public SecurityContext getContext() {
104 	final AccessControlContext acc0 = AccessController.getContext();
105 	final AccessControlContext racc = (AccessControlContext)
106 	    AccessController.doPrivileged(new PrivilegedAction() {
107 		public Object run() {
108 		    DomainCombiner comb0 = acc0.getDomainCombiner();
109 		    AccessControlContext acc1 =
110 			new AccessControlContext(acc0, new Combiner(comb0));
111 		    AccessControlContext acc2 = (AccessControlContext)
112 			AccessController.doPrivileged(new PrivilegedAction() {
113 				public Object run() {
114 				    return AccessController.getContext();
115 				}
116 			    }, acc1);
117 		    return new AccessControlContext(acc2, comb0);
118 		}
119 	    });
120 	return (SecurityContext) AccessController.doPrivileged(
121 	    new PrivilegedAction() {
122 		public Object run() {
123 		    return Security.getContext();
124 		}
125 	    }, racc);
126     }
127 
128     /**
129      * DomainCombiner that uses specified combiner (if any) to combine in
130      * restrictedPD.
131      */
132     private class Combiner implements DomainCombiner {
133 	private final DomainCombiner combiner;
134 
135 	Combiner(DomainCombiner combiner) {
136 	    this.combiner = combiner;
137 	}
138 
139 	public ProtectionDomain[] combine(ProtectionDomain[] current,
140 					  ProtectionDomain[] assigned)
141 	{
142 	    if (combiner != null) {
143 		return combiner.combine(new ProtectionDomain[]{restrictedPD},
144 					assigned);
145 	    } else if (assigned == null) {
146 		return new ProtectionDomain[]{restrictedPD};
147 	    }
148 	    ProtectionDomain[] pds = new ProtectionDomain[assigned.length + 1];
149 	    pds[0] = restrictedPD;
150 	    System.arraycopy(assigned, 0, pds, 1, assigned.length);
151 	    return pds;
152 	}
153     }
154 
155     /** Returns a string representation of this object. */
156     public String toString() {
157 	StringBuffer sb =
158 	    new StringBuffer("BasicUntrustedObjectAccessController{");
159 	for (int i = 0; i < permissions.length; i++) {
160 	    if (i > 0) {
161 		sb.append(", ");
162 	    }
163 	    sb.append(permissions[i]);
164 	}
165 	sb.append("}");
166 	return sb.toString();
167     }
168 
169     /**
170      * Returns <code>true</code> if the specified object and this object
171      * are both instances of this class that were constructed with equivalent
172      * permissions. The order of permissions in the array that was passed to
173      * the constructor is not significant.
174      */
175     public boolean equals(Object obj) {
176 	if (this == obj) {
177 	    return true;
178 	} else if (!(obj instanceof BasicUntrustedObjectSecurityContext)) {
179 	    return false;
180 	}
181 	Permission[] otherPerms =
182 	    ((BasicUntrustedObjectSecurityContext) obj).permissions;
183 	if (permissions.length != otherPerms.length) {
184 	    return false;
185 	}
186 	otherPerms = (Permission[]) otherPerms.clone();
187     perms:
188 	for (int i = permissions.length; --i >= 0; ) {
189 	    Permission p = permissions[i];
190 	    for (int j = i; j >= 0; j--) {
191 		if (p.equals(otherPerms[j])) {
192 		    otherPerms[j] = otherPerms[i];
193 		    continue perms;
194 		}
195 	    }
196 	    return false;
197 	}
198 	return true;
199     }
200 
201     /** Returns a hash code value for this object. */
202     public int hashCode() {
203 	int hash = 0;
204 	for (int i = permissions.length; --i >= 0; ) {
205 	    hash += permissions[i].hashCode();
206 	}
207 	return hash;
208     }
