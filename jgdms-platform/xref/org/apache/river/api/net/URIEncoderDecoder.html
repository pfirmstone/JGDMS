31  /**
32   * This class is used to encode a string using the format required by {@code
33   * application/x-www-form-urlencoded} MIME content type. It contains helper
34   * methods used by the Uri class, and performs encoding and decoding in a
35   * slightly different way than {@code URLEncoder} and {@code URLDecoder}.
36   * @since 3.0.0
37   */
38  class URIEncoderDecoder {
39  
40      static final String digits = "0123456789ABCDEF"; //$NON-NLS-1$
41  
42      static final String encoding = "UTF8"; //$NON-NLS-1$
43      
44      // Map of escaped unreserved characters defined in RFC3986
45      private static final Map<String,Character> legalEscaped;
46      
47      static {
48          legalEscaped = new HashMap<String,Character>();
49          char [] legals = "abcdefghijklmnopqrstuvwyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789-._~".toCharArray();
50          StringBuilder buf = new StringBuilder(12);
51          int l = legals.length;
52          for (int i = 0; i< l; i++){
53              char ch = legals[i];
54              byte[] bytes;
55              try {
56                  bytes = new String(new char[] { ch }).getBytes(encoding);
57              } catch (UnsupportedEncodingException ex) {
58                  Logger.getLogger(URIEncoderDecoder.class.getName()).log(Level.SEVERE, null, ex);
59                  continue;
60              }
61              for (int j = 0; j < bytes.length; j++) {
62                  buf.append('%');
63                  buf.append(digits.charAt((bytes[j] & 0xf0) >> 4));
64                  buf.append(digits.charAt(bytes[j] & 0xf));
65              }
66              legalEscaped.put(buf.toString(), Character.valueOf(ch));
67              buf.delete(0, buf.length());
68          }
69      }

73      /**
74       * Validate a string by checking if it contains any characters other than:
75       * 1. letters ('a'..'z', 'A'..'Z') 2. numbers ('0'..'9') 3. characters in
76       * the legalset parameter 4. others (unicode characters that are not in
77       * US-ASCII set, and are not ISO Control or are not ISO Space characters)
78       * <p>
79       * called from {@code URI.Helper.parseURI()} to validate each component
80       *
81       * @param s
82       *            {@code java.lang.String} the string to be validated
83       * @param legal
84       *            {@code java.lang.String} the characters allowed in the String
85       *            s
86       */
87      static void validate(String s, String legal) throws URISyntaxException {
88          for (int i = 0; i < s.length();) {
89              char ch = s.charAt(i);
90              if (ch == '%') {
91                  do {
92                      if (i + 2 >= s.length()) {
93                          throw new URISyntaxException(s, Messages.getString("luni.7D"), //$NON-NLS-1$
94                                  i);
95                      }
96                      int d1 = Character.digit(s.charAt(i + 1), 16);
97                      int d2 = Character.digit(s.charAt(i + 2), 16);
98                      if (d1 == -1 || d2 == -1) {
99                          throw new URISyntaxException(s, Messages.getString("luni.7E", //$NON-NLS-1$
100                                 s.substring(i, i + 3)), i);
101                     }
102 
103                     i += 3;
104                 } while (i < s.length() && s.charAt(i) == '%');
105 
106                 continue;
107             }
108             if (    !((ch >= 'a' && ch <= 'z') 
109                     || (ch >= 'A' && ch <= 'Z')
110                     || (ch >= '0' && ch <= '9') 
111                     || legal.indexOf(ch) > -1 )
112                 ) 
113             {
114                 throw new URISyntaxException(s, Messages.getString("luni.7F"), i); //$NON-NLS-1$
115             }
116             i++;
117         }
118     }

120     static void validateSimple(String s, String legal)
121             throws URISyntaxException {
122         for (int i = 0; i < s.length();) {
123             char ch = s.charAt(i);
124             if (!((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z')
125                     || (ch >= '0' && ch <= '9') || legal.indexOf(ch) > -1)) {
126                 throw new URISyntaxException(s, Messages.getString("luni.7F"), i); //$NON-NLS-1$
127             }
128             i++;
129         }
130     }

132     /**
133      * All characters except letters ('a'..'z', 'A'..'Z') and numbers ('0'..'9')
134      * and legal characters are converted into their hexidecimal value prepended
135      * by '%'.
136      * <p>
137      * For example: '#' -&gt; %23
138      * Other characters, which are unicode chars that are not US-ASCII, and are
139      * not ISO Control or are not ISO Space chars, are preserved.
140      * <p>
141      * Called from {@code URI.quoteComponent()} (for multiple argument
142      * constructors)
143      *
144      * @param s
145      *            java.lang.String the string to be converted
146      * @param legal
147      *            java.lang.String the characters allowed to be preserved in the
148      *            string s
149      * @return java.lang.String the converted string
150      */
151     static String quoteIllegal(String s, String legal)
152             throws UnsupportedEncodingException {
153         StringBuilder buf = new StringBuilder(s.length() + 24);
154         for (int i = 0; i < s.length(); i++) {
155             char ch = s.charAt(i);
156             if (       
157                        (ch >= 'a' && ch <= 'z')
158                     || (ch >= 'A' && ch <= 'Z')
159                     || (ch >= '0' && ch <= '9')
160                     || legal.indexOf(ch) > -1
161                 ) 
162             {
163                 buf.append(ch);
164             } else {
165                 byte[] bytes = new String(new char[] { ch }).getBytes(encoding);
166                 for (int j = 0; j < bytes.length; j++) {
167                     buf.append('%');
168                     buf.append(digits.charAt((bytes[j] & 0xf0) >> 4));
169                     buf.append(digits.charAt(bytes[j] & 0xf));
170                 }
171             }
172         }
173         return buf.toString();
174     }
175 
176     /**
177      * Other characters, which are Unicode chars that are not US-ASCII, and are
178      * not ISO Control or are not ISO Space chars are not preserved. They are
179      * converted into their hexidecimal value prepended by '%'.
180      * <p>
181      * For example: Euro currency symbol -&gt; "%E2%82%AC".
182      * <p>
183      * Called from URI.toASCIIString()
184      *
185      * @param s
186      *            java.lang.String the string to be converted
187      * @return java.lang.String the converted string
188      */
189     static String encodeOthers(String s) throws UnsupportedEncodingException {
190         StringBuilder buf = new StringBuilder(s.length()*9);
191         for (int i = 0; i < s.length(); i++) {
192             char ch = s.charAt(i);
193             if (ch <= 127) {
194                 buf.append(ch);
195             } else {
196                 byte[] bytes = new String(new char[] { ch }).getBytes(encoding);
197                 for (int j = 0; j < bytes.length; j++) {
198                     buf.append('%');
199                     buf.append(digits.charAt((bytes[j] & 0xf0) >> 4));
200                     buf.append(digits.charAt(bytes[j] & 0xf));
201                 }
202             }
203         }
204         return buf.toString();
205     }

207     /**
208      * Decodes the string argument which is assumed to be encoded in the {@code
209      * x-www-form-urlencoded} MIME content type using the UTF-8 encoding scheme.
210      * <p>
211      *'%' and two following hex digit characters are converted to the
212      * equivalent byte value. All other characters are passed through
213      * unmodified.
214      * <p>
215      * e.g. "A%20B%20C %24%25" -&gt; "A B C $%"
216      * <p>
217      * Called from URI.getXYZ() methods
218      * 
219      * @param s
220      *            java.lang.String The encoded string.
221      * @return java.lang.String The decoded version.
222      */
223     static String decode(String s) throws UnsupportedEncodingException {
224 
225         StringBuilder result = new StringBuilder(s.length());
226         ByteArrayOutputStream out = new ByteArrayOutputStream();
227         int l = s.length();
228         for (int i = 0; i < l;) {
229             char c = s.charAt(i);
230             if (c == '%') {
231                 out.reset();
232                 do {
233                     if (i + 2 >= l) {
234                         throw new IllegalArgumentException(Messages.getString(
235                                 "luni.80", i)); //$NON-NLS-1$
236                     }
237                     int d1 = Character.digit(s.charAt(i + 1), 16);
238                     int d2 = Character.digit(s.charAt(i + 2), 16);
239                     if (d1 == -1 || d2 == -1) {
240                         throw new IllegalArgumentException(Messages.getString(
241                                 "luni.81", s.substring(i, i + 3), //$NON-NLS-1$
242                                 String.valueOf(i)));
243                     }
244                     out.write((byte) ((d1 << 4) + d2));
245                     i += 3;
246                 } while (i < l && s.charAt(i) == '%');
247                 result.append(out.toString(encoding));
248                 continue;
249             }
250             result.append(c);
251             i++;
252         }
253         return result.toString();
254     }

256     static String decodeUnreserved(String s) throws URISyntaxException {
257         StringBuilder result = new StringBuilder(s.length());
258         StringBuilder pct_encoded = new StringBuilder(12);
259         int l = s.length();
260         for (int i = 0; i < l;) {
261             char c = s.charAt(i);
262             if (c == '%') {
263                     do {
264                         if (i + 2 >= l) {
265                             throw new URISyntaxException(s, Messages.getString("luni.80", i), i);
266                         }
267                         char c1 = s.charAt(i+1);
268                         char c2 = s.charAt(i+2);
269                         if (!isValidHexChar(c1) || !isValidHexChar(c2)) {
270                             throw new URISyntaxException(s, Messages.getString(
271                                 "luni.81", s.substring(i, i + 3), //$NON-NLS-1$
272                                 String.valueOf(i)), i);
273                         }
274                         pct_encoded.append('%').append(c1).append(c2);
275                         i += 3;
276                     } while (i < l && s.charAt(i) == '%');
277                     String encoded = pct_encoded.toString().toUpperCase(Locale.ENGLISH);
278                     Character ch = legalEscaped.get(encoded);
279                     if ( ch != null ){
280                         result.append(ch.charValue());
281                     } else {
282                         // Ensure all pct-encoded strings are uppercase.
283                         result.append(encoded);
284                     }
285                     pct_encoded.delete(0, pct_encoded.length());
286                     continue;
287             }
288             result.append(c);
289             i++;
290         }
291         return result.toString();
292     }
293     
294     private static boolean isValidHexChar(char c) {
295         return (c >= '0' && c <= '9') || (c >= 'A' && c <= 'F') || (c >= 'a' && c <= 'f');
296     }
