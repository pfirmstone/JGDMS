19  package org.apache.river.api.security;
20  
21  import org.apache.river.api.security.PolicyUtils.ExpansionFailedException;
22  import org.apache.river.impl.Messages;
23  
24  import java.util.Properties;
25  
26  /**
27   * Segments form a chain of String parts which represent a framework for
28   * building complex String's from Substrings.  Segments are chained at 
29   * creation time, the last Segment in a chain communicates back up the chain
30   * using the #next() method call.
31   * 
32   * The first segment in a chain must be checked each string build creation loop,
33   * until it is complete, at which point the loop should be terminated by the caller.  
34   * The last segment in a chain must have the next() method
35   * called at the end of the loop, this will propagate back up the chain, ensuring
36   * every possible unique string combination is produced, before the status of
37   * the first link becomes complete.
38   * 
39   * Segments are not thread safe.
40   * 
41   * Segments are comparable, but shouldn't be compared until after segmentation
42   * is completed, so initially should be stored in an unsorted Collection.
43   * 
44   * @author Peter Firmstone.
45   * @since 3.0.0
46   */
47  class Segment {
48      private Segment previous;
49      private Segment [] divisions;
50      private String original;
51  //    private int length = 0;
52      int counter = 0;
53      Status stat;
54      private StringBuilder sb;
55      private boolean toggle;
56      public Segment(String s, Segment preceed){
57          previous = preceed;
58          stat = Status.STRING;
59          divisions = null;
60          original = s;
61          sb = null;
62          toggle = false;
63      }
64  
65      @Override
66      public String toString(){
67          return original;
68      }
69  
70      public Status status(){
71          return stat;
72      }
73      
74      public boolean hasNext(){
75          if (allStringState()) return true;
76          if ( stat.equals(Status.COMPLETE) || stat.equals(Status.STRING)) {
77              return previous == null ? false : previous.hasNext();
78          }
79          return true;
80      }
81      
82      private boolean allStringState(){
83          if (toggle) return false;
84          toggle = true;
85          if ( ( previous == null || previous.allStringState()) && status().equals(Status.STRING)) return true;
86          return false;
87      }
88  
89      /**
90       * Must only be called on the last segment in the Set after sorting.
91       */
92      public String next(){
93          if (sb == null) sb = new StringBuilder(120);
94          else sb.delete(0, sb.capacity() -1 );
95          append(sb);
96          if ( stat.equals(Status.COMPLETE)){
97              if (previous != null ){
98                  previous.next();
99              }
100         } else if ( stat.equals(Status.MORE)){
101             if (divisions[counter].status().equals(Status.MORE)){
102                 divisions[counter].next();
103             } else {
104                 counter ++;
105             }
106             if (counter == divisions.length){
107                 stat = Status.COMPLETE;
108             }
109         } else if ( stat.equals(Status.STRING)&& previous != null){
110                 previous.next(); // ensures backward propagation.
111         }
112         return sb.toString();
113     }
114     
115     private StringBuilder append(StringBuilder sb){
116         if (previous != null) previous.append(sb);
117         if (divisions != null) divisions[counter].append(sb);
118         else sb.append(original);
119         return sb;
120     }
121     
122     private int sequenceNumber(){
123         int sequence = 1;
124         if (previous != null){
125             sequence = sequence + previous.sequenceNumber();
126         }
127         return sequence;
128     }
129 
130     /**
131      * Segments the current String by find Properties between the START_MARK and
132      * END_MARK and replacing them with their values, splitting them into separate
133      * Strings (that remain encapsulated in the Segment) if regex is non null.
134      * @param START_MARK
135      * @param END_MARK
136      * @param regex
137      * @param p
138      */
139     public void divideAndReplace(String START_MARK, String END_MARK,
140             String regex, Properties p) throws ExpansionFailedException{
141         if (previous != null) previous.divideAndReplace(START_MARK, END_MARK, regex, p);
142         if (divisions != null ){
143             int l = divisions.length;
144             for (int i = 0; i < l ; i++ ){
145                 divisions[i].divideAndReplace(START_MARK, END_MARK, regex, p);
146             }
147             // If divisions exist, then this Segment has already been processed.
148             return;
149         }
150         String orig = original; //original reference is replaced
151         Segment prev = previous; //previous reference too.
152         final int START_OFFSET = START_MARK.length();
153         final int END_OFFSET = END_MARK.length();
154         int start = orig.indexOf(START_MARK);
155         int end = 0;
156         int beginning = 0;
157         while (start >= 0) {
158             // Get the segment preceeding the key, or between keys.
159             Segment seg;
160             if ( start > beginning && beginning >= 0){
161                 seg = new Segment(orig.substring(beginning, start), prev);
162                 prev = seg;
163             }
164             end = orig.indexOf(END_MARK, start);
165             if (end >= 0) {
166                 String key = orig.substring(start + START_OFFSET, end);
167                 String value = p.getProperty(key);
168                 if (value != null) {
169                     seg = new Segment(value, prev);
170                     if (regex != null) {
171                         seg.split(regex);
172                         seg.status(Status.MORE);
173                     }
174                     prev = seg;
175                 } else {
176                     throw new ExpansionFailedException(Messages.getString("security.14F", key)); //$NON-NLS-1$
177                 }
178             }
179             beginning = end + END_OFFSET;
180             start = orig.indexOf(START_MARK, beginning);
181         }
182         // Now there could be a trailing string.
183         if (beginning < orig.length()){
184             // Use this to represent it.
185             previous = prev;
186             original = orig.substring(beginning);
187         } else if ( beginning == orig.length() && prev != null){
188             // Replace the last Segment in the list with this, after
189             // making it equal.  The reason for doing so is that a downstream
190             // Segment may reference this.
191             previous = prev.previous;
192             original = prev.original;
193             divisions = prev.divisions;
194 //            length = last.length;
195             stat = prev.stat;
196             
197         }
198     }
199     
200     /**
201      * Same as String.split(String regex) except that it stores the result
202      * internally.
203      */
204     private void split(String regex){
205         String [] prop = original.split(regex);
206         int l = prop.length;
207         divisions = new Segment[l];
208         for (int i = 0; i < l; i++ ){
209             // Divisions are parallel, they don't preceed or link to each other.
210             divisions[i] = new Segment(prop[i], null);
211         }
212     }
213     
214     private void status(Status status){
215         stat = status;
216     }
217     
218     public enum Status {
219         STRING, MORE, COMPLETE
220     }
221 }
