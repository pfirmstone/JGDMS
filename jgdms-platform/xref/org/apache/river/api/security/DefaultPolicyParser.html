73  /**
74   * This is a basic loader of policy files. It delegates lexical analysis to 
75   * a pluggable scanner and converts received tokens to a set of 
76   * {@link org.apache.river.api.security.PermissionGrant PermissionGrant's}. 
77   * For details of policy format, which should be identical to Sun's Java Policy
78   * files see the 
79   * {@link org.apache.river.api.security.ConcurrentPolicyFile default policy description}.
80   * <br>
81   * For ordinary uses, this class has just one public method <code>parse()</code>, 
82   * which performs the main task.
83   * Extensions of this parser may redefine specific operations separately, 
84   * by overriding corresponding protected methods. 
85   * <br>
86   * This implementation is effectively thread-safe, as it has no field references 
87   * to data being processed (that is, passes all the data as method parameters).
88   * 
89   * @see org.apache.river.api.security.ConcurrentPolicyFile
90   * @see org.apache.river.api.security.DefaultPolicyScanner
91   * @see org.apache.river.api.security.PermissionGrant
92   */
93  public class DefaultPolicyParser implements PolicyParser {

121     /**
122      * This is the main business method. It manages loading process as follows:
123      * the associated scanner is used to parse the stream to a set of 
124      * {@link org.apache.river.api.security.DefaultPolicyScanner.GrantEntry composite tokens},
125      * then this set is iterated and each token is translated to a PermissionGrant.
126      * Semantically invalid tokens are ignored, the same as void PermissionGrant's.
127      * <br>
128      * A policy file may refer to some KeyStore(s), and in this case the first
129      * valid reference is initialized and used in processing tokens.   
130      * 
131      * @param location an URL of a policy file to be loaded
132      * @param system system properties, used for property expansion
133      * @return a collection of PermissionGrant objects, may be empty
134      * @throws Exception IO error while reading location or file syntax error 
135      */
136     public Collection<PermissionGrant> parse(URL location, Properties system)
137             throws Exception {
138         boolean resolve = PolicyUtils.canExpandProperties();
139         Reader r = new BufferedReader(new InputStreamReader(
140                 AccessController
141                         .doPrivileged(new PolicyUtils.URLLoader(location)), "UTF-8"));
142 
143         Collection<GrantEntry> grantEntries = new HashSet<GrantEntry>();
144         List<KeystoreEntry> keystores = new ArrayList<KeystoreEntry>();
145 
146         try {
147             scanner.scanStream(r, grantEntries, keystores); // modifies keystores
148         }
149         finally {
150             r.close();
151         }
152 
153         //XXX KeyStore could be loaded lazily...
154         KeyStore ks = initKeyStore(keystores, location, system, resolve);
155 
156         Collection<PermissionGrant> result = new HashSet<PermissionGrant>();
157 	for (DefaultPolicyScanner.GrantEntry ge : grantEntries) {
158 	    try {
159 		PermissionGrant pe = resolveGrant(ge, ks, system, resolve);
160 		if (!pe.isVoid()) {
161 		    result.add(pe);
162 		}
163 	    }
164 	    catch (Exception e) {
165 		if ( e instanceof SecurityException ) throw (SecurityException) e;
166 		// Bad policy syntax usually results in difficult to trace
167 		// problems, the sooner we fail, the sooner the problem
168 		// is evident, for example in an operating environment
169 		// with a misconfigured policy, it's better to fail 
170 		// immediately than to experience a problem later.
171 		log(Level.CONFIG, "security.1A9", new Object[]{ge}, e);
172 	    }
173 	}
174         
175         return result;
176     }

178     /**
179      * Translates GrantEntry token to PermissionGrant object. It goes step by step, 
180      * trying to resolve each component of the GrantEntry:
181      * <ul>
182      * <li> If <code>codebase</code> is specified, expand it and construct an URL.
183      * <li> If <code>signers</code> is specified, expand it and obtain 
184      * corresponding Certificates.
185      * <li> If <code>principals</code> collection is specified, iterate over it. 
186      * For each PrincipalEntry, expand name and if no class specified, 
187      * resolve actual X500Principal from a KeyStore certificate; otherwise keep it 
188      * as UnresolvedPrincipal. 
189      * <li> Iterate over <code>permissions</code> collection. For each PermissionEntry,
190      * try to resolve (see method 
191      * {@link #resolvePermission(DefaultPolicyScanner.PermissionEntry, DefaultPolicyScanner.GrantEntry, KeyStore, Properties, boolean) resolvePermission()}) 
192      * a corresponding permission. If resolution failed, ignore the PermissionEntry. 
193      * </ul>
194      * In fact, property expansion in the steps above is conditional and is ruled by
195      * the parameter <i>resolve</i>.  
196      * <br>
197      * Finally a new PermissionGrant is created, which associates the trinity 
198      * of resolved URL, Certificates and Principals to a set of granted Permissions.
199      * 
200      * @param ge GrantEntry token to be resolved
201      * @param ks KeyStore for resolving Certificates, may be <code>null</code> 
202      * @param system system properties, used for property expansion 
203      * @param resolve flag enabling/disabling property expansion
204      * @return resolved PermissionGrant
205      * @throws Exception if unable to resolve codebase, signers or principals 
206      * of the GrantEntry
207      * @see DefaultPolicyScanner.PrincipalEntry
208      * @see DefaultPolicyScanner.PermissionEntry
209      * @see org.apache.river.api.security.PolicyUtils
210      */
211     PermissionGrant resolveGrant(DefaultPolicyScanner.GrantEntry ge,
212             KeyStore ks, Properties system, boolean resolve) throws Exception {
213         if ( ge == null ) return null;
214         List<String> codebases = new ArrayList<String>(8);
215         Certificate[] signers = null;
216         Set<Principal> principals = new HashSet<Principal>();
217         Set<Permission> permissions = new HashSet<Permission>();
218         String cb = ge.getCodebase(null);
219         String signerString = ge.getSigners();
220         if ( cb != null ) {
221             if ( resolve ) {
222                 try {
223                     Collection<String> cbstr = expandURLs(cb, system);
224                     Iterator<String> it = cbstr.iterator();
225                     while (it.hasNext()){
226                         codebases.add(getURI(it.next()));
227                     }
228                 } catch (ExpansionFailedException e) {
229                     log(Level.CONFIG, "security.1A7", new Object[]{e.getMessage()});
230                 }
231             } else {
232                 codebases.add(getURI(cb));
233             }
234         }
235 	String[] aliases = new String[0];
236         if ( signerString != null) {
237 	    try {
238 		if (resolve) {
239 		    signerString = PolicyUtils.expand(signerString, system);
240 		}
241 	    } catch (ExpansionFailedException e){
242 		log(Level.CONFIG, "security.1A6", new Object[]{e.getMessage()});
243 	    }
244             
245 	    StringTokenizer snt = new StringTokenizer(signerString, ",");
246 	    List<String> alias = new ArrayList<String>(snt.countTokens());
247 	    while (snt.hasMoreTokens()){
248 		alias.add(snt.nextToken().trim());
249 	    }
250 	    aliases = alias.toArray(new String[alias.size()]);
251 	    signers = resolveSigners(ks, aliases);
252         }
253         if (ge.getPrincipals(null) != null) {
254             String principalName;
255             String principalClass;
256 	    for (PrincipalEntry pe : ge.getPrincipals(system)) {
257 		principalName = pe.getName();
258 		principalClass = pe.getKlass();
259 		try {
260 		    if (resolve) {
261 			principalName = PolicyUtils.expand(principalName, system);
262 		    }
263 		} catch (ExpansionFailedException e){
264 		    log(Level.CONFIG, "security.1A4", new Object[]{e.getMessage()});
265 		}
266 		if (principalClass == null) {
267 		    principals.add(getPrincipalByAlias(ks, principalName));
268 		} else {
269 		    principals.add(new UnresolvedPrincipal(principalClass, principalName));
270 		}
271 	    }
272         }
273         Collection<PermissionEntry> pec = ge.getPermissions();
274         if (pec != null) {
275             Iterator<PermissionEntry> iter = pec.iterator();
276             while ( iter.hasNext()) {
277                 DefaultPolicyScanner.PermissionEntry pe = iter.next();
278                 try {
279                     permissions.add(resolvePermission(pe, ge, ks, system, resolve));
280                 } catch (ExpansionFailedException e){
281 		    log(Level.CONFIG, "security.1A5", new Object[]{pe.toString(),e.getMessage()});
282 		} catch (Exception e) {
283                     if ( e instanceof SecurityException ) throw (SecurityException) e;
284                     log(Level.CONFIG, "security.1A5", new Object[]{pe.toString(),e.getMessage()});
285                 }
286             }
287         }
288         PermissionGrantBuilder pgb = PermissionGrantBuilder.newBuilder();
289         Iterator<String> iter = codebases.iterator();
290         while (iter.hasNext()){
291             pgb.uri(iter.next());
292         }
293 	
294         return pgb
295             .certificates(signers, aliases)
296             .principals(principals.toArray(new Principal[principals.size()]))
297             .permissions(permissions.toArray(new Permission[permissions.size()]))
298             .context(PermissionGrantBuilder.URI)
299             .build();
300     }

302     String getURI(String uriString) throws MalformedURLException, URISyntaxException{
303         // We do this to support windows, this is to ensure that path
304         // capitalisation is correct and illegal strings are escaped correctly.
305         if (uriString == null) return null;
306         return Uri.fixWindowsURI(uriString);
307     }

309     Segment segment(String s, Properties p) throws ExpansionFailedException{
310         final String ARRAY_START_MARK = "${{";
311         final String ARRAY_END_MARK = "}}";
312         final String ARRAY_SEPARATOR = p.getProperty("path.separator");
313         final String START_MARK = "${"; //$NON-NLS-1$
314         final String END_MARK = "}"; //$NON-NLS-1$
315         Segment primary = new Segment(s, null);
316         primary.divideAndReplace(ARRAY_START_MARK, ARRAY_END_MARK,
317                 ARRAY_SEPARATOR, p);
318         primary.divideAndReplace(START_MARK, END_MARK, null, p);
319         // Repeat twice for nested properties
320         primary.divideAndReplace(START_MARK, END_MARK, null, p);
321         primary.divideAndReplace(START_MARK, END_MARK, null, p);
322         return primary;
323     }

325     Collection<String> expandURLs(String s, Properties p) throws ExpansionFailedException{
326         Segment seg = segment(s,p);
327         Collection<String> urls = new ArrayList<String>();
328         while ( seg.hasNext() ){
329 //            urls.add(seg.next().replace(File.separatorChar, '/'));
330             urls.add(seg.next());
331         }
332         return urls;
333     }

335     /**
336      * Translates PermissionEntry token to Permission object.
337      * First, it performs general expansion for non-null <code>name</code> and
338      * properties expansion for non-null <code>name</code>, <code>action</code> 
339      * and <code>signers</code>.
340      * Then, it obtains signing Certificates(if any), tries to find a class specified by 
341      * <code>klass</code> name and instantiate a corresponding permission object.
342      * If class is not found or it is signed improperly, returns UnresolvedPermission.
343      *
344      * @param pe PermissionEntry token to be resolved
345      * @param ge parental GrantEntry of the PermissionEntry 
346      * @param ks KeyStore for resolving Certificates, may be <code>null</code>
347      * @param system system properties, used for property expansion
348      * @param resolve flag enabling/disabling property expansion
349      * @return resolved Permission object, either of concrete class or UnresolvedPermission
350      * @throws Exception if failed to expand properties, 
351      * or to get a Certificate, 
352      * or to newBuilder an instance of a successfully found class 
353      */
354     Permission resolvePermission(
355             DefaultPolicyScanner.PermissionEntry pe,
356             DefaultPolicyScanner.GrantEntry ge, KeyStore ks, Properties system,
357             boolean resolve) throws Exception {
358         String className = pe.getKlass(), name=pe.getName(), 
359                 actions=pe.getActions(), signer=pe.getSigners();
360         if (name != null) {
361             name = PolicyUtils.expandGeneral(name, new PermissionExpander(ge, ks));
362         }
363         if (resolve) {
364             if (name != null) {
365                 name = PolicyUtils.expand(name, system);
366             }
367             if (actions != null) {
368                 actions = PolicyUtils.expand(actions, system);
369             }
370             if (signer != null) {
371                 signer = PolicyUtils.expand(signer, system);
372             }
373         }
374         Certificate[] signers = (signer == null) ? null : resolveSigners(
375                 ks, signer);
376         try {
377             Class<?> klass = Class.forName(className);
378             if (PolicyUtils.matchSubset(signers, klass.getSigners())) {
379                 return PolicyUtils.instantiatePermission(klass, name, actions);
380             }
381         }
382         catch (ClassNotFoundException cnfe) {}
383         //maybe properly signed class will be loaded later
384         return new UnresolvedPermission(className, name, actions, signers);
385     }

387     /** 
388      * Specific handler for expanding <i>self</i> and <i>alias</i> protocols. 
389      */
390     class PermissionExpander implements PolicyUtils.GeneralExpansionHandler {
391 
392         // Store KeyStore
393         private final KeyStore ks;
394 
395         // Store GrantEntry
396         private final DefaultPolicyScanner.GrantEntry ge;
397 
398         /** 
399          * Combined setter of all required fields. 
400          */
401         PermissionExpander(DefaultPolicyScanner.GrantEntry ge,
402                 KeyStore ks) {
403             this.ge = ge;
404             this.ks = ks;
405         }
406 
407         /**
408          * Resolves the following protocols:
409          * <dl>
410          * <dt>self
411          * <dd>Denotes substitution to a principal information of the parental 
412          * GrantEntry. Returns a space-separated list of resolved Principals 
413          * (including wildcarded), formatting each as <b>class &quot;name&quot;</b>.
414          * If parental GrantEntry has no Principals, throws ExpansionFailedException.
415          * <dt>alias:<i>name</i>
416          * <dd>Denotes substitution of a KeyStore alias. Namely, if a KeyStore has 
417          * an X.509 certificate associated with the specified name, then returns 
418          * <b>javax.security.auth.x500.X500Principal &quot;<i>DN</i>&quot;</b> string, 
419          * where <i>DN</i> is a certificate's subject distinguished name.  
420          * </dl>
421          * @throws ExpansionFailedException - if protocol is other than 
422          * <i>self</i> or <i>alias</i>, or if data resolution failed 
423          */
424         public String resolve(String protocol, String data)
425                 throws PolicyUtils.ExpansionFailedException {
426 
427             if ("self".equals(protocol)) { //$NON-NLS-1$
428                 //need expanding to list of principals in grant clause 
429                 if (ge.getPrincipals(null) != null && !ge.getPrincipals(null).isEmpty()) {
430                     StringBuilder sb = new StringBuilder();
431 		    for (DefaultPolicyScanner.PrincipalEntry pr : ge.getPrincipals(null)) {
432 			if (pr.getKlass() == null) {
433 			    // aliased X500Principal
434 			    try {
435 				sb.append(pc2str(getPrincipalByAlias(ks, pr.getName())));
436 			    }
437 			    catch (KeyStoreException e) {
438 				throw new PolicyUtils.ExpansionFailedException(
439 					Messages.getString("security.143", pr.getName()), e); //$NON-NLS-1$
440 			    } catch (CertificateException e) {
441 				throw new PolicyUtils.ExpansionFailedException(
442 					Messages.getString("security.143", pr.getName()), e); //$NON-NLS-1$
443 			    }
444 			} else {
445 			    sb.append(pr.getKlass()).append(" \"").append(pr.getName()) //$NON-NLS-1$
446 				    .append("\" "); //$NON-NLS-1$
447 			}
448 		    }
449                     return sb.toString();
450                 } else {
451                     throw new PolicyUtils.ExpansionFailedException(
452                             Messages.getString("security.144")); //$NON-NLS-1$
453                 }
454             }
455             if ("alias".equals(protocol)) { //$NON-NLS-1$
456                 try {
457                     return pc2str(getPrincipalByAlias(ks, data));
458                 }
459                 catch (KeyStoreException e) {
460                     throw new PolicyUtils.ExpansionFailedException(
461                             Messages.getString("security.143", data), e); //$NON-NLS-1$
462                 } catch (CertificateException e) {
463 		    throw new PolicyUtils.ExpansionFailedException(
464 			    Messages.getString("security.143", data), e); //$NON-NLS-1$
465 		}
466             }
467             throw new PolicyUtils.ExpansionFailedException(
468                     Messages.getString("security.145", protocol)); //$NON-NLS-1$
469         }
470 
471         // Formats a string describing the passed Principal. 
472         private String pc2str(Principal pc) {
473             String klass = pc.getClass().getName();
474             String name = pc.getName();
475             StringBuilder sb = new StringBuilder(klass.length() + name.length()
476                     + 5);
477             return sb.append(klass).append(" \"").append(name).append("\"") //$NON-NLS-1$ //$NON-NLS-2$
478                     .toString();
479         }
480     }

482     /**
483      * Takes a comma-separated list of aliases and obtains corresponding 
484      * certificates.
485      * @param ks KeyStore for resolving Certificates, may be <code>null</code> 
486      * @param signers comma-separated list of certificate aliases, 
487      * must be not <code>null</code>
488      * @return an array of signing Certificates
489      * @throws Exception if KeyStore is <code>null</code> 
490      * or if it failed to provide a certificate  
491      */
492     Certificate[] resolveSigners(KeyStore ks, String signers)
493             throws Exception {
494         if (ks == null) {
495             throw new KeyStoreException(Messages.getString("security.146", //$NON-NLS-1$
496                     signers));
497         }
498 
499         Collection<Certificate> certs = new ArrayList<Certificate>();
500         StringTokenizer snt = new StringTokenizer(signers, ","); //$NON-NLS-1$
501         while (snt.hasMoreTokens()) {
502             //XXX cache found certs ??
503             certs.add(ks.getCertificate(snt.nextToken().trim()));
504         }
505         return certs.toArray(new Certificate[certs.size()]);
506     }
507     
508     Certificate[] resolveSigners(KeyStore ks, String[] signers) throws KeyStoreException{
509 	if (signers == null || signers.length == 0) return new Certificate[0];
510 	Collection<Certificate> certs = new ArrayList<Certificate>(signers.length);
511 	for (int i=0,l=signers.length; i<l; i++){
512 	    certs.add(ks.getCertificate(signers[i]));
513 	}
514 	return certs.toArray(new Certificate[certs.size()]);
515     }

517     /**
518      * Returns a subject's X500Principal of an X509Certificate, 
519      * which is associated with the specified keystore alias. 
520      * @param ks KeyStore for resolving Certificate, may be <code>null</code>
521      * @param alias alias to a certificate
522      * @return X500Principal with a subject distinguished name
523      * @throws KeyStoreException if KeyStore is <code>null</code> 
524      * or if it failed to provide a certificate
525      * @throws CertificateException if found certificate is not 
526      * an X509Certificate 
527      */
528     Principal getPrincipalByAlias(KeyStore ks, String alias)
529             throws KeyStoreException, CertificateException {
530 
531         if (ks == null) {
532             throw new KeyStoreException(
533                     Messages.getString("security.147", alias)); //$NON-NLS-1$
534         }
535         //XXX cache found certs ??
536         Certificate x509 = ks.getCertificate(alias);
537         if (x509 instanceof X509Certificate) {
538             return ((X509Certificate) x509).getSubjectX500Principal();
539         } else {
540             throw new CertificateException(Messages.getString("security.148", //$NON-NLS-1$
541                     alias, x509));
542         }
543     }

545     /**
546      * Returns the first successfully loaded KeyStore, from the specified list of
547      * possible locations. This method iterates over the list of KeystoreEntries;
548      * for each entry expands <code>url</code> and <code>type</code>,
549      * tries to construct instances of specified URL and KeyStore and to load 
550      * the keystore. If it is loaded, returns the keystore, otherwise proceeds to 
551      * the next KeystoreEntry. 
552      * <br>
553      * <b>Note:</b> an url may be relative to the policy file location or absolute.
554      * @param keystores list of available KeystoreEntries
555      * @param base the policy file location
556      * @param system system properties, used for property expansion
557      * @param resolve flag enabling/disabling property expansion
558      * @return the first successfully loaded KeyStore or <code>null</code>
559      */
560     KeyStore initKeyStore(List<KeystoreEntry>keystores,
561             URL base, Properties system, boolean resolve) {
562 	for (KeystoreEntry ke : keystores) {
563 	    try {
564 		String url = ke.getUrl(), type = ke.getType();
565 		if (resolve) {
566 		    url = PolicyUtils.expandURL(url, system);
567 		    if (type != null) {
568 			type = PolicyUtils.expand(type, system);
569 		    }
570 		}
571 		if (type == null || type.length() == 0) {
572 		    type = KeyStore.getDefaultType();
573 		}
574 		KeyStore ks = KeyStore.getInstance(type);
575 		URL location = new URL(base, url);
576 		InputStream is = AccessController
577 			.doPrivileged(new PolicyUtils.URLLoader(location));
578 		try {
579 		    ks.load(is, null);
580 		}
581 		finally {
582 		    is.close();
583 		}
584 		return ks;
585 	    }
586 	    catch (ExpansionFailedException e) {
587 		log(Level.CONFIG, "security.8A", e);
588 	    } catch (KeyStoreException e) {
589 		log(Level.CONFIG, "security.8A", e);
590 	    } catch (PrivilegedActionException e) {
591 		log(Level.CONFIG, "security.8A", e);
592 	    } catch (IOException e) {
593 		log(Level.CONFIG, "security.8A", e);
594 	    } catch (NoSuchAlgorithmException e) {
595 		log(Level.CONFIG, "security.8A", e);
596 	    } catch (CertificateException e) {
597 		log(Level.CONFIG, "security.8A", e);
598 	    }
599 	}
600         return null;
601     }
