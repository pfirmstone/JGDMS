138     /**
139      * Substitutes all entries like ${some.key}, found in specified string, 
140      * for specified values.
141      * If some key is unknown, throws ExpansionFailedException. 
142      * @param str the string to be expanded
143      * @param properties available key-value mappings 
144      * @return expanded string
145      * @throws ExpansionFailedException
146      */
147     public static String expand(String str, Properties properties)
148             throws ExpansionFailedException {
149         final String START_MARK = "${"; //$NON-NLS-1$
150         final String END_MARK = "}"; //$NON-NLS-1$
151         final int START_OFFSET = START_MARK.length();
152         final int END_OFFSET = END_MARK.length();
153 
154         StringBuilder result = new StringBuilder(str);
155         int start = result.indexOf(START_MARK);
156         while (start >= 0) {
157             int end = result.indexOf(END_MARK, start);
158             if (end >= 0) {
159                 String key = result.substring(start + START_OFFSET, end);
160                 String value = properties.getProperty(key);
161                 if (value != null) {
162                     result.replace(start, end + END_OFFSET, value);
163                     start += value.length();
164                 } else {
165                     throw new ExpansionFailedException(Messages.getString("security.14F", key)); //$NON-NLS-1$
166                 }
167             }
168             start = result.indexOf(START_MARK, start);
169         }
170         return result.toString();
171     }

184     /**
185      * Handy shortcut for 
186      * <code>expand(str, properties).replace(File.separatorChar, '/')</code>.
187      * @see #expand(String, Properties)
188      */
189     static String expandURL(String str, Properties properties)
190             throws ExpansionFailedException {
191         return expand(str, properties).replace(File.separatorChar, '/');
192     }

194     /**
195      * Normalizes URLs to standard ones, eliminating pathname symbols.
196      * 
197      * @param codebase -
198      *            the original URL.
199      * @return - the normalized URL.
200      * @throws URISyntaxException  
201      */
202     static URI normalizeURL(URL codebase) throws URISyntaxException {
203         if (codebase == null) return null;
204         if ("file".equals(codebase.getProtocol())) { //$NON-NLS-1$
205             try {
206                 if (codebase.getHost().length() == 0) {
207                     String path = codebase.getFile();
208 
209                     if (path.length() == 0) {
210                         // codebase is "file:"
211                         path = "*";
212                     }
213                     return Uri.uriToURI(Uri.fileToUri(new File(path)));
214                 } else {
215                     // codebase is "file://<smth>"
216                     return Uri.uriToURI(Uri.urlToUri(codebase));
217                 }
218             } catch (Exception e) {
219                 if ( e instanceof SecurityException ) throw (SecurityException) e;
220                 // Ignore
221             }
222         }
223         return Uri.uriToURI(Uri.urlToUri(codebase));
224     }

226     /**
227      * Converts a file path to URI without accessing file system
228      * (like {File#toURI()} does).
229      * 
230      * @param path -
231      *            file path.
232      * @return - the resulting URI.
233      * @throws URISyntaxException
234      */
235     static URI filePathToURI(String path) throws URISyntaxException {
236         if (File.separatorChar == '\\' && path != null){
237             path = path.replace(File.separatorChar, '/');
238             path = path.toUpperCase();
239         }
240         if ( path != null && !path.startsWith("/")) { //$NON-NLS-1$
241             return new URI("file", null, //$NON-NLS-1$
242                     new StringBuilder(path.length() + 1).append('/')
243                             .append(path).toString(), null, null);
244         }
245         return new URI("file", null, path, null, null); //$NON-NLS-1$
246     }

248     /**
249      * Instances of this interface are intended for resolving  
250      * generalized expansion expressions, of the form ${{protocol:data}}. 
251      * Such functionality is applicable to security policy files, for example.
252      * @see org.apache.river.api.security.PolicyUtils#expandGeneral(String, GeneralExpansionHandler)
253      */
254     static interface GeneralExpansionHandler {
255 
256         /**
257          * Resolves general expansion expressions of the form ${{protocol:data}}.
258          * @param protocol denotes type of resolution
259          * @param data data to be resolved, optional (may be null)
260          * @return resolved value, must not be null
261          * @throws PolicyUtils.ExpansionFailedException if expansion is impossible
262          */
263         String resolve(String protocol, String data)
264                 throws ExpansionFailedException;
265     }

267     /**
268      * Substitutes all entries like ${{protocol:data}}, found in specified string, 
269      * for values resolved by passed handler.
270      * The data part may be empty, and in this case expression 
271      * may have simplified form, as ${{protocol}}.
272      * If some entry cannot be resolved, throws ExpansionFailedException;
273      * @param str the string to be expanded
274      * @param handler the handler to resolve data denoted by protocol  
275      * @return expanded string
276      * @throws ExpansionFailedException
277      */
278     static String expandGeneral(String str,
279             GeneralExpansionHandler handler) throws ExpansionFailedException {
280         final String START_MARK = "${{"; //$NON-NLS-1$
281         final String END_MARK = "}}"; //$NON-NLS-1$
282         final int START_OFFSET = START_MARK.length();
283         final int END_OFFSET = END_MARK.length();
284 
285         StringBuilder result = new StringBuilder(str);
286         int start = result.indexOf(START_MARK);
287         while (start >= 0) {
288             int end = result.indexOf(END_MARK, start);
289             if (end >= 0) {
290                 String key = result.substring(start + START_OFFSET, end);
291                 int separator = key.indexOf(':');
292                 String protocol = (separator >= 0) ? key
293                         .substring(0, separator) : key;
294                 String data = (separator >= 0) ? key.substring(separator + 1)
295                         : null;
296                 String value = handler.resolve(protocol, data);
297                 result.replace(start, end + END_OFFSET, value);
298                 start += value.length();
299             }
300             start = result.indexOf(START_MARK, start);
301         }
302         return result.toString();
303     }

340     /**
341      * Obtains a list of locations for a policy or configuration provider.
342      * The search algorithm is as follows:
343      * <ol>
344      * <li> Look in security properties for keys of form <code>prefix + n</code>, 
345      * where <i>n</i> is an integer and <i>prefix</i> is a passed parameter. 
346      * Sequence starts with <code>n=1</code>, and keeps incrementing <i>n</i> 
347      * until next key is not found. <br> 
348      * For each obtained key, try to construct an URL instance. On success, 
349      * add the URL to the list; otherwise ignore it.
350      * <li>
351      *         If security settings do not prohibit (through 
352      *         {@link #POLICY_ALLOW_DYNAMIC the &quot;policy.allowSystemProperty&quot; property}) 
353      *         to use additional policy location, read the system property under the 
354      *         passed key parameter. If property exists, it may designate a file or 
355      *         an absolute URL. Thus, first check if there is a file with that name, 
356      *         and if so, convert the pathname to URL. Otherwise, try to instantiate   
357      *         an URL directly. If succeeded, append the URL to the list 
358      * <li>
359      *         If the additional location from the step above was specified to the 
360      *         system via &quot;==&quot; (i.e. starts with '='), discard all URLs above 
361      *         and use this only URL.
362      * </ol> 
363      * <b>Note:</b> all property values (both security and system) related to URLs are
364      * subject to {@link #expand(String, Properties) property expansion}, regardless 
365      * of the &quot;policy.expandProperties&quot; security setting.  
366      * 
367      * @param system system properties
368      * @param systemUrlKey key to additional policy location
369      * @param securityUrlPrefix prefix to numbered locations in security properties
370      * @return array of URLs to provider's configuration files, may be empty.
371      */
372     static URL[] getPolicyURLs(final Properties system,
373             final String systemUrlKey, final String securityUrlPrefix) {
374 
375         final List<URL> urls = new ArrayList<URL>();
376         boolean dynamicOnly = false;
377         URL dynamicURL = null;
378 
379         //first check if policy is set via system properties
380         if (!Util.equalsIgnoreCase(FALSE, Security.getProperty(POLICY_ALLOW_DYNAMIC))) {
381             String location = system.getProperty(systemUrlKey);
382             if (location != null) {
383                 if (location.startsWith("=")) { //$NON-NLS-1$
384                     //overrides all other urls
385                     dynamicOnly = true;
386                     location = location.substring(1);
387                 }
388                 try {
389                     location = expandURL(location, system);
390                     // location can be a file, but we need an url...
391                     final File f = new File(location);
392                     if (f.exists()) {
393                         dynamicURL = f.toURI().toURL();
394                     } 
395                     if (dynamicURL == null) {
396                         dynamicURL = new URL(location);
397                     }
398                 } catch (Exception e) {
399                     if ( e instanceof SecurityException ) throw (SecurityException) e;
400                     // TODO: log error
401                     System.err.println("Error detecting system policy location: "+e);
402                 }
403             }
404         }
405         //next read urls from security.properties 
406         if (!dynamicOnly) {
407             int i = 1;
408             while (true) { 
409                 String location = Security.getProperty(new StringBuilder(
410                                 securityUrlPrefix).append(i++).toString());
411                 if (location == null) {
412                     break;
413                 }
414                 try {
415                     location = expandURL(location, system);
416                     URL anURL = new URL(location);
417                     if (anURL != null) {
418                         urls.add(anURL);
419                     }
420                 }
421                 catch (Exception e) {
422                     if ( e instanceof SecurityException ) throw (SecurityException) e;
423                     // TODO: log error
424                     System.err.println("Error detecting security policy location: "+e);
425                 }
426             }
427         }
428         if (dynamicURL != null) {
429             urls.add(dynamicURL);
430         }
431         return urls.toArray(new URL[urls.size()]);
432     }

434     // Empty set of arguments to default constructor of a Permission.
435     private static final Class[] NO_ARGS = {};
436 
437     // One-arg set of arguments to default constructor of a Permission.
438     private static final Class[] ONE_ARGS = { String.class };
439 
440     // Two-args set of arguments to default constructor of a Permission.
441     private static final Class[] TWO_ARGS = { String.class, String.class };
442 
443     /**
444      * Tries to find a suitable constructor and instantiate a new Permission
445      * with specified parameters.  
446      *
447      * @param targetType class of expected Permission instance
448      * @param targetName name of expected Permission instance
449      * @param targetActions actions of expected Permission instance
450      * @return a new Permission instance
451      * @throws IllegalArgumentException if no suitable constructor found
452      * @throws InstantiationException any exception thrown by Constructor.newInstance()
453      */
454     static Permission instantiatePermission(Class<?> targetType,
455             String targetName, String targetActions)
456             throws InstantiationException, IllegalAccessException, 
457             IllegalArgumentException, InvocationTargetException 
458     {
459 
460         // let's guess the best order for trying constructors
461         Class[][] argTypes;
462         Object[][] args;
463         if (targetActions != null) {
464             argTypes = new Class[][] { TWO_ARGS, ONE_ARGS, NO_ARGS };
465             args = new Object[][] { { targetName, targetActions },
466                     { targetName }, {} };
467         } else if (targetName != null) {
468             argTypes = new Class[][] { ONE_ARGS, TWO_ARGS, NO_ARGS };
469             args = new Object[][] { { targetName },
470                     { targetName, /*targetActions*/ null }, {} };
471         } else {
472             argTypes = new Class[][] { NO_ARGS, ONE_ARGS, TWO_ARGS };
473             args = new Object[][] { {}, { /*targetName*/ null },
474                     { /*targetName*/ null, /*targetActions*/ null } };
475         }
476 
477         // finally try to instantiate actual permission
478         for (int i = 0; i < argTypes.length; i++) {
479             try {
480                 Constructor<?> ctor = targetType.getConstructor(argTypes[i]);
481                 return (Permission)ctor.newInstance(args[i]);
482             }
483             catch (NoSuchMethodException ignore) {}
484         }
485         throw new IllegalArgumentException(
486                 Messages.getString("security.150", targetType));//$NON-NLS-1$
487     }

489     /**
490      * Checks whether the objects from <code>what</code> array are all
491      * presented in <code>where</code> array.
492      * 
493      * @param what first array, may be <code>null</code> 
494      * @param where  second array, may be <code>null</code>
495      * @return <code>true</code> if the first array is <code>null</code>
496      * or if each and every object (ignoring null values) 
497      * from the first array has a twin in the second array; <code>false</code> otherwise
498      */
499     public static boolean matchSubset(Object[] what, Object[] where) {
500         if (what == null) {
501             return true;
502         }
503 
504         for (int i = 0; i < what.length; i++) {
505             if (what[i] != null) {
506                 if (where == null) {
507                     return false;
508                 }
509                 boolean found = false;
510                 for (int j = 0; j < where.length; j++) {
511                     if (what[i].equals(where[j])) {
512                         found = true;
513                         break;
514                     }
515                 }
516                 if (!found) {
517                     return false;
518                 }
519             }
520         }
521         return true;
522     }
