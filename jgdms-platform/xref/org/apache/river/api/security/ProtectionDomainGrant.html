34  /**
35   * ProtectionDomainGrant's become void if serialized, since ProtectionDomain's
36   * cannot be serialized.
37   * 
38   * @author Peter Firmstone
39   * @since 3.0.0
40   */
41  class ProtectionDomainGrant extends PrincipalGrant {
42      private static final long serialVersionUID = 1L;
43      protected final WeakReference<ProtectionDomain> domain;
44      private final int hashCode;
45      
46      @SuppressWarnings("unchecked")
47      ProtectionDomainGrant(WeakReference<ProtectionDomain> domain, Principal[] groups, 
48              Permission[] perm){
49          super(groups, perm);
50          this.domain = domain;
51          int hash = 7;
52          hash = 13 * hash + (this.domain != null ? this.domain.hashCode() : 0);
53          hash = 13 * hash + super.hashCode();
54          hashCode = hash;
55      }
56      
57      @Override
58      public boolean equals(Object o){
59          if (o == null) return false;
60          if (o == this) return true;
61  	if (o.hashCode() != this.hashCode()) return false;
62          if (o instanceof ProtectionDomainGrant){
63              ProtectionDomainGrant c = (ProtectionDomainGrant) o;
64              if ( !super.equals(o)) return false;
65              if ( domain !=null && domain.equals(c.domain)) return true;
66          }
67          return false;
68      }
69  
70      @Override
71      public int hashCode() {
72          return hashCode;
73      }
74      
75      public String toString(){
76          StringBuilder sb = new StringBuilder(400);
77          sb.append(super.toString())
78            .append("ProtectionDomain: \n");
79          ProtectionDomain pd = null;
80          if (domain != null){ pd = domain.get();
81              if (pd != null){
82                  sb.append(pd.toString());
83              } else {
84                  sb.append("Grant is void - ProtectionDomain is null");
85              }
86          } else {
87              sb.append("Grant applies to all ProtectionDomain's");
88          }
89          return sb.toString();
90      }

93      @Override
94      public boolean implies(ProtectionDomain pd){
95  //        if ((domain == null) && (pals.isEmpty())) return true;
96  //        if (pd == null) return false;
97          return impliesProtectionDomain(pd) && implies(getPrincipals(pd));
98  	
99      }

101     @Override
102     public boolean implies(ClassLoader cl, Principal[] pal) {
103         return impliesClassLoader(cl) && implies(pal);
104     }

106     @Override
107     public boolean implies(CodeSource codeSource, Principal[] pal) {
108         return impliesCodeSource(codeSource) && implies(pal);
109     }

111     /*
112      * Checks if passed ProtectionDomain matches this PermissionGrant. 
113      * Non-null ProtectionDomain's are
114      * compared with equals() and if false are compared by ClassLoader and
115      * CodeSource, in case of new PermissionDomain's created by a DomainCombiner
116      */   
117     // for grant
118     private boolean impliesProtectionDomain(ProtectionDomain pd) {
119         // ProtectionDomain comparison
120         if (domain == null) return true; // Dynamic grant compatibility, opposite of CodeSource and Certificate grant's
121         if (pd == null) return false;       
122         if (domain.get() == null ) return false; // grant is void.
123         if ( pd.equals(domain.get())) return true; // pd not null fast reference comparison
124         if ( impliesClassLoader(pd.getClassLoader()) 
125                 && impliesCodeSource(pd.getCodeSource()))
126         {
127             return true;
128         }
129         return false;
130     }
131 
132     // This is here for revoke and for new ProtectionDomain's created by the
133     // DomainCombiner such as those in the SubjectDomainCombiner.
134     private boolean impliesClassLoader(ClassLoader cl) {
135         if (domain == null ) return true;  // Dynamic grant compatibility
136         if (cl == null) return false;       
137         if (domain.get() == null ) return false; // is void.
138         ClassLoader thisloader = domain.get().getClassLoader(); // pd not null
139         if (thisloader != null){
140             return thisloader.equals(cl);
141         }
142         return false; // System loader if null.
143     }
144     // This is here for revoke and for new ProtectionDomain's created by the
145     // DomainCombiner such as those in the SubjectDomainCombiner.
146     private boolean impliesCodeSource(CodeSource codeSource) {
147         ProtectionDomain pd = domain != null ? domain.get(): null;
148         if (pd == null) return false; // is void - why did I have true?
149         // The CodeSource is not normalized, if the PD was created by a domain
150         // combiner it is likely that the CodeSource will have the same referent.
151         CodeSource cs = pd.getCodeSource(); // Don't normalise CodeSource.
152         if (cs == codeSource) return true; // same reference.
153         if (cs == null && codeSource == null) return true; // if both null, when pd exists.
154         if (cs == null) return false; // Null cs indicates system domain, does not imply.
155         // Most won't get to here, since domain combiners shouldn't duplicate CodeSource.
156         // But just in case...
157         // Don't use CodeSource.equals() because that causes DNS Lookup.
158         Certificate[] myCerts = cs.getCertificates();
159         Certificate[] hisCerts = codeSource.getCertificates();
160         if ( myCerts != null && !Arrays.equals(myCerts, hisCerts)) return false;
161         try {
162             URI myLocation = cs.getLocation().toURI();
163             URI hisLocation = codeSource.getLocation().toURI();
164             if (myLocation.equals(hisLocation)) return true;
165         } catch (URISyntaxException ex) {
166             // We only compare URL if we can't compare URI
167             URL myLocation = cs.getLocation();
168             URL hisLocation = codeSource.getLocation();
169             // Only use string representation to compare, DNS cache poisioning
170             // presents a security risk, so URL.equals isn't used.
171             if (myLocation.toExternalForm().equals(hisLocation.toExternalForm())) return true;
172         }
173         return false;
174     }

176     @Override
177     public boolean impliesEquivalent(PermissionGrant grant) {
178 	if (!(grant instanceof ProtectionDomainGrant)) return false;
179 	if (!super.impliesEquivalent(grant)) return false;
180 	ProtectionDomain dom = domain.get();
181 	return dom == null ? false 
182 		: dom.equals(((ProtectionDomainGrant)grant).domain.get());
183     }
184     
185     @Override
186     public boolean isDyanamic() {
187 	return true;
188     }
189 
190 
191     @Override
192     public boolean isVoid() {        
193         if ( super.isVoid()) return true;
194         if ( domain != null && domain.get() == null) return true;
195         return false;
196     }

198     @Override
199     public PermissionGrantBuilder getBuilderTemplate() {
200         PermissionGrantBuilder pgb = super.getBuilderTemplate();
201         if ( pgb instanceof PermissionGrantBuilderImp ){
202          PermissionGrantBuilderImp pgbi = (PermissionGrantBuilderImp) pgb;
203                 pgbi.setDomain(domain)
204                 .context(PermissionGrantBuilder.PROTECTIONDOMAIN);
205         }
206         return pgb;
207     }

209     //writeReplace method for serialization proxy pattern
210     private Object writeReplace() {
211         return getBuilderTemplate();
212     }

216             throws InvalidObjectException{
217         throw new InvalidObjectException("PermissionGrantBuilder required");
218     }
