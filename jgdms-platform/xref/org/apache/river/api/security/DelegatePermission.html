44  /**
45   * Permissions such as SocketPermission or FilePermission guard a resource
46   * but allow that resource to escape the control of the SecurityManager
47   * and Policy provider, this prevents them from being revoked. DelegatePermission
48   * supports Li Gong's method guard pattern, by allowing developers to implement
49   * a method guard delegate object, that checks for the DelegatePermission on
50   * every invocation, without allowing a reference to the resource to permanently
51   * escape the control of the SecurityManager or Policy provider.
52   * <p>
53   * A DelegatePermission represents any other Permission, called a candidate
54   * Permission.  A user granted a DelegatePermission does not have the privilege
55   * of the candidate Permission, although a user with a candidate Permission
56   * has the privilege of the DelegatePermission that represents the candidate, 
57   * while the @ref DelegateSecurityManager is in force.
58   * <p>
59   * A DelegatePermission requires a method guard delegate to encapsulate a privileged
60   * resource. The developer is responsible for developing the method guard wrapper, an 
61   * example for SocketFactory can be found on Apache River's svn.
62   * <p>
63   * A method guard delegates ProtectionDomain is granted the 
64   * candidate permission, the security delegate allows any user granted a
65   * DelegatePermission to utilise the functions that its candidate Permission
66   * guards, when the user no longer has the DelegatePermission, the method guard
67   * delegate no longer allows the user to access the functions guarded by the
68   * candidate permission.  A security delegate has the responsibility to
69   * prevent security sensitive objects guarded by the candidate permission from
70   * escaping.  In order to do so, a security delegate utilises Li Gong's 
71   * proposed method guard pattern.
72   * <p>
73   * Security Delegates enable sensitive objects to be used by code that isn't
74   * fully trusted you may want to monitor, such as a 
75   * file write that is limited by the number of bytes written, or a Permission
76   * to write a file, that we might decide to retract or revoke if a user
77   * does something we don't like, such as exceed a pre set limit or behave
78   * in a manner we would like to avoid, such as hogging network bandwidth.
79   * <p>
80   * If the SecurityManager installed doesn't implement DelegateSecurityManager,
81   * DelegatePermission's will be disabled.  This allows delegate's
82   * to be included in code, the decision to utilise delegate functionality may
83   * delayed until runtime or deployment.
84   * <p>
85   * The DelegatePermissionCollection returned by newPermissionCollection() is not
86   * synchronized, this decision was made because PermissionCollection's are 
87   * usually accessed from within a heterogenous PermissionCollection like 
88   * Permissions that synchronizes anyway.  The decision made for the
89   * PermissionCollection contract to be synchronized has been broken deliberately
90   * in this case, existing PermissionCollection implementations don't cleanly
91   * protect their internal state with synchronization, since the Enumeration
92   * returned by elements() will throw a ConcurrentModificationException if in a 
93   * loop when Permission's are being added to a PermissionCollection.  In this
94   * case external synchronization must be used.
95   * <p>
96   * Serialization has been implemented so the implementation is not
97   * tied to the serialized form, instead serialization proxy's are used.
98   * <p>
99   * A candidate permission is referred to as a target in the following 
100  * explanation of DelegatePermission policy file syntax.
101  * <p>
102  * The syntax of the target name approximates that used for specifying
103  * permissions in the default security policy file; it is listed below using
104  * the same grammar notation employed by <i>The Java(TM) Language
105  * Specification</i>:
106  * <pre>
107  * <i>Target</i>:
108  *   <i>DelimiterDeclaration</i>opt <i>Permissions</i> ;opt
109  *   
110  * <i>DelimiterDeclaration</i>:
111  *   delim = <i>DelimiterCharacter</i>
112  *   
113  * <i>Permissions</i>:
114  *   <i>Permission</i>
115  *   <i>Permissions</i> ; <i>Permission</i>
116  *   
117  * <i>Permission</i>:
118  *   <i>PermissionClassName</i>
119  *   <i>PermissionClassName Name</i>
120  *   <i>PermissionClassName Name</i> , <i>Actions</i>
121  *   
122  * <i>PermissionClassName</i>:
123  *   <i>ClassName</i>
124  *   
125  * <i>Name</i>:
126  *   <i>DelimitedString</i>
127  *   
128  * <i>Actions</i>:
129  *   <i>DelimitedString</i>
130  * </pre>
131  * The production for <i>ClassName</i> is the same as that used in <i>The
132  * Java Language Specification</i>.  <i>DelimiterCharacter</i> can be any
133  * unquoted non-whitespace character other than ';' (single and
134  * double-quote characters themselves are allowed).  If
135  * <i>DelimiterCharacter</i> is not specified, then the double-quote
136  * character is the default delimiter.  <i>DelimitedString</i> is the same
137  * as the <i>StringLiteral</i> production in <i>The Java Language
138  * Specification</i>, except that it is delimited by the
139  * <i>DelimiterDeclaration</i>-specified (or default) delimiter character
140  * instead of the double-quote character exclusively.
141  * <p>
142  * Note that if the double-quote character is used as the delimiter and the
143  * name or actions strings of specified permissions themselves contain nested
144  * double-quote characters, then those characters must be escaped (or in some
145  * cases doubly-escaped) appropriately.  For example, the following policy file
146  * entry would yield a <code>GrantPermission</code> containing a
147  * <code>FooPermission</code> in which the target name would include the word
148  * "quoted" surrounded by double-quote characters:
149  * <pre>
150  * permission org.apache.river.api.security.DelegatePermission
151  *     "FooPermission \"a \\\"quoted\\\" string\"";
152  * </pre>
153  * For comparison, the following policy file entry which uses a custom
154  * delimiter would yield an equivalent <code>GrantPermission</code>:
155  * <pre>
156  * permission org.apache.river.api.security.DelegatePermission
157  *     "delim=| FooPermission |a \"quoted\" string|";
158  * </pre>
159  * Some additional example policy file permissions:
160  * <pre>
161  * // allow granting of permission to listen for and accept connections
162  * permission org.apache.river.api.security.DelegatePermission
163  *     "java.net.SocketPermission \"localhost:1024-\", \"accept,listen\"";
164  *
165  * // allow granting of permissions to read files under /foo, /bar directories
166  * permission org.apache.river.api.security.DelegatePermission 
167  *     "delim=' java.io.FilePermission '/foo/-', 'read'; java.io.FilePermission '/bar/-', 'read'";
168  * </pre>
169  * 
170  * @author Peter Firmstone
171  * @since 3.0.0
172  */

188      * like SocketPermission and to allow caching.
189      * 
190      * @param p Permission to be represented.
191      * @return DelegatePermission

193     public static Permission get(Permission p){
194 	Permission del = (Permission) instances.get(p);
195 	if ( del == null ){
196 	    del = new DelegatePermission(p);
197             @SuppressWarnings("unchecked")
198 	    Permission existed = (Permission) instances.putIfAbsent(p, del);
199 	    if ( existed != null ){
200 		del = existed;
201 	    }
202 	}
203 	return del;
204     }

217     /**
218      * Parses permission information from given GrantPermission name string.
219      * Throws an IllegalArgumentException if the name string is misformatted.
220      */
221     private static PermissionEntry[] parsePermissions(String s) {
222 	try {
223 	    ArrayList l = new ArrayList();
224 	    StreamTokenizer st = createTokenizer(s);
225 	    char delim = '"';
226 
227 	    if (st.nextToken() == StreamTokenizer.TT_WORD &&
228 		st.sval.equals("delim"))
229 	    {
230 		if (st.nextToken() == '=') {
231 		    if (st.nextToken() == StreamTokenizer.TT_WORD) {
232 			if (st.sval.length() > 1) {
233 			    throw new IllegalArgumentException(
234 				"excess delimiter characters");
235 			}
236 			delim = st.sval.charAt(0);
237 		    } else {
238 			delim = (char) st.ttype;
239 		    }
240 		    if (delim == ';') {
241 			throw new IllegalArgumentException(
242 			    "illegal delimiter ';'");
243 		    }
244 		} else {	// rewind
245 		    st = createTokenizer(s);
246 		}
247 		st.nextToken();
248 	    }
249 	    st.quoteChar(delim);
250 
251 	    do {
252 		String type, name = null, actions = null;
253 
254 		if (st.ttype != StreamTokenizer.TT_WORD) {
255 		    throw new IllegalArgumentException(
256 			"expected permission type");
257 		}
258 		type = st.sval;
259 		
260 		// REMIND: allow unquoted name/actions?
261 		st.nextToken();
262 		if (st.ttype == StreamTokenizer.TT_EOF || st.ttype == ';') {
263 		    l.add(new PermissionEntry(type, null, null, null));
264 		    continue;
265 		} else if (st.ttype == delim) {
266 		    name = st.sval;
267 		} else {
268 		    throw new IllegalArgumentException(
269 			"expected permission name or ';'");
270 		}
271 		
272 		st.nextToken();
273 		if (st.ttype == StreamTokenizer.TT_EOF || st.ttype == ';') {
274 		    l.add(new PermissionEntry(type, name, null, null));
275 		    continue;
276 		} else if (st.ttype != ',') {
277 		    throw new IllegalArgumentException("expected ',' or ';'");
278 		}
279 
280 		if (st.nextToken() != delim) {
281 		    throw new IllegalArgumentException(
282 			"expected permission actions");
283 		}
284 		actions = st.sval;
285 		
286 		st.nextToken();
287 		if (st.ttype == StreamTokenizer.TT_EOF || st.ttype == ';') {
288 		    l.add(new PermissionEntry(type, name, actions, null));
289 		    continue;
290 		} else {
291 		    throw new IllegalArgumentException("expected ';'");
292 		}
293 
294 	    } while (st.nextToken() != StreamTokenizer.TT_EOF);
295 
296 	    return (PermissionEntry[]) l.toArray(new PermissionEntry[l.size()]);
297 	} catch (IOException ex) {
298 	    throw new InternalError(ex);
299 	}
300     }

302     /**
303      * Returns tokenizer for parsing given string.  The tokenizer is configured
304      * similarly to that used by sun.security.provider.PolicyParser, except
305      * that comments are disabled and no quote character is set (yet).
306      */
307     private static StreamTokenizer createTokenizer(String s) {
308 	StreamTokenizer st = new StreamTokenizer(new StringReader(s));
309 	st.resetSyntax();
310 	st.wordChars('a', 'z');
311 	st.wordChars('A', 'Z');
312 	st.wordChars('.', '.');
313 	st.wordChars('0', '9');
314 	st.wordChars('_', '_');
315 	st.wordChars('$', '$');
316 	st.wordChars(128 + 32, 255);
317 	st.whitespaceChars(0, ' ');
318 	st.lowerCaseMode(false);
319 	st.ordinaryChar('/');
320 	st.slashSlashComments(false);
321 	st.slashStarComments(false);
322 	return st;
323     }

333 	if (permission == null) return false;
334 	if (!(permission instanceof DelegatePermission)) return false;
335 	if ( permission.getClass() != this.getClass()) return false;

408     private static class DelegatePermissionCollection extends PermissionCollection {
409 	private static final long serialVersionUID = 1L;
410 	private final transient PermissionCollection candidates;
411 	private final Set<Permission> delegates;
412 	
413 	DelegatePermissionCollection(){
414 	    candidates = new Permissions();
415 	    delegates = new HashSet<Permission>(32);
416 	}
417 
418 	@Override
419 	public void add(Permission permission) {
420 	    if (! (permission instanceof DelegatePermission))
421 	    throw new IllegalArgumentException("invalid permission: "+ permission);
422 	if (isReadOnly())
423 	    throw new SecurityException("attempt to add a Permission to a " +
424 		    "readonly PermissionCollection");
425 	    delegates.add(permission);
426 	    candidates.add(((DelegatePermission) permission).getPermission());
427 	}
428 
429 	@Override
430 	public boolean implies(Permission permission) {
431 	    if ( !(permission instanceof DelegatePermission)) return false;
432 	    return candidates.implies(
433 		    ((DelegatePermission )permission).getPermission());
434 	}
435 
436 	@Override
437 	public Enumeration<Permission> elements() {
438 	    return Collections.enumeration(delegates);
439 	}
