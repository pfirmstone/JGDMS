18  package org.apache.river.api.security;
19  
20  import java.security.Permission;
21  import org.apache.river.api.common.Beta;
22  
23  /**
24   * Jar files that include a META-INF/PERMISSIONS.LIST file,
25   * allow a {@link net.jini.security.ProxyPreparer} to grant these permissions dynamically.
26   * <p>
27   * The PERMISSIONS.LIST file must be UTF-8 encoded, the format of the file 
28   * is line based, length is not limited, however lines must be readable by 
29   * {@link java.io.BufferedReader#readLine() }.
30   * <p>
31   * <h2>Syntax:</h2>
32   * Comments begin with # or //.
33   * <p>
34   * A permission is represented by three whitespace delimited strings:
35   * <ul>
36   * <li>type - The fully qualified class name of the permission. The class must be a subclass
37   *        of <code>java.security.Permission</code> and must define a
38   *        2-argument constructor that takes a <i>name</i> string and an
39   *        <i>actions</i> string.
40   * 
41   * <li>name - The permission name that will be passed as the first argument
42   *        to the constructor of the <code>Permission</code> class identified
43   *        by <code>type</code>.
44   * 
45   * <li>actions - The permission actions that will be passed as the second
46   *        argument to the constructor of the <code>Permission</code> class
47   *        identified by <code>type</code>.
48   * </ul>
49   * <p>
50   * The encoded format is:
51   *
52   * <pre>
53   * (type)
54   * </pre>
55   *
56   * or
57   *
58   * <pre>
59   * (type &quot;name&quot;)
60   * </pre>
61   *
62   * or
63   *
64   * <pre>
65   * (type &quot;name&quot; &quot;actions&quot;)
66   * </pre>
67   *
68   * where <i>name</i> and <i>actions</i> are strings that must be encoded for
69   * proper parsing. Specifically, the <code>&quot;</code>,<code>\</code>,
70   * carriage return, and line feed characters must be escaped using
71   * <code>\&quot;</code>, <code>\\</code>,<code>\r</code>, and
72   * <code>\n</code>, respectively.
73   *
74   * <p>
75   * The encoded string contains no leading or trailing whitespace characters.
76   * A single space character is used between <i>type</i> and
77   * &quot;<i>name</i>&quot; and between &quot;<i>name</i>&quot; and
78   * &quot;<i>actions</i>&quot;.
79   * 
80   * @author peter
81   * @since 3.0.0
82   */
83  @Beta
84  public interface AdvisoryDynamicPermissions {
85      
86      public static final Permission [] DEFAULT_PERMISSIONS = new Permission [0];
87      
88      /**
89       * Advisory permissions for smart proxy's, the client may or may not grant them.
90       * This method is typically called from a 
91       * {@link net.jini.security.ProxyPreparer} implementation.  If the client
92       * chooses to grant these permissions, it may do so to a specific proxy 
93       * with Principals belonging to a specific ClassLoader after authenticating
94       * the proxy's principal.
95       * 
96       * @return unshared array containing only advisory permissions and no null references.
97       * 
98       */
99      public Permission [] getPermissions();
100 }
