31       * The resulting class descriptor is not fully functional; it can only be
32       * used as input to the ObjectInputStream.resolveClass() and
33       * ObjectStreamClass.initNonProxy() methods.
34       */
35      static ObjectStreamClass convert(ObjectStreamClassInformation o) 
36  	    throws IOException, ClassNotFoundException {
37  	ByteArrayOutputStream bao = new ByteArrayOutputStream();
38  	ObjectOutputStream dao = new ObjectOutputStream(bao);
39  	o.write(dao, true);
40  	dao.flush();
41  	byte[] bytes = bao.toByteArray();
42  	ClassDescriptorConversionObjectInputStream pois 
43  		= new ClassDescriptorConversionObjectInputStream(
44  			new ByteArrayInputStream(bytes)
45  		);
46  	return pois.readClassDescriptor();
47      }

49      static ObjectStreamClassInformation convert(ObjectStreamClass o) 
50  	    throws IOException, ClassNotFoundException {
51  	ByteArrayOutputStream bao = new ByteArrayOutputStream();
52  	ClassDescriptorConversionObjectOutputStream coos 
53  		= new ClassDescriptorConversionObjectOutputStream(bao);
54  	coos.writeClassDescriptor(o);
55  	coos.flush();
56  	byte[] bytes = bao.toByteArray();
57  	ObjectInputStream in = new ObjectInputStream(new ByteArrayInputStream(bytes));
58  	ObjectStreamClassInformation result = new ObjectStreamClassInformation();
59  	result.read(in);
60  	result.readFields(in);
61  	return result;
62      }

80      @Override
81      public String toString() {
82  	String endLine = "\n";
83  	StringBuilder b = new StringBuilder(512);
84  	b.append("Name: ").append(fullyQualifiedClassName).append(endLine).append("Externalizable? ").append(externalizable).append(endLine).append("Serializable? ").append(serializable).append(endLine).append("Has writeObject() data? ").append(hasWriteObjectData).append(endLine).append("Has block external data? ").append(hasBlockExternalData).append(endLine).append("Is Enum? ").append(isEnum).append(endLine);
85  	if (fields != null) {
86  	    for (int i = 0, l = fields.length; i < l; i++) {
87  		if (fields[i] != null) {
88  		    b.append("Field name: ").append(fields[i].getName()).append(endLine).append("Field Type Code: ").append(fields[i].getTypeCode()).append(endLine).append("Field offset: ").append(fields[i].getOffset()).append(endLine);
89  		}
90  	    }
91  	}
92  	b.append("Has Handle? ").append(hasHandle).append(endLine).append("Handle: ").append(handle).append(endLine).append("Primitive data size: ").append(primDataSize).append(endLine).append("Number of Object fields: ").append(numObjFields).append(endLine);
93  	return b.toString();
94      }

96      /**
97       * Writes class descriptor information to given DataOutputStream.
98       */
99      void write(ObjectOutputStream out, boolean replaceHandleWithObject) throws IOException {
100 	out.writeUTF(fullyQualifiedClassName);
101 	out.writeLong(serialVer);
102 	byte flags = 0;
103 	if (externalizable) {
104 	    flags |= ObjectStreamConstants.SC_EXTERNALIZABLE;
105 	    flags |= ObjectStreamConstants.SC_BLOCK_DATA; // Stream protocol version 1 isn't supported.
106 	} else if (serializable) {
107 	    flags |= ObjectStreamConstants.SC_SERIALIZABLE;
108 	}
109 	if (hasWriteObjectData) {
110 	    flags |= ObjectStreamConstants.SC_WRITE_METHOD;
111 	}
112 	if (isEnum) {
113 	    flags |= ObjectStreamConstants.SC_ENUM;
114 	}
115 	out.writeByte(flags);
116 	out.writeShort(fields.length);
117 	for (int i = 0, l = fields.length; i < l; i++) {
118 	    ObjectStreamField f = fields[i];
119 	    out.writeByte(f.getTypeCode());
120 	    out.writeUTF(f.getName());
121 	    if (!f.isPrimitive()) {
122 		String typeString = f.getTypeString();
123 		if (typeString == null) {
124 		    out.writeByte(ObjectStreamConstants.TC_NULL);
125 		} else if (hasHandle && !replaceHandleWithObject) {
126 		    out.writeByte(ObjectStreamConstants.TC_REFERENCE);
127 		    out.writeInt(handle);
128 		} else {
129 		    out.writeByte(ObjectStreamConstants.TC_STRING);
130 		    out.writeUTF(typeString);
131 		}
132 	    }
133 	}

136     /**
137      * Reads non-proxy class descriptor information from given DataInputStream.
138      */
139     void read(ObjectInputStream in) throws IOException, ClassNotFoundException {
140 	//    	System.out.println("read in class descriptor");
141 	fullyQualifiedClassName = in.readUTF();
142 	if (fullyQualifiedClassName.length() == 0) {
143 	    // luni.07 = The stream is corrupted
144 	    throw new IOException(Messages.getString("luni.07")); //$NON-NLS-1$
145 	}
146 	serialVer = in.readLong();
147 	byte flags = in.readByte();
148 	hasWriteObjectData = ((flags & ObjectStreamConstants.SC_WRITE_METHOD) != 0);
149 	hasBlockExternalData = ((flags & ObjectStreamConstants.SC_BLOCK_DATA) != 0);
150 	externalizable = ((flags & ObjectStreamConstants.SC_EXTERNALIZABLE) != 0);
151 	boolean sflag = (flags & ObjectStreamConstants.SC_SERIALIZABLE) != 0;
152 	if (externalizable && sflag) {
153 	    throw new InvalidClassException(fullyQualifiedClassName, "serializable and externalizable flags conflict");
154 	}
155 	serializable = externalizable || sflag;
156 	isEnum = ((flags & ObjectStreamConstants.SC_ENUM) != 0);
157 	if (isEnum && serialVer != 0L) {
158 	    throw new InvalidClassException(fullyQualifiedClassName, "enum descriptor has non-zero serialVersionUID: " + serialVer);
159 	}
160     }

239     void readFields(ObjectInputStream in) throws IOException {
240 	//    	System.out.println("readFields");
241 	int numFields = in.readShort();
242 	if (isEnum && numFields != 0) {
243 	    throw new InvalidClassException(fullyQualifiedClassName, "enum descriptor has non-zero field count: " + numFields);
244 	}
245 	fields = ((numFields > 0) ? new ObjectField[numFields] : ObjectStreamClass.NO_FIELDS);
246 	for (int i = 0; i < numFields; i++) {
247 	    char tcode = (char) in.readByte();
248 	    String fname = in.readUTF();
249 	    String signature = null;
250 	    if ((tcode == 'L') || (tcode == '[')) {
251 		byte streamConstant = (in instanceof AtomicMarshalInputStream) ? ((AtomicMarshalInputStream) in).nextTC() : in.readByte();
252 		switch (streamConstant) {
253 		    case ObjectStreamConstants.TC_NULL:
254 			//    			System.out.println("TC_NULL");
255 			break;
256 		    case ObjectStreamConstants.TC_REFERENCE:
257 			//    			System.out.println("TC_REFERENCE");
258 			if (in instanceof AtomicMarshalInputStream) {
259 			    signature = (String) ((AtomicMarshalInputStream) in).readCyclicReference();
260 			} else {
261 			    throw new StreamCorruptedException("TC_REFERENCE is not supported for this stream");
262 			}
263 			break;
264 		    case ObjectStreamConstants.TC_STRING:
265 			//    			System.out.println("TC_STRING");
266 			if (in instanceof AtomicMarshalInputStream) {
267 			    signature = (String) ((AtomicMarshalInputStream) in).readNewString(isEnum);
268 			} else {
269 			    signature = in.readUTF();
270 			}
271 			break;
272 		    case ObjectStreamConstants.TC_LONGSTRING:
273 			//    			System.out.println("TC_LONGSTRING");
274 			if (in instanceof AtomicMarshalInputStream) {
275 			    signature = (String) ((AtomicMarshalInputStream) in).readNewLongString(isEnum);
276 			} else {
277 			    throw new UnsupportedOperationException("Cannot read long UTF string from this stream");
278 			}
279 		    default:
280 			throw new StreamCorruptedException("Stream failed in ObjectStreamClass descriptor");
281 		}
282 	    } else {
283 		signature = new String(new char[]{tcode});
284 	    }
285 	    try {
286 		fields[i] = new ObjectField(fname, signature, false);
287 	    } catch (RuntimeException e) {
288 		IOException ex = new InvalidClassException(fullyQualifiedClassName, "invalid descriptor for field " + fname);
289 		ex.initCause(e);
290 		throw ex;
291 	    }
292 	}
293 	primDataSize = 0;
294 	numObjFields = 0;
295 	int firstObjIndex = -1;
296 	for (int i = 0, l = fields.length; i < l; i++) {
297 	    ObjectField f = (ObjectField) fields[i];
298 	    switch (f.getTypeCode()) {
299 		case 'Z':
300 		case 'B':
301 		    f.setOffset(primDataSize++);
302 		    break;
303 		case 'C':
304 		case 'S':
305 		    f.setOffset(primDataSize);
306 		    primDataSize += 2;
307 		    break;
308 		case 'I':
309 		case 'F':
310 		    f.setOffset(primDataSize);
311 		    primDataSize += 4;
312 		    break;
313 		case 'J':
314 		case 'D':
315 		    f.setOffset(primDataSize);
316 		    primDataSize += 8;
317 		    break;
318 		case '[':
319 		case 'L':
320 		    f.setOffset(numObjFields++);
321 		    if (firstObjIndex == -1) {
322 			firstObjIndex = i;
323 		    }
324 		    break;
325 		default:
326 		    throw new InternalError();
327 	    }
328 	}
329 	if (firstObjIndex != -1 && firstObjIndex + numObjFields != fields.length) {
330 	    throw new InvalidClassException(fullyQualifiedClassName, "illegal field order");
331 	}

356 
357 	private final Field field;
358 	private boolean unshared;
359 
360 	static String checkName(String name) {
361 	    if (name == null) {
362 		throw new NullPointerException();
363 	    }
364 	    return name;
365 	}
366 
367 	ObjectField(String name, Class<?> type) {
368 	    this(name, type, false);
369 	}
370 
371 	ObjectField(String name, Class<?> type, boolean unshared) {
372 	    super(checkName(name), type, unshared);
373 	    field = null;
374 	    this.unshared = unshared;
375 	}
376 
377 	static Class<?> getType(String signature) {
378 	    if (signature == null) {
379 		throw new IllegalArgumentException("illegal signature, cannot be null");
380 	    }
381 	    switch (signature.charAt(0)) {
382 		case 'B':
383 		    return Byte.TYPE;
384 		case 'C':
385 		    return Character.TYPE;
386 		case 'D':
387 		    return Double.TYPE;
388 		case 'F':
389 		    return Float.TYPE;
390 		case 'I':
391 		    return Integer.TYPE;
392 		case 'J':
393 		    return Long.TYPE;
394 		case 'L':
395 		case '[':
396 		    return Object.class;
397 		case 'S':
398 		    return Short.TYPE;
399 		case 'Z':
400 		    return Boolean.TYPE;
401 		default:
402 		    throw new IllegalArgumentException("illegal signature: " + signature);
403 	    }
404 	}
405 
406 	/**
407 	 * Creates an ObjectField representing a field with the given name,
408 	 * signature and unshared setting.
409 	 */
410 	ObjectField(String name, String signature, boolean unshared) {
411 	    this(name, getType(signature), unshared);
412 	}

414 	static Class<?> type(Class<?> ftype, boolean showType) {
415 	    return (showType || ftype.isPrimitive()) ? ftype : Object.class;
416 	}

418 	/**
419 	 * Creates an ObjectField representing the given field with the
420 	 * specified unshared setting. For compatibility with the behavior of
421 	 * earlier serialization implementations, a "showType" parameter is
422 	 * necessary to govern whether or not a getType() call on this
423 	 * ObjectField (if non-primitive) will return Object.class (as opposed
424 	 * to a more specific reference type).
425 	 */
426 	ObjectField(Field field, boolean unshared, boolean showType) {
427 	    this(field.getName(), type(field.getType(), showType), unshared);
428 	}
