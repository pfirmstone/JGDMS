7   package org.apache.river.api.io;
8   
9   import java.io.IOException;
10  import java.io.ObjectOutputStream;
11  import java.io.ObjectStreamException;
12  import java.io.ObjectStreamField;
13  import java.io.Serializable;
14  import java.util.Iterator;
15  import java.util.Map;
16  import java.util.Properties;
17  import org.apache.river.api.io.AtomicSerial.GetArg;
18  
19  /**
20   *
21   * @author peter
22   */
23  @Serializer(replaceObType = Properties.class)
24  @AtomicSerial
25  class PropertiesSerializer implements Serializable {
26      private static final long serialVersionUID = 1L;
27      
28      /**
29       * By defining serial persistent fields, we don't need to use transient fields.
30       * All fields can be final and this object becomes immutable.
31       */
32      private static final ObjectStreamField[] serialPersistentFields = 
33  	{
34  	    new ObjectStreamField("m", MapSerializer.class)
35  	};
36      
37      
38      private final Properties p;
39      private final MapSerializer m;
40      
41      PropertiesSerializer(Properties p){
42  	this.p = p;
43  	m = new MapSerializer(p);
44      }
45      
46      PropertiesSerializer(GetArg arg) throws IOException{
47  	this(getProperties(arg.get("m", null, MapSerializer.class)));
48      }
49      
50      static Properties getProperties(MapSerializer ms){
51  	Properties p = new Properties();
52  	Iterator<Map.Entry> it = ms.entrySet().iterator();
53  	while (it.hasNext()){
54  	    Map.Entry e = it.next();
55  	    String key = (String) e.getKey();
56  	    String value = (String) e.getValue();
57  	    p.put(key, value);
58  	}
59  	return p;
60      }
61      
62      Object readResolve() throws ObjectStreamException {
63  	return p;
64      }
65      
66      /**
67       * @serialData 
68       * @param out
69       * @throws IOException 
70       */
71      private void writeObject(ObjectOutputStream out) throws IOException {
72  	ObjectOutputStream.PutField pf = out.putFields();
73  	pf.put("m", m);
74  	out.writeFields();
75      }
76  }
