51  /**
52   * This AtomicMarshalOutputStream, replaces a number of Java Object's in the stream 
53   * with Serializer's that ordinarily would not be deserializable by 
54   * AtomicMarshalInputStream or would not be safe to be deserialized, this 
55   * includes, but is not limited to Java Collections classes, Throwable 
56   * subclasses and object versions of primitive values.
57   * 
58   * @author peter
59   */
60  public class AtomicMarshalOutputStream extends MarshalOutputStream {
61      private static final Logger logger = Logger.getLogger("org.apache.river.api.io");
62      
63      final DelegateObjectOutputStream d;
64      
65      final ClassLoader defaultLoader;
66      
67      public AtomicMarshalOutputStream(OutputStream out, Collection context) throws IOException {
68  	this(out, null, context, false);
69      }
70      
71      /**
72       *
73       * @param out
74       * @param context
75       * @param objectOutputStreamMode if true stream format is compatible with 
76       * ObjectOutputStream, if false, it's compatible with MarshalOutputStream.
77       * @throws IOException
78       */
79      public AtomicMarshalOutputStream(OutputStream out, ClassLoader defaultLoader, Collection context, boolean objectOutputStreamMode) throws IOException{
80  	super(context);
81  	this.defaultLoader= defaultLoader;
82  	d = new DelegateObjectOutputStream(out, this, objectOutputStreamMode);
83  	d.enableReplaceObject(true);
84      }
85      
86      @Override
87      public final void writeObjectOverride(Object obj) throws IOException {
88  	d.writeObject(obj);
89  	// This is where we check the number of Object's cached and
90  	// reset if we're getting towards half our limit.
91  	// One day this limit will become too small.
92  	if (d.numObjectsCached > 32768) reset();
93      }
94      
95      @Override
96      public void writeUnshared(Object obj) throws IOException {
97  	d.writeUnshared(obj);
98      }

220     /**
221      * Reset will disregard the state of any objects already written to the
222      * stream.  The state is reset to be the same as a new ObjectOutputStream.
223      * The current point in the stream is marked as reset so the corresponding
224      * ObjectInputStream will be reset at the same point.  Objects previously
225      * written to the stream will not be referred to as already being in the
226      * stream.  They will be written to the stream again.
227      *
228      * @throws  IOException if reset() is invoked while serializing an object.
229      */
230     @Override
231     public void reset() throws IOException {
232 	d.numObjectsCached = 0;
233 	d.reset();
234     }

251     private static final class DelegateObjectOutputStream extends ObjectOutputStream {
252     
253 	final Map<Class,Class> serializers;
254 	final AtomicMarshalOutputStream aout;
255 	int numObjectsCached = 0;
256 	final boolean objectOutputStreamMode;
257 	boolean enableReplaceObject;
258 	
259 	DelegateObjectOutputStream(OutputStream out, AtomicMarshalOutputStream aout, boolean objectOutputStreamMode) throws IOException{
260 	    super(out);
261 	    this.aout = aout;
262 	    this.objectOutputStreamMode = objectOutputStreamMode;
263 	    this.serializers = new HashMap<Class,Class>(24);
264 	    serializers.put(Byte.class, ByteSerializer.class);
265 	    serializers.put(Short.class, ShortSerializer.class);
266 	    serializers.put(Integer.class, IntSerializer.class);
267 	    serializers.put(Long.class, LongSerializer.class);
268 	    serializers.put(Double.class, DoubleSerializer.class);
269 	    serializers.put(Float.class, FloatSerializer.class);
270 	    serializers.put(Character.class, CharSerializer.class);
271 	    serializers.put(Boolean.class, BooleanSerializer.class);
272 	    serializers.put(Properties.class, PropertiesSerializer.class);
273 	    serializers.put(URL.class, URLSerializer.class);
274 	    serializers.put(URI.class, URISerializer.class);
275 	    serializers.put(UID.class, UIDSerializer.class);
276 	    serializers.put(File.class, FileSerializer.class);
277 	    serializers.put(MarshalledObject.class, MarshalledObjectSerializer.class);
278 	    serializers.put(ActivationGroupDesc.class, ActivationGroupDescSerializer.class);
279 	    serializers.put(ActivationGroupID.class, ActivationGroupIDSerializer.class);
280 	    serializers.put(ActivationDesc.class, ActivationDescSerializer.class);
281 	    serializers.put(CommandEnvironment.class, CmdEnv.class);
282 	    serializers.put(StackTraceElement.class, StackTraceElementSerializer.class);
283             serializers.put(X500Principal.class, X500PrincipalSerializer.class);
284 	    serializers.put(Date.class, DateSerializer.class);
285 	}

304 	/**
305 	 * Important objects that cannot be safely serialized are substituted.
306 	 */
307 	@Override
308 	public Object replaceObject(Object obj) throws IOException {
309 	    numObjectsCached++;
310 	    if (enableReplaceObject) {
311 		return aout.replaceObject(obj);
312 	    } else {
313 		return defaultReplaceObject(obj);
314 	    }
315 	}
316 	
317 	Object defaultReplaceObject(Object obj) throws IOException {
318 	    /* If we have a ProxyAccessor, lets replace it now, because it's
319 	     * likely to support AtomicSerial, unlike other serializers that don't.
320 	     */
321 	    logger.log(Level.FINEST, "Object in stream instance of: {0}", obj.getClass());
322 	    try {
323 		if (obj instanceof DynamicProxyCodebaseAccessor ){
324 		    logger.log(Level.FINEST, "Object in stream instance of DynamicProxyCodebaseAccessor");
325 		    obj = 
326 		    ProxySerializer.create(
327 			    (DynamicProxyCodebaseAccessor) obj,
328 			    aout.defaultLoader,
329 			    aout.getObjectStreamContext()
330 		    );
331 		} else if (obj instanceof ProxyAccessor ) {
332 		    logger.log(Level.FINEST, "Object in stream instance of ProxyAccessor");
333 		    obj = 
334 		    ProxySerializer.create(
335 			    (ProxyAccessor) obj,
336 			    aout.defaultLoader,
337 			    aout.getObjectStreamContext()
338 		    );
339 		}
340 	    } catch (IOException e) {
341 		logger.log(Level.FINE, "Unable to create ProxyAccessorSerializer", e);
342 		throw e;
343 	    }
344 	    Class c = obj.getClass();
345 	    Class s = serializers.get(c);
346 	    if (c.isAnnotationPresent(AtomicSerial.class)){} // Ignore
347 	    else if (c.isAnnotationPresent(AtomicExternal.class)){} // Ignore
348 	    // REMIND: stateless objects, eg EmptySet?
349 	    else if (s != null){
350 		try {
351 		    Constructor constructor = s.getDeclaredConstructor(c);
352 		    obj = constructor.newInstance(obj);
353 		} catch (NoSuchMethodException ex) {
354 		    logger.log(Level.FINE, "Unable to contruct serializer", ex);
355 		} catch (SecurityException ex) {
356 		    logger.log(Level.FINE, "Unable to contruct serializer", ex);
357 		} catch (InstantiationException ex) {
358 		    logger.log(Level.FINE, "Unable to contruct serializer", ex);
359 		} catch (IllegalAccessException ex) {
360 		    logger.log(Level.FINE, "Unable to contruct serializer", ex);
361 		} catch (IllegalArgumentException ex) {
362 		    logger.log(Level.FINE, "Unable to contruct serializer", ex);
363 		} catch (InvocationTargetException ex) {
364 		    logger.log(Level.FINE, "Unable to contruct serializer", ex);
365 		}
366 	    }
367 	    else if (obj instanceof Map) obj = new MapSerializer((Map) obj);
368 	    else if (obj instanceof Set) obj = new SetSerializer((Set) obj);
369 	    else if (obj instanceof Collection) obj = new ListSerializer((Collection) obj);
370 	    else if (obj instanceof Permission) obj = new PermissionSerializer((Permission) obj);
371 	    else if (obj instanceof Throwable) obj = new ThrowableSerializer((Throwable) obj);
372 	    logger.log(Level.FINEST, "Returning object in stream instance of: {0}", obj.getClass());
373 	    return obj;
374 	}
