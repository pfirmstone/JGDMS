19  package org.apache.river.discovery;
20  
21  import java.io.IOException;
22  import java.nio.ByteBuffer;
23  import net.jini.core.constraint.InvocationConstraints;
24  import net.jini.io.UnsupportedConstraintException;
25  
26  
27  /**
28   * Interface implemented by classes which decode multicast announcement data
29   * and additionally support delayed constraint checking.
30   *
31   * @author Sun Microsystems, Inc.
32   * @since 2.1
33   */
34  public interface DelayedMulticastAnnouncementDecoder
35      extends MulticastAnnouncementDecoder
36  {
37  
38      /**
39       * Decodes the multicast announcement data contained in the given buffer in
40       * a manner that satisfies the specified absolute constraints, returning a
41       * {@link MulticastAnnouncement} instance that contains the decoded data,
42       * with constraint checking optionally delayed.
43       * <code>null</code> constraints are considered equivalent to empty
44       * constraints.  Constraint checking may be delayed using the
45       * <code>delayConstraintCheck</code> flag.
46       * <p> If the <code>delayConstraintCheck</code> flag is <code>true</code>,
47       * the method behaves as follows:<ul>
48       * <li> Some of the specified constraints may not be checked before this
49       * method returns; the returned <code>MulticastAnnouncement</code>'s
50       * {@link MulticastAnnouncement#checkConstraints checkConstraints}
51       * method must be invoked to complete checking of all the constraints.
52       * <li> Constraints which must be checked before accessor methods of the
53       * returned <code>MulticastAnnouncement</code> can be invoked are always
54       * checked before this method returns.</ul>
55       * If <code>delayConstraintCheck</code> is <code>false</code>,
56       * all the specified constraints are checked before this method returns.
57       *
58       * @param buf a buffer containing the packet data to decode.  The contents
59       * of <code>buf</code> may be used on subsequent invocations of the returned
60       * <code>MulticastAnnouncement</code> instance's
61       * <code>checkConstraints</code> method.  The caller must ensure that the
62       * contents of <code>buf</code> are not modified before invocation of the
63       * <code>checkConstraints</code> method.  Additionally, multicast
64       * announcement data must begin at position zero of <code>buf</code>.
65       * @param constraints the constraints to apply when decoding the data, or
66       * <code>null</code>
67       * @param delayConstraintCheck flag to control delayed constraint checking
68       * @return the decoded multicast announcement data
69       * @throws IOException if an error occurs in interpreting the data
70       * @throws UnsupportedConstraintException if unable to satisfy the
71       * specified constraints
72       * @throws SecurityException if the given constraints cannot be satisfied
73       * due to insufficient caller permissions
74       * @throws NullPointerException if <code>buf</code> is <code>null</code>
75       */
76      MulticastAnnouncement decodeMulticastAnnouncement(
77  					    ByteBuffer buf,
78  					    InvocationConstraints constraints,
79                                              boolean delayConstraintCheck)
80  	throws IOException;
81  }
