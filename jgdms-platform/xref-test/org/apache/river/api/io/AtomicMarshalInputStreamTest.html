34      @Test
35      public void testReadObject() throws Exception {
36  	System.out.println("available");
37  	ByteArrayOutputStream baos = new ByteArrayOutputStream();
38  	ObjectOutputStream oos = new AtomicMarshalOutputStream(baos, null);
39  	Have expResult = Have.SERVICE;
40  	oos.writeObject(expResult);
41  	oos.flush();
42  	byte [] bits = baos.toByteArray();
43  	ByteArrayInputStream bais = new ByteArrayInputStream(bits);
44  	ObjectInputStream instance = AtomicMarshalInputStream.create(bais, null, false, null, null);
45  	Have result = (Have) instance.readObject();
46  	assertEquals(expResult, result);
47      }

51  	System.out.println("Exception test");
52  	ByteArrayOutputStream baos = new ByteArrayOutputStream();
53  	ObjectOutputStream oos = new AtomicMarshalOutputStream(baos, null);
54  	UnmarshalException ue = new UnmarshalException("Synthetic");
55  	oos.writeObject(ue);
56  	oos.flush();
57  	byte [] bits = baos.toByteArray();
58  	ByteArrayInputStream bais = new ByteArrayInputStream(bits);
59  	ObjectInputStream in = AtomicMarshalInputStream.create(bais, null, false, null, null);
60  	UnmarshalException ue2 = (UnmarshalException) in.readObject();

65       * Test of createObjectInputStream method, of class AtomicMarshalInputStream.
66       */
67      @Test
68      public void testObjectOutputStreamCompatibility() throws Exception {
69  	System.out.println("test ObjectOutputStream compatibility");
70  	MarshalledInstance mi = new MarshalledInstance("Test object");
71  	ByteArrayOutputStream baos = null;
72  	byte[] data;
73  	ObjectOutputStream oos = null;
74  	ByteArrayInputStream bais = null;
75  	ObjectInputStream ois = null;
76  	try {
77  	    baos = new ByteArrayOutputStream();
78  	    oos = new ObjectOutputStream(new BufferedOutputStream(baos));
79  	    oos.writeObject(mi);
80  	    oos.flush();
81  	    data = baos.toByteArray();
82  	    bais = new ByteArrayInputStream(data);
83  	    ois = AtomicMarshalInputStream.createObjectInputStream(new BufferedInputStream(bais),
84  		    null,
85  		    false,
86  		    null,
87  		    null
88  	    );
89  	    MarshalledInstance result = (MarshalledInstance) ois.readObject();
90  	    assertEquals(mi, result);
91  	} finally {
92  	    try {
93  		if (oos != null) oos.close();
94  		else if (baos != null) baos.close();
95  		if (ois != null) ois.close();
96  		else if (bais != null) bais.close();
97  	    } catch (IOException e){}
98  	}
99      }
