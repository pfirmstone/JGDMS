140     @Test
141     public void TestQueueFuture() throws InterruptedException 
142     {
143         System.out.println("testQueueFuture");
144         Queue<Future> que = RC.queue(
145                 new ConcurrentLinkedQueue<Referrer<Future>>(), 
146                 Ref.TIME, 20L);
147         Future f = new F();
148         que.offer(f);
149         Thread.sleep(100L);
150         assertTrue(f.isCancelled());
151     }
