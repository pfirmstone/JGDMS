43  /**
44   * ReferenceProcessor is responsible for creation and collection of References 
45   * on behalf of Reference Collection implementations.
46   *
47   * @param <T> 
48   * @author peter
49   */
50  class ReferenceProcessor<T> implements ReferenceQueuingFactory<T, Referrer<T>> {
51      
52      private final static ScheduledExecutorService garbageCleaner =
53              Executors.newScheduledThreadPool(1, new SystemThreadFactory());
54      // Map to register newly created object references.
55      private final static Map<Reference,ScheduledFuture> finalizerTasks = 
56              new ConcurrentHashMap<Reference,ScheduledFuture>();
57      // Finalizer queue to advise cancellation of ScheduledFuture's, 
58      // when their ReferenceProcessor has been collected.
59      private final static ReferenceQueue<Reference> phantomQueue = 
60              new ReferenceQueue<Reference>();
61      static {
62          // Finizer Task to cancel unneeded tasks.
63          garbageCleaner.scheduleAtFixedRate(
64                  new FinalizerTask(phantomQueue, finalizerTasks), 
65                  5L, 5L, TimeUnit.MINUTES
66                  );
67      }
68      
69      private final Collection<Referrer<T>> col;
70      private final Object colLock;
71      private final RefQueue<T> queue;
72      private final Ref type;
73      private final Lock queueLock;
74      private final boolean gcThreads;
75      private volatile boolean started = false;
76      
77      ReferenceProcessor(Collection<Referrer<T>> col, Ref type, RefQueue<T> queue, boolean gcThreads, Object lock){
78          if (col == null || type == null ) throw new NullPointerException("collection or reference type cannot be null");
79          this.col = col;
80          colLock = lock;
81          this.type = type;
82          this.queue = type == Ref.STRONG ? null : queue;
83          this.gcThreads = gcThreads;
84          queueLock = new ReentrantLock();
85      }
86      
87      /**
88       * Register with executor service and finaliser for cleanup.
89       * @param GcInterval time interval between scheduled cleaning runs.
90       */
91      public void start(long GcInterval){
92         if (started) return; // Start once only.
93         synchronized (this){
94             if (started) return;
95             started = true;
96         }
97         if (queue == null) return;
98         long enqDelay = GcInterval * (9L/10L);
99         // Enque garbage task preceeds cleaner task slightly, so we don't 
100        // consume too much memory with massive collections.
101        // It would be more efficient for the enque task to perform removal
102        // from the collection while iterating, however it has been left
103        // this way in case we want to combine time and soft or weak reference
104        // behaviour in some way.
105        ScheduledFuture task;
106        task =  gcThreads || type.equals(Ref.TIME)? 
107 	       garbageCleaner.scheduleAtFixedRate(
108 		       new CleanerTask(col, queue),
109 		       GcInterval,
110 		       GcInterval,
111 		       TimeUnit.MILLISECONDS
112 	       ) 
113 	       : null;
114        scheduleFinaliserTask(task);
115        task = type.equals(Ref.TIME)? 
116 	       garbageCleaner.scheduleAtFixedRate(
117 		       new EnqueGarbageTask(col, colLock),
118 		       enqDelay,
119 		       GcInterval,
120 		       TimeUnit.MILLISECONDS
121 	       )
122                : null;
123        scheduleFinaliserTask(task);
124     }
125     
126     private void scheduleFinaliserTask(ScheduledFuture task){
127         if ( task != null ){
128            // Register with finaliser.
129             @SuppressWarnings("unchecked")
130            Reference r = new PhantomReference(this, phantomQueue);
131            finalizerTasks.put(r, task);
132        }
133     }
134 
135     @Override
136     public T pseudoReferent(Referrer<T> u) {
137         throw new UnsupportedOperationException("Not supported.");
138     }
139 
140     @Override
141     public Referrer<T> referenced(T w, boolean enque, boolean temporary) {
142         if (w == null) return null;
143         if (temporary) return ReferenceFactory.singleUseForLookup(w, type);
144         return ReferenceFactory.create(w, enque == true ? queue : null, type);
145     }
146 
147     @Override
148     public void processQueue() {
149         if (queue == null || gcThreads) return;
150         Object t = null;
151         /*
152          * The reason for using an explicit lock is if another thread is
153          * removing the garbage, we don't want to prevent all other threads
154          * accessing the underlying collection, when it blocks on poll,
155          * this means that some client threads will receive null values 
156          * on occassion, but this is a small price to pay.  
157          * Might have to employ the null object pattern.
158          */
159         if ( queueLock.tryLock()){
160             try {
161                 while ( (t = queue.poll()) != null){
162                     col.remove(t);
163                 }
164             }finally{
165                 queueLock.unlock();
166             }
167         }
168     }
169     
170     private static class EnqueGarbageTask implements Runnable{
171         private final Collection col;
172         private final Object lock; // This could be the same as col, or a map.
173         
174         EnqueGarbageTask(Collection c, Object lock){
175             col = c;
176             this.lock = lock;
177         }
178         
179         public void run() {
180             long time = System.nanoTime();
181 	    if (lock instanceof ConcurrentMap ||
182 		lock instanceof ConcurrentSkipListSet ||
183 		lock instanceof CopyOnWriteArrayList ||
184 		lock instanceof CopyOnWriteArraySet ||
185 		lock instanceof ConcurrentLinkedQueue ||
186 		lock instanceof ConcurrentLinkedDeque) 
187 	    {
188 		Iterator it = col.iterator();
189 		while (it.hasNext()){
190 		    Object r = it.next();
191 		    if (r instanceof TimeBomb) {
192 			((TimeBomb)r).updateClock(time);
193 		    }
194 		}
195 	    } else {
196 		synchronized (lock){
197 		    Iterator it = col.iterator();
198 		    while (it.hasNext()){
199 			Object r = it.next();
200 			if (r instanceof TimeBomb) {
201 			    ((TimeBomb)r).updateClock(time);
202 			}
203 		    }
204 		}
205 	    }
206         }
207         
208     }
209     
210     private static class CleanerTask implements Runnable {
211         
212         private final Collection col;
213         private final RefQueue queue;
214         
215         private CleanerTask(Collection c, RefQueue queue){
216             col = c;
217             this.queue = queue;
218         }
219         
220         @Override
221         public void run() {
222             try {
223                 for ( Object t = queue.poll(); t != null; t = queue.poll()){ 
224                     col.remove(t);
225 		    if (t instanceof Referrer){
226 			Object referent = ((Referrer)t).get();
227 			if (referent instanceof AutoCloseable){
228 			   try{
229 			       // Release any resources held by the referent.
230 			       ((AutoCloseable) referent).close();
231 			   } catch (Exception ex){} // Ignore
232                 }
233 		    }
234                 }
235             }catch(Exception e){
236                 e.printStackTrace(System.err);
237             }
238         }
239     
240     }
241     
242     private static class FinalizerTask implements Runnable {
243         
244         private final ReferenceQueue phantomQueue;
245         private final Map<Reference,ScheduledFuture> finalizerTasks ;
246         
247         private FinalizerTask(ReferenceQueue queue, 
248                 Map<Reference,ScheduledFuture> tasks){
249             phantomQueue = queue;
250             finalizerTasks = tasks;
251         }
252 
253         @Override
254         public void run() {
255             Reference p;
256             while ( (p = phantomQueue.poll()) != null){
257                 ScheduledFuture sf = finalizerTasks.remove(p);
258                 if (sf !=null) sf.cancel(true);
259                 // phantom reference is eligible for gc we don't have to
260                 // clear it, but might as well.
261                 p.clear();
262             }
263         }
264         
265     }

267     private static class SystemThreadFactory implements ThreadFactory{
268         private static final ThreadGroup g;
269         
270         static {
271             ThreadGroup tg = Thread.currentThread().getThreadGroup();
272             g = AccessController.doPrivileged( new ThreadGroupAction(tg));
273         }
274 
275         private SystemThreadFactory(){
276         }
277         
278         @Override
279         public Thread newThread(Runnable r) {
280             return AccessController.doPrivileged( new CreateThread(g, r));
281         }
282         
283     }
284     
285     private static class ThreadGroupAction implements PrivilegedAction<ThreadGroup>{
286         private ThreadGroup tg;
287         
288         ThreadGroupAction(ThreadGroup g){
289             tg = g;
290         }
291         public ThreadGroup run() {
292             try {
293                 ThreadGroup parent = tg.getParent();
294                 while (parent != null){
295                     tg = parent;
296                     parent = tg.getParent();
297                 }
298             }catch (SecurityException e){
299                 Logger.getLogger(ReferenceProcessor.class.getName()).log(Level.FINE, "Unable to get parent thread group", e);
300             }
301             return tg;
302         }
303     }
304     
305     private static class CreateThread implements PrivilegedAction<Thread>{
306         private ThreadGroup g;
307         private Runnable r;
308         
309         CreateThread(ThreadGroup g, Runnable r){
310             this.g = g;
311             this.r = r;
312         }
313         public Thread run() {
314             Thread t = new Thread(g, r);
315             try {
316                 t.setContextClassLoader(null);
317                 t.setPriority(Thread.MAX_PRIORITY);
318             } catch (SecurityException e){
319                 Logger.getLogger(ReferenceProcessor.class.getName()).log(Level.FINE, "Unable to set ContextClassLoader or Priority", e);
320             }
321             return t;
322         }
323         
324     }
