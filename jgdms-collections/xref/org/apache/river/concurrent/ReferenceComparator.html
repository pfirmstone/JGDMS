16  package org.apache.river.concurrent;
17  
18  import java.util.Comparator;
19  
20  /**
21   *
22   * @param <T> 
23   * @author Peter Firmstone.
24   */
25  class ReferenceComparator<T> extends AbstractReferenceComparator<T> {
26      private Comparator<? super T> comparator;
27      ReferenceComparator(Comparator<? super T> comparator){
28          if ( comparator == null ) throw new IllegalArgumentException("Null value prohibited");
29          this.comparator = comparator;
30      }
31      
32      Comparator<? super T> get(){
33          return comparator;
34      }
35  }
