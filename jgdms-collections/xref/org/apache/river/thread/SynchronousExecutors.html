44  /**
45   * The intent of this Executor is to share a single thread pool among tasks with
46   * dependencies that prevent them running concurrently.
47   * 
48   * @author peter
49   */
50  public class SynchronousExecutors implements Startable {
51      private static final Logger logger = Logger.getLogger("org.apache.river.impl");
52      private final Lock distributorLock;
53      private final Condition workToDo;
54      private final List<Queue<Callable>> queues ;
55      private final Distributor distributor;
56      private final Thread distributorThread;
57      private final AtomicBoolean distributorWaiting;
58      
59      public SynchronousExecutors(ScheduledExecutorService pool){
60          queues = new ArrayList<Queue<Callable>>(24);
61          distributorLock = new ReentrantLock();
62          workToDo = distributorLock.newCondition();
63          distributorWaiting = new AtomicBoolean(false);
64          distributor = new Distributor(queues, pool, distributorLock, workToDo, distributorWaiting);
65          distributorThread = new Thread(distributor ,"SynchronousQueueArray distributor");
66          distributorThread.setDaemon(false);
67      }
68      
69      void addQueue(Queue<Callable> queue){
70          synchronized (queues){
71              queues.add(queue);
72          }
73      }
74      
75      boolean removeQueue(Object queue){
76          synchronized (queues){
77              return queues.remove(queue);
78          }
79      }

83          distributorThread.start();
84      }

87          distributorThread.interrupt();
88      }

90      /**
91       * The ExecutorService returned, supports a subset of ExecutorService
92       * methods, the intent of this executor is to serialize the execution
93       * of tasks, it is up to the BlockingQueue or caller to ensure order, only 
94       * one task will execute at a time, that task will be retried if it fails,
95       * using a back off strategy of 1, 5 and 10 seconds, followed by 1, 1 and 5
96       * minutes thereafter forever, no other task will execute until the task
97       * at the head of the queue is completed successfully.
98       * 
99       * Tasks submitted must implement Callable, Runnable is not supported.
100      * 
101      * @param <T>
102      * @param queue
103      * @return 
104      */
105     public <T> ExecutorService newSerialExecutor(BlockingQueue<Callable<T>> queue){
106         QueueWrapper que = new QueueWrapper<T>(queue);
107         ExecutorService serv = new SerialExecutor<T>(que, distributorWaiting, distributorLock, workToDo, this);
108         addQueue(que);
109         return serv;
110     }

157         @Override
158         public <T> Future<T> submit(Callable<T> task) {
159             if (terminating) throw new RejectedExecutionException("ExecutorService is shutting down");
160             if (task == null) throw new NullPointerException("task cannot be null");
161             Task t = new Task<T>(task, queue, lock, workToDo);
162             if (queue.offer(t)){
163                 if (waiting.get() && !queue.stalled){
164                     lock.lock();
165                     try {
166                         workToDo.signalAll();
167                     } finally {
168                         lock.unlock();
169                     }
170                 }
171                 return t;
172             }
173             throw new RejectedExecutionException("task rejected, queue likely full");
174         }

219     private static class Distributor implements Runnable {
220         
221         private final List<Queue<Callable>> queues;
222         private final ScheduledExecutorService executor;
223         private final Lock lock;
224         private final Condition workToDo;
225         private final AtomicBoolean waiting;
226         
227         Distributor(List<Queue<Callable>> queues, ScheduledExecutorService executor, Lock lock, Condition workToDo, AtomicBoolean waiting){
228             this.queues = queues;
229             this.executor = executor;
230             this.lock = lock;
231             this.workToDo = workToDo;
232             this.waiting = waiting;
233         }
234 
235         @Override
236         public void run() {
237             int nullCount = 0; // sequence of null tasks
238             int size;
239             List<Callable> tasks = new ArrayList<Callable>(64);
240             try {
241                 while (!Thread.currentThread().isInterrupted()){
242                     try {
243                         Queue<Callable> queue;
244                         synchronized (queues){
245                             size = queues.size();
246                             for (int i = 0; i < size; i++){
247                                 queue = queues.get(i);
248                                 Callable task = queue != null ? queue.peek() : null;
249                                 if (task != null) tasks.add(task);
250                                 else nullCount++;
251                             }
252                         }
253 
254                         Iterator <Callable> it = tasks.iterator();
255                         while (it.hasNext()){
256                             Callable task = it.next();
257                             long delay = 0;
258                             if (task instanceof Task) delay = ((Task)task).delay();
259                             if (delay == 0) executor.submit(task);
260                             else executor.schedule(task, delay, TimeUnit.MILLISECONDS);
261                             nullCount = 0; // reset null count
262                         }
263                         tasks.clear();
264                         if (nullCount >= size ) {
265                             // Time for a nap.
266 			    boolean elapsed = true;
267                             lock.lock();
268                             try {
269                                 waiting.set(true);
270                                 elapsed = !workToDo.await(2, TimeUnit.SECONDS);
271                             } catch (InterruptedException ex) {
272                                 Thread.currentThread().interrupt(); // restore
273                             } finally {
274                                 waiting.set(false);
275                                 lock.unlock();
276 				if (!elapsed) logger.log(Level.FINEST, "Spurious wakeup occurred");
277                             }
278                         }
279                     } catch (Exception e){
280                         System.out.println(e);
281                         logger.log(Level.FINE, "Exception thrown by distributor: {0}", e);
282                     }
283                 }// end while
284             } finally {
285                 executor.shutdown();
286             }
287         }
288     }

290     private static class QueueWrapper<T> extends AbstractQueue<Callable<T>> implements Queue<Callable<T>>{
291         
292         final ReentrantLock lock; // lock to control the head of the queue.
293         final Queue<Callable<T>> queue;
294         Callable<T> peek;// Only ever accessed by distributor thread.
295         boolean stalled;
296         
297         QueueWrapper(Queue<Callable<T>> queue){
298             this.queue = queue;
299             lock = new ReentrantLock();
300             peek = null;
301             stalled = false;
302         }
303 
304         @Override
305         public Iterator<Callable<T>> iterator() {
306             return queue.iterator();
307         }
308 
309         @Override
310         public int size() {
311             return queue.size();
312         }
313 
314         @Override
315         public boolean offer(Callable<T> e) {
316             return queue.offer(e);
317         }
318 
319         @Override
320         public Callable<T> poll() {
321             lock.lock();
322             try {
323                 return queue.poll();
324             } finally {
325                 lock.unlock();
326             }
327         }
328 
329         @Override
330         public Callable<T> peek() {
331             boolean locked = lock.tryLock();
332             if (!locked) return null; // Pretend queue empty so director tries another queue.
333             try {
334                 if (peek == null) {
335                     peek = queue.peek();
336                     return peek;
337                 } else {
338                     stalled = true;
339                     return null; // Pretend queue empty so director tries another queue.
340                 }
341             } finally {
342                 lock.unlock();
343             }
344         }

348     private static class Task<T> implements Callable<T>, Future<T>, Comparable<Task> {
349         
350         /**
351          * Default delay backoff times. 
352          *
353          * @see #retryTime 
354          */
355         private static final long[] delays = {
356              0, // First value is never read
357              TimeUnit.SECONDS.toMillis(1),
358              TimeUnit.SECONDS.toMillis(5),
359              TimeUnit.SECONDS.toMillis(10),
360              TimeUnit.MINUTES.toMillis(1),
361              TimeUnit.MINUTES.toMillis(1),
362              TimeUnit.MINUTES.toMillis(5)
363         };
364         
365         
366         volatile boolean complete = false;
367         volatile boolean cancelled = false;
368         volatile T result = null;
369         volatile Exception exception = null;
370         private final Callable<T> task;
371         private final QueueWrapper queue;
372         private final Lock executorLock;
373         private final Condition waiting;
374         private final Condition resultAwait;
375         private final boolean comparable;
376         private int attempt;
377         
378         Task(Callable<T> c, QueueWrapper wrapper, Lock executorLock, Condition distributorWaiting){
379             task = c;
380             comparable = task instanceof Comparable;
381             queue = wrapper;
382             this.waiting = distributorWaiting;
383             resultAwait = queue.lock.newCondition();
384             attempt = 0;
385             this.executorLock = executorLock;
386         }
387         
388         /**
389          * Return the next time at which we should make another attempt.
390          * This is <em>not</em> an interval, but the actual time.
391          * <p>
392          * The implementation is free to do as it pleases with the policy
393          * here.  The default implementation is to delay using intervals of
394          * 1 second, 5 seconds, 10 seconds, and 1 minute between
395          * attempts, and then retrying every five minutes forever.
396          * <p>
397          * The default implementation assumes it is being called from
398          * the default <code>run</code> method and that the current thread
399          * holds the lock on this object. If the caller does
400          * not own the lock the result is undefined and could result in an
401          * exception.
402          */
403         long delay() {
404             int index = (attempt < delays.length ? attempt : delays.length - 1); 
405             return delays[index];
406         }
407 
408         public T call() throws Exception {
409             if (cancelled) return null;
410             boolean reschedule = false;
411             queue.lock.lock();
412             try {
413                 result = task.call();
414                 if (((Task)queue.peek).task == task 
415                         && task == ((Task)queue.queue.peek()).task)
416                 {
417                     queue.queue.poll(); // Remove successfully completed task from queue.
418                 } 
419                 queue.peek = null; // set peek to null to unblock queue.
420                 queue.stalled = false;
421                 executorLock.lock();
422                 try {
423                     waiting.signalAll();
424                 } finally {
425                     executorLock.unlock();
426                 }
427                 complete = true;
428                 resultAwait.signalAll();
429                 return result;
430             } catch (Exception e) {
431                 exception = e;
432                 reschedule = true;
433                 throw e;
434             } finally {
435                 try {
436                     if (reschedule) {
437                         attempt++;
438                         queue.peek = null; // set peek to null to unblock queue.
439                         executorLock.lock();
440                         try {
441                             waiting.signalAll();
442                         } finally {
443                             executorLock.unlock();
444                         }
445                     }
446                 } finally {
447                     queue.lock.unlock();
448                 }
449             }
450         }

467         @Override
468         public T get() throws InterruptedException, ExecutionException {
469             if (complete) {
470                 if (exception != null) throw new ExecutionException(exception);
471                 return result;
472             }
473             queue.lock.lock();
474             try {
475                 while (!complete){
476                     resultAwait.await();
477                 }
478                 return result;
479             } finally {
480                 queue.lock.unlock();
481             }
482         }
483 
484         @Override
485         public T get(long timeout, TimeUnit unit) throws InterruptedException, ExecutionException, TimeoutException {
486             if (complete) return result;
487             long begin = System.currentTimeMillis();
488             if (queue.lock.tryLock(timeout, unit)){
489                 try {
490                     while (!complete){
491                         long remain = unit.toMillis(timeout) - (System.currentTimeMillis() - begin);
492                         if ( 1L > remain ) {
493                             if (exception != null) throw new ExecutionException(exception);
494                             throw new TimeoutException(
495                                     "Timed out while waiting for result");
496                         }
497                         boolean elapsed = !resultAwait.await(remain, TimeUnit.MILLISECONDS);
498 			if (elapsed) {
499 			    if (!complete) throw new TimeoutException(
500 				    "Timed out while waiting for result");
501 			} else if (Thread.currentThread().isInterrupted()) {
502 			   
503 			}
504                     }
505                     return result;
506                 } finally {
507                     queue.lock.unlock();
508                 }
509             } else {
510                 throw new TimeoutException("Timed out while waiting for lock");
511             }
512         }

514         @Override
515         public int compareTo(Task o) {
516             if (comparable) {
517                 return ((Comparable) task).compareTo(o.task);
518             }
519             int h1 = task.hashCode();
520             int h2 = o.task.hashCode();
521             if ( h1 < h2 ) return -1;
522             if ( h1 > h2 ) return 1;
523             return 0;
524         }
