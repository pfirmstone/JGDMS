19  package org.apache.river.collection;
20  
21  import java.lang.ref.Reference;
22  import java.lang.ref.ReferenceQueue;
23  import java.lang.ref.SoftReference;
24  import java.lang.ref.WeakReference;
25  import java.util.HashMap;
26  import java.util.LinkedList;
27  import java.util.List;
28  import java.util.Map;
29  
30  /**
31   * A table that maps a key containing a weak reference to one or more values
32   * containing soft references.  The weakly-referenced object in a key are
33   * compared for identity using ==.  Entries are removed when either weak
34   * references in keys or soft references in values are cleared.  Callers must
35   * insure that instances of this class are not accessed concurrently while they
36   * are being modified.
37   *
38   * @author Sun Microsystems, Inc.
39   * @since 2.0
40   */
41  public final class WeakSoftTable {
42  
43      /* -- Fields -- */
44  
45      /** Hash table mapping WeakKeys to Lists of SoftValues. */
46      private final Map hash = new HashMap();
47  
48      /** Reference queue for cleared keys and values. */
49      private final ReferenceQueue queue = new ReferenceQueue();
50  
51      /* -- Inner classes -- */
52  
53      /**
54       * Implemented by classes to permit copying instances into reference queues
55       * when they are added to the table and to remove them from the table when
56       * the references are cleared.
57       */
58      public interface RemovableReference {
59  
60  	/**
61  	 * Returns a copy of this instance registered with the specified queue.
62  	 *
63  	 * @param queue the queue with which this instance should be registered
64  	 * @return the copy
65  	 */
66  	RemovableReference copy(ReferenceQueue queue);
67  
68  	/**
69  	 * Called with the containing map when this instance's reference is
70  	 * cleared, to remove the associated entry from the map.
71  	 *
72  	 * @param map the map from which this newly cleared instance should be
73  	 *	  removed
74  	 */
75  	void cleared(Map map);
76      }
77  
78      /**
79       * A key that maintains a weak reference to an object which should be
80       * compared by object identity.
81       */
82      public static class WeakKey extends WeakReference
83  	implements RemovableReference
84      {
85  	/** Whether the key was null, as opposed to being cleared. */
86  	private final boolean nullKey;
87  
88  	/**
89  	 * The hash code of the key.  Store it so that we can still use it when
90  	 * comparing hash buckets in the hash table after the key has been
91  	 * cleared.
92  	 */
93  	private final int hashCode;
94  
95  	/**
96  	 * Creates a key that holds a weak reference to the argument and
97  	 * compares it using ==.
98  	 */
99  	public WeakKey(Object key) {
100 	    super(key);
101 	    nullKey = key == null;
102 	    hashCode = getClass().hashCode() ^ System.identityHashCode(key);
103 	}
104 
105 	/** Creates a copy of the key registered with the queue. */
106 	protected WeakKey(WeakKey weakKey, ReferenceQueue queue) {
107 	    super(weakKey.get(), queue);
108 	    nullKey = weakKey.nullKey;
109 	    hashCode = weakKey.hashCode;
110 	}
111 
112 	public RemovableReference copy(ReferenceQueue queue) {
113 	    return new WeakKey(this, queue);
114 	}
115 
116 	public void cleared(Map map) {
117 	    map.remove(this);
118 	}
119 
120 	public int hashCode() {
121 	    return hashCode;
122 	}
123 
124 	/**
125 	 * Returns true if the argument is an instance of the same concrete
126 	 * class, and if both objects had null keys, or if neither object has
127 	 * had its weak key cleared and their values are ==.
128 	 */
129 	public boolean equals(Object o) {
130 	    if (this == o) {
131 		return true;
132 	    } else if (o == null || o.getClass() != getClass()) {
133 		return false;
134 	    }
135 	    WeakKey weakKey = (WeakKey) o;
136 	    if (weakKey.nullKey != nullKey) {
137 		return false;
138 	    } else if (nullKey) {
139 		return true;
140 	    }
141 	    Object key = weakKey.get();
142 	    return key != null && key == get();
143 	}
144     }
145 
146     /** A value that maintains a soft reference to an object. */
147     public static class SoftValue extends SoftReference
148 	implements RemovableReference
149     {
150 	/**
151 	 * The associated key.  Used to remove the hash table entry when the
152 	 * value is cleared.
153 	 */
154 	protected final WeakKey key;
155 
156 	/**
157 	 * Creates a value for the associated key that retains a soft reference
158 	 * to value.
159 	 */
160 	public SoftValue(WeakKey key, Object value) {
161 	    super(value);
162 	    this.key = key;
163 	}
164 
165 	/** Creates a copy of the value registered with the queue. */
166 	protected SoftValue(SoftValue softValue, ReferenceQueue queue) {
167 	    super(softValue.get(), queue);
168 	    key = softValue.key;
169 	}
170 
171 	public RemovableReference copy(ReferenceQueue queue) {
172 	    return new SoftValue(this, queue);
173 	}
174 
175 	public void cleared(Map map) {
176 	    List list = (List) map.get(key);
177 	    if (list != null && list.remove(this) && list.isEmpty()) {
178 		map.remove(key);
179 	    }
180 	}
181     }
182 
183     /* -- Constructors -- */
184 
185     /** Creates an instance of this class. */
186     public WeakSoftTable() { }
187 
188     /* -- Methods -- */
189 
190     /**
191      * Removes all invalidated entries from the map, that is, removes all
192      * entries whose keys or values have been discarded.  This method should be
193      * invoked once by each public mutator in this class.
194      */
195     private void processQueue() {
196 	Reference ref;
197 	while ((ref = queue.poll()) != null) {
198 	    if (ref instanceof RemovableReference) {
199 		((RemovableReference) ref).cleared(hash);
200 	    }
201 	}
202     }
203 
204     /**
205      * Returns the value associated with the specified key and index, or null
206      * if not found.  Values are stored in order, so if a given index is null,
207      * then values for higher index values will also be null.
208      */
209     public SoftValue get(WeakKey key, int index) {
210 	List list = (List) hash.get(key);
211 	if (list != null && index < list.size()) {
212 	    return (SoftValue) list.get(index);
213 	} else {
214 	    return null;
215 	}
216     }
217 
218     /**
219      * Associates an additional value with the specified key.  The value is
220      * added at the next open index.
221      */
222     public void add(WeakKey key, SoftValue value) {
223 	processQueue();
224 	List list = (List) hash.get(key);
225 	if (list == null) {
226 	    list = new LinkedList();
227 	    hash.put(key.copy(queue), list);
228 	}
229 	list.add(value.copy(queue));
230     }
231 
232     /**
233      * Removes and returns the index'th value associated with the specified
234      * key.  Returns null if the item is not found.
235      */
236     public SoftValue remove(WeakKey key, int index) {
237 	processQueue();
238 	List list = (List) hash.get(key);
239 	if (list != null && index < list.size()) {
240 	    return (SoftValue) list.remove(index);
241 	} else {
242 	    return null;
243 	}
244     }
245 
246     /** Returns a string representation of this object. */
247     public String toString() {
248 	return hash.toString();
249     }
250 }
