65  /**
66   * An invocation handler for activatable remote objects.  If the client
67   * constraints of this activatable invocation handler are not
68   * <code>null</code>, then the invocation handler's underlying proxy (if
69   * any) must implement {@link RemoteMethodControl} or a remote invocation
70   * will fail with an {@link UnsupportedConstraintException}.
71   *
72   * @author Sun Microsystems, Inc.
73   * @since 2.0
74   *
75   * This implementation recognizes the following system property:
76   * <ul>
77   * <li><code>org.apache.river.activation.enableActivateGrant</code> - This
78   * property is interpreted as a <code>boolean</code> value (see {@link
79   * Boolean#getBoolean Boolean.getBoolean}). If <code>true</code>, this
80   * implementation invokes {@link Security#grant(Class,Class) Security.grant}
81   * as defined in the specification.
82   * </ul>
83   *
84   * <p>This implementation's {@link #invoke invoke} method throws {@link
85   * IllegalArgumentException} if a remote invocation is to be made and
86   * the <code>proxy</code> argument is an instance of an interface
87   * whose binary name is
88   * <code>javax.management.MBeanServerConnection</code> or any of the
89   * names produced by the following procedure:
90   *
91   * <blockquote>
92   *
93   * For each resource named
94   * <code>org/apache/river/proxy/resources/InvocationHandler.moreProhibitedProxyInterfaces</code>
95   * that is visible to the system class loader, the contents of the
96   * resource are parsed as UTF-8 text to produce a list of interface
97   * names.  The resource must contain a list of binary names of
98   * interfaces, one per line.  Space and tab characters surrounding
99   * each name, as well as blank lines, are ignored.  The comment
100  * character is <tt>'#'</tt>; on each line, all characters starting
101  * with the first comment character are ignored.
102  *
103  * </blockquote>
104  *
105  * <p>This implementation uses the {@link Logger} named
106  * <code>net.jini.activation.ActivatableInvocationHandler</code> to log
107  * information at the following levels:
108  *
109  * <table summary="Describes what is logged by ActivatableInvocationHandler
110  * at various logging levels" border=1 cellpadding=5>
111  *
112  * <tr> <th> Level <th> Description
113  *
114  * <tr> <td> {@link Levels#FAILED FAILED} <td> exception thrown from final
115  * attempt to communicate a remote call
116  * 
117  * <tr> <td> {@link Levels#FAILED FAILED} <td> exception thrown activating
118  * the object
119  * 
120  * <tr> <td> {@link Levels#HANDLED HANDLED} <td> exception caught in
121  * attempt to communicate a remote call
122  *
123  * </table>
124  **/
125 @AtomicSerial

193     /**
194      * Creates an instance with the specified activation identifier, a
195      * possibly-<code>null</code> underlying proxy, and <code>null</code>
196      * client constraints. If the underlying proxy implements {@link
197      * RemoteMethodControl} and its constraints are not <code>null</code>,
198      * the underlying proxy of this instance is a copy of that proxy with
199      * <code>null</code> constraints. 
200      *
201      * @param	id the activation identifier
202      * @param	underlyingProxy an underlying proxy, or <code>null</code>
203      * @throws	NullPointerException if <code>id</code> is <code>null</code>
204      **/
205     public ActivatableInvocationHandler(ActivationID id,
206 					Remote underlyingProxy)
207     {
208 	if (id == null) {
209 	    throw new NullPointerException("id is null");
210 	} 
211 	this.id = id;
212 	this.uproxy = underlyingProxy;
213 	this.clientConstraints = null;
214 
215 	/*
216 	 * If underlying proxy's constraints are non-null,
217 	 * set underlying proxy to a copy of the underlying
218 	 * proxy with null constraints.
219 	 */
220 	if (uproxy instanceof RemoteMethodControl) {
221 	    MethodConstraints uproxyConstraints =
222 		((RemoteMethodControl) uproxy).getConstraints();
223 	    if (uproxyConstraints != null) {
224 		uproxy = (Remote) ((RemoteMethodControl) uproxy).
225 		    setConstraints(null);
226 	    }
227 	}
228     }

230     private static MethodConstraints checkConstraints(Remote uproxy, MethodConstraints clientConstraints)
231 	    throws InvalidObjectException 
232     {
233 	if (!hasConsistentConstraints(uproxy, clientConstraints)) {
234 		StringBuilder sb = new StringBuilder(128);
235 		sb.append("inconsistent constraints between underlying proxy and invocation handler\n")
236 		  .append("Proxy:\n");
237 		if (uproxy instanceof RemoteMethodControl)
238 		  sb.append(((RemoteMethodControl)uproxy).getConstraints());
239 		sb.append("InvocationHandler:\n");
240 		sb.append(clientConstraints);
241 		throw new InvalidObjectException(sb.toString());
242 	    }
243 	return clientConstraints;
244     }

246     /**
247      * Returns true if the constraints on the underlying proxy (if it
248      * implements {@link RemoteMethodControl}) are equivalent to the
249      * constraints of this invocation handler, or if the underlying proxy
250      * does not implement RemoteMethodControl.
251      */
252     private static boolean hasConsistentConstraints(Remote uproxy, MethodConstraints clientConstraints) {
253 	if (uproxy instanceof RemoteMethodControl) {
254 	    MethodConstraints uproxyConstraints =
255 		((RemoteMethodControl) uproxy).getConstraints();
256 	    return (clientConstraints == null ?
257 		    uproxyConstraints == null :
258 		    clientConstraints.equals(uproxyConstraints));
259 	} else {
260 	    return true;
261 	}
262     }

299     /**
300      * Processes a method invocation made on the encapsulating
301      * proxy instance, <code>proxy</code>, and returns the result.
302      * This method is invoked when a method is invoked on a proxy
303      * instance that this handler is associated with.
304      *
305      * <p>If the specified method is one of the following
306      * <code>java.lang.Object</code> methods, it will be processed as follows:
307      * </p><ul>
308      * <li>{@link Object#equals equals}: returns
309      * <code>true</code> if the argument is an
310      * instance of a dynamically generated {@link Proxy}
311      * class that implements the same ordered set of interfaces as the
312      * specified proxy, and this invocation handler is equal to the invocation
313      * handler of that parameter, and returns <code>false</code> otherwise.
314      * <li>{@link Object#hashCode hashCode}: returns the hash code for the
315      * proxy. 
316      * <li>{@link Object#toString toString}: returns a string
317      * representation of the specified <code>proxy</code> object.
318      * </ul>
319      *
320      * <p>If the specified method is {@link RemoteMethodControl#setConstraints
321      * RemoteMethodControl.setConstraints}, then if <code>proxy</code> is an
322      * instance of a dynamic proxy class containing this invocation
323      * handler, returns a new proxy containing a copy of this invocation
324      * handler with the same activation identifier, the new specified
325      * client constraints (<code>args[0]</code>), and the current
326      * underlying proxy, or if the current underlying proxy implements
327      * {@link RemoteMethodControl}, a copy of that proxy with the new
328      * specified client constraints.  An exception is thrown if
329      * <code>proxy</code> is not an instance of a dynamic proxy class
330      * containing this invocation handler.
331      * 
332      * <p>If the specified method is {@link RemoteMethodControl#getConstraints
333      * RemoteMethodControl.getConstraints}, returns the client constraints.
334      * 
335      * <p>If the specified method is
336      * {@link TrustEquivalence#checkTrustEquivalence
337      * TrustEquivalence.checkTrustEquivalence}, returns <code>true</code> if
338      * the argument (<code>args[0]</code>) is an instance of a dynamic proxy
339      * class (that is, a class generated by {@link Proxy}) that implements the
340      * same interfaces as the specified proxy and calling the
341      * {@link #checkTrustEquivalence checkTrustEquivalence} method of this
342      * invocation handler with the invocation handler of that argument
343      * returns <code>true</code>, and returns <code>false</code> otherwise.
344      *
345      * <p>For all other methods, a remote invocation is made as follows:
346      *
347      * <p>A single set of absolute constraints (if any) is used for the
348      * duration of the remote invocation, including any activation that may
349      * occur.
350      *
351      * </p><ul>
352      * <li>If the underlying proxy is non-<code>null</code>, the method is
353      * invoked as follows:
354      *
355      * <ul>
356      * <li>If the client constraints of this object are not
357      * <code>null</code> and the underlying proxy does not implement {@link
358      * RemoteMethodControl} then an {@link UnsupportedConstraintException}
359      * is thrown.
360      *
361      * <li>If <code>method</code>'s declaring class is not
362      * <code>public</code>, the underlying proxy is an instance of the
363      * <code>method</code>'s declaring class, and the underlying proxy's
364      * class is <code>public</code>, then a <code>public</code> method with
365      * the same name and parameter types is obtained from the underlying
366      * proxy's class, and if such a method exists, that method is
367      * reflectively invoked on the underlying proxy passing it the
368      * specified <code>args</code> and the result is returned; otherwise if
369      * such a method doesn't exist an <code>ActivateFailedException</code>
370      * is thrown with {@link NoSuchMethodException} as the cause.
371      *
372      * <li>Otherwise, the original <code>method</code> is reflectively
373      * invoked on the underlying proxy passing it the specified
374      * <code>args</code>.
375      * </ul>
376      *
377      * <p>If this reflective invocation throws an exception other than
378      * {@link IllegalAccessException}, {@link IllegalArgumentException},
379      * the <code>ActivateFailedException</code> described above, or
380      * an {@link InvocationTargetException} containing {@link
381      * ConnectException}, {@link ConnectIOException}, {@link
382      * NoSuchObjectException}, or {@link UnknownHostException}, then if the
383      * exception an {@link InvocationTargetException} the contained
384      * exception is thrown to the caller, otherwise the exception is thrown
385      * directly.
386      *
387      * </p><li>If the underlying proxy is <code>null</code> or if the
388      * reflective invocation does not throw an exception to the caller as
389      * described above:
390      * <ul>
391      *
392      * <li>If permitted by some implementation-specific mechanism,
393      * dynamically grants permissions to the class loader of the
394      * activation identifier's class by invoking {@link
395      * Security#grant(Class,Class) Security.grant} passing the class of the
396      * proxy and the class of the activation identifier.  If this
397      * invocation throws an {@link UnsupportedOperationException}, the
398      * exception is ignored.
399      *
400      * <li>A new proxy is obtained by invoking the {@link
401      * ActivationID#activate activate} method on the activation identifier,
402      * passing <code>false</code> as the argument.  That method must return
403      * an instance of a dynamic {@link Proxy} class, with an invocation
404      * handler that is an instance of this class, containing the same
405      * activation identifier.  If the returned proxy does not meet this
406      * criteria, then an {@link ActivateFailedException} is thrown.  If the
407      * <code>activate</code> call throws {@link ConnectException}, then
408      * a new <code>ConnectException</code> is thrown with the original
409      * <code>ConnectException</code> as the cause.  If the
410      * <code>activate</code> call throws {@link RemoteException}, then
411      * {@link ConnectIOException} is thrown with the
412      * <code>RemoteException</code> as the cause.  If the
413      * <code>activate</code> call throws {@link UnknownObjectException}, then
414      * {@link NoSuchObjectException} is thrown with the
415      * <code>UnknownObjectException</code> as the cause.  Finally, if the
416      * <code>activate</code> call throws {@link ActivationException},
417      * then {@link ActivateFailedException} is thrown with the
418      * <code>ActivationException</code> as the cause.
419      *
420      * <li>If a valid, new proxy is returned by the <code>activate</code>
421      * call, the underlying proxy of the new proxy is obtained from the new
422      * proxy's activatable invocation handler.  If the obtained underlying
423      * proxy implements <code>RemoteMethodControl</code>, this invocation
424      * handler's underlying proxy is set to a copy of the obtained
425      * underlying proxy with the client constraints of this instance.
426      * Otherwise, this invocation handler's underlying proxy is set to the
427      * obtained underlying proxy.
428      *
429      * <li>The reflective invocation is then retried (as above) on the new
430      * underlying proxy.  Activation and retry can occur up to three
431      * times. On subsequent attempts, <code>true</code> will be passed to
432      * the activation identifier's <code>activate</code> method, if passing
433      * <code>false</code> returned the same underlying proxy as before or
434      * if <code>NonExistantObjectException</code> was thrown by the call to the
435      * underlying proxy.
436      *
437      * <li>If the final attempt at reflective invocation throws
438      * <code>IllegalAccessException</code> or
439      * <code>IllegalArgumentException</code> an
440      * <code>ActivateFailedException</code> is thrown with the original
441      * exception as the cause.  If this reflective invocation throws
442      * <code>InvocationTargetException</code>, the contained target
443      * exception is thrown.
444      * </ul>
445      * </ul>
446      *
447      * <p>The implementation of remote method invocation defined by this class
448      * preserves at-most-once call semantics: the remote call either does not
449      * execute, partially executes, or executes exactly once at the remote
450      * site. Note that for remote calls to activatable objects, arguments may
451      * be marshalled more than once.
452      *
453      * <p>The semantics of this method are unspecified if the arguments could
454      * not have been produced by an instance of some valid dynamic proxy
455      * class containing this invocation handler.
456      * This method throws {@link IllegalArgumentException} if
457      * <code>proxy</code> is an instance of
458      * <code>InvocationHandler</code> or, if a remote invocation is to be
459      * made, any of the superinterfaces of <code>proxy</code>'s class
460      * have a method with the same name and parameter types as
461      * <code>method</code> but that does not declare
462      * <code>RemoteException</code> or a superclass of
463      * <code>RemoteException</code> in its <code>throws</code> clause
464      * (even if such a method is not a member of any of the direct
465      * superinterfaces of <code>proxy</code>'s class because of
466      * overriding).
467      *
468      * @throws	Throwable {@inheritDoc}
469      * @see	java.lang.reflect.UndeclaredThrowableException
470      **/
471     @Override
472     public Object invoke(Object proxy, Method method, Object[] args)
473 	throws Throwable
474     {
475 	Class decl = method.getDeclaringClass();
476 	if (proxy instanceof InvocationHandler) {
477 	    throw new IllegalArgumentException(
478 				    "proxy cannot be an invocation handler");
479 	} else if (decl == Object.class) {
480 	    return invokeObjectMethod(proxy, method, args);
481 	} else if (decl == RemoteMethodControl.class) {
482 	    return invokeRemoteMethodControlMethod(proxy, method, args);
483 	} else if (decl == TrustEquivalence.class) {
484 	    return invokeTrustEquivalenceMethod(proxy, method, args);
485 	} else {
486 	    return invokeRemoteMethod(proxy, method, args);
487 	}
488     }

490     /**
491      * Handles java.lang.Object methods.
492      **/
493     private Object invokeObjectMethod(Object proxy,
494 				      Method method,
495 				      Object[] args)
496     {
497 	String name = method.getName();
498 
499 	if (name.equals("hashCode")) {
500 	    return Integer.valueOf(hashCode());
501 
502 	} else if (name.equals("equals")) {
503 	    Object obj = args[0];
504 	    boolean b =
505 		proxy == obj ||
506 		(obj != null &&
507 		 Util.sameProxyClass(proxy, obj) &&
508 		 equals(Proxy.getInvocationHandler(obj)));
509 	    return Boolean.valueOf(b);
510 
511 	} else if (name.equals("toString")) {
512 	    return proxyToString(proxy);
513 
514 	} else {
515 	    throw new IllegalArgumentException(
516 		"unexpected Object method: " + method);
517 	}
518     }

520     /**
521      * Handles RemoteMethodControl methods.
522      **/
523     private Object invokeRemoteMethodControlMethod(Object proxy,
524 						   Method method,
525 						   Object[] args)
526 	throws Throwable
527     {
528 	String name = method.getName();
529 	
530 	if (name.equals("setConstraints")) {
531 	    if (Proxy.getInvocationHandler(proxy) != this) {
532 		throw new IllegalArgumentException("not proxy for this");
533 	    }
534 	    Remote newProxy;
535 	    synchronized (this) {
536 		newProxy = uproxy;
537 	    }
538 	    MethodConstraints mc = (MethodConstraints) args[0];
539 	    if (newProxy instanceof RemoteMethodControl) {
540 		newProxy = (Remote)
541 		    ((RemoteMethodControl) newProxy).setConstraints(mc);
542 	    }
543 	    Class proxyClass = proxy.getClass();
544 	    return Proxy.newProxyInstance(
545 		getProxyLoader(proxyClass),
546 		proxyClass.getInterfaces(),
547 		new ActivatableInvocationHandler(id, newProxy, mc));
548 	    // IllegalArgumentException means proxy argument was bogus
549 		    
550 	} else if (name.equals("getConstraints")) {
551 	    return clientConstraints;
552 	    
553 	} else {
554 	    throw new AssertionError(method);
555 	}
556     }

558     /**
559      * Handles TrustEquivalence methods.
560      **/
561     private Object invokeTrustEquivalenceMethod(Object proxy,
562 						Method method,
563 						Object[] args)
564     {
565 	String name = method.getName();
566 	if (name.equals("checkTrustEquivalence")) {
567 	    Object obj = args[0];
568 	    boolean b =
569 		proxy == obj ||
570 		(obj != null &&
571 		 Util.sameProxyClass(proxy, obj) &&
572 		 checkTrustEquivalence(Proxy.getInvocationHandler(obj)));
573 	    return Boolean.valueOf(b);
574 
575 	} else {
576 	    throw new AssertionError(method);
577 	}
578     }

580     /**
581      * Handles remote methods.
582      **/
583     private Object invokeRemoteMethod(Object proxy,
584 				      Method method,
585 				      Object[] args)
586 	throws Throwable
587     {    
588 	Util.checkProxyRemoteMethod(proxy.getClass(), method);
589 	/*
590 	 * Make the relative constraints (if any) on the method absolute.
591 	 * If there are relative constraints, create new method constraints
592 	 * to set on the underlying proxy before invoking a method.
593 	 */
594 	MethodConstraints convertedConstraints = null;
595 	if (clientConstraints != null) {
596 	    InvocationConstraints relativeConstraints =
597 		clientConstraints.getConstraints(method);
598 	    InvocationConstraints absoluteConstraints =
599 		relativeConstraints.makeAbsolute();
600 	    if (relativeConstraints != absoluteConstraints) {
601 		convertedConstraints =
602 		    new BasicMethodConstraints(absoluteConstraints);
603 	    }
604 	}
605 	
606 	boolean force = false;
607 	Remote currProxy;
608 	Failure failure = null;
609 	
610 	/*
611 	 * Attempt object activation if underlying proxy is unknown.
612 	 */
613 	synchronized (this) {
614 	    if (uproxy == null) {
615 		activate(force, proxy, method);
616 		force = true;
617 	    }
618 	    currProxy = uproxy;
619 	}
620 	 
621 	for (int retries = MAX_RETRIES; --retries >= 0; ) {
622 	    if (logger.isLoggable(Levels.HANDLED)) {
623 		if (failure != null) {
624 		    logThrow(Levels.HANDLED, "outbound call",
625 			     "invokeRemoteMethod",
626 			     method, failure.exception);
627 		}
628 	    }
629 	    
630 	    if ((clientConstraints != null) &&
631 		!(currProxy instanceof RemoteMethodControl))
632 	    {
633 		throw new UnsupportedConstraintException(
634  		    "underlying proxy does not implement RemoteMethodControl");
635 	    }
636 
637 	    /*
638 	     * Set constraints on target proxy only if relative constraints
639 	     * were made absolute.
640 	     */
641 	    Remote targetProxy =
642 		(convertedConstraints == null ?
643 		 currProxy :
644 		 (Remote) ((RemoteMethodControl) currProxy).
645 		 setConstraints(convertedConstraints));
646 
647 	    /*
648 	     * Invoke method on target proxy.
649 	     */
650 	    Object result = invokeMethod(targetProxy, method, args);
651 	    if (result instanceof Failure) {
652 		failure = (Failure) result;
653 		if (!failure.retry || retries <= 0) {
654 		    break;
655 		}
656 	    } else {
657 		return result;
658 	    }
659 
660 	    /*
661 	     * Reattempt activation.
662 	     */
663 	    synchronized (this) {
664 		if (uproxy == null || currProxy.equals(uproxy)) {
665 		    activate(force, proxy, method);
666 		    if (currProxy.equals(uproxy) &&
667 			(failure.exception instanceof NoSuchObjectException) &&
668 			!force)
669 		    {
670 			activate(true, proxy, method);
671 		    }
672 		    force = true;
673 		} else {
674 		    force = false;
675 		}
676 		currProxy = uproxy;
677 	    }
678 	}
679 	
680 	if (logger.isLoggable(Levels.FAILED)) {
681 	    logThrow(Levels.FAILED, "outbound call", "invokeRemoteMethod",
682 		     method, failure.exception);
683 	}
684 	throw failure.exception;
685     }

766     /**
767      * Reflectively invokes the method on the supplied <code>proxy</code>
768      * as follows:
769      * 
770      * <p>If the <code>method</code>'s declaring class is not
771      * <code>public</code>, the proxy is an instance of the
772      * <code>method</code>'s declaring class, and the proxy class is
773      * <code>public</code>, a <code>public</code> method with the same name and
774      * parameter types is obtained from the proxy class, and if such a
775      * method exists, that method is
776      * reflectively invoked on the proxy passing it the specified
777      * <code>args</code> and the result is returned, otherwise if such a
778      * method doesn't exist an <code>ActivateFailedException</code>
779      * is thrown with {@link NoSuchMethodException} as the cause.
780      *
781      * <p>Otherwise, the original <code>method</code> is reflectively
782      * invoked on the proxy passing it the specified
783      * <code>args</code> and the result is returned.
784      *
785      * <p>If the reflective invocation throws
786      * <code>IllegalAccessException</code> or
787      * <code>IllegalArgumentException</code>, an
788      * <code>ActivateFailedException</code> exception is thrown with the
789      * original exception as the cause. If the reflective invocation throws
790      * {@link InvocationTargetException} the contained target exception is
791      * thrown.
792      *
793      * @param 	proxy a proxy
794      * @param	m a method
795      * @param	args arguments
796      * @return	result of reflective invocation
797      * @throws	ActivateFailedException if the reflective invocation throws
798      *		<code>IllegalAccessException</code> or
799      *		<code>IllegalArgumentException</code>
800      * @throws	Throwable if the reflective invocation throws
801      *		<code>InvocationTargetException</code>, the contained
802      *		target exception is thrown
803      **/
804     private static Object invokeMethod0(Object proxy, Method m, Object[] args)
805 	throws Throwable
806     {
807 	Class mcl = m.getDeclaringClass();
808 	if (!Modifier.isPublic(mcl.getModifiers())) {
809 	    Class cl = proxy.getClass();
810 	    if (mcl.isAssignableFrom(cl) &&
811 		Modifier.isPublic(cl.getModifiers()))
812 	    {
813 		try {
814 		    m = cl.getMethod(m.getName(), m.getParameterTypes());
815 		} catch (NoSuchMethodException nsme) {
816 		    throw new ActivateFailedException("bad proxy").
817 			initCause(nsme);
818 		}
819 	    }
820 	}
821 
822 	try {
823 	    return m.invoke(proxy, args);
824 	} catch (IllegalAccessException e) {
825 	    throw new ActivateFailedException("bad proxy").initCause(e);
826 	} catch (IllegalArgumentException e) {
827 	    throw new ActivateFailedException("bad proxy").initCause(e);
828 	} catch (InvocationTargetException e) {
829 	    throw e.getTargetException();
830 	}
831     }

833     /**
834      * Returns a proxy trust iterator for an activatable object that is
835      * suitable for use by {@link
836      * net.jini.security.proxytrust.ProxyTrustVerifier}.
837      *
838      * <p>The iterator produces the current underlying proxy on each
839      * iteration.  The iterator produces up to three elements, but after
840      * the first element, iteration terminates unless the exception set by
841      * a call to {@link ProxyTrustIterator#setException setException} on
842      * the previous iteration is an instance of {@link ConnectException},
843      * {@link ConnectIOException}, {@link NoSuchObjectException}, or {@link
844      * UnknownHostException}.
845      *
846      * <p>On each iteration, if the current underlying proxy is
847      * <code>null</code> or the same as the underlying proxy produced by
848      * the previous iteration:
849      *
850      * <p>A new proxy is obtained by invoking the {@link
851      * ActivationID#activate activate} method on the activation identifier,
852      * passing <code>false</code> as the argument.  That method must return
853      * an instance of a dynamic {@link Proxy} class, with an invocation
854      * handler that is an instance of this class, containing the same
855      * activation identifier.  If this activation
856      * throws one of the following exceptions, the exception is thrown
857      * by the <code>next</code> method of the iterator and the iteration
858      * terminates:
859      *
860      * <blockquote>If the proxy returned by the <code>activate</code> call
861      * does not meet the criteria listed above, then an {@link
862      * ActivateFailedException} is thrown.  If the <code>activate</code>
863      * call throws {@link RemoteException}, then {@link ConnectIOException}
864      * is thrown with the <code>RemoteException</code> as the cause.  If
865      * the <code>activate</code> call throws {@link UnknownHostException},
866      * then {@link NoSuchObjectException} is thrown with the
867      * <code>UnknownHostException</code> as the cause.  Finally, if the
868      * <code>activate</code> call throws {@link ActivationException}, then
869      * {@link ActivateFailedException} is thrown with the
870      * <code>ActivationException</code> as the cause.
871      * </blockquote>
872      *
873      * <p>If a valid, new proxy is returned by the <code>activate</code>
874      * call, the underlying proxy of the new proxy is obtained from the new
875      * proxy's activatable invocation handler.  If the obtained underlying
876      * proxy implements <code>RemoteMethodControl</code>, this invocation
877      * handler's underlying proxy is set to a copy of the obtained
878      * underlying proxy with the client constraints of this instance.
879      * Otherwise, this invocation handler's underlying proxy is set to the
880      * obtained underlying proxy.
881      *
882      * <p>On the first call to the activation identifier's
883      * <code>activate</code> method, <code>false</code> is passed as an
884      * argument; on subsequent calls <code>true</code> will be passed, if
885      * passing <code>false</code> returned the same underlying proxy as
886      * before (when compared using the <code>equals</code> method) or if
887      * the exception passed to <code>setException</code> is an instance of
888      * <code>NonExistantObjectException</code>. If an activation attempt results
889      * in an exception, that exception is thrown by the <code>next</code>
890      * method of the iterator and iteration terminates.
891      *
892      * @return a proxy trust iterator suitable for use by
893      * <code>ProxyTrustVerifier</code>
894      **/
895     protected ProxyTrustIterator getProxyTrustIterator() {
896 	return new ProxyTrustIterator() {
897 	    private int retries = MAX_RETRIES + 1;
898 	    private boolean force = false;
899 	    private Remote currProxy = null;
900 	    private Exception fail = null;
901 	    private RemoteException ex = null;
902 	    private boolean advance = true;
903 
904             @Override
905 	    public synchronized boolean hasNext() {
906 		if (advance) {
907 		    advance = false;
908 		    if (--retries < 0) {
909 		    } else if (retries == MAX_RETRIES ||
910 			       ex instanceof NoSuchObjectException ||
911 			       ex instanceof ConnectException ||
912 			       ex instanceof UnknownHostException ||
913 			       ex instanceof ConnectIOException)
914 		    {
915 			try {
916 			    synchronized
917 				(ActivatableInvocationHandler.this)
918 			    {
919 				if (uproxy == null || uproxy.equals(currProxy))
920 				{
921 				    activate(force, null, getPtiMethod);
922 				    if (uproxy.equals(currProxy) &&
923 					ex instanceof NoSuchObjectException &&
924 					!force)
925 				    {
926 					activate(true, null, getPtiMethod);
927 				    }
928 				    force = true;
929 				} else {
930 				    force = false;
931 				}
932 				currProxy = uproxy;
933 			    }
934 			} catch (Exception e) {
935 			    fail = e;
936 			    retries = 0;
937 			}
938 		    } else {
939 			retries = -1;
940 			if (!(ex == null ||
941 			      ex instanceof MarshalException ||
942 			      ex instanceof ServerException ||
943 			      ex instanceof ServerError))
944 			{
945 			    synchronized (ActivatableInvocationHandler.this) {
946 				if (currProxy.equals(uproxy)) {
947 				    uproxy = null;
948 				}
949 			    }
950 			}
951 		    }
952 		}
953 		return retries >= 0;
954 	    }
955 
956             @Override
957 	    public synchronized Object next() throws RemoteException {
958 		if (!hasNext()) {
959 		    throw new NoSuchElementException();
960 		}
961 		advance = true;
962 		if (fail == null) {
963 		    return currProxy;
964 		} else if (fail instanceof RemoteException) {
965 		    throw (RemoteException) fail;
966 		} else {
967 		    throw (RuntimeException) fail;
968 		}
969 	    }
970 
971             @Override
972 	    public synchronized void setException(RemoteException e) {
973 		if (e == null) {
974 		    throw new NullPointerException("exception is null");
975 		} else if (retries > MAX_RETRIES || !advance || fail != null) {
976 		    throw new IllegalStateException();
977 		}
978 		ex = e;
979 	    }
980 	};
981     }

984      * Returns <code>true</code> if the specified object (which is not
985      * yet known to be trusted) is equivalent in trust, content, and
986      * function to this known trusted object, and <code>false</code>
987      * otherwise.
988      *
989      * <p><code>ActivatableInvocationHandler</code> implements this
990      * method as follows:
991      *
992      * <p>This method returns <code>true</code> if and only if the
993      * following conditions are met:
994      * <ul>
995      * <li> The specified object has the same class as this object.
996      * <li> This object's activation identifier implements {@link
997      * TrustEquivalence}.
998      * <li> Invoking the <code>checkTrustEquivalence</code> method on
999      * this object's activation identifier passing the specified object's
1000      * activation identifier returns <code>true</code>.
1001      * <li> The client constraints in the specified object are equal to the
1002      * ones in this object.
1003      * </ul>
1004      *
1005      * <p>The underlying proxy of the specified object is set to
1006      * <code>null</code> if this method returns <code>true</code> and any
1007      * of the following conditions are met:
1008      * <ul>
1009      * <li> This object's underlying proxy is <code>null</code>.
1010      * <li> This object's underlying proxy is not an instance of {@link
1011      * TrustEquivalence}.
1012      * <li> Invoking the <code>checkTrustEquivalence</code> method on this
1013      * object's underlying proxy, passing the underlying proxy of the
1014      * specified object, returns <code>false</code>.
1015      * </ul>
1016      **/
1017     @Override
1018     public boolean checkTrustEquivalence(Object obj) {
1019 	if (this == obj) {
1020 	    return true;
1021 	} else if (!(obj instanceof ActivatableInvocationHandler) ||
1022 		   !(id instanceof TrustEquivalence))
1023 	{
1024 	    return false;
1025 	}
1026 	ActivatableInvocationHandler other =
1027 	    (ActivatableInvocationHandler) obj;
1028 	if (!((TrustEquivalence) id).checkTrustEquivalence(other.id) ||
1029 	    (clientConstraints != other.clientConstraints &&
1030 	     (clientConstraints == null ||
1031 	      !clientConstraints.equals(other.clientConstraints))))
1032 	{
1033 	    return false;
1034 	}
1035 	Remote currProxy;
1036 	synchronized (this) {
1037 	    currProxy = uproxy;
1038 	}
1039 	Remote otherProxy;
1040 	synchronized (other) {
1041 	    otherProxy = other.uproxy;
1042 	}
1043 	if (currProxy == null || !(currProxy instanceof TrustEquivalence) ||
1044 	    !((TrustEquivalence) currProxy).checkTrustEquivalence(otherProxy))
1045 	{
1046 	    synchronized (other) {
1047 		other.uproxy = null;
1048 	    }
1049 	}
1050 	return true;
1051     }

1076      * of uproxy is updated to the value returned by calling the id's
1077      * activate method.  Note: The caller should be synchronized on "this"
1078      * while calling this method.

1080      * @param force boolean to pass to activation id's activate method
1081      * @param proxy outer proxy from which dynamic grants are inherited,
1082      *	      or null
1083      **/
1084     private void activate0(boolean force, Object proxy)
1085 	throws RemoteException
1086     {
1087 	assert Thread.holdsLock(this);
1088 	
1089 	uproxy = null;
1090 	try {
1091 	    if (proxy != null && enableGrant) {
1092 		try {
1093 		    Security.grant(proxy.getClass(), id.getClass());
1094 		} catch (UnsupportedOperationException uoe) {
1095 		}
1096 	    }
1097 	    Object newProxy = id.activate(force);
1098 	    if (!Proxy.isProxyClass(newProxy.getClass())) {
1099 		throw new ActivateFailedException("invalid proxy");
1100 	    }
1101 	    
1102 	    InvocationHandler obj = Proxy.getInvocationHandler(newProxy);
1103 	    if (!(obj instanceof ActivatableInvocationHandler)) {
1104 		throw new ActivateFailedException("invalid proxy handler");
1105 	    }
1106 	    ActivatableInvocationHandler handler =
1107 		(ActivatableInvocationHandler) obj;
1108 	    if (!id.equals(handler.id)) {
1109 		throw new ActivateFailedException("unexpected activation id");
1110 	    }
1111 	    
1112 	    Remote newUproxy;
1113             synchronized (handler){
1114                 newUproxy = handler.uproxy;
1115             }
1116 	    if (newUproxy == null) {
1117 		throw new ActivateFailedException("null underlying proxy");
1118 	    } else if (newUproxy instanceof RemoteMethodControl) {
1119 		newUproxy = (Remote) ((RemoteMethodControl) newUproxy).
1120 		    		setConstraints(clientConstraints);
1121 	    }
1122 	    uproxy = newUproxy;
1123 	    
1124 	} catch (ConnectException e) {
1125 	    throw new ConnectException("activation failed", e);
1126 	} catch (RemoteException e) {
1127 	    throw new ConnectIOException("activation failed", e);
1128 	} catch (UnknownObjectException e) {
1129 	    throw new net.jini.export.NonExistantObjectException("object not registered", e);
1130 	} catch (ActivationException e) {
