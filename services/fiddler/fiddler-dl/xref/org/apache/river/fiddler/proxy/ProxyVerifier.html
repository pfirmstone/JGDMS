18  package org.apache.river.fiddler.proxy;
19  
20  import java.io.IOException;
21  import java.io.Serializable;
22  import java.rmi.RemoteException;
23  import net.jini.core.constraint.MethodConstraints;
24  import net.jini.core.constraint.RemoteMethodControl;
25  import net.jini.id.ReferentUuid;
26  import net.jini.id.Uuid;
27  import net.jini.security.TrustVerifier;
28  import net.jini.security.proxytrust.TrustEquivalence;
29  import org.apache.river.api.io.AtomicSerial;
30  import org.apache.river.api.io.AtomicSerial.GetArg;
31  
32  /** This class defines a trust verifier for the proxies related to the 
33   *  Fiddler implementation of the lookup discovery service.
34   *
35   * @see net.jini.security.TrustVerifier
36   *
37   * @author Sun Microsystems, Inc.
38   *
39   * @since 2.0
40   */
41  @AtomicSerial
42  public final class ProxyVerifier implements Serializable, TrustVerifier {
43  
44      private static final long serialVersionUID = 2L;
45  
46      /** The canonical instance of the inner proxy to the service. This
47       *  instance will be used by the <code>isTrusted</code> method 
48       *  as the known trusted object used to determine whether or not a
49       *  given proxy is equivalent in trust, content, and function.
50       *
51       *  @serial
52       */
53      private final RemoteMethodControl innerProxy;
54      /** 
55       * The unique identifier associated with the backend server referenced
56       * by the <code>innerProxy</code>, used for comparison with the IDs
57       * extracted from the smart proxies being verified.
58       *
59       * @serial
60       */
61      private final Uuid proxyID;
62  
63      /** Constructs an instance of <code>TrustVerifier</code> that can be
64       *  used to determine whether or not a given proxy is equivalent in
65       *  trust, content, and function to the service's <code>innerProxy</code>
66       *  referenced in the class constructed here.
67       *
68       * @param innerProxy canonical instance of the inner proxy to the service.
69       *                   The <code>isTrustedObject</code> method will
70       *                   determine whether or not proxies input to that
71       *                   method are equivalent in trust, content, and function
72       *                   to this object.
73       * @param proxyID    instance of <code>Uuid</code> containing the unique
74       *                   identifier associated with the backend server
75       *                   referenced by the <code>innerProxy</code> paramater.
76       * 
77       * @throws UnsupportedOperationException if <code>innerProxy</code> does
78       *         not implement both {@link RemoteMethodControl} and {@link
79       *         TrustEquivalence}
80       */
81      public ProxyVerifier(Fiddler innerProxy, Uuid proxyID) {
82          this(innerProxy, proxyID, check(innerProxy, proxyID));
83          
84      }//end constructor
85      
86      /**
87       * {@link AtomicSerial} constructor.
88       * @param arg
89       * @throws IOException 
90       */
91      ProxyVerifier(GetArg arg) throws IOException {
92  	this((Fiddler) arg.get("innerProxy", null), (Uuid) arg.get("proxyID", null));
93      }
94      
95      private ProxyVerifier(Fiddler innerProxy, Uuid proxyID, boolean check){
96  	this.innerProxy = (RemoteMethodControl)innerProxy;
97          this.proxyID = proxyID;
98      }
99      
100     private static boolean check(Fiddler innerProxy, Uuid proxyID){
101         if( !(innerProxy instanceof RemoteMethodControl) ) {
102             throw new UnsupportedOperationException
103                          ("cannot construct verifier - canonical inner "
104                           +"proxy is not an instance of RemoteMethodControl");
105         } else if( !(innerProxy instanceof TrustEquivalence) ) {
106             throw new UnsupportedOperationException
107                              ("cannot construct verifier - canonical inner "
108                               +"proxy is not an instance of TrustEquivalence");
109         }//endif
110 	return true;
111     }
112 
113     /** Returns <code>true</code> if the specified proxy object (that is
114      *  not yet known to be trusted) is equivalent in trust, content, and
115      *  function to the canonical inner proxy object referenced in this
116      *  class; otherwise returns <code>false</code>.
117      *
118      * @param obj proxy object that will be compared to this class' stored
119      *            canonical proxy to determine whether or not the given
120      *            proxy object is equivalent in trust, content, and function.
121      *            
122      * @return <code>true</code> if the specified object (that is not yet
123      *                           known to be trusted) is equivalent in trust,
124      *                           content, and function to the canonical inner
125      *                           proxy object referenced in this class;
126      *                           otherwise returns <code>false</code>.
127      *
128      * @throws NullPointerException if any argument is <code>null</code>
129      */
130     public boolean isTrustedObject(Object obj,
131                                    TrustVerifier.Context ctx)
132                                                        throws RemoteException
133     {
134         /* Validate the arguments */
135         if (obj == null || ctx == null) {
136             throw new NullPointerException("arguments must not be null");
137 	}//endif
138         /* Prepare the input proxy object for trust verification. The types
139          * of proxies, specific to the service, that this method will
140          * handle are:
141          *  - ConstrainableFiddlerProxy
142          *  - ConstrainableFiddlerRegistration
143          *  - ConstrainableFiddlerLease
144          *  - ConstrainableFiddlerAdminProxy
145          */
146         RemoteMethodControl inputProxy;
147         Uuid inputProxyID;
148         if( obj instanceof FiddlerProxy.ConstrainableFiddlerProxy ) {
149             inputProxy = (RemoteMethodControl)((FiddlerProxy)obj).server;
150             inputProxyID = ((ReferentUuid)obj).getReferentUuid();
151         } else if
152           (obj instanceof FiddlerRegistration.ConstrainableFiddlerRegistration)
153         {
154             FiddlerRegistration reg = (FiddlerRegistration)obj;
155             if( !this.isTrustedObject( (reg.eventReg).getSource(), ctx) ) {
156                 return false;
157             }//endif
158             if( !this.isTrustedObject( (reg.eventReg).getLease(), ctx) ) {
159                 return false;
160             }//endif
161             inputProxy = (RemoteMethodControl)reg.server;
162             /* FiddlerRegistration doesn't carry a proxyID. Default to the
163              * cannonical proxyID to avoid complicated handler logic below.
164              */
165             inputProxyID = proxyID;
166         } else if( obj instanceof FiddlerLease.ConstrainableFiddlerLease ) {
167             inputProxy = (RemoteMethodControl)((FiddlerLease)obj).server;
168             inputProxyID = ((FiddlerLease)obj).getServerID();
169         } else if
170              (obj instanceof FiddlerAdminProxy.ConstrainableFiddlerAdminProxy)
171         {
172             inputProxy = (RemoteMethodControl)((FiddlerAdminProxy)obj).server;
173             inputProxyID = ((ReferentUuid)obj).getReferentUuid();
174         } else if( obj instanceof RemoteMethodControl ) {
175             /* This block handles the case where the inner proxy itself 
176              * (rather than an outer proxy that contains the inner proxy)
177              * is being verified for trust. Unlike most of the outer proxies,
178              * the inner proxy doesn't provide a means for obtaining the UUID
179              * of the associated backend server. Thus, to avoid complicated
180              * handler logic, for this case the cannonical proxyID is used
181              * in the trust equivalence comparison that is performed below.
182              */
183             inputProxy = (RemoteMethodControl)obj;
184             inputProxyID = proxyID;
185         } else {
186             return false;
187         }//endif
188         /* Get the client constraints currently set on the input proxy */
189         final MethodConstraints mConstraints 
190                                         = inputProxy.getConstraints();
191         /* Create a copy of the canonical proxy with its method constraints
192          * replaced with the method constraints of the input proxy.
193          */
194         final TrustEquivalence constrainedInnerProxy =
195              (TrustEquivalence)innerProxy.setConstraints(mConstraints);
196         /* With respect to trust, content, and function, test whether the
197          * input proxy is equivalent to the canonical inner proxy that has
198          * the same method constraints as that input proxy, and verify that
199          * the canonical ID of the backend server is equivalent to the ID
200          * extracted from the input proxy; return the result.
201          */
202         return (    constrainedInnerProxy.checkTrustEquivalence(inputProxy)
203                  && proxyID.equals(inputProxyID) );
204     }//end isTrustedObject
205 
206 }//end class ProxyVerifier
