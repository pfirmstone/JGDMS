59  /**
60   * Initialization common to all modes in which instances of this service
61   *  runs: activatable/persistent, non-activatable/persistent, and
62   *  transient (non-activatable /non-persistent).
63   *
64   */
65  class FiddlerInit {
66      LookupDiscoveryManager discoveryMgr;
67      ProxyPreparer listenerPreparer;
68      ProxyPreparer locatorToJoinPreparer;
69      ProxyPreparer locatorToDiscoverPreparer;
70      ProxyPreparer recoveredListenerPreparer;
71      ProxyPreparer recoveredLocatorToJoinPreparer;
72      ProxyPreparer recoveredLocatorToDiscoverPreparer;
73      String persistDir;
74      ReliableLog log = null;
75      DiscoveryManagement joinMgrLDM;
76      long leaseMax;
77      ExecutorService executorService;
78      ActivationSystem activationSystem;
79      Exporter serverExporter;
80      LocalLogHandler logHandler;
81      ActivationID activationID = null;
82      // These three fields are used by the Starter.start() implementation.
83      boolean persistent;
84      Configuration config;
85      AccessControlContext context;
86      LoginContext loginContext;
87      String codebase;
88      String certFactoryType;
89      String certPathEncoding;
90      byte[] encodedCerts;
91      
92      FiddlerInit(Configuration config,
93                  boolean persistent, 
94                  ActivationID activID, 
95                  LoginContext loginContext) throws IOException,
96                                                          ConfigurationException,
97                                                          ActivationException
98      {
99          try {
100             this.config = config;
101             this.persistent = persistent;
102             this.loginContext = loginContext;
103             context = AccessController.getContext();
104            /* Note that two discovery managers are retrieved/created in this
105              * method. One manager is used internally by this service to provide
106              * the lookup discovery capabilities offered to the clients that
107              * register to use this service. That manager (discoveryMgr) is
108              * considered part of this service's implementation, and thus is
109              * created rather than retrieved from the configuration. When
110              * created here, this manager is configured to initially discover
111              * NO_GROUPS and NO LOCATORS. The sets of groups and locators to
112              * discover on behalf of the registered clients will be populated
113              * during recovery; and then the discovery mechanism provided by
114              * this manager will be started later, after recovery and
115              * configuration retrieval has occurred, and after this service
116              * has been exported. Thus, it is important that this manager be
117              * created prior to recovery; as is done below.
118              * 
119              * The second discovery manager (joinMgrLDM) is passed to the join
120              * manager that is employed by this service to advertise itself to
121              * clients through lookup services. This discovery manager is
122              * retrieved from the configuration, and must satisfy the following
123              * requirements: must be an instance of both DiscoveryGroupManagement
124              * and DiscoveryLocatorManagement, and should be initially configured
125              * to discover NO_GROUPS and NO LOCATORS. This discovery manager is
126              * retrieved from the configuration after recovery of any persistent
127              * state, and after retrieval of any initial configuration items.
128              */
129             discoveryMgr = new LookupDiscoveryManager
130                                               (DiscoveryGroupManagement.NO_GROUPS,
131                                                new LookupLocator[0],
132                                                null,
133                                                config);
134 
135             /* Get the proxy preparers for the remote event listeners */
136             listenerPreparer = (ProxyPreparer)Config.getNonNullEntry
137                                                        (config,
138                                                         FiddlerImpl.COMPONENT_NAME,
139                                                         "listenerPreparer",
140                                                         ProxyPreparer.class,
141                                                         new BasicProxyPreparer());
142             /* Get the proxy preparers for the lookup locators to join */
143             locatorToJoinPreparer = (ProxyPreparer)Config.getNonNullEntry
144                                                        (config,
145                                                         FiddlerImpl.COMPONENT_NAME,
146                                                         "locatorToJoinPreparer",
147                                                         ProxyPreparer.class,
148                                                         new BasicProxyPreparer());
149             /* Get the proxy preparers for the lookup locators to discover */
150             locatorToDiscoverPreparer = (ProxyPreparer)Config.getNonNullEntry
151                                                      (config,
152                                                       FiddlerImpl.COMPONENT_NAME,
153                                                       "locatorToDiscoverPreparer",
154                                                       ProxyPreparer.class,
155                                                       new BasicProxyPreparer());
156             if(persistent) {
157                 /* Retrieve the proxy preparers that will only be applied during
158                  * the state recovery process below, when any previously stored
159                  * listeners or locators are recovered from persistent storage.
160                  */
161                 recoveredListenerPreparer = 
162                  (ProxyPreparer)Config.getNonNullEntry(config,
163                                                        FiddlerImpl.COMPONENT_NAME,
164                                                        "recoveredListenerPreparer",
165                                                        ProxyPreparer.class,
166                                                        new BasicProxyPreparer());
167                 recoveredLocatorToJoinPreparer =
168                  (ProxyPreparer)Config.getNonNullEntry
169                                                  (config,
170                                                   FiddlerImpl.COMPONENT_NAME,
171                                                   "recoveredLocatorToJoinPreparer",
172                                                   ProxyPreparer.class,
173                                                   new BasicProxyPreparer());
174                 recoveredLocatorToDiscoverPreparer = 
175                  (ProxyPreparer)Config.getNonNullEntry
176                                              (config,
177                                               FiddlerImpl.COMPONENT_NAME,
178                                               "recoveredLocatorToDiscoverPreparer",
179                                               ProxyPreparer.class,
180                                               new BasicProxyPreparer());
181 
182 
183                 /* Get the log directory for persisting this service's state */
184                 persistDir = (String)Config.getNonNullEntry(config,
185                                                             FiddlerImpl.COMPONENT_NAME,
186                                                             "persistenceDirectory",
187                                                             String.class);
188                 /* Recover the state that was persisted on prior runs (if any) */
189                 logHandler = new LocalLogHandler();
190                 log = new ReliableLog(persistDir, logHandler);
191             }//endif(persistent)
192 
193             /* Get the various configurable constants */
194             leaseMax = Config.getLongEntry(config,
195                                            FiddlerImpl.COMPONENT_NAME,
196                                            "leaseMax",
197                                            FiddlerImpl.MAX_LEASE, 0, Long.MAX_VALUE);
198 
199             /* Get a general-purpose task manager for this service 
200              * LinkedBlockingQueue is unbounded for that reason*/
201             executorService = Config.getNonNullEntry(config,
202                                              FiddlerImpl.COMPONENT_NAME,
203                                              "executorService",
204                                              ExecutorService.class,
205                                              new ThreadPoolExecutor(10,10,15,TimeUnit.SECONDS, new LinkedBlockingQueue<Runnable>(), new NamedThreadFactory("Fiddler Executor", false)) );
206             /* Get the discovery manager to pass to this service's join manager. */
207             try {
208                 joinMgrLDM = Config.getNonNullEntry(config,
209                                                     FiddlerImpl.COMPONENT_NAME,
210                                                     "discoveryManager",
211                                                     DiscoveryManagement.class);
212                 if( joinMgrLDM instanceof DiscoveryGroupManagement ) {
213                     String[] groups0 =
214                                ((DiscoveryGroupManagement)joinMgrLDM).getGroups();
215                     if(    (groups0 == DiscoveryGroupManagement.ALL_GROUPS)
216                         || (groups0.length != 0) )
217                     {
218                         throw new ConfigurationException
219                                      ("discoveryManager entry must be configured "
220                                       +"to initially discover/join NO_GROUPS");
221                     }//endif
222                 } else {// !(joinMgrLDM instanceof DiscoveryGroupManagement)
223                     throw new ConfigurationException
224                                            ("discoveryManager entry must "
225                                             +"implement DiscoveryGroupManagement");
226                 }//endif
227                 if( joinMgrLDM instanceof DiscoveryLocatorManagement ) {
228                     LookupLocator[] locs0 =
229                             ((DiscoveryLocatorManagement)joinMgrLDM).getLocators();
230                     if( (locs0 != null) && (locs0.length != 0) ) {
231                         throw new ConfigurationException
232                                      ("discoveryManager entry must be configured "
233                                       +"to initially discover/join no locators");
234                     }//endif
235                 } else {// !(joinMgrLDM instanceof DiscoveryLocatorManagement)
236                     throw new ConfigurationException
237                                          ("discoveryManager entry must "
238                                           +"implement DiscoveryLocatorManagement");
239                 }//endif
240             } catch (NoSuchEntryException e) {
241                 joinMgrLDM
242                    = new LookupDiscoveryManager(DiscoveryGroupManagement.NO_GROUPS,
243                                                 new LookupLocator[0], null,config);
244             }
245 
246             /* Handle items and duties related to exporting this service. */
247             ServerEndpoint endpoint = TcpServerEndpoint.getInstance(0);
248             InvocationLayerFactory ilFactory = 
249 		    new AtomicILFactory(
250 			null,
251 			null,
252 			Fiddler.class.getClassLoader()
253 		    );
254             Exporter defaultExporter = new BasicJeriExporter(endpoint,
255                                                              ilFactory,
256                                                              false,
257                                                              true);
258             /* For the activatable server */
259             if(activID != null) {
260                 ProxyPreparer aidPreparer =
261                   (ProxyPreparer)Config.getNonNullEntry(config,
262                                                         FiddlerImpl.COMPONENT_NAME,
263                                                         "activationIdPreparer",
264                                                         ProxyPreparer.class,
265                                                         new BasicProxyPreparer());
266                 ProxyPreparer aSysPreparer = 
267                   (ProxyPreparer)Config.getNonNullEntry(config,
268                                                         FiddlerImpl.COMPONENT_NAME,
269                                                         "activationSystemPreparer",
270                                                         ProxyPreparer.class,
271                                                         new BasicProxyPreparer());
272                 activationID = (ActivationID)aidPreparer.prepareProxy
273                                                                    (activID);
274                 activationSystem = (ActivationSystem)aSysPreparer.prepareProxy
275 			    (net.jini.activation.ActivationGroup.getSystem());
276                 defaultExporter = new ActivationExporter(activationID,
277                                                          defaultExporter);
278             }//endif(activationID != null)
279 
280             /* Get the exporter that will be used to export this service */
281             try {
282                 serverExporter = (Exporter)Config.getNonNullEntry(config,
283                                                                   FiddlerImpl.COMPONENT_NAME,
284                                                                   "serverExporter",
285                                                                   Exporter.class,
286                                                                   defaultExporter,
287                                                                   activationID);
288             } catch(ConfigurationException e) {// exception, use default
289                 throw new ExportException("Configuration exception while "
290                                           +"retrieving service's exporter",
291                                           e);
292             }
293 	    
294 	    codebase = Config.getNonNullEntry(config, FiddlerImpl.COMPONENT_NAME,
295 		    "Codebase_Annotation", String.class, "");
296 	    certFactoryType = Config.getNonNullEntry(config, FiddlerImpl.COMPONENT_NAME,
297 		    "Codebase_CertFactoryType", String.class, "X.509");
298 	    certPathEncoding = Config.getNonNullEntry(config, FiddlerImpl.COMPONENT_NAME,
299 		    "Codebase_CertPathEncoding", String.class, "PkiPath");
300 	    encodedCerts = Config.getNonNullEntry(config, FiddlerImpl.COMPONENT_NAME,
301 		    "Codebase_Certs", byte[].class, new byte[0]);
302 	
303         } catch(Throwable e) {
304             cleanupInitFailure();
305             handleActivatableInitThrowable(e);
306         }
307         
308     }
309     
310     /* Called in the constructor when failure occurs during the initialization
311      * process. Un-does any work that may have already been completed; for
312      * example, un-exports the service if it has already been exported,
313      * terminates any threads that may have been started, etc.
314      */
315     private void cleanupInitFailure() {
316         if(executorService != null)  {
317             try {
318                 executorService.shutdown();
319             } catch(Throwable t) { }
320         }//endif
321         
322         if(joinMgrLDM != null)  {
323             try {
324                 joinMgrLDM.terminate();
325             } catch(Throwable t) { }
326         }//endif
327 
328         if(discoveryMgr != null)  {
329             try {
330                 discoveryMgr.terminate();
331             } catch(Throwable t) { }
332         }//endif
333         // No threads started, no need to interrupt.
334     }//end cleanupInitFailure
335     
336     /* Convenience method called in the constructor or the activatable version
337      * of this service when failure occurs during the initialization process.
338      * Logs and rethrows the given <code>Throwable</code> so the constructor
339      * doesn't have to.
340      */
341     private static void handleActivatableInitThrowable(Throwable t) 
342                                             throws IOException,
343                                                    ActivationException,
344                                                    ConfigurationException
345     {
346         handleInitThrowable(t);
347         if (t instanceof ActivationException) {
348             throw (ActivationException)t;
349         } else {
350             throw new AssertionError(t);
351         }//endif
352     }//end handleInitThrowable
353     
354     /* Convenience method called in the constructor or the non-activatable 
355      * version of this service when failure occurs during the initialization
356      * process. Logs and rethrows the given <code>Throwable</code> so the
357      * constructor doesn't have to.
358      */
359     private static void handleInitThrowable(Throwable t) 
360                                             throws IOException,
361                                                    ConfigurationException
362     {
363         FiddlerImpl.problemLogger.log(Level.SEVERE, "cannot initialize the service", t);
364         if (t instanceof IOException) {
365             throw (IOException)t;
366         } else if (t instanceof ConfigurationException) {
367             throw (ConfigurationException)t;
368         } else if (t instanceof RuntimeException) {
369             throw (RuntimeException)t;
370         } else if (t instanceof Error) {
371             throw (Error)t;
372         }//endif
373     }//end handleInitThrowable
374 }
