18  package org.apache.river.fiddler;
19  
20  import net.jini.config.ConfigurationException;
21  
22  import javax.security.auth.login.LoginException;
23  
24  import java.rmi.activation.ActivationException;
25  import java.rmi.activation.ActivationID;
26  import java.rmi.MarshalledObject;
27  import java.io.IOException;
28  import net.jini.export.DynamicProxyCodebaseAccessor;
29  
30  /**
31   * Convenience class intended for use with the 
32   * {@link org.apache.river.start.ServiceStarter} framework to start
33   * an implementation of Fiddler that is activatable, and which will
34   * log its state information to persistent storage.
35   *
36   * @author Sun Microsystems, Inc.
37   * @since 2.0
38   */
39  public class ActivatableFiddlerImpl extends FiddlerImpl implements DynamicProxyCodebaseAccessor {
40  
41      /**
42       * Constructs a new instance of <code>FiddlerImpl</code> that is 
43       * activatable, and which will persist its state.
44       * <p>
45       * A constructor having this signature is required for the class to
46       * be activatable. This constructor is automatically called by the 
47       * activation group when the service is activated.
48       *
49       * @param activationID the activation ID generated by the activation
50       *                     system and assigned to the instance of the server
51       *                     being activated
52       * @param data         state data (represented as a 
53       *                     <code>MarshalledObject</code>) which is needed to
54       *                     re-activate this server
55       *
56       * @throws IOException            this exception can occur when there is
57       *                                a problem recovering data from disk,
58       *                                exporting the server that's being
59       *                                activated, or when unmarshalling the
60       *                                given <code>data</code> parameter.
61       * @throws ConfigurationException this exception can occur when a
62       *                                problem occurs while retrieving an item
63       *                                from the <code>Configuration</code>
64       *                                generated from the contents of the
65       *                                given <code>data</code> parameter
66       * @throws ActivationException    this exception can occur when a problem
67       *                                occurs while activating the service
68       * @throws LoginException         this exception occurs when authentication
69       *                                fails while performing a JAAS login for
70       *                                this service
71       * @throws ClassNotFoundException this exception can occur while 
72       *                                unmarshalling the given <code>data</code>
73       *                                parameter; when a class needed in the
74       *                                unmarshalling process cannot be found.
75       * @throws ClassCastException     this exception can occur while
76       *                                unmarshalling the given <code>data</code>
77       *                                parameter; when the contents of that
78       *                                parameter is not a <code>String</code>
79       *                                array.
80       */
81      ActivatableFiddlerImpl(ActivationID activationID,
82                             MarshalledObject data)
83                                                 throws IOException,
84                                                        ActivationException,
85                                                        ConfigurationException,
86                                                        LoginException,
87                                                        ClassNotFoundException
88      {
89          super(activationID, data);
90      }//end constructor
91  
92  }//end class ActivatableFiddlerImpl
