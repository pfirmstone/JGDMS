142     /**
143      * Start the manager. Start looking for lookup and registering
144      * with them.
145      * @param config object to use to obtain
146      *               <code>DiscoveryManagement</code> object, and if
147      *               this is the initial incarnation of this service, 
148      *               the object used to get the initial set of groups,
149      *               locators, and deployer defined attributes.
150      * @param service The proxy object to register with lookups.
151      * @param baseAttributes Any attributes the implementation wants
152      *                       attached, only used if this is the
153      *                       initial incarnation.
154      * @throws IOException if the is problem persisting the 
155      *         initial state or in starting discovery.
156      * @throws ConfigurationException if the configuration
157      *         is invalid.
158      */
159     void startManager(Configuration config, Object service,
160 		      ServiceID serviceID, Entry[] baseAttributes) 
161 	throws IOException, ConfigurationException
162     {
163 	if (operationsLogger.isLoggable(Level.FINER)) {
164             operationsLogger.entering(JoinStateManager.class.getName(), 
165 	        "startManager",
166 	        new Object[] { config, service, serviceID, 
167 		    Arrays.asList(baseAttributes)});
168 	}
169 	if (serviceID == null || serviceUuid == null)
170             throw new AssertionError(
171 	       "serviceID and serviceUuid must be set");
172     
173         // Default do nothing preparer
174         final ProxyPreparer defaultPreparer =
175             new net.jini.security.BasicProxyPreparer();
176         lookupLocatorPreparer =
177             (ProxyPreparer)Config.getNonNullEntry(config,
178                 TxnManager.MAHALO, "lookupLocatorPreparer",
179                 ProxyPreparer.class, defaultPreparer);
180         if(initlogger.isLoggable(Level.CONFIG)) {
181 	    initlogger.log(Level.CONFIG, "lookupLocatorPreparer: {0}",
182 	         lookupLocatorPreparer);
183 	}
184 //TODO - defer creation of default LDM
185 	dm = (DiscoveryManagement)
186 	    Config.getNonNullEntry(config, TxnManager.MAHALO,
187 		"discoveryManager", DiscoveryManagement.class, 
188 		new LookupDiscoveryManager(
189                     DiscoveryGroupManagement.NO_GROUPS, null, null, config));
190         if(initlogger.isLoggable(Level.CONFIG)) {
191 	    initlogger.log(Level.CONFIG, "discoveryManager: {0}", dm);
192 	}
193 
194 	if (dm instanceof DiscoveryGroupManagement) {
195 	    // Verify proper initial state ---> NO_GROUPS
196             String[] groups =
197                        ((DiscoveryGroupManagement)dm).getGroups();
198             if( (groups == DiscoveryGroupManagement.ALL_GROUPS) || 
199 	        (groups.length != 0) )
200             {
201                 throw new ConfigurationException(
202 		    "discoveryManager entry must be configured " +
203 		    "to initially discover/join NO_GROUPS");
204             }//endif
205     
206         } else { 
207             throw new ConfigurationException("Entry for component " +
208 		TxnManager.MAHALO + ", name " +
209 		"discoveryManager must implement " +
210 	        "net.jini.discovery.DiscoveryGroupManagement");
211 	}
212 	
213 	if (dm instanceof DiscoveryLocatorManagement) {
214             LookupLocator[] locs =
215                     ((DiscoveryLocatorManagement)dm).getLocators();
216             if( (locs != null) && (locs.length != 0) ) {
217                 throw new ConfigurationException
218                     ("discoveryManager entry must be initially"
219 		    + "configured with no locators");
220             }//endif
221 	} else {
222 	    throw new ConfigurationException("Entry for component " +
223 		TxnManager.MAHALO + ", name " +
224 		"discoveryManager must implement " +
225 		"net.jini.discovery.DiscoveryLocatorManagement");
226 	}
227 
228 	// if this is the first incarnation, consult config for groups,
229 	// locators and attributes.
230 	if (initial) {
231             if(initlogger.isLoggable(Level.FINEST)) {
232 	        initlogger.log(Level.FINEST, "Obtaining initial values");
233 	    }
234 	    groups = (String[])
235 		config.getEntry(TxnManager.MAHALO,
236 		    "initialLookupGroups", String[].class, 
237 		    new String[] { "" }); //default to public group
238             if(initlogger.isLoggable(Level.CONFIG)) {
239 	        initlogger.log(Level.CONFIG, "Obtaining initial groups: {0}",
240 		(groups==null?
241 		    Arrays.asList(new String[] {"<ALL_GROUPS>"}):
242 		    Arrays.asList(groups)));
243 	    }
244 	    locators = (LookupLocator[])
245 		Config.getNonNullEntry(config, TxnManager.MAHALO,
246 		     "initialLookupLocators", LookupLocator[].class, 
247                      new LookupLocator[0]);
248             if(initlogger.isLoggable(Level.CONFIG)) {
249 	        initlogger.log(Level.CONFIG, "Obtaining initial locators: {0}",
250 		Arrays.asList(locators));
251 	    }
252 	    final Entry[] cAttrs = (Entry[])
253 		Config.getNonNullEntry(config, TxnManager.MAHALO,
254 		    "initialLookupAttributes", Entry[].class, new Entry[0]);
255             if(initlogger.isLoggable(Level.CONFIG)) {
256 	        initlogger.log(Level.CONFIG, "Obtaining initial attributes: {0}",
257 		Arrays.asList(cAttrs));
258 	    }
259 	    if (cAttrs.length == 0) {
260 		attributes = baseAttributes;
261 	    } else {
262 		attributes = new Entry[cAttrs.length + baseAttributes.length];
263                 System.arraycopy(baseAttributes, 0, attributes,
264                                  0, baseAttributes.length);
265                 System.arraycopy(cAttrs, 0, attributes,
266                                  baseAttributes.length, cAttrs.length);
267 	    }
268             if(initlogger.isLoggable(Level.FINEST)) {
269 	        initlogger.log(Level.FINEST, "Combined attributes: {0}",
270 		Arrays.asList(attributes));
271 	    }
272 
273         } else {
274             /* recovery : if there are any locators get and
275              * use recoveredLookupLocatorPreparer
276              */
277             if(initlogger.isLoggable(Level.FINEST)) {
278 	        initlogger.log(Level.FINEST, "Recovered locators: {0}",
279 		Arrays.asList(locators));
280 	    }
281             if (locators.length > 0) {
282                 final ProxyPreparer recoveredLookupLocatorPreparer =
283                     (ProxyPreparer)Config.getNonNullEntry(config,
284                         TxnManager.MAHALO,
285                         "recoveredLookupLocatorPreparer", ProxyPreparer.class,
286                          defaultPreparer);
287                 if(initlogger.isLoggable(Level.CONFIG)) {
288 	            initlogger.log(Level.CONFIG, "recoveredLookupLocatorPreparer: {0}",
289 		    recoveredLookupLocatorPreparer);
290                 }
291 		final List prepared = new java.util.LinkedList();
292                 for (int i=0; i<locators.length; i++) {
293                     try {
294                         prepared.add(recoveredLookupLocatorPreparer.
295                                      prepareProxy(locators[i]));
296                     } catch (Throwable t) {
297                         if(initlogger.isLoggable(Levels.HANDLED)) {
298 	                    initlogger.log(Levels.HANDLED, 
299 			    "Exception re-preparing LookupLocator: {0}. " 
300 			        + "Dropping locator.",
301 			    locators[i]);
302 			}
303                         if(initlogger.isLoggable(Levels.HANDLED)) {
304 	                    initlogger.log(Levels.HANDLED, 
305 			    "Preparer exception: ", t);
306 			}
307                     }
308                 }
309                 locators =
310                     (LookupLocator[])prepared.toArray(new LookupLocator[0]);
311 	    }
312 	}
313 
314 	// Now that we have groups & locators (either from 
315 	// a previous incarnation or from the config) start discovery.
316         if(initlogger.isLoggable(Level.FINEST)) {
317 	    initlogger.log(Level.FINEST, "Setting groups and locators");
318 	}
319 	((DiscoveryGroupManagement)dm).setGroups(groups);
320 	((DiscoveryLocatorManagement)dm).setLocators(locators);
321 
322         if(initlogger.isLoggable(Level.FINEST)) {
323 	    initlogger.log(Level.FINEST, "Creating JoinManager");
324         }
325 	mgr = new JoinManager(service, attributes, serviceID, 
326             dm, null, config);
327         // Once we are running we don't need the attributes,
328         // locators, and groups fields, null them out (the
329         // state is in the mgr and dm.
330         attributes = null;
331         groups = null;
332         locators = null;
333 
334 	// Now that we have state, make sure it is written to disk.
335         if(initlogger.isLoggable(Level.FINEST)) {
336 	    initlogger.log(Level.FINEST, "Taking snapshot");
337 	}
338 	update();
339 	if (operationsLogger.isLoggable(Level.FINER)) {
340             operationsLogger.exiting(JoinStateManager.class.getName(), 
341 	        "startManager");
342 	}
343     }

372     /** 
373      * Make a good faith attempt to terminate
374      * discovery, and cancel any lookup registrations.  */
375     public void stop() {
376 	if (operationsLogger.isLoggable(Level.FINER)) {
377             operationsLogger.entering(JoinStateManager.class.getName(), 
378 	        "stop");
379 	}
380 	// Unregister with lookup
381 
382 	// Terminate the JoinManager first so it will not call
383 	// into the dm after it has been terminated.
384 	if (mgr != null)
385 	    mgr.terminate();
386 
387 	if (dm != null) 
388 	    dm.terminate();
389 	if (operationsLogger.isLoggable(Level.FINER)) {
390             operationsLogger.exiting(JoinStateManager.class.getName(), 
391 	        "stop");
392 	}
393     }

404             operationsLogger.exiting(JoinStateManager.class.getName(), 
405 	        "destroy");
406 	}
407     }

651     public void setLookupLocators(LookupLocator[] locators) 
652         throws RemoteException
653     {
654 	if (operationsLogger.isLoggable(Level.FINER)) {
655             operationsLogger.entering(JoinStateManager.class.getName(), 
656 	        "setLookupLocators");
657 	}
658         prepareLocators(locators);
659 	((DiscoveryLocatorManagement)dm).setLocators(locators);
660 	update();
661 	if (operationsLogger.isLoggable(Level.FINER)) {
662             operationsLogger.exiting(JoinStateManager.class.getName(), 
663 	        "setLookupLocators");
664 	}
665     }

667     /**
668      * Apply <code>lookupLocatorPreparer</code> to each locator in the
669      * array, replacing the original locator with the result of the
670      * <code>prepareProxy</code> call. If call fails with an exception
671      * throw that exception.
672      * @param locators the <code>LookupLocator</code>s to be prepared.
673      * @throws RemoteException if preparation of any of the locators
674      *         does.
675      * @throws SecurityException if preparation of any of the locators
676      *         does.
677      */
678     private void prepareLocators(LookupLocator[] locators)
679         throws RemoteException
680     {
681 	if (operationsLogger.isLoggable(Level.FINER)) {
682             operationsLogger.entering(JoinStateManager.class.getName(), 
683 	        "prepareLocators");
684 	}
685         for (int i = 0; i<locators.length; i++) {
686             locators[i] = (LookupLocator)lookupLocatorPreparer.prepareProxy(
687                 locators[i]);
688 	}
689 	if (operationsLogger.isLoggable(Level.FINER)) {
690             operationsLogger.exiting(JoinStateManager.class.getName(), 
691 	        "prepareLocators");
692 	}
693     }

712     /**
713      * Utility method to write out an array of entities to an
714      * <code>ObjectOutputStream</code>.  Can be recovered by a call
715      * to <code>readAttributes()</code>
716      * <p>
717      * Packages each attribute in its own <code>MarshalledObject</code> so
718      * a bad codebase on an attribute class will not corrupt the whole array.
719      */
720     // @see JoinAdminActivationState#readAttributes
721     static private void writeAttributes(Entry[] attributes,
722                                         ObjectOutput out)
723         throws IOException
724     {
725         // Need to package each attribute in its own marshaled object,
726         // this makes sure that the attribute's code base is preserved
727         // and when we unpack to discard attributes who's codebase
728         // has been lost without throwing away those we can still deal with.
729          
730         out.writeInt(attributes.length);
731         for (int i=0; i<attributes.length; i++) {
732             out.writeObject(
733                 new MarshalledInstance(attributes[i]).convertToMarshalledObject());
734 	}
735     }
736  
737     /**
738      * Utility method to read in an array of entities from a
739      * <code>ObjectInputStream</code>.  Array should have been written
740      * by a call to <code>writeAttributes()</code>
741      * <p>
742      *   
743      * Will try and recover as many attributes as possible.
744      * Attributes which can't be recovered won't be returned but they
745      * will remain in the log.
746      * 
747      */
748     // @see JoinAdminActivationState#writeAttributes
749     static private Entry[] readAttributes(ObjectInput in)
750         throws IOException, ClassNotFoundException
751     {
752         final List entries = new java.util.LinkedList();
753         final int objectCount = in.readInt();
754         for (int i=0; i<objectCount; i++) {
755             try {
756                 MarshalledObject mo = (MarshalledObject)in.readObject();
757                 entries.add(new MarshalledInstance(mo).get(false));
758             } catch (IOException e) {
759                 if(initlogger.isLoggable(Levels.HANDLED)) {
760 	            initlogger.log(Levels.HANDLED, 
761 		    "Exception getting service attribute ... skipping", e);
762 		}
763             } catch (ClassNotFoundException e) {
764                 if(initlogger.isLoggable(Levels.HANDLED)) {
765 	            initlogger.log(Levels.HANDLED, 
766 		    "Exception getting service attribute ... skipping", e);
767 		}
768             }
769         }
770  
771         return (Entry[])entries.toArray(new Entry[0]);
772     }

779     public void snapshot(OutputStream out) throws IOException {

781         oostream.writeObject(serviceUuid);
782         writeAttributes(mgr.getAttributes(), oostream);

784         oostream.writeObject(((DiscoveryGroupManagement)dm).getGroups());
785 	oostream.flush();
786     }
