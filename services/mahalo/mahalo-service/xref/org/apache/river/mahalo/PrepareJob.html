35  /**
36   * An implementation of a <code>org.apache.river.mahalo.Job</code> which
37   * interacts with a set of
38   * <code>net.jini.core.transaction.server.TransactionParticipant</code>s
39   * to inform them to vote.
40   *
41   * @author Sun Microsystems, Inc.
42   *
43   * @see org.apache.river.mahalo.Job
44   * @see org.apache.river.mahalo.ParticipantTask
45   * @see net.jini.core.transaction.Transaction
46   * @see net.jini.core.transaction.server.TransactionParticipant
47   */
48  class PrepareJob extends Job implements TransactionConstants {
49      final ServerTransaction tr;
50      final ClientLog log;
51      final ParticipantHandle[] handles;
52      final int maxtries = 5;
53      /** Logger for operations related messages */
54      private static final Logger operationsLogger =
55         TxnManagerImpl.operationsLogger;
56  
57      /** Logger for persistence related messages */
58      private static final Logger persistenceLogger =
59          TxnManagerImpl.persistenceLogger;
60  
61      /**
62       * Constructs an <code>PrepareJob</code>
63       *
64       *
65       * @param tr The <code>Transaction</code> whose participants
66       *           will be instructed to vote
67       *
68       * @param pool The <code>TaskManager</code> which provides the
69       *             threads used for interacting with participants.
70       *
71       * @param log  The <code>ClientLog</code> used for recording
72       *             recovery data.
73       *
74       * @param handles The array of participants which will be contacted
75       *                and informed to vote
76       *
77       * @see org.apache.river.thread.TaskManager
78       * @see org.apache.river.mahalo.log.ClientLog
79       * @see net.jini.core.transaction.server.TransactionParticipant
80       */
81      public PrepareJob(Transaction tr, ExecutorService pool, WakeupManager wm, ClientLog log, ParticipantHandle[] handles, AccessControlContext context) {
82  	super(pool, wm, context);
83  
84  	if (log == null)
85  	    throw new IllegalArgumentException("PrepareJob: PrepareJob: " +
86  							"log is null");
87  
88  	this.log = log;
89  
90  	if (!(tr instanceof ServerTransaction))
91  	    throw new IllegalArgumentException("PrepareJob: PrepareJob: " +
92  					"must be a ServerTransaction");
93  
94  	this.tr =  (ServerTransaction) tr;
95  
96  	if (handles == null)
97  	    throw new IllegalArgumentException("PrepareJob: PrepareJob: " +
98  					"must have participants");
99  
100 	if (handles.length == 0)
101 	    throw new IllegalArgumentException("PrepareJob: PrepareJob: " +
102 					"must have participants");
103 
104 	this.handles = handles;
105     }
106 
107 
108     /**
109      * The work to be performed by each <code>TaskManager.Task</code>
110      * is provided by the <code>Job</code> that creates it.
111      * The work performed by a task belonging to the AbortJob
112      * contacts a participant, instructs it to vote and
113      * log appropriately.
114      *
115      * @param who The task performing the work
116      *
117      * @param param A parameter, of the task's choosing, useful
118      *              in performing work.
119      *
120      * @see org.apache.river.mahalo.Job
121      * @see org.apache.river.thread.TaskManager.Task
122      */
123     Object doWork(Runnable who, Object param) {
124         if (operationsLogger.isLoggable(Level.FINER)) {
125             operationsLogger.entering(PrepareJob.class.getName(),
126                 "doWork", new Object[] {who, param});
127 	}
128 	ParticipantHandle handle = (ParticipantHandle)param;
129 	TransactionParticipant par = null;
130 
131         //check if a vote already exists because it was
132         //recovered from the log. In this situation,
133         //we do not need to log this info since it
134         //exists in the log which was used for recovery...
135  
136         int vote = handle.getPrepState();
137  
138         switch (vote) {
139             case COMMITTED:
140             case NOTCHANGED:
141             case ABORTED:
142             case PREPARED:
143                 if (operationsLogger.isLoggable(Level.FINER)) {
144                     operationsLogger.exiting(
145 		        PrepareJob.class.getName(), 
146 			"doWork", Integer.valueOf(vote));
147 		}
148 		return Integer.valueOf(vote);
149         }
150  
151         //...otherwise, explicitly instruct the participant to
152         //prepare after unpacking it and checking against the
153         //max retry threshold
154  
155         if (par == null)
156             par = handle.getPreParedParticipant();
157  
158         //If you have exhausted the max retry threshold
159         //stop, so that no further attempts are made.
160  
161 	try {
162             if (attempt(who) > maxtries) {
163                 if (operationsLogger.isLoggable(Level.FINER)) {
164                     operationsLogger.exiting(
165 		        PrepareJob.class.getName(),"doWork",
166 		        Integer.valueOf(ABORTED));
167 		}
168 	        return Integer.valueOf(ABORTED);
169             }
170 	} catch (JobException je) {
171             if (operationsLogger.isLoggable(Level.FINER)) {
172                 operationsLogger.exiting(
173 		    PrepareJob.class.getName(),"doWork", null);
174 	    }
175 	    return null;
176 	}
177  
178  
179         //At this point, if participant is null, there
180         //must be an error unpacking, so retry later
181         if (par == null) {
182             if (operationsLogger.isLoggable(Level.FINER)) {
183                 operationsLogger.exiting(
184 		    PrepareJob.class.getName(),"doWork", null);
185 	    }
186             return null;
187         }
188  
189         //Here we actually need to ask the participant to
190         //prepare.  Note the RemoteException causes a
191         //retry. Here we only log info for the cases
192         //where a final outcome is available.
193  
194         Object response = null;
195  
196         try {
197             vote = par.prepare(tr.mgr, tr.id);
198             response = Integer.valueOf(vote);
199         } catch (TransactionException bte) {
200             vote = ABORTED;
201             response = Integer.valueOf(vote);
202         } catch (RemoteException re) {
203         } catch (RuntimeException rte) {
204 	    vote = ABORTED;
205 	    response = Integer.valueOf(vote);
206 	}
207 
208         if (response != null) {
209 	    handle.setPrepState(vote);
210             try {
211                 log.write( new PrepareRecord(handle, vote));
212             } catch (org.apache.river.mahalo.log.LogException le) {
213                 //the full package name used to disambiguate
214                 //the LogException
215                 if (persistenceLogger.isLoggable(Level.WARNING)) {
216                     persistenceLogger.log(Level.WARNING,
217 	            "Problem writing PrepareRecord.", le);
218 	        }
219 //TODO - ignore?		
220             }
221             if (operationsLogger.isLoggable(Level.FINER)) {
222                 operationsLogger.exiting(
223 		   PrepareJob.class.getName(),"doWork", response);
224 	    }
225 
226 	    return response;
227         }
228  
229 	if (operationsLogger.isLoggable(Level.FINER)) {
230             operationsLogger.exiting(
231 		PrepareJob.class.getName(),"doWork", null);
232 	}
233 	return null;
234     }

237     /**
238      * Creates the <code>TaskManager.Task</code>s necessary to
239      * inform participants to vote.
240      */
241     Runnable[] createTasks() {
242         int length = handles.length;
243 	Runnable[] tmp = new Runnable[length];
244 
245 	for (int i = 0; i < length; i++) {
246 	    tmp[i] = 
247 	        new ParticipantTask(getPool(), getMgr(), this, handles[i]);
248 	}
249 
250 	return tmp;
251     }

254     /**
255      * Gathers partial results submitted by tasks and produces
256      * a single outcome.
257      *
258      * @see org.apache.river.mahalo.Job
259      */
260     Object computeResult() throws JobException {
261         if (operationsLogger.isLoggable(Level.FINER)) {
262             operationsLogger.entering(PrepareJob.class.getName(),
263                 "computeResult");
264 	}
265 	try {
266 	    if (!isCompleted(0))
267 	        throw new ResultNotReadyException("Cannot compute result " +
268 					"since there are jobs pending");
269 	} catch (JobNotStartedException jnse) {
270 	    throw new ResultNotReadyException("Cannot compute result since" +
271 					   " jobs were not created");
272 	}
273 
274 	int prepstate = NOTCHANGED;
275 	int tmp = 0;
276 
277         
278         Iterator i = results.values().iterator();
279         checkresults:
280         while (i.hasNext()){
281                 tmp = ((Integer) i.next()).intValue();
282 	    switch(tmp) {
283 	      case NOTCHANGED:
284 		//Does not affect the prepstate
285 	        break;
286 
287 	      case ABORTED:
288 		//Causes all further checks to end
289 		//while marking ABORTED. A single abort
290 		//aborts the whole transaction.
291 
292 		prepstate = ABORTED;
293                 break checkresults;
294 
295 	      case PREPARED:
296 		//changes the state to PREPARED only
297 		//if currently NOTCHANGED
298 		if (prepstate == NOTCHANGED)
299 		    prepstate = PREPARED;
300 		break;
301 	    }
302 	}
303         
304         
305 	Integer r = Integer.valueOf(prepstate);
306         if (operationsLogger.isLoggable(Level.FINER)) {
307             operationsLogger.exiting(PrepareJob.class.getName(),
308                 "computeResult", r);
309 	}
310 	return r;
311     }
