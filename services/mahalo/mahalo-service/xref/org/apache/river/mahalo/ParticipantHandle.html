18  package org.apache.river.mahalo;
19  
20  import java.io.IOException;
21  import java.io.InvalidObjectException;
22  import java.io.ObjectOutputStream;
23  import java.io.Serializable;
24  import java.rmi.RemoteException;
25  import java.util.logging.Level;
26  import java.util.logging.Logger;
27  import net.jini.core.transaction.server.TransactionConstants;
28  import net.jini.core.transaction.server.TransactionParticipant;
29  import net.jini.security.ProxyPreparer;
30  import org.apache.river.api.io.AtomicSerial;
31  import org.apache.river.api.io.AtomicSerial.GetArg;
32  
33  
34  /**
35   *
36   * @author Sun Microsystems, Inc.
37   *
38   */
39  @AtomicSerial
40  class ParticipantHandle implements Serializable, TransactionConstants {
41      static final long serialVersionUID = -1776073824495304317L;
42  
43      /**
44       * Cached reference to prepared participant.
45       */
46      private volatile transient TransactionParticipant preparedPart;
47  
48      /**
49       * @serial
50       */
51      private final StorableObject storedpart;
52  
53      /**
54       * @serial
55       */
56      private final long crashcount;
57  
58      /**
59       * @serial
60       */
61      private int prepstate;
62  
63      /** Logger for persistence related messages */
64      private static final Logger persistenceLogger = 
65          TxnManagerImpl.persistenceLogger;
66  
67  
68      /**
69       * Create a new node that is equivalent to that node
70       */
71      ParticipantHandle(TransactionParticipant preparedPart, 
72          long crashcount) 
73  	throws RemoteException 
74      {
75          this(check(preparedPart), preparedPart, crashcount, ACTIVE);
76      }
77      
78      ParticipantHandle(GetArg arg) throws IOException {
79  	this(check(arg), 
80  		(TransactionParticipant) arg.get("preparedPart", null),
81  		arg.get("crashcount", 0),
82  		arg.get("prepstate", 0));
83      }
84      
85      private ParticipantHandle(boolean check, TransactionParticipant preparedPart, 
86          long crashcount, int prepstate) throws RemoteException {
87          StorableObject storedpart = null;
88  	try {
89  	    storedpart = new StorableObject(preparedPart);
90  	    this.preparedPart = preparedPart;
91  	} catch (RemoteException re) {
92   	    if (persistenceLogger.isLoggable(Level.WARNING)) {
93                  persistenceLogger.log(Level.WARNING,
94  		    "Cannot store the TransactionParticipant", re);
95  	    }
96  	    crashcount = 0;
97  	    //REMIND:  suspect we are supposed to rethrow exception here?
98  	}
99  	this.crashcount = crashcount;
100         this.storedpart = storedpart;
101 	this.prepstate = prepstate;
102     }
103 
104     private static boolean check(AtomicSerial.GetArg arg) throws IOException {
105 	try {
106 	    return check(arg.get("preparedPart", null));
107 	} catch (IllegalArgumentException ex){
108 	    InvalidObjectException e = new InvalidObjectException("Invariants unsatisfied");
109 	    e.initCause(ex);
110 	    throw e;
111 	}
112     }
113     
114     private static boolean check(Object preparedPart){
115 	 if (preparedPart == null) 
116 	    throw new NullPointerException(
117 	        "TransactionParticipant argument cannot be null");
118 	 return true;
119     }
120 
121     long getCrashCount() {
122 	return crashcount;
123     }
124 
125     synchronized TransactionParticipant getPreParedParticipant() {
126 	return preparedPart;
127     }
128 
129     // Only called by service initialization code 
130     void restoreTransientState(ProxyPreparer recoveredListenerPreparer) 
131         throws RemoteException
132     {
133         if (recoveredListenerPreparer == null) 
134 	    throw new NullPointerException(
135 	        "Preparer argument cannot be null");
136 	/*
137 	 * ProxyPreparation potentially make remote calls. So,
138 	 * need to make sure that locks aren't being held across this 
139 	 * invocation.
140 	 */
141 	preparedPart = (TransactionParticipant)
142 	    recoveredListenerPreparer.prepareProxy(storedpart.get());
143     }
144 
145     StorableObject getStoredPart() {
146 	return storedpart;
147     }
148 
149     synchronized void setPrepState(int state) {
150 	switch (state) {
151 	    case PREPARED:
152 	    case NOTCHANGED:
153 	    case COMMITTED:
154 	    case ABORTED:
155 		break;
156 	    default:
157 		throw new IllegalArgumentException("ParticipantHandle: " +
158 			    "setPrepState: cannot set to " + 
159 		    	    org.apache.river.constants.TxnConstants.getName(state));
160 	}
161 
162 	this.prepstate = state;
163     }
164 
165     synchronized int getPrepState() {
166 	return prepstate;
167     }
168     
169     private synchronized void writeObject(ObjectOutputStream out) throws IOException {
170 	out.defaultWriteObject();
171     }
172     
173     /**
174      * Return the <code>hashCode</code> of the 
175      * embedded <code>TransactionParticipant</code>.
176      */
177     public int hashCode() {
178         return preparedPart.hashCode();
179     }
180 
181     public boolean equals(Object that) {
182 	if (this == that) 
183 	    return true;
184 	if (that == null) 
185 	    return false;
186 	if (that.getClass() != getClass()) 
187 	    return false;
188 
189         ParticipantHandle h = (ParticipantHandle)that; 
190         return preparedPart.equals(h.preparedPart);
191     }
192 }
