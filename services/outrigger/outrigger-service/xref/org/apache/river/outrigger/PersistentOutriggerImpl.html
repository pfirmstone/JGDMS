18  package org.apache.river.outrigger;
19  
20  import java.io.IOException;
21  import java.rmi.MarshalledObject;
22  import java.rmi.activation.ActivationID;
23  import java.rmi.activation.ActivationException;
24  import javax.security.auth.login.LoginException;
25  import net.jini.config.ConfigurationException;
26  import org.apache.river.start.lifecycle.LifeCycle;
27  import net.jini.io.MarshalledInstance;
28  
29  /**
30   * <code>OutriggerServerWrapper</code> subclass for
31   * persistent servers.
32   *
33   * @author Sun Microsystems, Inc.
34   * @since 2.0
35   */
36  class PersistentOutriggerImpl extends OutriggerServerWrapper {
37      /**
38       * Create a new incarnation of an activatable
39       * <code>OutriggerServerImpl</code> server.
40       * @param activationID of the server, may be <code>null</code>.
41       * @param data an array of <code>String</code>s (packaged in
42       *        a marshalled object) that will be used 
43       *        to obtain a <code>Configuration</code>.
44       * @throws IOException if there is problem recovering data
45       *         from disk, exporting the server, or unpacking
46       *         <code>data</code>.
47       * @throws ClassCastException if the value of <code>data.get()</code>
48       *         is not an array of <code>String</code>s.
49       * @throws ConfigurationException if the <code>Configuration</code> is 
50       *         malformed.  
51       * @throws ActivationException if activatable and there
52       *         is a problem getting a reference to the activation system.
53       * @throws LoginException if the <code>loginContext</code> specified
54       *         in the configuration is non-null and throws 
55       *         an exception when login is attempted.
56       * @throws ClassNotFoundException if the classes of the objects
57       *         encapsulated inside <code>data</code> can not be found.
58       */
59      protected PersistentOutriggerImpl(ActivationID activationID, 
60  			    MarshalledObject data) 
61  	throws IOException, ConfigurationException, LoginException,
62  	       ActivationException, ClassNotFoundException
63      {
64  	super(activationID, (String[]) new MarshalledInstance(data).get(false));
65  	allowCalls();
66      }
67  
68      /**
69       * Create a new non-activatable, persistent space.
70       * The space will be implemented by a new
71       * <code>OutriggerServerImpl()</code> server instance.
72       * @param configArgs set of strings to be used to obtain a
73       *                   <code>Configuration</code>.
74       * @param lifeCycle the object to notify when this
75       *                  service is destroyed.
76       * @throws IOException if there is problem recovering data
77       *         from disk or exporting the server for the space.
78       * @throws ConfigurationException if the configuration is 
79       *         malformed.  
80       * @throws LoginException if the <code>loginContext</code> specified
81       *         in the configuration is non-null and throws 
82       *         an exception when login is attempted.
83       */
84      PersistentOutriggerImpl(String[] configArgs, LifeCycle lifeCycle) 
85          throws IOException, ConfigurationException, LoginException
86      {
87  	super(configArgs, lifeCycle, true);
88  	allowCalls();
89      }
90  }
