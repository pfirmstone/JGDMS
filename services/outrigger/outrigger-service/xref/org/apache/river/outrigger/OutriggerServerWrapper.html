58  /**
59   * For various reasons there is code that we would like
60   * to run before every incoming remote call. To accomplish
61   * this we wrap the server in an object that will run
62   * the common code and then delegate to the server to
63   * do the actual work. This is a base class for these
64   * wrappers.
65   *
66   * @author Sun Microsystems, Inc.
67   * @since 2.0
68   */
69  public class OutriggerServerWrapper 
70      implements OutriggerServer, ServerProxyTrust, ProxyAccessor, Startable,
71  	ServiceProxyAccessor,
72  	       ServiceAttributesAccessor, ServiceIDAccessor, CodebaseAccessor
73  {
74      /** The object being delegated to */     
75      private final OutriggerServerImpl delegate;

83      /** 
84       * If non-null cause incoming calls to immediately throw this
85       * exception. Takes presidents over <code>holdCalls</code>. This
86       * field is only set to an <code>Error</code>,
87       * <code>RemoteException</code>, or <code>RuntimeException</code>
88       * and thus can be thrown by an of <code>OutriggerServer</code>'s
89       * methods.  
90       */
91      private Throwable failCallsWith;

93      /**
94       * Create an <code>OutriggerServerWrapper</code> that
95       * will delegate to a non-activatable <code>OutriggerServerImpl</code>
96       * created with the specified configuration and wrapped by
97       * <code>this</code>.
98       * @param configArgs set of strings to be used to obtain a
99       *                   <code>Configuration</code>.
100      * @param lifeCycle the object to notify when this
101      *                  service is destroyed.
102      * @param persistent If <code>true</code> will throw an 
103      *                   <code>ConfigurationException</code>
104      *                   if there is no persistence directory or
105      *                   store specified in the configuration.
106      * @throws IOException if there is problem exporting the server.
107      * @throws ConfigurationException if the <code>Configuration</code> is 
108      *         malformed.
109      * @throws LoginException if the <code>loginContext</code> specified
110      *         in the configuration is non-null and throws 
111      *         an exception when login is attempted.
112      */
113     protected OutriggerServerWrapper(String[] configArgs, LifeCycle lifeCycle, 
114 			   boolean persistent) 
115 	throws IOException, ConfigurationException, LoginException
116     {
117 	try {
118 	    delegate = new OutriggerServerImpl(null, lifeCycle, configArgs,
119 					       persistent, this);
120 	} catch (ActivationException e) {
121 	    throw new AssertionError(e);
122 	}
123     }

125     /**
126      * Create an <code>OutriggerServerWrapper</code> that
127      * will delegate to an <code>OutriggerServerImpl</code>
128      * created with the specified argument and wrapped by <code>this</code>.
129      * @param activationID of the server, must not be <code>null</code>.
130      * @param configArgs set of strings to be used to obtain a
131      *                   <code>Configuration</code>.
132      * @throws IOException if there is problem recovering data
133      *         from disk, exporting the server, or unpacking 
134      *         <code>data</code>.
135      * @throws ConfigurationException if the <code>Configuration</code> is 
136      *         malformed.
137      * @throws ActivationException if activatable and there
138      *         is a problem getting a reference to the activation system.
139      * @throws LoginException if the <code>loginContext</code> specified
140      *         in the configuration is non-null and throws 
141      *         an exception when login is attempted.
142      * @throws NullPointerException if <code>activationID</code>
143      *         is <code>null</code>.
144      */
145     OutriggerServerWrapper(ActivationID activationID, String[] configArgs) 
146 	throws IOException, ConfigurationException, LoginException,
147 	       ActivationException
148     {
149 	if (activationID == null)
150 	    throw new NullPointerException("activationID must be non-null");
151 	delegate = new OutriggerServerImpl(activationID, null, configArgs,
152 					   true, this);
153     }

155     /**
156      * Cause incoming calls to block until further notice.
157      */
158     synchronized void holdCalls() {
159 	failCallsWith = null;
160 	allowCalls = false;
161 	notifyAll();
162     }
163 
164     /**
165      * Cause in new or blocked calls to fail with
166      * the specified exception.
167      * @throws IllegalArgumentException if <code>t</code>
168      * is not an <code>Error</code>, <code>RemoteException</code>,
169      * or <code>RuntimeException</code>.
170      * @throws NullPointerException if <code>t</code> is <code>null</code>.
171      */
172     synchronized void rejectCalls(Throwable t) {
173 	if (t == null)
174 	    throw new NullPointerException("Throwable must not be null");
175 
176 	if (!((t instanceof Error) || 
177 	      (t instanceof RuntimeException) ||
178 	      (t instanceof RemoteException)))
179 	    throw new IllegalArgumentException("t must be an exception " +
180 		"that can be thrown from any of OutriggerServer's methods");
181 
182 	failCallsWith = t;
183 	allowCalls = true;
184 	notifyAll();
185     }
186 
187     /**
188      * Allow incoming calls.
189      */
190     protected synchronized void allowCalls() {
191 	failCallsWith = null;
192 	allowCalls = true;
193 	notifyAll();
194     }
195     
196     /**
197      * Block until calls are allowed, or until calls
198      * are to be rejected.
199      * @throws RemoteException If calls are being rejected
200      *         with <code>RemoteException</code>s.
201      * @throws RuntimeException If calls are being rejected
202      *         with <code>RuntimeException</code>s.
203      * @throws Error If calls are being rejected
204      *         with <code>Error</code>s.
205      */
206     private synchronized void gate() throws RemoteException {
207 	while (!allowCalls || failCallsWith != null) {
208 	    if (failCallsWith != null) {
209 		if (failCallsWith instanceof RemoteException)
210 		    throw (RemoteException)failCallsWith;
211 		else if (failCallsWith instanceof Error)
212 		    throw (Error)failCallsWith;
213 		else if (failCallsWith instanceof RuntimeException) 
214 		    throw (RuntimeException)failCallsWith;
215 		else
216 		    throw new AssertionError("Wrapper trying to " +
217 					     "throw " + failCallsWith);
218 	    }
219 
220 	    if (!allowCalls) {
221 		try {
222 		    wait();
223 		} catch (InterruptedException e) {
224 		    throw new 
225 			InternalSpaceException("gate method interrupted");
226 		}
227 	    }
228 	}
229     }
