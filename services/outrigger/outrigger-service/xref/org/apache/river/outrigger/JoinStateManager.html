122     /**
123      * Start the manager. Start looking for lookup and registering
124      * with them.
125      * @param config object to use to obtain
126      *               <code>DiscoveryManagement</code> object, and if
127      *               this is the initial incarnation of this service, 
128      *               the object used to get the initial set of groups,
129      *               locators, and deployer defined attributes.
130      * @param log object used to persist the manager's state, may be
131      *            <code>null</code>.
132      * @param serviceID The <code>ServiceID</code> to register
133      *                  under.
134      * @param service The proxy object to register with lookups.
135      * @param baseAttributes Any attributes the implementation wants
136      *                       attached, only used if this is the
137      *                       initial incarnation.
138      * @throws IOException if the is problem persisting the 
139      *         initial state or in starting discovery.
140      * @throws ConfigurationException if the configuration
141      *         is invalid.
142      * @throws NullPointerException if <code>config</code>, 
143      *         or <code>serviceID</code> is  <code>null</code>.
144      */
145     void startManager(Configuration config, LogOps log, Object service,
146 		      ServiceID serviceID, Entry[] baseAttributes) 
147 	throws IOException, ConfigurationException
148     {
149 	// Default do nothing preparer
150 	final ProxyPreparer defaultPreparer = 
151 	    new net.jini.security.BasicProxyPreparer();
152 
153 	if (serviceID == null)
154 	    throw new NullPointerException("serviceID can't be null");
155 
156 	this.log = log;
157 
158 	lookupLocatorPreparer = 
159 	    (ProxyPreparer)Config.getNonNullEntry(config, 
160 		OutriggerServerImpl.COMPONENT_NAME, "lookupLocatorPreparer",
161 		ProxyPreparer.class, defaultPreparer);
162 
163 	dgm = (DiscoveryGroupManagement)
164 	    Config.getNonNullEntry(config, 
165 		OutriggerServerImpl.COMPONENT_NAME, "discoveryManager",
166 		DiscoveryGroupManagement.class, 
167 		new LookupDiscoveryManager(
168                     DiscoveryGroupManagement.NO_GROUPS, null, null,
169 		    config));
170 
171 	if (!(dgm instanceof DiscoveryManagement))
172 	    throw throwNewConfigurationException("Entry for component " +
173 		OutriggerServerImpl.COMPONENT_NAME + ", name " +
174 		"discoveryManager must implement " +
175 	        "net.jini.discovery.DiscoveryGroupManagement");
176 
177 	if (!(dgm instanceof DiscoveryLocatorManagement))
178 	    throw throwNewConfigurationException("Entry for component " +
179 		OutriggerServerImpl.COMPONENT_NAME + ", name " +
180 		"discoveryManager must implement " +
181 		"net.jini.discovery.DiscoveryLocatorManagement");
182 
183 	final String[] toCheck = dgm.getGroups();
184 	if (toCheck == null || toCheck.length != 0)
185 	    throw throwNewConfigurationException("Entry for component " +
186 		OutriggerServerImpl.COMPONENT_NAME + ", name " +
187 		"discoveryManager must be initially configured with no " +
188                 "groups");
189 
190 	if (((DiscoveryLocatorManagement)dgm).getLocators().length != 0)
191 	    throw throwNewConfigurationException("Entry for component " +
192                 OutriggerServerImpl.COMPONENT_NAME + ", name " +
193 		"discoveryManager must be initially configured with no " +
194 		"locators");
195 
196 	// if this is the first incarnation, consult config for groups,
197 	// locators and attributes.
198 	if (initial) {
199 	    groups = (String[])
200 		config.getEntry(OutriggerServerImpl.COMPONENT_NAME,
201 		    "initialLookupGroups", String[].class,
202 		    new String[]{""});
203 
204 	    locators = (LookupLocator[])
205 		Config.getNonNullEntry(config, 
206 		    OutriggerServerImpl.COMPONENT_NAME,
207                     "initialLookupLocators", LookupLocator[].class, 
208                     new LookupLocator[0]);
209 
210 	    final Entry[] cAttrs = (Entry[])
211 		Config.getNonNullEntry(config, 
212                     OutriggerServerImpl.COMPONENT_NAME,
213 		    "initialLookupAttributes", Entry[].class, new Entry[0]);
214 
215 	    if (cAttrs.length == 0) {
216 		attributes = baseAttributes;
217 	    } else {
218 		attributes = 
219 		    new Entry[cAttrs.length + baseAttributes.length];
220 		System.arraycopy(baseAttributes, 0, attributes, 
221 				 0, baseAttributes.length);
222 		System.arraycopy(cAttrs, 0, attributes, 
223 				 baseAttributes.length, cAttrs.length);
224 	    }
225 	} else {
226 	    /* recovery : if there are any locators get and
227 	     * use recoveredLookupLocatorPreparer
228 	     */
229 	    if (locators.length > 0) {
230 		final ProxyPreparer recoveredLookupLocatorPreparer = 
231 		    (ProxyPreparer)Config.getNonNullEntry(config, 
232 		        OutriggerServerImpl.COMPONENT_NAME,
233 			"recoveredLookupLocatorPreparer", ProxyPreparer.class,
234 			 defaultPreparer);
235 
236 		final List prepared = new java.util.LinkedList();
237 		for (int i=0; i<locators.length; i++) {
238 		    final LookupLocator locator = locators[i];
239 		    try {
240 			prepared.add(recoveredLookupLocatorPreparer.
241 				     prepareProxy(locator));
242 		    } catch (Throwable t) {
243 			logger.log(Level.INFO,
244 			    "Encountered exception preparing lookup locator " +
245 			    "for " + locator + ", dropping locator", t);
246 		    }
247 		}
248 
249 		locators = 
250 		    (LookupLocator[])prepared.toArray(new LookupLocator[0]);
251 	    }
252 	}
253 
254 	// Now that we have groups & locators (either from 
255 	// a previous incarnation or from the config) start discovery.
256 	if (logger.isLoggable(Level.CONFIG)) {
257 	    if (groups == null) {
258 		logger.log(Level.CONFIG, "joining all groups");
259 	    } else if (groups.length == 0) {
260 		logger.log(Level.CONFIG, "joining no groups");	
261 	    } else {
262 		final StringBuffer buf = new StringBuffer();
263 		buf.append("joining groups:");
264 		for (int i=0; i<groups.length; i++) {
265 		    if (i != 0)
266 			buf.append(",");
267 
268 		    buf.append("\"");
269 		    buf.append(groups[i]);
270 		    buf.append("\"");
271 		}
272 		logger.log(Level.CONFIG, buf.toString());
273 	    }
274 
275 	    if (locators.length == 0) {
276 		logger.log(Level.CONFIG, "joining no specific registrars");
277 	    } else {
278 		final StringBuffer buf = new StringBuffer();
279 		buf.append("joining the specific registrars:");
280 		for (int i=0; i<locators.length; i++) {
281 		    if (i != 0)
282 			buf.append(", ");
283 
284 		    buf.append(locators[i]);
285 		}
286 		logger.log(Level.CONFIG, buf.toString());
287 	    }
288 
289 	    if (attributes.length == 0) {
290 		logger.log(Level.CONFIG, "registering no attributes");
291 	    } else {
292 		final StringBuffer buf = new StringBuffer();
293 		buf.append("registering the attributes:");
294 		for (int i=0; i<attributes.length; i++) {
295 		    if (i != 0)
296 			buf.append(", ");
297 
298 		    buf.append(attributes[i]);
299 		}
300 		logger.log(Level.CONFIG, buf.toString());		
301 	    }
302 	}
303 
304 	dgm.setGroups(groups);
305 	((DiscoveryLocatorManagement)dgm).setLocators(locators);
306 
307 	mgr = new JoinManager(service, attributes, serviceID, 
308 			      (DiscoveryManagement)dgm, null, config);
309 
310 	// Once we are running we don't need the attributes,
311 	// locators, and groups fields, null them out (the
312 	// state is in the mgr and dgm.
313 	attributes = null;
314 	groups = null;
315 	locators = null;
316 
317 	// Now that we have state, make sure it is written to disk.
318 	update();
319     }

321     /** 
322      * Make a good faith attempt to terminate
323      * discovery, and cancel any lookup registrations.  */
324     public void destroy() {
325 	// Unregister with lookup
326 
327 	// Terminate the JoinManager first so it will not call
328 	// into the dgm after it has been terminated.
329 	if (mgr != null)
330 	    mgr.terminate();
331 
332 	if (dgm != null) 
333 	    ((DiscoveryManagement)dgm).terminate();
334     }

511     /**
512      * Apply <code>lookupLocatorPreparer</code> to each locator in the
513      * array, replacing the original locator with the result of the
514      * <code>prepareProxy</code> call. If call fails with an exception
515      * throw that exception.  
516      * @param locators the <code>LookupLocator</code>s to be prepared.
517      * @throws RemoteException if preparation of any of the locators
518      *         does.
519      * @throws SecurityException if preparation of any of the locators
520      *         does.
521      */
522     private void prepareLocators(LookupLocator[] locators) 
523         throws RemoteException 
524     {
525 	for (int i = 0; i<locators.length; i++) 
526 	    locators[i] = (LookupLocator)lookupLocatorPreparer.prepareProxy(
527 		locators[i]);
528     }

530     /**
531      * Utility method to write out an array of entities to an
532      * <code>ObjectOutputStream</code>.  Can be recovered by a call
533      * to <code>readAttributes()</code>
534      * <p>
535      * Packages each attribute in its own <code>MarshalledObject</code> so
536      * a bad codebase on an attribute class will not corrupt the whole array.
537      */
538     // @see JoinAdminActivationState#readAttributes
539     static private void writeAttributes(Entry[] attributes,
540                                         ObjectOutputStream out)
541         throws IOException
542     {
543         // Need to package each attribute in its own marshaled object,
544         // this makes sure that the attribute's code base is preserved
545         // and when we unpack to discard attributes who's codebase
546         // has been lost without throwing away those we can still deal with.
547          
548         out.writeInt(attributes.length);
549         for (int i=0; i<attributes.length; i++) {
550             out.writeObject(
551                 new MarshalledInstance(attributes[i]).convertToMarshalledObject());
552 	}
553     }
554  
555     /**
556      * Utility method to read in an array of entities from a
557      * <code>ObjectInputStream</code>.  Array should have been written
558      * by a call to <code>writeAttributes()</code>
559      * <p>
560      *   
561      * Will try and recover as many attributes as possible.
562      * Attributes which can't be recovered won't be returned but they
563      * will remain in the log.
564      */
565     // @see JoinAdminActivationState#writeAttributes
566     static private Entry[] readAttributes(ObjectInputStream in)
567         throws IOException, ClassNotFoundException
568     {
569         final List entries = new java.util.LinkedList();
570         final int objectCount = in.readInt();
571         for (int i=0; i<objectCount; i++) {
572             try {
573                 MarshalledObject mo = (MarshalledObject)in.readObject();
574                 entries.add( new MarshalledInstance(mo).get(false));
575             } catch (IOException e) {
576 		logger.log(Level.INFO, "Encountered IOException recovering " +
577                     "attribute, dropping attribute", e);
578             } catch (ClassNotFoundException e) {
579 		logger.log(Level.INFO, "Encountered ClassNotFoundException " +
580 		    "recovering attribute, dropping attribute", e);
581             }
582         }
583  
584         return (Entry[])entries.toArray(new Entry[0]);
585     }

591     // inherit doc comment
592     public void store(ObjectOutputStream out) throws IOException {
593         writeAttributes(mgr.getAttributes(), out);
594         out.writeObject(((DiscoveryLocatorManagement)dgm).getLocators());
595         out.writeObject(dgm.getGroups());
596     }
597 
598     // inherit doc comment
599     public JoinStateManager restore(ObjectInputStream in) 
600 	throws IOException, ClassNotFoundException 
601     {
602 	initial = false;
603         attributes = readAttributes(in);
604         locators   = (LookupLocator [])in.readObject();
605         groups     = (String [])in.readObject();
606         return this;
607     }

609     /**
610      * Construct, log, and throw a new ConfigurationException with
611      * the given message.
612      */
613     private static ConfigurationException throwNewConfigurationException(
614             String msg) 
615 	throws ConfigurationException
616     {
617 	final ConfigurationException e = new ConfigurationException(msg);
618 
619 	if (logger.isLoggable(Levels.FAILED)) {
620 	    logger.log(Levels.FAILED, msg, e);
621 	}
622 
623 	throw e;
624     }

626     /** 
627      * Propagate an IOException by wrapping it in a RuntimeException.
628      * Performs appropriate logging.
629      */
630     private static RuntimeException propagateIOException(String msg,
631 							 IOException nested)
632     {
633 	final RuntimeException e = new RuntimeException(msg, nested);
634 	if (logger.isLoggable(Levels.FAILED)) {
635 	    logger.log(Levels.FAILED, msg, e);
636 	}
637 
638 	throw e;
639     }
