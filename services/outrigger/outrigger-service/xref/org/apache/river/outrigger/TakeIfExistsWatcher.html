20  import java.util.Map;
21  import java.util.Set;
22  import java.util.WeakHashMap;
23  import net.jini.core.transaction.TransactionException;
24  import net.jini.id.Uuid;
25  import net.jini.space.InternalSpaceException;
26  import org.apache.river.outrigger.proxy.EntryRep;

28  /**
29   * Subclass of <code>QueryWatcher</code> for <code>takeIfExists</code>
30   * queries.  Resolves with the first matching transition where the
31   * entry is visible to the associated transaction and the entry is
32   * still available, or of the locked entry set goes empty.
33   */
34  class TakeIfExistsWatcher extends SingletonQueryWatcher 
35      implements IfExistsWatcher, Transactable
36  {
37      /**
38       * The set of entries that would match but are currently
39       * unavailable (e.g. they are locked). We only keep
40       * the ids, not the entries themselves.
41       */
42      private final Set<Uuid> lockedEntries;
43  
44      /**
45       * Set <code>true</code> once the query thread is 
46       * done processing the backlog. Once this is 
47       * <code>true</code> it is ok to resolve if
48       * <code>lockedEntries</code> is empty.
49       */
50      private boolean backlogFinished = false;
51  
52      /**
53       * If non-null the transaction this query is
54       * being performed under. If <code>null</code> 
55       * this query is not associated with a transaction.
56       */
57      private final Txn txn;
58  
59      /**
60       * Set of entries (represented by <code>EntryHolder</code>s) that
61       * we would have liked to return, but have been provisionally
62       * removed.
63       */
64      private final Set<EntryHandle> provisionallyRemovedEntrySet;
65  
66      /**
67       * Create a new <code>TakeIfExistsWatcher</code>.
68       * @param expiration the initial expiration time
69       *        for this <code>TransitionWatcher</code> in 
70       *        milliseconds since the beginning of the epoch.
71       * @param timestamp the value that is used
72       *        to sort <code>TransitionWatcher</code>s.
73       * @param startOrdinal the highest ordinal associated
74       *        with operations that are considered to have occurred 
75       *        before the operation associated with this watcher.
76       * @param lockedEntries Set of entries (by their IDs)
77       *        that match but are unavailable. Must be non-empty.
78       *        Keeps a reference to this object.
79       * @param provisionallyRemovedEntrySet If the watcher encounters
80       *        an entry that can not be read/taken because it has been
81       *        provisionally removed then its handle will be placed in
82       *        this <code>WeakHashMap</code> as a key (with null as the
83       *        value).  May be <code>null</code> in which case
84       *        provisionally removed entries will not be
85       *        recorded. Ensures that object is only accessed by one
86       *        thread at a time
87       * @param txn If the query is being performed under
88       *        a transaction the <code>Txn</code> object
89       *        associated with that transaction.
90       * @throws NullPointerException if <code>lockedEntries</code> is
91       *         <code>null</code>.
92       */
93      TakeIfExistsWatcher(long expiration, long timestamp, 
94  	 long startOrdinal, Set<Uuid> lockedEntries, 
95           Set<EntryHandle> provisionallyRemovedEntrySet, Txn txn)
96      {
97  	super(expiration, timestamp, startOrdinal);
98  
99  	if (lockedEntries == null) 
100 	    throw new NullPointerException("lockedEntries must be non-null");
101 	
102 	this.lockedEntries = lockedEntries;
103 	this.txn = txn;
104 	this.provisionallyRemovedEntrySet = provisionallyRemovedEntrySet; 
105     }
106 
107     boolean isInterested(EntryTransition transition, long ordinal) {
108 	/* If we are unresolved pretty much all transitions are
109 	 * interesting because we may need to update
110 	 * lockedEntries.
111 	 * 
112 	 * Note, !isResolved() without the lock will result only in
113 	 * false positives, not false negatives - it will only
114 	 * cause isInterested() to return false if we are resolved,
115 	 * we may still return true if we are resolved though.  
116 	 */
117 	return (ordinal>startOrdinal) && !isResolved();
118     }
119 
120     synchronized void process(EntryTransition transition, long now) {
121 	if (isResolved())
122 	    return; // Already done.
123 
124 	final EntryHandle handle = transition.getHandle();
125 	final EntryRep rep = handle.rep();
126 	final boolean isAvailable = transition.isAvailable();
127 	final TransactableMgr transitionTxn = transition.getTxn();
128 
129 	/* If it at one time it was available to our transaction
130 	 * it may still be, try to get it.
131 	 */
132 	if (isAvailable &&
133 	    ((null == transitionTxn) || txn == transitionTxn)) {
134 	    /* Is it still available? */
135 	    if (getServer().attemptCapture(handle, txn, true, null, 
136 		provisionallyRemovedEntrySet, now, this)) 
137 	    {
138 		// Got it
139 		resolve(handle, null);
140 	    } else {
141 		/* Must not have been able to get it. Either
142 		 * locked under a conflicting lock, in which
143 		 * case it should be in our lockedEntries set, or
144 		 * it has been removed, in which it still needs
145 		 * to be in our lockedEntries since it may have
146 		 * been replaced before being removed.
147 		 */
148 		lockedEntries.add(rep.id());		 
149 	    }
150 	} else if (isAvailable) { // but it is not visible to txn
151 	    /* If we are here then at one time it must have been was
152 	     * available but not visible to us, implying that it was an
153 	     * entry written under a transaction and is interesting to
154 	     * us, but not yet visible. We need to add it lockedEntries
155 	     * even if has been removed since it could have gotten
156 	     * replaced before it was removed. If we did not
157 	     * add it we would be acting on the future.
158 	     */
159 	    lockedEntries.add(rep.id());
160 	} else {
161 	    /* Must not be available, transition must mark
162 	     * the resolution of the transaction in such away
163 	     * that the entry has been removed, remove it 
164 	     * from the set and see if that makes the set empty.
165 	     */
166 	    lockedEntries.remove(rep.id());
167 	    if (backlogFinished && lockedEntries.isEmpty())
168 		resolve(null, null);
169 	}
170     }
171 
172     synchronized boolean catchUp(EntryTransition transition, long now) {
173 	if (isResolved())
174 	    return true; // Already done.
175 
176 	final EntryHandle handle = transition.getHandle();
177 	final EntryRep rep = handle.rep();
178 	final boolean isAvailable = transition.isAvailable();
179 	final TransactableMgr transitionTxn = transition.getTxn();
180 
181 	/* If it at one time it was available to our transaction
182 	 * it may still be, try to get it.
183 	 */
184 	if (isAvailable &&
185 	    ((null == transitionTxn) || txn == transitionTxn)) {
186 	    /* Is it still available? Try to get it. attemptCapture will
187 	     * add the entry to lockedEntries for us if we could not
188 	     * get it and it is still in the space (but locked).
189 	     * Nothing will be added if it has been removed outright. 
190 	     * This is ok even though we are peaking into the future -
191 	     * we won't act on that information until the future
192 	     * comes to pass.
193 	     */
194 	    if (getServer().attemptCapture(handle, txn, true,
195 		lockedEntries, provisionallyRemovedEntrySet, now, this)) 
196 	    {
197 		// Got it
198 		resolve(handle, null);
199 		return true;
200 	    }
201 
202 	    // did not resolve
203 	    return false;
204 	}
205 
206 	if (isAvailable) { // but it is not visible to txn
207 	    /* If we are here then at one time it must have been was
208 	     * available but not visible to us, implying that it was
209 	     * an entry written under a transaction and is interesting
210 	     * to us, but not yet visible. We only add if it has not
211 	     * already been removed.  It might have gotten replaced
212 	     * before removal, but since we won't let this query get
213 	     * resolved with a definitive null before we get past the
214 	     * point in the journal where it was removed it is ok to
215 	     * never put it in (and if we did put it in it might never
216 	     * get removed since process() may have already processed
217 	     * the removal). We don't need to check to see it the
218 	     * entry has been provisionally removed since if has been
219 	     * provisional removal does not put in entries in the
220 	     * journal and if it is provisionally removed it has not
221 	     * yet been removed so the remove recored has not yet been
222 	     * created (must less processed).
223 	     */
224 	    synchronized (handle) {
225 		if (!handle.removed()) {
226 		    lockedEntries.add(rep.id());
227 		}
228 
229 		/* If it has been removed, there is no way it
230 		 * will be interesting to us again, ever.
231 		 */
232 	    }
233 	    // Either way, still not resolved.
234 	    return false;
235 	}
236 
237 	/* Must not be available, transition must mark
238 	 * the resolution of the transaction in such away
239 	 * that the entry has been removed, remove it 
240 	 * from the set (don't need to check for empty
241 	 * because we haven't gotten to the point
242 	 * where we can resolve with a definitive null.)
243 	 */
244 	lockedEntries.remove(rep.id());
245 	return false;
246     }
247 
248     /**
249      * Once the backlog is complete we can resolve if 
250      * lockedEntries is/becomes empty.
251      */
252     public synchronized void caughtUp() {
253 	backlogFinished = true;
254 	
255 	if (isResolved())
256 	    return; // Don't much mater.
257 
258 	if (lockedEntries.isEmpty())
259 	    resolve(null, null);	
260     }
261 
262     public synchronized boolean isLockedEntrySetEmpty() {
263 	if (!isResolved())
264 	    throw new IllegalStateException("Query not yet resolved");	
265 	return lockedEntries.isEmpty();
266     }

268     /**
269      * If a transaction ends in the middle of a query we want
270      * to throw an exception to the client making the query
271      * not the <code>Txn</code> calling us here.)
272      */
273     public synchronized int prepare(TransactableMgr mgr,
274 				    OutriggerServerImpl space) 
275     {
276 	assert txn != null:"Transactable method called on a " +
277 	    "non-transactional TakeIfExistsWatcher";
278 
279 	// only throw an exception if we are not resolved.
280 	if (!isResolved()) {
281 	    // Query still in progress, kill it
282 	    resolve(null, new TransactionException("completed while " +
283 						   "operation in progress"));
284 	}
285 
286 	// If this object has made changes they have been recorded elsewhere
287 	return NOTCHANGED;
288     }
289 
290     /**
291      * This should never happen since we always return
292      * <code>NOTCHANGED</code> from <code>prepare</code>.
293      */
294     public void commit(TransactableMgr mgr, OutriggerServerImpl space) {
295 	throw new InternalSpaceException("committing a blocking query");
296 			   
297     }
298 
299     /**
300      * If a transaction ends in the middle of a query we want
301      * to throw an exception to the client making the query 
302      * (not the <code>Txn</code> calling us here.)
303      */
304     public void abort(TransactableMgr mgr, OutriggerServerImpl space) {
305 	// prepare does the right thing, and should forever
306 	prepare(mgr, space);
307     }
