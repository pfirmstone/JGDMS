18  package org.apache.river.norm.proxy;
19  
20  import org.apache.river.landlord.ConstrainableLandlordLease;
21  import org.apache.river.landlord.Landlord;
22  import org.apache.river.landlord.LandlordProxyVerifier;
23  import java.io.IOException;
24  import java.io.InvalidObjectException;
25  import java.io.ObjectInputStream;
26  import java.io.Serializable;
27  import java.rmi.RemoteException;
28  import net.jini.core.constraint.MethodConstraints;
29  import net.jini.core.constraint.RemoteMethodControl;
30  import net.jini.id.Uuid;
31  import net.jini.security.TrustVerifier;
32  import net.jini.security.proxytrust.TrustEquivalence;
33  import org.apache.river.api.io.AtomicSerial;
34  import org.apache.river.api.io.AtomicSerial.GetArg;
35  
36  /** Defines a trust verifier for the smart proxies of a Norm server. */
37  @AtomicSerial
38  public final class ProxyVerifier implements Serializable, TrustVerifier {
39      private static final long serialVersionUID = 2;
40  
41      /**
42       * The Norm server proxy.
43       *
44       * @serial
45       */
46      private final RemoteMethodControl serverProxy;
47  
48      /**
49       * The unique ID for the Norm server.
50       *
51       * @serial
52       */
53      private final Uuid serverUuid;
54  
55      /**
56       * Returns a verifier for the smart proxies of a Norm server with the
57       * specified proxy and unique ID.
58       *
59       * @param serverProxy the Norm server proxy
60       * @param serverUuid the unique ID for the Norm server
61       * @throws UnsupportedOperationException if <code>serverProxy</code> does
62       *	       not implement both {@link RemoteMethodControl} and {@link
63       *	       TrustEquivalence}
64       */
65      public ProxyVerifier(NormServer serverProxy, Uuid serverUuid) {
66  	this(serverProxy, serverUuid, check(serverProxy, serverUuid));
67      }
68      
69      private static boolean check(NormServer serverProxy, Uuid serverUuid){
70  	if (!(serverProxy instanceof RemoteMethodControl)) {
71  	    throw new UnsupportedOperationException(
72  		"Verifier requires service proxy to implement " +
73  		"RemoteMethodControl");
74  	} else if (!(serverProxy instanceof TrustEquivalence)) {
75  	    throw new UnsupportedOperationException(
76  		"Verifier requires service proxy to implement " +
77  		"TrustEquivalence");
78  	}
79  	return true;
80      }
81      
82      ProxyVerifier(GetArg arg) throws IOException {
83  	this((NormServer) arg.get("serverProxy", null),
84  	    (Uuid) arg.get("serverUuid", null),
85  	    check(arg));
86      }
87      
88      private static boolean check(GetArg arg) throws IOException {
89  	NormServer serverProxy = (NormServer) arg.get("serverProxy", null);
90  	Uuid serverUuid = (Uuid) arg.get("serverUuid", null);
91  	try {
92  	    return check(serverProxy, serverUuid);
93  	} catch(UnsupportedOperationException e){
94  	    InvalidObjectException ex = new InvalidObjectException("invariants unsatisfied");
95  	    ex.initCause(e);
96  	    throw ex;
97  	}
98      }
99      
100     private ProxyVerifier(NormServer serverProxy, Uuid serverUuid, boolean check){
101 	this.serverProxy = (RemoteMethodControl) serverProxy;
102 	this.serverUuid = serverUuid;
103     }
104 
105     /**
106      * @throws NullPointerException {@inheritDoc}
107      */
108     public boolean isTrustedObject(Object obj, TrustVerifier.Context ctx)
109 	throws RemoteException
110     {
111 	if (obj == null || ctx == null) {
112 	    throw new NullPointerException("Arguments must not be null");
113 	} else if (obj instanceof ConstrainableLandlordLease) {
114 	    return new LandlordProxyVerifier(
115 		(Landlord) serverProxy, serverUuid).isTrustedObject(obj, ctx);
116 	}
117 	RemoteMethodControl otherServerProxy;
118 	if (obj instanceof SetProxy.ConstrainableSetProxy) {
119 	    if (!isTrustedObject(((SetProxy) obj).getRenewalSetLease(), ctx)) {
120 		return false;
121 	    }
122 	    otherServerProxy =
123 		(RemoteMethodControl) ((AbstractProxy) obj).server;
124 	} else if (obj instanceof AdminProxy.ConstrainableAdminProxy ||
125 		   obj instanceof NormProxy.ConstrainableNormProxy)
126 	{
127 	    if (!serverUuid.equals(((AbstractProxy) obj).uuid)) {
128 		return false;
129 	    }
130 	    otherServerProxy =
131 		(RemoteMethodControl) ((AbstractProxy) obj).server;
132 	} else if (obj instanceof RemoteMethodControl) {
133 	    otherServerProxy = (RemoteMethodControl) obj;
134 	} else {
135 	    return false;
136 	}
137 	MethodConstraints mc = otherServerProxy.getConstraints();
138 	TrustEquivalence trusted =
139 	    (TrustEquivalence) serverProxy.setConstraints(mc);
140 	return trusted.checkTrustEquivalence(otherServerProxy);
141     }
142 
143     /** Require server proxy to implement TrustEquivalence. */
144     private void readObject(ObjectInputStream in)
145 	throws IOException, ClassNotFoundException
146     {
147 	in.defaultReadObject();
148 	if (!(serverProxy instanceof TrustEquivalence)) {
149 	    throw new InvalidObjectException(
150 		"serverProxy must implement TrustEquivalence");
151 	}
152     }
153 }
