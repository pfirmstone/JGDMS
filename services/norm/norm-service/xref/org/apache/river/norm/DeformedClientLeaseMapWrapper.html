18  package org.apache.river.norm;
19  
20  import java.util.HashMap;
21  import java.util.Map;
22  import java.rmi.RemoteException;
23  
24  import net.jini.core.lease.LeaseException;
25  import net.jini.core.lease.LeaseMapException;
26  
27  import org.apache.river.lease.AbstractLeaseMap;
28  import java.util.concurrent.ConcurrentMap;
29  import net.jini.core.lease.Lease;
30  
31  /**
32   * An implementation of LeaseMap that holds exactly one lease.  Used when
33   * we have a deformed lease that we won't be able to batch with anyone
34   * else and we want to call ClientLeaseWrapper.renew() instead of
35   * LeaseMap.renewAll().
36   * <p>
37   * Provides hooks for synchronization and data associated with each
38   * client lease while allowing us to use
39   * <code>LeaseRenewalManager</code>.  Objects of this class are
40   * returned by <code>createLeaseMap</code> calls made on
41   * <code>ClientLeaseWrapper</code> objects that are deformed.
42   *
43   * @author Sun Microsystems, Inc.
44   * @see ClientLeaseWrapper 
45   */
46  class DeformedClientLeaseMapWrapper extends AbstractLeaseMap {
47      private static final long serialVersionUID = 1L;
48  
49      /**
50       * Create a DeformedClientLeaseMapWrapper.
51       * @param lease a Wrapper for the lease that wants to be renewed.
52       *              May be deformed.
53       * @param duration the duration to associate with lease
54       */
55      DeformedClientLeaseMapWrapper(ClientLeaseWrapper lease, long duration) {
56  	super(new HashMap<Lease, Long>(1), lease, duration);	
57      }
58  
59      // inherit javadoc
60      public void cancelAll() {
61  	throw new UnsupportedOperationException(
62  	     "ClientLeaseMapWrapper.cancelAll: " + 
63  	     "LRS should not being canceling client leases");
64      }
65  
66      // inherit javadoc
67      public void renewAll() throws LeaseMapException, RemoteException {
68  	if (map instanceof ConcurrentMap){
69              renewAl();
70          } else {
71              synchronized (mapLock){
72                  renewAl();
73              }
74          }
75      }
76      
77      private void renewAl() throws LeaseMapException, RemoteException {
78  	Map.Entry entry = (Map.Entry) map.entrySet().iterator().next();
79          ClientLeaseWrapper l = 
80  	    (ClientLeaseWrapper) entry.getKey();
81  	long d = ((Long) entry.getValue()).longValue();
82  	try {
83  	    l.renew(d);
84  	} catch (LeaseException e) {
85  	    final Map<Lease, Throwable> m = new HashMap<Lease, Throwable>(1);
86  	    m.put(l, e);
87  	    throw new LeaseMapException(e.getMessage(), m); 
88  	} 
89      }
90  
91      // inherit javadoc
92      public boolean canContainKey(Object key) {
93  	// DeformedClientLeaseMapWrapper can only be created with exactly one
94  	// lease, and currently they can only contain one lease, so
95  	// return false unless the lease passed in is the one we already
96  	// have.
97  	return map.containsKey(key);
98      }
99  }
