18  package org.apache.river.norm;
19  
20  import org.apache.river.thread.InterruptedStatusThread;
21  
22  import java.util.logging.Level;
23  import java.util.logging.Logger;
24  
25  import net.jini.core.lease.Lease;
26  import net.jini.core.lease.LeaseDeniedException;
27  import net.jini.lease.DesiredExpirationListener;
28  import net.jini.lease.LeaseRenewalEvent;
29  
30  /**
31   * Object that transfers events from the Lease Renewal Manager to the
32   * rest of Norm Server.  Isolates the renewal manager from having to
33   * block on the snapshot locks.
34   *
35   * @author Sun Microsystems, Inc.
36   */
37  class LRMEventListener extends InterruptedStatusThread 
38                         implements DesiredExpirationListener 
39  {
40      /** Logger for logging messages for this class */
41      private static final Logger logger = Logger.getLogger("org.apache.river.norm");
42  
43      /** Ref to the main server object which has all the top level methods */
44      volatile private NormServerBaseImpl server;
45  
46      /**
47       * Queue we use to decouple the reception of events from the lease
48       * renewal manager from the scheduling of the sending of remote
49       * events and modifying our internal tables (which both require
50       * obtaining serious locks).
51       */
52      final private Queue queue = new Queue();
53  
54      /** Any events that hold this object are ignored */
55      final static LeaseDeniedException EXPIRED_SET_EXCEPTION = 
56  	new LeaseDeniedException("Set Expired");
57  	
58      /**
59       * Simple constructor
60       */
61      LRMEventListener() {
62          super("LRM Event Listener");
63  	setDaemon(false);
64      }
65      
66      /**
67       * Set only once after construction.
68       * @param server Object that will make the actual internal updates and
69       * schedule the sending of remote events
70       */
71      void setServer(NormServerBaseImpl server){
72          synchronized (this){
73              if (this.server == null) this.server = server;
74          }
75      }
76  
77      //////////////////////////////////////////////////
78      // Methods required by the LeaseListener interface 
79  
80      // Inherit java doc from super type
81      public void notify(LeaseRenewalEvent e) {	
82  	// Drop if the exception field is == to EXPIRED_SET_EXCEPTION, this
83  	// implies that lease could not be renewed because the wrapper 
84  	// has determined that the set has expired.
85  	if (e.getException() == EXPIRED_SET_EXCEPTION) 
86  	    return;
87  
88  	// Paranoia, check to make sure that lease is one of wrapped
89  	// client lease...if it's not, ignore the event
90  	final Lease l = e.getLease();
91  	if (l instanceof ClientLeaseWrapper) {
92  	    final ClientLeaseWrapper clw = (ClientLeaseWrapper) l;
93  	    queue.enqueue(new Discriminator(clw, true));
94  	}
95      }
96  
97      //////////////////////////////////////////////////////////////
98      // Methods required by the DesiredExpirationListener interface
99  	
100     // Inherit java doc from super type
101     public void expirationReached(LeaseRenewalEvent e) {
102 	// Paranoia, check to make sure that lease is one of wrapped
103 	// client lease...if it's not, ignore the event
104 	final Lease l = e.getLease();
105 	if (l instanceof ClientLeaseWrapper) {
106 	    final ClientLeaseWrapper clw = (ClientLeaseWrapper) l;
107 	    queue.enqueue(new Discriminator(clw, false)) ;
108 	}
109     }
110 
111     public void run() {
112 	// Loop taking items off the queue and pass them to the server
113 	while (!hasBeenInterrupted()) {
114 	    try {
115 		final Discriminator d = (Discriminator) queue.dequeue();
116 
117 		if (d.isFailure) {
118 		    server.renewalFailure(d.clw);
119 		} else {
120 		    server.desiredExpirationReached(d.clw);
121 		}
122 
123 	    } catch (InterruptedException e) {
124 		// Someone wants this thread dead -- just return
125 		return;
126 	    } catch (RuntimeException e) {
127 		logger.log(Level.INFO,
128 			   "Exception in LRMEventListener Notifier while " +
129 			   "processing an event from the LRM -- " +
130 			   "attempting to continue",
131 			   e);
132 
133 	    } catch (Error e) {
134 		logger.log(Level.INFO,
135 			   "Exception in LRMEventListener Notifier while " +
136 			   "processing an event from the LRM -- " +
137 			   "attempting to continue",
138 			   e);
139 	    }
140 	}
141     }
142 
143     /**
144      * Trivial container class to tell us if we are processing the given
145      * wrapper because of a failure event or a desired expiration
146      * reached event.
147      */
148     static private class Discriminator {
149 	/** true if this wrapper is associated with a renewal failure event */
150 	final private boolean isFailure;
151 
152 	/** The wrapped leases associated with the event */
153 	final private ClientLeaseWrapper clw;
154 
155 	private Discriminator(ClientLeaseWrapper clw, boolean isFailure) {
156 	    this.isFailure = isFailure;
157 	    this.clw = clw;
158 	}
159     }
160 }
