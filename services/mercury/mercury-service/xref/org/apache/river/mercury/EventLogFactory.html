37  /**
38   * This class serves as a factory for generating <tt>EventLogIterator</tt>
39   * objects. The iterator objects are cached so that subsequent calls 
40   * for the same iterator return the same object.
41   *
42   * @author Sun Microsystems, Inc.
43   *
44   * @since 1.1
45   */
46  class EventLogFactory {
47  
48      /** <tt>Map</tt> that contains references to generated iterators */
49      private HashMap iterators = new HashMap();
50  
51      /** 
52       * Method to return the iterator object for the designated 
53       * <tt>Uuid</tt>. The <tt>File</tt> argument determines where the
54       * persistence store will be maintained and is only used for the
55       * first instance of the iterator. (Note that upon recovery
56       * from a restart/crash the <tt>Uuid</tt> will already exist so the
57       * <tt>logPath</tt> arg will not be used.)
58       * Subsequent calls for the same <tt>Uuid</tt> will return the 
59       * original object.
60       */
61      public EventLogIterator iterator(Uuid uuid, File logPath) {
62  
63          // Try to get reference from cache
64          EventLogIteratorImpl eli = (EventLogIteratorImpl)iterators.get(uuid);
65          if (eli == null) { // doesn't exist, so create one
66              eli = new EventLogIteratorImpl(uuid, logPath);
67  	    try { 
68  		eli.init(); // initialize the iterator
69  	    } catch (IOException ioe) {
70  		// ignore ... the next usage of this
71  		// object will throw IOException
72  	    }
73              iterators.put(uuid, eli); // add to cache
74  	}
75              
76  	return eli;
77      }
78      
79      /** 
80       * Method to return the iterator object for the designated 
81       * <tt>Uuid</tt>. 
82       * Subsequent calls for the same <tt>Uuid</tt> will return the 
83       * original object.
84       */
85      public EventLogIterator iterator(Uuid uuid) {
86  
87          // Try to get reference from cache
88          EventLogIteratorImpl eli = (EventLogIteratorImpl)iterators.get(uuid);
89          if (eli == null) { // doesn't exist, so create one
90              eli = new EventLogIteratorImpl(uuid);
91  	    try { 
92  		eli.init(); // initialize the iterator
93  	    } catch (IOException ioe) {
94  		throw new InternalMailboxException(
95                      "Received unexpected IOException from"
96                      + " a non-persistent log", ioe);
97  	    }
98              iterators.put(uuid, eli); // add to cache
99  	}
100             
101 	return eli;
102     }
103 
104     /**
105      * Remove the <TT>EventLogIterator</TT> associated with the given
106      * <TT>Uuid</TT>. This is (presumably) called to flush an existing
107      * <TT>EventLogIterator</TT> whose storage location has changed.
108      * This way, the next call to <TT>iterator</TT> will produce a new
109      * object instead of returning the cached version.
110      */
111     public void remove(Uuid uuid) {
112         iterators.remove(uuid); // remove from cache, if it's there
113     }
114 
115     /**
116      * Private class which implements the <tt>EventLogIterator</tt>
117      * interface. This class delegates to an <tt>EventLog</tt> for most
118      * of its functionality.
119      */
120     private static class EventLogIteratorImpl 
121 	implements EventLogIterator
122     {
123       
124         /** The associated <tt>Uuid</tt> for this iterator */
125         private final Uuid uuid;
126 
127         /** The associated <tt>EventLog</tt> for this iterator */
128         private EventLog log = null; 
129 
130         /** 
131          * Simple constructor that assigns the <tt>Uuid</tt>
132          * field to the appropriate internal field and creates
133          * a  persistent <tt>EventLog</tt> object using the provided
134          * <tt>File</tt> argument.
135          */
136         EventLogIteratorImpl(Uuid id, File logPath) {
137             uuid = id;
138 	    log = new PersistentEventLog(id, logPath);
139         }
140 
141         /** 
142          * Simple constructor that assigns the <tt>Uuid</tt>
143          * field to the appropriate internal field and creates
144 	 * a transient event log.
145 	 */
146         EventLogIteratorImpl(Uuid id) {
147             uuid = id;
148 	    log = new TransientEventLog(id);
149         }
150 
151 	// Inherit documentation from supertype
152         public void init() throws IOException {
153 	    log.init();
154 	}
155 
156         // Inherit documentation from supertype
157         public void add(RemoteEvent evt) throws IOException, 
158 		IllegalArgumentException 
159 	{
160             log.add(evt);
161 	}
162             
163         // Inherit documentation from supertype
164         public boolean hasNext()  throws IOException {
165             return !log.isEmpty();
166         }
167 
168         // Inherit documentation from supertype
169         public RemoteEvent next() throws IOException, 
170 		ClassNotFoundException, NoSuchElementException 
171 	{
172             return log.next();
173         }
174 
175         // Inherit documentation from supertype
176         public void remove() throws IOException, IllegalStateException {
177             log.remove();
178         }
179 
180         // Inherit documentation from supertype
181         public RemoteEventData[] readAhead(int maxEvents) 
182             throws IOException, ClassNotFoundException 
183         {
184             return log.readAhead(maxEvents);
185         }
186 
187         // Inherit documentation from supertype
188         public void moveAhead(Object cookie) throws IOException {
189             log.moveAhead(cookie);
190         }
191         
192         // Inherit documentation from supertype
193         public void destroy() throws IOException {
194             log.close();
195 	    log.delete();
196         }
197     }
198 }
