43  /**
44   * A MailboxProxy is a proxy for the event mailbox service.  
45   * This is the object passed to clients of this service.
46   * It implements the <code>PullEventMailbox</code> and the 
47   * <code>Administrable</code> interfaces.
48   *
49   * @author Sun Microsystems, Inc.
50   *
51   * @since 1.1
52   */
53  @AtomicSerial
54  public class MailboxProxy implements PullEventMailbox,
55      Administrable, Serializable, ReferentUuid, ProxyAccessor
56  {
57  
58      private static final long serialVersionUID = 2L;
59  
60      /**
61       * The reference to the event mailbox service implementation
62       *
63       * @serial
64       */
65      final MailboxBackEnd mailbox;
66  
67      /**
68       * The proxy's <code>Uuid</code>
69       *
70       * @serial
71       */
72      final Uuid proxyID;
73  
74      /**
75       * Creates a mailbox proxy, returning an instance
76       * that implements RemoteMethodControl if the server does too.
77       *
78       * @param mailbox the server proxy
79       * @param id the ID of the server
80       */
81      public static MailboxProxy create(MailboxBackEnd mailbox, Uuid id) {
82          if (mailbox == null || id == null) {
83              throw new IllegalArgumentException("Cannot accept null arguments");
84          }
85          if (mailbox instanceof RemoteMethodControl) {
86              return new ConstrainableMailboxProxy(mailbox, id, null);
87          } else {
88              return new MailboxProxy(mailbox, id);
89          }
90      }
91  
92      /** Convenience constructor. */
93      private MailboxProxy(MailboxBackEnd mailbox, Uuid proxyID) {
94  	this.mailbox = mailbox;
95  	this.proxyID = proxyID;
96      }
97  
98      MailboxProxy(GetArg arg) throws IOException {
99  	this(check(arg),(Uuid) arg.get("proxyID", null));
100     }
101     
102     private static MailboxBackEnd check(GetArg arg) throws IOException {
103 	MailboxBackEnd mailbox = (MailboxBackEnd) arg.get("mailbox", null);
104 	Uuid proxyID = (Uuid) arg.get("proxyID", null);
105 	/* Verify server */
106         if(mailbox == null) {
107             throw new InvalidObjectException("MailboxProxy.readObject "
108                                              +"failure - mailbox "
109                                              +"field is null");
110         }//endif
111         /* Verify proxyID */
112         if(proxyID == null) {
113             throw new InvalidObjectException("MailboxProxy.proxyID "
114                                              +"failure - proxyID "
115                                              +"field is null");
116         }//endif
117 	return mailbox;
118     }
119 
120     // inherit javadoc from parent
121     public MailboxRegistration register(long duration) 
122         throws RemoteException, LeaseDeniedException {
123         // Check for a bad argument
124         // Note that -1 (i.e. Lease.ANY) is a valid request
125         if (duration < 1 && duration != Lease.ANY)
126             throw new IllegalArgumentException(
127                 "Duration values must be positive");
128         return mailbox.register(duration);
129     }
130 
131     // inherit javadoc from parent
132     public MailboxPullRegistration pullRegister(long duration) 
133         throws RemoteException, LeaseDeniedException {
134         // Check for a bad argument
135         // Note that -1 (i.e. Lease.ANY) is a valid request
136         if (duration < 1 && duration != Lease.ANY)
137             throw new IllegalArgumentException(
138                 "Duration values must be positive");
139         return mailbox.pullRegister(duration);
140     }

142     // inherit javadoc from parent
143     public Object getAdmin() throws RemoteException {
144         return mailbox.getAdmin();
145     }

147     /* From net.jini.id.ReferentUuid */
148     /**
149      * Returns the universally unique identifier that has been assigned to the
150      * resource this proxy represents.
151      *
152      * @return the instance of <code>Uuid</code> that is associated with the
153      *         resource this proxy represents. This method will not return
154      *         <code>null</code>.
155      *
156      * @see net.jini.id.ReferentUuid
157      */
158     public Uuid getReferentUuid() {
159         return proxyID;
160     }
161 
162     /** Proxies for servers with the same proxyID have the same hash code. */
163     public int hashCode() {
164 	return proxyID.hashCode();
165     }
166 
167     /** 
168      * Proxies for servers with the same <code>proxyID</code> are 
169      * considered equal. 
170      */
171     public boolean equals(Object o) {
172 	return ReferentUuids.compare(this,o);
173     }
174     
175     /** When an instance of this class is deserialized, this method is
176      *  automatically invoked. This implementation of this method validates
177      *  the state of the deserialized instance.
178      *
179      * @throws InvalidObjectException if the state of the
180      *         deserialized instance of this class is found to be invalid.
181      */
182     private void readObject(ObjectInputStream s)
183                                throws IOException, ClassNotFoundException
184     {
185         s.defaultReadObject();
186         /* Verify server */
187         if(mailbox == null) {
188             throw new InvalidObjectException("MailboxProxy.readObject "
189                                              +"failure - mailbox "
190                                              +"field is null");
191         }//endif
192         /* Verify proxyID */
193         if(proxyID == null) {
194             throw new InvalidObjectException("MailboxProxy.proxyID "
195                                              +"failure - proxyID "
196                                              +"field is null");
197         }//endif
198     }//end readObject
199 
200     /** During deserialization of an instance of this class, if it is found
201      *  that the stream contains no data, this method is automatically
202      *  invoked. Because it is expected that the stream should always
203      *  contain data, this implementation of this method simply declares
204      *  that something must be wrong.
205      *
206      * @throws InvalidObjectException to indicate that there
207      *         was no data in the stream during deserialization of an
208      *         instance of this class; declaring that something is wrong.
209      */
210     private void readObjectNoData() throws ObjectStreamException {
211         throw new InvalidObjectException("no data found when attempting to "
212                                          +"deserialize MailboxProxy instance");
213     }//end readObjectNoData

216     public Object getProxy() {
217 	return mailbox;

221     /** A subclass of MailboxProxy that implements RemoteMethodControl. */
222     @AtomicSerial
223     final static class ConstrainableMailboxProxy extends MailboxProxy
224         implements RemoteMethodControl
225     {
226         private static final long serialVersionUID = 1L;
227 
228         /** Creates an instance of this class. */
229         private ConstrainableMailboxProxy(MailboxBackEnd mailbox, Uuid uuid,
230             MethodConstraints methodConstraints)
231         {
232             super(constrainServer(mailbox, methodConstraints),
233                   uuid);
234         }
235 
236 	ConstrainableMailboxProxy(GetArg arg) throws IOException {
237 	    super(check(arg));
238 	}
239 	
240 	private static GetArg check(GetArg arg) throws IOException{
241 	    MailboxProxy mp = new MailboxProxy(arg);
242 	    // Verify that the server implements RemoteMethodControl
243             if( !(mp.mailbox instanceof RemoteMethodControl) ) {
244                 throw new InvalidObjectException(
245 		    "MailboxAdminProxy.readObject failure - mailbox " +
246 		    "does not implement constrainable functionality ");
247             }//endif
248 	    return arg;
249 	}
250 
251        /**
252          * Returns a copy of the server proxy with the specified client
253          * constraints and methods mapping.
254          */
255         private static MailboxBackEnd constrainServer(
256             MailboxBackEnd mailbox,
257             MethodConstraints methodConstraints)
258         {
259             return (MailboxBackEnd)
260                 ((RemoteMethodControl)mailbox).setConstraints(methodConstraints);
261         }
262 
263         /** {@inheritDoc} */
264         public RemoteMethodControl setConstraints(
265             MethodConstraints constraints)
266         {
267             return new ConstrainableMailboxProxy(mailbox, proxyID,
268                 constraints);
269         }
270 
271         /** {@inheritDoc} */
272         public MethodConstraints getConstraints() {
273             return ((RemoteMethodControl) mailbox).getConstraints();
274         }

280          * Returns a proxy trust iterator that is used in
281          * <code>ProxyTrustVerifier</code> to retrieve this object's
282          * trust verifier.
283          */
284         private ProxyTrustIterator getProxyTrustIterator() {
285             return new SingletonProxyTrustIterator(mailbox);
286         }//end getProxyTrustIterator
287 	
288 	/** Performs various functions related to the trust verification
289          *  process for the current instance of this proxy class, as
290          *  detailed in the description for this class.
291          *
292          * @throws <code>InvalidObjectException</code> if any of the
293          *         requirements for trust verification (as detailed in the
294          *         class description) are not satisfied.
295          */
296         private void readObject(ObjectInputStream s)
297                                    throws IOException, ClassNotFoundException
298         {
299 	    /* Note that basic validation of the fields of this class was
300              * already performed in the readObject() method of this class'
301              * super class.
302              */
303             s.defaultReadObject();
304 	    // Verify that the server implements RemoteMethodControl
305             if( !(mailbox instanceof RemoteMethodControl) ) {
306                 throw new InvalidObjectException(
307 		    "MailboxAdminProxy.readObject failure - mailbox " +
308 		    "does not implement constrainable functionality ");
309             }//endif
310         }//end readObject
