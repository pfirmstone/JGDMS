61      // documentation inherited from MulticastRequestEncoder
62      public void encodeMulticastRequest(MulticastRequest request,
63  				       DatagramBufferFactory bufs,
64  				       InvocationConstraints constraints)
65  	throws IOException
66      {
67  	if (request == null || bufs == null) {
68  	    throw new NullPointerException();
69  	}
70  	try {
71  	    X500Constraints cons = X500Constraints.process(constraints, true);
72  
73  	    // REMIND: instead iterate through constraint-designated principals
74  	    X500PrivateCredential[] creds = getPrivateCredentials();
75  	    X500PrivateCredential chosen = null;
76  	    int best = -1;
77  	    SecurityException se = null;
78  	    for (int i = 0; i < creds.length; i++) {
79  		X500PrivateCredential c = creds[i];
80  		X500Principal p = c.getCertificate().getSubjectX500Principal();
81  		int score = cons.checkClientPrincipal(p);
82  		if (score < 0) {
83  		    if (logger.isLoggable(Level.FINEST)) {
84  			logger.log(Level.FINEST,
85  				   "skipping disallowed principal {0}",
86  				   new Object[]{ p });
87  		    }
88  		    continue;
89  		}
90  		try {
91  		    checkAuthenticationPermission(p, "connect");
92  		} catch (SecurityException e) {
93  		    se = e;
94  		    if (logger.isLoggable(Level.FINE)) {
95  			logger.log(Level.FINE,
96  				   "not authorized to use principal {0}",
97  				   new Object[]{ p });
98  		    }
99  		    continue;
100 		}
101 		if (score > best) {
102 		    chosen = c;
103 		    best = score;
104 		}
105 	    }
106 	    if (chosen == null) {
107 		UnsupportedConstraintException uce = 
108 		    new UnsupportedConstraintException(
109 			"unsupported constraints: " + constraints);
110 		if (se != null) {
111 		    // At least one principal was rejected due to lack
112 		    // of authentication permissions
113 		    secureThrow(se, uce);		    
114 		}
115 		throw uce;
116 	    }
117 	    if (logger.isLoggable(Level.FINEST)) {
118 		logger.log(Level.FINEST, "using principal {0}",
119 			   new Object[]{ chosen });
120 	    }
121 	    SigningBufferFactory sbf = new SigningBufferFactory(bufs, chosen);
122 	    Plaintext.encodeMulticastRequest(request, sbf);
123 	    sbf.sign();
124 
125 	} catch (IOException e) {
126 	    throw e;
127 	} catch (SecurityException e) {
128 	    throw e;
129 	} catch (Exception e) {
130 	    throw new DiscoveryProtocolException(null, e);
131 	}
132     }

134     // documentation inherited from MulticastAnnouncementDecoder
135     public MulticastAnnouncement decodeMulticastAnnouncement(
136 					    ByteBuffer buf,
137 					    InvocationConstraints constraints,
138                                             boolean delayConstraintCheck)
139 	throws IOException
140     {
141 	try {
142 	    int len = buf.getInt();
143 	    ByteBuffer data = buf.duplicate();
144 	    data.limit(data.position() + len);
145 	    buf.position(data.limit());
146 
147 	    X500Principal p = new X500Principal(Plaintext.getUtf(buf));
148 	    ByteBuffer signed = (ByteBuffer) data.duplicate().position(0);
149 	    MulticastAnnouncement ma =
150 		Plaintext.decodeMulticastAnnouncement(data);
151 
152             ma = new X500MulticastAnnouncement(ma, constraints, p, 
153 					       buf.duplicate(), signed);
154             if (!delayConstraintCheck) {
155                 ma.checkConstraints();
156             }
157             return ma;
158         } catch (IOException e) {
159             throw e;
160         } catch (SecurityException e) {
161             throw e;
162         } catch (Exception e) {
163             throw new DiscoveryProtocolException(null, e);
164         }
165     }

167     public MulticastAnnouncement decodeMulticastAnnouncement(ByteBuffer buf,
168                                             InvocationConstraints constraints)
169         throws IOException
170     {
171         return decodeMulticastAnnouncement(buf, constraints, false);

174     private class X500MulticastAnnouncement extends MulticastAnnouncement {
175 	private final InvocationConstraints constraints;
176 	private final X500Principal p;
177 	private final ByteBuffer signature;
178 	private final ByteBuffer signed;
179 	private X500MulticastAnnouncement(MulticastAnnouncement plainMA,
180 					  InvocationConstraints constraints,
181 					  X500Principal p,
182 					  ByteBuffer signature,
183 					  ByteBuffer signed)
184 	{
185 	    super(plainMA.getSequenceNumber(),  plainMA.getHost(), 
186 		    plainMA.getPort(), plainMA.getGroups(),
187 		    plainMA.getServiceID());
188 	    this.constraints = constraints;
189 	    this.p = p;
190 	    this.signature = signature;
191 	    this.signed = signed;
192 	}
193 
194 	public void checkConstraints() throws IOException {
195 	    try {
196 		X500Constraints cons = X500Constraints.process(constraints,
197 							       false);
198 		if (cons.checkServerPrincipal(p) < 0) {
199 		    throw new UnsupportedConstraintException(
200 			"principal not allowed: " + p);
201 		}
202 		Certificate cert = getCertificate(p);
203 		if (cert == null) {
204 		    throw new DiscoveryProtocolException(
205 			"unknown principal: " + p);
206 		}
207 		if (logger.isLoggable(Level.FINEST)) {
208 		    logger.log(Level.FINEST, "mapped principal {0} to {1}",
209 			       new Object[]{ p, cert });
210 		}
211 		if (!verify(signed.duplicate(), signature.duplicate(), 
212 			    cert.getPublicKey()))
213 		{
214 		    throw new DiscoveryProtocolException(
215 			"signature verification failed: " + p);
216 		}
217 	    } catch (IOException e) {
218 		throw e;
219 	    } catch (SecurityException e) {
220 		throw e;
221 	    } catch (Exception e) {
222 		throw new DiscoveryProtocolException(null, e);
223 	    }       
224 	}
225     }
