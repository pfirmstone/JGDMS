64      // documentation inherited from MulticastRequestDecoder
65      public MulticastRequest decodeMulticastRequest(
66  					    ByteBuffer buf,
67  					    InvocationConstraints constraints,
68  					    ClientSubjectChecker checker,
69                                              boolean delayConstraintCheck)
70  	throws IOException
71      {
72  	try {
73  	    int len = buf.getInt();
74  	    ByteBuffer data = buf.duplicate();
75  	    data.limit(data.position() + len);
76  	    buf.position(data.limit());
77  
78  	    X500Principal p = new X500Principal(Plaintext.getUtf(buf));
79  	    ByteBuffer signed = (ByteBuffer) data.duplicate().position(0);
80  	    MulticastRequest mr = Plaintext.decodeMulticastRequest(data);
81              
82              mr = new X500MulticastRequest(mr, constraints, p, buf.duplicate(),
83  					  signed, checker);
84              if (!delayConstraintCheck) {
85                  mr.checkConstraints();
86              }
87              return mr;
88  	} catch (IOException e) {
89  	    throw e;
90  	} catch (SecurityException e) {
91  	    throw e;
92  	} catch (Exception e) {
93  	    throw new DiscoveryProtocolException(null, e);
94  	}
95      }

97      // documentation inherited from MulticastAnnouncementEncoder
98      public void encodeMulticastAnnouncement(MulticastAnnouncement announcement,
99  					    DatagramBufferFactory bufs,
100 					    InvocationConstraints constraints)
101 	throws IOException
102     {
103 	if (announcement == null || bufs == null) {
104 	    throw new NullPointerException();
105 	}
106 	try {
107 	    X500Constraints cons = X500Constraints.process(constraints, false);
108 
109 	    // REMIND: instead iterate through constraint-designated principals
110 	    X500PrivateCredential[] creds = getPrivateCredentials();
111 	    X500PrivateCredential chosen = null;
112 	    int best = -1;
113 	    SecurityException se = null;
114 	    for (int i = 0; i < creds.length; i++) {
115 		X500PrivateCredential c = creds[i];
116 		X500Principal p = c.getCertificate().getSubjectX500Principal();
117 		int score = cons.checkServerPrincipal(p);
118 		if (score < 0) {
119 		    if (logger.isLoggable(Level.FINEST)) {
120 			logger.log(Level.FINEST,
121 				   "skipping disallowed principal {0}",
122 				   new Object[]{ p });
123 		    }
124 		    continue;
125 		}
126 		try {
127 		    checkAuthenticationPermission(p, "accept");
128 		} catch (SecurityException e) {
129 		    se = e;
130 		    if (logger.isLoggable(Level.FINE)) {
131 			logger.log(Level.FINE,
132 				   "not authorized to use principal {0}",
133 				   new Object[]{ p });
134 		    }
135 		    continue;
136 		}
137 		if (score > best) {
138 		    chosen = c;
139 		    best = score;
140 		}
141 	    }
142 	    if (chosen == null) {
143 		UnsupportedConstraintException uce = 
144 		    new UnsupportedConstraintException(
145 			"unsupported constraints: " + constraints);
146 		if (se != null) {
147 		    // At least one principal was rejected due to lack
148 		    // of authentication permissions.
149 		    secureThrow(se, uce);		    
150 		}
151 		throw uce;
152 	    }
153 	    if (logger.isLoggable(Level.FINEST)) {
154 		logger.log(Level.FINEST, "using principal {0}",
155 			   new Object[]{ chosen });
156 	    }
157 	    SigningBufferFactory sbf = new SigningBufferFactory(bufs, chosen);
158 	    Plaintext.encodeMulticastAnnouncement(announcement, sbf);
159 	    sbf.sign();
160 
161 	} catch (IOException e) {
162 	    throw e;
163 	} catch (SecurityException e) {
164 	    throw e;
165 	} catch (Exception e) {
166 	    throw new DiscoveryProtocolException(null, e);
167 	}
168     }

170     public MulticastRequest decodeMulticastRequest(ByteBuffer buf,
171                                             InvocationConstraints constraints,
172                                             ClientSubjectChecker checker)
173         throws IOException
174     {
175         return decodeMulticastRequest(buf, constraints, checker, false);

178     private class X500MulticastRequest extends MulticastRequest {
179 	private final InvocationConstraints constraints;
180 	private final X500Principal p;
181 	private final ByteBuffer signature;
182 	private final ByteBuffer signed;
183 	private final ClientSubjectChecker checker;
184 	private X500MulticastRequest(MulticastRequest plainMR,
185 				    InvocationConstraints constraints,
186 				    X500Principal p,
187 				    ByteBuffer signature,
188 				    ByteBuffer signed,
189 				    ClientSubjectChecker checker)
190 	{
191 	    super(plainMR.getHost(), plainMR.getPort(), plainMR.getGroups(), 
192 		    plainMR.getServiceIDs());
193 	    this.constraints = constraints;
194 	    this.p = p;
195 	    this.signature = signature;
196 	    this.signed = signed;
197 	    this.checker = checker;
198 	}
199 
200 	public void checkConstraints() throws IOException {
201 	    try {
202 		X500Constraints cons = X500Constraints.process(constraints,
203 							       true);
204 		if (cons.checkClientPrincipal(p) < 0) {
205 		    throw new UnsupportedConstraintException(
206 			"principal not allowed: " + p);
207 		}
208 		Certificate cert = getCertificate(p);
209 		if (cert == null) {
210 		    throw new DiscoveryProtocolException(
211 			"unknown principal: " + p);
212 		}
213 		if (logger.isLoggable(Level.FINEST)) {
214 		    logger.log(Level.FINEST, "mapped principal {0} to {1}",
215 			       new Object[]{ p, cert });
216 		}
217 		if (checker != null) {
218 		    checker.checkClientSubject(new Subject(
219 			true,
220 			Collections.singleton(p),
221 			Collections.singleton(cert),
222 			Collections.EMPTY_SET));
223 		}
224 		if (!verify(signed.duplicate(), signature.duplicate(), 
225 					cert.getPublicKey())) {
226 		    throw new DiscoveryProtocolException(
227 			"signature verification failed: " + p);
228 		}
229 	    } catch (IOException e) {
230 		throw e;
231 	    } catch (SecurityException e) {
232 		throw e;
233 	    } catch (Exception e) {
234 		throw new DiscoveryProtocolException(null, e);
235 	    }
236 	}
237     }
