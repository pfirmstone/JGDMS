18  package org.apache.river.discovery.https;
19  
20  import java.io.IOException;
21  import java.net.DatagramPacket;
22  import java.net.Socket;
23  import java.nio.ByteBuffer;
24  import java.util.Collection;
25  import net.jini.core.constraint.InvocationConstraints;
26  import org.apache.river.discovery.ClientSubjectChecker;
27  import org.apache.river.discovery.Discovery;
28  import org.apache.river.discovery.EncodeIterator;
29  import org.apache.river.discovery.MulticastAnnouncement;
30  import org.apache.river.discovery.MulticastRequest;
31  import org.apache.river.discovery.UnicastResponse;
32  import org.apache.river.discovery.internal.UnicastClient;
33  import org.apache.river.discovery.internal.UnicastServer;
34  import aQute.bnd.annotation.headers.RequireCapability;
35  import aQute.bnd.annotation.headers.ProvideCapability;
36  
37  /**
38   * This exists as a separate https protocol because the jini discovery protocol
39   * cannot perform the required handshake negotiation to select a provider 
40   * as specified in DJ.2.6.6 and DJ.2.6.7 through firewalls and https proxy servers.
41   * 
42   * As such this is not an implementation of the
43   * jini discovery protocol.  It is simply https unicast discovery.
44   * 
45   * There are no providers of https unicast discovery as there is with
46   * Discovery v2.
47   * 
48   * It can be used by passing a https://host:port string to LookupLocator's 
49   * constructor.
50   * 
51   * @author Peter Firmstone
52   */
53  @RequireCapability(
54  	ns="osgi.extender",
55  	filter="(osgi.extender=osgi.serviceloader.registrar)")
56  @ProvideCapability(
57  	ns="osgi.serviceloader",
58  	name="org.apache.river.discovery.Discovery")
59  public class DiscoveryUnicastHTTPS extends Discovery {
60       
61      private final UnicastServer server = new Server();
62      private final UnicastClient client = new Client();
63  
64      /**
65       * Unsupported.
66       * 
67       * @throws UnsupportedOperationException
68       */
69      @Override
70      public EncodeIterator encodeMulticastRequest(
71              MulticastRequest request,
72              int maxPacketSize,
73              InvocationConstraints constraints) 
74      {
75          throw new UnsupportedOperationException("Not supported.");
76      }
77  
78      /**
79       * Unsupported.
80       * 
81       * @throws UnsupportedOperationException
82       */
83      @Override
84      public MulticastRequest decodeMulticastRequest(
85              DatagramPacket packet,
86              InvocationConstraints constraints,
87              ClientSubjectChecker checker) throws IOException 
88      {
89          throw new UnsupportedOperationException("Not supported.");
90      }
91  
92      /**
93       * Unsupported.
94       * 
95       * @throws UnsupportedOperationException
96       */
97      @Override
98      public EncodeIterator encodeMulticastAnnouncement(
99              MulticastAnnouncement announcement,
100             int maxPacketSize,
101             InvocationConstraints constraints) 
102     {
103         throw new UnsupportedOperationException("Not supported.");
104     }
105 
106     /**
107      * Unsupported.
108      * 
109      * @throws UnsupportedOperationException
110      */
111     @Override
112     public MulticastAnnouncement decodeMulticastAnnouncement(
113             DatagramPacket packet,
114             InvocationConstraints constraints) throws IOException 
115     {
116         throw new UnsupportedOperationException("Not supported.");
117     }
118 
119     @Override
120     public UnicastResponse doUnicastDiscovery(
121             Socket socket,
122             InvocationConstraints constraints,
123             ClassLoader defaultLoader,
124             ClassLoader verifierLoader,
125             Collection context) throws IOException, ClassNotFoundException 
126     {
127         ByteBuffer sent = ByteBuffer.allocate(256);
128         ByteBuffer received = ByteBuffer.allocate(256);
129         return client.doUnicastDiscovery(socket, constraints, defaultLoader,
130                 verifierLoader, context, sent, received);
131     }
132 
133     @Override
134     public void handleUnicastDiscovery(
135             UnicastResponse response,
136             Socket socket, 
137             InvocationConstraints constraints,
138             ClientSubjectChecker checker,
139             Collection context) throws IOException 
140     {
141         ByteBuffer sent = ByteBuffer.allocate(256);
142         ByteBuffer received = ByteBuffer.allocate(256);
143         server.handleUnicastDiscovery(response, socket, constraints, checker,
144                 context, received, sent);
145     }
146     
147     public String toString(){
148 	return "UnicastHTTPS";
149     }
150     
151 }
