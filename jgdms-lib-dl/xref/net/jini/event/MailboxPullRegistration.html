18  package net.jini.event;
19  
20  import java.rmi.RemoteException;
21  import java.util.Collection;
22  
23  import net.jini.core.event.RemoteEvent;
24  import net.jini.core.event.RemoteEventListener;
25  import net.jini.core.lease.Lease;
26  
27  
28  /**
29   * The <code>MailboxPullRegistration</code> defines the interface through which
30   * a client manages its registration and its notification processing. 
31   * Event mailbox clients use this interface to:
32   * <UL>
33   * <LI>Manage the {@link Lease} for this registration.
34   * <LI>Obtain a {@link RemoteEventListener} reference that
35   *     can be registered with event generators. This listener will
36   *     then store any received events for this registration.
37   * <LI>Synchronously or asynchronously collect any event notifications 
38   *     stored by this particular registration.
39   * </UL>
40   *
41   * @author Sun Microsystems, Inc.
42   *
43   * @since 2.1
44   */
45  public interface MailboxPullRegistration extends MailboxRegistration {
46  
47      /**
48       * Retrieves stored notifications, if any. 
49       *
50       * @return A <code>RemoteEventIterator</code> that can be used to retrieve
51       * event notifications from the mailbox service.
52       *
53       * @throws java.rmi.RemoteException if there is
54       *  a communication failure between the client and the service.
55       */
56      public RemoteEventIterator getRemoteEvents()
57  		    throws RemoteException;
58      /**
59       * Adds the provided collection of unknown events to this registration.
60       * The mailbox will then propagate an 
61       * {@link net.jini.core.event.UnknownEventException}
62       * back to any event generator that attempts to deliver an event with an 
63       * identifier-source combination held in a registration's unknown 
64       * exception list. 
65       *
66       * @param unknownEvents  A <code>Collection</code> of unknown events 
67       *         to be associated with this registration.
68       *
69       * @throws java.rmi.RemoteException if there is
70       *  a communication failure between the client and the service.
71       */
72      public void addUnknownEvents(Collection<? extends RemoteEvent> unknownEvents)
73  		    throws RemoteException;
74  }
