18  package net.jini.lookup.entry;
19  
20  import net.jini.entry.AbstractEntry;
21  import java.util.Set;
22  import java.rmi.MarshalledObject;
23  import java.io.IOException;
24  import java.security.AccessController;
25  import java.security.PrivilegedAction;
26  import org.apache.river.lookup.util.ConsistentSet;
27  import net.jini.io.MarshalledInstance;
28  
29  /**
30   * <CODE>Entry</CODE> that enables a UI for a service to be associated
31   * with the service in the attribute sets of the service item.
32   * <CODE>UIDescriptor</CODE> holds a marshalled UI factory object, as
33   * well as a role string, a sub-role string, and set of attributes that describe the
34   * UI generated by the factory.
35   *
36   * @author Bill Venners
37   */
38  public class UIDescriptor extends AbstractEntry {
39  
40      private static final long serialVersionUID = -4862086482826820670L;
41  
42      /**
43       * Gives the fully qualified name of the interface that represents
44       * the role of the UI generated by the marshalled UI factory.
45       * If the client program unmarshals the UI factory and invokes a factory method, the
46       * UI returned by the factory method must implement the role the interface specified by
47       * <CODE>role</CODE>.
48       *
49       * <P>
50       * For a client program to be able to use a UI, the client has to have prior knowledge
51       * of the UI semantics, which is defined by the UI's role type. Thus, for a client
52       * to be able to use a UI, the client must understand the semantics
53       * of the type whose fully qualified name is given in the <CODE>String</CODE>
54       * referenced from the <CODE>role</CODE> field of that UI's <CODE>UIDescriptor</CODE>.
55       *
56       * <P>
57       * For example, two role types that are defined in the <CODE>net.jini.lookup.ui</CODE>
58       * package by the Jini Service UI Specification are <CODE>MainUI</CODE>, for a main UI
59       * to a Jini service, and <CODE>AdminUI</CODE>, for an administration UI. Other role types
60       * may be defined by the Jini Service UI Specification and by individual Jini service API
61       * specifications.
62       *
63       * <P>
64       * As the strings referenced from the <CODE>role</CODE> field are Java type names, they
65       * are intended to be manipulated by client programs only. They should not be shown to a user.
66       */
67      public String role;
68  
69      /**
70       * A <code>String</code> to facilitate searching whose value represents the main UI toolkit (for example
71       * Swing or AWT) which the produced UI makes use of. The value to which this field should
72       * be set is defined by the semantics of the factory type. (This field is intended
73       * to facilitate searches. For example, a client can search for all blender services that have Swing
74       * MainUI's.)
75       */
76      public String toolkit;
77  
78      /**
79       * An unmodifiable set of objects that describe the UI generated by the marshalled UI factory.
80       */
81      public Set attributes;
82  
83      /**
84       * The <CODE>get()</CODE> method of this <CODE>MarshalledObject</CODE>
85       * must return an object that implements one or more UI factory interfaces. The actual
86       * UI factory type or types implemented by the returned object
87       * must be described by a <CODE>UIFactoryTypes</CODE> attribute placed in
88       * the attributes set of this <CODE>UIDescriptor</CODE>.
89       */
90      public MarshalledObject factory;
91  
92      /**
93       * Constructs a <CODE>UIDescriptor</CODE> with all fields set to <CODE>null</CODE>.
94       */
95      public UIDescriptor() {
96      }
97  
98      /**
99       * Constructs a <CODE>UIDescriptor</CODE> with the fields set to passed values.
100      * This constructor copies the contents of the passed attributes <code>Set</code> into a
101      * serializable read-only <code>Set</code> that has a
102      * consistent serialized form across all VMs, and initializes the <code>attributes</code>
103      * field with the consistent <code>Set</code>.
104      *
105      * @param role the role
106      * @param toolkit the toolkit
107      * @param attributes the attributes
108      * @param factory the factory
109      */
110     public UIDescriptor(String role, String toolkit, Set attributes,
111         MarshalledObject factory) {
112 
113         this.role = role;
114         this.toolkit = toolkit;
115         if (attributes != null) {
116             this.attributes = new ConsistentSet(attributes);
117         }
118         else {
119             this.attributes = null;
120         }
121         this.factory = factory;
122     }
123 
124     /**
125      * A convenience method for unmarshalling the UI factory stored
126      * in the <CODE>MarshalledObject</CODE> referenced from the
127      * <CODE>factory</CODE> field. This method saves a reference
128      * to the current context class loader, sets the context class loader
129      * to the class loader passed as <CODE>parentLoader</CODE>, invokes
130      * <CODE>get()</CODE> on the marshalled object, then resets the
131      * context class loader to the saved reference before returning
132      * the object produced by <CODE>get()</CODE>.
133      *
134      * <P>The class loader
135      * passed in <CODE>parentLoader</CODE> should be able to load classes
136      * needed when the UI interacts with the <CODE>roleObject</CODE> passed as the first
137      * parameter to the factory method. For example, if the <CODE>roleObject</CODE> is
138      * the service item (as it is for the <CODE>MainUI</CODE> and <CODE>AdminUI</CODE>
139      * roles), the class loader passed in <CODE>parentLoader</CODE> could be
140      * the class loader with which the service proxy object referenced
141      * from the service item's <CODE>service</CODE> field was loaded.
142      * For example:
143      *
144      * <PRE>
145      * Object uiFactory = uiDescriptor.getUIFactory(
146      *     serviceItem.service.getClass().getClassLoader());
147      * </PRE>
148      *
149      * @param parentLoader the parent ClassLoader 
150      * @throws java.io.IOException if an IOException occurs
151      * @throws java.lang.ClassNotFoundException if class not found.
152      * @throws NullPointerException if <CODE>parentLoader</CODE> is null.
153      * @return the unmarshalled UI factory
154      */
155     public final Object getUIFactory(final ClassLoader parentLoader)
156         throws IOException, ClassNotFoundException {
157 
158         if (parentLoader == null) {
159             throw new NullPointerException();
160         }
161 
162         Object uiFactory;
163         final Thread currentThread = Thread.currentThread();
164 
165         // First, get a reference to the current context class loader,
166         // so it can be restored after the unmarshalling
167         ClassLoader original = currentThread.getContextClassLoader();
168         try {
169             currentThread.setContextClassLoader(parentLoader);
170             uiFactory = new MarshalledInstance(factory).get(false);
171         }
172         finally {
173             currentThread.setContextClassLoader(original);
174         }
175         return uiFactory;
176     }
177 }
