18  package net.jini.lookup.entry;
19  
20  import java.beans.Beans;
21  import java.io.IOException;
22  import net.jini.core.entry.Entry;
23  
24  /**
25   * Utility class for handling JavaBeans(TM) components that relate to Entry 
26   * classes in the Jini lookup service.
27   * 
28   * @author Sun Microsystems, Inc.
29   */
30  public class EntryBeans {
31      /**
32       * Create a bean of the appropriate type for the given Entry
33       * object, and link it to the object for immediate manipulation.
34       *
35       * @param ent the Entry for which to create and link a bean
36       * @return a new bean of the right type, linked to the given Entry
37       * @exception ClassNotFoundException no bean class of the
38       * appropriate type could be found
39       * @exception ClassCastException the bean class does not implement
40       * the EntryBean interface
41       * @exception IOException the JavaBeans component could not be instantiated
42       */
43      public static EntryBean createBean(Entry ent)
44  	throws ClassNotFoundException, IOException
45      {
46  	String beanClass = ent.getClass().getName() + "Bean";
47  	Object obj = null;
48  	try {
49  	    obj = Beans.instantiate(ent.getClass().getClassLoader(),
50  				    beanClass);
51  	} catch (ClassNotFoundException e) {
52  	    // Ignore any ClassNotFoundException thrown here.
53  	} // Allow the instantiate method to throw an IOException.	
54  	if (obj == null)
55  	    obj = Beans.instantiate(null, beanClass);
56  	EntryBean entBean = (EntryBean)Beans.getInstanceOf(obj,
57  							   EntryBean.class);
58  	entBean.makeLink(ent);
59  	return entBean;
60      }
61  
62      /**
63       * Return the class of JavaBeans component that corresponds to a particular 
64       * Entry class.  The class passed in as argument must implement the
65       * Entry interface.
66       *
67       * @param c the class for which a JavaBeans component should be found
68       * @return corresponding JavaBeans component class
69       * @exception ClassNotFoundException no such class could be found
70       * @exception ClassCastException the class does not implement
71       * the EntryBean interface
72       * @see EntryBean
73       */
74      public static Class getBeanClass(Class c) throws ClassNotFoundException {
75  	if (!Entry.class.isAssignableFrom(c))
76  	    throw new IllegalArgumentException("class does not implement net.jini.core.entry.Entry");
77  	String beanClassName = c.getName() + "Bean";
78  	Class beanClass = null;
79  	try {
80  	    beanClass = c.getClassLoader().loadClass(beanClassName);
81  	} catch (ClassNotFoundException e) {
82  	    // ignore
83  	}
84  	if (beanClass == null)
85  	    beanClass = Class.forName(beanClassName);
86  	if (!EntryBean.class.isAssignableFrom(beanClass))
87  	    throw new ClassCastException("JavaBeans component class does not implement EntryBean interface");
88  	return beanClass;
89      }
90  }
