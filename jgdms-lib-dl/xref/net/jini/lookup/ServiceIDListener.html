18  package net.jini.lookup;
19  
20  /**
21   * Interface used by a service that does not yet have a service ID, for
22   * callback from the JoinManager when a service ID is assigned.
23   * 
24   * @author Sun Microsystems, Inc.
25   *
26   * @see JoinManager
27   */
28  public interface ServiceIDListener extends java.util.EventListener {
29      /** 
30       * Called when the JoinManager gets a valid ServiceID from a lookup 
31       * service.
32       *
33       *@param serviceID  the service ID assigned by the lookup service.
34       */
35      void serviceIDNotify(net.jini.core.lookup.ServiceID serviceID);
36  }
