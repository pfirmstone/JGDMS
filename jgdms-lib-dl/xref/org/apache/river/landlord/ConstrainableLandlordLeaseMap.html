18  package org.apache.river.landlord;
19  
20  import net.jini.core.lease.Lease;
21  import net.jini.core.constraint.MethodConstraints;
22  import net.jini.core.constraint.RemoteMethodControl;
23  import net.jini.id.Uuid;
24  import net.jini.id.ReferentUuid;
25  import org.apache.river.proxy.ConstrainableProxyUtil;
26  
27  /**
28   * Constrainable sub-class of <code>LandlordLeaseMap</code>.
29   * @author Sun Microsystems, Inc.
30   * @since 2.0
31   */
32  final public class ConstrainableLandlordLeaseMap extends LandlordLeaseMap {
33      /** 
34       * Create a new <code>ConstrainableLandlordLeaseMap</code>.
35       * @param landlord Reference to the entity that created 
36       *                 <code>lease</code>.  Assumes that any
37       *                 necessary method constraints have been
38       *                 attached.
39       * @param landlordUuid a universally unique id that has been
40       *                 assigned to the server granting of the lease.
41       *                 Ideally the <code>Uuid</code> {@link
42       *                 ReferentUuid#getReferentUuid landlord.getUuid} would
43       *                 return if <code>landlord</code> implemented
44       *                 {@link ReferentUuid}. Used to determine when
45       *                 leases can be added to this map.
46       * @param lease    First lease to be placed in the map.  It is
47       *                 assumed that <code>canContainKey(lease)</code>
48       *                 would be <code>true</code>.  Must work with the 
49       *                 landlord protocol.
50       * @param duration The duration the lease should be renewed for if 
51       *                 <code>renewAll</code> is called.
52       * @throws ClassCastException if <code>landlord</code>
53       *                 does not implement <code>RemoteMethodControl</code>.
54       * @throws NullPointerException if landlord is <code>null</code>.
55       */
56      ConstrainableLandlordLeaseMap(Landlord landlord, Uuid landlordUuid, 
57  				  LandlordLease lease, long duration) 
58      {
59  	super(chLandlord(landlord), landlordUuid, lease, duration);
60  	
61      }
62      
63      private static Landlord chLandlord(Landlord landlord) throws ClassCastException {
64          if (!(landlord instanceof RemoteMethodControl))
65  	    throw new ClassCastException("landlord must implement " +
66  					 "RemoteMethodControl");
67          return landlord;
68      }
69  
70  
71      // doc inherited from super
72      public boolean canContainKey(Object key) {
73  	if (!super.canContainKey(key))
74  	    return false;	
75  
76  	// Same landlord, check to see if we have comparable constraints.
77  	if (!(key instanceof ConstrainableLandlordLease))
78  	    return false;
79  
80  	// The key's constraints
81  	final MethodConstraints lmc = 
82  	    ((ConstrainableLandlordLease)key).getConstraints();
83  
84  	// Our constraints
85  	final MethodConstraints omc = 
86  	    ((RemoteMethodControl)landlord()).getConstraints();
87  
88  	// Are they equivalent (after applying the map)?
89  	return ConstrainableProxyUtil.equivalentConstraints(
90  	    lmc, omc, ConstrainableLandlordLease.leaseMapMethodMapArray);
91  
92      }
93  }
