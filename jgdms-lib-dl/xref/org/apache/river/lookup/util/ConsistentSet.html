19  package org.apache.river.lookup.util;
20  
21  import java.io.IOException;
22  import java.io.Serializable;
23  import java.util.AbstractSet;
24  import java.util.Arrays;
25  import java.util.Collection;
26  import java.util.HashSet;
27  import java.util.Iterator;
28  import java.util.NoSuchElementException;
29  import java.util.Set;
30  import org.apache.river.api.io.AtomicSerial;
31  import org.apache.river.api.io.AtomicSerial.GetArg;
32  
33  /**
34   * An implementation of the <code>java.util.Set</code> interface that has
35   * a serialized form consistent in all virtual machines. <code>ConsistentSet</code>
36   * instances are unmodifiable. All mutator methods, such as <code>add</code> and
37   * <code>remove</code>, throw <code>UnsupportedOperationException</code>.
38   * This class permits the <code>null</code> element.
39   *
40   * <p>
41   * Although instances of this class are unmodifiable, they are not necessarily
42   * immutable. If a client retrieves a mutable object contained in a
43   * <code>ConsistentSet</code> and mutates that object, the client in effect
44   * mutates the state of the <code>ConsistentSet</code>. In this case, the
45   * serialized form of the <code>ConsistentSet</code> will also most likely
46   * have been mutated. A <code>ConsistentSet</code> that contains only immutable
47   * objects will maintain a consistent serialized form indefinitely. But a 
48   * <code>ConsistentSet</code> that contains mutable objects will maintain a
49   * consistent serialized form only so long as the mutable objects are not
50   * mutated.
51   *
52   * @author Bill Venners
53   */
54  @AtomicSerial
55  public class ConsistentSet<T> extends AbstractSet<T> implements Serializable {
56  
57      private static final long serialVersionUID = -533615203387369436L;
58  
59      /**
60       * @serial An array of the <code>Object</code> elements contained in
61       *     this <code>ConsistentSet</code>.
62       */
63      private final T[] elements;
64  
65      /**
66       * Constructs a new, empty <code>ConsistentSet</code>. All instances
67       * of <code>ConsistentSet</code> are unmodifiable.
68       */
69      @SuppressWarnings("unchecked")
70      public ConsistentSet() {
71          elements = (T[]) new Object[0];
72      }
73  
74      /**
75       * Constructs a new <code>ConsistentSet</code> containing the elements
76       * in the passed collection. All instances of <code>ConsistentSet</code>
77       * are unmodifiable.
78       *
79       * @param init the collection whose elements are to be placed into this set.
80       * @throws NullPointerException if the passed <code>init</code> reference
81       *     is <code>null</code>
82       */
83      @SuppressWarnings("unchecked")
84      public ConsistentSet(Collection<T> init) {
85  	this(elements(init));
86      }
87  
88      public ConsistentSet(GetArg arg) throws IOException {
89  	this(elements(Arrays.asList((T[])arg.get("elements", null))));
90      }
91      
92      public static <T> T[] elements(Collection<T> init){
93  	
94          if (init == null) {
95              throw new NullPointerException();
96          }
97  
98          // Put the collection in a HashSet to get rid of duplicates
99          Set<T> tempSet = new HashSet<T>(init);
100 
101         return tempSet.toArray((T[]) new Object[init.size()]);
102     }
103 
104     private ConsistentSet(T[] elements){
105 	this.elements = elements;
106     }
107 
108     /**
109      * Returns an <code>iterator</code> over the elements in this set. The elements
110      * are returned in no particular order. Because all instances of
111      * <code>ConsistentSet</code> are unmodifiable, the <code>remove</code> method 
112      * of the returned <code>Iterator</code> throws
113      * <code>UnsupportedOperationException</code>.
114      *
115      * @return an <code>Iterator</code> over the elements in this
116      *     <code>ConsistentSet</code>.
117      */
118     @Override
119     public Iterator<T> iterator() {
120 
121         return new Iterator<T>() {
122 
123             private int nextPos = 0;
124 
125             @Override
126             public boolean hasNext() {
127                 return nextPos < elements.length;
128             }
129 
130             @Override
131             public T next() {
132 		if (!hasNext()) throw new NoSuchElementException();
133                 T next = elements[nextPos];
134                 ++nextPos;
135                 return next;
136             }
137 
138             @Override
139             public void remove() {
140                 throw new IllegalArgumentException();
141             }
142         };
143     }
144 
145     /**
146      * Returns the number of elements in this <code>ConsistentSet</code> (its cardinality).
147      *
148      * @return the number of elements in this <code>ConsistentSet</code> (its cardinality).
149      */
150     @Override
151     public int size() {
152         return elements.length;
153     }
154 }
