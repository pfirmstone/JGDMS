19  package org.apache.river.lookup.util;
20  
21  import java.io.IOException;
22  import java.io.Serializable;
23  import java.util.Map;
24  import org.apache.river.api.io.AtomicSerial;
25  import org.apache.river.api.io.AtomicSerial.GetArg;
26  
27  /**
28   * An implementation of the <code>java.util.Map.Entry</code> interface that has
29   * a serialized form consistent in all virtual machines. <code>ConsistentMapEntry</code>
30   * instances are unmodifiable. The <code>setValue</code> mutator method
31   * throws <code>UnsupportedOperationException</code>.  This class permits <code>null</code>
32   * for values and keys.
33   *
34   * <p>
35   * Although instances of this class are unmodifiable, they are not necessarily
36   * immutable. If a client retrieves a mutable object (either a key or value) contained in a
37   * <code>ConsistentMapEntry</code> and mutates that object, the client in effect
38   * mutates the state of the <code>ConsistentMapEntry</code>. In this case, the
39   * serialized form of the <code>ConsistentMapEntry</code> will most likely also
40   * have been mutated. A <code>ConsistentMapEntry</code> that contains only immutable
41   * objects will maintain a consistent serialized form indefinitely. But a 
42   * <code>ConsistentMapEntry</code> that contains mutable objects will maintain a
43   * consistent serialized form only so long as the mutable objects are not
44   * mutated.
45   *
46   * @author Bill Venners
47   */
48  @AtomicSerial
49  final class ConsistentMapEntry<K,V> implements Map.Entry<K,V>, Serializable {
50  
51      private static final long serialVersionUID = -8633627011729114409L;
52  
53      /**
54       * @serial An <code>Object</code> key, or <code>null</code>
55       */
56      private final K key;
57  
58      /**
59       * @serial An <code>Object</code> value, or <code>null</code>
60       */
61      private final V value;
62  
63      /**
64       * {@link AtomicSerial} constructor.
65       * @param arg
66       * @throws IOException 
67       * @throws ClassCastException if types don't match <K,V>
68       */
69      public ConsistentMapEntry(GetArg arg) throws IOException {
70  	this((K) arg.get("key", null), (V) arg.get("value", null));
71      }
72  
73      /**
74       * Constructs a new <code>ConsistentMapEntry</code> with passed
75       * <code>key</code> and <code>value</code>. <code>null</code> is
76       * allowed in either (or both) parameters.
77       *
78       * @param key the key (<code>null</code> key is OK)
79       * @param value the value (<code>null</code> value is OK) associated with the key
80       */
81      public ConsistentMapEntry(K key, V value) {
82  
83          this.key = key;
84          this.value = value;
85      }
86  
87      /**
88       * Returns the key.
89       *
90       * @return the key.
91       */
92      @Override
93      public K getKey() {
94          return key;
95      }
96  
97      /**
98       * Returns the value.
99       *
100      * @return the value.
101      */
102     @Override
103     public V getValue() {
104         return value;
105     }
106 
107     /**
108      * Replaces the value corresponding to this entry with the specified value. Because
109      * all instances of this class are unmodifiable, this method always throws
110      * <code>UnsupportedOperationException</code>.
111      *
112      * @throws UnsupportedOperationException always
113      */
114     @Override
115     public V setValue(V value) {
116         throw new UnsupportedOperationException();
117     }
118 
119     /**
120      * Compares the specified object (the <CODE>Object</CODE> passed
121      * in <CODE>o</CODE>) with this <CODE>ConsistentMapEntry</CODE>
122      * object for equality. Returns true if the specified object
123      * is not <code>null</code>, if the specified object's class is
124      * <CODE>ConsistentMapEntry</CODE>, if the keys of this object and
125      * the specified object are either both <code>null</code> or semantically
126      * equal, and the values of this object and the specified object are either
127      * both <code>null</code> or semantically equal.
128      *
129      * @param o the object to compare against
130      * @return <code>true</code> if the objects are the semantically equal,
131      *     <code>false</code> otherwise.
132      */
133     @Override
134     public boolean equals(Object o) {
135 
136         if (o == null) {
137             return false;
138         }
139 
140         if (o == this) {
141             return true;
142         }
143 
144         // TODO: ASK JOSH SHOULD EQUALS CHECK FOR INSTANCEOF OR EXACT CLASS?
145         if (o.getClass() != ConsistentMapEntry.class) {
146             return false;
147         }
148 
149         ConsistentMapEntry unmod = (ConsistentMapEntry) o;
150 
151         boolean keysEqual = equalsOrNull(key, unmod.key);
152         boolean valsEqual = equalsOrNull(value, unmod.value);
153 
154         return keysEqual && valsEqual;
155     }
156 
157     private static boolean equalsOrNull(Object o1, Object o2) {
158         return (o1 == null ? o2 == null : o1.equals(o2));
159     }
160 
161     /**    
162      * Returns the hash code value for this <CODE>ConsistentMapEntry</CODE> object.
163      *
164      * @return the hashcode for this object
165      */ 
166     @Override
167     public int hashCode() {
168 
169         int keyHash = (key == null ? 0 : key.hashCode());
170         int valueHash = (value == null ? 0 : value.hashCode());
171 
172         return keyHash ^ valueHash;
173     }
174 }
